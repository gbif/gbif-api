14    private static final EndpointCreatedComparator COMP = EndpointCreatedComparator.INSTANCE;
15    private final long now = System.currentTimeMillis();
16  
17    @Test
18    public void testSuccessfulSimpleComparisons() {
19      Endpoint e1 = new Endpoint();
20      Endpoint e2 = new Endpoint();
21  
22      e1.setCreated(new Date(now));
23      e2.setCreated(new Date(now));
24      assertEquals(0, COMP.compare(e1, e2));
25  
26      e2.setCreated(new Date(now-10000));
27      assertTrue(COMP.compare(e1, e2) > 0);
28  
29      e2.setCreated(new Date(now+10000));
30      assertTrue(COMP.compare(e1, e2) < 0);
31    }
32
