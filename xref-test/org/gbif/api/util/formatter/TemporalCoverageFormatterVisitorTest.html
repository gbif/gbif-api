14    @Test
15    public void testTemporalCoverageFormatterVisitor(){
16      TemporalCoverageFormatterVisitor formatter = new TestFormatter();
17      TemporalCoverage dateRange = new DateRange();
18      TemporalCoverage singleDate = new SingleDate();
19      TemporalCoverage verbatimTimePeriod = new VerbatimTimePeriod();
20  
21      assertEquals("dateRange", dateRange.acceptFormatter(formatter));
22      assertEquals("singleDate", singleDate.acceptFormatter(formatter));
23      assertEquals("verbatimTimePeriod", verbatimTimePeriod.acceptFormatter(formatter));
24    }
25  
26    /**
27     * Test implementation of TemporalCoverageFormatterVisitor to ensure all format() methods are reachable
28     */
29    private static class TestFormatter implements TemporalCoverageFormatterVisitor {
30  
31      @Override
32      public String format(DateRange t) {
33        return "dateRange";
34      }
35  
36      @Override
37      public String format(SingleDate t) {
38        return "singleDate";
39      }
40  
41      @Override

44      }
