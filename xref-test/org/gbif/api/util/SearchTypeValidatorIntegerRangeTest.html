16  @RunWith(Parameterized.class)
17  public class SearchTypeValidatorIntegerRangeTest {
18  
19    private final String arg;
20    private final Integer start;
21    private final Integer end;
22  
23    public SearchTypeValidatorIntegerRangeTest(String arg, Integer start, Integer end) {
24      this.arg = arg;
25      this.start = start;
26      this.end = end;
27    }
28  
29    @Parameterized.Parameters
30    public static Collection<Object[]> data() {
31      Object[][] data = {
32        {"10", null, null},
33        {"10,20", 10, 20},
34        {"10.1,20.0", null, null},
35        {"-1 , 0", -1, 0},
36        {"*,20", null, 20},
37        {"10, *", 10, null},
38        {"peter", null, null},
39        {"Puma concolor, 1882", null, null},
40      };
41      return Arrays.asList(data);
42    };
43  
44  
45    @Test
46    public void testRange() throws Exception {
47      try {
48        Range<Integer> range = SearchTypeValidator.parseIntegerRange(arg);
49        if (start == null && end == null) {
50          fail(arg + " supposed to be an invalid range");
51        } else {
52          // test parse result
53          if (range.hasLowerBound()) {
54            assertEquals(start, range.lowerEndpoint());
55          } else {
56            assertNull(start);
57          }
58  
59          if (range.hasUpperBound()) {
60            assertEquals(end, range.upperEndpoint());
61          } else {
62            assertNull(end);
63          }
64  
65          assertTrue(SearchTypeValidator.isRange(arg));
66        }
67  
68      } catch (IllegalArgumentException e) {
69        // expected?
70        if (start != null || end != null) {
71          fail(arg + " supposed to be a valid range");
72        }
73      }
74    }

76  }
