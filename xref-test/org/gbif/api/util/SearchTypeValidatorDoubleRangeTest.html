16  @RunWith(Parameterized.class)
17  public class SearchTypeValidatorDoubleRangeTest {
18  
19    private final String arg;
20    private final Double start;
21    private final Double end;
22  
23    public SearchTypeValidatorDoubleRangeTest(String arg, Double start, Double end) {
24      this.arg = arg;
25      this.start = start;
26      this.end = end;
27    }
28  
29    @Parameterized.Parameters
30    public static Collection<Object[]> data() {
31      Object[][] data = {
32        {"10.3", null, null},
33        {"10,20", 10d, 20d},
34        {"*,20", null, 20d},
35        {"10, *", 10d, null},
36        {"10.1,20.2", 10.1d, 20.2d},
37        {"-1,2.0432", -1d, 2.0432d},
38        {"peter", null, null},
39      };
40      return Arrays.asList(data);
41    };
42  
43  
44    @Test
45    public void testRange() throws Exception {
46      try {
47        Range<Double> range = SearchTypeValidator.parseDecimalRange(arg);
48        if (start == null && end == null) {
49          fail(arg + " supposed to be an invalid range");
50        } else {
51          // test parse result
52          if (range.hasLowerBound()) {
53            assertEquals(start, range.lowerEndpoint());
54          } else {
55            assertNull(start);
56          }
57  
58          if (range.hasUpperBound()) {
59            assertEquals(end, range.upperEndpoint());
60          } else {
61            assertNull(end);
62          }
63  
64          assertTrue(SearchTypeValidator.isRange(arg));
65        }
66  
67      } catch (IllegalArgumentException e) {
68        // expected?
69        if (start != null || end != null) {
70          fail(arg + " supposed to be a valid range");
71        }
72      }
73    }
