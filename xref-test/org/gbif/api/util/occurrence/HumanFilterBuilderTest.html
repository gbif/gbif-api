69    @Test
70    public void testAllParams() throws Exception {
71  
72      final String date = DateFormatUtils.ISO_DATE_FORMAT.format(new Date());
73      List<Predicate> ands = Lists.newArrayList();
74      for (OccurrenceSearchParameter p : OccurrenceSearchParameter.values()) {
75        if (p.type().isEnum()) {
76          if (p.type() == Country.class) {
77            ands.add(new EqualsPredicate(p, Country.DENMARK.getIso2LetterCode()));
78  
79          } else {
80            Class<Enum<?>> vocab = (Class<Enum<?>>) p.type();
81            // add a comparison for every possible enum value to test the resource bundle for completeness
82            List<Predicate> ors = Lists.newArrayList();
83            for (Enum<?> e : vocab.getEnumConstants()) {
84              ors.add(new EqualsPredicate(p, e.toString()));
85            }
86            ands.add(new DisjunctionPredicate(ors));
87          }
88  
89        } else if (p.type() == Date.class) {
90          ands.add(new EqualsPredicate(p, date));
91  
92        } else if (p.type() == Double.class) {
93          ands.add(new EqualsPredicate(p, "12.478"));
94  
95        } else if (p.type() == Integer.class) {
96          ands.add(new EqualsPredicate(p, "10"));
97  
98        } else if (p.type() == String.class) {
99          if (p == OccurrenceSearchParameter.GEOMETRY) {
100           ands.add(new EqualsPredicate(p, "POLYGON ((30 10, 10 20, 20 40, 40 40, 30 10))"));
101         } else {
102           ands.add(new EqualsPredicate(p, "Bernd Neumann"));
103         }
104 
105       } else if (p.type() == Boolean.class) {
106         ands.add(new EqualsPredicate(p, "true"));
107 
108       } else if (p.type() == UUID.class) {
109         ands.add(new EqualsPredicate(p, UUID.randomUUID().toString()));
110 
111       } else {
112         throw new IllegalStateException("Unknown SearchParameter type " + p.type());
113       }
114     }
115     ConjunctionPredicate and = new ConjunctionPredicate(ands);
116 
117     Map<OccurrenceSearchParameter, LinkedList<String>> x = builder.humanFilter(and);
118     for (OccurrenceSearchParameter p : x.keySet()) {
119       System.out.print(p + "\t");
120       System.out.println(x.get(p));
121     }
