40    @Test
41    public void testInferRank() {
42      for (Map.Entry<String, Rank> stringRankEntry : NAMES.entrySet()) {
43        assertEquals(stringRankEntry.getValue(), Rank.inferRank(stringRankEntry.getKey(), null, null, null, null));
44      }
45  
46      assertEquals(Rank.SPECIES, Rank.inferRank("Abies", "Abies", "alba", null, null));
47      assertEquals(Rank.SPECIES, Rank.inferRank("Abies", null, "alba", null, null));
48      assertEquals(Rank.INFRAGENERIC_NAME, Rank.inferRank(null, "Abies", null, null, null));
49      assertEquals(Rank.INFRAGENERIC_NAME, Rank.inferRank("", "Abies", null, null, null));
50      assertEquals(Rank.VARIETY, Rank.inferRank(null, "Abies", "alba", "var.", "alpina"));
51      assertEquals(Rank.SUBSPECIES, Rank.inferRank(null, "Abies", "alba", "ssp.", null));
52      assertEquals(Rank.SPECIES, Rank.inferRank(null, "Abies", null, "spec.", null));
53      assertEquals(Rank.SUPRAGENERIC_NAME, Rank.inferRank("Neurolaenodinae", null, null, "ib.", null));
54      assertEquals(Rank.SUPRAGENERIC_NAME, Rank.inferRank("Neurolaenodinae", null, null, "supersubtrib.", null));
55  
56      // should not be able to infer the correct family
57      assertEquals(Rank.UNRANKED, Rank.inferRank("Compositae", null, null, null, null));
58    }
