12  public class NomenclaturalStatusTest {
13  
14      @Test
15      public void testFromString() throws Exception {
16          assertEquals(NomenclaturalStatus.CONSERVED, NomenclaturalStatus.fromString("conserved"));
17          assertEquals(NomenclaturalStatus.CONSERVED, NomenclaturalStatus.fromString("conserved "));
18          assertEquals(NomenclaturalStatus.CONSERVED, NomenclaturalStatus.fromString("nom.cons."));
19          assertEquals(NomenclaturalStatus.CONSERVED, NomenclaturalStatus.fromString("nomcons"));
20          assertEquals(NomenclaturalStatus.CONSERVED, NomenclaturalStatus.fromString("nomen conservandum"));
21          assertEquals(NomenclaturalStatus.CONSERVED, NomenclaturalStatus.fromString("orth. cons."));
22          assertEquals(NomenclaturalStatus.CONSERVED, NomenclaturalStatus.fromString("conservandum"));
23  
24          assertNull(NomenclaturalStatus.fromString("carla"));
25          assertNull(NomenclaturalStatus.fromString(""));
26          assertNull(NomenclaturalStatus.fromString(null));
27      }
28  
29      @Test
30      public void testUniqueness() throws Exception {
31          Set<String> keys = Sets.newHashSet();
32          for (NomenclaturalStatus ns : NomenclaturalStatus.values()) {
33              if (ns.getAbbreviatedLabel() != null) {
34                  assertFalse(ns.name(), keys.contains(ns.getAbbreviatedLabel()));
35                  keys.add(ns.getAbbreviatedLabel());
36              }
37  
38              if (ns.getLatinLabel() != null) {
39                  assertFalse(ns.name(), keys.contains(ns.getLatinLabel()));
40                  keys.add(ns.getLatinLabel());
41              }
42          }
43      }
