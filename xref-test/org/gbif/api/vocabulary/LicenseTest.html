16  package org.gbif.api.vocabulary;

18  import org.junit.jupiter.api.Test;
19  
20  import static org.junit.jupiter.api.Assertions.assertEquals;
21  import static org.junit.jupiter.api.Assertions.assertFalse;
22  import static org.junit.jupiter.api.Assertions.assertNull;
23  import static org.junit.jupiter.api.Assertions.assertTrue;
24  
25  public class LicenseTest {
26  
27    @Test
28    public void testFromString() {
29      // positive matches
30      assertEquals(License.CC0_1_0, License.fromLicenseUrl("http://creativecommons.org/publicdomain/zero/1.0/legalcode").get());
31      assertEquals(License.CC0_1_0, License.fromLicenseUrl("HTTP://creativecommons.org/publicdomain/zero/1.0/legalcode").get());
32      assertEquals(License.CC0_1_0, License.fromLicenseUrl(" http://creativecommons.org/publicdomain/zero/1.0/legalcode ").get());
33      assertEquals(License.CC0_1_0, License.fromLicenseUrl("http://creativecommons.org/publicdomain/zero/1.0/legalcode/").get());
34      assertEquals(License.CC0_1_0, License.fromLicenseUrl("http://creativecommons.org/publicdomain/zero/1.0").get());
35      assertEquals(License.CC0_1_0, License.fromLicenseUrl("http://creativecommons.org/publicdomain/zero/1.0/").get());
36      assertEquals(License.CC0_1_0, License.fromLicenseUrl("HTTPS://CREATIVECOMMONS.ORG/PUBLICDOMAIN/ZERO/1.0/").get());
37      assertEquals(License.CC_BY_4_0, License.fromLicenseUrl("http://creativecommons.org/licenses/by/4.0/legalcode").get());
38      assertEquals(License.CC_BY_NC_4_0, License.fromLicenseUrl("http://creativecommons.org/licenses/by-nc/4.0/legalcode").get());
39  
40      // negative matches
41      assertNull(License.fromLicenseUrl("CC0").orElse(null));
42      assertNull(License.fromLicenseUrl("https://creativecommons.org/licenses/by/3.0/legalcode").orElse(null));
43      assertNull(License.fromLicenseUrl("https://creativecommons.org/licenses/by/3.0/").orElse(null));
44      assertNull(License.fromLicenseUrl("https://creativecommons.org/licenses/by-nc/2.0/legalcode").orElse(null));
45      assertNull(License.fromLicenseUrl("https://creativecommons.org/licenses/by-nc/2.0/").orElse(null));
46    }
47  
48    @Test
49    public void testConcreteLicense() {
50      assertFalse(License.UNSPECIFIED.isConcrete());
51      assertFalse(License.UNSUPPORTED.isConcrete());
52      assertTrue(License.CC0_1_0.isConcrete());
53    }
54  
55    @Test
56    public void testGetMostRestrictive(){
57      assertEquals(License.CC_BY_4_0, License.getMostRestrictive(License.CC0_1_0, License.CC_BY_4_0, License.UNSPECIFIED));
58      assertEquals(License.CC_BY_4_0, License.getMostRestrictive(License.CC_BY_4_0, License.CC0_1_0, License.UNSPECIFIED));
59  
60      assertEquals(License.CC_BY_NC_4_0, License.getMostRestrictive(License.CC_BY_4_0, License.CC_BY_NC_4_0, License.UNSPECIFIED));
61  
62      assertEquals(License.UNSPECIFIED, License.getMostRestrictive(License.UNSUPPORTED, License.CC0_1_0, License.UNSPECIFIED));
63      assertEquals(License.UNSPECIFIED, License.getMostRestrictive(License.UNSUPPORTED, License.UNSPECIFIED, License.UNSPECIFIED));
64      assertEquals(License.UNSPECIFIED, License.getMostRestrictive(License.CC_BY_NC_4_0, null, License.UNSPECIFIED));
65    }
66  
67    @Test
68    public void testLicenseOrdinalNumber() {
69      int numberOfConcreteLicense = 0;
70      for (License currLicense : License.values()) {
71        if (currLicense.isConcrete()) {
72          numberOfConcreteLicense++;
73        }
74      }
75  
76      //The ordinal number in the Enum implicitly defines the level of restriction.
77      assertEquals(3, numberOfConcreteLicense, "The License value ordinal test covers all values of the Concrete License");
78      assertTrue(License.CC0_1_0.ordinal() < License.CC_BY_4_0.ordinal());
79      assertTrue(License.CC_BY_4_0.ordinal() < License.CC_BY_NC_4_0.ordinal());
80    }
81  }
