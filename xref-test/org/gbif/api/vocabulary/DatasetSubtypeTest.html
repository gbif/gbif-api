52          fail("Enumeration present in both sets - can only be present in one");
53        } else if (!DatasetSubtype.CHECKLIST_DATASET_SUBTYPES.contains(e) && !DatasetSubtype.OCCURRENCE_DATASET_SUBTYPES
54          .contains(e)) {
55          fail("Enumeration not present in either set");
56        } else {
57          assertTrue(DatasetSubtype.CHECKLIST_DATASET_SUBTYPES.contains(e) || DatasetSubtype.OCCURRENCE_DATASET_SUBTYPES
