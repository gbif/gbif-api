16  package org.gbif.api.jackson;

18  import org.gbif.dwc.terms.DwcTerm;
19  import org.gbif.dwc.terms.Term;
20  
21  import java.io.IOException;
22  import java.util.AbstractMap;
23  import java.util.Arrays;
24  import java.util.Collections;
25  import java.util.List;
26  import java.util.Map;
27  
28  import org.junit.Test;
29  
30  import com.fasterxml.jackson.databind.ObjectMapper;
31  import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
32  import com.fasterxml.jackson.databind.annotation.JsonSerialize;
33  
34  import static junit.framework.TestCase.assertEquals;
35  
36  /**
37   * Unit tests for {@link MapEntrySerde}.
38   */
39  public class MapEntrySerdeTest {
40  
41    private static final ObjectMapper MAPPER = new ObjectMapper();
42  
43    @Test
44    public void testMapEntrySerde() throws IOException {
45  
46      MapEntryWrapper kv = new MapEntryWrapper("mykey", 18);
47      String json = MAPPER.writeValueAsString(kv);
48  
49      MapEntryWrapper rebuiltKeyValue = MAPPER.readValue(json, MapEntryWrapper.class);
50      assertEquals(kv.getSingleElement().getValue(), rebuiltKeyValue.getSingleElement().getValue());
51  
52      MapEntryListTermWrapper kvl = new MapEntryListTermWrapper(DwcTerm.country, 19);
53      json = MAPPER.writeValueAsString(kvl);
54  
55      MapEntryListTermWrapper rebuiltKeyValueList = MAPPER.readValue(json, MapEntryListTermWrapper.class);
56      assertEquals(2, rebuiltKeyValueList.getListTerm().size());
57  
58    }
59  
60    /**
61     * For testing purpose only.
62     */
63    static class MapEntryWrapper {
64  
65      private Map.Entry<Object, Object> singleElement;
66  
67      MapEntryWrapper() {}
68  
69      MapEntryWrapper(Object key, Object value){
70        singleElement = new AbstractMap.SimpleImmutableEntry<>(key, value);
71      }
72  
73      @JsonSerialize(using = MapEntrySerde.MapEntryJsonSerializer.class)
74      @JsonDeserialize(using = MapEntrySerde.MapEntryJsonDeserializer.class)
75      Map.Entry<Object, Object> getSingleElement() {
76        return singleElement;
77      }
78    }
79  
80    static class MapEntryListTermWrapper {
81  
82      private List<Map.Entry<Term, Object>> listTerm;
83  
84      MapEntryListTermWrapper() {}
85  
86      MapEntryListTermWrapper(Term key, Object value){
87        listTerm = Collections.unmodifiableList(
88          Arrays.asList(
89            new AbstractMap.SimpleImmutableEntry<>(key, value),
90            new AbstractMap.SimpleImmutableEntry<>(key, value)));
91      }
92  
93      @JsonSerialize(contentUsing = MapEntrySerde.MapEntryJsonSerializer.class)
94      @JsonDeserialize(contentUsing = MapEntrySerde.MapEntryJsonDeserializer.class)
95      List<Map.Entry<Term, Object>> getListTerm() {
96        return listTerm;
97      }
98    }
99  
100 }
