14  package org.gbif.api.jackson;
15  
16  import com.fasterxml.jackson.databind.ObjectMapper;
17  import org.gbif.api.model.checklistbank.search.NameUsageSearchParameter;
18  import org.gbif.api.model.common.search.Facet;
19  import org.gbif.api.model.event.search.EventSearchParameter;
20  import org.gbif.api.model.literature.search.LiteratureSearchParameter;
21  import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
22  import org.gbif.api.model.registry.search.DatasetSearchParameter;
23  import org.junit.jupiter.api.Assertions;
24  import org.junit.jupiter.api.Test;
25  
26  public class SearchParameterSerDeserTest {
27  
28    @Test
29    public void testFacetsSer() throws Exception {
30      Facet<OccurrenceSearchParameter> facet = new Facet<>(OccurrenceSearchParameter.ISLAND);
31  
32      ObjectMapper objectMapper = new ObjectMapper();
33      Assertions.assertEquals(
34          "{\"field\":\"ISLAND\",\"counts\":null}", objectMapper.writeValueAsString(facet));
35  
36      Facet<EventSearchParameter> eventFacet = new Facet<>(EventSearchParameter.ISLAND);
37      Assertions.assertEquals(
38          "{\"field\":\"ISLAND\",\"counts\":null}", objectMapper.writeValueAsString(eventFacet));
39  
40      Facet<DatasetSearchParameter> datasetFacet = new Facet<>(DatasetSearchParameter.CONTINENT);
41      Assertions.assertEquals(
42          "{\"field\":\"CONTINENT\",\"counts\":null}", objectMapper.writeValueAsString(datasetFacet));
43  
44      Facet<NameUsageSearchParameter> nameUsageFacet = new Facet<>(NameUsageSearchParameter.RANK);
45      Assertions.assertEquals(
46          "{\"field\":\"RANK\",\"counts\":null}", objectMapper.writeValueAsString(nameUsageFacet));
47  
48      Facet<LiteratureSearchParameter> literatureFacet = new Facet<>(LiteratureSearchParameter.DOI);
49      Assertions.assertEquals(
50          "{\"field\":\"DOI\",\"counts\":null}", objectMapper.writeValueAsString(literatureFacet));
51    }
52  }
