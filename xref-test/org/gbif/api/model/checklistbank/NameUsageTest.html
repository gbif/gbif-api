178   public void testSetLink() {
179     Identifier link = new Identifier();
180     link.setIdentifier("http://www.example.org");
181     link.setType(IdentifierType.URL);
182 
183     Identifier id = new Identifier();
184     id.setIdentifier("12345");
185     id.setType(IdentifierType.SOURCE_ID);
186 
187     NameUsage nu = new NameUsage();
188     nu.getIdentifiers().add(id);
189     assertNull(nu.getLink());
190     assertEquals("12345", nu.getSourceId());
191 
192     nu.getIdentifiers().add(link);
193     assertEquals("http://www.example.org", nu.getLink());
194 
195     nu.getIdentifiers().clear();
196     assertNull(nu.getLink());
197     assertNull(nu.getSourceId());
198     assertEquals(0, nu.getIdentifiers().size());
199 
200 
201     // sourceid
202     nu.setSourceId("99");
203     assertEquals(1, nu.getIdentifiers().size());
204     assertEquals("99", nu.getSourceId());
205     assertEquals("99", nu.getIdentifiers().get(0).getIdentifier());
206 
207     nu.setSourceId("991");
208     assertEquals(1, nu.getIdentifiers().size());
209     assertEquals("991", nu.getSourceId());
210     assertEquals("991", nu.getIdentifiers().get(0).getIdentifier());
211 
212 
213     // link
214     assertEquals(1, nu.getIdentifiers().size());
215     nu.addLink("http://www.example.org");
216     assertEquals("http://www.example.org", nu.getLink());
217 
218     nu.setLink("http://www.example.org/2");
219     assertEquals("http://www.example.org/2", nu.getLink());
220 
221     nu.addLink("http://www.example.org/3");
222     assertEquals("http://www.example.org/2", nu.getLink());
223     assertEquals(3, nu.getIdentifiers().size());

245   @Test
246   public void testJsonSerde() throws IOException {
247     final ObjectMapper mapper = new ObjectMapper();
248     mapper.enable(DeserializationConfig.Feature.ACCEPT_EMPTY_STRING_AS_NULL_OBJECT);
249     mapper.enable(SerializationConfig.Feature.INDENT_OUTPUT);
250     mapper.disable(SerializationConfig.Feature.WRITE_NULL_PROPERTIES);
251 
252     NameUsage u = new NameUsage();
253     u.setKey(123);
254     u.setDatasetKey(UUID.randomUUID());
255     u.setNumDescendants(321);
256     u.getNomenclaturalStatus().add(NomenclaturalStatus.AMBIGUOUS);
257     Identifier i = new Identifier();
258     i.setIdentifier("me");
259     u.addIdentifier(i);
260     u.setKingdomKey(6);
261     u.setKingdom("Plants");
262 
263     String json = mapper.writeValueAsString(u);
264     System.out.println(json);
265     assertEquals(u, mapper.readValue(json, NameUsage.class));
266   }
