14    @Test
15    public void testCanonicalName() throws Exception {
16      ParsedName pn = new ParsedName();
17      pn.setGenusOrAbove("Abies");
18      assertEquals("Abies", pn.canonicalName());
19  
20      pn.setSpecificEpithet("alba");
21      assertEquals("Abies alba", pn.canonicalName());
22  
23      pn = new ParsedName();
24      pn.setGenusOrAbove("Abies");
25      pn.setSpecificEpithet("alba");
26      pn.setRank(Rank.VARIETY);
27      pn.setAuthorship("Mill.");
28      pn.setBracketAuthorship("Carl.");
29      pn.setNotho(NamePart.GENERIC);
30      pn.setInfraSpecificEpithet("alpina");
31      pn.setYear("1887");
32      pn.setSensu("Döring");
33      pn.setRemarks("lost");
34      pn.setNomStatus("nom. illeg.");
35      pn.setAuthorsParsed(true);
36  
37      assertEquals("Abies alba alpina", pn.canonicalName());
38      assertEquals("×Abies alba var. alpina", pn.canonicalNameWithMarker());
39      assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887", pn.canonicalNameComplete());
40      assertEquals("Abies alba", pn.canonicalSpeciesName());
41      assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887 Döring, nom. illeg. [lost]", pn.fullName());
42    }

80    @Test
81    public void testSubgenus() throws Exception {
82      // Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013
83      ParsedName pn = new ParsedName();
84      pn.setGenusOrAbove("Brachyhypopomus");
85      pn.setInfraGeneric("Odontohypopomus");
86      pn.setAuthorship("Sullivan, Zuanon & Cox Fernandes");
87      pn.setYear("2013");
88      assertEquals("Odontohypopomus", pn.canonicalName());
89      assertEquals("Odontohypopomus", pn.canonicalNameWithMarker());
90      assertEquals("Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
91      // with given rank marker it is shown instead of brackets
92      pn.setRank(Rank.SUBGENUS);
93      assertEquals("Odontohypopomus", pn.canonicalName());
94      assertEquals("subgen. Odontohypopomus", pn.canonicalNameWithMarker());
95      assertEquals("Brachyhypopomus subgen. Odontohypopomus Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
96  
97      // Achillea sect. Ptarmica (Mill.) W.D.J.Koch
98      pn = new ParsedName();
99      pn.setGenusOrAbove("Achillea");
100     pn.setInfraGeneric("Ptarmica");
101     pn.setAuthorship("W.D.J.Koch");
102     pn.setBracketAuthorship("Mill.");
103 
104     assertEquals("Ptarmica", pn.canonicalName());
105     assertEquals("Achillea (Ptarmica) (Mill.) W.D.J.Koch", pn.canonicalNameComplete());
106     pn.setRank(Rank.SECTION);
107     assertEquals("Achillea sect. Ptarmica (Mill.) W.D.J.Koch", pn.canonicalNameComplete());
108   }

122 
123     pn.setYear("1904");

125 
126     pn.setBracketAuthorship("Carl.");

142   /**
143    * assert all build name methods return the same string
144    * @param pn
145    * @param name expected string
146    */
147   private void assertBuildName(ParsedName pn, String name) {
148     assertEquals(name, pn.fullName());
149     assertEquals(name, pn.canonicalName());
150     assertEquals(name, pn.canonicalNameComplete());
151     assertEquals(name, pn.canonicalNameWithMarker());
152   }
153 
154   private void assertBuildName(ParsedName pn, String full, String canonical, String canonicalComplete, String canonicalMarker) {
155     assertEquals(full, pn.fullName());
156     assertEquals(canonical, pn.canonicalName());
157     assertEquals(canonicalComplete, pn.canonicalNameComplete());
158     assertEquals(canonicalMarker, pn.canonicalNameWithMarker());
159   }
