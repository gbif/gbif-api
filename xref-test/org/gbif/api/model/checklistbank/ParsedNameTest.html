13    @Test
14    public void testCanonicalName() throws Exception {
15      ParsedName pn = new ParsedName();
16      pn.setGenusOrAbove("Abies");
17      assertEquals("Abies", pn.canonicalName());
18  
19      pn.setSpecificEpithet("alba");
20      assertEquals("Abies alba", pn.canonicalName());
21  
22      pn = new ParsedName();
23      pn.setGenusOrAbove("Abies");
24      pn.setSpecificEpithet("alba");
25      pn.setRank(Rank.VARIETY);
26      pn.setAuthorship("Mill.");
27      pn.setBracketAuthorship("Carl.");
28      pn.setNotho(NamePart.GENERIC);
29      pn.setInfraSpecificEpithet("alpina");
30      pn.setYear("1887");
31      pn.setSensu("Döring");
32      pn.setRemarks("lost");
33      pn.setNomStatus("nom. illeg.");
34      pn.setAuthorsParsed(true);
35  
36      assertEquals("Abies alba alpina", pn.canonicalName());
37      assertEquals("×Abies alba var. alpina", pn.canonicalNameWithMarker());
38      assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887", pn.canonicalNameComplete());
39      assertEquals("Abies alba", pn.canonicalSpeciesName());
40      assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887 Döring, nom. illeg. [lost]", pn.fullName());
41    }

62    @Test
63    public void testSubgenus() throws Exception {
64      // Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013
65      ParsedName pn = new ParsedName();
66      pn.setGenusOrAbove("Brachyhypopomus");
67      pn.setInfraGeneric("Odontohypopomus");
68      pn.setAuthorship("Sullivan, Zuanon & Cox Fernandes");
69      pn.setYear("2013");
70      assertEquals("Odontohypopomus", pn.canonicalName());
71      assertEquals("Odontohypopomus", pn.canonicalNameWithMarker());
72      assertEquals("Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
73      // with given rank marker it is shown instead of brackets
74      pn.setRank(Rank.SUBGENUS);
75      assertEquals("Odontohypopomus", pn.canonicalName());
76      assertEquals("subgen. Odontohypopomus", pn.canonicalNameWithMarker());
77      assertEquals("Brachyhypopomus subgen. Odontohypopomus Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
78  
79      // Achillea sect. Ptarmica (Mill.) W.D.J.Koch
80      pn = new ParsedName();
81      pn.setGenusOrAbove("Achillea");
82      pn.setInfraGeneric("Ptarmica");
83      pn.setAuthorship("W.D.J.Koch");
84      pn.setBracketAuthorship("Mill.");
85  
86      assertEquals("Ptarmica", pn.canonicalName());
87      assertEquals("Achillea (Ptarmica) (Mill.) W.D.J.Koch", pn.canonicalNameComplete());
88      pn.setRank(Rank.SECTION);
89      assertEquals("Achillea sect. Ptarmica (Mill.) W.D.J.Koch", pn.canonicalNameComplete());
90    }

124   /**
125    * assert all build name methods return the same string
126    * @param pn
127    * @param name expected string
128    */
129   private void assertBuildName(ParsedName pn, String name) {
130     assertEquals(name, pn.fullName());
131     assertEquals(name, pn.canonicalName());
132     assertEquals(name, pn.canonicalNameComplete());
133     assertEquals(name, pn.canonicalNameWithMarker());
134   }
135 
136   private void assertBuildName(ParsedName pn, String full, String canonical, String canonicalComplete, String canonicalMarker) {
137     assertEquals(full, pn.fullName());
138     assertEquals(canonical, pn.canonicalName());
139     assertEquals(canonicalComplete, pn.canonicalNameComplete());
140     assertEquals(canonicalMarker, pn.canonicalNameWithMarker());
141   }
