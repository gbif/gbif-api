12    @Test
13    public void testCanonicalName() throws Exception {
14      ParsedName pn = new ParsedName();
15      pn.setGenusOrAbove("Abies");
16      assertEquals("Abies", pn.canonicalName());
17  
18      pn.setSpecificEpithet("alba");
19      assertEquals("Abies alba", pn.canonicalName());
20  
21  
22      pn = new ParsedName();
23      pn.setGenusOrAbove("Abies");
24      pn.setSpecificEpithet("alba");
25      pn.setRank(Rank.VARIETY);
26      pn.setAuthorship("Mill.");
27      pn.setBracketAuthorship("Carl.");
28      pn.setNotho(NamePart.GENERIC);
29      pn.setInfraSpecificEpithet("alpina");
30      pn.setYear("1887");
31      pn.setSensu("Döring");
32      pn.setRemarks("lost");
33      pn.setNomStatus("nom. illeg.");
34      pn.setAuthorsParsed(true);
35  
36      assertEquals("Abies alba alpina", pn.canonicalName());
37      assertEquals("×Abies alba var. alpina", pn.canonicalNameWithMarker());
38      assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887", pn.canonicalNameComplete());
39      assertEquals("Abies alba", pn.canonicalSpeciesName());
40      assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887 Döring, nom. illeg. [lost]", pn.fullName());
41    }

46    @Test
47    public void testSubgenus() throws Exception {
48      // Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013
49      ParsedName pn = new ParsedName();
50      pn.setGenusOrAbove("Brachyhypopomus");
51      pn.setInfraGeneric("Odontohypopomus");
52      pn.setAuthorship("Sullivan, Zuanon & Cox Fernandes");
53      pn.setYear("2013");
54      assertEquals("Brachyhypopomus (Odontohypopomus)", pn.canonicalName());
55      assertEquals("Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
56      // with given rank marker it is shown instead of brackets
57      pn.setRank(Rank.SUBGENUS);
58      assertEquals("Brachyhypopomus (Odontohypopomus)", pn.canonicalName());
59      assertEquals("Brachyhypopomus subgen. Odontohypopomus Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
60  
61      // Achillea sect. Ptarmica (Mill.) W.D.J.Koch
62      pn = new ParsedName();
63      pn.setGenusOrAbove("Achillea");
64      pn.setInfraGeneric("Ptarmica");
65      pn.setAuthorship("W.D.J.Koch");
66      pn.setBracketAuthorship("Mill.");
67  
68      assertEquals("Achillea (Ptarmica)", pn.canonicalName());
69      pn.setRank(Rank.SECTION);
70      assertEquals("Achillea sect. Ptarmica (Mill.) W.D.J.Koch", pn.canonicalNameComplete());
71    }

105   /**
106    * assert all build name methods return the same string
107    * @param pn
108    * @param name expected string
109    */
110   private void assertBuildName(ParsedName pn, String name) {
111     assertEquals(name, pn.fullName());
112     assertEquals(name, pn.canonicalName());
113     assertEquals(name, pn.canonicalNameComplete());
114     assertEquals(name, pn.canonicalNameWithMarker());
115   }
116 
117   private void assertBuildName(ParsedName pn, String full, String canonical, String canonicalComplete, String canonicalMarker) {
118     assertEquals(full, pn.fullName());
119     assertEquals(canonical, pn.canonicalName());
120     assertEquals(canonicalComplete, pn.canonicalNameComplete());
121     assertEquals(canonicalMarker, pn.canonicalNameWithMarker());
122   }
