15    @Test
16    public void testCanonicalName() throws Exception {
17      ParsedName pn = new ParsedName();
18      pn.setGenusOrAbove("Abies");
19      assertEquals("Abies", pn.canonicalName());
20  
21      pn.setSpecificEpithet("alba");
22      assertEquals("Abies alba", pn.canonicalName());
23  
24      pn = new ParsedName();
25      pn.setGenusOrAbove("Abies");
26      pn.setSpecificEpithet("alba");
27      pn.setRank(Rank.VARIETY);
28      pn.setAuthorship("Mill.");
29      pn.setBracketAuthorship("Carl.");
30      pn.setNotho(NamePart.GENERIC);
31      pn.setInfraSpecificEpithet("alpina");
32      pn.setYear("1887");
33      pn.setSensu("Döring");
34      pn.setRemarks("lost");
35      pn.setNomStatus("nom. illeg.");
36      pn.setParsedPartially(true);
37  
38      assertEquals("Abies alba alpina", pn.canonicalName());
39      assertEquals("×Abies alba var. alpina", pn.canonicalNameWithMarker());
40      assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887", pn.canonicalNameComplete());
41      assertEquals("Abies alba", pn.canonicalSpeciesName());
42      assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887 Döring, nom. illeg. [lost]", pn.fullName());
43  
44      pn.setGenusOrAbove(null);
45      assertEquals("alba alpina", pn.canonicalName());
46      assertEquals("alba var. alpina", pn.canonicalNameWithMarker());
47      assertEquals("alba var. alpina (Carl.) Mill., 1887", pn.canonicalNameComplete());
48      assertNull(pn.canonicalSpeciesName());
49      assertEquals("alba var. alpina (Carl.) Mill., 1887 Döring, nom. illeg. [lost]", pn.fullName());
50    }

71    @Test
72    public void testIndet() throws Exception {
73      ParsedName pn = new ParsedName();
74      pn.setType(NameType.SCIENTIFIC);
75      pn.setGenusOrAbove("Abies");
76      assertFalse(pn.isIndetermined());
77  
78      pn.setRank(Rank.SPECIES);
79      assertTrue(pn.isIndetermined());
80  
81      for (NameType t : NameType.values()) {
82        pn.setType(t);
83        if (t.isParsable()) {
84          assertTrue(pn.isIndetermined());
85        } else {
86          assertFalse(pn.isIndetermined());
87        }
88      }
89      pn.setType(NameType.SCIENTIFIC);
90  
91      pn.setSpecificEpithet("vulgaris");
92      assertFalse(pn.isIndetermined());
93  
94      pn.setRank(Rank.SUBSPECIES);
95      assertTrue(pn.isIndetermined());
96  
97      pn.setInfraSpecificEpithet("kingkong");
98      assertFalse(pn.isIndetermined());
99  
100     for (Rank r : Rank.values()) {
101       if (r.isInfraspecific()) {
102         assertFalse(r.toString(), pn.isIndetermined());
103       }
104     }
105 
106     pn.setInfraSpecificEpithet(null);
107     for (Rank r : Rank.values()) {
108       if (r.isInfraspecific()) {
109         assertTrue(r.toString(), pn.isIndetermined());
110       }
111     }
112 
113     pn.setRank(Rank.SUBGENUS);
114     pn.setSpecificEpithet(null);
115     assertTrue(pn.isIndetermined());
116 
117     pn.setInfraGeneric("Mysubgenus");
118     assertFalse(pn.isIndetermined());
119 
120 
121     pn = new ParsedName();
122     pn.setType(NameType.SCIENTIFIC);
123     pn.setRank(Rank.SPECIES_AGGREGATE);
124     pn.setGenusOrAbove("Achillea");
125     pn.setSpecificEpithet("millefolium");
126     pn.setAuthorship("L.");
127     assertFalse(pn.isIndetermined());
128   }

150   @Test
151   public void testSubgenus() throws Exception {
152     // Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013
153     ParsedName pn = new ParsedName();
154     pn.setGenusOrAbove("Brachyhypopomus");
155     pn.setInfraGeneric("Odontohypopomus");
156     pn.setAuthorship("Sullivan, Zuanon & Cox Fernandes");
157     pn.setYear("2013");
158     assertEquals("Odontohypopomus", pn.canonicalName());
159     assertEquals("Odontohypopomus", pn.canonicalNameWithMarker());
160     assertEquals("Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
161     // with given rank marker it is shown instead of brackets
162     pn.setRank(Rank.SUBGENUS);
163     assertEquals("Odontohypopomus", pn.canonicalName());
164     assertEquals("subgen. Odontohypopomus", pn.canonicalNameWithMarker());
165     assertEquals("Brachyhypopomus subgen. Odontohypopomus Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
166 
167     // Achillea sect. Ptarmica (Mill.) W.D.J.Koch
168     pn = new ParsedName();
169     pn.setGenusOrAbove("Achillea");
170     pn.setInfraGeneric("Ptarmica");
171     pn.setAuthorship("W.D.J.Koch");
172     pn.setBracketAuthorship("Mill.");
173 
174     assertEquals("Ptarmica", pn.canonicalName());
175     assertEquals("Achillea (Ptarmica) (Mill.) W.D.J.Koch", pn.canonicalNameComplete());
176     pn.setRank(Rank.SECTION);
177     assertEquals("Achillea sect. Ptarmica (Mill.) W.D.J.Koch", pn.canonicalNameComplete());
178   }

269   @Test
270   public void testJsonSerde() throws IOException {
271     ParsedName pn = new ParsedName();
272     pn.setGenusOrAbove("Brachyhypopomus");
273     pn.setInfraGeneric("Odontohypopomus");
274     pn.setAuthorship("Sullivan, Zuanon & Cox Fernandes");
275     pn.setYear("2013");
276     pn.setRank(Rank.SUBGENUS);
277     SerdeTestUtils.testSerDe(pn, ParsedName.class);
278 
279     pn = new ParsedName();
280     pn.setKey(123);
281     pn.setType(NameType.SCIENTIFIC);
282     pn.setGenusOrAbove("Abax");
283     pn.setSpecificEpithet("carinatus");
284     pn.setInfraSpecificEpithet("carinatus");
285     pn.setBracketAuthorship("Duftschmid");
286     pn.setBracketYear("1812");
287     pn.setRank(Rank.INFRASUBSPECIFIC_NAME);
288     SerdeTestUtils.testSerDe(pn, ParsedName.class);
289 
290     pn.setAuthorship("Carlos & Kamera");
291     pn.setYear("1999");
292 
293     for (Rank rank : Rank.values()) {
294       pn.setRank(rank);
295       // UNRANKED intentionally becomes null!
296       if (rank.notOtherOrUnknown()) {
297         String json = SerdeTestUtils.testSerDe(pn, ParsedName.class);
298         assertTrue(json.contains("\"rankMarker\""));
299       }
300     }
301     // check that rankMarker=null gets removed from json
302     pn.setRank(null);
303     String json = SerdeTestUtils.serialize(pn);
304     assertFalse(json.contains("\"rankMarker\""));
305   }

307   /**
308    * assert all build name methods return the same string
309    * @param pn
310    * @param name expected string
311    */
312   private void assertBuildName(ParsedName pn, String name) {
313     assertEquals(name, pn.fullName());
314     assertEquals(name, pn.canonicalName());
315     assertEquals(name, pn.canonicalNameComplete());
316     assertEquals(name, pn.canonicalNameWithMarker());
317   }

319   private void assertBuildName(ParsedName pn, String full, String canonical, String canonicalComplete, String canonicalMarker) {
320     assertEquals("wrong canonicalName", canonical, pn.canonicalName());
321     assertEquals("wrong canonicalNameWithMarker", canonicalMarker, pn.canonicalNameWithMarker());
322     assertEquals("wrong canonicalNameComplete", canonicalComplete, pn.canonicalNameComplete());
323     assertEquals("wrong fullName", full, pn.fullName());
324   }
