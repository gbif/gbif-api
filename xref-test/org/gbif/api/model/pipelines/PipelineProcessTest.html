1   package org.gbif.api.model.pipelines;

3   import java.time.LocalDateTime;
4   import java.util.Objects;
5   
6   import org.junit.Test;
7   
8   import static org.junit.Assert.assertEquals;
9   
10  /** Tests the {@link PipelineProcess}. */
11  public class PipelineProcessTest {
12  
13    @Test
14    public void pipelineProcessComparatorTest() {
15      PipelineProcess p1 = new PipelineProcess();
16      PipelineProcess p2 = new PipelineProcess();
17  
18      // process without steps
19      PipelineExecution e1 = new PipelineExecution().setCreated(LocalDateTime.now());
20      p1.addExecution(e1);
21      PipelineExecution e2 = new PipelineExecution().setCreated(LocalDateTime.now());
22      p2.addExecution(e2);
23      assertEquals(
24          0, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
25  
26      // one process without steps
27      PipelineStep p1e1s1 =
28          new PipelineStep()
29              .setStarted(LocalDateTime.now().minusMinutes(30))
30              .setState(PipelineStep.Status.RUNNING);
31      e1.addStep(p1e1s1);
32      assertEquals(
33          -1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
34  
35      // both processes with steps
36      PipelineStep p2e2s1 =
37          new PipelineStep()
38              .setStarted(LocalDateTime.now().minusMinutes(35))
39              .setState(PipelineStep.Status.RUNNING);
40      e2.addStep(p2e2s1);
41      assertEquals(
42          1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
43  
44      // completed steps have less preference over running ones
45      PipelineStep p1e1s2 =
46          new PipelineStep()
47              .setStarted(LocalDateTime.now().minusMinutes(40))
48              .setState(PipelineStep.Status.COMPLETED);
49      e1.addStep(p1e1s2);
50      assertEquals(
51          1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
52  
53      // steps with null values
54      PipelineStep p1e1s3 = new PipelineStep();
55      e1.addStep(p1e1s3);
56      assertEquals(
57          1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
58  
59      // steps with start date but no state
60      PipelineStep p1e1s4 = new PipelineStep().setStarted(LocalDateTime.now().minusMinutes(15));
61      e1.addStep(p1e1s4);
62      assertEquals(
63          1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
64    }
65  }
