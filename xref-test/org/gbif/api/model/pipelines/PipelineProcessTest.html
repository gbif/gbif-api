16  package org.gbif.api.model.pipelines;
17  
18  import java.time.LocalDateTime;
19  import java.util.Objects;
20  
21  import org.junit.jupiter.api.Test;
22  
23  import static org.junit.jupiter.api.Assertions.assertEquals;
24  
25  /** Tests the {@link PipelineProcess}. */
26  public class PipelineProcessTest {
27  
28    @Test
29    public void pipelineProcessComparatorByLatestStepTest() {
30      PipelineProcess p1 = new PipelineProcess();
31      PipelineProcess p2 = new PipelineProcess();
32  
33      // process without steps
34      PipelineExecution e1 = new PipelineExecution().setCreated(LocalDateTime.now());
35      p1.addExecution(e1);
36      PipelineExecution e2 = new PipelineExecution().setCreated(LocalDateTime.now());
37      p2.addExecution(e2);
38      assertEquals(
39          0, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
40  
41      // one process without steps
42      PipelineStep p1e1s1 =
43          new PipelineStep()
44              .setStarted(LocalDateTime.now().minusMinutes(30))
45              .setState(PipelineStep.Status.RUNNING);
46      e1.addStep(p1e1s1);
47      assertEquals(
48          -1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
49  
50      // both processes with steps
51      PipelineStep p2e2s1 =
52          new PipelineStep()
53              .setStarted(LocalDateTime.now().minusMinutes(35))
54              .setState(PipelineStep.Status.RUNNING);
55      e2.addStep(p2e2s1);
56      assertEquals(
57          1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
58  
59      // completed steps have less preference over running ones
60      PipelineStep p1e1s2 =
61          new PipelineStep()
62              .setStarted(LocalDateTime.now().minusMinutes(40))
63              .setState(PipelineStep.Status.COMPLETED);
64      e1.addStep(p1e1s2);
65      assertEquals(
66          1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
67  
68      // steps with null values
69      PipelineStep p1e1s3 = new PipelineStep();
70      e1.addStep(p1e1s3);
71      assertEquals(
72          1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
73  
74      // steps with start date but no state
75      PipelineStep p1e1s4 = new PipelineStep().setStarted(LocalDateTime.now().minusMinutes(15));
76      e1.addStep(p1e1s4);
77      assertEquals(
78          1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
79    }
80  
81    @Test
82    public void pipelineProcessComparatorByLatestExecutionTest() {
83      PipelineProcess p1 = new PipelineProcess();
84      PipelineProcess p2 = new PipelineProcess();
85  
86      assertEquals(0, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_EXEUCTION_ASC));
87  
88      PipelineExecution e1 = new PipelineExecution().setCreated(LocalDateTime.now().minusMinutes(20));
89      p1.addExecution(e1);
90      assertEquals(-1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_EXEUCTION_ASC));
91  
92      PipelineExecution e2 = new PipelineExecution().setCreated(LocalDateTime.now().minusMinutes(30));
93      p2.addExecution(e2);
94      assertEquals(1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_EXEUCTION_ASC));
95  
96      PipelineExecution e3 = new PipelineExecution().setCreated(LocalDateTime.now().minusMinutes(10));
97      p2.addExecution(e3);
98      assertEquals(-1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_EXEUCTION_ASC));
99    }
100 }
