14  package org.gbif.api.model.predicate;
15  
16  import com.fasterxml.jackson.databind.ObjectMapper;
17  import com.fasterxml.jackson.databind.module.SimpleModule;
18  import java.io.File;
19  import java.io.IOException;
20  import org.gbif.api.model.common.search.SearchParameter;
21  import org.gbif.api.model.event.search.EventSearchParameter;
22  import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
23  import org.junit.jupiter.api.Assertions;
24  import org.junit.jupiter.api.Test;
25  
26  /** Test cases and examples of serialization or predicates using a mixing. */
27  public class PredicateDeSerTest {
28  
29    private static final ObjectMapper MAPPER = new ObjectMapper();
30  
31    static {
32      MAPPER.registerModule(
33          new SimpleModule()
34              .addDeserializer(
35                  SearchParameter.class, new SearchParameter.SearchParameterDeserializer())
36              .addDeserializer(
37                  OccurrenceSearchParameter.class,
38                  new OccurrenceSearchParameter.OccurrenceSearchParameterDeserializer())
39              .addDeserializer(
40                  EventSearchParameter.class,
41                  new EventSearchParameter.EventSearchParameterDeserializer()));
42    }
43  
44    private File getTestFile(String predicateFile) {
45      return new File(getClass().getResource("/predicate/" + predicateFile).getFile());
46    }
47  
48    @Test
49    public void deserTest() throws IOException {
50      EqualsPredicate<OccurrenceSearchParameter> eq =
51          new EqualsPredicate<>(OccurrenceSearchParameter.OCCURRENCE_STATUS, "present", false);
52      System.out.println(MAPPER.writeValueAsString(eq));
53  
54      assertPredicate("is_null.json");
55      assertPredicate("conjunction.json");
56      assertPredicate("within.json");
57      assertPredicate("equals_catalog_number.json");
58      assertPredicate("equals_date_range.json");
59      assertPredicate("like_catalog_number.json");
60      assertPredicate("and_with_not.json");
61      assertPredicate("conjunction_with_in.json");
62      assertPredicate("complex_conjunction_with_in.json");
63      assertPredicate("range.json");
64      assertPredicate("distance.json");
65      assertPredicate("is_null_event.json");
66    }
67  
68    @Test
69    public void eventSearchParameterTest() throws IOException {
70      ObjectMapper eventsMapper =
71          new ObjectMapper()
72              .registerModule(
73                  new SimpleModule()
74                      .addDeserializer(
75                          SearchParameter.class,
76                          new EventSearchParameter.EventSearchParameterDeserializer())
77                      .addDeserializer(
78                          EventSearchParameter.class,
79                          new EventSearchParameter.EventSearchParameterDeserializer()));
80  
81      Predicate predicate = eventsMapper.readValue(getTestFile("equals_event.json"), Predicate.class);
82      Assertions.assertEquals(
83          EventSearchParameter.YEAR, ((EqualsPredicate<EventSearchParameter>) predicate).getKey());
84    }
85  
86    private void assertPredicate(String fileName) throws IOException {
87      Predicate predicate = MAPPER.readValue(getTestFile(fileName), Predicate.class);
88      Assertions.assertNotNull(predicate);
89    }
90  }
