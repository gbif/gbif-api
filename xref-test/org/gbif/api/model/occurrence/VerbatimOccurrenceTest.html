61    @Test
62    public void testHasField() {
63      VerbatimOccurrence occ = new VerbatimOccurrence();
64  
65      occ.setVerbatimField(DwcTerm.catalogNumber, "abc123");
66      assertTrue(occ.hasVerbatimField(DwcTerm.catalogNumber));
67      assertFalse(occ.hasVerbatimField(DwcTerm.institutionCode));
68  
69      occ.setVerbatimField(DwcTerm.catalogNumber, " ");
70      assertTrue(occ.hasVerbatimField(DwcTerm.catalogNumber));
71  
72      occ.setVerbatimField(DwcTerm.catalogNumber, "");
73      assertFalse(occ.hasVerbatimField(DwcTerm.catalogNumber));
74  
75      occ.setVerbatimField(DwcTerm.catalogNumber, null);
76      assertFalse(occ.hasVerbatimField(DwcTerm.catalogNumber));
77    }

79    @Test
80    public void testFieldFromTermFactory() {
81      VerbatimOccurrence occ = new VerbatimOccurrence();
82      String prefix = "t_";
83      String colName = "t_dwc:basisOfRecord";
84      Term term = TermFactory.instance().findTerm(colName.substring(prefix.length()));
85      occ.setVerbatimField(term, "PreservedSpecimen");
86      assertEquals(1, occ.getVerbatimFields().size());
87      assertTrue(occ.hasVerbatimField(DwcTerm.basisOfRecord));
88      assertEquals("PreservedSpecimen", occ.getVerbatimField(DwcTerm.basisOfRecord));

149   @Test
150   public void testVerbatimMapSerde() throws Exception {
151     ObjectMapper mapper = new ObjectMapper();
152     mapper.enable(DeserializationConfig.Feature.ACCEPT_EMPTY_STRING_AS_NULL_OBJECT);
153     mapper.enable(SerializationConfig.Feature.INDENT_OUTPUT);
154     mapper.disable(SerializationConfig.Feature.WRITE_NULL_PROPERTIES);
155 
156     VerbatimOccurrence v = new VerbatimOccurrence();
157     v.setKey(7l);
158     v.setLastParsed(new Date());
159     v.setDatasetKey(UUID.randomUUID());
160 
161     for (Term term : DwcTerm.values()) {
162       v.setVerbatimField(term, RandomStringUtils.randomAlphabetic(20));
163     }
164     final int numDwcTerms = v.getVerbatimFields().size();
165 
166     for (Term term : DcTerm.values()) {
167       v.setVerbatimField(term, RandomStringUtils.randomAlphabetic(20));
168     }
169     final int numDcTerms = v.getVerbatimFields().size() - numDwcTerms;
170 
171     for (Term term : GbifTerm.values()) {
172       v.setVerbatimField(term, RandomStringUtils.randomAlphabetic(20));
173     }
174     final int numGbifTerms = v.getVerbatimFields().size() - numDwcTerms - numDcTerms;
175 
176     for (Term term : IucnTerm.values()) {
177       v.setVerbatimField(term, RandomStringUtils.randomAlphanumeric(20));
178     }
179     final int numIucnTerms = v.getVerbatimFields().size() - numDwcTerms - numDcTerms - numGbifTerms;
180 
181     v.setVerbatimField(DwcTerm.scientificName, "Abies alba");
182     v.setVerbatimField(DwcTerm.collectionCode, "BUGS");
183     v.setVerbatimField(DwcTerm.catalogNumber, "MD10782");
184     v.setVerbatimField(UnknownTerm.build("http://rs.un.org/terms/temperatur"), RandomStringUtils.randomAlphabetic(30));
185     v.setVerbatimField(UnknownTerm.build("http://rs.un.org/terms/co2"), RandomStringUtils.randomAlphabetic(30));
186     v.setVerbatimField(UnknownTerm.build("http://rs.un.org/terms/modified"), new Date().toString());
187     v.setVerbatimField(UnknownTerm.build("http://rs.un.org/terms/scientificName"),
188       RandomStringUtils.randomAlphabetic(30));
189     final int numOtherTerms = v.getVerbatimFields().size() - numDwcTerms - numDcTerms - numGbifTerms - numIucnTerms;
190 
191     final int numTerms = v.getVerbatimFields().size();
192 
193     assertEquals(numTerms, numDwcTerms + numDcTerms + numGbifTerms + numIucnTerms + numOtherTerms);
194 
195     String json = mapper.writeValueAsString(v);
196     System.out.println(json);
197 
198     VerbatimOccurrence v2 = mapper.readValue(json, VerbatimOccurrence.class);
199     // 5 terms
200     assertEquals(numTerms, v2.getVerbatimFields().size());
201 
202   }

204   @Test
205   public void testVerbatimExtensionsMapSerde() throws Exception {
206     ObjectMapper mapper = new ObjectMapper();
207     mapper.enable(DeserializationConfig.Feature.ACCEPT_EMPTY_STRING_AS_NULL_OBJECT);
208     mapper.enable(SerializationConfig.Feature.INDENT_OUTPUT);
209     mapper.disable(SerializationConfig.Feature.WRITE_NULL_PROPERTIES);
210 
211     Map<Term, String> verbatimRecord = new HashMap<Term, String>();
212     Date today = new Date();
213     verbatimRecord.put(DcTerm.created, IsoDateFormat.FULL.getDateFormat().format(today));
214     verbatimRecord.put(DcTerm.creator, "fede");
215     verbatimRecord.put(DcTerm.description, "testDescription");
216     verbatimRecord.put(DcTerm.format, "jpg");
217     verbatimRecord.put(DcTerm.license, "licenseTest");
218     verbatimRecord.put(DcTerm.publisher, "publisherTest");
219     verbatimRecord.put(DcTerm.title, "titleTest");
220     verbatimRecord.put(DcTerm.references, "http://www.gbif.org/");
221     verbatimRecord.put(DcTerm.identifier, "http://www.gbif.org/");
222 
223     VerbatimOccurrence v = new VerbatimOccurrence();
224     v.setKey(7l);
225     v.setLastParsed(new Date());
226     v.setDatasetKey(UUID.randomUUID());
227     Map<Extension, List<Map<Term, String>>> extensions = new HashMap<Extension, List<Map<Term, String>>>();
228     List<Map<Term, String>> verbatimRecords = new ArrayList<>();
229     verbatimRecords.add(verbatimRecord);
230     extensions.put(Extension.MULTIMEDIA, verbatimRecords);
231     v.setExtensions(extensions);
232 
233     String json = mapper.writeValueAsString(v);
234     System.out.println(json);
235 
236     VerbatimOccurrence v2 = mapper.readValue(json, VerbatimOccurrence.class);
237     assertNotNull(v2.getExtensions());
238     assertTrue(!v2.getExtensions().get(Extension.MULTIMEDIA).isEmpty());
239     assertEquals(v2.getExtensions().get(Extension.MULTIMEDIA).get(0), verbatimRecord);
240   }
