33    private static final OccurrenceSearchParameter TEST_KEY = OccurrenceSearchParameter.CATALOG_NUMBER;
34    private static final String TEST_VALUE = "bar";
35    private static final OccurrenceSearchParameter COMP_PARAM = OccurrenceSearchParameter.ELEVATION;
36    private static final String COMP_VALUE = "100";
37  
38    @Test
39    public void testEmptyConstructor() {
40      assertThrows(IllegalArgumentException.class, () -> new EqualsPredicate(TEST_KEY, "", false));
41    }
42  
43    @Test
44    public void testEquals() {
45      SimplePredicate ep1 = new EqualsPredicate(TEST_KEY, TEST_VALUE, false);
46      SimplePredicate ep2 = new EqualsPredicate(TEST_KEY, TEST_VALUE, false);
47      SimplePredicate gt1 = new GreaterThanPredicate(COMP_PARAM, COMP_VALUE);
48      SimplePredicate gt2 = new GreaterThanPredicate(COMP_PARAM, COMP_VALUE);
49      SimplePredicate lt1 = new LessThanPredicate(COMP_PARAM, COMP_VALUE);
50      SimplePredicate lt2 = new LessThanPredicate(COMP_PARAM, COMP_VALUE);
51  
52      // All of them are equal to themselves
53      assertThat(ep1, equalTo(ep1));
54      assertThat(gt1, equalTo(gt1));
55      assertThat(lt1, equalTo(lt1));
56  
57      // But unequal among themselves
58      assertThat(ep1, both(not(equalTo(gt1))).and(not(equalTo(lt1))));
59      assertThat(gt1, both(not(equalTo(ep1))).and(not(equalTo(lt1))));
60      assertThat(lt1, both(not(equalTo(ep1))).and(not(equalTo(gt1))));
61  
62      // And certainly unequal to any other Predicate
63      assertThat(ep1, not(equalTo(mock(Predicate.class))));
64  
65      assertThat(ep1, equalTo(ep2));
66      assertThat(gt1, equalTo(gt2));
67      assertThat(lt1, equalTo(lt2));
68  
69      ep2 = new EqualsPredicate(TEST_KEY, "FOOBAR", false);
70      assertThat(ep1, not(equalTo(ep2)));
71    }
72  
73    @Test
74    public void testGoodDoubleValue() {
75      new EqualsPredicate(OccurrenceSearchParameter.DECIMAL_LATITUDE, "32", false);
76      new EqualsPredicate(OccurrenceSearchParameter.DECIMAL_LATITUDE, "32.032", false);
77      new EqualsPredicate(OccurrenceSearchParameter.ELEVATION, "-10,5.5", false);
78      new EqualsPredicate(OccurrenceSearchParameter.ELEVATION, "*,200", false);
79      new EqualsPredicate(OccurrenceSearchParameter.ELEVATION, "100,*", false);
80    }

87    @Test
88    public void testHashcode() {
89      SimplePredicate sp1 = new EqualsPredicate(TEST_KEY, TEST_VALUE, false);
90      SimplePredicate sp2 = new EqualsPredicate(TEST_KEY, TEST_VALUE, false);
91      SimplePredicate sp3 = new LessThanPredicate(COMP_PARAM, COMP_VALUE);
92      SimplePredicate sp4 = new GreaterThanPredicate(COMP_PARAM, COMP_VALUE);
93  
94      assertThat(sp1.hashCode(), equalTo(sp2.hashCode()));
95      assertThat(sp1.hashCode(), not(equalTo(sp3.hashCode())));
96      assertThat(sp3.hashCode(), equalTo(sp4.hashCode()));
97    }
