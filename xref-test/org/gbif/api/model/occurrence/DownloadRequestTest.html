156   @Test
157   public void testSerde() throws IOException {
158     PredicateDownloadRequest d = newDownload(new EqualsPredicate(OccurrenceSearchParameter.CATALOG_NUMBER, "b", false));
159     PredicateDownloadRequest d2;
160     try (ByteArrayOutputStream baos = new ByteArrayOutputStream()) {
161       MAPPER.writeValue(baos, d);
162       d2 = MAPPER.readValue(baos.toByteArray(), PredicateDownloadRequest.class);

164       fail(e.getMessage());
165       throw e;
166     }
167 
168     assertEquals(d, d2);

336   /**
337    * Test three extension situations: known and supported, known but not supported, unknown.
338    */
339   @Test
340   public void downloadWithExtensionTest() throws Exception {
341     String requestTemplate = "{\n"
342       + "  \"creator\":\"gbif_user\",\n"
343       + "  \"notification_address\":[\"gbif_user@mailinator.com\"],\n"
344       + "  \"created\":\"2024\",\n"
345       + "  \"format\": \"DWCA\",\n"
346       + "  \"predicate\":{\n"
347       + "    \"type\":\"equals\",\n"
348       + "    \"key\":\"TAXON_KEY\",\n"
349       + "    \"value\":\"212\"\n"
350       + "  },\n"
351       + "  %s\n"
352       + "}";
353 
354     DownloadRequest request;
355 
356     // Known and supported extension
357     request = MAPPER.readValue(String.format(requestTemplate, "'verbatimExtensions':['http://rs.tdwg.org/dwc/terms/MeasurementOrFact']".replace("'", "\"")), DownloadRequest.class);
358     assertEquals("gbif_user", request.getCreator());
359     assertEquals(DownloadFormat.DWCA, request.getFormat());
360     assertEquals("gbif_user@mailinator.com", request.getNotificationAddressesAsString());
361     assertEquals(EqualsPredicate.class, ((PredicateDownloadRequest) request).getPredicate().getClass());
362     assertEquals("212", ((EqualsPredicate) ((PredicateDownloadRequest) request).getPredicate()).getValue());
363     assertEquals(Extension.MEASUREMENT_OR_FACT, ((PredicateDownloadRequest) request).getVerbatimExtensions().iterator().next());
364 
365     // Known and *unsupported* extension
366     try {
367       MAPPER.readValue(String.format(requestTemplate, "'verbatimExtensions':['http://zooarchnet.org/dwc/terms/ChronometricDate']".replace("'", "\"")), DownloadRequest.class);
368       fail();
369     } catch (Exception e) {
370     }
371 
372     // Unknown extension
373     try {
374       MAPPER.readValue(String.format(requestTemplate, "'verbatimExtensions':['http://example.org/nothing']".replace("'", "\"")), DownloadRequest.class);
375       fail();
376     } catch (Exception e) {
377     }
378 
379     // Extension enum value — not supported as these are a bit too internal
380     try {
381       MAPPER.readValue(String.format(requestTemplate, "'verbatimExtensions':['MEASUREMENT_OR_FACT']".replace("'", "\"")), DownloadRequest.class);
382       fail();
383     } catch (Exception e) {
384     }
385 
386     // Null value
387     try {
388       MAPPER.readValue(String.format(requestTemplate, "'verbatimExtensions':[null]".replace("'", "\"")), DownloadRequest.class);
389       fail();
390     } catch (Exception e) {
391     }
392   }
