25    @Test
26    public void testValidations() {
27      ValidatorFactory validatorFactory =
28        Validation.byProvider(ApacheValidationProvider.class).configure().buildValidatorFactory();
29      Validator validator = validatorFactory.getValidator();
30  
31      Dataset ds = new Dataset();
32  
33      // 3 non-mandatory fields that don't validate
34      ds.setTitle("B"); // too short
35      ds.setHomepage(URI.create("www.gbif.org")); // doesn't start with http or https
36      ds.setLogoUrl(URI.create("file:///tmp/aha")); // bad http URI
37  
38      // All mandatory fields missing include:
39      // ds.setType(?);
40      // ds.setInstallationKey(?);
41      // ds.setPublishingOrganizationKey(?);
42  
43      // perform validation
44      Set<ConstraintViolation<Dataset>> violations = validator.validate(ds);
45      assertTrue("Violations were expected", !violations.isEmpty());
46  
47      // ensure all 6 expected violations are caught
48      Set<String> propertiesInViolation =
49        Sets.newHashSet("title", "homepage", "logoUrl", "type", "installationKey", "publishingOrganisationKey");
50      assertEquals(6, violations.size());
51      for (ConstraintViolation<?> cv : violations) {
52        propertiesInViolation.contains(cv.getPropertyPath().toString());
53      }
54  
55      // fix non-mandatory fields that don't validate
56      ds.setTitle("Rooftop bugs");
57      ds.setHomepage(URI.create("http://www.gbif.org"));
58      ds.setLogoUrl(URI.create("http://www.gbif.org/logo.png"));
59  
60      // add all mandatory fields that were missing
61      ds.setType(DatasetType.SAMPLING_EVENT);
62      ds.setInstallationKey(UUID.randomUUID());
63      ds.setPublishingOrganizationKey(UUID.randomUUID());
64  
65      // perform validation again
66      violations = validator.validate(ds);
67      assertTrue("No violations were expected", violations.isEmpty());
68    }

70    @Test
71    public void testLenientEquals() {
72      Dataset ds1 = new Dataset();
73      ds1.setMaintenanceUpdateFrequency(MaintenanceUpdateFrequency.DAILY);
74      ds1.setMaintenanceDescription("Daily, except for holidays");
75      ds1.setLicense(License.CC_BY_4_0);
76      ds1.setCreated(new Date());
77  
78      Dataset ds2 = new Dataset();
79      ds2.setMaintenanceUpdateFrequency(MaintenanceUpdateFrequency.DAILY);
80      ds2.setMaintenanceDescription("Daily, except for holidays");
81      ds2.setLicense(License.CC0_1_0); // different license
82      ds1.setCreated(new Date()); // different created date!
83  
84      assertTrue(ds1.lenientEquals(ds2)); // true because lenient equals excludes
85    }
