31  public class UserTest {
32    private Validator validator = Validation.buildDefaultValidatorFactory().getValidator();
33  
34    @Test
35    public void testHasRole() throws Exception {
36      User u = new User();
37      u.setUserName("betty");
38      u.setEmail("betty@gbif.org");
39      u.setFirstName("Betty Ford");
40  
41      assertFalse(u.hasRole(UserRole.USER));
42      assertFalse(u.hasRole(UserRole.REGISTRY_EDITOR));
43      assertFalse(u.hasRole(UserRole.ADMIN));
44      assertFalse(u.hasRole(null));
45  
46      u.addRole(UserRole.USER);
47      assertTrue(u.hasRole(UserRole.USER));
48      assertFalse(u.hasRole(UserRole.REGISTRY_EDITOR));
49      assertFalse(u.hasRole(UserRole.ADMIN));
50      assertFalse(u.hasRole(null));
51  
52      u.addRole(UserRole.ADMIN);
53      assertTrue(u.hasRole(UserRole.USER));
54      assertFalse(u.hasRole(UserRole.REGISTRY_EDITOR));
55      assertTrue(u.hasRole(UserRole.ADMIN));
56      assertFalse(u.hasRole(null));
57    }
58  
59    @Test
60    public void testIsAdmin() throws Exception {
61      User u = new User();
62      u.setUserName("betty");
63      u.setEmail("betty@gbif.org");
64      u.setFirstName("Betty Ford");
65  
66      assertFalse(u.isAdmin());
67  
68      u.addRole(UserRole.ADMIN);
69      assertTrue(u.isAdmin());
70    }

72    @Test
73    public void testValidation() throws Exception {
74      User u = new User();
75      u.setKey(100);
76      u.setUserName("be");
77      u.setEmail("betty@gbif.org");
78      u.setFirstName("Betty");
79      u.setLastName("Ford");
80      assertValidation("userName", u);
81  
82      u.setUserName("betty");
83      u.setFirstName(null);
84      assertValidation("firstName", u);
85  
86      u.setFirstName("Betty");
87      assertValidation(null, u);
88  
89      u.setEmail("betty at gbif.org");
90      assertValidation("email", u);
91    }
92  
93    private void assertValidation(String property, User u) {
94      Set<ConstraintViolation<User>> err = validator.validate(u);
95      System.out.print(err);
96      if (property == null) {
97        assertEquals(0, err.size());
98      } else {
99        assertEquals(1, err.size());
100       assertEquals(property, err.iterator().next().getPropertyPath().toString());
101     }
102   }
