015  @Test
016  public void testLineComparatorCSV() {
017    LineComparator comp = new LineComparator(3, ",", '"');
018    List<String> lines = new ArrayList<String>();
019    String l1 = "121,432423,9099053,\"Frieda karla L.,Ahrens\"";
020    String l2 = "adshjhg,fsdfsd,fdsfdsfsd,Forunkel forunculus,Janssen";
021    String l3 = ",,,,zzz";
022    String l4 = "321,2453,432423,berndef,ahrene32";
023    lines.add(l1);
024    lines.add(l2);
025    lines.add(l3);
026    lines.add(l4);
027
028    Collections.sort(lines, comp);
029
030    assertEquals(l2, lines.get(0));
031    assertEquals(l1, lines.get(1));
032    assertEquals(l4, lines.get(2));
033    assertEquals(l3, lines.get(3));
034  }

078  // Direct copy from From IPT codebase
079  // see https://code.google.com/p/gbif-providertoolkit/source/browse/trunk/gbif-ipt/src/main/java/org/gbif/ipt/task/GenerateDwca.java#93
080  private static final Comparator<String> IGNORE_CASE_COMPARATOR = Ordering.from(new Comparator<String>() {
081
082    public int compare(String o1, String o2) {
083      return o1.compareToIgnoreCase(o2);
084    }
085  }).nullsFirst();
086
087  /**
088   * Test for respecting equals and compareTo to ensure comparator respects java contracts.
089   * http://dev.gbif.org/issues/browse/POR-2730
090   */
091  @Test
092  public void testEqualsCompareToContract() throws IOException {
093
094    LineComparator comp = new LineComparator(0, ",", null, IGNORE_CASE_COMPARATOR);
095
096    String s1 = ",2,3"; // null in first instance
097    String s2 = ",2,3"; // null in first instance
098
099    // http://docs.oracle.com/javase/7/docs/api/java/util/Comparator.html#compare(T,%20T)
100    int sign1 = (int) Math.signum(comp.compare(s1, s2));
101    int sign2 = (int) Math.signum(comp.compare(s2, s1));
102
103    assertEquals(sign1 * -1, sign2);
104
105  }
