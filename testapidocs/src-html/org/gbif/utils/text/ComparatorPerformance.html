017package org.gbif.utils.text;
018
019import org.gbif.utils.file.FileUtils;
020
021import java.io.File;
022import java.io.IOException;
023import java.util.Arrays;
024import java.util.Collections;
025import java.util.Comparator;
026import java.util.LinkedList;
027import java.util.List;
028
029import com.google.common.collect.Lists;
030import com.google.common.collect.Ordering;
031import org.junit.Test;
032
033/**
034 * @author markus
035 */
036public class ComparatorPerformance {
037
038  private final String ENCODING = "UTF-8";
039
040  /**
041   * Comparing performance for various file soring methods.
042   * <p/>
043   * Executed on a MacPro with 9GB, 8-core 3GHz and 1TB disk
044   * sorting a 207MB large text file made from concatenation of irmng.tail:
045   * <p/>
046   * Sorting with unix sort took 5817 ms
047   * Sorting with org.gbif.utils.text.StringComparator and 10k lines in memory (200 parts) took 48968 ms
048   * Sorting with org.gbif.utils.text.CCollationComparator and 10k lines in memory (200 parts) took 49858 ms
049   * Sorting with org.gbif.utils.text.StringComparator and 100k lines in memory (20 parts) took 17962 ms
050   * Sorting with org.gbif.utils.text.CCollationComparator and 100k lines in memory (20 parts) took 14046 ms
051   * Sorting with org.gbif.utils.text.StringComparator and 1000k lines in memory (2 parts) took 15492 ms
052   * Sorting with org.gbif.utils.text.CCollationComparator and 1000k lines in memory (2 parts) took 14317 ms
053   */
054
055  @Test
056  public void testFileSorting() throws IOException, InstantiationException, IllegalAccessException {
057    // 10MB text file, big file used in results above was concatenated from this one
058    File source = FileUtils.getClasspathFile("sorting/irmng.tail");
059    File sorted = File.createTempFile("gbif-common-file-sort2", "sorted.txt");
060    sorted.deleteOnExit();
061    FileUtils futils = new FileUtils();
062
063    // test unix sort
064    long start = System.currentTimeMillis();
065    futils.sort(source, sorted, ENCODING, 0, "\t", null, "\n", 0);
066    long end = System.currentTimeMillis();
067    System.out.println(String.format("Sorting with unix sort took %s ms", (end - start)));
068
069    // sort with comparator to test
070    List<Comparator<String>> comparators = availableComparators();
071
072    for (Integer linesInMen : Arrays.asList(100000)) {
073      futils.setLinesPerMemorySort(linesInMen);
074      for (Comparator<String> comp : comparators) {
075
076        start = System.currentTimeMillis();
077        futils.sortInJava(source, sorted, ENCODING, comp, 0);
078        end = System.currentTimeMillis();
079
080        System.out.println(
081          String.format("Sorting with %s and %s lines in memory took %s ms", comp.getClass().getName(), linesInMen, (end - start)));
082      }
083    }
084  }
085
086  private List<Comparator<String>> availableComparators() {
087    List<Comparator<String>> comparators = Lists.newArrayList();
088    comparators.add(new StringComparator());
089    comparators.add(Ordering.<String>natural().nullsFirst());
090    return comparators;
091  }
092
093  @Test
094  public void testVariousComparators() throws IOException, InstantiationException, IllegalAccessException {
095    // sort with comparator to test
096    List<Comparator<String>> comparators = availableComparators();
097    for (Comparator<String> comp : comparators) {
098      LinkedList<String> source = FileUtils.streamToList(FileUtils.classpathStream("sorting/irmng.tail"));
099      long start = System.currentTimeMillis();
100      Collections.sort(source, comp);
101      long end = System.currentTimeMillis();
102      System.out.println("Sorting with " + comp.getClass().getName() + " took " + (end - start) + " ms");
103    }
104  }
105}
