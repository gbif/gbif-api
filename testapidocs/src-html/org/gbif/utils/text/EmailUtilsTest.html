017package org.gbif.utils.text;
018
019import org.gbif.utils.text.EmailUtils.EmailWithName;
020
021import org.junit.Test;
022
023import static org.junit.Assert.assertEquals;
024import static org.junit.Assert.assertTrue;
025
026/**
027 * @author markus
028 */
029public class EmailUtilsTest {
030
031  @Test
032  public void testEmailPatterns() {
033    EmailWithName en = EmailUtils.parseEmail("mdoering@gbif.org");
034    assertTrue(en.name == null);
035    assertEquals("mdoering@gbif.org", en.email);
036
037    en = EmailUtils.parseEmail("Manuel Möglich <manuel.moeglich@googlemail.com>");
038    assertEquals("Manuel Möglich", en.name);
039    assertEquals("manuel.moeglich@googlemail.com", en.email);
040
041    en = EmailUtils.parseEmail("Manuel Möglich");
042    assertEquals("Manuel Möglich", en.name);
043    assertTrue(en.email == null);
044
045    en = EmailUtils.parseEmail("Manuel Möglich manuel.moeglich@googlemail.com");
046    assertEquals("Manuel Möglich", en.name);
047    assertEquals("manuel.moeglich@googlemail.com", en.email);
048
049  }
050}
