017package org.gbif.utils.file;
018
019import java.io.File;
020import java.io.IOException;
021import java.nio.charset.Charset;
022
023import org.apache.commons.lang3.StringUtils;
024import org.junit.Test;
025
026import static org.gbif.utils.file.CharsetDetection.detectEncoding;
027
028import static org.junit.Assert.assertEquals;
029
030/**
031 * @author markus
032 */
033public class CharsetDetectionTest {
034
035  @Test
036  public void testCP1252Encoding() throws IOException {
037    File test = FileUtils.getClasspathFile("charsets/cp1252-test.txt");
038
039    Charset encoding = detectEncoding(test);
040
041    assertEquals("windows1252", encoding.displayName().replace("-", "").toLowerCase());
042  }
043
044
045  @Test
046  public void testEncodingDetection() throws IOException {
047    String[] files =
048      new String[] {"iso-8859-1_names.txt", "macroman_names.txt", "utf-16BE_bom_names.txt", "utf-16BE_names.txt",
049        "utf-16LE_bom_names.txt", "utf-16LE_names.txt", "utf-8_bom_names.txt", "utf-8_names.txt",
050        "windows1252_names.txt"};
051    for (String fn : files) {
052      File test = FileUtils.getClasspathFile("charsets/" + fn);
053
054      Charset encoding = detectEncoding(test);
055      String expected = StringUtils.substringBefore(fn, "_");
056
057      // x-MacRoman is alias for MacRoman used on unix, therefore remove x-
058      assertEquals(expected.replace("-", "").toLowerCase(),
059        encoding.displayName().toLowerCase().replace("x-mac", "mac").replace("-", ""));
060    }
061  }
062
063  @Test
064  public void testEncodingDetectionKyles() throws IOException {
065    String[] files =
066      new String[] {"utf-8_arabic.csv", "utf-8_japanese.csv", "utf-8_korean.csv", "utf-8_latin.csv", "utf-8_no-bom.csv",
067        "utf-8_traditional-chinese.csv", "utf-8_.csv", "utf-16BE_no-bom.csv", "utf-16BE_.csv",
068        "utf-16LE_little-endian-no-bom.csv", "utf-16LE_little-endian.csv"};
069    for (String fn : files) {
070      File test = FileUtils.getClasspathFile("charsets/kyle/" + fn);
071
072      Charset encoding = detectEncoding(test);
073      String expected = StringUtils.substringBefore(fn, "_");
074
075      assertEquals(expected.replace("-", "").toLowerCase(), encoding.displayName().toLowerCase().replace("-", ""));
076    }
077  }
078}
