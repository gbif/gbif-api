001package org.gbif.utils.file;
002
003import java.io.File;
004import java.io.FileNotFoundException;
005import java.io.IOException;
006
007import com.google.common.collect.Sets;
008import org.junit.Test;
009
010import static org.junit.Assert.assertEquals;
011
012public class ResourcesUtilTest {
013
014  private InputStreamUtils isu = new InputStreamUtils();
015
016  @Test
017  public void testCopyResources() throws Exception {
018    File tmp = FileUtils.createTempDir();
019    tmp.deleteOnExit();
020    ResourcesUtil.copy(tmp, "", true, "testNOT_EXISTING.txt", "test1/test.txt", "test1/test2/test.txt");
021
022    // test
023    File t1 = new File(tmp, "test1/test.txt");
024    File t2 = new File(tmp, "test1/test2/test.txt");
025    org.apache.commons.io.FileUtils.contentEquals(t1, t2);
026    assertTestFile(t1);
027    assertTestFile(t2);
028  }
029
030  @Test(expected = IOException.class)
031  public void testCopyResourcesThrowing() throws Exception {
032    File tmp = FileUtils.createTempDir();
033    tmp.deleteOnExit();
034    ResourcesUtil.copy(tmp, "", false, "testNOT_EXISTING.txt", "test1/test.txt", "test1/test2/test.txt");
035
036    // test
037    File t1 = new File(tmp, "test1/test.txt");
038    File t2 = new File(tmp, "test1/test2/test.txt");
039    org.apache.commons.io.FileUtils.contentEquals(t1, t2);
040    assertTestFile(t1);
041    assertTestFile(t2);
042  }
043
044  @Test
045  public void testList() throws Exception {
046    assertEquals(Sets.newHashSet("test.txt", "test2"), Sets.newHashSet(ResourcesUtil.list(ResourcesUtil.class, "test1")));
047    assertEquals(Sets.newHashSet("test.txt"), Sets.newHashSet(ResourcesUtil.list(ResourcesUtil.class, "test1/test2")));
048    assertEquals(Sets.newHashSet("utf16be.xml", "utf16le.xml", "utf8.xml", "utf8bom.xml"), Sets.newHashSet(ResourcesUtil.list(ResourcesUtil.class, "sax")));
049    assertEquals(Sets.newHashSet(), Sets.newHashSet(ResourcesUtil.list(ResourcesUtil.class, "abba")));
050  }
051
052  private void assertTestFile(File tf) throws FileNotFoundException {
053    assertEquals("hallo\n", isu.readEntireStream(FileUtils.getInputStream(tf)));
054  }
055}
