052    while (it.hasNext()) {
053      String x = it.next();

055      assertEquals(sorted.poll(), x);
056    }
057  }

201  @Test
202  public void testSortingWithNonFirstIdColumn() throws IOException {
203    File source = FileUtils.getClasspathFile("sorting/TDB_104.csv");
204    File sorted = File.createTempFile("gbif-common-file-sort", "sorted.txt");
205    sorted.deleteOnExit();
206    FileUtils futils = new FileUtils();

208
209    // read file
210    BufferedReader br = new BufferedReader(new InputStreamReader(new FileInputStream(sorted), "UTF-8"));
211    int line = 0;
212    while (true) {
213      line++;
214      String row = br.readLine();
215      if (row == null) {
216        break;
217      }
218
219      if (line == 1) {
220        assertEquals(
221          "taxonRank;scientificName;scientificNameAuthorship;taxonID;parentNameUsageID;vernacularName;taxonomicStatus",
222          row);
223      } else if (line == 2) {
224        // row 2 and 3 have the same ids - only test if the id is correct (actual order of those 2 records can differ)
225        Iterator<String> columns = Splitter.on(";").split(row).iterator();
226        columns.next();
227        columns.next();
228        columns.next();
229        assertEquals("urn:lsid:luomus.fi:taxonconcept:0071b855-3d23-4fdc-b2e0-8464c22d752a:1", columns.next());
230
231      } else if (line == 100) {
232        assertEquals(
233          "species;Ctenochira angulata;(Thomson, 1883) ;urn:lsid:luomus.fi:taxonconcept:4adcf436-a0d2-4940-9155-220ffc6f5859:1;urn:lsid:luomus.fi:taxonconcept:817994ea-b58b-4deb-973f-9fa99c537f8a:1;;valid",
234          row);
235      }
236    }

239  /**
240   * Test that ensures the chunk file is deleted at the end of sortInJava method. Otherwise, unwanted chunk files
241   * will be left over.
242   */
243  @Test
244  public void testSortInJava() throws IOException {
245    File source = FileUtils.getClasspathFile("sorting/taxon.txt");
246    File sorted = File.createTempFile("gbif-common-file-sort", "taxon_sorted.txt");
247    sorted.deleteOnExit();
248    FileUtils futils = new FileUtils();
249    Comparator<String> lineComparator = new LineComparator(0, "\t");
250    futils.sortInJava(source, sorted, ENCODING, lineComparator, 1);
251
252    // the chunk file should NOT exist
253    File chunkFile = new File(source.getParent(), "taxon_0txt");
254    assertFalse(chunkFile.exists());
255
256    // the sorted file should exist
257    System.out.println(sorted.getAbsolutePath());
258    assertTrue(sorted.exists());
259
260    // read file
261    BufferedReader br = new BufferedReader(new InputStreamReader(new FileInputStream(sorted), "UTF-8"));
262    int line = 0;
263    while (true) {
264      line++;
265      String row = br.readLine();
266      if (row == null) {
267        break;
268      }
269      // first line (smallest ID)
270      if (line == 2) {
271        assertTrue(row.startsWith("118701359"));
272      }
273      // last line (largest ID)
274      else if (line == 8) {
275        assertTrue(row.startsWith("120320038"));
276      }
277    }
