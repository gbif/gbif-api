101   * @throws IOException
102   */
103  @Test
104  public void testCsvMultiline() throws IOException, ParseException {
105    File csv = FileUtils.getClasspathFile("csv/csv_quote_endline.csv");
106
107    try (TabularDataFileReader<List<String>> reader = TabularFiles.newTabularFileReader(
108            Files.newBufferedReader(csv.toPath(), StandardCharsets.UTF_8), ',', true)) {
109
110      //before we start reading, those methods are expected to return 0
111      assertEquals(0, reader.getLastRecordNumber());
112      assertEquals(0, reader.getLastRecordLineNumber());
113
114      int numberOfRows = 0;
115      List<String> rec = reader.read();
116      while(rec != null){
117        numberOfRows++;
118        rec = reader.read();
119      }
120
121      assertEquals(3, numberOfRows);
122      assertEquals(3, reader.getLastRecordNumber());
123      assertEquals(7, reader.getLastRecordLineNumber());
124    }
125  }

186  /**
187   * Test extracting a CSV file containing embedded JSON, which itself contains escaped quotes.
188   *
189   * JSON value like: { "test": "value, \"like\" this" }
190   *
191   * Would become in CSV: "{ ""test"": ""value, \""like\"" this"" }"
192   */
193  @Test
194  public void testCsvJsonEscapedQuotes() throws IOException, ParseException {
195    File csv = FileUtils.getClasspathFile("csv/csv_json_escaped_quotes2.csv");
196    try (TabularDataFileReader<List<String>> reader = TabularFiles.newTabularFileReader(
197      Files.newBufferedReader(csv.toPath(), StandardCharsets.UTF_8), ',', "\n", '"', true)) {
198
199      List<String> atom = reader.read();
200      assertEquals(3, atom.size());
201      assertEquals("779", atom.get(0));
202      // Without the Java escapes: {"chronostratigraphy": "Cretaceous, Early Cretaceous, Albian - Late Cretaceous, Cenomanian", "cataloguedescription": "Very worn vertebra. Old catalogue says \"fragments of bone\".", "created": "2009-05-13", "barcode": "010039076", "project": "eMesozoic", "determinationnames": "Ornithocheirus", "subdepartment": "Vertebrates", "lithostratigraphy": "Selborne Group, Upper Greensand Formation, Cambridge Greensand Member", "imagecategory": ["Register;Specimen"]}
203      assertEquals("{\"jsonKey\": \"jsonValue\"}", atom.get(1));
204      assertEquals("Cambridge, Cambridge", atom.get(2));
205
206      atom = reader.read();
207      assertEquals("{\"jsonKey\": \"jsonValue with a \"quote\" in the middle (invalid JSON)\"}", atom.get(1));
208
209      atom = reader.read();
210      assertEquals("{\"jsonKey\": \"jsonValue with a \\\"quote\\\" in the middle (valid JSON)\"}", atom.get(1));
211
212    }
213  }

215  /**
216   * TSV cannot encode tabs or newlines, so there is no escape character.
217   */
218  @Test
219  public void testTsvBackslashes() throws IOException, ParseException {
220    File csv = FileUtils.getClasspathFile("tabular/with_backslashes.tsv");
221    try (TabularDataFileReader<List<String>> reader = TabularFiles.newTabularFileReader(
222      Files.newBufferedReader(csv.toPath(), StandardCharsets.UTF_8), '\t', "\n", null, true)) {
223
224      List<String> atom = reader.read();
225      assertEquals(2, atom.size());
226      assertEquals("key", atom.get(0));
227      assertEquals("value", atom.get(1));
228      atom = reader.read();
229      assertEquals("Around 1\\4 mile along the road", atom.get(1));
230      atom = reader.read();
231      assertEquals("Near the Cloud\\Mitchell county line", atom.get(1));
232      atom = reader.read();
233      assertEquals("{\"jKey\": \"jValue with \\\"quotes\\\"\"}", atom.get(1));
234    }
235  }
