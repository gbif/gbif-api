001package org.gbif.utils.file.tabular;
002
003import org.gbif.utils.file.FileUtils;
004
005import java.io.File;
006import java.io.FileInputStream;
007import java.util.List;
008
009import org.junit.Test;
010
011import static org.junit.Assert.assertEquals;
012
013/**
014 * Unit tests for {@link TabularDataFileReader}.
015 */
016public class TabularDataFileReaderTest {
017
018  @Test
019  public void testCsvAllwaysQuotes() throws Exception {
020    File csv = FileUtils.getClasspathFile("csv/csv_optional_quotes_excel2008.csv");
021
022    TabularDataFileReader<List<String>> reader = TabularFiles.newTabularFileReader(new FileInputStream(csv), ',', true);
023
024    List<String> rec = reader.read();
025    assertEquals(3, rec.size());
026    assertEquals("1", rec.get(0));
027    assertEquals("This has a, comma", rec.get(2));
028
029    rec = reader.read();
030    assertEquals("I say this is only a \"quote\"", rec.get(2));
031
032    reader.close();
033  }
034
035}
