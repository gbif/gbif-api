001package org.gbif.utils.file;
002
003import java.net.MalformedURLException;
004import java.net.URL;
005
006import org.junit.Test;
007
008import static org.junit.Assert.assertTrue;
009
010/**
011 * Testing the data layout
012 */
013public class TestFileUtils {
014
015  @Test
016  public void testSourceRepo() throws MalformedURLException {
017    URL url = new URL("http://www.gbif.org:8081/~markus/testdata/geobotany.pdf");
018    System.out.println(FileUtils.toFilePath(url));
019    assertTrue(FileUtils.toFilePath(url).equals("www.gbif.org:8081/~markus/testdata/geobotany.pdf"));
020
021    url = new URL("ftp://ftp.gbif.org/testdata/markus/geobotany.pdf");
022    System.out.println(FileUtils.toFilePath(url));
023    assertTrue(FileUtils.toFilePath(url).equals("ftp.gbif.org/__ftp__/testdata/markus/geobotany.pdf"));
024  }
025}
