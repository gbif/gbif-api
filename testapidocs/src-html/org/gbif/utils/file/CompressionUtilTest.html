038  public static File createTempDirectory() throws IOException {
039
040    final File temp = File.createTempFile("temp", Long.toString(System.nanoTime()));
041
042    if (!temp.delete()) {
043      throw new IOException("Could not delete temp file: " + temp.getAbsolutePath());
044    }
045
046    if (!temp.mkdir()) {
047      throw new IOException("Could not create temp directory: " + temp.getAbsolutePath());
048    }
049    temp.deleteOnExit();
050
051    return temp;
052  }
053
054  private static void assureEqualContent(List<File> result, String metaContent, String dataContent) throws IOException {
055    for (File f : result) {
056      String x = FileUtils.readFileToString(f, "utf-8");
057      if ("meta.xml".equals(f.getName())) {
058        assertEquals(metaContent, x);
059      } else if ("quote_in_quote.csv".equals(f.getName())) {
060        assertEquals(dataContent, x);
061      } else {
062        fail("unexpected file");
063      }
064    }
065  }
066
067  public File classpathFile(String path) {
068    File f = null;
069    // relative path. Use classpath instead
070    URL url = getClass().getClassLoader().getResource(path);
071    if (url != null) {
072      f = new File(url.getFile());
073    }
074    return f;
075  }

077  @Test
078  public void testDecompress() throws IOException {
079    // meta.xml
080    File meta = classpathFile("compression/archive/meta.xml");
081    String metaContent = FileUtils.readFileToString(meta, "utf-8");
082    // quote_in_quote.csv
083    File data = classpathFile("compression/archive/quote_in_quote.csv");
084    String dataContent = FileUtils.readFileToString(data, "utf-8");
085
086    File tmpDir = createTempDirectory();
087    File testArchiveFile = classpathFile("compression/archive.zip");
088    List<File> result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
089    assertEquals(2, result.size());
090    assureEqualContent(result, metaContent, dataContent);
091
092    FileUtils.cleanDirectory(tmpDir);
093    testArchiveFile = classpathFile("compression/archive.tgz");
094    result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
095    assertEquals(2, result.size());
096    assureEqualContent(result, metaContent, dataContent);
097
098    FileUtils.cleanDirectory(tmpDir);
099    testArchiveFile = classpathFile("compression/archive-zip.dat");
100    result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
101    assertEquals(2, result.size());
102    assureEqualContent(result, metaContent, dataContent);
103
104    FileUtils.cleanDirectory(tmpDir);
105    testArchiveFile = classpathFile("compression/archive-tgz.dat");
106    result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
107    assertEquals(2, result.size());
108    assureEqualContent(result, metaContent, dataContent);
109
110    FileUtils.cleanDirectory(tmpDir);
111    testArchiveFile = classpathFile("compression/cate.zip");
112    result = CompressionUtil.decompressFile(tmpDir, testArchiveFile);
113    assertEquals(3, result.size());
114  }

211    List<File> files = CompressionUtil.unzipFile(tmpDir, testZippedFolder, true);

213    // assert wrapping root directory is removed
214    File rootDir = new File(tmpDir, "res1");

216    assertTrue(new File(tmpDir, "dwca.zip").exists());
217    assertTrue(new File(tmpDir, "eml.xml").exists());
218    assertTrue(new File(tmpDir, "publication.log").exists());
219    assertTrue(new File(tmpDir, "resource.xml").exists());
220    assertTrue(new File(tmpDir, "test4.rtf").exists());
221
222    // assert subdirectory sources was preserved
223    File sourceDir = new File(tmpDir, "sources");
224    assertTrue(sourceDir.isDirectory());
225    assertTrue(sourceDir.exists());
226    assertEquals(1, sourceDir.listFiles().length);
227    assertTrue(new File(sourceDir, "occurrence.txt").exists());
228
229    // assert subdirectory dwca was preserved
230    File dwcaDir = new File(tmpDir, "dwca");
231    assertTrue(dwcaDir.isDirectory());

236    assertTrue(new File(dwcaDir, "meta.xml").exists());
237    assertTrue(new File(dwcaDir, "eml.xml").exists());
238
239    // assert hidden files and directories are removed
240    assertFalse(new File(tmpDir, ".svn").exists());
241    assertFalse(new File(tmpDir, "/sources/.svn").exists());
242    // assert .DS_Store removed
243    assertFalse(new File(tmpDir, ".DS_Store").exists());
244    // assert __MACOSX removed

260  @Test
261  public void testZipFolder() throws IOException {
262    File zipWithDirs = File.createTempFile("aha", ".zip");
263    System.out.println(zipWithDirs.getAbsolutePath());
264
265    File zipWithoutDirs = File.createTempFile("aha", ".zip");
266    System.out.println(zipWithoutDirs.getAbsolutePath());
267    //tmp.deleteOnExit();
268    File testFolder = classpathFile("charsets");
269    // remember how many files we have in the root folder, exlcuding files in subdirectories
270    final int rootFileNum = testFolder.listFiles().length;
271
272    CompressionUtil.zipDir(testFolder, zipWithDirs, true);
273    CompressionUtil.zipDir(testFolder, zipWithoutDirs, false);
274
275    assertNotEquals(zipWithDirs.length(), zipWithoutDirs.length());
276
277    // now decompress the subdir zip and make sure we get the same amount of root files
278    File tmpDir = org.gbif.utils.file.FileUtils.createTempDir();
279    tmpDir.deleteOnExit();
280    CompressionUtil.unzipFile(tmpDir, zipWithDirs, true);
281    int decompressedRootFileNum = tmpDir.listFiles().length;
282    assertEquals(rootFileNum, decompressedRootFileNum);
283  }
