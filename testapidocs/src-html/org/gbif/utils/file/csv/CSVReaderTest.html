036  @Test
037  public void testCsvQuotedDelimiter() throws IOException {
038    String rows =
039            "12,\"not real\"\n"
040                    + "13,not \"real\"\n"
041                    + "\"14\",noting\n"
042                    + "15,\"not \"\"real\"\"\"\n"
043                    + "16,\"no, this is \"\"real\"\"\"\n";
044
045    InputStream stream = new ByteArrayInputStream(rows.getBytes(Charsets.UTF_8));
046    CSVReader reader = new CSVReader(stream, UTF8, ",", '"', 0);
047
048    String[] rec = reader.next();
049    assertEquals(2, rec.length);
050    assertEquals("12", rec[0]);
051    assertEquals("not real", rec[1]);
052
053    rec = reader.next();
054    assertEquals(2, rec.length);
055    assertEquals("13", rec[0]);
056    assertEquals("not \"real\"", rec[1]);
057
058    rec = reader.next();
059    assertEquals(2, rec.length);
060    assertEquals("14", rec[0]);
061    assertEquals("noting", rec[1]);
062
063    rec = reader.next();
064    assertEquals(2, rec.length);
065    assertEquals("15", rec[0]);
066    assertEquals("not \"real\"", rec[1]);
067
068    rec = reader.next();
069    assertEquals(2, rec.length);
070    assertEquals("16", rec[0]);
071    assertEquals("no, this is \"real\"", rec[1]);
072
073    assertFalse(reader.hasNext());
074  }

094    assertEquals("What though, \"if you have a quote\" and a comma", atom[2]);
095
096    atom = reader.next();
097    assertEquals("What, if we have a \"quote, which has a comma, or 2\"", atom[2]);
098
099    reader.close();
100  }

102  /**
103   * tests the csv reader with different number of header rows on the same file and compares the 4th line in the text
104   * file for each of them
105   */
106  @Test
107  public void testHeaderRows() throws IOException {
108    File source = FileUtils.getClasspathFile("csv/iucn100.csv");
109    // assert the headers are the same, no matter how many rows we skip for the iterator
110    CSVReader reader = new CSVReader(source, UTF8, ",", '"', 1);
111    reader.next();
112    reader.next();
113    String[] row4h1 = reader.next();
114    reader.close();
115
116    reader = new CSVReader(source, UTF8, ",", '"', 0);
117    reader.next();
118    reader.next();
119    reader.next();
120    String[] row4h0 = reader.next();
121    reader.close();
122
123    reader = new CSVReader(source, UTF8, ",", '"', 3);
124    String[] row4h3 = reader.next();
125    reader.close();
126
127    assertTrue(row4h0.length == row4h1.length);
128    assertTrue(row4h0.length == row4h3.length);
129    int idx = row4h0.length;
130    while (idx > 0) {
131      idx--;
132      assertEquals(row4h0[idx], row4h1[idx]);
133      assertEquals(row4h0[idx], row4h3[idx]);
134    }
135  }

153  @Test
154  public void testIgnoreEmptyLines() throws IOException {
155    File csv = FileUtils.getClasspathFile("csv/empty_line.tab");
156    CSVReader reader = new CSVReader(csv, UTF8, "\t", null, 1);
157    String[] ids = {"1", "5", "10", "12", "14", "20"};
158    int row = 0;
159    while (reader.hasNext()) {
160      String[] rec = reader.next();
161      assertEquals(ids[row], rec[0]);
162      row++;
163    }
164    assertTrue(reader.getEmptyLines().size() > 1);
165    assertTrue(reader.getEmptyLines().contains(6));
166    assertTrue(reader.getEmptyLines().contains(9));
167  }
