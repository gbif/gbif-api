037  @Test
038  public void testCsvQuotedDelimiter() throws IOException {
039    String rows =
040            "12,\"not real\"\n"
041                    + "13,not \"real\"\n"
042                    + "\"14\",noting\n"
043                    + "15,\"not \"\"real\"\"\"\n"
044                    + "16,\"no, this is \"\"real\"\"\"\n";
045
046    InputStream stream = new ByteArrayInputStream(rows.getBytes(Charsets.UTF_8));
047    CSVReader reader = new CSVReader(stream, UTF8, ",", '"', 0);
048
049    String[] rec = reader.next();
050    assertEquals(2, rec.length);
051    assertEquals("12", rec[0]);
052    assertEquals("not real", rec[1]);
053
054    rec = reader.next();
055    assertEquals(2, rec.length);
056    assertEquals("13", rec[0]);
057    assertEquals("not \"real\"", rec[1]);
058
059    rec = reader.next();
060    assertEquals(2, rec.length);
061    assertEquals("14", rec[0]);
062    assertEquals("noting", rec[1]);
063
064    rec = reader.next();
065    assertEquals(2, rec.length);
066    assertEquals("15", rec[0]);
067    assertEquals("not \"real\"", rec[1]);
068
069    rec = reader.next();
070    assertEquals(2, rec.length);
071    assertEquals("16", rec[0]);
072    assertEquals("no, this is \"real\"", rec[1]);
073
074    assertFalse(reader.hasNext());
075  }

103  /**
104   * tests the csv reader with different number of header rows on the same file and compares the 4th line in the text
105   * file for each of them
106   */
107  @Test
108  public void testHeaderRows() throws IOException {
109    File source = FileUtils.getClasspathFile("csv/iucn100.csv");
110    // assert the headers are the same, no matter how many rows we skip for the iterator
111    CSVReader reader = new CSVReader(source, UTF8, ",", '"', 1);
112    reader.next();
113    reader.next();
114    String[] row4h1 = reader.next();
115    reader.close();
116
117    reader = new CSVReader(source, UTF8, ",", '"', 0);
118    reader.next();
119    reader.next();
120    reader.next();
121    String[] row4h0 = reader.next();
122    reader.close();
123
124    reader = new CSVReader(source, UTF8, ",", '"', 3);
125    String[] row4h3 = reader.next();
126    reader.close();
127
128    assertTrue(row4h0.length == row4h1.length);
129    assertTrue(row4h0.length == row4h3.length);
130    int idx = row4h0.length;
131    while (idx > 0) {
132      idx--;
133      assertEquals(row4h0[idx], row4h1[idx]);
134      assertEquals(row4h0[idx], row4h3[idx]);
135    }
136  }

153  @Test
154  public void testIgnoreEmptyLines() throws IOException {
155    File csv = FileUtils.getClasspathFile("csv/empty_line.tab");
156    CSVReader reader = new CSVReader(csv, UTF8, "\t", null, 1);
157    String[] ids = {"1", "5", "10", "12", "14", "20", "21", "", "30"};
158    int row = 0;
159    while (reader.hasNext()) {
160      String[] rec = reader.next();
161      assertEquals(ids[row], rec[0]);
162      row++;
163    }
164    assertTrue(reader.getEmptyLines().size() > 1);
165    assertTrue(reader.getEmptyLines().contains(6));
166    assertTrue(reader.getEmptyLines().contains(12));
167  }
