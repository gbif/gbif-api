001package org.gbif.utils.file.csv;
002
003import org.gbif.utils.file.FileUtils;
004
005import java.io.File;
006import java.io.IOException;
007
008import org.junit.Test;
009
010import static org.junit.Assert.assertEquals;
011import static org.junit.Assert.assertNull;
012import static org.junit.Assert.assertTrue;
013
014/**
015 *
016 */
017public class CSVReaderFactoryTest {
018
019  @Test
020  public void detectCsvAlwaysQuoted() throws IOException {
021    File source = FileUtils.getClasspathFile("csv/csv_always_quoted.csv");
022    CSVReader reader = CSVReaderFactory.build(source);
023    assertEquals(",", reader.delimiter);
024    assertEquals(new Character('"'), reader.quoteChar);
025    assertEquals(1, reader.headerRows);
026    reader.close();
027  }
028
029  /**
030   * We dont want unquoted CSVs, See detectCsvOptionallyQuoted()
031   */
032  public void detectCsvUnquoted() throws IOException {
033    String[] files = {"csv/csv_unquoted.txt"};
034    for (String fn : files) {
035      File source = FileUtils.getClasspathFile(fn);
036      CSVReader reader = CSVReaderFactory.build(source);
037      assertEquals(",", reader.delimiter);
038      assertNull(reader.quoteChar);
039      assertEquals(1, reader.headerRows);
040      reader.close();
041    }
042  }
043
044  @Test
045  public void detectPipe() throws IOException {
046    String[] files = new String[] {"csv/pipe_separator.txt"};
047    for (String fn : files) {
048      File source = FileUtils.getClasspathFile(fn);
049      CSVReader reader = CSVReaderFactory.build(source);
050      assertEquals("|", reader.delimiter);
051      assertNull(reader.quoteChar);
052      assertEquals(1, reader.headerRows);
053      reader.close();
054    }
055  }
056
057  @Test
058  public void detectSemicolon() throws IOException {
059    String[] files = {"csv/semicolon_separator.csv"};
060    for (String fn : files) {
061      File source = FileUtils.getClasspathFile(fn);
062      CSVReader reader = CSVReaderFactory.build(source);
063      assertEquals(";", reader.delimiter);
064      assertNull(reader.quoteChar);
065      assertEquals(1, reader.headerRows);
066      reader.close();
067    }
068  }
069
070  /**
071   * As CSV files with rare optional quotes are hard to detect but cause problems
072   * we prefer to default to the " quotation in case comma seperated files are used.
073   * This is why test detectCsvUnquoted() is outcommented right now!
074   */
075  @Test
076  public void detectCsvOptionallyQuoted() throws IOException {
077    String[] files = {"csv/csv_optional_quotes_puma.csv", "csv/csv_optional_quotes_excel2008.csv",
078            "csv/csv_incl_single_quotes.csv", "csv/iucn100.csv", "csv/csv_unquoted.txt", "csv/csv_unquoted_coordinates.txt"};
079    for (String fn : files) {
080      File source = FileUtils.getClasspathFile(fn);
081      CSVReader reader = CSVReaderFactory.build(source);
082      assertEquals(",", reader.delimiter);
083      assertEquals(new Character('"'), reader.quoteChar);
084      assertEquals(1, reader.headerRows);
085      reader.close();
086    }
087  }
088
089  @Test
090  public void detectTab() throws IOException {
091    String[] files = {"csv/ipni.tab.txt", "csv/tab_separated_generic.txt", "csv/iucn100.tab.txt", "csv/ebird.tab.txt",
092            "csv/empty_line.tab", "csv/irmng.tail", "csv/MOBOT.tab.csv"};
093    for (String fn : files) {
094      File source = FileUtils.getClasspathFile(fn);
095      CSVReader reader = CSVReaderFactory.build(source);
096      assertEquals("\t", reader.delimiter);
097      assertNull(reader.quoteChar);
098      assertEquals(1, reader.headerRows);
099      reader.close();
100    }
101  }
102
103  @Test
104  public void detectTabQuoted() throws IOException {
105    String[] files = {"csv/eol/my_darwincore_tab_separated_quoted.txt",
106            "csv/eol/my_dataobject_tab_separated_quoted.txt", "csv/borza_tab_separated_quoted.txt"};
107    for (String fn : files) {
108      File source = FileUtils.getClasspathFile(fn);
109      CSVReader reader = CSVReaderFactory.build(source);
110      assertEquals("\t", reader.delimiter);
111      assertTrue(reader.quoteChar == '"');
112      assertEquals(1, reader.headerRows);
113      reader.close();
114    }
115  }
