001package org.gbif.utils.collection;
002
003import java.util.HashMap;
004import java.util.LinkedList;
005import java.util.Map;
006import java.util.Queue;
007
008import org.junit.Test;
009
010import static org.junit.Assert.assertEquals;
011
012public class MapUtilsTest {
013
014
015  @Test
016  public void testMap() {
017    Map<String, Double> map = new HashMap<String, Double>(4);
018    map.put("A", 99.5);
019    map.put("B", 67.4);
020    map.put("C", 67.5);
021    map.put("D", 67.3);
022
023    Map<String, Double> mapSorted = MapUtils.sortByValue(map);
024
025    assertEquals(4, mapSorted.size());
026
027    Queue<String> expected = new LinkedList<String>();
028    expected.add("D");
029    expected.add("B");
030    expected.add("C");
031    expected.add("A");
032    for (String k : mapSorted.keySet()) {
033      assertEquals(expected.poll(), k);
034    }
035  }
036
037  @Test
038  public void testMap2() {
039    Map<Integer, String> map = new HashMap<Integer, String>(4);
040    map.put(5, "C");
041    map.put(4, "B");
042    map.put(99, "A");
043    map.put(3, "D");
044
045    Map<Integer, String> mapSorted = MapUtils.sortByValue(map);
046
047    assertEquals(4, mapSorted.size());
048
049    Queue<Integer> expected = new LinkedList<Integer>();
050    expected.add(99);
051    expected.add(4);
052    expected.add(5);
053    expected.add(3);
054    for (Integer k : mapSorted.keySet()) {
055      assertEquals(expected.poll(), k);
056    }
057  }
058}
