001package org.gbif.utils.collection;
002
003import java.util.HashSet;
004import java.util.Iterator;
005import java.util.Set;
006
007public class HashSetPerformance {
008
009  private static final int TIMES = 100000;
010  private static final int MAX = 500000;
011
012  public static void main(String[] argv) {
013    // first, get the JIT going
014    test(false, new CompactHashSet());
015    test(false, new HashSet());
016
017    // then, do real timings
018    System.out.println("*** HashSet ***");
019    for (int ix = 0; ix < 3; ix++) {
020      test(true, new HashSet());
021    }
022    System.out.println("*** CompactHashSet ***");
023    for (int ix = 0; ix < 3; ix++) {
024      test(true, new CompactHashSet());
025    }
026
027  }
028
029  public static void test(boolean output, Set set) {
030    long start = System.currentTimeMillis();
031
032    if (output) {
033      System.gc();
034      System.gc();
035    }
036    long before = Runtime.getRuntime().totalMemory() - Runtime.getRuntime().freeMemory();
037
038    // add
039    for (int ix = 0; ix < TIMES; ix++) {
040      set.add(new Long(Math.round(Math.random() * MAX)));
041    }
042
043    if (output) {
044      System.gc();
045      System.gc();
046      long after = Runtime.getRuntime().totalMemory() - Runtime.getRuntime().freeMemory();
047      System.out.println("Memory before: " + before);
048      System.out.println("Memory after: " + after);
049      System.out.println("Memory usage: " + (after - before));
050    }
051
052    // lookup
053    int count = 0;
054    for (int ix = 0; ix < TIMES; ix++) {
055      Long number = new Long(Math.round(Math.random() * MAX));
056      if (set.contains(number)) {
057        count++;
058      }
059    }
060
061    // iterate
062    Iterator it = set.iterator();
063    while (it.hasNext()) {
064      Long number = (Long) it.next();
065    }
066
067    // remove
068    for (int ix = 0; ix < TIMES; ix++) {
069      Long number = new Long(Math.round(Math.random() * MAX));
070      set.remove(number);
071    }
072
073    if (output) {
074      System.out.println("TIME: " + (System.currentTimeMillis() - start));
075    }
076  }
077}
