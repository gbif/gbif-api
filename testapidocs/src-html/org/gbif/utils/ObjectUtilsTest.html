001package org.gbif.utils;
002
003import java.util.Collection;
004
005import com.google.common.collect.Lists;
006import org.junit.Test;
007
008import static org.junit.Assert.assertEquals;
009import static org.junit.Assert.assertNull;
010
011/**
012 *
013 */
014public class ObjectUtilsTest {
015
016  @Test
017  public void testCoalesce() throws Exception {
018    assertNull(ObjectUtils.coalesce());
019    assertNull(ObjectUtils.coalesce((Integer)null));
020    assertNull(ObjectUtils.coalesce((Integer)null, (Integer)null));
021
022    assertEquals((Integer)13, ObjectUtils.coalesce(null, 13));
023    assertEquals((Integer)13, ObjectUtils.coalesce(null, 13, 14));
024    assertEquals((Integer)13, ObjectUtils.coalesce(13, 15));
025  }
026
027  @Test
028  public void testCoalesce1() throws Exception {
029    assertNull(ObjectUtils.coalesce((Collection) null));
030    assertNull(ObjectUtils.coalesce(Lists.newArrayList()));
031    assertNull(ObjectUtils.coalesce(Lists.newArrayList(null, null)));
032
033    assertEquals((Integer)13, ObjectUtils.coalesce(Lists.newArrayList(null, null, 13)));
034    assertEquals((Integer)13, ObjectUtils.coalesce(Lists.newArrayList(null, null, 13, 14)));
035    assertEquals((Integer)13, ObjectUtils.coalesce(Lists.newArrayList(13, null)));
036
037  }
038}
