001package org.gbif.utils.html;
002
003import com.google.common.collect.ImmutableSet;
004import org.junit.Test;
005import org.owasp.html.PolicyFactory;
006import org.owasp.html.Sanitizers;
007
008import static junit.framework.TestCase.assertEquals;
009import static org.junit.Assert.assertTrue;
010
011/**
012 * Unit test for BeanHtmlSanitizer.
013 *
014 */
015public class BeanHtmlSanitizerTest {
016
017  @Test
018  public void testBeanHtmlSanitizer(){
019    //Pre-packaged HTML sanitizer policies that allows common block elements including <p>, <h1>, etc.
020    PolicyFactory policy = Sanitizers.BLOCKS;
021    TestBean testBean = new TestBean();
022    testBean.setCount(1);
023    testBean.setComment("Help to find your way on the Internet");
024    testBean.setUnsafeHtml("<p>      <a href=\"http://perdu.com/\" class=\"important\">" +
025            "        Help" +
026            "      </a></p>");
027
028    //demonstrate that inner beans are NOT sanitized
029    TestInnerBean innerBean = new TestInnerBean();
030    innerBean.setInnerString("<h1><script type=\"text/javascript\">echo(\"oops\");<script></h1");
031    testBean.setInnerBean(innerBean);
032
033    BeanHtmlSanitizer.sanitize(testBean, policy);
034
035    assertEquals("<p>Help</p>", testBean.getUnsafeHtml().replaceAll(" ", ""));
036    //ensure other fields were untouched
037    assertEquals("Help to find your way on the Internet", testBean.getComment());
038    assertEquals(1, testBean.getCount());
039    assertEquals("<h1><script type=\"text/javascript\">echo(\"oops\");<script></h1", testBean.getInnerBean().getInnerString());
040
041    // but we can sanitize the inner bean explicitly
042    BeanHtmlSanitizer.sanitize(innerBean, policy);
043    assertEquals("<h1></h1>", testBean.getInnerBean().getInnerString());
044  }
045
046  @Test
047  public void testBeanHtmlSanitizerExclusionList(){
048    PolicyFactory policy = Sanitizers.BLOCKS;
049    TestBean testBean = new TestBean();
050    testBean.setCount(1);
051    testBean.setComment("Help to find your way on the Internet");
052    testBean.setUnsafeHtml("<p>      <a href=\"http://perdu.com/\" class=\"important\">" +
053            "        Help" +
054            "      </a></p>");
055    // ask to exclude "unsafeHtml" property from sanitization
056    BeanHtmlSanitizer.sanitize(testBean, policy, ImmutableSet.of("unsafeHtml"));
057    assertTrue(testBean.getUnsafeHtml().contains("<a href=\"http://perdu.com/\" class=\"important\">"));
058  }
059
060
061  /**
062   * Requires to be public for testing BeanHtmlSanitizer.
063   *
064   */
065  public static class TestBean{
066    private String unsafeHtml;
067    private String comment;
068    private int count;
069
070    private TestInnerBean innerBean;
071
072    public String getUnsafeHtml() {
073      return unsafeHtml;
074    }
075
076    public void setUnsafeHtml(String unsafeHtml) {
077      this.unsafeHtml = unsafeHtml;
078    }
079
080    public String getComment() {
081      return comment;
082    }
083
084    public void setComment(String comment) {
085      this.comment = comment;
086    }
087
088    public int getCount() {
089      return count;
090    }
091
092    public void setCount(int count) {
093      this.count = count;
094    }
095
096    public TestInnerBean getInnerBean() {
097      return innerBean;
098    }
099
100    public void setInnerBean(TestInnerBean innerBean) {
101      this.innerBean = innerBean;
102    }
103
104  }
105
106  public static class TestInnerBean{
107    private String innerString;
108
109    public String getInnerString() {
110      return innerString;
111    }
112
113    public void setInnerString(String innerString) {
114      this.innerString = innerString;
115    }
116  }
