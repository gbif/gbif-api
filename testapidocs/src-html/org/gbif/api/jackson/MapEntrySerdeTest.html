001package org.gbif.api.jackson;
002
003import org.gbif.dwc.terms.DwcTerm;
004import org.gbif.dwc.terms.Term;
005
006import java.io.IOException;
007import java.util.AbstractMap;
008import java.util.List;
009import java.util.Map;
010
011import com.google.common.collect.ImmutableList;
012import org.codehaus.jackson.map.ObjectMapper;
013import org.codehaus.jackson.map.annotate.JsonDeserialize;
014import org.codehaus.jackson.map.annotate.JsonSerialize;
015import org.junit.Test;
016
017import static junit.framework.TestCase.assertEquals;
018
019/**
020 * Unit tests for {@link MapEntrySerde}.
021 */
022public class MapEntrySerdeTest {
023
024  private static final ObjectMapper MAPPER = new ObjectMapper();
025
026  @Test
027  public void testMapEntrySerde() throws IOException {
028
029    MapEntryWrapper kv = new MapEntryWrapper("mykey", 18);
030    String json = MAPPER.writeValueAsString(kv);
031
032    MapEntryWrapper rebuiltKeyValue = MAPPER.readValue(json, MapEntryWrapper.class);
033    assertEquals(kv.getSingleElement().getValue(), rebuiltKeyValue.getSingleElement().getValue());
034
035    MapEntryListTermWrapper kvl = new MapEntryListTermWrapper(DwcTerm.country, 19);
036    json = MAPPER.writeValueAsString(kvl);
037
038    MapEntryListTermWrapper rebuiltKeyValueList = MAPPER.readValue(json, MapEntryListTermWrapper.class);
039    assertEquals(2, rebuiltKeyValueList.getListTerm().size());
040
041  }
042
043  /**
044   * For testing purpose only.
045   */
046  static class MapEntryWrapper {
047
048    private Map.Entry<Object, Object> singleElement;
049
050    MapEntryWrapper() {}
051
052    MapEntryWrapper(Object key, Object value){
053      singleElement = new AbstractMap.SimpleImmutableEntry<>(key, value);
054    }
055
056    @JsonSerialize(using = MapEntrySerde.MapEntryJsonSerializer.class)
057    @JsonDeserialize(using = MapEntrySerde.MapEntryJsonDeserializer.class)
058    Map.Entry<Object, Object> getSingleElement() {
059      return singleElement;
060    }
061  }
062
063  static class MapEntryListTermWrapper {
064
065    private List<Map.Entry<Term, Object>> listTerm;
066
067    MapEntryListTermWrapper() {}
068
069    MapEntryListTermWrapper(Term key, Object value){
070      listTerm = ImmutableList.of(new AbstractMap.SimpleImmutableEntry<>(key, value),
071              new AbstractMap.SimpleImmutableEntry<>(key, value));
072    }
073
074    @JsonSerialize(contentUsing = MapEntrySerde.MapEntryJsonSerializer.class)
075    @JsonDeserialize(contentUsing = MapEntrySerde.MapEntryJsonDeserializer.class)
076    List<Map.Entry<Term, Object>> getListTerm() {
077      return listTerm;
078    }
079  }
080
081}
