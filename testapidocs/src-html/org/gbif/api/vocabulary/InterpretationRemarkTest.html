001package org.gbif.api.vocabulary;
002
003import java.io.IOException;
004import java.util.Arrays;
005import java.util.Collections;
006import java.util.HashSet;
007import java.util.List;
008import java.util.Set;
009import java.util.stream.Collectors;
010
011import com.google.common.reflect.ClassPath;
012import org.junit.Test;
013
014import static org.junit.Assert.assertFalse;
015import static org.junit.Assert.assertTrue;
016import static org.junit.Assert.fail;
017
018/**
019 * Checks InterpretationRemark implementation under "org.gbif.api.vocabulary".
020 */
021public class InterpretationRemarkTest {
022
023  private static final String ROOT_PACKAGE = "org.gbif.api.vocabulary";
024
025  private static final List<Class<?>> IMPLEMENTING_CLASSES = loadInterpretationRemarkImplementations();
026
027  /**
028   * Get all {@link Class} that implement the {@link InterpretationRemark} interface from ROOT_PACKAGE.
029   *
030   * @return
031   */
032  private static List<Class<?>> loadInterpretationRemarkImplementations() {
033    try {
034      Set<ClassPath.ClassInfo> classes =
035              ClassPath.from(InterpretationRemarkTest.class.getClassLoader()).getTopLevelClasses(ROOT_PACKAGE);
036
037      return classes.stream()
038              .map(ClassPath.ClassInfo::load)
039              .filter(c -> Arrays.asList(c.getInterfaces()).contains(InterpretationRemark.class))
040              .collect(Collectors.toList());
041    } catch (IOException e) {
042      fail(e.getMessage());
043    }
044    return Collections.emptyList();
045  }
046
047  /**
048   * Checks all classes (enumerations) that implement {@link InterpretationRemark} to ensure they have unique entry
049   * names.
050   * {@link InterpretationRemark} implementations can be used as keys and can also be serialized as String. For this
051   * reason we want to ensure we have unique entries among implementations. This is mainly to avoid confusion when we
052   * do aggregation.
053   */
054  @Test
055  public void testInterpretationRemarkImplementations() {
056    Set<String> interpretationRemarks = new HashSet<>();
057    Set<String> interpretationRemarksId = new HashSet<>();
058    IMPLEMENTING_CLASSES.forEach(cl -> Arrays.asList((InterpretationRemark[]) cl.getEnumConstants()).forEach(
059            ir -> {
060              assertTrue("Enumeration value " + ir + " is unique within all InterpretationRemark implementations.",
061                      interpretationRemarks.add(ir.toString()));
062              assertTrue("Enumeration value " + ir + " with getID " + ir.getId() + " is unique within all InterpretationRemark implementations.",
063                      interpretationRemarksId.add(ir.getId()));
064            }
065    )
066    );
067  }
068
069  @Test
070  public void testInterpretationRemarkNotNulls() {
071    IMPLEMENTING_CLASSES.forEach(cl -> Arrays.asList((InterpretationRemark[]) cl.getEnumConstants()).forEach(
072            ir -> assertTrue("InterpretationRemark implementations return a not null value for getSeverity()",
073                    ir.getSeverity() != null)
074            )
075    );
076  }
077
078  @Test
079  public void testInterpretationRemarkDeprecated() {
080    assertTrue(OccurrenceIssue.COORDINATE_PRECISION_UNCERTAINTY_MISMATCH.isDeprecated());
081    assertFalse(OccurrenceIssue.BASIS_OF_RECORD_INVALID.isDeprecated());
082  }
083
084}
