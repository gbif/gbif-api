040  @Test
041  public void testInferRank() {
042    for (Map.Entry<String, Rank> stringRankEntry : NAMES.entrySet()) {
043      assertEquals(stringRankEntry.getValue(), Rank.inferRank(stringRankEntry.getKey(), null, null, null, null));
044    }
045
046    assertEquals(Rank.SPECIES, Rank.inferRank("Abies", "Abies", "alba", null, null));
047    assertEquals(Rank.SPECIES, Rank.inferRank("Abies", null, "alba", null, null));
048    assertEquals(Rank.INFRAGENERIC_NAME, Rank.inferRank(null, "Abies", null, null, null));
049    assertEquals(Rank.INFRAGENERIC_NAME, Rank.inferRank("", "Abies", null, null, null));
050    assertEquals(Rank.VARIETY, Rank.inferRank(null, "Abies", "alba", "var.", "alpina"));
051    assertEquals(Rank.PATHOVAR, Rank.inferRank(null, "Pseudomonas", "syringae ", "pv.", "aceris"));
052    assertEquals(Rank.SUBSPECIES, Rank.inferRank(null, "Abies", "alba", "ssp.", null));
053    assertEquals(Rank.SPECIES, Rank.inferRank(null, "Abies", null, "spec.", null));
054    assertEquals(Rank.SUPRAGENERIC_NAME, Rank.inferRank("Neurolaenodinae", null, null, "ib.", null));
055    assertEquals(Rank.SUPRAGENERIC_NAME, Rank.inferRank("Neurolaenodinae", null, null, "supersubtrib.", null));
056    assertEquals(Rank.BIOVAR, Rank.inferRank(null, "Pseudomonas", "syringae ", "bv.", "aceris"));
057    assertEquals(Rank.BIOVAR, Rank.inferRank(null, "Thymus", "vulgaris", "biovar", "geraniol"));
058    assertEquals(Rank.CHEMOFORM, Rank.inferRank(null, "Thymus", "vulgaris", "ct.", "geraniol"));
059    assertEquals(Rank.CHEMOFORM, Rank.inferRank(null, "Thymus", "vulgaris", "chemoform", "geraniol"));
060    assertEquals(Rank.CHEMOVAR, Rank.inferRank(null, "Thymus", "vulgaris", "chemovar", "geraniol"));
061    assertEquals(Rank.SEROVAR, Rank.inferRank(null, "Thymus", "vulgaris", "serovar", "geraniol"));
062
063    // should not be able to infer the correct family
064    assertEquals(Rank.UNRANKED, Rank.inferRank("Compositae", null, null, null, null));
065  }

215  @Test
216  public void testHigher() {
217    assertFalse(Rank.SUPERFAMILY.higherThan(Rank.KINGDOM));
218    assertFalse(Rank.SUPERFAMILY.higherThan(Rank.KINGDOM));
219    assertFalse(Rank.SPECIES.higherThan(Rank.SUBGENUS));
220
221    assertTrue(Rank.INFRASPECIFIC_NAME.higherThan(Rank.VARIETY));
222    assertTrue(Rank.SUPERFAMILY.higherThan(Rank.FAMILY));
223    assertTrue(Rank.SPECIES.higherThan(Rank.VARIETY));
224    assertTrue(Rank.GENUS.higherThan(Rank.INFRAGENERIC_NAME));
225
226    int expectedHigher = Rank.DWC_RANKS.size() - 1;
227    for (Rank r : Rank.DWC_RANKS) {
228      int higherCount = 0;
229      for (Rank r2 : Rank.DWC_RANKS) {
230        if (r.higherThan(r2)) {
231          higherCount++;
232        }
233      }
234      assertEquals(expectedHigher, higherCount);
235      expectedHigher--;
236    }
237
238    // questionable
239    assertFalse(Rank.INFORMAL.higherThan(Rank.VARIETY));
240    assertFalse(Rank.UNRANKED.higherThan(Rank.VARIETY));
241  }
