001package org.gbif.api.vocabulary;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006
007public class ContactTypeTest {
008
009  @Test
010  public void testFromString() throws Exception {
011    assertEquals(ContactType.POINT_OF_CONTACT, ContactType.fromString("POINT_OF_CONTACT"));
012    assertEquals(ContactType.POINT_OF_CONTACT, ContactType.fromString("POINTOFCONTACT"));
013    assertEquals(ContactType.POINT_OF_CONTACT, ContactType.fromString("point_of_contact"));
014    assertEquals(ContactType.POINT_OF_CONTACT, ContactType.fromString("pointOfContact"));
015    assertEquals(ContactType.POINT_OF_CONTACT, ContactType.fromString("pointofcontact"));
016    assertEquals(ContactType.CURATOR, ContactType.fromString("curator"));
017  }
018
019}
