029  @Test
030  public void testFromIsoCode() throws Exception {

034
035  @Test
036  public void testGetIso2LetterCode() throws Exception {
037    Set<String> codes = Sets.newHashSet();
038    for (Country l : Country.values()) {
039      assertNotNull(l.getIso2LetterCode());
040      // make sure its upper case
041      assertEquals(l.getIso2LetterCode().toUpperCase(), l.getIso2LetterCode());
042      // make sure its unique
043      assertFalse(codes.contains(l.getIso2LetterCode()));
044      codes.add(l.getIso2LetterCode());
045    }
046    assertEquals("DE", Country.GERMANY.getIso2LetterCode());
047    assertEquals("GB", Country.UNITED_KINGDOM.getIso2LetterCode());
048  }

078  @Test
079  public void testIsCustomCode() throws Exception {
080    for (Country l : Country.values()) {
081      if (l.isOfficial()) {
082        assertFalse(Country.isCustomCode(l.getIso2LetterCode()));
083        assertFalse(Country.isCustomCode(l.getIso3LetterCode()));
084      } else {
085        assertTrue(Country.isCustomCode(l.getIso2LetterCode()));
086        assertTrue(Country.isCustomCode(l.getIso3LetterCode()));
087      }
088    }
089  }

091  @Test
092  public void testIsOfficial() throws Exception {
093    for (Country l : Country.OFFICIAL_COUNTRIES) {
094      assertTrue(l.isOfficial());
095    }
096    int officialCountries = Country.OFFICIAL_COUNTRIES.size();
097    int allCountries = Country.values().length;
098    assertTrue(allCountries > officialCountries);
099    assertEquals(249, officialCountries);
100  }

102  @Test
103  public void testgetTitle() throws Exception {
104    for (Country l : Country.values()) {
105      assertNotNull(l.getTitle());
106      assertTrue(l.getTitle().length() > 2);
107    }
108    assertEquals("United Kingdom", Country.UNITED_KINGDOM.getTitle());
109    assertEquals("Germany", Country.GERMANY.getTitle());
110  }

112  /**
113   * A container of Countries using 2 properties.
114   */
115  private static class Container {
116
117    public Country country;
118    public Country island; // verify that the names don't matter
119
120    public Container(Country country, Country island) {
121      this.country = country;
122      this.island = island;
123    }
124
125    // used by serializer below
126    public Container() {
127    }
128  }
129
130  @Test
131  public void testSerDe() {
132    ObjectMapper mapper = new ObjectMapper();
133    Container source = new Container(Country.DENMARK, Country.FALKLAND_ISLANDS);
134    try {
135      String serialized = mapper.writeValueAsString(source);
136      Container target = mapper.readValue(serialized.getBytes(), Container.class);
137      assertEquals("country differs", source.country, target.country);
138      assertEquals("island differs", source.island, target.island);
139    } catch (Exception e) {
140      fail(e.getMessage());
141    }
142  }
