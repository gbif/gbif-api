001package org.gbif.api.vocabulary;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006
007public class MaintenanceUpdateFrequencyTest {
008
009  @Test
010  public void testFromString() throws Exception {
011    // with spaces
012    assertEquals(MaintenanceUpdateFrequency.AS_NEEDED, MaintenanceUpdateFrequency.fromString("asNeeded"));
013    assertEquals(MaintenanceUpdateFrequency.AS_NEEDED, MaintenanceUpdateFrequency.fromString("as needed"));
014    assertEquals(MaintenanceUpdateFrequency.AS_NEEDED, MaintenanceUpdateFrequency.fromString("as-needed"));
015    assertEquals(MaintenanceUpdateFrequency.AS_NEEDED, MaintenanceUpdateFrequency.fromString("ASNEEDED"));
016    assertEquals(MaintenanceUpdateFrequency.OTHER_MAINTENANCE_PERIOD, MaintenanceUpdateFrequency.fromString("otherMaintenancePeriod"));
017    assertEquals(MaintenanceUpdateFrequency.NOT_PLANNED, MaintenanceUpdateFrequency.fromString("notPlanned"));
018    // typo matches vocabulary
019    assertEquals(MaintenanceUpdateFrequency.UNKOWN, MaintenanceUpdateFrequency.fromString("unkown"));
020  }
021}
