001package org.gbif.api.vocabulary;

003import org.junit.Test;
004
005import static org.junit.Assert.fail;
006
007/**
008 * Ensure that roles defined in {@link UserRole} and {@link AppRole} are unique.
009 */
010public class RolesTest {
011
012  @Test
013  public void testRolesUniqueness() {
014    for(UserRole ur : UserRole.values()) {
015      try{
016        AppRole.valueOf(ur.name());
017        fail(ur.name() + " is defined in UserRole AND AppRole.");
018      }
019      catch (IllegalArgumentException iaEx) {
020        //expected
021      }
022    }
023  }
024}
