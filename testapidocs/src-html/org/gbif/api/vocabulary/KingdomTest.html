001package org.gbif.api.vocabulary;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006
007public class KingdomTest {
008
009  @Test
010  public void testGetScientificName() throws Exception {
011    assertEquals("Plantae", Kingdom.PLANTAE.scientificName());
012    assertEquals("Animalia", Kingdom.ANIMALIA.scientificName());
013    assertEquals("Chromista", Kingdom.CHROMISTA.scientificName());
014    assertEquals("incertae sedis", Kingdom.INCERTAE_SEDIS.scientificName());
015  }
016
017  @Test
018  public void testNubUsageID() throws Exception {
019    assertEquals((Integer)0, Kingdom.INCERTAE_SEDIS.nubUsageID());
020    assertEquals((Integer)1, Kingdom.ANIMALIA.nubUsageID());
021    assertEquals((Integer)6, Kingdom.PLANTAE.nubUsageID());
022  }
023
024  @Test
025  public void testClbNameID() throws Exception {
026    assertEquals((Integer)9, Kingdom.INCERTAE_SEDIS.clbNameID());
027    assertEquals((Integer)1, Kingdom.ANIMALIA.clbNameID());
028    assertEquals((Integer)6, Kingdom.PLANTAE.clbNameID());
029  }
030}
