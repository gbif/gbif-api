001package org.gbif.api.vocabulary;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006
007public class KingdomTest {
008
009  @Test
010  public void testGetScientificName() throws Exception {
011    assertEquals("Plantae", Kingdom.PLANTAE.scientificName());
012    assertEquals("Animalia", Kingdom.ANIMALIA.scientificName());
013    assertEquals("Chromista", Kingdom.CHROMISTA.scientificName());
014    assertEquals("incertae sedis", Kingdom.INCERTAE_SEDIS.scientificName());
015  }
016
017  @Test
018  public void testNubUsageKey() throws Exception {
019    assertEquals(0, Kingdom.INCERTAE_SEDIS.nubUsageKey());
020    assertEquals(1, Kingdom.ANIMALIA.nubUsageKey());
021    assertEquals(6, Kingdom.PLANTAE.nubUsageKey());
022
023    for (Kingdom k : Kingdom.values()) {
024      assertEquals(k.ordinal(), k.nubUsageKey());
025    }
026  }
027
028  @Test
029  public void testByKey() throws Exception {
030    assertEquals(Kingdom.INCERTAE_SEDIS, Kingdom.byNubUsageKey(0));
031    assertEquals(Kingdom.ANIMALIA, Kingdom.byNubUsageKey(1));
032    assertEquals(Kingdom.PLANTAE, Kingdom.byNubUsageKey(6));
033    assertEquals(Kingdom.VIRUSES, Kingdom.byNubUsageKey(8));
034  }
035
036  @Test(expected = IllegalArgumentException.class)
037  public void testByKeyException() throws Exception {
038    Kingdom.byNubUsageKey(9);
039  }
040
041  @Test(expected = IllegalArgumentException.class)
042  public void testByKeyException2() throws Exception {
043    Kingdom.byNubUsageKey(-1);
044  }
045}
