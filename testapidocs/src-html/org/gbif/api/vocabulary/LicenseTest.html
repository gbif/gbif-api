001package org.gbif.api.vocabulary;
002
003
004import org.junit.Test;
005
006import static org.junit.Assert.assertEquals;
007import static org.junit.Assert.assertFalse;
008import static org.junit.Assert.assertNull;
009import static org.junit.Assert.assertTrue;
010
011public class LicenseTest {
012
013  @Test
014  public void testFromString() throws Exception {
015    // positive matches
016    assertEquals(License.CC0_1_0, License.fromLicenseUrl("http://creativecommons.org/publicdomain/zero/1.0/legalcode").get());
017    assertEquals(License.CC0_1_0, License.fromLicenseUrl("HTTP://creativecommons.org/publicdomain/zero/1.0/legalcode").get());
018    assertEquals(License.CC0_1_0, License.fromLicenseUrl(" http://creativecommons.org/publicdomain/zero/1.0/legalcode ").get());
019    assertEquals(License.CC0_1_0, License.fromLicenseUrl("http://creativecommons.org/publicdomain/zero/1.0/legalcode/").get());
020    assertEquals(License.CC0_1_0, License.fromLicenseUrl("http://creativecommons.org/publicdomain/zero/1.0").get());
021    assertEquals(License.CC0_1_0, License.fromLicenseUrl("http://creativecommons.org/publicdomain/zero/1.0/").get());
022    assertEquals(License.CC0_1_0, License.fromLicenseUrl("HTTPS://CREATIVECOMMONS.ORG/PUBLICDOMAIN/ZERO/1.0/").get());
023    assertEquals(License.CC_BY_4_0, License.fromLicenseUrl("http://creativecommons.org/licenses/by/4.0/legalcode").get());
024    assertEquals(License.CC_BY_NC_4_0, License.fromLicenseUrl("http://creativecommons.org/licenses/by-nc/4.0/legalcode").get());
025
026    // negative matches
027    assertNull(License.fromLicenseUrl("CC0").orNull());
028    assertNull(License.fromLicenseUrl("https://creativecommons.org/licenses/by/3.0/legalcode").orNull());
029    assertNull(License.fromLicenseUrl("https://creativecommons.org/licenses/by/3.0/").orNull());
030    assertNull(License.fromLicenseUrl("https://creativecommons.org/licenses/by-nc/2.0/legalcode").orNull());
031    assertNull(License.fromLicenseUrl("https://creativecommons.org/licenses/by-nc/2.0/").orNull());
032  }
033
034  @Test
035  public void testConcreteLicense() {
036    assertFalse(License.UNSPECIFIED.isConcrete());
037    assertFalse(License.UNSUPPORTED.isConcrete());
038    assertTrue(License.CC0_1_0.isConcrete());
039  }
040
041  @Test
042  public void testGetMostRestrictive(){
043    assertEquals(License.CC_BY_4_0, License.getMostRestrictive(License.CC0_1_0, License.CC_BY_4_0, License.UNSPECIFIED));
044    assertEquals(License.CC_BY_4_0, License.getMostRestrictive(License.CC_BY_4_0, License.CC0_1_0, License.UNSPECIFIED));
045
046    assertEquals(License.CC_BY_NC_4_0, License.getMostRestrictive(License.CC_BY_4_0, License.CC_BY_NC_4_0, License.UNSPECIFIED));
047
048    assertEquals(License.UNSPECIFIED, License.getMostRestrictive(License.UNSUPPORTED, License.CC0_1_0, License.UNSPECIFIED));
049    assertEquals(License.UNSPECIFIED, License.getMostRestrictive(License.UNSUPPORTED, License.UNSPECIFIED, License.UNSPECIFIED));
050    assertEquals(License.UNSPECIFIED, License.getMostRestrictive(License.CC_BY_NC_4_0, null, License.UNSPECIFIED));
051  }
052
053  @Test
054  public void testLicenseOrdinalNumber() {
055    int numberOfConcreteLicense = 0;
056    for (License currLicense : License.values()) {
057      if (currLicense.isConcrete()) {
058        numberOfConcreteLicense++;
059      }
060    }
061
062    //The ordinal number in the Enum implicitly defines the level of restriction.
063    assertEquals("The License value ordinal test covers all values of the Concrete License", 3, numberOfConcreteLicense);
064    assertTrue(License.CC0_1_0.ordinal() < License.CC_BY_4_0.ordinal());
065    assertTrue(License.CC_BY_4_0.ordinal() < License.CC_BY_NC_4_0.ordinal());
066  }
067}
