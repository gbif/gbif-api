001package org.gbif.api.model.crawler;
002
003import java.util.UUID;
004
005import org.junit.Test;
006
007import static org.junit.Assert.*;
008
009public class OccurrenceValidationReportTest {
010
011  @Test
012  public void testEmpty() {
013    OccurrenceValidationReport report = new OccurrenceValidationReport(0, 0, 0, 0, 0, true);
014    assertNull(report.getInvalidationReason());
015    assertTrue(report.isValid());
016  }
017
018  @Test
019  public void testGoodTripletsNoOccId() {
020    OccurrenceValidationReport report = new OccurrenceValidationReport(100, 100, 0, 0, 100, true);
021    assertNull(report.getInvalidationReason());
022    assertTrue(report.isValid());
023  }
024
025  @Test
026  public void testMissingTripletsGoodOccId() {
027    OccurrenceValidationReport report = new OccurrenceValidationReport(100, 0, 100, 100, 0, true);
028    assertNull(report.getInvalidationReason());
029    assertTrue(report.isValid());
030  }
031
032  @Test
033  public void testDuplicateTripletsGoodOccId() {
034    OccurrenceValidationReport report = new OccurrenceValidationReport(100, 80, 0, 100, 0, true);
035    assertNull(report.getInvalidationReason());
036    assertTrue(report.isValid());
037  }
038
039  @Test
040  public void testMissingTripletsNoOccId() {
041    OccurrenceValidationReport report = new OccurrenceValidationReport(100, 70, 30, 0, 100, true);
042    assertFalse(report.isValid());
043    assertEquals(
044      "Archive invalid because [30% invalid triplets is > than threshold of 25%; 100 records without an occurrence id "
045      + "(should be 0)]",
046      report.getInvalidationReason());
047  }
048
049  @Test
050  public void testDuplicateTripletsNoOccId() {
051    OccurrenceValidationReport report = new OccurrenceValidationReport(100, 80, 0, 0, 100, false);
052    assertEquals(
053      "Archive invalid because [20 duplicate triplets detected; 100 records without an occurrence id (should be 0)]",
054      report.getInvalidationReason());
055    assertFalse(report.isValid());
056  }
057
058  @Test
059  public void testBadEverything() {
060    OccurrenceValidationReport report = new OccurrenceValidationReport(100, 50, 28, 80, 5, true);
061    assertEquals(
062      "Archive invalid because [28% invalid triplets is > than threshold of 25%; 22 duplicate triplets detected; 5 "
063      + "records without an occurrence id (should be 0); 15 duplicate occurrence ids detected]",
064      report.getInvalidationReason());
065    assertFalse(report.isValid());
066  }
067}
