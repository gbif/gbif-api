001package org.gbif.api.model.crawler;
002
003import org.gbif.api.vocabulary.Origin;
004import org.gbif.api.vocabulary.Rank;
005
006import com.google.common.collect.ImmutableMap;
007import com.google.common.collect.Lists;
008import org.junit.Test;
009
010import static org.junit.Assert.*;
011
012public class NormalizerStatsTest {
013
014    @Test
015    public void testGetCountByRank() throws Exception {
016        NormalizerStats stats = new NormalizerStats(1, 2, 12, 1,
017                ImmutableMap.of(Origin.DENORMED_CLASSIFICATION, 2, Origin.IMPLICIT_NAME, 10, Origin.SOURCE, 50),
018                ImmutableMap.of(Rank.KINGDOM, 2, Rank.GENUS, 10, Rank.SPECIES, 50),
019                Lists.<String>newArrayList());
020        assertEquals(62, stats.getCount());
021        assertEquals(12, stats.getSynonyms());
022        assertEquals(2, stats.getDepth());
023        assertEquals(1, stats.getRoots());
024        assertEquals(1, stats.getIgnored());
025
026        assertEquals(0, stats.getCountByOrigin(Origin.AUTO_RECOMBINATION));
027        assertEquals(50, stats.getCountByOrigin(Origin.SOURCE));
028        assertEquals(2, stats.getCountByOrigin(Origin.DENORMED_CLASSIFICATION));
029        assertEquals(10, stats.getCountByOrigin(Origin.IMPLICIT_NAME));
030
031        assertEquals(50, stats.getCountByRank(Rank.SPECIES));
032        assertEquals(10, stats.getCountByRank(Rank.GENUS));
033        assertEquals(2, stats.getCountByRank(Rank.KINGDOM));
034        assertEquals(0, stats.getCountByRank(Rank.FAMILY));
035    }
036
037}
