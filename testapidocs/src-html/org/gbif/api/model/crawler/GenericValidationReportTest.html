001package org.gbif.api.model.crawler;
002
003import com.google.common.collect.Lists;
004import org.junit.Test;
005
006import static org.junit.Assert.*;
007
008public class GenericValidationReportTest {
009
010  @Test(expected = NullPointerException.class)
011  public void testConstructor() throws Exception {
012    GenericValidationReport report = new GenericValidationReport(10, true, null, null);
013  }
014
015  @Test(expected = NullPointerException.class)
016  public void testConstructor2() throws Exception {
017    GenericValidationReport report = new GenericValidationReport(10, true, Lists.<String>newArrayList(), null);
018  }
019
020  @Test(expected = NullPointerException.class)
021  public void testConstructor3() throws Exception {
022    GenericValidationReport report = new GenericValidationReport(10, true, null, Lists.<Integer>newArrayList());
023  }
024
025  @Test
026  public void testIsValid() throws Exception {
027    assertTrue(new GenericValidationReport(10, true, Lists.<String>newArrayList(), Lists.<Integer>newArrayList()).isValid());
028    assertTrue(new GenericValidationReport(0, true, Lists.<String>newArrayList(), Lists.<Integer>newArrayList()).isValid());
029    assertFalse(new GenericValidationReport(10, true, Lists.<String>newArrayList(), Lists.<Integer>newArrayList(1)).isValid());
030    assertFalse(new GenericValidationReport(10, true, Lists.<String>newArrayList(), Lists.<Integer>newArrayList(1,2,3,4)).isValid());
031    assertFalse(new GenericValidationReport(10, true, Lists.<String>newArrayList("r32"), Lists.<Integer>newArrayList(1,2)).isValid());
032    assertFalse(new GenericValidationReport(10, true, Lists.<String>newArrayList("r32"), Lists.<Integer>newArrayList()).isValid());
033  }
034
035}
