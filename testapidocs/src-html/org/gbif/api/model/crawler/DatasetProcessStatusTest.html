016package org.gbif.api.model.crawler;
017
018import org.gbif.api.vocabulary.EndpointType;
019
020import java.io.IOException;
021import java.net.URI;
022import java.util.Date;
023import java.util.UUID;
024
025import org.codehaus.jackson.map.ObjectMapper;
026import org.junit.Test;
027
028import static org.junit.Assert.assertEquals;
029
030public class DatasetProcessStatusTest {
031
032  @Test
033  public void testEquals() throws IOException {
034    UUID uuid = UUID.randomUUID();
035    Date now = new Date();
036    DatasetProcessStatus s1 = DatasetProcessStatus.builder()
037      .datasetKey(uuid)
038      .startedCrawling(now)
039      .crawlJob(new CrawlJob(uuid, EndpointType.BIOCASE, URI.create("http://www.foo.com"), 1, null))
040      .crawlContext("foo")
041      .processStateOccurrence(ProcessState.RUNNING)
042      .processStateChecklist(ProcessState.EMPTY)
043      .declaredCount(100L)
044      .pagesCrawled(10L)
045      .pagesFragmentedSuccessful(11L)
046      .pagesFragmentedError(12L)
047      .fragmentsEmitted(13L)
048      .fragmentsReceived(14L)
049      .fragmentsProcessed(15L)
050      .rawOccurrencesPersistedNew(16L)
051      .rawOccurrencesPersistedUnchanged(17L)
052      .rawOccurrencesPersistedUpdated(18L)
053      .rawOccurrencesPersistedError(19L)
054      .verbatimOccurrencesPersistedSuccessful(20L)
055      .verbatimOccurrencesPersistedError(21L)
056      .interpretedOccurrencesPersistedSuccessful(22L)
057      .interpretedOccurrencesPersistedError(23L)
058      .build();
059
060    DatasetProcessStatus s2 = DatasetProcessStatus.builder()
061      .datasetKey(uuid)
062      .startedCrawling(new Date(now.getTime()))
063      .crawlJob(new CrawlJob(uuid, EndpointType.BIOCASE, URI.create("http://www.foo.com"), 1, null))
064      .crawlContext("foo")
065      .processStateOccurrence(ProcessState.RUNNING)
066      .processStateChecklist(ProcessState.EMPTY)
067      .declaredCount(100L)
068      .pagesCrawled(10L)
069      .pagesFragmentedSuccessful(11L)
070      .pagesFragmentedError(12L)
071      .fragmentsEmitted(13L)
072      .fragmentsReceived(14L)
073      .fragmentsProcessed(15L)
074      .rawOccurrencesPersistedNew(16L)
075      .rawOccurrencesPersistedUnchanged(17L)
076      .rawOccurrencesPersistedUpdated(18L)
077      .rawOccurrencesPersistedError(19L)
078      .verbatimOccurrencesPersistedSuccessful(20L)
079      .verbatimOccurrencesPersistedError(21L)
080      .interpretedOccurrencesPersistedSuccessful(22L)
081      .interpretedOccurrencesPersistedError(23L)
082      .build();
083
084    assertEquals(s1, s2);
085    assertEquals(s1.hashCode(), s2.hashCode());
086
087    ObjectMapper om = new ObjectMapper();
088
089    byte[] bytes = om.writeValueAsBytes(s1);
090    DatasetProcessStatus s3 = om.readValue(bytes, DatasetProcessStatus.class);
091    assertEquals(s1, s3);
092    assertEquals(s1.hashCode(), s3.hashCode());
093  }
094
095
096}
