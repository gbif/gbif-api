023public class GbifUserTest {
024
025  private Validator validator = Validation.buildDefaultValidatorFactory().getValidator();
026
027  @Test
028  public void testHasRole() throws Exception {
029    GbifUser u = new GbifUser();
030    u.setUserName("betty");
031    u.setEmail("betty@gbif.org");
032    u.setFirstName("Betty Ford");
033
034    assertFalse(u.hasRole(UserRole.USER));
035    assertFalse(u.hasRole(UserRole.REGISTRY_EDITOR));
036    assertFalse(u.hasRole(UserRole.ADMIN));
037    assertFalse(u.hasRole(null));
038
039    u.addRole(UserRole.USER);
040    assertTrue(u.hasRole(UserRole.USER));
041    assertFalse(u.hasRole(UserRole.REGISTRY_EDITOR));
042    assertFalse(u.hasRole(UserRole.ADMIN));
043    assertFalse(u.hasRole(null));
044
045    u.addRole(UserRole.ADMIN);
046    assertTrue(u.hasRole(UserRole.USER));
047    assertFalse(u.hasRole(UserRole.REGISTRY_EDITOR));
048    assertTrue(u.hasRole(UserRole.ADMIN));
049    assertFalse(u.hasRole(null));
050  }
051
052  @Test
053  public void testValidation() throws Exception {
054    GbifUser u = new GbifUser();
055    u.setKey(100);
056    u.setUserName("be");
057    u.setEmail("betty@gbif.org");
058    u.setFirstName("Betty");
059    u.setLastName("Ford");
060    u.setPasswordHash("a");
061    assertValidation("userName", u);
062
063    u.setUserName("betty");
064    u.setEmail("betty at gbif.org");
065    assertValidation("email", u);
066  }
067
068  @Test
069  public void testEmailValidation() throws Exception {
070    GbifUser u = new GbifUser();
071    u.setKey(100);
072    u.setUserName("betty");
073    u.setEmail("b.betty@pre.name-dash.org");
074    u.setFirstName("Betty");
075    u.setLastName("Ford");
076    u.setPasswordHash("a");
077    assertValidation(null, u);
078  }
079
080  private void assertValidation(String property, GbifUser u) {
081    Set<ConstraintViolation<GbifUser>> err = validator.validate(u);
082    System.out.print(err);
083    if (property == null) {
084      assertEquals(0, err.size());
085    } else {
086      assertEquals(1, err.size());
087      assertEquals(property, err.iterator().next().getPropertyPath().toString());
088    }
089  }
