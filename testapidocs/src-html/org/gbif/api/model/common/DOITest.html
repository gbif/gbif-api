001package org.gbif.api.model.common;
002
003import org.codehaus.jackson.map.ObjectMapper;
004import org.junit.Test;
005
006import static org.junit.Assert.assertEquals;
007import static org.junit.Assert.assertFalse;
008import static org.junit.Assert.assertTrue;
009import static org.junit.Assert.fail;

011public class DOITest {
012
013  @Test
014  public void testPrefixConstructor() throws Exception {
015    DOI d = new DOI("10.1234", "1ASCDU");
016    assertEquals(d, new DOI("doi:10.1234/1ASCDU"));
017  }
018
019  @Test
020  public void testCreate() throws Exception {
021    DOI d = new DOI("http://dx.doi.org/10.1234/1ASCDU");
022
023    assertSame(d, "http://dx.doi.org/urn:doi:10.1234/1ASCDU");
024    assertSame(d, "http://doi.org/urn:doi:10.1234/1ASCDU");
025    assertSame(d, "http://dx.doi.org/doi:10.1234/1ASCDU");
026    assertSame(d, "http://doi.org/doi:10.1234/1ASCDU");
027    assertSame(d, "http://dx.doi.org/10.1234/1ASCDU");
028    assertSame(d, "http://doi.org/10.1234/1ASCDU");
029    assertSame(d, "https://dx.doi.org/10.1234/1ASCDU");
030    assertSame(d, "https://doi.org/10.1234/1ASCDU");
031
032    assertSame(d, "doi:10.1234/1ASCDU");
033    assertSame(d, "urn:doi:10.1234/1ASCDU");
034    assertSame(d, "doi:10.1234/1ASCDU");
035    assertSame(d, "urn:doi:10.1234/1ASCDU");
036    assertSame(d, "doi:10.1234/1ascdu");
037    assertSame(d, "urn:doi:10.1234/1ascdu");
038    assertSame(d, "10.1234/1ascdu");
039    assertSame(d, "10.1234/1ASCDU");
040
041    // try subdivisions
042    new DOI("http://dx.doi.org/10.1234.999/1ASCDU");
043    new DOI("http://dx.doi.org/10.1234.999.123/1ASCDU");
044    new DOI("http://dx.doi.org/10.1234.999.123/1ASCDU/more/and/more");
045    new DOI("http://dx.doi.org/10.1234.999.123/1ASCDU.more.and/more");
046    new DOI("http://dx.doi.org/10.1234.999.123/unicode-isällöwéd_h&$e");
047
048    // try invalid ones
049    assertInvalid("uri:doi:10.1234/1ascdu");
050    assertInvalid("doi::10.1234/1ascdu");
051    assertInvalid("10/1234/1ascdu");
052    assertInvalid("doi");
053    assertInvalid("Tim Robertson");
054    assertInvalid("http://dx.doi.org");
055    assertInvalid("http://dx.doi.org/10");
056    assertInvalid("http://doi.org");
057    assertInvalid("http://doi.org/10.453621432764678123");
058    assertInvalid("http://dx.doi.org/11.1234.999/1ASCDU");
059  }
060
061  @Test
062  public void testUrlConversion() throws Exception {
063    DOI d = new DOI("http://doi.org/10.1000/456%23789");
064    assertEquals("456#789", d.getSuffix());
065    assertSame(d, "10.1000/456#789");
066  }
067
068  private void assertSame(DOI doi, String doi2) {
069    DOI d2 = new DOI(doi2);
070    assertEquals(d2, doi);
071  }
072
073  private void assertInvalid(String doi) {
074    try {
075      new DOI(doi);
076      fail("Invalid DOI expected: " + doi);
077    } catch (IllegalArgumentException e) {
078      // expected!
079    }
080  }
081
082
083
084  /**
085   * A container of Languages using 2 properties.
086   */
087  private static class Container {
088
089    public DOI doi1;
090    public DOI doi2; // verify that the names don't matter
091
092    public Container(DOI doi1, DOI doi2) {
093      this.doi1 = doi1;
094      this.doi2 = doi2;
095    }
096
097    // used by serializer below
098    public Container() {
099    }
100  }
101
102  @Test
103  public void testSerDe() {
104    ObjectMapper mapper = new ObjectMapper();
105    Container source = new Container(new DOI("doi:10.1038/nature.2014.16460"), new DOI("http://dx.doi.org/10.1034/gbif.2014.xscdf2"));
106
107    try {
108      String json = mapper.writeValueAsString(source);
109      assertTrue("DOIs should start with doi:10.", json.contains("\"doi:10."));
110      assertFalse("DOIs should not use http resolvers", json.contains("http"));
111      assertFalse("DOIs should not use http resolvers", json.contains("doi.org"));
112      System.out.println(json);
113      Container target = mapper.readValue(json, Container.class);
114      assertEquals("doi1 differs", source.doi1, target.doi1);
115      assertEquals("doi2 differs", source.doi2, target.doi2);
116    } catch (Exception e) {
117      fail(e.getMessage());
118    }
119  }
120}
