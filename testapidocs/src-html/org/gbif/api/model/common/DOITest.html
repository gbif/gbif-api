001package org.gbif.api.model.common;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006import static org.junit.Assert.fail;
007
008public class DOITest {
009
010  @Test
011  public void testCreate() throws Exception {
012    DOI d = new DOI("http://dx.doi.org/10.1234/1ASCDU");
013
014    assertSame(d, "http://dx.doi.org/urn:doi:10.1234/1ASCDU");
015    assertSame(d, "http://doi.org/urn:doi:10.1234/1ASCDU");
016    assertSame(d, "http://dx.doi.org/doi:10.1234/1ASCDU");
017    assertSame(d, "http://doi.org/doi:10.1234/1ASCDU");
018    assertSame(d, "http://dx.doi.org/10.1234/1ASCDU");
019    assertSame(d, "http://doi.org/10.1234/1ASCDU");
020    assertSame(d, "https://dx.doi.org/10.1234/1ASCDU");
021    assertSame(d, "https://doi.org/10.1234/1ASCDU");
022
023    assertSame(d, "doi:10.1234/1ASCDU");
024    assertSame(d, "urn:doi:10.1234/1ASCDU");
025    assertSame(d, "doi:10.1234/1ASCDU");
026    assertSame(d, "urn:doi:10.1234/1ASCDU");
027    assertSame(d, "doi:10.1234/1ascdu");
028    assertSame(d, "urn:doi:10.1234/1ascdu");
029    assertSame(d, "10.1234/1ascdu");
030    assertSame(d, "10.1234/1ASCDU");
031
032    // try subdivisions
033    new DOI("http://dx.doi.org/10.1234.999/1ASCDU");
034    new DOI("http://dx.doi.org/10.1234.999.123/1ASCDU");
035    new DOI("http://dx.doi.org/10.1234.999.123/1ASCDU/more/and/more");
036    new DOI("http://dx.doi.org/10.1234.999.123/1ASCDU.more.and/more");
037    new DOI("http://dx.doi.org/10.1234.999.123/unicode-isällöwéd_h&$e");
038
039    // try invalid ones
040    assertInvalid("uri:doi:10.1234/1ascdu");
041    assertInvalid("doi::10.1234/1ascdu");
042    assertInvalid("10/1234/1ascdu");
043    assertInvalid("doi");
044    assertInvalid("Tim Robertson");
045    assertInvalid("http://dx.doi.org");
046    assertInvalid("http://dx.doi.org/10");
047    assertInvalid("http://doi.org");
048    assertInvalid("http://doi.org/10.453621432764678123");
049    assertInvalid("http://dx.doi.org/11.1234.999/1ASCDU");
050  }
051
052  @Test
053  public void testUrlConversion() throws Exception {
054    DOI d = new DOI("http://doi.org/10.1000/456%23789");
055    assertEquals("456#789", d.getSuffix());
056    assertSame(d, "10.1000/456#789");
057  }
058
059  private void assertSame(DOI doi, String doi2) {
060    DOI d2 = new DOI(doi2);
061    assertEquals(d2, doi);
062  }
063
064  private void assertInvalid(String doi) {
065    try {
066      new DOI(doi);
067      fail("Invalid DOI expected: " + doi);
068    } catch (IllegalArgumentException e) {
069      // expected!
070    }
071  }
072}
