023  @Test
024  public void testIsParsable() throws Exception {
025    assertFalse(DOI.isParsable(null));
026    assertFalse(DOI.isParsable(""));
027    assertFalse(DOI.isParsable("10.1234.1ASCDU"));
028    assertFalse(DOI.isParsable("DOI:123"));
029    assertFalse(DOI.isParsable("   "));
030
031    assertTrue(DOI.isParsable("http://dx.doi.org/urn:doi:10.1234/1ASCDU"));
032    assertTrue(DOI.isParsable("http://doi.org/urn:doi:10.1234/1ASCDU"));
033    assertTrue(DOI.isParsable("http://dx.doi.org/doi:10.1234/1ASCDU"));
034    assertTrue(DOI.isParsable("http://doi.org/doi:10.1234/1ASCDU"));
035    assertTrue(DOI.isParsable("http://dx.doi.org/10.1234/1ASCDU"));
036    assertTrue(DOI.isParsable("http://doi.org/10.1234/1ASCDU"));
037    assertTrue(DOI.isParsable("https://dx.doi.org/10.1234/1ASCDU"));
038    assertTrue(DOI.isParsable("https://doi.org/10.1234/1ASCDU"));
039
040    assertTrue(DOI.isParsable("doi:10.1234/1ASCDU"));
041    assertTrue(DOI.isParsable("urn:doi:10.1234/1ASCDU"));
042    assertTrue(DOI.isParsable("doi:10.1234/1ASCDU"));
043    assertTrue(DOI.isParsable("urn:doi:10.1234/1ASCDU"));
044    assertTrue(DOI.isParsable("doi:10.1234/1ascdu"));
045    assertTrue(DOI.isParsable("urn:doi:10.1234/1ascdu"));
046    assertTrue(DOI.isParsable("10.1234/1ascdu"));
047    assertTrue(DOI.isParsable("10.1234/1ASCDU"));
048
049    // try subdivisions
050    assertTrue(DOI.isParsable("http://dx.doi.org/10.1234.999/1ASCDU"));
051    assertTrue(DOI.isParsable("http://dx.doi.org/10.1234.999.123/1ASCDU"));
052    assertTrue(DOI.isParsable("http://dx.doi.org/10.1234.999.123/1ASCDU/more/and/more"));
053    assertTrue(DOI.isParsable("http://dx.doi.org/10.1234.999.123/1ASCDU.more.and/more"));
054    assertTrue(DOI.isParsable("http://dx.doi.org/10.1234.999.123/unicode-isällöwéd_h&$e"));
055  }

062    assertSame(d, "http://doi.org/urn:doi:10.1234/1ASCDU");
063    assertSame(d, "http://dx.doi.org/doi:10.1234/1ASCDU");
064    assertSame(d, "http://doi.org/doi:10.1234/1ASCDU");
065    assertSame(d, "http://dx.doi.org/10.1234/1ASCDU");
066    assertSame(d, "http://doi.org/10.1234/1ASCDU");
067    assertSame(d, "https://dx.doi.org/10.1234/1ASCDU");
068    assertSame(d, "https://doi.org/10.1234/1ASCDU");
069
070    assertSame(d, "doi:10.1234/1ASCDU");
071    assertSame(d, "urn:doi:10.1234/1ASCDU");
072    assertSame(d, "doi:10.1234/1ASCDU");
073    assertSame(d, "urn:doi:10.1234/1ASCDU");
074    assertSame(d, "doi:10.1234/1ascdu");
075    assertSame(d, "urn:doi:10.1234/1ascdu");
076    assertSame(d, "10.1234/1ascdu");
077    assertSame(d, "10.1234/1ASCDU");
078
079    // try subdivisions
080    new DOI("http://dx.doi.org/10.1234.999/1ASCDU");
081    new DOI("http://dx.doi.org/10.1234.999.123/1ASCDU");
082    new DOI("http://dx.doi.org/10.1234.999.123/1ASCDU/more/and/more");
083    new DOI("http://dx.doi.org/10.1234.999.123/1ASCDU.more.and/more");
084    new DOI("http://dx.doi.org/10.1234.999.123/unicode-isällöwéd_h&$e");

086    // try invalid ones
087    assertInvalid("uri:doi:10.1234/1ascdu");
088    assertInvalid("doi::10.1234/1ascdu");
089    assertInvalid("10/1234/1ascdu");
090    assertInvalid("doi");
091    assertInvalid("Tim Robertson");
092    assertInvalid("http://dx.doi.org");

095    assertInvalid("http://doi.org/10.453621432764678123");
096    assertInvalid("http://dx.doi.org/11.1234.999/1ASCDU");

122  /**
123   * A container of Languages using 2 properties.
124   */
125  private static class Container {
126
127    public DOI doi1;
128    public DOI doi2; // verify that the names don't matter
129
130    public Container(DOI doi1, DOI doi2) {
131      this.doi1 = doi1;
132      this.doi2 = doi2;
133    }
134
135    // used by serializer below
136    public Container() {
137    }
138
139    @Override
140    public int hashCode() {
141      return Objects.hashCode(doi1, doi2);
142    }
143
144    @Override
145    public boolean equals(Object obj) {
146      if (this == obj) {
147        return true;
148      }
149      if (obj == null || getClass() != obj.getClass()) {
150        return false;
151      }
152      final Container other = (Container) obj;
153      return Objects.equal(this.doi1, other.doi1) && Objects.equal(this.doi2, other.doi2);
154    }
155  }

157  @Test
158  public void testSerDe() throws IOException {
159    DOI d = new DOI("10.1234", "1ASCDU");
160    SerdeTestUtils.testSerDe(d, DOI.class);
161
162    Container source = new Container(new DOI("doi:10.1038/nature.2014.16460"), new DOI("http://dx.doi.org/10.1034/gbif.2014.xscdf2"));
163
164    try {
165      String json = SerdeTestUtils.testSerDe(source, Container.class);
166      assertTrue("DOIs should start with doi:10.", json.contains("\"doi:10."));
167      assertFalse("DOIs should not use http resolvers", json.contains("http"));
168      assertFalse("DOIs should not use http resolvers", json.contains("doi.org"));
169      System.out.println(json);
170    } catch (Exception e) {
171      fail(e.getMessage());
172    }
173  }
