024  @Test
025  public void testIsParsable() throws Exception {
026    assertFalse(DOI.isParsable(null));
027    assertFalse(DOI.isParsable(""));
028    assertFalse(DOI.isParsable("10.1234.1ASCDU"));
029    assertFalse(DOI.isParsable("DOI:123"));
030    assertFalse(DOI.isParsable("   "));
031
032    assertTrue(DOI.isParsable("http://dx.doi.org/urn:doi:10.1234/1ASCDU"));
033    assertTrue(DOI.isParsable("http://doi.org/urn:doi:10.1234/1ASCDU"));
034    assertTrue(DOI.isParsable("http://dx.doi.org/doi:10.1234/1ASCDU"));
035    assertTrue(DOI.isParsable("http://doi.org/doi:10.1234/1ASCDU"));
036    assertTrue(DOI.isParsable("http://dx.doi.org/10.1234/1ASCDU"));
037    assertTrue(DOI.isParsable("http://doi.org/10.1234/1ASCDU"));
038    assertTrue(DOI.isParsable("https://dx.doi.org/10.1234/1ASCDU"));
039    assertTrue(DOI.isParsable("https://doi.org/10.1234/1ASCDU"));
040
041    assertTrue(DOI.isParsable("doi:10.1234/1ASCDU"));
042    assertTrue(DOI.isParsable("urn:doi:10.1234/1ASCDU"));
043    assertTrue(DOI.isParsable("doi:10.1234/1ASCDU"));
044    assertTrue(DOI.isParsable("urn:doi:10.1234/1ASCDU"));
045    assertTrue(DOI.isParsable("doi:10.1234/1ascdu"));
046    assertTrue(DOI.isParsable("urn:doi:10.1234/1ascdu"));
047    assertTrue(DOI.isParsable("10.1234/1ascdu"));
048    assertTrue(DOI.isParsable("10.1234/1ASCDU"));
049
050    // This is real.
051    assertTrue(DOI.isParsable("10.1643/0045-8511(2007)2007[699:Tnvsot]2.0.Co;2"));
052    assertTrue(DOI.isParsable("doi:10.1643/0045-8511(2007)2007[699:Tnvsot]2.0.Co;2"));
053    assertTrue(DOI.isParsable("https://doi.org/10.1643/0045-8511(2007)2007%5b699%3aTnvsot%5d2.0.Co%3b2"));
054
055    // try subdivisions
056    assertTrue(DOI.isParsable("http://dx.doi.org/10.1234.999/1ASCDU"));
057    assertTrue(DOI.isParsable("http://dx.doi.org/10.1234.999.123/1ASCDU"));
058    assertTrue(DOI.isParsable("http://dx.doi.org/10.1234.999.123/1ASCDU/more/and/more"));
059    assertTrue(DOI.isParsable("http://dx.doi.org/10.1234.999.123/1ASCDU.more.and/more"));
060    assertTrue(DOI.isParsable("http://dx.doi.org/10.1234.999.123/unicode-isällöwéd_h&$e"));
061  }

107  @Test

109    DOI d = new DOI("http://doi.org/10.1000/456%23789");
110    assertEquals("456#789", d.getSuffix());

131  /**
132   * A container of Languages using 2 properties.
133   */
134  private static class Container {
135
136    public DOI doi1;
137    public DOI doi2; // verify that the names don't matter
138
139    public Container(DOI doi1, DOI doi2) {
140      this.doi1 = doi1;
141      this.doi2 = doi2;
142    }
143
144    // used by serializer below
145    public Container() {
146    }
147
148    @Override
149    public int hashCode() {
150      return Objects.hashCode(doi1, doi2);
151    }
152
153    @Override
154    public boolean equals(Object obj) {
155      if (this == obj) {
156        return true;
157      }
158      if (obj == null || getClass() != obj.getClass()) {
159        return false;
160      }
161      final Container other = (Container) obj;
162      return Objects.equal(this.doi1, other.doi1) && Objects.equal(this.doi2, other.doi2);
163    }
164  }

166  @Test
167  public void testSerDe() throws IOException {
168    DOI d = new DOI("10.1234", "1ASCDU");
169    SerdeTestUtils.testSerDe(d, DOI.class);
170
171    Container source = new Container(new DOI("doi:10.1038/nature.2014.16460"), new DOI("http://dx.doi.org/10.1034/gbif.2014.xscdf2"));
172    try {
173      String json = SerdeTestUtils.testSerDe(source, Container.class);
174      assertFalse("DOIs should be shown without a scheme", json.contains("doi:10."));
175      assertFalse("DOIs should not use http resolvers", json.contains("http"));
176      assertFalse("DOIs should not use http resolvers", json.contains("doi.org"));
177      System.out.println(json);
178    } catch (Exception e) {
179      fail(e.getMessage());
180    }
181  }
