016package org.gbif.api.model.common;
017
018import org.gbif.api.vocabulary.UserRole;
019
020import javax.validation.Validation;
021import javax.validation.Validator;
022
023import org.junit.Test;
024
025import static org.junit.Assert.assertFalse;
026import static org.junit.Assert.assertTrue;
027
028public class UserPrincipalTest {
029  private Validator validator = Validation.buildDefaultValidatorFactory().getValidator();
030
031  @Test
032  public void testHasRole() throws Exception {
033    User user = new User();
034    user.setKey(100);
035    user.setUserName("betty");
036    user.setEmail("betty@gbif.org");
037    user.setFirstName("Betty");
038    user.setLastName("Ford");
039    UserPrincipal u = new UserPrincipal(user);
040
041    assertFalse(u.hasRole("USER"));
042    assertFalse(u.hasRole("User"));
043    assertFalse(u.hasRole("admin"));
044
045    user.addRole(UserRole.USER);
046    assertTrue(u.hasRole("user"));
047    assertTrue(u.hasRole("USER"));
048    assertTrue(u.hasRole("User"));
049    assertFalse(u.hasRole("admin"));
050
051    user.addRole(UserRole.ADMIN);
052    assertTrue(u.hasRole("User"));
053    assertTrue(u.hasRole("admin"));
054    assertTrue(u.hasRole("Admin"));
055  }
056
057}
