016package org.gbif.api.model.common;
017
018import org.gbif.api.SerdeTestUtils;
019import org.gbif.api.vocabulary.UserRole;
020
021import java.io.IOException;
022import java.util.Set;
023import javax.validation.ConstraintViolation;
024import javax.validation.Validation;
025import javax.validation.Validator;
026
027import org.junit.Test;
028
029import static org.junit.Assert.assertEquals;
030import static org.junit.Assert.assertFalse;
031import static org.junit.Assert.assertTrue;
032
033public class UserTest {
034  private Validator validator = Validation.buildDefaultValidatorFactory().getValidator();
035
036  @Test
037  public void testHasRole() throws Exception {
038    User u = new User();
039    u.setUserName("betty");
040    u.setEmail("betty@gbif.org");
041    u.setFirstName("Betty Ford");
042
043    assertFalse(u.hasRole(UserRole.USER));
044    assertFalse(u.hasRole(UserRole.REGISTRY_EDITOR));
045    assertFalse(u.hasRole(UserRole.ADMIN));
046    assertFalse(u.hasRole(null));
047
048    u.addRole(UserRole.USER);
049    assertTrue(u.hasRole(UserRole.USER));
050    assertFalse(u.hasRole(UserRole.REGISTRY_EDITOR));
051    assertFalse(u.hasRole(UserRole.ADMIN));
052    assertFalse(u.hasRole(null));
053
054    u.addRole(UserRole.ADMIN);
055    assertTrue(u.hasRole(UserRole.USER));
056    assertFalse(u.hasRole(UserRole.REGISTRY_EDITOR));
057    assertTrue(u.hasRole(UserRole.ADMIN));
058    assertFalse(u.hasRole(null));
059  }
060
061  @Test
062  public void testIsAdmin() throws Exception {
063    User u = new User();
064    u.setUserName("betty");
065    u.setEmail("betty@gbif.org");
066    u.setFirstName("Betty Ford");
067
068    assertFalse(u.isAdmin());
069
070    u.addRole(UserRole.ADMIN);
071    assertTrue(u.isAdmin());
072  }
073
074  @Test
075  public void testValidation() throws Exception {
076    User u = new User();
077    u.setKey(100);
078    u.setUserName("be");
079    u.setEmail("betty@gbif.org");
080    u.setFirstName("Betty");
081    u.setLastName("Ford");
082    assertValidation("userName", u);
083
084    u.setUserName("betty");
085    u.setFirstName(null);
086    assertValidation("firstName", u);
087
088    u.setFirstName("Betty");
089    assertValidation(null, u);
090
091    u.setEmail("betty at gbif.org");
092    assertValidation("email", u);
093  }
094
095  private void assertValidation(String property, User u) {
096    Set<ConstraintViolation<User>> err = validator.validate(u);
097    System.out.print(err);
098    if (property == null) {
099      assertEquals(0, err.size());
100    } else {
101      assertEquals(1, err.size());
102      assertEquals(property, err.iterator().next().getPropertyPath().toString());
103    }
104  }
105
106  @Test
107  public void testSerDe() throws IOException {
108    User u = new User();
109    u.setKey(100);
110    u.setUserName("be");
111    u.setEmail("betty@gbif.org");
112    u.setFirstName("Betty");
113    u.setLastName("Ford");
114    SerdeTestUtils.testSerDe(u, User.class);
115  }
116
117  @Test
118  public void testName() throws IOException {
119    User u = new User();
120    u.setKey(100);
121    u.setUserName("be");
122    u.setEmail("betty@gbif.org");
123    u.setFirstName("Betty");
124    u.setLastName("Ford");
125    assertEquals("Betty Ford", u.getName());
126  }
127
128}
