018import org.gbif.api.SerdeTestUtils;
019import org.gbif.api.vocabulary.UserRole;
020
021import java.io.IOException;
022import java.util.Map;
023import java.util.Set;
024import javax.validation.ConstraintViolation;

027
028import com.google.common.collect.Maps;
029import org.junit.Test;
030
031import static org.junit.Assert.assertEquals;

033import static org.junit.Assert.assertTrue;
034
035public class UserTest {
036  private Validator validator = Validation.buildDefaultValidatorFactory().getValidator();
037
038  @Test
039  public void testHasRole() throws Exception {
040    User u = new User();
041    u.setUserName("betty");
042    u.setEmail("betty@gbif.org");
043    u.setFirstName("Betty Ford");
044
045    assertFalse(u.hasRole(UserRole.USER));
046    assertFalse(u.hasRole(UserRole.REGISTRY_EDITOR));
047    assertFalse(u.hasRole(UserRole.ADMIN));
048    assertFalse(u.hasRole(null));
049
050    u.addRole(UserRole.USER);
051    assertTrue(u.hasRole(UserRole.USER));
052    assertFalse(u.hasRole(UserRole.REGISTRY_EDITOR));
053    assertFalse(u.hasRole(UserRole.ADMIN));
054    assertFalse(u.hasRole(null));
055
056    u.addRole(UserRole.ADMIN);
057    assertTrue(u.hasRole(UserRole.USER));
058    assertFalse(u.hasRole(UserRole.REGISTRY_EDITOR));
059    assertTrue(u.hasRole(UserRole.ADMIN));
060    assertFalse(u.hasRole(null));
061  }
062
063  @Test
064  public void testIsAdmin() throws Exception {
065    User u = new User();
066    u.setUserName("betty");
067    u.setEmail("betty@gbif.org");
068    u.setFirstName("Betty Ford");
069
070    assertFalse(u.isAdmin());
071
072    u.addRole(UserRole.ADMIN);
073    assertTrue(u.isAdmin());
074  }
075
076  @Test
077  public void testValidation() throws Exception {
078    User u = new User();
079    u.setKey(100);
080    u.setUserName("be");
081    u.setEmail("betty@gbif.org");
082    u.setFirstName("Betty");
083    u.setLastName("Ford");
084    assertValidation("userName", u);
085
086    u.setUserName("betty");
087    u.setFirstName(null);
088    assertValidation("firstName", u);
089
090    u.setFirstName("Betty");
091    assertValidation(null, u);
092
093    u.setEmail("betty at gbif.org");
094    assertValidation("email", u);
095  }
096
097  private void assertValidation(String property, User u) {
098    Set<ConstraintViolation<User>> err = validator.validate(u);
099    System.out.print(err);
100    if (property == null) {
101      assertEquals(0, err.size());
102    } else {
103      assertEquals(1, err.size());
104      assertEquals(property, err.iterator().next().getPropertyPath().toString());
105    }
106  }
