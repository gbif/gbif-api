016package org.gbif.api.model.registry;
017
018import java.net.URI;
019import java.util.HashSet;
020import java.util.Set;
021
022import javax.validation.ConstraintViolation;
023import javax.validation.Validation;
024import javax.validation.Validator;
025import javax.validation.ValidatorFactory;
026
027import org.junit.jupiter.api.Test;
028
029import static org.junit.jupiter.api.Assertions.assertFalse;
030import static org.junit.jupiter.api.Assertions.assertTrue;
031
032public class NetworkTest {
033
034  @Test
035  public void testValidations() {
036    ValidatorFactory validatorFactory = Validation.buildDefaultValidatorFactory();
037    Validator validator = validatorFactory.getValidator();
038
039    Network net = new Network();
040    net.setTitle("a"); // too short
041    net.setLogoUrl(URI.create("file:///tmp/aha")); // bad http URI
042    Set<ConstraintViolation<Network>> violations = validator.validate(net);
043    assertFalse(violations.isEmpty(), "Violations were expected");
044
045    // ensure all expected properties are caught
046    Set<String> propertiesInViolation = new HashSet<>();
047    propertiesInViolation.add("title");
048    propertiesInViolation.add("logoUrl");
049    for (ConstraintViolation<?> cv : violations) {
050      propertiesInViolation.remove(cv.getPropertyPath().toString());
051    }
052    assertTrue(propertiesInViolation.isEmpty(), "Properties incorrectly passed validation " + propertiesInViolation);
053  }
054}
