023public class DatasetTest {
024
025  @Test
026  public void testValidations() {
027    ValidatorFactory validatorFactory =
028      Validation.byProvider(ApacheValidationProvider.class).configure().buildValidatorFactory();
029    Validator validator = validatorFactory.getValidator();
030
031    Dataset ds = new Dataset();
032
033    // 3 non-mandatory fields that don't validate
034    ds.setTitle("B"); // too short
035    ds.setHomepage(URI.create("www.gbif.org")); // doesn't start with http or https
036    ds.setLogoUrl(URI.create("file:///tmp/aha")); // bad http URI
037
038    // All mandatory fields missing include:
039    // ds.setType(?);
040    // ds.setInstallationKey(?);
041    // ds.setPublishingOrganizationKey(?);
042
043    // perform validation
044    Set<ConstraintViolation<Dataset>> violations = validator.validate(ds);
045    assertTrue("Violations were expected", !violations.isEmpty());
046
047    // ensure all 6 expected violations are caught
048    Set<String> propertiesInViolation =
049      Sets.newHashSet("title", "homepage", "logoUrl", "type", "installationKey", "publishingOrganisationKey");
050    assertEquals(6, violations.size());
051    for (ConstraintViolation<?> cv : violations) {
052      propertiesInViolation.contains(cv.getPropertyPath().toString());
053    }
054
055    // fix non-mandatory fields that don't validate
056    ds.setTitle("Rooftop bugs");
057    ds.setHomepage(URI.create("http://www.gbif.org"));
058    ds.setLogoUrl(URI.create("http://www.gbif.org/logo.png"));
059
060    // add all mandatory fields that were missing
061    ds.setType(DatasetType.SAMPLING_EVENT);
062    ds.setInstallationKey(UUID.randomUUID());
063    ds.setPublishingOrganizationKey(UUID.randomUUID());
064
065    // perform validation again
066    violations = validator.validate(ds);
067    assertTrue("No violations were expected", violations.isEmpty());
068  }
069
070  @Test
071  public void testLenientEquals() {
072    Dataset ds1 = new Dataset();
073    ds1.setMaintenanceUpdateFrequency(MaintenanceUpdateFrequency.DAILY);
074    ds1.setMaintenanceDescription("Daily, except for holidays");
075    ds1.setLicense(License.CC_BY_4_0);
076    ds1.setCreated(new Date());
077
078    Dataset ds2 = new Dataset();
079    ds2.setMaintenanceUpdateFrequency(MaintenanceUpdateFrequency.DAILY);
080    ds2.setMaintenanceDescription("Daily, except for holidays");
081    ds2.setLicense(License.CC0_1_0); // different license
082    ds1.setCreated(new Date()); // different created date!
083
084    assertTrue(ds1.lenientEquals(ds2)); // true because lenient equals excludes
085  }
