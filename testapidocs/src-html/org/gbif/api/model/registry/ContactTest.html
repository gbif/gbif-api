016package org.gbif.api.model.registry;
017
018import org.junit.Test;
019
020import static org.junit.Assert.assertEquals;
021import static org.junit.Assert.assertTrue;
022
023public class ContactTest {
024
025  @Test
026  public void testUserId() throws Exception {
027    Contact c = new Contact();
028
029    c.addUserId(null, null);
030    assertTrue(c.getUserId().isEmpty());
031
032    c.addUserId(null, "3421423");
033    assertEquals("3421423", c.getUserId().get(0));
034
035    c.addUserId("", "3421423");
036    assertEquals("3421423", c.getUserId().get(1));
037
038    // Directory prefix absent from the id.
039    c.addUserId("gbif", "3421423");
040    assertEquals("gbif:3421423", c.getUserId().get(2));
041
042    c.addUserId("http://orcid.org/", "3421423");
043    assertEquals("http://orcid.org/3421423", c.getUserId().get(3));
044
045    c.addUserId("http://orcid.org", "3421423");
046    assertEquals("http://orcid.org/3421423", c.getUserId().get(4));
047
048    c.addUserId("http://orcid.org", "de/12-3421423");
049    assertEquals("http://orcid.org/de/12-3421423", c.getUserId().get(5));
050
051    c.addUserId("https://scholar.google.com/citations?user=", "jvW0IrIAAAAJ");
052    assertEquals("https://scholar.google.com/citations?user=jvW0IrIAAAAJ", c.getUserId().get(6));
053
054    c.addUserId("https://www.linkedin.com/in/", "john-smith-12345");
055    assertEquals("https://www.linkedin.com/in/john-smith-12345", c.getUserId().get(7));
056
057    c.addUserId("https://www.linkedin.com/profile/view?id=", "AAkAAABiOnwBeoX3a3wKqe4IEqDkJ_ifoVj1234");
058    assertEquals("https://www.linkedin.com/profile/view?id=AAkAAABiOnwBeoX3a3wKqe4IEqDkJ_ifoVj1234", c.getUserId().get(8));
059
060    c.addUserId(", ,", ", ,");
061    assertEquals(", ,", c.getUserId().get(9));
062
063    // Directory prefix included in the id
064    c.addUserId("gbif", "gbif:105539");
065    assertEquals("gbif:105539", c.getUserId().get(10));
066
067    c.addUserId("http://orcid.org/", "http://orcid.org/0000-0003-0623-6682");
068    assertEquals("http://orcid.org/0000-0003-0623-6682", c.getUserId().get(11));
069
070    c.addUserId("http://orcid.org", "http://orcid.org/0000-0003-0623-6682");
071    assertEquals("http://orcid.org/0000-0003-0623-6682", c.getUserId().get(12));
072
073    c.addUserId("https://orcid.org", "http://orcid.org/0000-0003-0623-6682");
074    assertEquals("http://orcid.org/0000-0003-0623-6682", c.getUserId().get(13));
075
076    c.addUserId("http://orcid.org", "https://orcid.org/0000-0003-0623-6682");
077    assertEquals("https://orcid.org/0000-0003-0623-6682", c.getUserId().get(14));
078
079    c.addUserId("http://orcid.org", "orcid.org/0000-0003-0623-6682");
080    assertEquals("http://orcid.org/0000-0003-0623-6682", c.getUserId().get(15));
081
082    c.addUserId("https://orcid.org", "orcid.org/0000-0003-0623-6682");
083    assertEquals("https://orcid.org/0000-0003-0623-6682", c.getUserId().get(16));
084
085    c.addUserId("https://scholar.google.com/citations?user=", "https://scholar.google.com/citations?user=jvW0IrIAAAAJ");
086    assertEquals("https://scholar.google.com/citations?user=jvW0IrIAAAAJ", c.getUserId().get(17));
087
088    // Just to check this test is written correctly.
089    assertEquals(18, c.getUserId().size());
090  }
091
092  @Test
093  public void testGetCompleteName() {
094    Contact c = new Contact();
095    assertTrue(c.computeCompleteName().equals(""));
096
097    c.setFirstName("FirstName ");
098    assertTrue(c.computeCompleteName().equals("FirstName"));
099
100    c.setLastName("LastName");
101    assertTrue(c.computeCompleteName().equals("FirstName LastName"));
102
103    c.setFirstName(" ");
104    assertTrue(c.computeCompleteName().equals("LastName"));
105  }
106
107}
