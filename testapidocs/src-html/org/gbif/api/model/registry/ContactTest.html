001package org.gbif.api.model.registry;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006import static org.junit.Assert.assertTrue;
007
008public class ContactTest {
009
010  @Test
011  public void testUserId() throws Exception {
012    Contact c = new Contact();
013
014    c.addUserId(null, null);
015    assertTrue(c.getUserId().isEmpty());
016
017    c.addUserId(null, "3421423");
018    assertEquals("3421423", c.getUserId().get(0));
019
020    c.addUserId("", "3421423");
021    assertEquals("3421423", c.getUserId().get(1));
022
023    c.addUserId("gbif", "3421423");
024    assertEquals("gbif:3421423", c.getUserId().get(2));
025
026    c.addUserId("http://orcid.org/", "3421423");
027    assertEquals("http://orcid.org/3421423", c.getUserId().get(3));
028
029    c.addUserId("http://orcid.org", "3421423");
030    assertEquals("http://orcid.org/3421423", c.getUserId().get(4));
031
032    c.addUserId("http://orcid.org", "de/12-3421423");
033    assertEquals("http://orcid.org/de/12-3421423", c.getUserId().get(5));
034
035    c.addUserId("https://scholar.google.com/citations?user=", "jvW0IrIAAAAJ");
036    assertEquals("https://scholar.google.com/citations?user=jvW0IrIAAAAJ", c.getUserId().get(6));
037
038    c.addUserId("https://www.linkedin.com/in/", "john-smith-12345");
039    assertEquals("https://www.linkedin.com/in/john-smith-12345", c.getUserId().get(7));
040
041    c.addUserId("https://www.linkedin.com/profile/view?id=", "AAkAAABiOnwBeoX3a3wKqe4IEqDkJ_ifoVj1234");
042    assertEquals("https://www.linkedin.com/profile/view?id=AAkAAABiOnwBeoX3a3wKqe4IEqDkJ_ifoVj1234", c.getUserId().get(8));
043
044    c.addUserId(", ,", ", ,");
045    assertEquals(", ,", c.getUserId().get(9));
046  }
047
048  @Test
049  public void testGetCompleteName() {
050    Contact c = new Contact();
051    assertTrue(c.computeCompleteName().equals(""));
052
053    c.setFirstName("FirstName ");
054    assertTrue(c.computeCompleteName().equals("FirstName"));
055
056    c.setLastName("LastName");
057    assertTrue(c.computeCompleteName().equals("FirstName LastName"));
058
059    c.setFirstName(" ");
060    assertTrue(c.computeCompleteName().equals("LastName"));
061  }
062
063}
