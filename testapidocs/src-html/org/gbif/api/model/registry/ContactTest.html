001package org.gbif.api.model.registry;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006import static org.junit.Assert.assertTrue;
007
008public class ContactTest {
009
010  @Test
011  public void testUserId() throws Exception {
012    Contact c = new Contact();
013
014    c.addUserId(null, null);
015    assertTrue(c.getUserId().isEmpty());
016
017    c.addUserId(null, "3421423");
018    assertEquals("3421423", c.getUserId().get(0));
019
020    c.addUserId("", "3421423");
021    assertEquals("3421423", c.getUserId().get(1));
022
023    // Directory prefix absent from the id.
024    c.addUserId("gbif", "3421423");
025    assertEquals("gbif:3421423", c.getUserId().get(2));
026
027    c.addUserId("http://orcid.org/", "3421423");
028    assertEquals("http://orcid.org/3421423", c.getUserId().get(3));
029
030    c.addUserId("http://orcid.org", "3421423");
031    assertEquals("http://orcid.org/3421423", c.getUserId().get(4));
032
033    c.addUserId("http://orcid.org", "de/12-3421423");
034    assertEquals("http://orcid.org/de/12-3421423", c.getUserId().get(5));
035
036    c.addUserId("https://scholar.google.com/citations?user=", "jvW0IrIAAAAJ");
037    assertEquals("https://scholar.google.com/citations?user=jvW0IrIAAAAJ", c.getUserId().get(6));
038
039    c.addUserId("https://www.linkedin.com/in/", "john-smith-12345");
040    assertEquals("https://www.linkedin.com/in/john-smith-12345", c.getUserId().get(7));
041
042    c.addUserId("https://www.linkedin.com/profile/view?id=", "AAkAAABiOnwBeoX3a3wKqe4IEqDkJ_ifoVj1234");
043    assertEquals("https://www.linkedin.com/profile/view?id=AAkAAABiOnwBeoX3a3wKqe4IEqDkJ_ifoVj1234", c.getUserId().get(8));
044
045    c.addUserId(", ,", ", ,");
046    assertEquals(", ,", c.getUserId().get(9));
047
048    // Directory prefix included in the id
049    c.addUserId("gbif", "gbif:105539");
050    assertEquals("gbif:105539", c.getUserId().get(10));
051
052    c.addUserId("http://orcid.org/", "http://orcid.org/0000-0003-0623-6682");
053    assertEquals("http://orcid.org/0000-0003-0623-6682", c.getUserId().get(11));
054
055    c.addUserId("http://orcid.org", "http://orcid.org/0000-0003-0623-6682");
056    assertEquals("http://orcid.org/0000-0003-0623-6682", c.getUserId().get(12));
057
058    c.addUserId("https://orcid.org", "http://orcid.org/0000-0003-0623-6682");
059    assertEquals("http://orcid.org/0000-0003-0623-6682", c.getUserId().get(13));
060
061    c.addUserId("http://orcid.org", "https://orcid.org/0000-0003-0623-6682");
062    assertEquals("https://orcid.org/0000-0003-0623-6682", c.getUserId().get(14));
063
064    c.addUserId("http://orcid.org", "orcid.org/0000-0003-0623-6682");
065    assertEquals("http://orcid.org/0000-0003-0623-6682", c.getUserId().get(15));
066
067    c.addUserId("https://orcid.org", "orcid.org/0000-0003-0623-6682");
068    assertEquals("https://orcid.org/0000-0003-0623-6682", c.getUserId().get(16));
069
070    c.addUserId("https://scholar.google.com/citations?user=", "https://scholar.google.com/citations?user=jvW0IrIAAAAJ");
071    assertEquals("https://scholar.google.com/citations?user=jvW0IrIAAAAJ", c.getUserId().get(17));
072
073    // Just to check this test is written correctly.
074    assertEquals(18, c.getUserId().size());
075  }
076
077  @Test
078  public void testGetCompleteName() {
079    Contact c = new Contact();
080    assertTrue(c.computeCompleteName().equals(""));
081
082    c.setFirstName("FirstName ");
083    assertTrue(c.computeCompleteName().equals("FirstName"));
084
085    c.setLastName("LastName");
086    assertTrue(c.computeCompleteName().equals("FirstName LastName"));
087
088    c.setFirstName(" ");
089    assertTrue(c.computeCompleteName().equals("LastName"));
090  }
091
092}
