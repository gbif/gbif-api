016package org.gbif.api.model.registry;
017
018import org.gbif.api.vocabulary.Language;
019
020import java.net.URI;
021import java.util.Collections;
022import java.util.HashSet;
023import java.util.Set;
024import java.util.UUID;
025
026import javax.validation.ConstraintViolation;
027import javax.validation.Validation;
028import javax.validation.Validator;
029import javax.validation.ValidatorFactory;
030
031import org.junit.Test;
032
033import static org.junit.Assert.assertTrue;
034
035public class OrganizationTest {
036
037  @Test
038  public void testValidations() {
039    ValidatorFactory validatorFactory = Validation.buildDefaultValidatorFactory();
040    Validator validator = validatorFactory.getValidator();
041
042    Organization org = new Organization();
043    org.setTitle("a");  // too short
044    org.setHomepage(Collections.singletonList(URI.create("www.gbif.org"))); // doesn't start with http or https
045    org.setLogoUrl(URI.create("file:///tmp/aha")); // bad http URI
046
047    // perform validation
048    Set<ConstraintViolation<Organization>> violations = validator.validate(org);
049    assertTrue("Violations were expected", !violations.isEmpty());
050
051    // ensure all expected properties are caught
052    Set<String> propertiesInViolation = new HashSet<>();
053    propertiesInViolation.add("title");
054    propertiesInViolation.add("logoUrl");
055
056    for (ConstraintViolation<?> cv : violations) {
057      propertiesInViolation.remove(cv.getPropertyPath().toString());
058    }
059    assertTrue("Properties incorrectly passed validation " + propertiesInViolation, propertiesInViolation.isEmpty());
060
061    // fix validation problems
062    org.setTitle("Academy of Natural Sciences");
063    org.setHomepage(Collections.singletonList(URI.create("http://www.gbif.org")));
064    org.setLogoUrl(URI.create("http://www.gbif.org/logo.png"));
065    org.setLanguage(Language.ENGLISH);
066    org.setEndorsingNodeKey(UUID.randomUUID());
067
068    // perform validation again
069    violations = validator.validate(org);
070    assertTrue("No violations were expected", violations.isEmpty());
071  }
072}
