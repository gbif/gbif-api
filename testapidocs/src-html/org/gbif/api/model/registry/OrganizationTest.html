001package org.gbif.api.model.registry;
002
003import org.gbif.api.vocabulary.Language;
004
005import java.net.URI;
006import java.util.Set;
007import java.util.UUID;
008import javax.validation.ConstraintViolation;
009import javax.validation.Validation;
010import javax.validation.Validator;
011import javax.validation.ValidatorFactory;
012
013import com.google.common.collect.Lists;
014import com.google.common.collect.Sets;
015import org.apache.bval.jsr303.ApacheValidationProvider;
016import org.junit.Test;
017
018import static org.junit.Assert.assertTrue;
019
020public class OrganizationTest {
021
022  @Test
023  public void testValidations() {
024    ValidatorFactory validatorFactory =
025      Validation.byProvider(ApacheValidationProvider.class).configure().buildValidatorFactory();
026    Validator validator = validatorFactory.getValidator();
027
028    Organization org = new Organization();
029    org.setTitle("a");  // too short
030    org.setHomepage(Lists.newArrayList(URI.create("www.gbif.org"))); // doesn't start with http or https
031    org.setLogoUrl(URI.create("file:///tmp/aha")); // bad http URI
032
033    // perform validation
034    Set<ConstraintViolation<Organization>> violations = validator.validate(org);
035    assertTrue("Violations were expected", !violations.isEmpty());
036
037    // ensure all expected properties are caught
038    Set<String> propertiesInViolation = Sets.newHashSet("title", "logoUrl");
039    for (ConstraintViolation<?> cv : violations) {
040      propertiesInViolation.remove(cv.getPropertyPath().toString());
041    }
042    assertTrue("Properties incorrectly passed validation " + propertiesInViolation, propertiesInViolation.isEmpty());
043
044    // fix validation problems
045    org.setTitle("Academy of Natural Sciences");
046    org.setHomepage(Lists.newArrayList(URI.create("http://www.gbif.org")));
047    org.setLogoUrl(URI.create("http://www.gbif.org/logo.png"));
048    org.setLanguage(Language.ENGLISH);
049    org.setEndorsingNodeKey(UUID.randomUUID());
050
051    // perform validation again
052    violations = validator.validate(org);
053    assertTrue("No violations were expected", violations.isEmpty());
054  }
055}
