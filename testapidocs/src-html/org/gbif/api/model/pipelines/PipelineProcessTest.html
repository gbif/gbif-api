001package org.gbif.api.model.pipelines;
002
003import java.time.LocalDateTime;
004import java.util.Objects;
005
006import org.junit.Test;
007
008import static org.junit.Assert.assertEquals;
009
010/** Tests the {@link PipelineProcess}. */
011public class PipelineProcessTest {
012
013  @Test
014  public void pipelineProcessComparatorTest() {
015    PipelineProcess p1 = new PipelineProcess();
016    PipelineProcess p2 = new PipelineProcess();
017
018    // process without steps
019    PipelineExecution e1 = new PipelineExecution().setCreated(LocalDateTime.now());
020    p1.addExecution(e1);
021    PipelineExecution e2 = new PipelineExecution().setCreated(LocalDateTime.now());
022    p2.addExecution(e2);
023    assertEquals(
024        0, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
025
026    // one process without steps
027    PipelineStep p1e1s1 =
028        new PipelineStep()
029            .setStarted(LocalDateTime.now().minusMinutes(30))
030            .setState(PipelineStep.Status.RUNNING);
031    e1.addStep(p1e1s1);
032    assertEquals(
033        -1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
034
035    // both processes with steps
036    PipelineStep p2e2s1 =
037        new PipelineStep()
038            .setStarted(LocalDateTime.now().minusMinutes(35))
039            .setState(PipelineStep.Status.RUNNING);
040    e2.addStep(p2e2s1);
041    assertEquals(
042        1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
043
044    // completed steps have less preference over running ones
045    PipelineStep p1e1s2 =
046        new PipelineStep()
047            .setStarted(LocalDateTime.now().minusMinutes(40))
048            .setState(PipelineStep.Status.COMPLETED);
049    e1.addStep(p1e1s2);
050    assertEquals(
051        1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
052
053    // steps with null values
054    PipelineStep p1e1s3 = new PipelineStep();
055    e1.addStep(p1e1s3);
056    assertEquals(
057        1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
058
059    // steps with start date but no state
060    PipelineStep p1e1s4 = new PipelineStep().setStarted(LocalDateTime.now().minusMinutes(15));
061    e1.addStep(p1e1s4);
062    assertEquals(
063        1, Objects.compare(p1, p2, PipelineProcess.PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC));
064  }
065}
