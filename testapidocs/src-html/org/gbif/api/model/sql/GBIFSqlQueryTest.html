001package org.gbif.api.model.sql;
002
003import org.gbif.api.model.occurrence.sql.GBIFSqlQuery;
004import org.gbif.api.model.occurrence.sql.SqlDownloadExportFormat;
005
006import java.util.Arrays;
007import java.util.Collection;
008
009import javax.validation.ValidationException;
010
011import org.junit.Assert;
012import org.junit.Test;
013import org.junit.runner.RunWith;
014import org.junit.runners.Parameterized;
015
016/**
017 * Testing GBIFSqlQuery to verify parsing of SQL and formats expected.
018 */
019@RunWith(Parameterized.class)
020public class GBIFSqlQueryTest {
021
022  public static final String VALIDATION_ERROR_MESSAGE =
023    "Failure reading export format please specify TSV or AVRO";
024
025  @Parameterized.Parameters
026  public static Collection<Object[]> inputs() {
027    return Arrays.asList(new Object[][] {
028      {"SELECT * from occ WHERE X > 23 EXPORT AS CSV", "", null},
029      {"SELECT * from occ WHERE X > 23 ", "SELECT * from occ WHERE X > 23", SqlDownloadExportFormat.TSV},
030      {"SELECT * from occ WHERE X > 23 EXPORT AS TSV", "SELECT * from occ WHERE X > 23", SqlDownloadExportFormat.TSV },
031      {"SELECT * from occ WHERE X > 23 EXPORT   AS    TSV", "SELECT * from occ WHERE X > 23", SqlDownloadExportFormat.TSV },
032      {"SELECT * from occ WHERE X > 23 EXPORT   AS    tsv", "SELECT * from occ WHERE X > 23", SqlDownloadExportFormat.TSV },
033      {"SELECT * from occ WHERE X > 23 EXPORT AS AVRO", "SELECT * from occ WHERE X > 23", SqlDownloadExportFormat.AVRO },
034      {"SELECT * from occ WHERE X > 23 export as AVRO", "SELECT * from occ WHERE X > 23", SqlDownloadExportFormat.AVRO },
035      {"SELECT * from occ WHERE X > 23 EXPORT   as    AVRO", "SELECT * from occ WHERE X > 23", SqlDownloadExportFormat.AVRO },
036      {"SELECT * from occ WHERE X > 23 EXPORT   AS    Avro", "SELECT * from occ WHERE X > 23", SqlDownloadExportFormat.AVRO }
037    });
038  }
039
040  private String gbifSql;
041  private String uncheckedHiveSQL;
042  private SqlDownloadExportFormat format;
043
044  public GBIFSqlQueryTest(String gbifSql, String uncheckedHiveSQL, SqlDownloadExportFormat format){
045    this.gbifSql = gbifSql;
046    this.uncheckedHiveSQL = uncheckedHiveSQL;
047    this.format = format;
048  }
049
050  @Test
051  public void testExport(){
052    GBIFSqlQuery query = null;
053    try {
054      query = GBIFSqlQuery.create(gbifSql);
055    }
056    catch(Exception exception){
057      Assert.assertEquals(format, query);
058      Assert.assertTrue(exception instanceof ValidationException);
059      Assert.assertEquals(VALIDATION_ERROR_MESSAGE, exception.getMessage());
060      return;
061    }
062    System.out.println(query.getUncheckedHiveQuery());
063    System.out.println(query.getSqlDownloadExportFormat());
064    System.out.println(query.getSqlDownloadExportFormat().getTemplate());
065
066    Assert.assertEquals(uncheckedHiveSQL, query.getUncheckedHiveQuery());
067    Assert.assertEquals(format, query.getSqlDownloadExportFormat());
068  }
069}
