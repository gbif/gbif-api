095  @Test
096  public void testHigherClassificationMap() {
097    NameUsage nu = new NameUsage();
098    nu.setKey(6);
099    nu.setScientificName("Vicia faba L.");
100    nu.setParentKey(10);
101    nu.setParent("Vicia subgen. Vicia");
102    nu.setRank(Rank.SPECIES);
103    nu.setKingdomKey(1);
104    nu.setKingdom("Plantae");
105    nu.setGenus("Vicia");
106    nu.setGenusKey(3);
107    nu.setOrder("Fabales");
108    nu.setOrderKey(4);
109    nu.setSpecies("Vicia faba");
110    nu.setSpeciesKey(6);
111    nu.setPhylum("Magnoliophyta");
112    nu.setPhylumKey(5);
113
114    assertEquals(5, nu.getHigherClassificationMap().size());
115    assertTrue(nu.getHigherClassificationMap().containsKey(10));
116    assertFalse(nu.getHigherClassificationMap().containsKey(6));
117
118    Queue<Integer> expectedKeyOrder = new LinkedList<Integer>();
119    expectedKeyOrder.add(1);
120    expectedKeyOrder.add(5);
121    expectedKeyOrder.add(4);
122    expectedKeyOrder.add(3);
123    expectedKeyOrder.add(10);
124
125    for (Object key : nu.getHigherClassificationMap().keySet()) {
126      assertEquals(expectedKeyOrder.poll(), key);
127    }
128    assertTrue(expectedKeyOrder.isEmpty());
129  }

174  @Test
175  public void testTaxonomicStatus() {
176    NameUsage nu = new NameUsage();
177    nu.setKey(6);
178    nu.setScientificName("Vicia faba L.");
179    nu.setTaxonomicStatus(null);
180    assertNull(nu.getTaxonomicStatus());
181    assertFalse(nu.isSynonym());
182
183    nu.setTaxonomicStatus(TaxonomicStatus.PROPARTE_SYNONYM);
184    assertEquals(TaxonomicStatus.PROPARTE_SYNONYM, nu.getTaxonomicStatus());
185    assertTrue(nu.isSynonym());
186
187    nu.setTaxonomicStatus(TaxonomicStatus.SYNONYM);
188    assertEquals(TaxonomicStatus.SYNONYM, nu.getTaxonomicStatus());
189    assertTrue(nu.isSynonym());
190
191    nu.setTaxonomicStatus(TaxonomicStatus.DOUBTFUL);
192    assertEquals(TaxonomicStatus.DOUBTFUL, nu.getTaxonomicStatus());
193    assertFalse(nu.isSynonym());
194
195    nu.setProParteKey(432);
196    assertTrue(nu.isProParte());
197
198    nu.setTaxonomicStatus(TaxonomicStatus.HETEROTYPIC_SYNONYM);
199    assertEquals(TaxonomicStatus.HETEROTYPIC_SYNONYM, nu.getTaxonomicStatus());
200    assertTrue(nu.isSynonym());
201  }
