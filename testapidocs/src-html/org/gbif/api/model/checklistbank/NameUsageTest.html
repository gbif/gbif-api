108  @Test
109  public void testHigherClassificationMap() {
110    NameUsage nu = new NameUsage();
111    nu.setKey(6);
112    nu.setScientificName("Vicia faba L.");
113    nu.setParentKey(10);
114    nu.setParent("Vicia subgen. Vicia");
115    nu.setRank(Rank.SPECIES);
116    nu.setKingdomKey(1);
117    nu.setKingdom("Plantae");
118    nu.setGenus("Vicia");
119    nu.setGenusKey(3);
120    nu.setOrder("Fabales");
121    nu.setOrderKey(4);
122    nu.setSpecies("Vicia faba");
123    nu.setSpeciesKey(6);
124    nu.setPhylum("Magnoliophyta");
125    nu.setPhylumKey(5);
126
127    assertEquals(5, nu.getHigherClassificationMap().size());
128    assertTrue(nu.getHigherClassificationMap().containsKey(10));
129    assertFalse(nu.getHigherClassificationMap().containsKey(6));
130
131    Queue<Integer> expectedKeyOrder = new LinkedList<Integer>();
132    expectedKeyOrder.add(1);
133    expectedKeyOrder.add(5);
134    expectedKeyOrder.add(4);
135    expectedKeyOrder.add(3);
136    expectedKeyOrder.add(10);
137
138    for (Object key : nu.getHigherClassificationMap().keySet()) {
139      assertEquals(expectedKeyOrder.poll(), key);
140    }
141    assertTrue(expectedKeyOrder.isEmpty());
142  }

178  @Test
179  public void testSetLink() {
180    Identifier link = new Identifier();
181    link.setIdentifier("http://www.example.org");
182    link.setType(IdentifierType.URL);
183
184    Identifier id = new Identifier();
185    id.setIdentifier("12345");
186    id.setType(IdentifierType.SOURCE_ID);
187
188    NameUsage nu = new NameUsage();
189    nu.getIdentifiers().add(id);
190    assertNull(nu.getLink());
191    assertEquals("12345", nu.getSourceId());
192
193    nu.getIdentifiers().add(link);
194    assertEquals("http://www.example.org", nu.getLink());
195
196    nu.getIdentifiers().clear();
197    assertNull(nu.getLink());
198    assertNull(nu.getSourceId());
199    assertEquals(0, nu.getIdentifiers().size());
200
201
202    // sourceid
203    nu.setSourceId("99");
204    assertEquals(1, nu.getIdentifiers().size());
205    assertEquals("99", nu.getSourceId());
206    assertEquals("99", nu.getIdentifiers().get(0).getIdentifier());
207
208    nu.setSourceId("991");
209    assertEquals(1, nu.getIdentifiers().size());
210    assertEquals("991", nu.getSourceId());
211    assertEquals("991", nu.getIdentifiers().get(0).getIdentifier());
212
213
214    // references
215    assertEquals(1, nu.getIdentifiers().size());
216    nu.setReferences(URI.create("http://www.example.org"));
217    assertEquals(1, nu.getIdentifiers().size());
218    assertEquals("http://www.example.org", nu.getLink());
219    assertEquals(nu.getLink(), nu.getReferences().toString());
220
221    nu.setReferences(URI.create("http://www.example.org/2"));
222    assertEquals(1, nu.getIdentifiers().size());
223    assertEquals("http://www.example.org/2", nu.getLink());
224    assertEquals(nu.getLink(), nu.getReferences().toString());
225  }

246  @Test
247  public void testJsonSerde() throws IOException {
248    final ObjectMapper mapper = new ObjectMapper();
249    mapper.enable(DeserializationConfig.Feature.ACCEPT_EMPTY_STRING_AS_NULL_OBJECT);
250    mapper.enable(SerializationConfig.Feature.INDENT_OUTPUT);
251    mapper.disable(SerializationConfig.Feature.WRITE_NULL_PROPERTIES);
252
253    NameUsage u = new NameUsage();
254    u.setKey(123);
255    u.setDatasetKey(UUID.randomUUID());
256    u.setNumDescendants(321);
257    u.getNomenclaturalStatus().add(NomenclaturalStatus.AMBIGUOUS);
258    Identifier i = new Identifier();
259    i.setIdentifier("me");
260    u.addIdentifier(i);
261    u.setKingdomKey(6);
262    u.setKingdom("Plants");
263
264    String json = mapper.writeValueAsString(u);
265    System.out.println(json);
266    assertEquals(u, mapper.readValue(json, NameUsage.class));
267  }
