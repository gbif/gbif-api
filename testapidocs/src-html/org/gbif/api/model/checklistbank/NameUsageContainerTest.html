044  @Test
045  public void testConstructor() {
046    final UUID checklistKey = UUID.randomUUID();
047    NameUsage src = new NameUsage();
048    src.setKey(1234);
049    src.setDatasetKey(checklistKey);
050    src.setKingdom("Animalia");
051    src.setScientificName("Puma concolor Linné");
052    src.setReferences(URI.create("http://wisdom.org/1234"));
053    src.setSourceTaxonKey(12345);
054    src.getIssues().add(NameUsageIssue.ACCEPTED_NAME_MISSING);
055    src.getIssues().add(NameUsageIssue.BACKBONE_MATCH_NONE);
056    src.getIssues().add(NameUsageIssue.BASIONYM_AUTHOR_MISMATCH);
057
058    NameUsageContainer nu = new NameUsageContainer(src);
059    Identifier id1 = new Identifier();
060    id1.setIdentifier("1234");
061    id1.setType(IdentifierType.GBIF_PORTAL);
062    nu.getIdentifiers().add(id1);
063
064    assertEquals(checklistKey, nu.getDatasetKey());
065    assertEquals((Integer) 1234, nu.getKey());
066    assertEquals((Integer) 12345, nu.getSourceTaxonKey());
067    assertEquals("http://wisdom.org/1234", nu.getReferences().toString());
068    assertEquals("Animalia", nu.getKingdom());
069    assertEquals("Puma concolor Linné", nu.getScientificName());
070    assertEquals(1, nu.getIdentifiers().size());
071
072    assertEquals("1234", nu.getIdentifiers().get(0).getIdentifier());
073    assertEquals(IdentifierType.GBIF_PORTAL, nu.getIdentifiers().get(0).getType());
074
075    assertNull(nu.getAccepted());
076    assertNull(nu.getAcceptedKey());
077    assertNull(nu.getAccordingTo());
078    assertNull(nu.getNameType());
079    assertNull(nu.getKingdomKey());
080
081    assertEquals(0, nu.getMedia().size());
082    assertEquals(0, nu.getDescriptions().size());
083    assertEquals(0, nu.getDistributions().size());
084    assertEquals(0, nu.getReferenceList().size());
085    assertEquals(3, nu.getIssues().size());
086  }

088  @Test
089  public void testEquals() {
090    Identifier id1 = new Identifier();
091    id1.setIdentifier("http://www.example.org");
092    id1.setType(IdentifierType.URL);
093
094    NameUsageContainer nu1 = new NameUsageContainer();
095    nu1.setKey(123);
096    nu1.getIdentifiers().add(id1);
097    nu1.setKingdom("Animalia");
098
099    NameUsageContainer nu2 = new NameUsageContainer();
100    nu2.setKey(123);
101    nu2.getIdentifiers().add(id1);
102    nu2.setKingdom("Animalia");
103
104    NameUsageMediaObject ru1 = new NameUsageMediaObject();
105    ru1.setTitle("Puma concolor");
106    ru1.setSourceTaxonKey(111);
107    NameUsageMediaObject ru2 = new NameUsageMediaObject();
108    ru2.setTitle("Puma concolor");
109    ru2.setSourceTaxonKey(111);
110
111    List<NameUsageMediaObject> lru1 = Lists.newArrayList();
112    lru1.add(ru1);
113    List<NameUsageMediaObject> lru2 = Lists.newArrayList();
114    lru2.add(ru2);
115
116    nu1.setMedia(lru1);
117    nu2.setMedia(lru2);
118
119    assertEquals(nu1, nu2);
120
121    nu2.setKey(124);
122
123    assertFalse(nu1.equals(nu2));
124  }

159    NameUsageContainer nu = new NameUsageContainer();

163    SpeciesProfile sp = new SpeciesProfile();
164    sp.setExtinct(true);

169    sp.setExtinct(null);

171
172    nu.getSpeciesProfiles().clear();
173
174    for (int i = 1; i < 10; i++) {

176      sp.setExtinct(i % 2 == 0);
177      nu.getSpeciesProfiles().add(sp);

179

181    assertFalse(nu.isExtinct());
182
183    sp = new SpeciesProfile();
184    sp.setExtinct(true);

187    assertTrue(nu.isExtinct());

210  @Test
211  public void testHigherClassificationMap() {
212    NameUsageContainer nu = new NameUsageContainer();
213    nu.setKey(6);
214    nu.setScientificName("Vicia faba L.");
215    nu.setParentKey(10);
216    nu.setParent("Vicia subgen. Vicia");
217    nu.setRank(Rank.SPECIES);
218    nu.setKingdomKey(1);
219    nu.setKingdom("Plantae");
220    nu.setGenus("Vicia");
221    nu.setGenusKey(3);
222    nu.setOrder("Fabales");
223    nu.setOrderKey(4);
224    nu.setSpecies("Vicia faba");
225    nu.setSpeciesKey(6);
226    nu.setPhylum("Magnoliophyta");
227    nu.setPhylumKey(5);
228
229    assertEquals(5, nu.getHigherClassificationMap().size());
230    assertTrue(nu.getHigherClassificationMap().containsKey(10));
231    assertFalse(nu.getHigherClassificationMap().containsKey(6));
232
233    Queue<Integer> expectedKeyOrder = new LinkedList<Integer>();
234    expectedKeyOrder.add(1);
235    expectedKeyOrder.add(5);
236    expectedKeyOrder.add(4);
237    expectedKeyOrder.add(3);
238    expectedKeyOrder.add(10);
239
240    for (Object key : nu.getHigherClassificationMap().keySet()) {
241      assertEquals(expectedKeyOrder.poll(), key);
242    }
243    assertTrue(expectedKeyOrder.isEmpty());
244  }

398  @Test
399  public void testThreats() {
400    NameUsageContainer nu = new NameUsageContainer();
401    nu.setKey(6);
402
403    assertEquals(0, nu.getDistributions().size());
404    assertEquals(0, nu.getThreatStatus().size());
405
406    Distribution d = new Distribution();
407    d.setThreatStatus(ThreatStatus.ENDANGERED);
408    nu.getDistributions().add(d);
409    assertEquals(1, nu.getDistributions().size());
410    assertEquals(1, nu.getThreatStatus().size());
411    assertEquals(ThreatStatus.ENDANGERED, nu.getThreatStatus().iterator().next());
412
413    d = new Distribution();
414    d.setThreatStatus(ThreatStatus.ENDANGERED);
415    nu.getDistributions().add(d);
416    assertEquals(2, nu.getDistributions().size());
417    assertEquals(1, nu.getThreatStatus().size());
418    assertEquals(ThreatStatus.ENDANGERED, nu.getThreatStatus().iterator().next());
419
420    d = new Distribution();
421    d.setThreatStatus(ThreatStatus.CRITICALLY_ENDANGERED);
422    nu.getDistributions().add(d);
423    assertEquals(3, nu.getDistributions().size());
424    assertEquals(2, nu.getThreatStatus().size());
425    assertEquals(ThreatStatus.ENDANGERED, nu.getThreatStatus().iterator().next());
426  }
