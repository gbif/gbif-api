044  @Test
045  public void testConstructor() {
046    final UUID checklistKey = UUID.randomUUID();
047    NameUsage src = new NameUsage();
048    src.setKey(1234);
049    src.setDatasetKey(checklistKey);
050    src.setKingdom("Animalia");
051    src.setScientificName("Puma concolor Linné");
052
053    Identifier id1 = new Identifier();
054    id1.setIdentifier("1234");
055    id1.setType(IdentifierType.GBIF_PORTAL);
056    id1.setUsageKey(src.getKey());
057    src.addIdentifier(id1);
058
059    src.setReferences(URI.create("http://wisdom.org/1234"));
060    src.setSourceId("12345");
061
062    NameUsageContainer nu = new NameUsageContainer(src);
063
064    assertEquals(checklistKey, nu.getDatasetKey());
065    assertEquals((Integer) 1234, nu.getKey());
066    assertEquals((Integer) 12345, nu.getSourceKey());
067    assertEquals("http://wisdom.org/1234", nu.getReferences().toString());
068    assertEquals("Animalia", nu.getKingdom());
069    assertEquals("Puma concolor Linné", nu.getScientificName());
070    assertEquals(1, nu.getIdentifiers().size());
071
072    assertEquals("1234", nu.getIdentifiers().get(0).getIdentifier());
073    assertEquals(IdentifierType.GBIF_PORTAL, nu.getIdentifiers().get(0).getType());
074    assertEquals((Integer) 1234, nu.getIdentifiers().get(0).getUsageKey());
075
076    assertNull(nu.getAccepted());
077    assertNull(nu.getAcceptedKey());
078    assertNull(nu.getAccordingTo());
079    assertNull(nu.getNameType());
080    assertNull(nu.getKingdomKey());
081
082    assertEquals(0, nu.getImages().size());
083    assertEquals(0, nu.getDescriptions().size());
084    assertEquals(0, nu.getDistributions().size());
085    assertEquals(0, nu.getReferenceList().size());
086  }

088  @Test
089  public void testEquals() {
090    Identifier id1 = new Identifier();

093
094    NameUsageContainer nu1 = new NameUsageContainer();
095    nu1.setKey(123);
096    nu1.getIdentifiers().add(id1);
097    nu1.setKingdom("Animalia");
098
099    NameUsageContainer nu2 = new NameUsageContainer();
100    nu2.setKey(123);
101    nu2.getIdentifiers().add(id1);
102    nu2.setKingdom("Animalia");
103
104    Image ru1 = new Image();
105    ru1.setDatasetKey(Constants.NUB_DATASET_KEY);
106    ru1.setTitle("Puma concolor");
107    ru1.setUsageKey(111);
108    Image ru2 = new Image();
109    ru2.setDatasetKey(Constants.NUB_DATASET_KEY);
110    ru2.setTitle("Puma concolor");
111    ru2.setUsageKey(111);
112
113    List<Image> lru1 = new ArrayList<Image>();
114    lru1.add(ru1);
115    List<Image> lru2 = new ArrayList<Image>();
116    lru2.add(ru2);
117
118    nu1.setImages(lru1);
119    nu2.setImages(lru2);
120
121    assertEquals(nu1, nu2);
122
123    nu2.setKey(124);
124
125    assertFalse(nu1.equals(nu2));
126  }

159  @Test
160  public void testExtinct() {
161    NameUsageContainer nu = new NameUsageContainer();
162    nu.setKey(6);
163    assertNull(nu.isExtinct());
164
165    SpeciesProfile sp = new SpeciesProfile();
166    sp.setExtinct(true);
167    nu.getSpeciesProfiles().add(sp);
168    assertTrue(nu.isExtinct());
169    sp.setExtinct(false);
170    assertFalse(nu.isExtinct());
171    sp.setExtinct(null);
172    assertNull(nu.isExtinct());
173
174    nu.getSpeciesProfiles().clear();
175
176    for (int i = 1; i < 10; i++) {
177      sp = new SpeciesProfile();
178      sp.setExtinct(i % 2 == 0);
179      nu.getSpeciesProfiles().add(sp);
180    }
181
182    assertEquals(9, nu.getSpeciesProfiles().size());
183    assertFalse(nu.isExtinct());
184
185    sp = new SpeciesProfile();
186    sp.setExtinct(true);
187    nu.getSpeciesProfiles().add(sp);
188    assertEquals(10, nu.getSpeciesProfiles().size());
189    assertTrue(nu.isExtinct());
190  }

212  @Test
213  public void testHigherClassificationMap() {
214    NameUsageContainer nu = new NameUsageContainer();
215    nu.setKey(6);
216    nu.setScientificName("Vicia faba L.");
217    nu.setParentKey(10);
218    nu.setParent("Vicia subgen. Vicia");
219    nu.setRank(Rank.SPECIES);
220    nu.setKingdomKey(1);
221    nu.setKingdom("Plantae");
222    nu.setGenus("Vicia");
223    nu.setGenusKey(3);
224    nu.setOrder("Fabales");
225    nu.setOrderKey(4);
226    nu.setSpecies("Vicia faba");
227    nu.setSpeciesKey(6);
228    nu.setPhylum("Magnoliophyta");
229    nu.setPhylumKey(5);
230
231    assertEquals(5, nu.getHigherClassificationMap().size());
232    assertTrue(nu.getHigherClassificationMap().containsKey(10));
233    assertFalse(nu.getHigherClassificationMap().containsKey(6));
234
235    Queue<Integer> expectedKeyOrder = new LinkedList<Integer>();
236    expectedKeyOrder.add(1);
237    expectedKeyOrder.add(5);
238    expectedKeyOrder.add(4);
239    expectedKeyOrder.add(3);
240    expectedKeyOrder.add(10);
241
242    for (Object key : nu.getHigherClassificationMap().keySet()) {
243      assertEquals(expectedKeyOrder.poll(), key);
244    }
245    assertTrue(expectedKeyOrder.isEmpty());
246  }

367  @Test
368  public void testThreats() {
369    NameUsageContainer nu = new NameUsageContainer();
370    nu.setKey(6);
371
372    assertEquals(0, nu.getDistributions().size());
373    assertEquals(0, nu.getThreatStatus().size());
374
375    Distribution d = new Distribution();
376    d.setThreatStatus(ThreatStatus.ENDANGERED);
377    nu.getDistributions().add(d);
378    assertEquals(1, nu.getDistributions().size());
379    assertEquals(1, nu.getThreatStatus().size());
380    assertEquals(ThreatStatus.ENDANGERED, nu.getThreatStatus().iterator().next());
381
382    d = new Distribution();
383    d.setThreatStatus(ThreatStatus.ENDANGERED);
384    nu.getDistributions().add(d);
385    assertEquals(2, nu.getDistributions().size());
386    assertEquals(1, nu.getThreatStatus().size());
387    assertEquals(ThreatStatus.ENDANGERED, nu.getThreatStatus().iterator().next());
388
389    d = new Distribution();
390    d.setThreatStatus(ThreatStatus.CRITICALLY_ENDANGERED);
391    nu.getDistributions().add(d);
392    assertEquals(3, nu.getDistributions().size());
393    assertEquals(2, nu.getThreatStatus().size());
394    assertEquals(ThreatStatus.ENDANGERED, nu.getThreatStatus().iterator().next());
395  }
