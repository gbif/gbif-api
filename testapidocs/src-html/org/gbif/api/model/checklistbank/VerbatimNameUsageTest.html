047

050    assertTrue(v1.hasExtension(DwcTerm.Identification));
051    assertFalse(v1.hasExtension(DwcTerm.MeasurementOrFact));

176  @Test
177  public void testVerbatimMapSerde() throws Exception {
178    ObjectMapper mapper = new ObjectMapper();
179    mapper.enable(DeserializationConfig.Feature.ACCEPT_EMPTY_STRING_AS_NULL_OBJECT);
180    mapper.enable(SerializationConfig.Feature.INDENT_OUTPUT);
181    mapper.disable(SerializationConfig.Feature.WRITE_NULL_PROPERTIES);
182
183    VerbatimNameUsage v = new VerbatimNameUsage();
184    v.setKey(7);
185    v.setLastCrawled(new Date());
186
187
188    for (Term term : DwcTerm.values()) {
189      v.setCoreField(term, RandomStringUtils.randomAlphabetic(20));
190    }
191    final int numDwcTerms = v.getFields().size();
192
193    for (Term term : DcTerm.values()) {
194      v.setCoreField(term, RandomStringUtils.randomAlphabetic(20));
195    }
196    final int numDcTerms = v.getFields().size() - numDwcTerms;
197
198    for (Term term : GbifTerm.values()) {
199      v.setCoreField(term, RandomStringUtils.randomAlphabetic(20));
200    }
201    final int numGbifTerms = v.getFields().size() - numDwcTerms - numDcTerms;
202
203    for (Term term : IucnTerm.values()) {
204      v.setCoreField(term, RandomStringUtils.randomAlphanumeric(20));
205    }
206    final int numIucnTerms = v.getFields().size() - numDwcTerms - numDcTerms - numGbifTerms;
207
208    v.setCoreField(DwcTerm.scientificName, "Abies alba");
209    v.setCoreField(DwcTerm.collectionCode, "BUGS");
210    v.setCoreField(DwcTerm.catalogNumber, "MD10782");
211    v.setCoreField(UnknownTerm.build("http://rs.un.org/terms/temperatur"), RandomStringUtils.randomAlphabetic(30));
212    v.setCoreField(UnknownTerm.build("http://rs.un.org/terms/co2"), RandomStringUtils.randomAlphabetic(30));
213    v.setCoreField(UnknownTerm.build("http://rs.un.org/terms/modified"), new Date().toString());
214    v.setCoreField(UnknownTerm.build("http://rs.un.org/terms/scientificName"), RandomStringUtils.randomAlphabetic(30));
215    final int numOtherTerms = v.getFields().size() - numDwcTerms - numDcTerms - numGbifTerms - numIucnTerms;
216
217    final int numTerms = v.getFields().size();
218
219    assertEquals(numTerms, numDwcTerms + numDcTerms + numGbifTerms + numIucnTerms + numOtherTerms);
220
221    String json = mapper.writeValueAsString(v);
222    System.out.println(json);
223
224    VerbatimNameUsage v2 = mapper.readValue(json, VerbatimNameUsage.class);
225    // 5 terms
226    assertEquals(numTerms, v2.getFields().size());
227
228  }

231  @Test
232  public void testVerbatimExtensionsMapSerde() throws Exception {
233    ObjectMapper mapper = new ObjectMapper();
234    mapper.enable(DeserializationConfig.Feature.ACCEPT_EMPTY_STRING_AS_NULL_OBJECT);
235    mapper.enable(SerializationConfig.Feature.INDENT_OUTPUT);
236    mapper.disable(SerializationConfig.Feature.WRITE_NULL_PROPERTIES);
237
238    Map<Term, String> verbatimRecord = new HashMap<Term, String>();
239    Date today = new Date();
240    verbatimRecord.put(DcTerm.created, IsoDateParsingUtils.IsoDateFormat.FULL.getDateFormat().format(today));
241    verbatimRecord.put(DcTerm.creator, "fede");
242    verbatimRecord.put(DcTerm.description, "testDescription");
243    verbatimRecord.put(DcTerm.format, "jpg");
244    verbatimRecord.put(DcTerm.license, "licenseTest");
245    verbatimRecord.put(DcTerm.publisher, "publisherTest");
246    verbatimRecord.put(DcTerm.title, "titleTest");
247    verbatimRecord.put(DcTerm.references, "http://www.gbif.org/");
248    verbatimRecord.put(DcTerm.identifier, "http://www.gbif.org/");
249
250    VerbatimNameUsage v = new VerbatimNameUsage();
251    v.setKey(7);
252    v.setLastCrawled(new Date());
253    Map<Extension, List<Map<Term, String>>> extensions = new HashMap<Extension, List<Map<Term, String>>>();
254    List<Map<Term, String>> verbatimRecords = Lists.newArrayList();
255    verbatimRecords.add(verbatimRecord);
256    extensions.put(Extension.MULTIMEDIA, verbatimRecords);
257    v.setExtensions(extensions);
258
259
260    String json = mapper.writeValueAsString(v);
261    System.out.println(json);
262
263    VerbatimNameUsage v2 = mapper.readValue(json, VerbatimNameUsage.class);
264    assertNotNull(v2.getExtensions());
265    assertTrue(!v2.getExtensions().get(Extension.MULTIMEDIA).isEmpty());
266    assertEquals(v2.getExtensions().get(Extension.MULTIMEDIA).get(0), verbatimRecord);
