015  @Test
016  public void testCanonicalName() throws Exception {
017    ParsedName pn = new ParsedName();
018    pn.setGenusOrAbove("Abies");
019    assertEquals("Abies", pn.canonicalName());
020
021    pn.setSpecificEpithet("alba");
022    assertEquals("Abies alba", pn.canonicalName());
023
024    pn = new ParsedName();
025    pn.setGenusOrAbove("Abies");
026    pn.setSpecificEpithet("alba");
027    pn.setRank(Rank.VARIETY);
028    pn.setAuthorship("Mill.");
029    pn.setBracketAuthorship("Carl.");
030    pn.setNotho(NamePart.GENERIC);
031    pn.setInfraSpecificEpithet("alpina");
032    pn.setYear("1887");
033    pn.setSensu("Döring");
034    pn.setRemarks("lost");
035    pn.setNomStatus("nom. illeg.");
036    pn.setAuthorsParsed(true);
037
038    assertEquals("Abies alba alpina", pn.canonicalName());
039    assertEquals("×Abies alba var. alpina", pn.canonicalNameWithMarker());
040    assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887", pn.canonicalNameComplete());
041    assertEquals("Abies alba", pn.canonicalSpeciesName());
042    assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887 Döring, nom. illeg. [lost]", pn.fullName());
043
044    pn.setGenusOrAbove(null);
045    assertEquals("alba alpina", pn.canonicalName());
046    assertEquals("alba var. alpina", pn.canonicalNameWithMarker());
047    assertEquals("alba var. alpina (Carl.) Mill., 1887", pn.canonicalNameComplete());
048    assertNull(pn.canonicalSpeciesName());
049    assertEquals("alba var. alpina (Carl.) Mill., 1887 Döring, nom. illeg. [lost]", pn.fullName());
050  }

071  @Test
072  public void testIndet() throws Exception {
073    ParsedName pn = new ParsedName();
074    pn.setGenusOrAbove("Abies");
075    assertFalse(pn.isIndetermined());
076
077    pn.setRank(Rank.SPECIES);
078    assertTrue(pn.isIndetermined());
079
080    pn.setSpecificEpithet("vulgaris");
081    assertFalse(pn.isIndetermined());
082
083    pn.setRank(Rank.SUBSPECIES);
084    assertTrue(pn.isIndetermined());
085
086    pn.setInfraSpecificEpithet("kingkong");
087    assertFalse(pn.isIndetermined());
088
089    for (Rank r : Rank.values()) {
090      if (r.isInfraspecific()) {
091        assertFalse(r.toString(), pn.isIndetermined());
092      }
093    }
094
095    pn.setInfraSpecificEpithet(null);
096    for (Rank r : Rank.values()) {
097      if (r.isInfraspecific()) {
098        assertTrue(r.toString(), pn.isIndetermined());
099      }
100    }
101
102    pn.setRank(Rank.SUBGENUS);
103    pn.setSpecificEpithet(null);
104    assertTrue(pn.isIndetermined());
105
106    pn.setInfraGeneric("Mysubgenus");
107    assertFalse(pn.isIndetermined());
108  }

130  @Test
131  public void testSubgenus() throws Exception {
132    // Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013
133    ParsedName pn = new ParsedName();
134    pn.setGenusOrAbove("Brachyhypopomus");
135    pn.setInfraGeneric("Odontohypopomus");
136    pn.setAuthorship("Sullivan, Zuanon & Cox Fernandes");
137    pn.setYear("2013");
138    assertEquals("Odontohypopomus", pn.canonicalName());
139    assertEquals("Odontohypopomus", pn.canonicalNameWithMarker());
140    assertEquals("Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
141    // with given rank marker it is shown instead of brackets
142    pn.setRank(Rank.SUBGENUS);
143    assertEquals("Odontohypopomus", pn.canonicalName());
144    assertEquals("subgen. Odontohypopomus", pn.canonicalNameWithMarker());
145    assertEquals("Brachyhypopomus subgen. Odontohypopomus Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
146
147    // Achillea sect. Ptarmica (Mill.) W.D.J.Koch
148    pn = new ParsedName();
149    pn.setGenusOrAbove("Achillea");
150    pn.setInfraGeneric("Ptarmica");
151    pn.setAuthorship("W.D.J.Koch");
152    pn.setBracketAuthorship("Mill.");
153
154    assertEquals("Ptarmica", pn.canonicalName());
155    assertEquals("Achillea (Ptarmica) (Mill.) W.D.J.Koch", pn.canonicalNameComplete());
156    pn.setRank(Rank.SECTION);
157    assertEquals("Achillea sect. Ptarmica (Mill.) W.D.J.Koch", pn.canonicalNameComplete());
158  }

226  @Test
227  public void testJsonSerde() throws IOException {
228    ParsedName pn = new ParsedName();
229    pn.setGenusOrAbove("Brachyhypopomus");
230    pn.setInfraGeneric("Odontohypopomus");
231    pn.setAuthorship("Sullivan, Zuanon & Cox Fernandes");
232    pn.setYear("2013");
233    pn.setRank(Rank.SUBGENUS);
234    SerdeTestUtils.testSerDe(pn, ParsedName.class);
235
236    pn = new ParsedName();
237    pn.setKey(123);
238    pn.setType(NameType.SCIENTIFIC);
239    pn.setGenusOrAbove("Abax");
240    pn.setSpecificEpithet("carinatus");
241    pn.setInfraSpecificEpithet("carinatus");
242    pn.setBracketAuthorship("Duftschmid");
243    pn.setBracketYear("1812");
244    pn.setRank(Rank.INFRASUBSPECIFIC_NAME);
245    SerdeTestUtils.testSerDe(pn, ParsedName.class);
246
247    pn.setAuthorship("Carlos & Kamera");
248    pn.setYear("1999");
249
250    for (Rank rank : Rank.values()) {
251      pn.setRank(rank);
252      // UNRANKED intentionally becomes null!
253      if (rank.notOtherOrUnknown()) {
254        String json = SerdeTestUtils.testSerDe(pn, ParsedName.class);
255        assertTrue(json.contains("\"rankMarker\""));
256      }
257    }
258    // check that rankMarker=null gets removed from json
259    pn.setRank(null);
260    String json = SerdeTestUtils.serialize(pn);
261    assertFalse(json.contains("\"rankMarker\""));
262  }

264  /**
265   * assert all build name methods return the same string
266   * @param pn
267   * @param name expected string
268   */
269  private void assertBuildName(ParsedName pn, String name) {
270    assertEquals(name, pn.fullName());
271    assertEquals(name, pn.canonicalName());
272    assertEquals(name, pn.canonicalNameComplete());
273    assertEquals(name, pn.canonicalNameWithMarker());
274  }
275
276  private void assertBuildName(ParsedName pn, String full, String canonical, String canonicalComplete, String canonicalMarker) {
277    assertEquals("wrong fullName", full, pn.fullName());
278    assertEquals("wrong canonicalName", canonical, pn.canonicalName());
279    assertEquals("wrong canonicalNameComplete", canonicalComplete, pn.canonicalNameComplete());
280    assertEquals("wrong canonicalNameWithMarker", canonicalMarker, pn.canonicalNameWithMarker());
281  }
