014  @Test
015  public void testCanonicalName() throws Exception {
016    ParsedName pn = new ParsedName();
017    pn.setGenusOrAbove("Abies");
018    assertEquals("Abies", pn.canonicalName());
019
020    pn.setSpecificEpithet("alba");
021    assertEquals("Abies alba", pn.canonicalName());
022
023    pn = new ParsedName();
024    pn.setGenusOrAbove("Abies");
025    pn.setSpecificEpithet("alba");
026    pn.setRank(Rank.VARIETY);
027    pn.setAuthorship("Mill.");
028    pn.setBracketAuthorship("Carl.");
029    pn.setNotho(NamePart.GENERIC);
030    pn.setInfraSpecificEpithet("alpina");
031    pn.setYear("1887");
032    pn.setSensu("Döring");
033    pn.setRemarks("lost");
034    pn.setNomStatus("nom. illeg.");
035    pn.setAuthorsParsed(true);
036
037    assertEquals("Abies alba alpina", pn.canonicalName());
038    assertEquals("×Abies alba var. alpina", pn.canonicalNameWithMarker());
039    assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887", pn.canonicalNameComplete());
040    assertEquals("Abies alba", pn.canonicalSpeciesName());
041    assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887 Döring, nom. illeg. [lost]", pn.fullName());
042  }

095  @Test
096  public void testSubgenus() throws Exception {
097    // Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013
098    ParsedName pn = new ParsedName();
099    pn.setGenusOrAbove("Brachyhypopomus");
100    pn.setInfraGeneric("Odontohypopomus");
101    pn.setAuthorship("Sullivan, Zuanon & Cox Fernandes");
102    pn.setYear("2013");
103    assertEquals("Odontohypopomus", pn.canonicalName());
104    assertEquals("Odontohypopomus", pn.canonicalNameWithMarker());
105    assertEquals("Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
106    // with given rank marker it is shown instead of brackets
107    pn.setRank(Rank.SUBGENUS);
108    assertEquals("Odontohypopomus", pn.canonicalName());
109    assertEquals("subgen. Odontohypopomus", pn.canonicalNameWithMarker());
110    assertEquals("Brachyhypopomus subgen. Odontohypopomus Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
111
112    // Achillea sect. Ptarmica (Mill.) W.D.J.Koch
113    pn = new ParsedName();
114    pn.setGenusOrAbove("Achillea");
115    pn.setInfraGeneric("Ptarmica");
116    pn.setAuthorship("W.D.J.Koch");
117    pn.setBracketAuthorship("Mill.");
118
119    assertEquals("Ptarmica", pn.canonicalName());
120    assertEquals("Achillea (Ptarmica) (Mill.) W.D.J.Koch", pn.canonicalNameComplete());
121    pn.setRank(Rank.SECTION);
122    assertEquals("Achillea sect. Ptarmica (Mill.) W.D.J.Koch", pn.canonicalNameComplete());
123  }

176  /**
177   * assert all build name methods return the same string
178   * @param pn
179   * @param name expected string
180   */
181  private void assertBuildName(ParsedName pn, String name) {
182    assertEquals(name, pn.fullName());
183    assertEquals(name, pn.canonicalName());
184    assertEquals(name, pn.canonicalNameComplete());
185    assertEquals(name, pn.canonicalNameWithMarker());
186  }
187
188  private void assertBuildName(ParsedName pn, String full, String canonical, String canonicalComplete, String canonicalMarker) {
189    assertEquals(full, pn.fullName());
190    assertEquals(canonical, pn.canonicalName());
191    assertEquals(canonicalComplete, pn.canonicalNameComplete());
192    assertEquals(canonicalMarker, pn.canonicalNameWithMarker());
193  }
