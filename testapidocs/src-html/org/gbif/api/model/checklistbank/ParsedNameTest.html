001package org.gbif.api.model.checklistbank;
002
003import org.gbif.api.vocabulary.NamePart;
004import org.gbif.api.vocabulary.Rank;
005
006import org.junit.Test;
007
008import static org.junit.Assert.assertEquals;
009
010public class ParsedNameTest {
011
012  @Test
013  public void testCanonicalName() throws Exception {
014    ParsedName pn = new ParsedName();
015    pn.setGenusOrAbove("Abies");
016    assertEquals("Abies", pn.canonicalName());
017
018    pn.setSpecificEpithet("alba");
019    assertEquals("Abies alba", pn.canonicalName());
020
021
022    pn = new ParsedName();
023    pn.setGenusOrAbove("Abies");
024    pn.setSpecificEpithet("alba");
025    pn.setRank(Rank.VARIETY);
026    pn.setAuthorship("Mill.");
027    pn.setBracketAuthorship("Carl.");
028    pn.setNotho(NamePart.GENERIC);
029    pn.setInfraSpecificEpithet("alpina");
030    pn.setYear("1887");
031    pn.setSensu("D?ring");
032    pn.setRemarks("lost");
033    pn.setNomStatus("nom. illeg.");
034    pn.setAuthorsParsed(true);
035
036    assertEquals("Abies alba alpina", pn.canonicalName());
037    assertEquals("?Abies alba var. alpina", pn.canonicalNameWithMarker());
038    assertEquals("?Abies alba var. alpina (Carl.) Mill., 1887", pn.canonicalNameComplete());
039    assertEquals("Abies alba", pn.canonicalSpeciesName());
040    assertEquals("?Abies alba var. alpina (Carl.) Mill., 1887 D?ring, nom. illeg. [lost]", pn.fullName());
041  }
042}
