013  @Test
014  public void testCanonicalName() throws Exception {
015    ParsedName pn = new ParsedName();
016    pn.setGenusOrAbove("Abies");
017    assertEquals("Abies", pn.canonicalName());
018
019    pn.setSpecificEpithet("alba");
020    assertEquals("Abies alba", pn.canonicalName());
021
022    pn = new ParsedName();
023    pn.setGenusOrAbove("Abies");
024    pn.setSpecificEpithet("alba");
025    pn.setRank(Rank.VARIETY);
026    pn.setAuthorship("Mill.");
027    pn.setBracketAuthorship("Carl.");
028    pn.setNotho(NamePart.GENERIC);
029    pn.setInfraSpecificEpithet("alpina");
030    pn.setYear("1887");
031    pn.setSensu("Döring");
032    pn.setRemarks("lost");
033    pn.setNomStatus("nom. illeg.");
034    pn.setAuthorsParsed(true);
035
036    assertEquals("Abies alba alpina", pn.canonicalName());
037    assertEquals("×Abies alba var. alpina", pn.canonicalNameWithMarker());
038    assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887", pn.canonicalNameComplete());
039    assertEquals("Abies alba", pn.canonicalSpeciesName());
040    assertEquals("×Abies alba var. alpina (Carl.) Mill., 1887 Döring, nom. illeg. [lost]", pn.fullName());
041  }

071  @Test
072  public void testSubgenus() throws Exception {
073    // Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013
074    ParsedName pn = new ParsedName();
075    pn.setGenusOrAbove("Brachyhypopomus");
076    pn.setInfraGeneric("Odontohypopomus");
077    pn.setAuthorship("Sullivan, Zuanon & Cox Fernandes");
078    pn.setYear("2013");
079    assertEquals("Odontohypopomus", pn.canonicalName());
080    assertEquals("Odontohypopomus", pn.canonicalNameWithMarker());
081    assertEquals("Brachyhypopomus (Odontohypopomus) Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
082    // with given rank marker it is shown instead of brackets
083    pn.setRank(Rank.SUBGENUS);
084    assertEquals("Odontohypopomus", pn.canonicalName());
085    assertEquals("subgen. Odontohypopomus", pn.canonicalNameWithMarker());
086    assertEquals("Brachyhypopomus subgen. Odontohypopomus Sullivan, Zuanon & Cox Fernandes, 2013", pn.canonicalNameComplete());
087
088    // Achillea sect. Ptarmica (Mill.) W.D.J.Koch
089    pn = new ParsedName();
090    pn.setGenusOrAbove("Achillea");
091    pn.setInfraGeneric("Ptarmica");
092    pn.setAuthorship("W.D.J.Koch");
093    pn.setBracketAuthorship("Mill.");
094
095    assertEquals("Ptarmica", pn.canonicalName());
096    assertEquals("Achillea (Ptarmica) (Mill.) W.D.J.Koch", pn.canonicalNameComplete());
097    pn.setRank(Rank.SECTION);
098    assertEquals("Achillea sect. Ptarmica (Mill.) W.D.J.Koch", pn.canonicalNameComplete());
099  }

101  @Test
102  public void testBuildName() throws Exception {
103    ParsedName pn = new ParsedName();
104    pn.setGenusOrAbove("Pseudomonas");
105    assertBuildName(pn, "Pseudomonas");
106
107    pn.setSpecificEpithet("syringae");
108    assertBuildName(pn, "Pseudomonas syringae");
109
110    pn.setAuthorsParsed(true);
111    pn.setAuthorship("Van Hall");
112    assertBuildName(pn, "Pseudomonas syringae Van Hall", "Pseudomonas syringae", "Pseudomonas syringae Van Hall", "Pseudomonas syringae");
113
114    pn.setYear("1904");
115    assertBuildName(pn, "Pseudomonas syringae Van Hall, 1904", "Pseudomonas syringae", "Pseudomonas syringae Van Hall, 1904", "Pseudomonas syringae");
116
117    pn.setBracketAuthorship("Carl.");
118    assertBuildName(pn, "Pseudomonas syringae (Carl.) Van Hall, 1904", "Pseudomonas syringae", "Pseudomonas syringae (Carl.) Van Hall, 1904", "Pseudomonas syringae");
119
120    pn.setRank(Rank.PATHOVAR);
121    pn.setInfraSpecificEpithet("aceris");
122    pn.setBracketAuthorship(null);
123    assertBuildName(pn, "Pseudomonas syringae pv. aceris Van Hall, 1904", "Pseudomonas syringae aceris", "Pseudomonas syringae pv. aceris Van Hall, 1904", "Pseudomonas syringae pv. aceris");
124
125    pn.setStrain("CFBP 2339");
126    assertBuildName(pn, "Pseudomonas syringae pv. aceris Van Hall, 1904 CFBP 2339", "Pseudomonas syringae aceris", "Pseudomonas syringae pv. aceris Van Hall, 1904 CFBP 2339", "Pseudomonas syringae pv. aceris CFBP 2339");
127
128    pn.setYear(null);
129    pn.setAuthorship(null);
130    assertBuildName(pn, "Pseudomonas syringae pv. aceris CFBP 2339", "Pseudomonas syringae aceris", "Pseudomonas syringae pv. aceris CFBP 2339", "Pseudomonas syringae pv. aceris CFBP 2339");
131  }

133  /**
134   * assert all build name methods return the same string
135   * @param pn
136   * @param name expected string
137   */
138  private void assertBuildName(ParsedName pn, String name) {
139    assertEquals(name, pn.fullName());
140    assertEquals(name, pn.canonicalName());
141    assertEquals(name, pn.canonicalNameComplete());
142    assertEquals(name, pn.canonicalNameWithMarker());
143  }
144
145  private void assertBuildName(ParsedName pn, String full, String canonical, String canonicalComplete, String canonicalMarker) {
146    assertEquals(full, pn.fullName());
147    assertEquals(canonical, pn.canonicalName());
148    assertEquals(canonicalComplete, pn.canonicalNameComplete());
149    assertEquals(canonicalMarker, pn.canonicalNameWithMarker());
150  }
