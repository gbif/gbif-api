055      + " \"creator\":\"" + TEST_USER + "\", "

074      + " \"predicate\": null"

077  private static final ObjectMapper MAPPER = new ObjectMapper();
078
079  private static PredicateDownloadRequest newDownload(Predicate p) {
080    return newDownload(p, TEST_USER);
081  }
082
083  private static PredicateDownloadRequest newDownload(Predicate p, String user) {
084    return new PredicateDownloadRequest(p, user, Lists.newArrayList(TEST_EMAIL), false, DownloadFormat.DWCA);
085  }

087  @Test
088  public void testAvailable() {
089    //When a Download is created it has RUNNING as its status
090    Download download = new Download();
091    download.setStatus(Download.Status.RUNNING);
092    assertFalse(download.isAvailable());
093
094    //Status changed
095    download.setStatus(Download.Status.SUCCEEDED);
096    assertTrue(download.isAvailable());
097  }
098
099  @Test
100  public void testBasic() {
101    Predicate p = mock(Predicate.class);
102    PredicateDownloadRequest dl = newDownload(p);
103
104    assertThat(dl.getNotificationAddresses(), IsCollectionContaining.hasItem(TEST_EMAIL));
105    assertThat(dl.getPredicate(), equalTo(p));
106  }
107
108  @Test
109  public void testEquals() {
110    Predicate p = mock(Predicate.class);
111
112    DownloadRequest dl1 = newDownload(p);
113    DownloadRequest dl2 = newDownload(p);
114
115    assertThat(dl1, both(equalTo(dl1)).and(equalTo(dl2)));
116  }
117
118  @Test
119  public void testHashcode() {
120    Predicate p = mock(Predicate.class);
121
122    PredicateDownloadRequest dl1 = newDownload(p);
123    PredicateDownloadRequest dl2 = newDownload(p);
124
125    assertThat(dl1.hashCode(), both(equalTo(dl1.hashCode())).and(equalTo(dl2.hashCode())));
126
127    dl2 = newDownload(p, TEST_EMAIL);
128    assertThat(dl1.hashCode(), not(equalTo(dl2.hashCode())));
129  }

131  @Test
132  public void testSerde() throws IOException {
133    PredicateDownloadRequest d = newDownload(new EqualsPredicate(OccurrenceSearchParameter.CATALOG_NUMBER, "b"));
134    try(ByteArrayOutputStream baos = new ByteArrayOutputStream()){
135      MAPPER.writeValue(baos, d);
136      PredicateDownloadRequest d2 = MAPPER.readValue(baos.toByteArray(), PredicateDownloadRequest.class);
137      assertEquals(d, d2);
138    } catch (Throwable e) { // closer must catch Throwable
139      fail(e.getMessage());
140      throw e;
141    }
142  }

144  @Test
145  public void testSQLDownloadSerde() throws IOException {
146    SqlDownloadRequest request = MAPPER.readValue(SQL_REQUEST, SqlDownloadRequest.class);
147    assertEquals(TEST_USER, request.getCreator());
148    assertNotNull(request.getSql());
149    assertEquals(DownloadFormat.SQL, request.getFormat());
150    assertTrue(request.getSendNotification());
151    assertEquals(TEST_EMAIL, request.getNotificationAddressesAsString());
152  }
