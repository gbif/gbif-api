013package org.gbif.api.model.occurrence;
014
015import org.gbif.api.model.occurrence.predicate.EqualsPredicate;
016import org.gbif.api.model.occurrence.predicate.Predicate;
017import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
018
019import java.io.ByteArrayOutputStream;
020import java.util.Date;
021
022import com.google.common.collect.Lists;
023import com.google.common.io.Closeables;
024import org.codehaus.jackson.map.ObjectMapper;
025import org.hamcrest.core.IsCollectionContaining;
026import org.junit.Test;
027
028import static org.hamcrest.CoreMatchers.equalTo;
029import static org.hamcrest.CoreMatchers.not;
030import static org.junit.Assert.assertEquals;
031import static org.junit.Assert.assertFalse;
032import static org.junit.Assert.assertThat;
033import static org.junit.Assert.assertTrue;
034import static org.junit.Assert.fail;
035import static org.junit.matchers.JUnitMatchers.both;
036import static org.mockito.Mockito.mock;
037
038public class DownloadRequestTest {
039
040  private static final String TEST_USER = "user@gbif.org";
041  private static final String TEST_EMAIL = "test@gbif.org";
042
043  private static DownloadRequest newDownload(Predicate p) {
044    return newDownload(p, TEST_USER);
045  }
046
047  private static DownloadRequest newDownload(Predicate p, String user) {
048    return newDownload(p, user, null);
049  }
050
051  private static DownloadRequest newDownload(Predicate p, String user, Date completed) {
052    return new DownloadRequest(p, user, Lists.newArrayList(TEST_EMAIL));
053  }
054
055  @Test
056  public void testAvailable() {
057    Download download = new Download();
058    download.setStatus(Download.Status.RUNNING);
059    assertFalse(download.isAvailable());
060
061    download.setStatus(Download.Status.SUCCEEDED);
062    assertTrue(download.isAvailable());
063  }
064
065  @Test
066  public void testBasic() {
067    Predicate p = mock(Predicate.class);
068    DownloadRequest dl = newDownload(p);
069
070    assertThat(dl.getNotificationAddresses(), IsCollectionContaining.hasItem(TEST_EMAIL));
071    assertThat(dl.getPredicate(), equalTo(p));
072  }
073
074
075  @Test
076  public void testEquals() {
077    Predicate p = mock(Predicate.class);
078
079    DownloadRequest dl1 = newDownload(p);
080    DownloadRequest dl2 = newDownload(p);
081
082    assertThat(dl1, both(equalTo(dl1)).and(equalTo(dl2)));
083  }
084
085
086  @Test
087  public void testHashcode() {
088    Predicate p = mock(Predicate.class);
089
090    DownloadRequest dl1 = newDownload(p);
091    DownloadRequest dl2 = newDownload(p);
092
093    assertThat(dl1.hashCode(), both(equalTo(dl1.hashCode())).and(equalTo(dl2.hashCode())));
094
095    dl2 = newDownload(p, TEST_EMAIL);
096    assertThat(dl1.hashCode(), not(equalTo(dl2.hashCode())));
097  }
098
099  @Test
100  public void testSerde() {
101    ObjectMapper mapper = new ObjectMapper();
102    DownloadRequest d = newDownload(new EqualsPredicate(OccurrenceSearchParameter.CATALOG_NUMBER, "b"));
103    ByteArrayOutputStream baos = new ByteArrayOutputStream();
104    try {
105      mapper.writeValue(baos, d);
106      baos.flush();
107      DownloadRequest d2 = mapper.readValue(baos.toByteArray(), DownloadRequest.class);
108      assertEquals(d, d2);
109
110    } catch (Exception e) {
111      fail(e.getMessage());
112    } finally {
113      Closeables.closeQuietly(baos);
114    }
115  }
116
117}
