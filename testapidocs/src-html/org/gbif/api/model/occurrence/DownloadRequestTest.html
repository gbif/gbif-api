013package org.gbif.api.model.occurrence;
014
015import org.gbif.api.model.occurrence.predicate.EqualsPredicate;
016import org.gbif.api.model.occurrence.predicate.Predicate;
017import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
018
019import java.io.ByteArrayOutputStream;
020import java.io.IOException;
021import java.util.Date;
022
023import com.google.common.collect.Lists;
024import com.google.common.io.Closer;
025import org.codehaus.jackson.map.ObjectMapper;
026import org.hamcrest.core.IsCollectionContaining;
027import org.junit.Test;
028
029import static org.hamcrest.CoreMatchers.equalTo;
030import static org.hamcrest.CoreMatchers.not;
031import static org.junit.Assert.assertEquals;
032import static org.junit.Assert.assertFalse;
033import static org.junit.Assert.assertThat;
034import static org.junit.Assert.assertTrue;
035import static org.junit.Assert.fail;
036import static org.junit.matchers.JUnitMatchers.both;
037import static org.mockito.Mockito.mock;
038
039public class DownloadRequestTest {
040
041  private static final String TEST_USER = "user@gbif.org";
042  private static final String TEST_EMAIL = "test@gbif.org";
043
044  private static DownloadRequest newDownload(Predicate p) {
045    return newDownload(p, TEST_USER);
046  }
047
048  private static DownloadRequest newDownload(Predicate p, String user) {
049    return newDownload(p, user, null);
050  }
051
052  private static DownloadRequest newDownload(Predicate p, String user, Date completed) {
053    return new DownloadRequest(p, user, Lists.newArrayList(TEST_EMAIL), false, DownloadFormat.DWCA);
054  }
055
056  @Test
057  public void testAvailable() {
058    Download download = new Download();
059    download.setStatus(Download.Status.RUNNING);
060    assertFalse(download.isAvailable());
061
062    download.setStatus(Download.Status.SUCCEEDED);
063    assertTrue(download.isAvailable());
064  }
065
066  @Test
067  public void testBasic() {
068    Predicate p = mock(Predicate.class);
069    DownloadRequest dl = newDownload(p);
070
071    assertThat(dl.getNotificationAddresses(), IsCollectionContaining.hasItem(TEST_EMAIL));
072    assertThat(dl.getPredicate(), equalTo(p));
073  }
074
075
076  @Test
077  public void testEquals() {
078    Predicate p = mock(Predicate.class);
079
080    DownloadRequest dl1 = newDownload(p);
081    DownloadRequest dl2 = newDownload(p);
082
083    assertThat(dl1, both(equalTo(dl1)).and(equalTo(dl2)));
084  }
085
086
087  @Test
088  public void testHashcode() {
089    Predicate p = mock(Predicate.class);
090
091    DownloadRequest dl1 = newDownload(p);
092    DownloadRequest dl2 = newDownload(p);
093
094    assertThat(dl1.hashCode(), both(equalTo(dl1.hashCode())).and(equalTo(dl2.hashCode())));
095
096    dl2 = newDownload(p, TEST_EMAIL);
097    assertThat(dl1.hashCode(), not(equalTo(dl2.hashCode())));
098  }
099
100  @Test
101  public void testSerde() throws IOException {
102    ObjectMapper mapper = new ObjectMapper();
103    DownloadRequest d = newDownload(new EqualsPredicate(OccurrenceSearchParameter.CATALOG_NUMBER, "b"));
104
105    Closer closer = Closer.create();
106    try {
107      ByteArrayOutputStream baos = closer.register(new ByteArrayOutputStream());
108      mapper.writeValue(baos, d);
109      baos.flush();
110      DownloadRequest d2 = mapper.readValue(baos.toByteArray(), DownloadRequest.class);
111      assertEquals(d, d2);
112
113    } catch (Throwable e) { // closer must catch Throwable
114      fail(e.getMessage());
115      throw closer.rethrow(e);
116
117    } finally {
118      closer.close();
119    }
120  }
121
122}
