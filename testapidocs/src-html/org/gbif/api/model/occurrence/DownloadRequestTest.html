316  /**
317   * Test three extension situations: known and supported, known but not supported, unknown.
318   */
319  @Test
320  public void downloadWithExtensionTest() throws Exception {
321    String requestTemplate = "{\n"
322      + "  \"creator\":\"gbif_user\",\n"
323      + "  \"notification_address\":[\"gbif_user@mailinator.com\"],\n"
324      + "  \"created\":\"2024\",\n"
325      + "  \"format\": \"DWCA\",\n"
326      + "  \"predicate\":{\n"
327      + "    \"type\":\"equals\",\n"
328      + "    \"key\":\"TAXON_KEY\",\n"
329      + "    \"value\":\"212\"\n"
330      + "  },\n"
331      + "  %s\n"
332      + "}";
333
334    DownloadRequest request;
335
336    // Known and supported extension
337    request = MAPPER.readValue(String.format(requestTemplate, "'verbatimExtensions':['http://rs.tdwg.org/dwc/terms/MeasurementOrFact']".replace("'", "\"")), DownloadRequest.class);
338    assertEquals("gbif_user", request.getCreator());
339    assertEquals(DownloadFormat.DWCA, request.getFormat());
340    assertEquals("gbif_user@mailinator.com", request.getNotificationAddressesAsString());
341    assertEquals(EqualsPredicate.class, ((PredicateDownloadRequest) request).getPredicate().getClass());
342    assertEquals("212", ((EqualsPredicate) ((PredicateDownloadRequest) request).getPredicate()).getValue());
343    assertEquals(Extension.MEASUREMENT_OR_FACT, ((PredicateDownloadRequest) request).getVerbatimExtensions().iterator().next());
344
345    // Known and *unsupported* extension
346    try {
347      MAPPER.readValue(String.format(requestTemplate, "'verbatimExtensions':['http://zooarchnet.org/dwc/terms/ChronometricDate']".replace("'", "\"")), DownloadRequest.class);
348      fail();
349    } catch (Exception e) {
350    }
351
352    // Unknown extension
353    try {
354      MAPPER.readValue(String.format(requestTemplate, "'verbatimExtensions':['http://example.org/nothing']".replace("'", "\"")), DownloadRequest.class);
355      fail();
356    } catch (Exception e) {
357    }
358
359    // Extension enum value — not supported as these are a bit too internal
360    try {
361      MAPPER.readValue(String.format(requestTemplate, "'verbatimExtensions':['MEASUREMENT_OR_FACT']".replace("'", "\"")), DownloadRequest.class);
362      fail();
363    } catch (Exception e) {
364    }
365
366    // Null value
367    try {
368      MAPPER.readValue(String.format(requestTemplate, "'verbatimExtensions':[null]".replace("'", "\"")), DownloadRequest.class);
369      fail();
370    } catch (Exception e) {
371    }
372  }
