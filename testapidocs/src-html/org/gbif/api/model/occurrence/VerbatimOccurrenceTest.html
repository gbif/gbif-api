057    occ.setVerbatimField(DwcTerm.catalogNumber, "");
058    assertFalse(occ.hasVerbatimField(DwcTerm.catalogNumber));
059
060    occ.setVerbatimField(DwcTerm.catalogNumber, null);

135  @Test
136  public void testVerbatimMapSerde() throws Exception {
137    ObjectMapper mapper = new ObjectMapper();
138    mapper.enable(DeserializationConfig.Feature.ACCEPT_EMPTY_STRING_AS_NULL_OBJECT);
139    mapper.enable(SerializationConfig.Feature.INDENT_OUTPUT);
140    mapper.disable(SerializationConfig.Feature.WRITE_NULL_PROPERTIES);
141
142    VerbatimOccurrence v = new VerbatimOccurrence();
143    v.setKey(7l);
144    v.setLastParsed(new Date());
145    v.setDatasetKey(UUID.randomUUID());
146
147
148    for (Term term : DwcTerm.values()) {
149      v.setVerbatimField(term, RandomStringUtils.randomAlphabetic(20));
150    }
151    final int numDwcTerms = v.getVerbatimFields().size();
152
153    for (Term term : DcTerm.values()) {
154      v.setVerbatimField(term, RandomStringUtils.randomAlphabetic(20));
155    }
156    final int numDcTerms = v.getVerbatimFields().size() - numDwcTerms;
157
158    for (Term term : GbifTerm.values()) {
159      v.setVerbatimField(term, RandomStringUtils.randomAlphabetic(20));
160    }
161    final int numGbifTerms = v.getVerbatimFields().size() - numDwcTerms - numDcTerms;
162
163    for (Term term : IucnTerm.values()) {
164      v.setVerbatimField(term, RandomStringUtils.randomAlphanumeric(20));
165    }
166    final int numIucnTerms = v.getVerbatimFields().size() - numDwcTerms - numDcTerms - numGbifTerms;
167
168    v.setVerbatimField(DwcTerm.scientificName, "Abies alba");
169    v.setVerbatimField(DwcTerm.collectionCode, "BUGS");
170    v.setVerbatimField(DwcTerm.catalogNumber, "MD10782");
171    v.setVerbatimField(UnknownTerm.build("http://rs.un.org/terms/temperatur"), RandomStringUtils.randomAlphabetic(30));
172    v.setVerbatimField(UnknownTerm.build("http://rs.un.org/terms/co2"), RandomStringUtils.randomAlphabetic(30));
173    v.setVerbatimField(UnknownTerm.build("http://rs.un.org/terms/modified"), new Date().toString());
174    v.setVerbatimField(UnknownTerm.build("http://rs.un.org/terms/scientificName"),
175      RandomStringUtils.randomAlphabetic(30));
176    final int numOtherTerms = v.getVerbatimFields().size() - numDwcTerms - numDcTerms - numGbifTerms - numIucnTerms;
177
178    final int numTerms = v.getVerbatimFields().size();
179
180    assertEquals(numTerms, numDwcTerms + numDcTerms + numGbifTerms + numIucnTerms + numOtherTerms);
181
182    String json = mapper.writeValueAsString(v);
183    System.out.println(json);
184
185    VerbatimOccurrence v2 = mapper.readValue(json, VerbatimOccurrence.class);
186    // 5 terms
187    assertEquals(numTerms, v2.getVerbatimFields().size());
188
189  }

192  @Test
193  public void testVerbatimExtensionsMapSerde() throws Exception {
194    ObjectMapper mapper = new ObjectMapper();
195    mapper.enable(DeserializationConfig.Feature.ACCEPT_EMPTY_STRING_AS_NULL_OBJECT);
196    mapper.enable(SerializationConfig.Feature.INDENT_OUTPUT);
197    mapper.disable(SerializationConfig.Feature.WRITE_NULL_PROPERTIES);
198
199    Map<Term, String> verbatimRecord = new HashMap<Term, String>();
200    Date today = new Date();
201    verbatimRecord.put(DcTerm.created, IsoDateFormat.FULL.getDateFormat().format(today));
202    verbatimRecord.put(DcTerm.creator, "fede");
203    verbatimRecord.put(DcTerm.description, "testDescription");
204    verbatimRecord.put(DcTerm.format, "jpg");
205    verbatimRecord.put(DcTerm.license, "licenseTest");
206    verbatimRecord.put(DcTerm.publisher, "publisherTest");
207    verbatimRecord.put(DcTerm.title, "titleTest");
208    verbatimRecord.put(DcTerm.references, "http://www.gbif.org/");
209    verbatimRecord.put(DcTerm.identifier, "http://www.gbif.org/");
210
211    VerbatimOccurrence v = new VerbatimOccurrence();
212    v.setKey(7l);
213    v.setLastParsed(new Date());
214    v.setDatasetKey(UUID.randomUUID());
215    Map<Extension, List<Map<Term, String>>> extensions = new HashMap<Extension, List<Map<Term, String>>>();
216    List<Map<Term, String>> verbatimRecords = Lists.newArrayList();
217    verbatimRecords.add(verbatimRecord);
218    extensions.put(Extension.MULTIMEDIA, verbatimRecords);
219    v.setExtensions(extensions);
220
221
222    String json = mapper.writeValueAsString(v);
223    System.out.println(json);
224
225    VerbatimOccurrence v2 = mapper.readValue(json, VerbatimOccurrence.class);
226    assertNotNull(v2.getExtensions());
227    assertTrue(!v2.getExtensions().get(Extension.MULTIMEDIA).isEmpty());
228    assertEquals(v2.getExtensions().get(Extension.MULTIMEDIA).get(0), verbatimRecord);
229

231  }
