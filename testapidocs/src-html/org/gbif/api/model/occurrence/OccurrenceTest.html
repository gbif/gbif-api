084  @Test
085  public void testEquals() {
086    Occurrence o1 = buildTestOccurrence();
087    Occurrence o2 = buildTestOccurrence();
088    Occurrence o3 = buildTestOccurrence();
089    o3.setCountry(Country.POLAND);
090
091    // All of them are equal to themselves
092    assertEquals(o1, o1);
093    assertEquals(o2, o2);
094    assertEquals(o3, o3);
095
096    // But not always equal among themselves
097    assertEquals(o1, o2);
098    assertNotEquals(o1, o3);
099    assertNotEquals(o2, o3);
100  }

252  public void testVerbatimMapSerde() throws Exception {
253    Occurrence o = new Occurrence();
254    o.setKey(7);
255    o.setLastParsed(new Date());
256    o.setDatasetKey(UUID.randomUUID());
257    o.setCountry(Country.ALBANIA);
258    o.setContinent(Continent.AFRICA);
259    o.setBasisOfRecord(BasisOfRecord.OBSERVATION);
260    o.setTaxonKey(212);
261    o.setDay(21);
262    o.setMonth(1);
263    o.setYear(1973);
264    o.setCoordinateAccuracy(0.002);
265
266    for (DwcTerm term : DwcTerm.values()) {
267      if (!term.isClass()) {
268        o.setVerbatimField(term, RandomStringUtils.randomAlphabetic(20));
269      }
270    }
271
272    for (DcTerm term : DcTerm.values()) {
273      if (!term.isClass()) {
274        o.setVerbatimField(term, RandomStringUtils.randomAlphabetic(20));
275      }
276    }
277
278    for (GbifTerm term : GbifTerm.values()) {
279      if (!term.isClass()) {
280        o.setVerbatimField(term, RandomStringUtils.randomAlphabetic(20));
281      }
282    }
283
284    for (IucnTerm term : IucnTerm.values()) {
285      if (!term.isClass()) {
286        o.setVerbatimField(term, RandomStringUtils.randomAlphabetic(20));
287      }
288    }
289
290    o.setVerbatimField(DwcTerm.scientificName, "Abies alba");
291    o.setVerbatimField(DwcTerm.collectionCode, "BUGS");
292    o.setVerbatimField(DwcTerm.catalogNumber, "MD10782");
293    o.setVerbatimField(UnknownTerm.build("http://rs.un.org/terms/temperatur"), RandomStringUtils.randomAlphabetic(30));
294    o.setVerbatimField(UnknownTerm.build("http://rs.un.org/terms/co2"), RandomStringUtils.randomAlphabetic(30));
295    o.setVerbatimField(UnknownTerm.build("http://rs.un.org/terms/modified"), new Date().toString());
296    o.setVerbatimField(UnknownTerm.build("http://rs.un.org/terms/scientificName"),
297        RandomStringUtils.randomAlphabetic(30));
298
299    String json = mapper.writeValueAsString(o);
300    System.out.println(json);
301
302    Occurrence o2 = mapper.readValue(json, Occurrence.class);
303
304    Sets.SetView<Term> diff = Sets.difference(o.getVerbatimFields().keySet(), o2.getVerbatimFields().keySet());
305
306    Iterator<Term> iter = diff.iterator();
307    while (iter.hasNext()) {
308      Term t = iter.next();
309      System.out.println("TERM DIFF: " + t.qualifiedName());
310    }
311
312    assertNull(o2.getVerbatimField(TermFactory.instance().findTerm("coordinateAccuracyInMeters")));
313    assertEquals(o.getVerbatimFields().size(), o2.getVerbatimFields().size());
314    assertTrue(diff.isEmpty());
315  }

320  @Test
321  public void testCustomSerlializations() throws Exception {
322    Occurrence o = new Occurrence();
323    o.setKey(7);
324    o.setCountry(Country.ALGERIA);
325    o.setClassKey(999);
326    o.setClazz("Insecta");
327    o.setVerbatimField(DwcTerm.recordedBy, "Sankt Nikolaus");
328
329    String json = mapper.writeValueAsString(o);
330    System.out.println(json);
331    assertTrue(json.contains("\"class\" : \"Insecta\""));
332    assertTrue(json.contains("\"countryCode\" : \"DZ\""));
333    assertTrue(json.contains("\"country\" : \"Algeria\""));
334
335    Occurrence o2 = mapper.readValue(json, Occurrence.class);
336
337    assertEquals(Country.ALGERIA, o2.getCountry());
338    assertEquals("Insecta", o2.getClazz());
339    assertNull(o2.getGeodeticDatum());
340
341    // now add coords, datum shows
342    o.setDecimalLatitude(12d);
343    o.setDecimalLongitude(23d);
344    json = mapper.writeValueAsString(o);
345    System.out.println(json);
346    assertTrue(json.contains("\"geodeticDatum\" : \"WGS84\""));
347
348    o2 = mapper.readValue(json, Occurrence.class);
349    assertEquals("WGS84", o2.getGeodeticDatum());
350
351    Sets.SetView<Term> diff = Sets.difference(o.getVerbatimFields().keySet(), o2.getVerbatimFields().keySet());
352
353    Iterator<Term> iter = diff.iterator();
354    while (iter.hasNext()) {
355      Term t = iter.next();
356      System.out.println(t.qualifiedName());
357    }
358
359    assertEquals(o.getVerbatimFields().size(), o2.getVerbatimFields().size());
360    assertTrue(diff.isEmpty());
361  }
