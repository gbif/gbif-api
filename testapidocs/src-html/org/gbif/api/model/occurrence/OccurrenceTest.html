016package org.gbif.api.model.occurrence;
017
018import org.gbif.api.util.ClassificationUtils;
019import org.gbif.api.vocabulary.Country;
020import org.gbif.api.vocabulary.EndpointType;
021
022import java.util.LinkedHashMap;
023import java.util.UUID;
024
025import org.junit.Test;
026
027import static org.junit.Assert.assertEquals;
028import static org.junit.Assert.assertFalse;
029import static org.junit.Assert.assertTrue;
030
031public class OccurrenceTest {
032
033  @Test
034  public void testGetHigherClassificationMap() throws Exception {
035    Occurrence occ = new Occurrence();
036    occ.setFamily("Plants family");
037    occ.setFamilyKey(16);
038    occ.setKingdom("Plants");
039    occ.setKingdomKey(6);
040
041    LinkedHashMap<Integer, String> clMap = ClassificationUtils.getHigherClassificationMap(occ);
042    assertEquals(2, clMap.size());
043    assertTrue(clMap.containsKey(16));
044    assertTrue(clMap.containsKey(6));
045    assertTrue(clMap.containsValue("Plants"));
046    assertTrue(clMap.containsValue("Plants family"));
047
048    occ.setTaxonKey(200);
049    clMap = ClassificationUtils.getHigherClassificationMap(occ);
050    assertEquals(2, clMap.size());
051
052    occ.setTaxonKey(16);
053    clMap = ClassificationUtils.getHigherClassificationMap(occ, occ.getTaxonKey(), null, null);
054    assertEquals(1, clMap.size());
055    assertTrue(clMap.containsKey(6));
056    assertTrue(clMap.containsValue("Plants"));
057    assertFalse(clMap.containsKey(16));
058    assertFalse(clMap.containsValue("Plants family"));
059  }
060
061  @Test
062  public void testProtocolAndPublishingCountry() {
063    Occurrence occ = new Occurrence();
064    occ.setKey(1);
065    occ.setDatasetKey(UUID.randomUUID());
066    occ.setProtocol(EndpointType.BIOCASE);
067    occ.setPublishingCountry(Country.AFGHANISTAN);
068
069    assertEquals(EndpointType.BIOCASE, occ.getProtocol());
070    assertEquals(Country.AFGHANISTAN, occ.getPublishingCountry());
071  }
072}
