016package org.gbif.api.model.occurrence.predicate;
017
018import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
019
020import java.util.ArrayList;
021import java.util.List;
022
023import com.google.common.collect.Lists;
024import org.junit.Test;
025
026import static org.hamcrest.CoreMatchers.equalTo;
027import static org.hamcrest.CoreMatchers.not;
028import static org.junit.Assert.assertThat;
029import static org.mockito.Mockito.mock;
030
031public class CompoundPredicateTest {
032
033  @Test(expected = IllegalArgumentException.class)
034  public void testEmptyPredicates() {
035    new ConjunctionPredicate(new ArrayList<Predicate>());
036  }
037
038  @Test
039  public void testEquals() {
040    List<Predicate> mocks = Lists.newArrayList(mock(Predicate.class), mock(Predicate.class));
041
042    // Coming from the same list so should be the same
043    Predicate cp1 = new ConjunctionPredicate(mocks);
044    Predicate cp2 = new ConjunctionPredicate(mocks);
045    assertThat(cp1, equalTo(cp2));
046    assertThat(cp1, equalTo(cp1));
047
048    // Now add one to the list, the underlying list for cp1 should not change so cp1 != cp2
049    mocks.add(mock(Predicate.class));
050    cp2 = new ConjunctionPredicate(mocks);
051    assertThat(cp1, not(equalTo(cp2)));
052
053    // Compare a Conjunction with a Disjunction
054    cp1 = new ConjunctionPredicate(mocks);
055    Predicate dp1 = new DisjunctionPredicate(mocks);
056    assertThat(cp1, not(equalTo(dp1)));
057    assertThat(dp1, not(equalTo(cp1)));
058    assertThat(dp1, equalTo(dp1));
059
060    Predicate dp2 = new DisjunctionPredicate(mocks);
061    assertThat(dp1, equalTo(dp2));
062  }
063
064  @Test
065  public void testToString() {
066    Predicate p = new ConjunctionPredicate(Lists.newArrayList(
067      new EqualsPredicate(OccurrenceSearchParameter.BASIS_OF_RECORD, "FOSSIL_SPECIMEN"),
068      new NotPredicate(
069        new DisjunctionPredicate(Lists.<Predicate>newArrayList(
070          new EqualsPredicate(OccurrenceSearchParameter.YEAR, "2001"),
071          new EqualsPredicate(OccurrenceSearchParameter.YEAR, "2000"),
072          new EqualsPredicate(OccurrenceSearchParameter.YEAR, "1999")
073        ))
074      ),
075      new EqualsPredicate(OccurrenceSearchParameter.TAXON_KEY, "212")
076      )
077    );
078
079    // make this doesnt throw an exception
080    p.toString();
081  }
082
083  @Test
084  public void testGetPredicates() {
085    List<Predicate> mocks = Lists.newArrayList(mock(Predicate.class), mock(Predicate.class));
086    ConjunctionPredicate predicate = new ConjunctionPredicate(mocks);
087    assertThat(mocks, equalTo(predicate.getPredicates()));
088  }
089
090  @Test
091  public void testHashcode() {
092    List<Predicate> mocks = Lists.newArrayList(mock(Predicate.class), mock(Predicate.class));
093
094    Predicate p1 = new ConjunctionPredicate(mocks);
095    Predicate p2 = new DisjunctionPredicate(mocks);
096
097    assertThat(p1.hashCode(), equalTo(p2.hashCode()));
098
099    mocks.add(mock(Predicate.class));
100    p2 = new DisjunctionPredicate(mocks);
101    assertThat(p1.hashCode(), not(equalTo(p2.hashCode())));
102  }
103
104  @Test(expected = NullPointerException.class)
105  public void testNullPredicate() {
106    new ConjunctionPredicate(null);
107  }
108
109}
