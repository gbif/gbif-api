013package org.gbif.api.model.occurrence.predicate;
014
015import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
016
017import java.util.UUID;
018
019import org.junit.Test;
020
021import static org.hamcrest.CoreMatchers.equalTo;
022import static org.hamcrest.CoreMatchers.not;
023import static org.junit.Assert.assertThat;
024import static org.junit.matchers.JUnitMatchers.both;
025import static org.mockito.Mockito.mock;
026
027public class SimplePredicateTest {
028
029  private static final OccurrenceSearchParameter TEST_KEY = OccurrenceSearchParameter.CATALOG_NUMBER;
030  private static final String TEST_VALUE = "bar";
031  private static final OccurrenceSearchParameter COMP_PARAM = OccurrenceSearchParameter.ELEVATION;
032  private static final String COMP_VALUE = "100";
033
034  @Test(expected = IllegalArgumentException.class)
035  public void testEmptyConstructor() {
036    new EqualsPredicate(TEST_KEY, "");
037  }
038
039  @Test
040  public void testEquals() {
041    SimplePredicate ep1 = new EqualsPredicate(TEST_KEY, TEST_VALUE);
042    SimplePredicate ep2 = new EqualsPredicate(TEST_KEY, TEST_VALUE);
043    SimplePredicate gt1 = new GreaterThanPredicate(COMP_PARAM, COMP_VALUE);
044    SimplePredicate gt2 = new GreaterThanPredicate(COMP_PARAM, COMP_VALUE);
045    SimplePredicate lt1 = new LessThanPredicate(COMP_PARAM, COMP_VALUE);
046    SimplePredicate lt2 = new LessThanPredicate(COMP_PARAM, COMP_VALUE);
047
048    // All of them are equal to themselves
049    assertThat(ep1, equalTo(ep1));
050    assertThat(gt1, equalTo(gt1));
051    assertThat(lt1, equalTo(lt1));
052
053    // But unequal among themselves
054    assertThat(ep1, both(not(equalTo(gt1))).and(not(equalTo(lt1))));
055    assertThat(gt1, both(not(equalTo(ep1))).and(not(equalTo(lt1))));
056    assertThat(lt1, both(not(equalTo(ep1))).and(not(equalTo(gt1))));
057
058    // And certainly unequal to any other Predicate
059    assertThat(ep1, not(equalTo(mock(Predicate.class))));
060
061    assertThat(ep1, equalTo(ep2));
062    assertThat(gt1, equalTo(gt2));
063    assertThat(lt1, equalTo(lt2));
064
065    ep2 = new EqualsPredicate(TEST_KEY, "FOOBAR");
066    assertThat(ep1, not(equalTo(ep2)));
067  }
068
069  @Test
070  public void testGoodDoubleValue() {
071    new EqualsPredicate(OccurrenceSearchParameter.DECIMAL_LATITUDE, "32");
072    new EqualsPredicate(OccurrenceSearchParameter.DECIMAL_LATITUDE, "32.032");
073  }
074
075  @Test
076  public void testGoodEnumValue() {
077    new EqualsPredicate(OccurrenceSearchParameter.BASIS_OF_RECORD, "PRESERVED_SPECIMEN");
078  }
079
080  @Test
081  public void testHashcode() {
082    SimplePredicate sp1 = new EqualsPredicate(TEST_KEY, TEST_VALUE);
083    SimplePredicate sp2 = new EqualsPredicate(TEST_KEY, TEST_VALUE);
084    SimplePredicate sp3 = new LessThanPredicate(COMP_PARAM, COMP_VALUE);
085    SimplePredicate sp4 = new GreaterThanPredicate(COMP_PARAM, COMP_VALUE);
086
087    assertThat(sp1.hashCode(), equalTo(sp2.hashCode()));
088    assertThat(sp1.hashCode(), not(equalTo(sp3.hashCode())));
089    assertThat(sp3.hashCode(), equalTo(sp4.hashCode()));
090  }
091
092  @Test(expected = NullPointerException.class)
093  public void testNullConstructor() {
094    new EqualsPredicate(null, null);
095  }
096
097  @Test(expected = IllegalArgumentException.class)
098  public void testOnlyEqualsAllowed() {
099    new LessThanPredicate(OccurrenceSearchParameter.DATASET_KEY, UUID.randomUUID().toString());
100  }
101
102  @Test
103  public void testSimplePredicateConstruction() {
104    SimplePredicate p = new EqualsPredicate(TEST_KEY, TEST_VALUE);
105    assertThat(p.getKey(), equalTo(TEST_KEY));
106    assertThat(p.getValue(), equalTo(TEST_VALUE));
107  }
108
109  @Test(expected = IllegalArgumentException.class)
110  public void testWrongDoubleValue() {
111    new EqualsPredicate(OccurrenceSearchParameter.DECIMAL_LATITUDE, "32..312");
112  }
113
114  @Test(expected = IllegalArgumentException.class)
115  public void testWrongEnumValue() {
116    new EqualsPredicate(OccurrenceSearchParameter.BASIS_OF_RECORD, "Specimen");
117  }
118
119  @Test(expected = IllegalArgumentException.class)
120  public void testWrongDoubleValue2() {
121    new EqualsPredicate(OccurrenceSearchParameter.ELEVATION, "0.6s");
122  }
123
124  @Test(expected = IllegalArgumentException.class)
125  public void testWrongUuidValue() {
126    new EqualsPredicate(OccurrenceSearchParameter.DATASET_KEY, "133-4312-443-2-43-32432423-");
127  }
