016package org.gbif.api.model.occurrence.predicate;
017
018import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
019
020import java.util.UUID;
021
022import org.junit.jupiter.api.Test;
023
024import static org.hamcrest.CoreMatchers.both;
025import static org.hamcrest.CoreMatchers.equalTo;
026import static org.hamcrest.CoreMatchers.not;
027import static org.hamcrest.MatcherAssert.assertThat;
028import static org.junit.jupiter.api.Assertions.assertThrows;
029import static org.mockito.Mockito.mock;
030
031public class SimplePredicateTest {
032
033  private static final OccurrenceSearchParameter TEST_KEY = OccurrenceSearchParameter.CATALOG_NUMBER;
034  private static final String TEST_VALUE = "bar";
035  private static final OccurrenceSearchParameter COMP_PARAM = OccurrenceSearchParameter.ELEVATION;
036  private static final String COMP_VALUE = "100";
037
038  @Test
039  public void testEmptyConstructor() {
040    assertThrows(IllegalArgumentException.class, () -> new EqualsPredicate(TEST_KEY, "", false));
041  }
042
043  @Test
044  public void testEquals() {
045    SimplePredicate ep1 = new EqualsPredicate(TEST_KEY, TEST_VALUE, false);
046    SimplePredicate ep2 = new EqualsPredicate(TEST_KEY, TEST_VALUE, false);
047    SimplePredicate gt1 = new GreaterThanPredicate(COMP_PARAM, COMP_VALUE);
048    SimplePredicate gt2 = new GreaterThanPredicate(COMP_PARAM, COMP_VALUE);
049    SimplePredicate lt1 = new LessThanPredicate(COMP_PARAM, COMP_VALUE);
050    SimplePredicate lt2 = new LessThanPredicate(COMP_PARAM, COMP_VALUE);
051
052    // All of them are equal to themselves
053    assertThat(ep1, equalTo(ep1));
054    assertThat(gt1, equalTo(gt1));
055    assertThat(lt1, equalTo(lt1));
056
057    // But unequal among themselves
058    assertThat(ep1, both(not(equalTo(gt1))).and(not(equalTo(lt1))));
059    assertThat(gt1, both(not(equalTo(ep1))).and(not(equalTo(lt1))));
060    assertThat(lt1, both(not(equalTo(ep1))).and(not(equalTo(gt1))));
061
062    // And certainly unequal to any other Predicate
063    assertThat(ep1, not(equalTo(mock(Predicate.class))));
064
065    assertThat(ep1, equalTo(ep2));
066    assertThat(gt1, equalTo(gt2));
067    assertThat(lt1, equalTo(lt2));
068
069    ep2 = new EqualsPredicate(TEST_KEY, "FOOBAR", false);
070    assertThat(ep1, not(equalTo(ep2)));
071  }
072
073  @Test
074  public void testGoodDoubleValue() {
075    new EqualsPredicate(OccurrenceSearchParameter.DECIMAL_LATITUDE, "32", false);
076    new EqualsPredicate(OccurrenceSearchParameter.DECIMAL_LATITUDE, "32.032", false);
077  }
078
079  @Test
080  public void testGoodEnumValue() {
081    new EqualsPredicate(OccurrenceSearchParameter.BASIS_OF_RECORD, "PRESERVED_SPECIMEN", false);
082  }
083
084  @Test
085  public void testHashcode() {
086    SimplePredicate sp1 = new EqualsPredicate(TEST_KEY, TEST_VALUE, false);
087    SimplePredicate sp2 = new EqualsPredicate(TEST_KEY, TEST_VALUE, false);
088    SimplePredicate sp3 = new LessThanPredicate(COMP_PARAM, COMP_VALUE);
089    SimplePredicate sp4 = new GreaterThanPredicate(COMP_PARAM, COMP_VALUE);
090
091    assertThat(sp1.hashCode(), equalTo(sp2.hashCode()));
092    assertThat(sp1.hashCode(), not(equalTo(sp3.hashCode())));
093    assertThat(sp3.hashCode(), equalTo(sp4.hashCode()));
094  }
095
096  @Test
097  public void testNullConstructor() {
098    assertThrows(NullPointerException.class, () -> new EqualsPredicate(null, null, false));
099  }
100
101  @Test
102  public void testOnlyEqualsAllowed() {
103    assertThrows(IllegalArgumentException.class, () -> new LessThanPredicate(OccurrenceSearchParameter.DATASET_KEY, UUID.randomUUID().toString()));
104  }
105
106  @Test
107  public void testSimplePredicateConstruction() {
108    SimplePredicate p = new EqualsPredicate(TEST_KEY, TEST_VALUE, false);
109    assertThat(p.getKey(), equalTo(TEST_KEY));
110    assertThat(p.getValue(), equalTo(TEST_VALUE));
111  }
