013package org.gbif.api.model.occurrence.predicate;
014
015import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
016
017import java.util.ArrayList;
018import java.util.List;
019
020import com.google.common.collect.Lists;
021import org.junit.Test;
022
023import static org.hamcrest.CoreMatchers.equalTo;
024import static org.hamcrest.CoreMatchers.not;
025import static org.junit.Assert.assertThat;
026import static org.junit.matchers.JUnitMatchers.both;
027import static org.mockito.Mockito.mock;
028
029public class InPredicateTest {
030
031  private static final OccurrenceSearchParameter TEST_KEY = OccurrenceSearchParameter.CATALOG_NUMBER;
032
033  @Test
034  public void testBasics() {
035    List<String> list = Lists.newArrayList("foo", "bar");
036    InPredicate p = new InPredicate(TEST_KEY, list);
037
038    assertThat(p.getKey(), equalTo(TEST_KEY));
039    assertThat((List<String>) p.getValues(), equalTo(list));
040
041    list.add("oink");
042    assertThat((List<String>) p.getValues(), not(equalTo(list)));
043  }
044
045  @Test(expected = NullPointerException.class)
046  public void testEmptyConstructor() {
047    new InPredicate(null, new ArrayList<String>());
048  }
049
050  @Test(expected = IllegalArgumentException.class)
051  public void testEmptyConstructor2() {
052    new InPredicate(OccurrenceSearchParameter.CATALOG_NUMBER, new ArrayList<String>());
053  }
054
055  @Test
056  public void testEquals() {
057    List<String> list = Lists.newArrayList("foo", "bar");
058    Predicate ip1 = new InPredicate(TEST_KEY, list);
059    Predicate ip2 = new InPredicate(TEST_KEY, list);
060
061    assertThat(ip1, both(equalTo(ip1)).and(equalTo(ip2)));
062
063    Predicate p = mock(Predicate.class);
064    assertThat(p, not(equalTo(ip1)));
065
066    list.add("oink");
067    ip2 = new InPredicate(TEST_KEY, list);
068    assertThat(ip2, not(equalTo(ip1)));
069  }
070
071  @Test
072  public void testHashcode() {
073    List<String> list = Lists.newArrayList("foo", "bar");
074    Predicate ip1 = new InPredicate(TEST_KEY, list);
075    Predicate ip2 = new InPredicate(TEST_KEY, list);
076
077    assertThat(ip1.hashCode(), both(equalTo(ip1.hashCode())).and(equalTo(ip2.hashCode())));
078
079    Predicate p = mock(Predicate.class);
080    assertThat(p.hashCode(), not(equalTo(ip1.hashCode())));
081
082    list.add("oink");
083    ip2 = new InPredicate(TEST_KEY, list);
084    assertThat(ip2.hashCode(), not(equalTo(ip1.hashCode())));
085  }
086
087  @Test(expected = NullPointerException.class)
088  public void testNullConstructor() {
089    new InPredicate(null, null);
090  }
091
092  @Test(expected = NullPointerException.class)
093  public void testNullConstructor2() {
094    new InPredicate(OccurrenceSearchParameter.CATALOG_NUMBER, null);
095  }
096
097}
