001package org.gbif.api.model.metrics.cube;
002
003import org.gbif.api.vocabulary.Country;
004import org.gbif.api.vocabulary.Kingdom;
005
006import java.util.UUID;
007
008import org.junit.Test;
009
010import static org.junit.Assert.assertEquals;
011import static org.junit.Assert.assertNotNull;
012
013/**
014 * Tests the type safety of the builder, and illustrates usage.
015 */
016public class ReadBuilderTest {
017
018  private static final class BooleanDimension extends Dimension<Boolean> {
019
020    public BooleanDimension() {
021      super("boolean", Boolean.class);
022    }
023  }
024
025  private static final class DoubleDimension extends Dimension<Double> {
026
027    public DoubleDimension() {
028      super("double", Double.class);
029    }
030  }
031
032  private static final class EnumDimension extends Dimension<Kingdom> {
033
034    public EnumDimension() {
035      super("enum", Kingdom.class);
036    }
037  }
038
039  private static final class FloatDimension extends Dimension<Float> {
040
041    public FloatDimension() {
042      super("float", Float.class);
043    }
044  }
045
046  private static final class IntDimension extends Dimension<Integer> {
047
048    public IntDimension() {
049      super("int", Integer.class);
050    }
051  }
052
053  private static final class StringDimension extends Dimension<String> {
054
055    public StringDimension() {
056      super("string", String.class);
057    }
058  }
059
060  private static final class UUIDDimension extends Dimension<UUID> {
061
062    public UUIDDimension() {
063      super("uuid", UUID.class);
064    }
065  }
066
067  private static final class CountryDimension extends Dimension<Country> {
068
069    public CountryDimension() {
070      super("country", Country.class);
071    }
072  }
073
074  @Test
075  public void testDimensions() {
076    ReadBuilder b = new ReadBuilder();
077    UUID uuid = UUID.randomUUID();
078
079    b.at(new StringDimension(), "1");
080    assertNotNull(b.build());
081    assertEquals(1, b.build().size());
082    assertEquals("1", b.build().get(new StringDimension()));
083
084    b.at(new IntDimension(), 1);
085    assertNotNull(b.build());
086    assertEquals(2, b.build().size());
087    assertEquals("1", b.build().get(new IntDimension()));
088
089    b.at(new FloatDimension(), 1f);
090    assertNotNull(b.build());
091    assertEquals(3, b.build().size());
092    assertEquals("1.0", b.build().get(new FloatDimension()));
093
094    b.at(new DoubleDimension(), 1d);
095    assertNotNull(b.build());
096    assertEquals(4, b.build().size());
097    assertEquals("1.0", b.build().get(new DoubleDimension()));
098
099    b.at(new UUIDDimension(), uuid);
100    assertNotNull(b.build());
101    assertEquals(5, b.build().size());
102    assertEquals(uuid.toString(), b.build().get(new UUIDDimension()));
103
104    b.at(new EnumDimension(), Kingdom.ANIMALIA);
105    assertNotNull(b.build());
106    assertEquals(6, b.build().size());
107    assertEquals(Kingdom.ANIMALIA.toString(), b.build().get(new EnumDimension()));
108
109    b.at(new BooleanDimension(), true);
110    assertNotNull(b.build());
111    assertEquals(7, b.build().size());
112    assertEquals(String.valueOf(true), b.build().get(new BooleanDimension()));
113
114    b.at(new CountryDimension(), Country.GERMANY);
115    assertNotNull(b.build());
116    assertEquals(8, b.build().size());
117    assertEquals("DE", b.build().get(new CountryDimension()));
118  }
119}
