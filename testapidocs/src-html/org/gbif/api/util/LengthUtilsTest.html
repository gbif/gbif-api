001package org.gbif.api.util;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006
007/**
008 *
009 */
010public class LengthUtilsTest {
011
012  @Test
013  public void testMetersToLatDegree() throws Exception {
014    assertEquals(0.009043, LengthUtils.metersToLatDegree(1000), 0.0000001);
015    assertEquals(0.000904,  LengthUtils.metersToLatDegree(100), 0.0000001);
016    assertEquals(0.00009, LengthUtils.metersToLatDegree(10), 0.0000001);
017    assertEquals(0.000045, LengthUtils.metersToLatDegree(5), 0.0000001);
018    assertEquals(0.000009, LengthUtils.metersToLatDegree(1), 0.0000001);
019    assertEquals(-0.000009, LengthUtils.metersToLatDegree(-1), 0.0000001);
020
021    assertEquals(0.0, LengthUtils.metersToLatDegree(0), 0.000001);
022  }
023
024  @Test
025  public void testLatDegreeToMeters() throws Exception {
026    assertEquals(1000, LengthUtils.latDegreeToMeters(0.009044), 0.1);
027    assertEquals(100,  LengthUtils.latDegreeToMeters(0.000904),  0.1);
028    assertEquals(10,   LengthUtils.latDegreeToMeters(0.00009), 0.1);
029    assertEquals(5,    LengthUtils.latDegreeToMeters(0.000045), 0.1);
030    assertEquals(1,    LengthUtils.latDegreeToMeters(0.000009), 0.1);
031
032    assertEquals(0.0, LengthUtils.latDegreeToMeters(0), 0.000001);
033  }
034}
