051  @Test
052  public void testDate() throws Exception {
053    final String lower = "1871-03-21";
054    final String upper = "1888-01-01";
055    Map<OccurrenceSearchParameter, LinkedList<String>> x = builder.humanFilter(
056      new ConjunctionPredicate(Lists.<Predicate>newArrayList(
057        new GreaterThanOrEqualsPredicate(OccurrenceSearchParameter.EVENT_DATE, lower),

066  /**
067   * test all available search parameters and make sure we have a bundle entry for all possible enum values
068   */
069  @Test
070  public void testAllParams() throws Exception {
071
072    final String date = DateFormatUtils.ISO_DATE_FORMAT.format(new Date());
073    List<Predicate> ands = Lists.newArrayList();
074    for (OccurrenceSearchParameter p : OccurrenceSearchParameter.values()) {
075      if (p.type().isEnum()) {
076        if (p.type() == Country.class) {
077          ands.add(new EqualsPredicate(p, Country.DENMARK.getIso2LetterCode()));
078
079        } else {
080          Class<Enum<?>> vocab = (Class<Enum<?>>) p.type();
081          // add a comparison for every possible enum value to test the resource bundle for completeness
082          List<Predicate> ors = Lists.newArrayList();
083          for (Enum<?> e : vocab.getEnumConstants()) {
084            ors.add(new EqualsPredicate(p, e.toString()));
085          }
086          ands.add(new DisjunctionPredicate(ors));
087        }
088
089      } else if (p.type() == Date.class) {
090        ands.add(new EqualsPredicate(p, date));
091
092      } else if (p.type() == Double.class) {
093        ands.add(new EqualsPredicate(p, "12.478"));
094
095      } else if (p.type() == Integer.class) {
096        ands.add(new EqualsPredicate(p, "10"));
097
098      } else if (p.type() == String.class) {
099        if (p == OccurrenceSearchParameter.GEOMETRY) {
100          ands.add(new EqualsPredicate(p, "POLYGON ((30 10, 10 20, 20 40, 40 40, 30 10))"));
101        } else {
102          ands.add(new EqualsPredicate(p, "Bernd Neumann"));
103        }
104
105      } else if (p.type() == Boolean.class) {
106        ands.add(new EqualsPredicate(p, "true"));
107
108      } else if (p.type() == UUID.class) {
109        ands.add(new EqualsPredicate(p, UUID.randomUUID().toString()));
110
111      } else {
112        throw new IllegalStateException("Unknown SearchParameter type " + p.type());
113      }
114    }
115    ConjunctionPredicate and = new ConjunctionPredicate(ands);
116
117    Map<OccurrenceSearchParameter, LinkedList<String>> x = builder.humanFilter(and);
118    for (OccurrenceSearchParameter p : x.keySet()) {
119      System.out.print(p + "\t");
120      System.out.println(x.get(p));
121    }
122  }

124  @Test
125  public void testHumanFilter() throws Exception {
126    List<Predicate> ors = Lists.newArrayList();
127    ors.add(new GreaterThanOrEqualsPredicate(OccurrenceSearchParameter.YEAR, "2000"));
128    ors.add(new GreaterThanPredicate(OccurrenceSearchParameter.YEAR, "2001"));
129    ors.add(new LessThanPredicate(OccurrenceSearchParameter.YEAR, "1750"));
130    ors.add(new LessThanOrEqualsPredicate(OccurrenceSearchParameter.YEAR, "1760"));
131    ors.add(new IsNotNullPredicate(OccurrenceSearchParameter.YEAR));
132    DisjunctionPredicate or = new DisjunctionPredicate(ors);
133
134
135    Map<OccurrenceSearchParameter, LinkedList<String>> x = builder.humanFilter(or);
136    assertEquals(5, x.get(OccurrenceSearchParameter.YEAR).size());
137    assertEquals(">= 2000", x.get(OccurrenceSearchParameter.YEAR).getFirst());
138    assertEquals("> 2001", x.get(OccurrenceSearchParameter.YEAR).get(1));
139    assertEquals("< 1750", x.get(OccurrenceSearchParameter.YEAR).get(2));
140    assertEquals("<= 1760", x.get(OccurrenceSearchParameter.YEAR).get(3));
141    assertEquals("is not null", x.get(OccurrenceSearchParameter.YEAR).getLast());
142  }

144  // http://www.gbif-dev.org/occurrence/search?COUNTRY=DE&BASIS_OF_RECORD=HUMAN_OBSERVATION
145  // &DATASET_KEY=7a22e1e4-f762-11e1-a439-00145eb45e9a

148  // &DATASET_KEY=85685a84-f762-11e1-a439-00145eb45e9a
149  // &DEPTH=0%2C*#
150  @Test
151  public void testHumanFilter2() {
152    List<Predicate> ors = Lists.newArrayList();
153    ors.add(new EqualsPredicate(OccurrenceSearchParameter.DATASET_KEY, "7a22e1e4-f762-11e1-a439-00145eb45e9a"));
154    ors.add(new EqualsPredicate(OccurrenceSearchParameter.DATASET_KEY, "78ff18a6-1c32-11e2-af65-00145eb45e9a"));
155    ors.add(new EqualsPredicate(OccurrenceSearchParameter.DATASET_KEY, "856734ce-f762-11e1-a439-00145eb45e9a"));
156    ors.add(new EqualsPredicate(OccurrenceSearchParameter.DATASET_KEY, "85685a84-f762-11e1-a439-00145eb45e9a"));
157
158    List<Predicate> ands = Lists.newArrayList();
159    ands.add(new EqualsPredicate(OccurrenceSearchParameter.BASIS_OF_RECORD, "HUMAN_OBSERVATION"));
160    ands.add(new EqualsPredicate(OccurrenceSearchParameter.COUNTRY, "DE"));
161    ands.add(new DisjunctionPredicate(ors));
162    ands.add(new GreaterThanOrEqualsPredicate(OccurrenceSearchParameter.DEPTH, "0"));
163
164    Map<OccurrenceSearchParameter, LinkedList<String>> x = builder.humanFilter(new ConjunctionPredicate(ands));
165    assertEquals(4, x.size());
166    assertEquals("Human Observation", x.get(OccurrenceSearchParameter.BASIS_OF_RECORD).getFirst());
167    assertEquals("Germany", x.get(OccurrenceSearchParameter.COUNTRY).getFirst());
168    assertEquals(4, x.get(OccurrenceSearchParameter.DATASET_KEY).size());
169    assertEquals(">= 0m", x.get(OccurrenceSearchParameter.DEPTH).getFirst());
170  }

172  @Test
173  public void testHumanFilterXmlEscape() throws Exception {
174    builder = new HumanFilterBuilder(null, null, true);
175    Predicate p = new EqualsPredicate(OccurrenceSearchParameter.COUNTRY, Country.AFGHANISTAN.getIso2LetterCode());
176
177    Map<OccurrenceSearchParameter, LinkedList<String>> x = builder.humanFilter(p);
178    assertEquals(1, x.size());
179    assertEquals(1, x.get(OccurrenceSearchParameter.COUNTRY).size());
180    assertEquals("Afghanistan", x.get(OccurrenceSearchParameter.COUNTRY).get(0));
181
182    List<Predicate> ors = Lists.newArrayList();
183    ors.add(new EqualsPredicate(OccurrenceSearchParameter.YEAR, "2000"));
184    ors.add(new EqualsPredicate(OccurrenceSearchParameter.YEAR, "2001"));
185    ors.add(new LessThanOrEqualsPredicate(OccurrenceSearchParameter.YEAR, "1760"));
186    DisjunctionPredicate or = new DisjunctionPredicate(ors);
187
188    List<Predicate> ands = Lists.newArrayList(p, or);
189
190    x = builder.humanFilter(new ConjunctionPredicate(ands));
191    assertEquals(2, x.size());
192    assertEquals(3, x.get(OccurrenceSearchParameter.YEAR).size());
193    assertEquals("&lt;= 1760", x.get(OccurrenceSearchParameter.YEAR).getLast());
194
195    NotPredicate noBirds = new NotPredicate(new EqualsPredicate(OccurrenceSearchParameter.TAXON_KEY, "212"));
196    ands.add(noBirds);
197
198    x = builder.humanFilter(new ConjunctionPredicate(ands));
199    assertEquals(3, x.size());
200    assertEquals(3, x.get(OccurrenceSearchParameter.YEAR).size());
201    assertEquals(1, x.get(OccurrenceSearchParameter.TAXON_KEY).size());
202    assertEquals("not (212)", x.get(OccurrenceSearchParameter.TAXON_KEY).getLast());
203  }

216  @Test
217  public void testRange() throws Exception {
218    List<Predicate> rangeAnd = Lists.newArrayList();
219    rangeAnd.add(new GreaterThanOrEqualsPredicate(OccurrenceSearchParameter.YEAR, "2000"));
220    rangeAnd.add(new LessThanOrEqualsPredicate(OccurrenceSearchParameter.YEAR, "2011"));
221    Predicate range = new ConjunctionPredicate(rangeAnd);
222
223    Map<OccurrenceSearchParameter, LinkedList<String>> x = builder.humanFilter(range);
224    assertEquals(1, x.size());
225    assertEquals(1, x.get(OccurrenceSearchParameter.YEAR).size());
226
227    // now test with other filters combined
228    Predicate eq = new EqualsPredicate(OccurrenceSearchParameter.TAXON_KEY, "212");
229    x = builder.humanFilter(new ConjunctionPredicate(Lists.newArrayList(eq, range)));
230    assertEquals(2, x.size());
231    assertEquals(1, x.get(OccurrenceSearchParameter.TAXON_KEY).size());
232    assertEquals(1, x.get(OccurrenceSearchParameter.YEAR).size());
233  }
