001package org.gbif.api.util.occurrence;
002
003import org.gbif.api.model.occurrence.predicate.ConjunctionPredicate;
004import org.gbif.api.model.occurrence.predicate.DisjunctionPredicate;
005import org.gbif.api.model.occurrence.predicate.EqualsPredicate;
006import org.gbif.api.model.occurrence.predicate.GreaterThanOrEqualsPredicate;
007import org.gbif.api.model.occurrence.predicate.GreaterThanPredicate;
008import org.gbif.api.model.occurrence.predicate.IsNotNullPredicate;
009import org.gbif.api.model.occurrence.predicate.LessThanOrEqualsPredicate;
010import org.gbif.api.model.occurrence.predicate.LessThanPredicate;
011import org.gbif.api.model.occurrence.predicate.NotPredicate;
012import org.gbif.api.model.occurrence.predicate.Predicate;
013import org.gbif.api.model.occurrence.predicate.WithinPredicate;
014import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
015import org.gbif.api.vocabulary.Country;
016
017import java.util.List;
018
019import com.google.common.collect.Lists;
020import org.junit.Test;
021
022import static org.junit.Assert.assertEquals;
023
024public class QueryParameterFilterBuilderTest {
025
026  @Test
027  public void testMultiValues() throws Exception {
028    QueryParameterFilterBuilder builder = new QueryParameterFilterBuilder();
029
030    Predicate p = new EqualsPredicate(OccurrenceSearchParameter.COUNTRY, Country.AFGHANISTAN.getIso2LetterCode());
031
032    String query = builder.queryFilter(p);
033    assertEquals("COUNTRY=AF", query);
034
035    List<Predicate> ors = Lists.newArrayList();
036    ors.add(new EqualsPredicate(OccurrenceSearchParameter.YEAR, "2000"));
037    ors.add(new EqualsPredicate(OccurrenceSearchParameter.YEAR, "2001"));
038    ors.add(new EqualsPredicate(OccurrenceSearchParameter.YEAR, "2002"));
039    DisjunctionPredicate or = new DisjunctionPredicate(ors);
040
041    query = builder.queryFilter(or);
042    assertEquals("YEAR=2000&YEAR=2001&YEAR=2002", query);
043  }
044
045  @Test(expected = IllegalArgumentException.class)
046  public void testNot() throws Exception {
047    QueryParameterFilterBuilder builder = new QueryParameterFilterBuilder();
048    NotPredicate noBirds = new NotPredicate(new EqualsPredicate(OccurrenceSearchParameter.TAXON_KEY, "212"));
049    builder.queryFilter(noBirds);
050  }
051
052  @Test(expected = IllegalArgumentException.class)
053  public void testLess() throws Exception {
054    QueryParameterFilterBuilder builder = new QueryParameterFilterBuilder();
055    builder.queryFilter(new LessThanPredicate(OccurrenceSearchParameter.YEAR, "1900"));
056  }
057
058  @Test(expected = IllegalArgumentException.class)
059  public void testGreater() throws Exception {
060    QueryParameterFilterBuilder builder = new QueryParameterFilterBuilder();
061    builder.queryFilter(new GreaterThanPredicate(OccurrenceSearchParameter.YEAR, "1900"));
062  }
063
064  @Test
065  public void testPolygon() throws Exception {
066    QueryParameterFilterBuilder builder = new QueryParameterFilterBuilder();
067    final String wkt = "POLYGON((30 10,10 20,20 40,40 40,30 10))";
068    String query = builder.queryFilter(new WithinPredicate(wkt));
069    assertEquals("GEOMETRY=30+10%2C10+20%2C20+40%2C40+40%2C30+10", query);
070  }
071
072  @Test
073  public void testRange() throws Exception {
074    QueryParameterFilterBuilder builder = new QueryParameterFilterBuilder();
075
076    List<Predicate> rangeAnd = Lists.newArrayList();
077    rangeAnd.add(new GreaterThanOrEqualsPredicate(OccurrenceSearchParameter.YEAR, "2000"));
078    rangeAnd.add(new LessThanOrEqualsPredicate(OccurrenceSearchParameter.YEAR, "2011"));
079    Predicate range = new ConjunctionPredicate(rangeAnd);
080
081    String query = builder.queryFilter(range);
082    assertEquals("YEAR=2000%2C2011", query);
083  }
084
085  @Test
086  public void testIsNotNull() throws Exception {
087    QueryParameterFilterBuilder builder = new QueryParameterFilterBuilder();
088    String query = builder.queryFilter(new IsNotNullPredicate(OccurrenceSearchParameter.MEDIA_TYPE));
089    assertEquals("MEDIA_TYPE=*", query);
090  }
091}
