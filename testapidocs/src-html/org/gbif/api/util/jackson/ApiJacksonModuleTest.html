001package org.gbif.api.util.jackson;
002
003import org.gbif.dwc.terms.DwcTerm;
004import org.gbif.dwc.terms.GbifTerm;
005import org.gbif.dwc.terms.Term;
006
007import java.io.IOException;
008
009import org.codehaus.jackson.map.ObjectMapper;
010import org.junit.Test;
011
012import static org.junit.Assert.assertEquals;
013
014public class ApiJacksonModuleTest {
015
016  @Test
017  public void testTermSerde() throws IOException {
018    ObjectMapper mapper = new ObjectMapper();
019    mapper.registerModule(new ApiJacksonModule());
020
021    for (Term t : DwcTerm.values()) {
022      String json = mapper.writeValueAsString(t);
023      assertEquals(t, mapper.readValue(json, Term.class));
024    }
025
026    for (Term t : GbifTerm.values()) {
027      String json = mapper.writeValueAsString(t);
028      assertEquals(t, mapper.readValue(json, Term.class));
029    }
030  }
031}
