001package org.gbif.api.util;
002
003import java.util.Arrays;
004import java.util.Collection;
005
006import com.google.common.collect.Range;
007import org.junit.Test;
008import org.junit.runner.RunWith;
009import org.junit.runners.Parameterized;
010
011import static org.junit.Assert.assertEquals;
012import static org.junit.Assert.assertNull;
013import static org.junit.Assert.assertTrue;
014import static org.junit.Assert.fail;
015
016@RunWith(Parameterized.class)
017public class SearchTypeValidatorIntegerRangeTest {
018
019  private final String arg;
020  private final Integer start;
021  private final Integer end;
022
023  public SearchTypeValidatorIntegerRangeTest(String arg, Integer start, Integer end) {
024    this.arg = arg;
025    this.start = start;
026    this.end = end;
027  }
028
029  @Parameterized.Parameters
030  public static Collection<Object[]> data() {
031    Object[][] data = {
032      {"10", null, null},
033      {"10,20", 10, 20},
034      {"10.1,20.0", null, null},
035      {"-1 , 0", -1, 0},
036      {"*,20", null, 20},
037      {"10, *", 10, null},
038      {"peter", null, null},
039      {"Puma concolor, 1882", null, null},
040    };
041    return Arrays.asList(data);
042  };
043
044
045  @Test
046  public void testRange() throws Exception {
047    try {
048      Range<Integer> range = SearchTypeValidator.parseIntegerRange(arg);
049      if (start == null && end == null) {
050        fail(arg + " supposed to be an invalid range");
051      } else {
052        // test parse result
053        if (range.hasLowerBound()) {
054          assertEquals(start, range.lowerEndpoint());
055        } else {
056          assertNull(start);
057        }
058
059        if (range.hasUpperBound()) {
060          assertEquals(end, range.upperEndpoint());
061        } else {
062          assertNull(end);
063        }
064
065        assertTrue(SearchTypeValidator.isRange(arg));
066      }
067
068    } catch (IllegalArgumentException e) {
069      // expected?
070      if (start != null || end != null) {
071        fail(arg + " supposed to be a valid range");
072      }
073    }
074  }
075
076}
