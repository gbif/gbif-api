001package org.gbif.api.util;
002
003import org.gbif.api.vocabulary.IdentifierType;
004
005import java.util.Arrays;
006import java.util.Collection;
007
008import org.junit.Test;
009import org.junit.runner.RunWith;
010import org.junit.runners.Parameterized;
011
012import static org.junit.Assert.assertEquals;
013import static org.junit.Assert.assertNull;
014
015@RunWith(value = Parameterized.class)
016public class IdentifierUtilsTest {
017
018  private String expected;
019  private String identifier;
020  private IdentifierType type;
021
022  /**
023   * A test is run for each set of parameters. To add a new test, just add a new set of parameters.
024   */
025  @Parameterized.Parameters
026  public static Collection<Object[]> getTestParameters() {
027    return Arrays.asList(new Object[][] { {null, "49c5b4ac-e3bf-401b-94b1-c94a2ad5c8d6", IdentifierType.UUID},
028      {"http://data.gbif.org/datasets/resource/ds1", "ds1", IdentifierType.GBIF_PORTAL}, {null, "local_1:dx:4", IdentifierType.UNKNOWN},
029      {"http://en.wikipedia.org/wiki/Handle_1", "http://en.wikipedia.org/wiki/Handle_1", IdentifierType.HANDLER},
030      {"urn:ds:acns:1", "urn:ds:acns:1", IdentifierType.URI},
031      {"http://ipt.gbif.org/resource.do?r=ds1", "http://ipt.gbif.org/resource.do?r=ds1", IdentifierType.URL},
032      {"ftp://ftp.gbif.org", "ftp://ftp.gbif.org", IdentifierType.FTP},
033      {"http://dx.doi.org/10.1016/S1097-2765(03)00225-9", "10.1016/S1097-2765(03)00225-9", IdentifierType.DOI},
034      {"http://lsid.tdwg.org/summary/132187", "132187", IdentifierType.LSID},});
035  }
036
037  public IdentifierUtilsTest(String expected, String identifier, IdentifierType type) {
038    this.expected = expected;
039    this.identifier = identifier;
040    this.type = type;
041  }
042
043  @Test
044  public void testGetIdentifierLink() {
045    assertEquals(expected, IdentifierUtils.getIdentifierLink(identifier, type));
046  }
047
048  @Test
049  public void testGetIdentifierLinkWithoutIdentifer() {
050    assertNull(IdentifierUtils.getIdentifierLink(null, type));
051  }
052
053  @Test
054  public void testGetIdentifierLinkWithoutType() {
055    assertNull(IdentifierUtils.getIdentifierLink(identifier, null));
056  }
057
058}
