001package org.gbif.api.util;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006import static org.junit.Assert.assertNull;
007
008public class UnicodeUtilsTest {
009
010    @Test
011    public void testAscii() throws Exception {
012        String x  = "À Á Â Ã Ä Å Ç È É Ê Ë Ì Í Î Ï Ð Ñ Ò Ó Ô Õ Ö Ø Ù Ú Û Ü Ý Ÿ à á â ã ä å ç è é ê ë ì í î ï ð ñ ò ó ô õ ö ø ù ú û ü ý ÿ";
013        String x2 = "A A A A A A C E E E E I I I I D N O O O O O O U U U U Y Y a a a a a a c e e e e i i i i d n o o o o o o u u u u y y";
014        assertEquals(x2, UnicodeUtils.ascii(x));
015        assertNull(UnicodeUtils.ascii(null));
016    }
017
018    @Test
019    public void testDecompose() throws Exception {
020        String x  = "Æ œ Œ Ĳ ĳ ǈ ǉ \u0238 \u0239 ß ﬆ ﬅ ﬀ ﬁ ﬂ ﬃ ﬄ";
021        String x2 = "Ae oe Oe Ij ij Lj lj db qp ss st ft ff fi fl ffi ffl";
022        assertEquals(x2, UnicodeUtils.decompose(x));
023        assertNull(UnicodeUtils.decompose(null));
024        assertEquals("fjaelje", UnicodeUtils.decompose("fjæǉe"));
025    }
026
027}
