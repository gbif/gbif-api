001package org.gbif.api.util;
002
003import java.util.Calendar;
004import java.util.Date;
005
006import org.junit.Assert;
007import org.junit.Test;
008
009import static org.junit.Assert.assertEquals;
010
011/**
012 * Test case for class OccurrenceDateParsingUtils.
013 */
014public class IsoDateParsingUtilsTest {
015
016  /**
017   * Test case for method {@link IsoDateParsingUtils#parseDateRange(String)}.
018   * Upper limit has an invalid date format.
019   */
020  @Test(expected = IllegalArgumentException.class)
021  public void isParseWrongDateRangeFormatTest() {
022    IsoDateParsingUtils.parseDateRange("2010-01,2000-10-33 00:00:00");
023  }
024
025  /**
026   * Test case for method {@link IsoDateParsingUtils#parseDateRange(String)}.
027   * Lower limit > upper limit.
028   */
029  @Test(expected = IllegalArgumentException.class)
030  public void isParseWrongDateRangeTest() {
031    IsoDateParsingUtils.parseDateRange("2010-01,2000-10");
032  }
033
034  /**
035   * Test case for method {@link IsoDateParsingUtils#parseDate(String)}.
036   */
037  @Test(expected = IllegalArgumentException.class)
038  public void isParseWrongDateTest() {
039    Date d = IsoDateParsingUtils.parseDate("2010-01-01-10");
040    IsoDateParsingUtils.parseDate("2010-01-01-10");
041  }
042
043  @Test(expected = IllegalArgumentException.class)
044  public void isParseNonExistingDateTest() {
045    IsoDateParsingUtils.parseDate("2010-21-01");
046  }
047
048  @Test(expected = IllegalArgumentException.class)
049  public void isParseNonExistingDateTest2() {
050    IsoDateParsingUtils.parseDate("2010-02-29");
051  }
052
053  @Test(expected = IllegalArgumentException.class)
054  public void isParseWrongDateWithTrailingStringTest() {
055    IsoDateParsingUtils.parseDate("2010 year");
056  }
057
058  @Test(expected = IllegalArgumentException.class)
059  public void isParseWrongYearTest() {
060    IsoDateParsingUtils.parseDate("10");
061  }
062
063  @Test(expected = IllegalArgumentException.class)
064  public void isParseWrongYearTest2() {
065    IsoDateParsingUtils.parseDate("10-10");
066  }
067
068  /**
069   * Test case for method {@link IsoDateParsingUtils#parseDate(String)}.
070   */
071  @Test(expected = IllegalArgumentException.class)
072  public void isParseWrongDateTimTest() {
073    IsoDateParsingUtils.parseDate("2010-01-01 22:00:00");
074  }
075
076  /**
077   * Test case for method {@link IsoDateParsingUtils#parseDateRange(String)}.
078   */
079  @Test
080  public void parseDateRangeTest() {
081    Assert.assertNotNull(IsoDateParsingUtils.parseDateRange("2000-01,2010"));
082    Assert.assertNotNull(IsoDateParsingUtils.parseDateRange("1900-09-23,2000"));
083    Assert.assertNotNull(IsoDateParsingUtils.parseDateRange("1999-10,1999-12"));
084    Assert.assertNotNull(IsoDateParsingUtils.parseDateRange("1999-10,*"));
085    Assert.assertNotNull(IsoDateParsingUtils.parseDateRange("*,1999-12"));
086    Assert.assertNotNull(IsoDateParsingUtils.parseDateRange("2000-10,*"));
087  }
088
089
090  /**
091   * Test case for method {@link IsoDateParsingUtils#parseDate(String)}.
092   */
093  @Test
094  public void parseDateTest() {
095    Assert.assertNotNull(IsoDateParsingUtils.parseDate("2000-01"));
096    Assert.assertNotNull(IsoDateParsingUtils.parseDate("2000"));
097    Assert.assertNotNull(IsoDateParsingUtils.parseDate("2000-01-01"));
098  }
099
100
101  /**
102   * Test case for method {@link IsoDateParsingUtils#toLastDayOfMonth(Date)}.
103   */
104  @Test
105  public void toLastDayOfMonthTest() {
106    Date lowerDate = IsoDateParsingUtils.parseDate("2000-01-01");
107    Calendar lastDayOfMonthCal = Calendar.getInstance();
108    lastDayOfMonthCal.setTime(IsoDateParsingUtils.toLastDayOfMonth(lowerDate));
109    assertEquals(2000, lastDayOfMonthCal.get(Calendar.YEAR));
110    assertEquals(0, lastDayOfMonthCal.get(Calendar.MONTH)); // January is 0, months start at index 0
111    assertEquals(31, lastDayOfMonthCal.get(Calendar.DAY_OF_MONTH));
112  }
113
114
115  /**
116   * Test case for method {@link IsoDateParsingUtils#toLastDayOfYear(Date)}.
117   */
118  @Test
119  public void toLastDayOfYearTest() {
120    Date lowerDate = IsoDateParsingUtils.parseDate("2000-01-01");
121    Calendar lastDayOfMonthCal = Calendar.getInstance();
122    lastDayOfMonthCal.setTime(IsoDateParsingUtils.toLastDayOfYear(lowerDate));
123    assertEquals(2000, lastDayOfMonthCal.get(Calendar.YEAR));
124    assertEquals(11, lastDayOfMonthCal.get(Calendar.MONTH)); // December is 11, months start at index 0
125    assertEquals(31, lastDayOfMonthCal.get(Calendar.DAY_OF_MONTH));
126  }
