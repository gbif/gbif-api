164  @ParameterizedTest
165  @MethodSource("data")
166  public void testValidateFine(SearchParameter parameter, String arg, boolean valid, boolean range) {
167    try {
168      SearchTypeValidator.validate(parameter, arg);
169      if (!valid) {
170        fail(arg + " supposed to be an invalid value for parameter " + parameter);
171      }
172    } catch (IllegalArgumentException e) {
173      if (valid) {
174        fail(arg + " supposed to be a valid value for parameter " + parameter + " " + e.getMessage());
175      }
176    }
177
178    // test isRange
179    assertEquals(range, SearchTypeValidator.isRange(arg), "Wrong isRange parsing of value " + arg);
180  }
