154  @Test
155  public void testValidateFine() throws Exception {
156    try {
157      SearchTypeValidator.validate(parameter, arg);
158      if (!valid) {
159        fail(arg + " supposed to be an invalid value for parameter " + parameter);
160      }
161    } catch (IllegalArgumentException e) {
162      if (valid) {
163        fail(arg + " supposed to be a valid value for parameter " + parameter);
164      }
165    }
166
167    // test isRange
168    assertEquals("Wrong isRange parsing of value " + arg, range, SearchTypeValidator.isRange(arg));
169  }
