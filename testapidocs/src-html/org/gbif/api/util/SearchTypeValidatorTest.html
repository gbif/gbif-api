163  @Test
164  public void testValidateFine() throws Exception {
165    try {
166      SearchTypeValidator.validate(parameter, arg);
167      if (!valid) {
168        fail(arg + " supposed to be an invalid value for parameter " + parameter);
169      }
170    } catch (IllegalArgumentException e) {
171      if (valid) {
172        fail(arg + " supposed to be a valid value for parameter " + parameter);
173      }
174    }
175
176    // test isRange
177    assertEquals("Wrong isRange parsing of value " + arg, range, SearchTypeValidator.isRange(arg));
178  }
