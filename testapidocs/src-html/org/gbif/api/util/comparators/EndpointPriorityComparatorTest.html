001package org.gbif.api.util.comparators;
002
003import org.gbif.api.model.registry.Endpoint;
004import org.gbif.api.vocabulary.EndpointType;
005
006import org.junit.Test;
007
008import static org.junit.Assert.assertEquals;
009import static org.junit.Assert.assertTrue;
010
011public class EndpointPriorityComparatorTest {
012
013  private static final EndpointPriorityComparator COMP = new EndpointPriorityComparator();
014
015  @Test
016  public void testSuccessfulSimpleComparisons() {
017    Endpoint e1 = new Endpoint();
018    Endpoint e2 = new Endpoint();
019
020    e1.setType(EndpointType.TAPIR);
021    e2.setType(EndpointType.BIOCASE);
022    assertTrue(COMP.compare(e1, e2) > 0);
023
024    e1.setType(EndpointType.BIOCASE);
025    e2.setType(EndpointType.BIOCASE);
026    assertEquals(0, COMP.compare(e1, e2));
027
028    e1.setType(EndpointType.BIOCASE);
029    e2.setType(EndpointType.TAPIR);
030    assertTrue(COMP.compare(e1, e2) < 0);
031
032    e1.setType(EndpointType.BIOCASE);
033    e2.setType(EndpointType.DIGIR_MANIS);
034    assertTrue(COMP.compare(e1, e2) > 0);
035
036    e1.setType(EndpointType.DWC_ARCHIVE);
037    e2.setType(EndpointType.TAPIR);
038    assertTrue(COMP.compare(e1, e2) > 0);
039
040    e1.setType(EndpointType.BIOCASE_XML_ARCHIVE);
041    e2.setType(EndpointType.BIOCASE);
042    assertTrue(COMP.compare(e1, e2) > 0);
043
044    e1.setType(EndpointType.DWC_ARCHIVE);
045    e2.setType(EndpointType.BIOCASE_XML_ARCHIVE);
046    assertTrue(COMP.compare(e1, e2) > 0);
047  }
048
049  @Test
050  public void testPriorityComparisons() {
051    Endpoint e1 = new Endpoint();
052    Endpoint e2 = new Endpoint();
053
054    e1.setType(EndpointType.TAPIR);
055    e2.setType(EndpointType.BIOCASE);
056    assertTrue(COMP.compare(e1, e2) > 0);
057
058    e1.setType(EndpointType.DIGIR_MANIS);
059    e2.setType(EndpointType.DIGIR_MANIS);
060    assertEquals(0, COMP.compare(e1, e2));
061  }
062
063  @Test(expected = ClassCastException.class)
064  public void testInvalidComparison1() {
065    Endpoint e1 = new Endpoint();
066    Endpoint e2 = new Endpoint();
067
068    e1.setType(EndpointType.TAPIR);
069    e2.setType(EndpointType.WMS);
070    COMP.compare(e1, e2);
071  }
072
073  @Test(expected = ClassCastException.class)
074  public void testInvalidComparison2() {
075    Endpoint e1 = new Endpoint();
076    Endpoint e2 = new Endpoint();
077
078    e1.setType(EndpointType.WMS);
079    e2.setType(EndpointType.TAPIR);
080    COMP.compare(e1, e2);
081  }
082
083
084}
