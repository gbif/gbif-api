016package org.gbif.api.util;
017
018import org.gbif.api.vocabulary.ContactType;
019import org.gbif.api.vocabulary.Country;
020import org.gbif.api.vocabulary.EndpointType;
021import org.gbif.api.vocabulary.IdentifierType;
022import org.gbif.api.vocabulary.TechnicalInstallationType;
023
024import java.util.Optional;
025
026import org.junit.jupiter.api.Test;
027
028import static org.junit.jupiter.api.Assertions.assertEquals;
029import static org.junit.jupiter.api.Assertions.assertThrows;
030
031public class VocabularyUtilsTest {
032
033  @Test
034  public void testParseContactType() {
035    assertEquals(ContactType.AUTHOR, VocabularyUtils.parseContactType("author"));
036    assertEquals(ContactType.ADMINISTRATIVE_POINT_OF_CONTACT,
037      VocabularyUtils.parseContactType("ADMINISTRATIVE POINT_of_CONTACT"));
038    assertThrows(IllegalArgumentException.class, () -> VocabularyUtils.parseContactType("bad"));
039  }
040
041  @Test
042  public void testParseEndpointType() {
043    assertEquals(EndpointType.TAPIR, VocabularyUtils.parseEndpointType("Tapir"));
044    assertEquals(EndpointType.DWC_ARCHIVE, VocabularyUtils.parseEndpointType("DWC_ARCHIVE"));
045    assertThrows(IllegalArgumentException.class, () -> VocabularyUtils.parseEndpointType("bad"));
046  }
047
048  @Test
049  public void testParseIdentifierType() {
050    assertEquals(IdentifierType.LSID, VocabularyUtils.parseIdentifierType("lsid"));
051    assertEquals(IdentifierType.DOI, VocabularyUtils.parseIdentifierType("d.o.i."));
052    assertThrows(IllegalArgumentException.class, () -> VocabularyUtils.parseIdentifierType("bad"));
053  }
054
055  @Deprecated
056  @Test
057  public void testParseTechnicalInstallationType() {
058    assertEquals(TechnicalInstallationType.IPT_INSTALLATION,
059      VocabularyUtils.parseTechnicalInstallationType("ipt installation"));
060    assertThrows(IllegalArgumentException.class, () -> VocabularyUtils.parseTechnicalInstallationType("bad"));
061  }
062
063  @Test
064  public void testReflectionLookup() {
065    assertEquals(Country.class, VocabularyUtils.lookupVocabulary(Country.class.getName()));
066  }
067
068  @Test
069  public void testReflectionLookupError() {
070    assertThrows(IllegalArgumentException.class, () -> VocabularyUtils.lookupVocabulary("WTFMate"));
071  }
072
073  @Test
074  public void testLookupUsingOptional() {
075    assertEquals(ContactType.AUTHOR, VocabularyUtils.lookup("author", ContactType.class).get());
076
077    //Optional.absent() should be returned for all the following values
078    assertEquals(Optional.empty(), VocabularyUtils.lookup("thor", ContactType.class));
079    assertEquals(Optional.empty(), VocabularyUtils.lookup("", ContactType.class));
080    assertEquals(Optional.empty(), VocabularyUtils.lookup(null, ContactType.class));
081  }
082}
