013package org.gbif.api.util;
014
015import org.gbif.api.vocabulary.Country;
016import org.gbif.api.vocabulary.Kingdom;
017import org.gbif.api.vocabulary.TechnicalInstallationType;
018import org.gbif.api.vocabulary.ContactType;
019import org.gbif.api.vocabulary.EndpointType;
020import org.gbif.api.vocabulary.IdentifierType;
021
022import java.util.Map;
023
024import com.google.common.base.Optional;
025import org.junit.Test;
026
027import static org.junit.Assert.assertEquals;
028import static org.junit.Assert.assertTrue;
029
030public class VocabularyUtilsTest {
031
032  @Test(expected = IllegalArgumentException.class)
033  public void testParseContactType() throws Exception {
034    assertEquals(ContactType.AUTHOR, VocabularyUtils.parseContactType("author"));
035    assertEquals(ContactType.ADMINISTRATIVE_POINT_OF_CONTACT,
036      VocabularyUtils.parseContactType("ADMINISTRATIVE POINT_of_CONTACT"));
037    VocabularyUtils.parseContactType("bad");
038  }
039
040  @Test(expected = IllegalArgumentException.class)
041  public void testParseEndpointType() throws Exception {
042    assertEquals(EndpointType.TAPIR, VocabularyUtils.parseEndpointType("Tapir"));
043    assertEquals(EndpointType.DWC_ARCHIVE, VocabularyUtils.parseEndpointType("DWC_ARCHIVE"));
044    VocabularyUtils.parseEndpointType("bad");
045  }
046
047  @Test(expected = IllegalArgumentException.class)
048  public void testParseIdentifierType() throws Exception {
049    assertEquals(IdentifierType.LSID, VocabularyUtils.parseIdentifierType("lsid"));
050    assertEquals(IdentifierType.DOI, VocabularyUtils.parseIdentifierType("d.o.i."));
051    VocabularyUtils.parseIdentifierType("bad");
052  }
053
054  @Test(expected = IllegalArgumentException.class)
055  public void testParseTechnicalInstallationType() throws Exception {
056    assertEquals(TechnicalInstallationType.IPT_INSTALLATION,
057      VocabularyUtils.parseTechnicalInstallationType("ipt installation"));
058    VocabularyUtils.parseTechnicalInstallationType("bad");
059  }
060
061  @Test
062  public void testReflectionLookup() {
063    assertEquals(Country.class, VocabularyUtils.lookupVocabulary(Country.class.getName()));
064  }
065
066  @Test(expected = IllegalArgumentException.class)
067  public void testReflectionLookupError() {
068    VocabularyUtils.lookupVocabulary("WTFMate");
069  }
070
071  @Test
072  public void testLookupUsingOptional() {
073    assertEquals(ContactType.AUTHOR, VocabularyUtils.lookup("author", ContactType.class).get());
074
075    //Optional.absent() should be returned for all the following values
076    assertEquals(Optional.absent(), VocabularyUtils.lookup("thor", ContactType.class));
077    assertEquals(Optional.absent(), VocabularyUtils.lookup("", ContactType.class));
078    assertEquals(Optional.absent(), VocabularyUtils.lookup(null, ContactType.class));
079  }
080
081  @Test
082  public void testListEnumerations(){
083    Map<String, Enum<?>[]> enums = VocabularyUtils.listEnumerations(Kingdom.class.getPackage().getName());
084    assertTrue(enums.containsKey(Kingdom.class.getSimpleName()));
085
086    // the current package should not include any enumerations
087    enums = VocabularyUtils.listEnumerations(getClass().getPackage().getName());
088    assertTrue(enums.isEmpty());
089
090    // a non-existing package should return an empty map
091    enums = VocabularyUtils.listEnumerations("a.b.c");
092    assertTrue(enums.isEmpty());
093  }
094}
