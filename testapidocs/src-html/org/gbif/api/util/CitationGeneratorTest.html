228        .getContacts()
229        .add(createContact("John D.", "Doe", "Awesome Organization", ContactType.ORIGINATOR));
230    // author with incomplete name
231    dataset.getContacts().add(createContact("Programmer", "Last", ContactType.PROGRAMMER));

237    assertEquals(
238        1, CitationGenerator.generateAuthorsName(getAuthors(dataset.getContacts())).size());

241  @Test
242  public void testRepeatedAuthor() {
243    Organization org = new Organization();
244    org.setTitle("Cited Organization");
245
246    Dataset dataset = getTestDatasetObject();
247    Contact contact1 = createContact("John D.", "Doe", ContactType.ORIGINATOR);
248    contact1.setUserId(Collections.singletonList("user1"));
249
250    Contact contact2 = createContact("John D.", "Doe", ContactType.METADATA_AUTHOR);
251    contact2.setUserId(Arrays.asList("user1", "user2"));
252
253    dataset.getContacts().add(contact1);
254    dataset.getContacts().add(contact2);
255
256    List<CitationContact> authors = getAuthors(dataset.getContacts());
257
258    // Only one author added
259    assertEquals(1, authors.size());
260
261    // The authors keeps the 2 roles
262    assertTrue(
263        authors
264            .get(0)
265            .getRoles()
266            .containsAll(EnumSet.of(ContactType.ORIGINATOR, ContactType.METADATA_AUTHOR)));
267
268    // The author has 2 users
269    assertTrue(authors.get(0).getUserId().containsAll(Arrays.asList("user1", "user2")));
270
271    // Repeated user is not added twice
272    assertEquals(authors.get(0).getUserId().size(), 2);
273
274    // we can only generate the name for one of them
275    assertEquals(
276        1, CitationGenerator.generateAuthorsName(getAuthors(dataset.getContacts())).size());
277  }

279  private Dataset getTestDatasetObject() {
280    Dataset dataset = new Dataset();
281    dataset.setTitle("Dataset to be cited");
282    dataset.setVersion("2.1");
283    dataset.setDoi(new DOI(TEST_PREFIX + "/abcd"));
284    dataset.setPubDate(
285        new Date(
286            LocalDate.of(2009, 2, 8).atStartOfDay(ZoneId.of("UTC")).toInstant().toEpochMilli()));
287
288    dataset.setType(DatasetType.SAMPLING_EVENT);
289
290    return dataset;
291  }
292
293  private Contact createContact(String firstName, String lastName, ContactType ct) {
294    return createContact(firstName, lastName, null, ct);
295  }
296
297  private Contact createContact(
298      String firstName, String lastName, String organization, ContactType ct) {
299    Contact c = new Contact();
300    c.setFirstName(firstName);
301    c.setLastName(lastName);
302    c.setOrganization(organization);
303    c.setType(ct);
304    return c;
305  }
