016  @Test
017  public void testTerm() {
018    assertEquals("taxonKey", GbifTerm.taxonKey.simpleName());
019    assertEquals("http://rs.gbif.org/terms/1.0/taxonKey", GbifTerm.taxonKey.qualifiedName());
020
021    assertEquals("lastInterpreted", GbifTerm.lastInterpreted.simpleName());
022    assertEquals("http://rs.gbif.org/terms/1.0/lastInterpreted", GbifTerm.lastInterpreted.qualifiedName());
023  }
024
025  @Test
026  public void testIsClass() {
027    assertTrue(GbifTerm.VernacularName.isClass());
028    assertTrue(GbifTerm.Distribution.isClass());
029    assertFalse(GbifTerm.datasetKey.isClass());
030  }
031
032  @Test
033  public void testNumberOfGroups() {
034    assertEquals(9, GbifTerm.GROUPS.length);
035  }
036
037  /**
038   * Test each group contains the expected number of terms, and contains no duplicates.
039   */
040  @Test
041  public void testListByGroup() {
042    List<GbifTerm> datasetTerms = GbifTerm.listByGroup(GbifTerm.GROUP_DATASET);
043    assertEquals(2, datasetTerms.size());
044    assertEquals(2, new HashSet<GbifTerm>(datasetTerms).size());
045
046    List<GbifTerm> occurrenceTerms = GbifTerm.listByGroup(DwcTerm.GROUP_OCCURRENCE);
047    assertEquals(13, occurrenceTerms.size());
048    assertEquals(13, new HashSet<GbifTerm>(occurrenceTerms).size());
049
050    List<GbifTerm> locationTerms = GbifTerm.listByGroup(DwcTerm.GROUP_LOCATION);
051    assertEquals(6, locationTerms.size());
052    assertEquals(6, new HashSet<GbifTerm>(locationTerms).size());
053
054    List<GbifTerm> rowTerms = GbifTerm.listByGroup(GbifTerm.GROUP_ROW_TYPE);
055    assertEquals(9, rowTerms.size());
056    assertEquals(9, new HashSet<GbifTerm>(rowTerms).size());
057
058    List<GbifTerm> distributionTerms = GbifTerm.listByGroup(GbifTerm.GROUP_SPECIES_DISTRIBUTION_EXTENSION);
059    assertEquals(2, distributionTerms.size());
060    assertEquals(2, new HashSet<GbifTerm>(distributionTerms).size());
061
062    List<GbifTerm> profileTerms = GbifTerm.listByGroup(GbifTerm.GROUP_SPECIES_PROFILE_EXTENSION);
063    assertEquals(10, profileTerms.size());
064    assertEquals(10, new HashSet<GbifTerm>(profileTerms).size());
065
066    List<GbifTerm> taxonTerms = GbifTerm.listByGroup(DwcTerm.GROUP_TAXON);
067    assertEquals(16, taxonTerms.size());
068    assertEquals(16, new HashSet<GbifTerm>(taxonTerms).size());
069
070    List<GbifTerm> crawlingTerms = GbifTerm.listByGroup(GbifTerm.GROUP_CRAWLING);
071    assertEquals(3, crawlingTerms.size());
072    assertEquals(3, new HashSet<GbifTerm>(crawlingTerms).size());
073
074    List<GbifTerm> vernacularTerms = GbifTerm.listByGroup(GbifTerm.GROUP_VERNACULAR_NAME_EXTENSION);
075    assertEquals(3, vernacularTerms.size());
076    assertEquals(3, new HashSet<GbifTerm>(vernacularTerms).size());
077  }
078
079  @Test
080  public void testGroupCoverage() {
081    HashSet<GbifTerm> arrayTerms = new HashSet<GbifTerm>();
082    for (GbifTerm t : GbifTerm.TAXONOMIC_TERMS) {
083      arrayTerms.add(t);
084      assertFalse(t.isClass());
085      assertEquals(DwcTerm.GROUP_TAXON, t.getGroup());
086    }
087
088    for (GbifTerm t : GbifTerm.listByGroup(DwcTerm.GROUP_TAXON)) {
089      if (!arrayTerms.contains(t)) {
090        assertTrue("Missing taxonomic term in GbifTerm.TAXONOMIC_TERMS: " + t.qualifiedName(), arrayTerms.contains(t));
091      }
092    }
093  }
094
095  @Test
096  public void testDeprecated() {
097    assertFalse(GbifTerm.gbifID.isDeprecated());
098    assertTrue(GbifTerm.coordinateAccuracy.isDeprecated());
099  }
