001package org.gbif.dwc.terms;
002
003import org.junit.Test;
004
005import java.net.URI;
006import java.util.HashSet;
007import java.util.Set;
008
009import static org.junit.Assert.assertEquals;
010import static org.junit.Assert.assertNotEquals;
011
012/**
013 *
014 */
015public class UnknownTermTest {
016
017  @Test
018  public void testBuild() throws Exception {
019    UnknownTerm.build("http://me.com/#me");
020    UnknownTerm.build("http://me.com/me");
021    UnknownTerm.build("http://me.org/me");
022
023
024    assertEquals(UnknownTerm.build("http://me.org/me"), UnknownTerm.build("http://me.org/me"));
025    assertEquals(UnknownTerm.build("me"), UnknownTerm.build("http://unknown.org/me"));
026  }
027
028  @Test
029  public void testNamespace() throws Exception {
030    assertEquals(URI.create("http://me.com"), UnknownTerm.build("http://me.com/me").namespace());
031    assertEquals(URI.create("https://me.com"), UnknownTerm.build("https://me.com/me").namespace());
032    assertEquals(URI.create("http://unknown.org"), UnknownTerm.build("me").namespace());
033  }
034
035  @Test(expected = IllegalArgumentException.class)
036  public void testBuildError() {
037    UnknownTerm.build("http://me.com/");
038  }
039
040  @Test
041  public void testBuildError2() {
042    Term u = UnknownTerm.build("gbif.org/verbatimLabel");
043    assertEquals("http://unknown.org/gbif.org/verbatimLabel", u.qualifiedName());
044    assertEquals("verbatimLabel", u.simpleName());
045  }
046
047  @Test
048  public void testPrefixedName() {
049    assertEquals("http://unknown.org/gbif/verbatimLabel", UnknownTerm.build("gbif:verbatimLabel").qualifiedName());
050  }
051
052  @Test
053  public void testQualifiedName() throws Exception {
054    Term me = UnknownTerm.build("http://me.com/#me");
055    assertEquals("http://me.com/#me", me.qualifiedName());
056
057    me = UnknownTerm.build("http://me.com/me");
058    assertEquals("http://me.com/me", me.qualifiedName());
059
060    me = UnknownTerm.build("http://me.org/me");
061    assertEquals("http://me.org/me", me.qualifiedName());
062  }
063
064  @Test
065  public void testSimpleName() throws Exception {
066    Term me1 = UnknownTerm.build("http://me.com/#me");
067    Term me2 = UnknownTerm.build("http://me.com/me");
068    Term me3 = UnknownTerm.build("http://me.org/me");
069
070    assertEquals("me", me1.simpleName());
071    assertEquals("me", me2.simpleName());
072    assertEquals("me", me3.simpleName());
073  }
074
075  @Test
076  public void testEquals() throws Exception {
077    Term me1 = UnknownTerm.build("http://me.com/#me");
078    Term me2 = UnknownTerm.build("http://me.com/me");
079    Term me3 = UnknownTerm.build("http://me.org/me");
080
081    assertNotEquals(me1, me2);
082    assertNotEquals(me1, me3);
083    assertNotEquals(me2, me3);
084
085    // only qualified names are relevant when deciding if a term is equal
086    Term me4 = UnknownTerm.build("http://me.org/me", "oscar");
087    assertNotEquals(me1, me4);
088    assertNotEquals(me2, me4);
089    assertEquals(me3, me4);
090
091  }
092
093  @Test
094  public void testSet() throws Exception {
095    Set<Term> terms = new HashSet<Term>();
096    terms.add(UnknownTerm.build("http://me.com/#me"));
097    terms.add(UnknownTerm.build("http://me.com/me"));
098    terms.add(UnknownTerm.build("http://me.org/me"));
099    terms.add(UnknownTerm.build("http://me.org/me", "oscar"));
100    terms.add(DcTerm.title);
101    terms.add(DcElement.title);
102    terms.add(DwcTerm.bed);
103
104    assertEquals(6, terms.size());
105  }
106}
