001package org.gbif.dwc.terms;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006import static org.junit.Assert.assertFalse;
007
008
009/**
010 * Generic tests for all Term implementing classes.
011 */
012public abstract class TermBaseTest<T extends Term> {
013
014  private static final TermFactory TERM_FACTORY = TermFactory.instance();
015
016  private final Class<T> clazz;
017  private final T[] values;
018  private final String[] prefixes;
019  private final String[] forbiddenChars;
020
021  public TermBaseTest(Class<T> clazz, String[] prefixes) {
022    this(clazz, prefixes, new String[]{"_","-"});
023  }
024
025  public TermBaseTest(Class<T> clazz, String[] prefixes, String[] forbiddenChars) {
026    this.clazz = clazz;
027    this.prefixes = prefixes;
028    this.forbiddenChars = forbiddenChars;
029    values = clazz.getEnumConstants();
030  }
031
032  @Test
033  public void testNames() throws Exception {
034    for (T t : values) {
035      for (String c : forbiddenChars) {
036        assertFalse("Term contains forbidden character " + c + " : " + t.simpleName(), t.simpleName().contains(c));
037        assertFalse("Term contains forbidden character " + c + " : " + t.qualifiedName(), t.qualifiedName().contains(c));
038      }
039    }
040  }
041
042  /**
043   * Test that iterates over all term values and uses the simple name alone as input to the factory and then verifies
044   * the Term returned is the same as the original one.
045   */
046  @Test
047  public void testTermEquality() throws Exception {
048    for (T t : values) {
049      Term found = TERM_FACTORY.findTerm(t.simpleName());
050      assertEquals(t, found);
051    }
052  }
053
054  @Test
055  public void testFindQualifiedTerm() {
056    for (T t : values) {
057      Term found = TERM_FACTORY.findTerm(t.qualifiedName());
058      assertEquals(t, found);
059    }
060  }
061
062  @Test
063  public void testPrefixedTerms() {
064    for (T t : values) {
065      for (String pre : prefixes) {
066        Term found = TERM_FACTORY.findTerm(pre+t.simpleName());
067        assertEquals(t, found);
068      }
069    }
070  }
071}
