001package org.gbif.dwc.terms;
002
003import java.util.ArrayList;
004import java.util.Collections;
005import java.util.List;
006
007import org.junit.Test;
008
009import static org.junit.Assert.assertEquals;
010
011public class TermComparatorTest {
012
013  @Test
014  public void testCompare() throws Exception {
015    TermComparator comp = new TermComparator();
016
017    List<Term> terms = new ArrayList<Term>();
018    terms.add(DwcTerm.basisOfRecord);
019    terms.add(DcTerm.title);
020    terms.add(DcTerm.references);
021    terms.add(DwcTerm.scientificName);
022    terms.add(IucnTerm.threatStatus);
023    terms.add(GbifTerm.canonicalName);
024    terms.add(GbifTerm.depth);
025
026    Collections.sort(terms, comp);
027
028    assertEquals(IucnTerm.threatStatus, terms.get(0));
029    assertEquals(DcTerm.references, terms.get(1));
030    assertEquals(DcTerm.title, terms.get(2));
031    assertEquals(GbifTerm.canonicalName, terms.get(3));
032    assertEquals(GbifTerm.depth, terms.get(4));
033    assertEquals(DwcTerm.basisOfRecord, terms.get(5));
034    assertEquals(DwcTerm.scientificName, terms.get(6));
035  }
036}
