001package org.gbif.dwc.terms;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006import static org.junit.Assert.assertFalse;
007
008public class DcElementTest {
009
010  private static final TermFactory TERM_FACTORY = TermFactory.instance();
011
012  @Test
013  public void testNames() throws Exception {
014    for (DcElement t : DcElement.values()) {
015      assertFalse("Bad term: " + t.simpleName(), t.simpleName().contains("_"));
016      assertFalse("Bad term: " + t.qualifiedName(), t.qualifiedName().contains("_"));
017      assertFalse("Bad term: " + t.qualifiedName(), t.isClass());
018    }
019  }
020
021  /**
022   * Test that iterates over all DcElement values and uses the qualified name alone as input to the factory and then
023   * verifies the Term returned is the same as the original one.
024   */
025  @Test
026  public void testTermEquality() throws Exception {
027    for (DcTerm t : DcTerm.values()) {
028      Term found = TERM_FACTORY.findTerm(t.qualifiedName());
029      assertEquals(t, found);
030    }
031  }
032
033  /**
034   * A lookup for "http://purl.org/dc/elements/1.1/rights" shouldn't return "http://purl.org/dc/terms/rights"
035   */
036  @Test
037  public void testFindTermWithValidNamespace() {
038    String qualifiedName = "http://purl.org/dc/elements/1.1/rights";
039    Term term = TERM_FACTORY.findTerm(qualifiedName);
040    assertEquals(qualifiedName, term.qualifiedName());
041  }
042
043  /**
044   * There are 15 terms in DcElement that share the same name with DcTerm elements. This test ensures, that
045   * when a term lookup is done with the prefix "dc", the DcElement under the /elements/1.1/ namespace
046   * is returned, not the DcTerm under the /terms namespace.
047   */
048  @Test
049  public void testFindTermWithDcPrefix() {
050    String name = "dc:rights";
051    Term term = TERM_FACTORY.findTerm(name);
052    assertEquals("http://purl.org/dc/elements/1.1/rights", term.qualifiedName());
053  }
054}
