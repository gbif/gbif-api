017package org.gbif.dwc.terms;
018
019import java.util.HashSet;
020import java.util.List;
021
022import org.junit.Test;
023
024import static org.junit.Assert.assertEquals;
025import static org.junit.Assert.assertFalse;
026import static org.junit.Assert.assertTrue;
027
028
029public class DwcTermTest extends TermBaseTest<DwcTerm> {
030
031  public DwcTermTest() {
032    super(DwcTerm.class, DwcTerm.PREFIXES);
033  }
034
035  @Test
036  public void testClassTerm() {
037    assertEquals("class", DwcTerm.class_.simpleName());
038    assertEquals("http://rs.tdwg.org/dwc/terms/class", DwcTerm.class_.qualifiedName());
039
040    assertEquals("order", DwcTerm.order.simpleName());
041    assertEquals("http://rs.tdwg.org/dwc/terms/order", DwcTerm.order.qualifiedName());
042  }
043
044  @Test
045  public void testIsClass() {
046    assertTrue(DwcTerm.Taxon.isClass());
047    assertTrue(DwcTerm.Event.isClass());
048    assertFalse(DwcTerm.scientificName.isClass());
049  }
050
051  @Test
052  public void testNumberOfGroups() {
053    assertEquals(10, DwcTerm.GROUPS.length);
054  }
055
056  /**
057   * Test each group contains the expected number of terms, and contains no duplicates.
058   */
059  @Test
060  public void testListByGroup() {
061    List<DwcTerm> occurrenceTerms = DwcTerm.listByGroup(DwcTerm.GROUP_OCCURRENCE);
062    assertEquals(22, occurrenceTerms.size());
063    assertEquals(22, new HashSet<DwcTerm>(occurrenceTerms).size());
064
065    List<DwcTerm> organismTerms = DwcTerm.listByGroup(DwcTerm.GROUP_ORGANISM);
066    assertEquals(8, organismTerms.size());
067    assertEquals(8, new HashSet<DwcTerm>(organismTerms).size());
068
069    List<DwcTerm> materialTerms = DwcTerm.listByGroup(DwcTerm.GROUP_MATERIAL_SAMPLE);
070    assertEquals(2, materialTerms.size());
071    assertEquals(2, new HashSet<DwcTerm>(materialTerms).size());
072
073    List<DwcTerm> eventTerms = DwcTerm.listByGroup(DwcTerm.GROUP_EVENT);
074    assertEquals(19, eventTerms.size());
075    assertEquals(19, new HashSet<DwcTerm>(eventTerms).size());
076
077    List<DwcTerm> locationTerms = DwcTerm.listByGroup(DwcTerm.GROUP_LOCATION);
078    assertEquals(44, locationTerms.size());
079    assertEquals(44, new HashSet<DwcTerm>(locationTerms).size());
080
081    List<DwcTerm> geologicalTerms = DwcTerm.listByGroup(DwcTerm.GROUP_GEOLOGICALCONTEXT);
082    assertEquals(19, geologicalTerms.size());
083    assertEquals(19, new HashSet<DwcTerm>(geologicalTerms).size());
084
085    List<DwcTerm> identificationTerms = DwcTerm.listByGroup(DwcTerm.GROUP_IDENTIFICATION);
086    assertEquals(9, identificationTerms.size());
087    assertEquals(9, new HashSet<DwcTerm>(identificationTerms).size());
088
089    List<DwcTerm> taxonTerms = DwcTerm.listByGroup(DwcTerm.GROUP_TAXON);
090    assertEquals(34, taxonTerms.size());
091    assertEquals(34, new HashSet<DwcTerm>(taxonTerms).size());
092
093    List<DwcTerm> measurementTerms = DwcTerm.listByGroup(DwcTerm.GROUP_MEASUREMENTORFACT);
094    assertEquals(10, measurementTerms.size());
095    assertEquals(10, new HashSet<DwcTerm>(measurementTerms).size());
096
097    List<DwcTerm> relationshipTerms = DwcTerm.listByGroup(DwcTerm.GROUP_RESOURCERELATIONSHIP);
098    assertEquals(8, relationshipTerms.size());
099    assertEquals(8, new HashSet<DwcTerm>(relationshipTerms).size());
100
101    List<DwcTerm> recordTerms = DwcTerm.listByGroup("Record");
102    assertEquals(11, recordTerms.size());
103    assertEquals(11, new HashSet<DwcTerm>(recordTerms).size());
104
105    List<DwcTerm> randomTerms = DwcTerm.listByGroup("Random");
106    assertEquals(0, randomTerms.size());
107    assertEquals(0, new HashSet<DwcTerm>(randomTerms).size());
108  }
109
110  @Test
111  public void testGroupCoverage() {
112    HashSet<DwcTerm> arrayTerms = new HashSet<DwcTerm>();
113    for (DwcTerm t : DwcTerm.TAXONOMIC_TERMS) {
114      arrayTerms.add(t);
115      assertFalse(t.isClass());
116      assertEquals(DwcTerm.GROUP_TAXON, t.getGroup());
117    }
118
119    for (DwcTerm t : DwcTerm.listByGroup(DwcTerm.GROUP_TAXON)) {
120      if (!arrayTerms.contains(t) && !t.isClass()) {
121        assertTrue("Missing taxonomic term in DwcTerm.TAXONOMIC_TERMS: " + t.qualifiedName(), arrayTerms.contains(t));
122      }
123    }
124  }
