016package org.gbif.dwc.terms;
017
018import java.util.HashSet;
019import java.util.List;
020
021import org.junit.jupiter.api.Test;
022
023import static org.junit.jupiter.api.Assertions.assertEquals;
024import static org.junit.jupiter.api.Assertions.assertFalse;
025import static org.junit.jupiter.api.Assertions.assertTrue;
026
027public class DwcTermTest extends TermBaseTest<DwcTerm> {
028
029  public DwcTermTest() {
030    super(DwcTerm.class);
031  }
032
033  @Test
034  public void testClassTerm() {
035    assertEquals("class", DwcTerm.class_.simpleName());
036    assertEquals("http://rs.tdwg.org/dwc/terms/class", DwcTerm.class_.qualifiedName());
037
038    assertEquals("order", DwcTerm.order.simpleName());
039    assertEquals("http://rs.tdwg.org/dwc/terms/order", DwcTerm.order.qualifiedName());
040  }
041
042  @Test
043  public void testIsClass() {
044    assertTrue(DwcTerm.Taxon.isClass());
045    assertTrue(DwcTerm.Event.isClass());
046    assertFalse(DwcTerm.scientificName.isClass());
047  }
048
049  @Test
050  public void testNumberOfGroups() {
051    assertEquals(10, DwcTerm.GROUPS.length);
052  }
053
054  /**
055   * Test each group contains the expected number of terms, and contains no duplicates.
056   */
057  @Test
058  public void testListByGroup() {
059    List<DwcTerm> occurrenceTerms = DwcTerm.listByGroup(DwcTerm.GROUP_OCCURRENCE);
060    assertEquals(22, occurrenceTerms.size());
061    assertEquals(22, new HashSet<>(occurrenceTerms).size());
062
063    List<DwcTerm> organismTerms = DwcTerm.listByGroup(DwcTerm.GROUP_ORGANISM);
064    assertEquals(8, organismTerms.size());
065    assertEquals(8, new HashSet<>(organismTerms).size());
066
067    List<DwcTerm> materialTerms = DwcTerm.listByGroup(DwcTerm.GROUP_MATERIAL_SAMPLE);
068    assertEquals(2, materialTerms.size());
069    assertEquals(2, new HashSet<>(materialTerms).size());
070
071    List<DwcTerm> eventTerms = DwcTerm.listByGroup(DwcTerm.GROUP_EVENT);
072    assertEquals(19, eventTerms.size());
073    assertEquals(19, new HashSet<>(eventTerms).size());
074
075    List<DwcTerm> locationTerms = DwcTerm.listByGroup(DwcTerm.GROUP_LOCATION);
076    assertEquals(44, locationTerms.size());
077    assertEquals(44, new HashSet<>(locationTerms).size());
078
079    List<DwcTerm> geologicalTerms = DwcTerm.listByGroup(DwcTerm.GROUP_GEOLOGICALCONTEXT);
080    assertEquals(19, geologicalTerms.size());
081    assertEquals(19, new HashSet<>(geologicalTerms).size());
082
083    List<DwcTerm> identificationTerms = DwcTerm.listByGroup(DwcTerm.GROUP_IDENTIFICATION);
084    assertEquals(9, identificationTerms.size());
085    assertEquals(9, new HashSet<>(identificationTerms).size());
086
087    List<DwcTerm> taxonTerms = DwcTerm.listByGroup(DwcTerm.GROUP_TAXON);
088    assertEquals(34, taxonTerms.size());
089    assertEquals(34, new HashSet<>(taxonTerms).size());
090
091    List<DwcTerm> measurementTerms = DwcTerm.listByGroup(DwcTerm.GROUP_MEASUREMENTORFACT);
092    assertEquals(10, measurementTerms.size());
093    assertEquals(10, new HashSet<>(measurementTerms).size());
094
095    List<DwcTerm> relationshipTerms = DwcTerm.listByGroup(DwcTerm.GROUP_RESOURCERELATIONSHIP);
096    assertEquals(8, relationshipTerms.size());
097    assertEquals(8, new HashSet<>(relationshipTerms).size());
098
099    List<DwcTerm> recordTerms = DwcTerm.listByGroup("Record");
100    assertEquals(11, recordTerms.size());
101    assertEquals(11, new HashSet<>(recordTerms).size());
102
103    List<DwcTerm> randomTerms = DwcTerm.listByGroup("Random");
104    assertEquals(0, randomTerms.size());
105    assertEquals(0, new HashSet<>(randomTerms).size());
106  }
107
108  @Test
109  public void testGroupCoverage() {
110    HashSet<DwcTerm> arrayTerms = new HashSet<>();
111    for (DwcTerm t : DwcTerm.TAXONOMIC_TERMS) {
112      arrayTerms.add(t);
113      assertFalse(t.isClass());
114      assertEquals(DwcTerm.GROUP_TAXON, t.getGroup());
115    }
116
117    for (DwcTerm t : DwcTerm.listByGroup(DwcTerm.GROUP_TAXON)) {
118      if (!arrayTerms.contains(t) && !t.isClass()) {
119        assertTrue(arrayTerms.contains(t), "Missing taxonomic term in DwcTerm.TAXONOMIC_TERMS: " + t.qualifiedName());
120      }
121    }
122  }
