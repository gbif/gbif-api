001package org.gbif.dwc.terms.jackson;
002
003import org.gbif.dwc.terms.DwcTerm;
004import org.gbif.dwc.terms.GbifTerm;
005import org.gbif.dwc.terms.Term;
006import org.gbif.dwc.terms.UnknownTerm;
007
008import java.io.IOException;
009import java.util.HashMap;
010import java.util.Map;
011
012import org.codehaus.jackson.map.ObjectMapper;
013import org.codehaus.jackson.map.annotate.JsonDeserialize;
014import org.codehaus.jackson.map.annotate.JsonSerialize;
015import org.junit.Test;
016
017import static org.junit.Assert.assertEquals;
018
019public class TermDeserializerTest {
020
021  /**
022   * Test class resembling a verbatim occurrence field map that causes json serde trouble.
023   */
024  static class Occ {
025    private Term term;
026    private Map<Term, String> data = new HashMap<Term, String>();
027
028    @JsonSerialize(keyUsing = TermKeySerializer.class)
029    @JsonDeserialize(keyUsing = TermKeyDeserializer.class)
030    public Map<Term, String> getData() {
031      return data;
032    }
033
034    public void setData(Map<Term, String> data) {
035      this.data = data;
036    }
037
038    public Term getTerm() {
039      return term;
040    }
041
042    public void setTerm(Term term) {
043      this.term = term;
044    }
045
046    @Override
047    public boolean equals(Object o) {
048      if (this == o) return true;
049      if (!(o instanceof Occ)) return false;
050
051      Occ occ = (Occ) o;
052
053      if (!data.equals(occ.data)) return false;
054      if (!term.equals(occ.term)) return false;
055
056      return true;
057    }
058
059    @Override
060    public int hashCode() {
061      int result = term.hashCode();
062      result = 31 * result + data.hashCode();
063      return result;
064    }
065
066  }
067
068  @Test
069  public void testTermSerde() throws IOException {
070    ObjectMapper mapper = new ObjectMapper();
071
072    for (Term t : DwcTerm.values()) {
073      String json = mapper.writeValueAsString(t);
074      assertEquals(t, mapper.readValue(json, Term.class));
075    }
076
077    for (Term t : GbifTerm.values()) {
078      String json = mapper.writeValueAsString(t);
079      assertEquals(t, mapper.readValue(json, Term.class));
080    }
081  }
082
083  @Test
084  public void testTermMap() throws IOException {
085    ObjectMapper mapper = new ObjectMapper();
086    Integer counter = 0;
087    Map<Term, Integer> terms = new HashMap<Term, Integer>();
088    for (Term t : DwcTerm.values()) {
089      terms.put(t, counter++);
090    }
091
092    for (Term t : GbifTerm.values()) {
093      terms.put(t, counter++);
094    }
095
096    String json = mapper.writeValueAsString(terms);
097    assertEquals(terms.size(), mapper.readValue(json, HashMap.class).size());
098  }
099
100  @Test
101  public void testOcc() throws IOException {
102    ObjectMapper mapper = new ObjectMapper();
103
104    final Term custom = UnknownTerm.build("http://me.com/mum", "mum");
105    Occ o = new Occ();
106    o.setTerm(custom);
107    o.getData().put(custom, "custom");
108    o.getData().put(GbifTerm.ageInDays, "21");
109
110    for (Term t : DwcTerm.values()) {
111      o.getData().put(t, t.simpleName());
112    }
113
114    String json = mapper.writeValueAsString(o);
115    Occ o2 = mapper.readValue(json, Occ.class);
116    System.out.println(o.equals(o2));
117
118    assertEquals(o, mapper.readValue(json, Occ.class));
119  }
120
121}
