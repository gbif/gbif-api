001package org.gbif.dwc.terms;
002
003import org.junit.Test;
004
005import static org.junit.Assert.assertEquals;
006import static org.junit.Assert.assertFalse;
007
008
009/**
010 *
011 */
012public class DcTermTest {
013
014  private static final TermFactory TERM_FACTORY = TermFactory.instance();
015
016  @Test
017  public void testNames() throws Exception {
018    for (DcTerm t : DcTerm.values()) {
019      assertFalse("Bad term: " + t.simpleName(), t.simpleName().contains("_"));
020      assertFalse("Bad term: " + t.qualifiedName(), t.qualifiedName().contains("_"));
021    }
022  }
023
024  /**
025   * Test that iterates over all DcTerm values and uses the simple name alone as input to the factory and then verifies
026   * the Term returned is the same as the original one.
027   */
028  @Test
029  public void testTermEquality() throws Exception {
030    for (DcTerm t : DcTerm.values()) {
031      Term found = TERM_FACTORY.findTerm(t.simpleName());
032      assertEquals(t, found);
033    }
034  }
035
036  /**
037   * A lookup for "http://purl.org/dc/terms/rights" shouldn't return "http://purl.org/dc/elements/1.1/rights".
038   */
039  @Test
040  public void testFindTermWithValidNamespace() {
041    String qualifiedName = "http://purl.org/dc/terms/rights";
042    Term term = TERM_FACTORY.findTerm(qualifiedName);
043    assertEquals(qualifiedName, term.qualifiedName());
044  }
045
046  /**
047   * There are 15 terms in DcElement that share the same name with DcTerm elements. This test ensures, that
048   * when a term lookup is done with the simple name only (no prefix, no namespace), the DcTerm under the /terms
049   * namespace is returned, not the DcElement under the /elements/1.1/rights namespace.
050   */
051  @Test
052  public void testFindTermWithNoNamespaceAndNoPrefix() {
053    String name = "rights";
054    Term term = TERM_FACTORY.findTerm(name);
055    assertEquals("http://purl.org/dc/terms/rights", term.qualifiedName());
056  }
057
058  /**
059   * There are 15 terms in DcElement that share the same name with DcTerm elements. This test ensures, that
060   * when a term lookup is done with the prefix "dcterms", the DcTerm under the /terms namespace
061   * is returned, not the DcElements term under the /elements/1.1/ namespace.
062   */
063  @Test
064  public void testFindTermWithDctermsPrefix() {
065    String name = "dcterms:rights";
066    Term term = TERM_FACTORY.findTerm(name);
067    assertEquals("http://purl.org/dc/terms/rights", term.qualifiedName());
068  }
069
070  /**
071   * There are 15 terms in DcElement that share the same name with DcTerm elements. This test ensures, that
072   * when a term lookup is done with the prefix "dct", the DcTerm under the /terms namespace
073   * is returned, not the DcElements term under the /elements/1.1/ namespace.
074   */
075  @Test
076  public void testFindTermWithDctPrefix() {
077    String name = "dct:rights";
078    Term term = TERM_FACTORY.findTerm(name);
079    assertEquals("http://purl.org/dc/terms/rights", term.qualifiedName());
080  }
081
082  /**
083   * There are 15 terms in DcElement that share the same name with DcTerm elements. This test ensures, that
084   * when a term lookup is done with the prefix "dcterm", the DcTerm under the /terms namespace
085   * is returned, not the DcElements term under the /elements/1.1/ namespace.
086   */
087  @Test
088  public void testFindTermWithDctermPrefix() {
089    String name = "dcterm:rights";
090    Term term = TERM_FACTORY.findTerm(name);
091    assertEquals("http://purl.org/dc/terms/rights", term.qualifiedName());
092  }
093}
