016  /**
017   * GBIF code assumes a term coming from any of the Term enumerations mostly have unique simple names.
018   * This tests verifies that!
019   *
020   * AcefTerm is known to overlap, so its excluded, see skipSimple in AcefTermTest.
021   */
022  @Test
023  public void testKnownTermUniqueness() {
024    Set<String> names = new HashSet<String>();
025
026    addTerms(names, DwcTerm.values());
027    addTerms(names, DcTerm.values());
028    addTerms(names, GbifTerm.values());
029    addTerms(names, GbifInternalTerm.values());
030    addTerms(names, IucnTerm.values());
031
032    addTerms(names, AcTerm.values());
033    addTerms(names, XmpTerm.values());
034    addTerms(names, XmpRightsTerm.values());
035  }
036
037  private void addTerms(Set<String> names, Term[] terms) {
038    for (Term t : terms) {
039      assertFalse("Duplicate simple name "+t.simpleName(), names.contains(t.simpleName()));
040      names.add(t.simpleName());
041    }
042  }

087  @Test
088  public void addUnknownTerm() {
089    TermFactory factory = TermFactory.instance();
090
091    Term me1 = factory.findTerm("http://me.com/#me");
092    Term me2 = factory.findTerm("http://me.com/me");
093    Term me3 = factory.findTerm("http://me.org/me");
094
095    assertNotEquals(me1, me2);
096    assertNotEquals(me1, me3);
097    assertNotEquals(me2, me3);
098  }

106  @Test
107  public void addSimpleTerm() {
108    TermFactory factory = TermFactory.instance();
109
110    Term hallo = factory.findTerm("hallo");
111    assertEquals(UnknownTerm.class, hallo.getClass());
112    assertEquals("http://unknown.org/hallo", hallo.qualifiedName());
113    assertEquals("hallo", hallo.simpleName());
114
115
116    Term tim = factory.findTerm("Tim");
117    assertEquals(UnknownTerm.class, tim.getClass());
118    assertEquals("http://unknown.org/Tim", tim.qualifiedName());
119    assertEquals("http://unknown.org", tim.namespace().toString());
120    assertEquals("Tim", tim.simpleName());
121
122    Term eva = factory.findTerm("tim:Eva");
123    assertEquals(UnknownTerm.class, eva.getClass());
124    assertEquals("http://unknown.org/tim/Eva", eva.qualifiedName());
125    assertEquals("http://unknown.org", tim.namespace().toString());
126    assertEquals("http://unknown.org/tim/Eva", eva.prefixedName());
127    assertEquals("Eva", eva.simpleName());
128
129    assertNotEquals(hallo, tim);
130  }

146  /**
147     * Not a real test, just a way of running many concurrent TermFactory.instance() calls to verify thread safety.
148     */
149  @Test
150  public void testMultithreadStart() throws InterruptedException {
151    int threadCount = 100;
152    ExecutorService tp = Executors.newFixedThreadPool(threadCount);
153    for (int i = 0; i < threadCount; i++) {
154      tp.submit(new TermFactoryLoader());
155    }
156    tp.shutdown();
157    tp.awaitTermination(30, TimeUnit.SECONDS);
158  }
159
160  private static class TermFactoryLoader implements Runnable {
161    @Override
162    public void run() {
163      TermFactory.instance();
164    }
165  }
