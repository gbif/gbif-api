001package org.gbif.dwc.terms;
002
003import org.codehaus.jackson.map.annotate.JsonDeserialize;
004import org.codehaus.jackson.map.annotate.JsonSerialize;
005import org.gbif.dwc.terms.jackson.TermDeserializer;
006import org.gbif.dwc.terms.jackson.TermSerializer;
007
008import java.io.Serializable;
009import java.net.URI;
010
011@JsonSerialize(using= TermSerializer.class)
012@JsonDeserialize(using= TermDeserializer.class)
013public interface Term extends Serializable {
014
015  /**
016   * A unique standard prefix representing the namespace.
017   * For example dwc.
018   */
019  String prefix();
020
021  /**
022   * The namespace the terms are in.
023   * Default implementations here expect the namespace to end with a slash.
024   */
025  URI namespace();
026
027  /**
028   * The simple term name without any namespace or paths.
029   * For example scientificName.
030   */
031  String simpleName();
032
033  /**
034   * The simple term name prefixed by a short unique namespace abbreviation.
035   * For example dwc:scientificName.
036   */
037  default String prefixedName() {
038    return prefix() + ":" + simpleName();
039  }
040
041  /**
042   * The full qualified term uri including the namespace.
043   * For example http://rs.tdwg.org/dwc/terms/scientificName.
044   */
045  default String qualifiedName() {
046    return namespace() + simpleName();
047  }
048
049  /**
050   * Informs if a term is generally used as a class term, i.e. defining rowTypes not properties.
051   * For example VernacularName, Taxon or Occurrence
052   * @return true if the term is defining a class instead of a property, e.g. Taxon
053   */
054  boolean isClass();
055
056}
