001package org.gbif.dwc.terms;
002
003import java.net.URI;
004
005public enum GadmTerm implements Term, AlternativeNames {
006  level0Gid(DwcTerm.GROUP_LOCATION),
007  level0Name(DwcTerm.GROUP_LOCATION),
008  level1Gid(DwcTerm.GROUP_LOCATION),
009  level1Name(DwcTerm.GROUP_LOCATION),
010  level2Gid(DwcTerm.GROUP_LOCATION),
011  level2Name(DwcTerm.GROUP_LOCATION),
012  level3Gid(DwcTerm.GROUP_LOCATION),
013  level3Name(DwcTerm.GROUP_LOCATION);
014
015  private static final String PREFIX = "gadm";
016  private static final String NS = "http://rs.gbif.org/terms/gadm/3.0/";
017  private static final URI NS_URI = URI.create(NS);
018
019  public final String[] normAlts;
020
021  GadmTerm(String... alternatives) {
022    normAlts = alternatives;
023  }
024
025  @Override
026  public String simpleName() {
027    return name();
028  }
029
030  @Override
031  public String[] alternativeNames() {
032    return normAlts;
033  }
034
035  @Override
036  public String toString() {
037    return prefixedName();
038  }
039
040  @Override
041  public String prefix() {
042    return PREFIX;
043  }
044
045  @Override
046  public URI namespace() {
047    return NS_URI;
048  }
049
050  @Override
051  public boolean isClass() {
052    return false;
053  }
054}
