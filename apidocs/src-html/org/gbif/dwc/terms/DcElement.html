001package org.gbif.dwc.terms;
002
003import java.net.URI;
004
005/**
006 * All Dublin Core terms with namespace http://purl.org/dc/elements/1.1/. All terms exist as DcTerm under a different
007 * namespace, but DcElement values are allowed to be free text, whereas DcTerm values must be a member of a specific
008 * class and therefore given by URI.
009 */
010public enum DcElement implements Term, AlternativeNames {
011  contributor,
012  coverage,
013  creator,
014  date,
015  description,
016  format,
017  identifier,
018  language,
019  publisher,
020  relation,
021  rights,
022  source,
023  subject,
024  title,
025  type;
026
027  private static final String PREFIX = "dc";
028  private static final String NS = "http://purl.org/dc/elements/1.1/";
029  private static final URI NS_URI = URI.create(NS);
030
031  public final String[] alternatives;
032
033  @Override
034  public String toString() {
035    return prefixedName();
036  }
037
038  @Override
039  public String simpleName() {
040    return name();
041  }
042
043  @Override
044  public String[] alternativeNames() {
045    return alternatives;
046  }
047
048  @Override
049  public boolean isClass() {
050    return false;
051  }
052
053  @Override
054  public String prefix() {
055    return PREFIX;
056  }
057
058  @Override
059  public URI namespace() {
060    return NS_URI;
061  }
062
063  DcElement(String... alternatives) {
064    this.alternatives = alternatives;
065  }
066}
