378  private final String groupName;
379  private final boolean isDeprecated;
380  public final String[] normAlts;
381
382  GbifTerm(String groupName, String... alternatives) {
383    this.groupName = groupName;
384    this.normAlts = alternatives;
385    boolean deprecatedAnnotationPresent = false;
386    try {
387      deprecatedAnnotationPresent = GbifTerm.class.getField(this.name()).isAnnotationPresent(Deprecated.class);
388    } catch (NoSuchFieldException ignore) { }
389    this.isDeprecated = deprecatedAnnotationPresent;
390  }
391
392  /**
393   * The simple term name without a namespace.
394   * For example taxonKey.
395   *
396   * @return simple term name
397   */
398  @Override
399  public String simpleName() {
400    return name();
401  }
402
403  /**
404   * Array of alternative simple names in use for the term.
405   *
406   * @return simple term name
407   */
408  @Override
409  public String[] alternativeNames() {
410    return normAlts;
411  }
412
413  /**
414   * The optional group the term is grouped in.
415   * For example Occurrence, Taxon, etc.
416   */
417  public String getGroup() {
418    return groupName;
419  }
420
421
422  /**
423   * List all terms that belong to a given group.
424   *
425   * @param group the group to list terms for
426   * @return the list of GBIF terms in the given group
427   */
428  public static List<GbifTerm> listByGroup(String group) {
429    List<GbifTerm> terms = new ArrayList<GbifTerm>();
430    for (GbifTerm t : GbifTerm.values()) {
431      if (t.getGroup().equalsIgnoreCase(group)) {
432        terms.add(t);
433      }
434    }
435    return terms;
436  }
437
438  @Override

443  @Override
444  public boolean isClass() {
445    return Character.isUpperCase(simpleName().charAt(0));
446  }
447
448  @Override
449  public String prefix() {
450    return PREFIX;
451  }
452
453  @Override
454  public URI namespace() {
455    return NS_URI;
456  }
457
458  /**
459   *
460   * @return true if the Term is annotated with @Deprecated.
461   */
462  public boolean isDeprecated(){

464  }
