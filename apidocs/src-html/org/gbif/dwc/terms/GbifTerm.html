043  /**
044   * Elevation in meters above sea level (altitude).
045   * <p>
046   * The elevation is the absolute vertical position of the observed location (z-coordinate).
047   * If depth is given or not will not impact the 3-dimensional position.
048   * For example a location 100m below the surface of a lake in 2000m altitude has a depth of 100 and
049   * an elevation of 1900.
050   * </p>
051   * <p>
052   * If minimum and maximum values are given the elevation is calculated using the equation:
053   * (minimumElevationInMeters + maximumElevationInMeters) / 2.
054   * For consistency and ease of use GBIF decided to always use a value in meters plus it's accurracy instead of
055   * min/max values which are sometimes used in Darwin Core. See also depth & distanceAboveSurface.
056   * </p>
057   */
058  elevation(DwcTerm.GROUP_OCCURRENCE),
059
060  /**
061   * Elevation accuracy is the uncertainty for the elevation in meters.
062   * <p>
063   * The elevation accuracy is calculated using the equation: (maximumElevationInMeters - minimumElevationInMeters) / 2
064   * in case a minimum and maximum verbatim value is given.
065   * </p>
066   */
067  elevationAccuracy(DwcTerm.GROUP_OCCURRENCE),
068
069  /**
070   * Depth in meters below the surface.
071   * <p>
072   * Complimentary and relative to elevation, depth indicates the distance to the earth surface, whether that is water
073   * or ground.
074   * For example a location 100m below the surface of a lake in 2000m altitude has a depth of 100 and
075   * an elevation of 1900.
076   * </p>
077   * <p>
078   * The depth is calculated using the equation: (minimumDepthInMeters + maximumDepthInMeters) / 2.
079   * For consistency and ease of use GBIF decided to always use a value in meters plus it's accurracy instead of
080   * min/max values which are sometimes used in Darwin Core. See also elevation & distanceAboveSurface.
081   * </p>
082   */
083  depth(DwcTerm.GROUP_OCCURRENCE),
084
085  /**
086   * Depth accuracy is the uncertainty for the depth in meters.
087   * <p>
088   * The depth accuracy is calculated using the equation: (maximumDepthInMeters - minimumDepthInMeters) / 2
089   * in case a minimum and maximum verbatim value is given.
090   * </p>
091   */
092  depthAccuracy(DwcTerm.GROUP_OCCURRENCE),
093
094  distanceAboveSurface(DwcTerm.GROUP_OCCURRENCE),
095  distanceAboveSurfaceAccuracy(DwcTerm.GROUP_OCCURRENCE),

133   * <p>
134   * In contrast dwc:taxonID is only used for the source ids similar to occurrenceID
135   * </p>

142  kingdomKey(DwcTerm.GROUP_TAXON),

151   */
152  classKey(DwcTerm.GROUP_TAXON),

154  /**
155   * The key to the accepted order in the GBIF backbone.
156   */
157  orderKey(DwcTerm.GROUP_TAXON),

159  /**
160   * The key to the accepted family in the GBIF backbone.

162  familyKey(DwcTerm.GROUP_TAXON),

176   * In case the taxonKey is of a higher rank than species (e.g. genus) speciesKey is null.

247  public static final String GROUP_DATASET = "Dataset";
248  public static final String GROUP_ROW_TYPE = "RowType";

251  public static final String GROUP_VERNACULAR_NAME_EXTENSION = "VernacularName";

257    GROUP_SPECIES_DISTRIBUTION_EXTENSION, GROUP_SPECIES_PROFILE_EXTENSION, DwcTerm.GROUP_TAXON,

268  private final String groupName;
269  public final String[] normAlts;
270
271  private GbifTerm(String groupName, String... alternatives) {
272    this.groupName = groupName;
273    normAlts = alternatives;
274  }
275
276  /**
277   * The full qualified term uri including the namespace.
278   * For example http://rs.gbif.org/terms/1.0/taxonKey.
279   *
280   * @return full qualified term uri
281   */
282  @Override
283  public String qualifiedName() {
284    return NS + simpleName();
285  }
286
287  /**
288   * The simple term name without a namespace.
289   * For example taxonKey.
290   *
291   * @return simple term name
292   */
293  @Override
294  public String simpleName() {
295    return name();
296  }
297
298  /**
299   * Array of alternative simple names in use for the term.
300   *
301   * @return simple term name
302   */
303  @Override
304  public String[] alternativeNames() {
305    return normAlts;
306  }
307
308  /**
309   * The optional group the term is grouped in.
310   * For example Occurrence, Taxon, etc.
311   */
312  public String getGroup() {
313    return groupName;
314  }
315
316
317  /**
318   * List all terms that belong to a given group.
319   *
320   * @param group the group to list terms for
321   * @return the list of GBIF terms in the given group
322   */
323  public static List<GbifTerm> listByGroup(String group) {
324    List<GbifTerm> terms = new ArrayList<GbifTerm>();
325    for (GbifTerm t : GbifTerm.values()) {
326      if (t.getGroup().equalsIgnoreCase(group)) {
327        terms.add(t);
328      }
329    }
330    return terms;
331  }
332
333  @Override
334  public String toString() {
335    return PREFIX + ":" + name();
336  }
337
338  /**
339   * @return true if the term is defining a class or row type instead of a property, e.g. VernacularName
340   */
341  public boolean isClass() {
342    return Character.isUpperCase(simpleName().charAt(0));
343  }
