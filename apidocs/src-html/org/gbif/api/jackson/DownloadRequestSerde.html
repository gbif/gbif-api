016package org.gbif.api.jackson;
017
018import org.gbif.api.model.occurrence.DownloadFormat;
019import org.gbif.api.model.occurrence.DownloadRequest;
020import org.gbif.api.model.occurrence.PredicateDownloadRequest;
021import org.gbif.api.model.occurrence.predicate.Predicate;
022import org.gbif.api.util.VocabularyUtils;
023
024import java.io.IOException;
025import java.util.ArrayList;
026import java.util.Arrays;
027import java.util.Collections;
028import java.util.List;
029import java.util.Optional;
030
031import org.slf4j.Logger;
032import org.slf4j.LoggerFactory;
033
034import com.fasterxml.jackson.core.JsonParser;
035import com.fasterxml.jackson.databind.DeserializationContext;
036import com.fasterxml.jackson.databind.JsonDeserializer;
037import com.fasterxml.jackson.databind.JsonNode;
038import com.fasterxml.jackson.databind.ObjectMapper;
039
040/**
041 * Download request deserializer.
042 * <p>
043 * For most of the time, the serialization has been to "notificationAddresses" and
044 * "sendNotification".  For a few months in 2018-2019, it was "notification_address" and
045 * "send_notification".
046 * <p>
047 * The API documentation has previously specified "notification_address" and "sendNotification".
048 * <p>
049 * We therefore accept all combinations.
050 * <p>
051 * https://github.com/gbif/portal-feedback/issues/2046
052 */
053public class DownloadRequestSerde extends JsonDeserializer<DownloadRequest> {
054
055  private static final String PREDICATE = "predicate";
056  private static final List<String> SEND_NOTIFICATION = Collections
057    .unmodifiableList(Arrays.asList("sendNotification", "send_notification"));
058  private static final List<String> NOTIFICATION_ADDRESSES =
059    Collections.unmodifiableList(
060      Arrays.asList("notificationAddresses", "notificationAddress", "notification_addresses",
061        "notification_address"));
062  private static final String CREATOR = "creator";
063  private static final String FORMAT = "format";
064  private static final Logger LOG = LoggerFactory.getLogger(DownloadRequestSerde.class);
065  private static final ObjectMapper MAPPER = new ObjectMapper();
066
067  @Override
068  public DownloadRequest deserialize(JsonParser jp, DeserializationContext ctxt) throws IOException {
069    JsonNode node = jp.getCodec().readTree(jp);
070    LOG.debug("DownloadRequest for deserialization: {}", node);
071    //at least one element must be defined
072    if (node.size() == 0) {
073      return null;
074    }
075    DownloadFormat format = Optional.ofNullable(node.get(FORMAT))
076      .map(n -> VocabularyUtils.lookupEnum(n.asText(), DownloadFormat.class)).orElse(DownloadFormat.DWCA);
077    String creator = Optional.ofNullable(node.get(CREATOR)).map(JsonNode::asText).orElse(null);
078
079    List<String> notificationAddresses = new ArrayList<>();
080    for (final String jsonKey : NOTIFICATION_ADDRESSES) {
081      notificationAddresses.addAll(Optional.ofNullable(node.get(jsonKey)).map(jsonNode -> {
082        try {
083          return Arrays.asList(MAPPER.treeToValue(jsonNode, String[].class));
084        } catch (Exception e) {
085          throw new RuntimeException(e);
086        }
087      }).orElse(new ArrayList<>()));
088    }
089
090    boolean sendNotification = false;
091    for (final String jsonKey : SEND_NOTIFICATION) {
092      sendNotification |= Optional.ofNullable(node.get(jsonKey)).map(JsonNode::asBoolean).orElse(Boolean.FALSE);
093    }
094
095    JsonNode predicate = Optional.ofNullable(node.get(PREDICATE)).orElse(null);
096    Predicate predicateObj = predicate == null ? null : MAPPER.treeToValue(predicate, Predicate.class);
097    return new PredicateDownloadRequest(predicateObj, creator, notificationAddresses, sendNotification, format);
098  }
099}
