001package org.gbif.api.jackson;
002
003import java.io.IOException;
004import java.util.ArrayList;
005import java.util.Arrays;
006import java.util.List;
007import java.util.Optional;
008import org.codehaus.jackson.JsonNode;
009import org.codehaus.jackson.JsonParser;
010import org.codehaus.jackson.JsonProcessingException;
011import org.codehaus.jackson.map.DeserializationContext;
012import org.codehaus.jackson.map.JsonDeserializer;
013import org.codehaus.jackson.map.ObjectMapper;
014import org.gbif.api.model.occurrence.DownloadFormat;
015import org.gbif.api.model.occurrence.DownloadRequest;
016import org.gbif.api.model.occurrence.PredicateDownloadRequest;
017import org.gbif.api.model.occurrence.SqlDownloadRequest;
018import org.gbif.api.model.occurrence.predicate.Predicate;
019import org.gbif.api.util.VocabularyUtils;
020import org.slf4j.Logger;
021import org.slf4j.LoggerFactory;
022import com.google.common.base.Throwables;
023
024/**
025 * 
026 * Download request deserializer.
027 *
028 */
029public class DownloadRequestSerde extends JsonDeserializer<DownloadRequest> {
030
031  private static final String PREDICATE = "predicate";
032  private static final String SQL = "sql";
033  private static final String SEND_NOTIFICATION = "send_notification";
034  private static final String NOTIFICATION_ADDRESS = "notification_address";
035  private static final String CREATOR = "creator";
036  private static final String FORMAT = "format";
037  private static final Logger LOG = LoggerFactory.getLogger(DownloadRequestSerde.class);
038  private static final ObjectMapper MAPPER = new ObjectMapper();
039
040  @Override
041  public DownloadRequest deserialize(JsonParser jp, DeserializationContext ctxt) throws IOException, JsonProcessingException {
042    JsonNode node = jp.getCodec().readTree(jp);
043    LOG.debug("DownloadRequest for deserialization: {}", node);
044    //at least one element must be defined
045    if (node.size() == 0) {
046      return null;
047    }
048    DownloadFormat format = Optional.ofNullable(node.get(FORMAT))
049      .map(n -> VocabularyUtils.lookupEnum(n.asText(), DownloadFormat.class)).orElse(DownloadFormat.DWCA);
050    String creator = Optional.ofNullable(node.get(CREATOR)).map(JsonNode::asText).orElse(null);
051
052    List<String> notificationAddress = Optional.ofNullable(node.get(NOTIFICATION_ADDRESS)).map(jsonNode -> {
053      try {
054        return Arrays.asList(MAPPER.readValue(jsonNode, String[].class));
055      } catch (Exception e) {
056       throw Throwables.propagate(e);
057      }
058    }).orElse(new ArrayList<>());
059
060    boolean sendNotification = Optional.ofNullable(node.get(SEND_NOTIFICATION)).map(JsonNode::asBoolean).orElse(Boolean.FALSE);
061
062    if (DownloadFormat.SQL == format) {
063      String sql = Optional.ofNullable(node.get(SQL)).map(JsonNode::asText).orElse(null);
064      return new SqlDownloadRequest(sql, creator, notificationAddress, sendNotification);
065    } else {
066      JsonNode predicate = Optional.ofNullable(node.get(PREDICATE)).orElse(null);
067      Predicate predicateObj = predicate == null ? null : MAPPER.readValue(predicate, Predicate.class);
068      return new PredicateDownloadRequest(predicateObj, creator, notificationAddress, sendNotification, format);
069    }
070  }
071}
