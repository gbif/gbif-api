001package org.gbif.api.jackson;
002
003import org.codehaus.jackson.JsonGenerator;
004import org.codehaus.jackson.JsonParser;
005import org.codehaus.jackson.JsonToken;
006import org.codehaus.jackson.map.DeserializationContext;
007import org.codehaus.jackson.map.JsonDeserializer;
008import org.codehaus.jackson.map.JsonSerializer;
009import org.codehaus.jackson.map.SerializerProvider;
010import org.codehaus.jackson.map.deser.std.DateDeserializer;
011
012import java.io.IOException;
013import java.text.SimpleDateFormat;
014import java.time.Instant;
015import java.util.Date;
016import java.util.TimeZone;
017
018/**
019 * Jackson {@link JsonSerializer} classes for {@link Date}s with specified formats.
020 */
021public class DateSerde {
022
023  private static SimpleDateFormat noTimezoneFormat = new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ss");
024  static {
025    noTimezoneFormat.setTimeZone(TimeZone.getTimeZone("UTC"));
026  }
027
028  /**
029   * Jackson {@link JsonSerializer} for {@link Date}.
030   */
031  public static class NoTimezoneDateJsonSerializer extends JsonSerializer<Date> {
032
033    @Override
034    public void serialize(Date value, JsonGenerator jgen, SerializerProvider provider) throws IOException {
035      if (value == null) {
036        // Empty fields aren't included in the JSON.
037        return;
038      } else {
039        jgen.writeString(noTimezoneFormat.format(value));
040      }
041    }
042  }
043
044  /**
045   * Jackson {@link JsonDeserializer} for {@link Date}s formatted above, falling back to the Jackson way.
046   */
047  public static class FlexibleDateJsonDeserializer extends DateDeserializer {
048
049    @Override
050    public Date deserialize(JsonParser jp, DeserializationContext ctxt) throws IOException {
051      if (jp.getCurrentToken() == JsonToken.VALUE_STRING) {
052        String text = jp.getText();
053        if (text.length() == 19) {
054          return Date.from(Instant.parse(text+"Z"));
055        } else {
056          return super.deserialize(jp, ctxt);
057        }
058      }
059      throw ctxt.mappingException("Expected String");
060    }
061  }
062}
