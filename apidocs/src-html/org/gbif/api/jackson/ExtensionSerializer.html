001package org.gbif.api.jackson;
002
003import org.gbif.api.vocabulary.Extension;
004
005import java.io.IOException;
006
007import org.codehaus.jackson.JsonGenerator;
008import org.codehaus.jackson.map.JsonSerializer;
009import org.codehaus.jackson.map.SerializerProvider;
010
011public class ExtensionSerializer extends JsonSerializer<Extension> {
012
013  @Override
014  public void serialize(Extension value, JsonGenerator jgen, SerializerProvider provider) throws IOException {
015    jgen.writeFieldName(value.getRowType());
016  }
017}
