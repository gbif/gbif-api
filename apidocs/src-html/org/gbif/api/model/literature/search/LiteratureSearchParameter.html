016package org.gbif.api.model.literature.search;
017
018import org.gbif.api.model.common.search.SearchParameter;
019import org.gbif.api.model.literature.LiteratureRelevance;
020import org.gbif.api.model.literature.LiteratureTopic;
021import org.gbif.api.model.literature.LiteratureType;
022import org.gbif.api.vocabulary.Country;
023
024import java.util.UUID;
025
026public enum LiteratureSearchParameter implements SearchParameter {
027  COUNTRIES_OF_RESEARCHER(Country.class),
028  COUNTRIES_OF_COVERAGE(Country.class),
029  LITERATURE_TYPE(LiteratureType.class),
030  RELEVANCE(LiteratureRelevance.class),
031  YEAR(Integer.class),
032  TOPICS(LiteratureTopic.class),
033  GBIF_DATASET_KEY(UUID.class), // dataset
034  PUBLISHING_ORGANIZATION_KEY(UUID.class), // publisher
035  PEER_REVIEW(Boolean.class),
036  OPEN_ACCESS(Boolean.class),
037  GBIF_DOWNLOAD_KEY(String.class), // download key
038  DOI(String.class),
039  SOURCE(String.class), // journal
040  PUBLISHER(String.class); // journal publisher
041
042  private final Class<?> type;
043
044  LiteratureSearchParameter(Class<?> type) {
045    this.type = type;
046  }
047
048  @Override
049  public Class<?> type() {
050    return type;
051  }
052}
