014package org.gbif.api.model.literature.search;
015
016import lombok.Getter;
017
018import lombok.Setter;
019
020import org.gbif.api.model.literature.LiteratureRelevance;
021import org.gbif.api.model.literature.LiteratureTopic;
022import org.gbif.api.model.literature.LiteratureType;
023import org.gbif.api.vocabulary.Country;
024import org.gbif.api.vocabulary.GbifRegion;
025import org.gbif.api.vocabulary.Language;
026
027import java.util.ArrayList;
028import java.util.Date;
029import java.util.HashMap;
030import java.util.HashSet;
031import java.util.List;
032import java.util.Map;
033import java.util.Set;
034import java.util.UUID;
035
036import com.fasterxml.jackson.annotation.JsonProperty;
037
038import io.swagger.v3.oas.annotations.media.Schema;
039
040@Getter
041@Setter
042@SuppressWarnings("unused")
043public class LiteratureSearchResult {
044
045  private String abstract_;
046
047  @Schema(description = "The date the literature item was found by GBIF's staff or automated processes.")
048  private String discovered;
049
050  @Schema(description = "The authors of the publication.")
051  private List<Map<String, Object>> authors = new ArrayList<>();
052
053  @Schema(description = "Countries or areas of focus of the study.")
054  private Set<Country> countriesOfCoverage = new HashSet<>();
055
056  @Schema(description = "Countries or areas of institution with which authors are affiliated.")
057  private Set<Country> countriesOfResearcher = new HashSet<>();
058
059  @Schema(description = "The date the publication was added to the GBIF literature database.")
060  private Date added;
061
062  @Schema(description = "The publication date of the publication.  See also `year`, `month` and `day`.")
063  private Date published;
064
065  @Schema(description = "The day of publication. See also `published`.", minimum = "1", maximum = "31")
066  private Integer day;
067
068  @Schema(description = "Keys of GBIF downloads referenced by the publication.")
069  private List<String> gbifDownloadKey = new ArrayList<>();
070
071  @Schema(description = "Keys of GBIF occurrences directly mentioned by the paper.")
072  private List<Long> gbifOccurrenceKey = new ArrayList<>();
073
074  @Schema(description = "Keys of taxa from the GBIF Backbone Taxonomy that are the focus of the paper.")
075  private List<Integer> gbifTaxonKey = new ArrayList<>();
076
077  @Schema(description = "Keys of higher taxa from the GBIF Backbone Taxonomy that are the focus of the paper.")
078  private List<Integer> gbifHigherTaxonKey = new ArrayList<>();
079
080  @Schema(description = "A list of GBIF network keys relevant to the publication.")
081  private List<UUID> gbifNetworkKey = new ArrayList<>();
082
083  @Schema(description = "A list of GBIF project identifiers relevant to the publication.")
084  private List<String> gbifProjectIdentifier = new ArrayList<>();
085
086  @Schema(description = "A list of GBIF programmes relevant to the publication.")
087  private List<String> gbifProgramme = new ArrayList<>();
088
089  @Schema(description = "The manner in which GBIF is cited in a paper.")
090  private String citationType;
091
092  @Schema(description = "GBIF regions (political divisions) related to the publication.")
093  private Set<GbifRegion> gbifRegion = new HashSet<>();
094
095  @Schema(description = "GBIF identifier for this literature item.")
096  private UUID id;
097
098  @Schema(description = "Identifiers (such as DOIs) for the literature item.")
099  private Map<String, Object> identifiers = new HashMap<>();
100
101  @Schema(description = "Keywords assigned to the literature item.")
102  private List<String> keywords = new ArrayList<>();
103
104  @Schema(description = "The language of the publication.")
105  private Language language;
106
107  @Schema(description = "Type of literature, e.g. journal article.")
108  private LiteratureType literatureType;
109
110  @Schema(description = "The month of publication. See also `published`.", minimum = "1", maximum = "12")
111  private Integer month;
112
113  @Schema(description = "Unstructured notes.")
114  private String notes;
115
116  @Schema(description = "Whether the publication is open access.")
117  private boolean openAccess;
118
119  @Schema(description = "Whether the publication has been peer reviewed.")
120  private boolean peerReview;
121
122  @Schema(description = "The publisher of the paper.")
123  private String publisher;
124
125  @Schema(description = "Relevance to GBIF community, see [literature relevance](https://www.gbif.org/faq?question=literature-relevance).")
126  private Set<LiteratureRelevance> relevance = new HashSet<>();
127
128  @Schema(description = "Journal of publication.")
129  private String source;
130
131  @Schema(description = "Various tags applied to the literature.")
132  private List<String> tags = new ArrayList<>();
133
134  @Schema(description = "The title of the publication.")
135  private String title;
136
137  @Schema(description = "Topics of the publication.")
138  private Set<LiteratureTopic> topics = new HashSet<>();
139
140  @Schema(description = "The date this literature entry was last modified.")
141  private Date modified;
142
143  @Schema(description = "Websites associated with the publication.")
144  private List<String> websites = new ArrayList<>();
145
146  @Schema(description = "The year of publication.  See also `published`.")
147  private Integer year;
148
149  @Schema(name = "abstract", description = "The abstract from the publication.")
150  @JsonProperty("abstract")
151  public String getAbstract() {
152    return abstract_;
153  }
154
155  public void setAbstract(String abstract_) {
156    this.abstract_ = abstract_;
157  }
158
159}
