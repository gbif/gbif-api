045  /**
046   * Comparator that sorts pipeline steps by start date and then by finished date in an ascending
047   * order.
048   */
049  public static final Comparator<PipelineStep> STEPS_BY_START_AND_FINISH_ASC =
050      (s1, s2) -> {
051        LocalDateTime started1 = s1 != null ? s1.getStarted() : null;
052        LocalDateTime started2 = s2 != null ? s2.getStarted() : null;
053
054        if (started1 == null) {
055          return (started2 == null) ? 0 : 1;
056        } else if (started2 == null) {
057          return -1;
058        }
059
060        int comparison = started1.compareTo(started2);
061        if (comparison != 0) {
062          return comparison;
063        }
064
065        LocalDateTime finished1 = s1.getFinished();
066        LocalDateTime finished2 = s2.getFinished();
067        if (finished1 == null) {
068          return (finished2 == null) ? 0 : 1;
069        } else if (finished2 == null) {
070          return -1;
071        }
072        return finished1.compareTo(finished2);
073      };

189  /** Inner class to store metrics. */
190  public static class MetricInfo implements Serializable {
191
192    private static final long serialVersionUID = 1872427841009786709L;
193
194    private String name;
195    private String value;
196
197    public MetricInfo(String name, String value) {
198      this.name = name;
199      this.value = value;
200    }
201
202    public String getName() {
203      return name;
204    }
205
206    public String getValue() {
207      return value;
208    }
209
210    @Override
211    public boolean equals(Object o) {
212      if (this == o) {
213        return true;
214      }
215      if (o == null || getClass() != o.getClass()) {
216        return false;
217      }
218      MetricInfo that = (MetricInfo) o;
219      return Objects.equals(name, that.name) && Objects.equals(value, that.value);
220    }
221
222    @Override
223    public int hashCode() {
224      return Objects.hash(name, value);
225    }
226
227    @Override
228    public String toString() {
229      return new StringJoiner(", ", MetricInfo.class.getSimpleName() + "[", "]")
230          .add("name='" + name + "'")
231          .add("value='" + value + "'")
232          .toString();
233    }
234  }
235
236  @Override
237  public boolean equals(Object o) {
238    if (this == o) return true;
239    if (o == null || getClass() != o.getClass()) return false;
240    PipelineStep that = (PipelineStep) o;
241    return key == that.key
242        && Objects.equals(type, that.type)
243        && Objects.equals(runner, that.runner)
244        && Objects.equals(started, that.started)
245        && Objects.equals(finished, that.finished)
246        && state == that.state
247        && Objects.equals(message, that.message)
248        && Objects.equals(metrics, that.metrics)
249        && Objects.equals(rerunReason, that.rerunReason)
250        && Objects.equals(createdBy, that.createdBy)
251        && Objects.equals(modified, that.modified)
252        && Objects.equals(modifiedBy, that.modifiedBy);
253  }
254
255  @Override
256  public int hashCode() {
257    return Objects.hash(
258        key,
259        type,
260        runner,
261        started,
262        finished,
263        state,
264        message,
265        metrics,
266        rerunReason,
267        createdBy,
268        modified,
269        modifiedBy);
270  }
271
272  @Override
273  public String toString() {
274    return new StringJoiner(", ", PipelineStep.class.getSimpleName() + "[", "]")
275        .add("key=" + key)
276        .add("type=" + type)
277        .add("runner=" + runner)
278        .add("started=" + started)
279        .add("finished=" + finished)
280        .add("state=" + state)
281        .add("message='" + message + "'")
282        .add("rerunReason='" + rerunReason + "'")
283        .add("createdBy='" + createdBy + "'")
284        .add("modified=" + modified)
285        .add("modifiedBy='" + modifiedBy + "'")
286        .add("metrics=" + metrics)
287        .toString();
288  }
289
290  @Override
291  public boolean lenientEquals(PipelineStep other) {
292    if (this == other) return true;
293    return Objects.equals(type, other.type)
294        && Objects.equals(runner, other.runner)
295        && Objects.equals(finished, other.finished)
296        && state == other.state
297        && Objects.equals(message, other.message)
298        && Objects.equals(metrics, other.metrics)
299        && Objects.equals(rerunReason, other.rerunReason);
300  }
