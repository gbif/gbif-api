016package org.gbif.api.model.pipelines;
017
018import org.gbif.api.jackson.LocalDateTimeSerDe;
019
020import java.io.Serializable;
021import java.time.LocalDateTime;
022import java.util.ArrayList;
023import java.util.Comparator;
024import java.util.List;
025import java.util.Objects;
026import java.util.Set;
027import java.util.StringJoiner;
028import java.util.TreeSet;
029
030import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
031import com.fasterxml.jackson.databind.annotation.JsonSerialize;
032
033import static org.gbif.api.model.pipelines.PipelineStep.STEPS_BY_START_AND_FINISH_ASC;
034
035/**
036 * Models a execution of a pipeline that can include one or more steps.
037 */
038public class PipelineExecution implements Serializable {
039
040  private long key;
041  private List<StepType> stepsToRun = new ArrayList<>();
042  private String rerunReason;
043  private String remarks;
044
045  @JsonSerialize(using = LocalDateTimeSerDe.LocalDateTimeSerializer.class)
046  @JsonDeserialize(using = LocalDateTimeSerDe.LocalDateTimeDeserializer.class)
047  private LocalDateTime created;
048
049  private String createdBy;
050  private Set<PipelineStep> steps = new TreeSet<>(STEPS_BY_START_AND_FINISH_ASC);
051
052  public static final Comparator<PipelineExecution> PIPELINE_EXECUTION_BY_CREATED_ASC =
053    (e1, e2) -> {
054      LocalDateTime created1 = e1 != null ? e1.getCreated() : null;
055      LocalDateTime created2 = e2 != null ? e2.getCreated() : null;
056
057      if (created1 == null) {
058        return created2 == null ? 0 : 1;
059      } else if (created2 == null) {
060        return -1;
061      }
062
063      return created1.compareTo(created2);
064    };
065
066  public long getKey() {
067    return key;
068  }
069
070  public PipelineExecution setKey(long key) {
071    this.key = key;
072    return this;
073  }
074
075  public List<StepType> getStepsToRun() {
076    return stepsToRun;
077  }
078
079  public PipelineExecution setStepsToRun(List<StepType> stepsToRun) {
080    this.stepsToRun = stepsToRun;
081    return this;
082  }
083
084  public String getRerunReason() {
085    return rerunReason;
086  }
087
088  public PipelineExecution setRerunReason(String rerunReason) {
089    this.rerunReason = rerunReason;
090    return this;
091  }
092
093  public String getRemarks() {
094    return remarks;
095  }
096
097  public PipelineExecution setRemarks(String remarks) {
098    this.remarks = remarks;
099    return this;
100  }
101
102  public LocalDateTime getCreated() {
103    return created;
104  }
105
106  public PipelineExecution setCreated(LocalDateTime created) {
107    this.created = created;
108    return this;
109  }
110
111  public String getCreatedBy() {
112    return createdBy;
113  }
114
115  public PipelineExecution setCreatedBy(String createdBy) {
116    this.createdBy = createdBy;
117    return this;
118  }
119
120  public Set<PipelineStep> getSteps() {
121    return steps;
122  }
123
124  public void setSteps(Set<PipelineStep> steps) {
125    this.steps.clear();
126    this.steps.addAll(steps);
127  }
128
129  public PipelineExecution addStep(PipelineStep step) {
130    steps.add(step);
131    return this;
132  }
133
134  @Override
135  public boolean equals(Object o) {
136    if (this == o) return true;
137    if (o == null || getClass() != o.getClass()) return false;
138    PipelineExecution that = (PipelineExecution) o;
139    return key == that.key
140      && Objects.equals(stepsToRun, that.stepsToRun)
141      && Objects.equals(rerunReason, that.rerunReason)
142      && Objects.equals(remarks, that.remarks)
143      && Objects.equals(created, that.created)
144      && Objects.equals(createdBy, that.createdBy)
145      && Objects.equals(steps, that.steps);
146  }
147
148  @Override
149  public int hashCode() {
150    return Objects.hash(key, stepsToRun, rerunReason, remarks, created, createdBy);
151  }
152
153  @Override
154  public String toString() {
155    return new StringJoiner(", ", PipelineExecution.class.getSimpleName() + "[", "]")
156      .add("key=" + key)
157      .add("stepsToRun=" + stepsToRun)
158      .add("rerunReason='" + rerunReason + "'")
159      .add("remarks='" + remarks + "'")
160      .add("created=" + created)
161      .add("createdBy='" + createdBy + "'")
162      .add("steps='" + steps + "'")
163      .toString();
164  }
165}
