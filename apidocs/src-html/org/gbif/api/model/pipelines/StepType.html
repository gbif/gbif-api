001package org.gbif.api.model.pipelines;
002
003/** Enum to represent the pipelines step names. */
004public enum StepType {
005
006  //Generic step name used to be a replace of the following three when client doesn't know the endpoint type of a dataset
007  TO_VERBATIM("toVerbatim", 1),
008  DWCA_TO_VERBATIM("dwcaToVerbatim", 1),
009  XML_TO_VERBATIM("xmlToVerbatim", 1),
010  ABCD_TO_VERBATIM("abcdToVerbatim", 1),
011  VERBATIM_TO_INTERPRETED("verbatimToInterpreted", 2),
012  INTERPRETED_TO_INDEX("interpretedToIndex", 3),
013  HDFS_VIEW("hdfsView", 3);
014
015  private String label;
016  private int executionOrder;
017
018  StepType(String label, int executionOrder) {
019    this.label = label;
020    this.executionOrder = executionOrder;
021  }
022
023  public int getExecutionOrder() {
024    return executionOrder;
025  }
026
027  public String getLabel() {
028    return label;
029  }
030}
