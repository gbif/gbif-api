014package org.gbif.api.model.pipelines;
015
016/** Enum to represent the pipelines step names. */
017public enum StepType {
018
019  // Generic step name used to be a replace of the following three when client doesn't know the endpoint type of a dataset
020  // Interpretaion
021  TO_VERBATIM("toVerbatim", 1),
022  DWCA_TO_VERBATIM("dwcaToVerbatim", 1),
023  XML_TO_VERBATIM("xmlToVerbatim", 1),
024  ABCD_TO_VERBATIM("abcdToVerbatim", 1),
025  VERBATIM_TO_INTERPRETED("verbatimToInterpreted", 2),
026  INTERPRETED_TO_INDEX("interpretedToIndex", 3),
027  HDFS_VIEW("hdfsView", 3),
028  FRAGMENTER("fragmenter", 3),
029  // Validator
030  VALIDATOR_UPLOAD_ARCHIVE("validatorUploadArchive", 1),
031  VALIDATOR_VALIDATE_ARCHIVE("validatorValidateArchive", 2),
032  VALIDATOR_DWCA_TO_VERBATIM("validatorDwcaToVerbatim", 3),
033  VALIDATOR_XML_TO_VERBATIM("validatorXmlToVerbatim", 3),
034  VALIDATOR_ABCD_TO_VERBATIM("validatorAbcdToVerbatim", 4),
035  VALIDATOR_VERBATIM_TO_INTERPRETED("validatorVerbatimToInterpreted", 4),
036  VALIDATOR_INTERPRETED_TO_INDEX("validatorInterpretedToIndex", 5),
037  VALIDATOR_COLLECT_METRICS("validatorCollectMetrics", 6);
038
039  private String label;
040  private int executionOrder;
041
042  StepType(String label, int executionOrder) {
043    this.label = label;
044    this.executionOrder = executionOrder;
045  }
046
047  public int getExecutionOrder() {
048    return executionOrder;
049  }
050
051  public String getLabel() {
052    return label;
053  }
054}
