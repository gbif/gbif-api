001package org.gbif.api.model.pipelines;
002
003import java.util.Set;
004
005import com.google.common.collect.ImmutableSet;
006
007/** Enum to represent the pipelines step names. */
008public enum StepType {
009  ALL("all", 0),
010  DWCA_TO_VERBATIM("dwcaToVerbatim", 1),
011  XML_TO_VERBATIM("xmlToVerbatim", 1),
012  ABCD_TO_VERBATIM("abcdToVerbatim", 1),
013  VERBATIM_TO_INTERPRETED("verbatimToInterpreted", 2),
014  INTERPRETED_TO_INDEX("interpretedToIndex", 3),
015  HDFS_VIEW("hdfsView", 3);
016
017  public static final Set<StepType> ALL_STEPS =
018      ImmutableSet.of(
019          XML_TO_VERBATIM,
020          ABCD_TO_VERBATIM,
021          DWCA_TO_VERBATIM,
022          VERBATIM_TO_INTERPRETED,
023          INTERPRETED_TO_INDEX,
024          HDFS_VIEW);
025
026  private String label;
027  private int executionOrder;
028
029  StepType(String label, int executionOrder) {
030    this.label = label;
031    this.executionOrder = executionOrder;
032  }
033
034  public int getExecutionOrder() {
035    return executionOrder;
036  }
037
038  public String getLabel() {
039    return label;
040  }
041}
