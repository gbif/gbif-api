014package org.gbif.api.model.pipelines;
015
016import lombok.AllArgsConstructor;
017import lombok.Getter;
018
019import java.util.Set;
020
021/** Enum to represent the pipelines step names. */
022@Getter
023@AllArgsConstructor
024public enum StepType {
025
026  // Ingestion
027
028  // Verbatim
029  TO_VERBATIM("toVerbatim"),
030  DWCA_TO_VERBATIM("dwcaToVerbatim"),
031  XML_TO_VERBATIM("xmlToVerbatim"),
032  ABCD_TO_VERBATIM("abcdToVerbatim"),
033  FRAGMENTER("fragmenter"),
034
035  // Occurrence interpretation
036  VERBATIM_TO_IDENTIFIER("verbatimToIdentifier"),
037  VERBATIM_TO_INTERPRETED("verbatimToInterpreted"),
038  INTERPRETED_TO_INDEX("interpretedToIndex"),
039  HDFS_VIEW("hdfsView"),
040
041  // Event interpretation
042  EVENTS_VERBATIM_TO_INTERPRETED("eventsVerbatimToInterpreted"),
043  EVENTS_INTERPRETED_TO_INDEX("eventsInterpretedToIndex"),
044  EVENTS_HDFS_VIEW("eventsHdfsView"),
045
046  // Validator
047  VALIDATOR_UPLOAD_ARCHIVE("validatorUploadArchive"),
048  VALIDATOR_VALIDATE_ARCHIVE("validatorValidateArchive"),
049  VALIDATOR_DWCA_TO_VERBATIM("validatorDwcaToVerbatim"),
050  VALIDATOR_XML_TO_VERBATIM("validatorXmlToVerbatim"),
051  VALIDATOR_ABCD_TO_VERBATIM("validatorAbcdToVerbatim"),
052  VALIDATOR_TABULAR_TO_VERBATIM("validatorTabularToVerbatim"),
053  VALIDATOR_VERBATIM_TO_INTERPRETED("validatorVerbatimToInterpreted"),
054  VALIDATOR_INTERPRETED_TO_INDEX("validatorInterpretedToIndex"),
055  VALIDATOR_COLLECT_METRICS("validatorCollectMetrics");
056
057  private final String label;
058
059  private static final Set<StepType> VERBATIM_TYPE =
060    Set.of(TO_VERBATIM, DWCA_TO_VERBATIM, XML_TO_VERBATIM, ABCD_TO_VERBATIM, FRAGMENTER);
061
062  private static final Set<StepType> OCCURRENCE_TYPE =
063    Set.of(VERBATIM_TO_IDENTIFIER, VERBATIM_TO_INTERPRETED, INTERPRETED_TO_INDEX, HDFS_VIEW);
064
065  private static final Set<StepType> EVENT_TYPE =
066    Set.of(EVENTS_VERBATIM_TO_INTERPRETED, EVENTS_INTERPRETED_TO_INDEX, EVENTS_HDFS_VIEW);
067
068  public static boolean isEventType(StepType type) {
069    return EVENT_TYPE.contains(type);
070  }
071
072  public static boolean isOccurrenceType(StepType type) {
073    return OCCURRENCE_TYPE.contains(type);
074  }
075
076  public static boolean isVerbatimType(StepType type) {
077    return VERBATIM_TYPE.contains(type);
078  }
079}
