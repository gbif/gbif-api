016package org.gbif.api.model.pipelines;
017
018import org.gbif.api.jackson.LocalDateTimeSerDe;
019import org.gbif.common.shaded.com.fasterxml.jackson.annotation.JsonInclude;
020
021import java.io.Serializable;
022import java.time.LocalDateTime;
023import java.util.*;
024
025import org.codehaus.jackson.map.annotate.JsonDeserialize;
026import org.codehaus.jackson.map.annotate.JsonSerialize;
027
028import static org.gbif.api.model.pipelines.PipelineExecution.PIPELINE_EXECUTION_BY_CREATED_ASC;
029import static org.gbif.api.model.pipelines.PipelineStep.Status.RUNNING;
030
031/** Models a pipeline process for a specific dataset and attempt. */
032public class PipelineProcess implements Serializable {
033
034  private static final long serialVersionUID = -3992826055732414678L;
035
036  @JsonInclude(JsonInclude.Include.NON_DEFAULT)
037  private long key;
038
039  private UUID datasetKey;
040  private String datasetTitle;
041  private int attempt;
042
043  @JsonSerialize(using = LocalDateTimeSerDe.LocalDateTimeSerializer.class)
044  @JsonDeserialize(using = LocalDateTimeSerDe.LocalDateTimeDeserializer.class)
045  private LocalDateTime created;
046
047  private String createdBy;
048  private Set<PipelineExecution> executions =
049      new TreeSet<>(PIPELINE_EXECUTION_BY_CREATED_ASC.reversed());
050
051  /**
052   * Comparator that sorts the pipeline processes by the created date of the latest execution.
053   */
054  public static final Comparator<PipelineProcess> PIPELINE_PROCESS_BY_LATEST_EXEUCTION_ASC =
055      (p1, p2) -> {
056        if (p1 == null) {
057          return p2 == null ? 0 : 1;
058        } else if (p2 == null) {
059          return -1;
060        }
061
062        PipelineExecution latestExecution1 =
063            p1.getExecutions().isEmpty() ? null : p1.getExecutions().iterator().next();
064        PipelineExecution latestExecution2 =
065            p2.getExecutions().isEmpty() ? null : p2.getExecutions().iterator().next();
066
067        return Objects.compare(
068            latestExecution1, latestExecution2, PIPELINE_EXECUTION_BY_CREATED_ASC);
069      };
070
071  /**
072   * Comparator that sorts pipeline processes by the start date of their latest step in an ascending
073   * order. The steps that are running have preference and we take into account only the steps of
074   * the latest execution.
075   */
076  public static final Comparator<PipelineProcess> PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC =
077      (p1, p2) -> {
078        if (p1 == null) {
079          return p2 == null ? 0 : 1;
080        } else if (p2 == null) {
081          return -1;
082        }
083
084        PipelineExecution latestExecution1 =
085            p1.getExecutions().isEmpty() ? null : p1.getExecutions().iterator().next();
086        PipelineExecution latestExecution2 =
087            p2.getExecutions().isEmpty() ? null : p2.getExecutions().iterator().next();
088
089        Optional<PipelineStep> lastStepOpt1 = Optional.empty();
090        if (latestExecution1 != null && latestExecution1.getSteps() != null) {
091          lastStepOpt1 =
092              latestExecution1.getSteps().stream()
093                  .filter(p -> p.getState() != null)
094                  .max(Comparator.comparing(PipelineStep::getStarted));
095        }
096
097        Optional<PipelineStep> lastStepOpt2 = Optional.empty();
098        if (latestExecution2 != null && latestExecution2.getSteps() != null) {
099          lastStepOpt2 =
100              latestExecution2.getSteps().stream()
101                  .filter(p -> p.getState() != null)
102                  .max(Comparator.comparing(PipelineStep::getStarted));
103        }
104
105        if (!lastStepOpt1.isPresent()) {
106          return !lastStepOpt2.isPresent() ? 0 : 1;
107        } else if (!lastStepOpt2.isPresent()) {
108          return -1;
109        }
110
111        PipelineStep step1 = lastStepOpt1.get();
112        PipelineStep step2 = lastStepOpt2.get();
113
114        if (step1.getStarted() == null) {
115          return step2.getStarted() == null ? 0 : 1;
116        } else if (step2.getStarted() == null) {
117          return -1;
118        }
119
120        // steps that are running have preference
121        if (step1.getState() == RUNNING) {
122          return step2.getState() == RUNNING ? step1.getStarted().compareTo(step2.getStarted()) : 1;
123        } else if (step2.getState() == RUNNING) {
124          return -1;
125        } else {
126          return step1.getStarted().compareTo(step2.getStarted());
127        }
128      };
129
130  public long getKey() {
131    return key;
132  }
133
134  public UUID getDatasetKey() {
135    return datasetKey;
136  }
137
138  public PipelineProcess setDatasetKey(UUID datasetKey) {
139    this.datasetKey = datasetKey;
140    return this;
141  }
142
143  public String getDatasetTitle() {
144    return datasetTitle;
145  }
146
147  public void setDatasetTitle(String datasetTitle) {
148    this.datasetTitle = datasetTitle;
149  }
150
151  public int getAttempt() {
152    return attempt;
153  }
154
155  public PipelineProcess setAttempt(int attempt) {
156    this.attempt = attempt;
157    return this;
158  }
159
160  public LocalDateTime getCreated() {
161    return created;
162  }
163
164  public PipelineProcess setCreated(LocalDateTime created) {
165    this.created = created;
166    return this;
167  }
168
169  public String getCreatedBy() {
170    return createdBy;
171  }
172
173  public PipelineProcess setCreatedBy(String createdBy) {
174    this.createdBy = createdBy;
175    return this;
176  }
177
178  public Set<PipelineExecution> getExecutions() {
179    return executions;
180  }
181
182  public void setExecutions(Set<PipelineExecution> executions) {
183    this.executions.clear();
184    this.executions.addAll(executions);
185  }
186
187  public PipelineProcess addExecution(PipelineExecution execution) {
188    executions.add(execution);
189    return this;
190  }
191
192  @Override
193  public String toString() {
194    return new StringJoiner(", ", PipelineProcess.class.getSimpleName() + "[", "]")
195        .add("key=" + key)
196        .add("datasetKey=" + datasetKey)
197        .add("datasetTitle=" + datasetTitle)
198        .add("attempt=" + attempt)
199        .add("created=" + created)
200        .add("createdBy='" + createdBy + "'")
201        .add("executions=" + executions)
202        .toString();
203  }
204
205  @Override
206  public boolean equals(Object o) {
207    if (this == o) return true;
208    if (o == null || getClass() != o.getClass()) return false;
209    PipelineProcess process = (PipelineProcess) o;
210    return attempt == process.attempt && Objects.equals(datasetKey, process.datasetKey);
211  }
212
213  @Override
214  public int hashCode() {
215    return Objects.hash(datasetKey, attempt);
216  }
217}
