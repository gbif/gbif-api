016package org.gbif.api.model.pipelines;
017
018import org.gbif.api.jackson.LocalDateTimeSerDe;
019
020import java.io.Serializable;
021import java.time.LocalDateTime;
022import java.util.Comparator;
023import java.util.Objects;
024import java.util.Optional;
025import java.util.Set;
026import java.util.StringJoiner;
027import java.util.TreeSet;
028import java.util.UUID;
029
030import com.fasterxml.jackson.annotation.JsonInclude;
031import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
032import com.fasterxml.jackson.databind.annotation.JsonSerialize;
033
034import static org.gbif.api.model.pipelines.PipelineExecution.PIPELINE_EXECUTION_BY_CREATED_ASC;
035import static org.gbif.api.model.pipelines.PipelineStep.Status.RUNNING;
036
037/**
038 * Models a pipeline process for a specific dataset and attempt.
039 */
040public class PipelineProcess implements Serializable {
041
042  private static final long serialVersionUID = -3992826055732414678L;
043
044  @JsonInclude(JsonInclude.Include.NON_DEFAULT)
045  private long key;
046
047  private UUID datasetKey;
048  private String datasetTitle;
049  private int attempt;
050
051  @JsonSerialize(using = LocalDateTimeSerDe.LocalDateTimeSerializer.class)
052  @JsonDeserialize(using = LocalDateTimeSerDe.LocalDateTimeDeserializer.class)
053  private LocalDateTime created;
054
055  private String createdBy;
056  private Set<PipelineExecution> executions =
057    new TreeSet<>(PIPELINE_EXECUTION_BY_CREATED_ASC.reversed());
058
059  /**
060   * Comparator that sorts the pipeline processes by the created date of the latest execution.
061   */
062  public static final Comparator<PipelineProcess> PIPELINE_PROCESS_BY_LATEST_EXEUCTION_ASC =
063    (p1, p2) -> {
064      if (p1 == null) {
065        return p2 == null ? 0 : 1;
066      } else if (p2 == null) {
067        return -1;
068      }
069
070      PipelineExecution latestExecution1 =
071        p1.getExecutions().isEmpty() ? null : p1.getExecutions().iterator().next();
072      PipelineExecution latestExecution2 =
073        p2.getExecutions().isEmpty() ? null : p2.getExecutions().iterator().next();
074
075      return Objects.compare(
076        latestExecution1, latestExecution2, PIPELINE_EXECUTION_BY_CREATED_ASC);
077    };
078
079  /**
080   * Comparator that sorts pipeline processes by the start date of their latest step in an ascending
081   * order. The steps that are running have preference and we take into account only the steps of
082   * the latest execution.
083   */
084  public static final Comparator<PipelineProcess> PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC =
085    (p1, p2) -> {
086      if (p1 == null) {
087        return p2 == null ? 0 : 1;
088      } else if (p2 == null) {
089        return -1;
090      }
091
092      PipelineExecution latestExecution1 =
093        p1.getExecutions().isEmpty() ? null : p1.getExecutions().iterator().next();
094      PipelineExecution latestExecution2 =
095        p2.getExecutions().isEmpty() ? null : p2.getExecutions().iterator().next();
096
097      Optional<PipelineStep> lastStepOpt1 = Optional.empty();
098      if (latestExecution1 != null && latestExecution1.getSteps() != null) {
099        lastStepOpt1 =
100          latestExecution1.getSteps().stream()
101            .filter(p -> p.getState() != null)
102            .max(Comparator.comparing(PipelineStep::getStarted));
103      }
104
105      Optional<PipelineStep> lastStepOpt2 = Optional.empty();
106      if (latestExecution2 != null && latestExecution2.getSteps() != null) {
107        lastStepOpt2 =
108          latestExecution2.getSteps().stream()
109            .filter(p -> p.getState() != null)
110            .max(Comparator.comparing(PipelineStep::getStarted));
111      }
112
113      if (!lastStepOpt1.isPresent()) {
114        return !lastStepOpt2.isPresent() ? 0 : 1;
115      } else if (!lastStepOpt2.isPresent()) {
116        return -1;
117      }
118
119      PipelineStep step1 = lastStepOpt1.get();
120      PipelineStep step2 = lastStepOpt2.get();
121
122      if (step1.getStarted() == null) {
123        return step2.getStarted() == null ? 0 : 1;
124      } else if (step2.getStarted() == null) {
125        return -1;
126      }
127
128      // steps that are running have preference
129      if (step1.getState() == RUNNING) {
130        return step2.getState() == RUNNING ? step1.getStarted().compareTo(step2.getStarted()) : 1;
131      } else if (step2.getState() == RUNNING) {
132        return -1;
133      } else {
134        return step1.getStarted().compareTo(step2.getStarted());
135      }
136    };
137
138  public long getKey() {
139    return key;
140  }
141
142  public UUID getDatasetKey() {
143    return datasetKey;
144  }
145
146  public PipelineProcess setDatasetKey(UUID datasetKey) {
147    this.datasetKey = datasetKey;
148    return this;
149  }
150
151  public String getDatasetTitle() {
152    return datasetTitle;
153  }
154
155  public void setDatasetTitle(String datasetTitle) {
156    this.datasetTitle = datasetTitle;
157  }
158
159  public int getAttempt() {
160    return attempt;
161  }
162
163  public PipelineProcess setAttempt(int attempt) {
164    this.attempt = attempt;
165    return this;
166  }
167
168  public LocalDateTime getCreated() {
169    return created;
170  }
171
172  public PipelineProcess setCreated(LocalDateTime created) {
173    this.created = created;
174    return this;
175  }
176
177  public String getCreatedBy() {
178    return createdBy;
179  }
180
181  public PipelineProcess setCreatedBy(String createdBy) {
182    this.createdBy = createdBy;
183    return this;
184  }
185
186  public Set<PipelineExecution> getExecutions() {
187    return executions;
188  }
189
190  public void setExecutions(Set<PipelineExecution> executions) {
191    this.executions.clear();
192    this.executions.addAll(executions);
193  }
194
195  public PipelineProcess addExecution(PipelineExecution execution) {
196    executions.add(execution);
197    return this;
198  }
199
200  @Override
201  public String toString() {
202    return new StringJoiner(", ", PipelineProcess.class.getSimpleName() + "[", "]")
203      .add("key=" + key)
204      .add("datasetKey=" + datasetKey)
205      .add("datasetTitle=" + datasetTitle)
206      .add("attempt=" + attempt)
207      .add("created=" + created)
208      .add("createdBy='" + createdBy + "'")
209      .add("executions=" + executions)
210      .toString();
211  }
212
213  @Override
214  public boolean equals(Object o) {
215    if (this == o) return true;
216    if (o == null || getClass() != o.getClass()) return false;
217    PipelineProcess process = (PipelineProcess) o;
218    return attempt == process.attempt && Objects.equals(datasetKey, process.datasetKey);
219  }
220
221  @Override
222  public int hashCode() {
223    return Objects.hash(datasetKey, attempt);
224  }
225}
