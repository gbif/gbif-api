001package org.gbif.api.model.pipelines;
002
003import org.gbif.api.jackson.LocalDateTimeSerDe;
004import org.gbif.common.shaded.com.fasterxml.jackson.annotation.JsonInclude;
005
006import java.io.Serializable;
007import java.time.LocalDateTime;
008import java.util.*;
009
010import org.codehaus.jackson.map.annotate.JsonDeserialize;
011import org.codehaus.jackson.map.annotate.JsonSerialize;
012
013import static org.gbif.api.model.pipelines.PipelineExecution.PIPELINE_EXECUTION_BY_CREATED_ASC;
014import static org.gbif.api.model.pipelines.PipelineStep.Status.RUNNING;
015
016/** Models a pipeline process for a specific dataset and attempt. */
017public class PipelineProcess implements Serializable {
018
019  private static final long serialVersionUID = -3992826055732414678L;
020
021  @JsonInclude(JsonInclude.Include.NON_DEFAULT)
022  private long key;
023
024  private UUID datasetKey;
025  private String datasetTitle;
026  private int attempt;
027
028  @JsonSerialize(using = LocalDateTimeSerDe.LocalDateTimeSerializer.class)
029  @JsonDeserialize(using = LocalDateTimeSerDe.LocalDateTimeDeserializer.class)
030  private LocalDateTime created;
031
032  private String createdBy;
033  private Set<PipelineExecution> executions =
034      new TreeSet<>(PIPELINE_EXECUTION_BY_CREATED_ASC.reversed());
035
036  /**
037   * Comparator that sorts the pipeline processes by the created date of the latest execution.
038   */
039  public static final Comparator<PipelineProcess> PIPELINE_PROCESS_BY_LATEST_EXEUCTION_ASC =
040      (p1, p2) -> {
041        if (p1 == null) {
042          return p2 == null ? 0 : 1;
043        } else if (p2 == null) {
044          return -1;
045        }
046
047        PipelineExecution latestExecution1 =
048            p1.getExecutions().isEmpty() ? null : p1.getExecutions().iterator().next();
049        PipelineExecution latestExecution2 =
050            p2.getExecutions().isEmpty() ? null : p2.getExecutions().iterator().next();
051
052        return Objects.compare(
053            latestExecution1, latestExecution2, PIPELINE_EXECUTION_BY_CREATED_ASC);
054      };
055
056  /**
057   * Comparator that sorts pipeline processes by the start date of their latest step in an ascending
058   * order. The steps that are running have preference and we take into account only the steps of
059   * the latest execution.
060   */
061  public static final Comparator<PipelineProcess> PIPELINE_PROCESS_BY_LATEST_STEP_RUNNING_ASC =
062      (p1, p2) -> {
063        if (p1 == null) {
064          return p2 == null ? 0 : 1;
065        } else if (p2 == null) {
066          return -1;
067        }
068
069        PipelineExecution latestExecution1 =
070            p1.getExecutions().isEmpty() ? null : p1.getExecutions().iterator().next();
071        PipelineExecution latestExecution2 =
072            p2.getExecutions().isEmpty() ? null : p2.getExecutions().iterator().next();
073
074        Optional<PipelineStep> lastStepOpt1 = Optional.empty();
075        if (latestExecution1 != null && latestExecution1.getSteps() != null) {
076          lastStepOpt1 =
077              latestExecution1.getSteps().stream()
078                  .filter(p -> p.getState() != null)
079                  .max(Comparator.comparing(PipelineStep::getStarted));
080        }
081
082        Optional<PipelineStep> lastStepOpt2 = Optional.empty();
083        if (latestExecution2 != null && latestExecution2.getSteps() != null) {
084          lastStepOpt2 =
085              latestExecution2.getSteps().stream()
086                  .filter(p -> p.getState() != null)
087                  .max(Comparator.comparing(PipelineStep::getStarted));
088        }
089
090        if (!lastStepOpt1.isPresent()) {
091          return !lastStepOpt2.isPresent() ? 0 : 1;
092        } else if (!lastStepOpt2.isPresent()) {
093          return -1;
094        }
095
096        PipelineStep step1 = lastStepOpt1.get();
097        PipelineStep step2 = lastStepOpt2.get();
098
099        if (step1.getStarted() == null) {
100          return step2.getStarted() == null ? 0 : 1;
101        } else if (step2.getStarted() == null) {
102          return -1;
103        }
104
105        // steps that are running have preference
106        if (step1.getState() == RUNNING) {
107          return step2.getState() == RUNNING ? step1.getStarted().compareTo(step2.getStarted()) : 1;
108        } else if (step2.getState() == RUNNING) {
109          return -1;
110        } else {
111          return step1.getStarted().compareTo(step2.getStarted());
112        }
113      };
114
115  public long getKey() {
116    return key;
117  }
118
119  public UUID getDatasetKey() {
120    return datasetKey;
121  }
122
123  public PipelineProcess setDatasetKey(UUID datasetKey) {
124    this.datasetKey = datasetKey;
125    return this;
126  }
127
128  public String getDatasetTitle() {
129    return datasetTitle;
130  }
131
132  public void setDatasetTitle(String datasetTitle) {
133    this.datasetTitle = datasetTitle;
134  }
135
136  public int getAttempt() {
137    return attempt;
138  }
139
140  public PipelineProcess setAttempt(int attempt) {
141    this.attempt = attempt;
142    return this;
143  }
144
145  public LocalDateTime getCreated() {
146    return created;
147  }
148
149  public PipelineProcess setCreated(LocalDateTime created) {
150    this.created = created;
151    return this;
152  }
153
154  public String getCreatedBy() {
155    return createdBy;
156  }
157
158  public PipelineProcess setCreatedBy(String createdBy) {
159    this.createdBy = createdBy;
160    return this;
161  }
162
163  public Set<PipelineExecution> getExecutions() {
164    return executions;
165  }
166
167  public void setExecutions(Set<PipelineExecution> executions) {
168    this.executions.clear();
169    this.executions.addAll(executions);
170  }
171
172  public PipelineProcess addExecution(PipelineExecution execution) {
173    executions.add(execution);
174    return this;
175  }
176
177  @Override
178  public String toString() {
179    return new StringJoiner(", ", PipelineProcess.class.getSimpleName() + "[", "]")
180        .add("key=" + key)
181        .add("datasetKey=" + datasetKey)
182        .add("datasetTitle=" + datasetTitle)
183        .add("attempt=" + attempt)
184        .add("created=" + created)
185        .add("createdBy='" + createdBy + "'")
186        .add("executions=" + executions)
187        .toString();
188  }
189
190  @Override
191  public boolean equals(Object o) {
192    if (this == o) return true;
193    if (o == null || getClass() != o.getClass()) return false;
194    PipelineProcess process = (PipelineProcess) o;
195    return attempt == process.attempt && Objects.equals(datasetKey, process.datasetKey);
196  }
197
198  @Override
199  public int hashCode() {
200    return Objects.hash(datasetKey, attempt);
201  }
202}
