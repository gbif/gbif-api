001package org.gbif.api.model.pipelines;
002
003import org.gbif.api.jackson.LocalDateTimeSerDe;
004import org.gbif.common.shaded.com.fasterxml.jackson.annotation.JsonInclude;
005
006import java.io.Serializable;
007import java.time.LocalDateTime;
008import java.util.*;
009
010import org.codehaus.jackson.map.annotate.JsonDeserialize;
011import org.codehaus.jackson.map.annotate.JsonSerialize;
012
013/** Base POJO model for the Pipelines status service */
014public class PipelineProcess implements Serializable {
015
016  private static final long serialVersionUID = -3992826055732414678L;
017
018  public static final Comparator<PipelineStep> STEPS_COMPARATOR =
019      (s1, s2) -> {
020        LocalDateTime started1 = s1 != null ? s1.getStarted() : null;
021        LocalDateTime started2 = s2 != null ? s2.getStarted() : null;
022
023        if (started1 == null) {
024          return (started2 == null) ? 0 : 1;
025        } else if (started2 == null) {
026          return -1;
027        } else {
028          int comparison = started1.compareTo(started2);
029          if (comparison != 0) {
030            return comparison;
031          } else {
032            LocalDateTime finished1 = s1.getFinished();
033            LocalDateTime finished2 = s2.getFinished();
034            if (finished1 == null) {
035              return (finished2 == null) ? 0 : 1;
036            } else if (finished2 == null) {
037              return -1;
038            } else {
039              return finished1.compareTo(finished2);
040            }
041          }
042        }
043      };
044
045  @JsonInclude(JsonInclude.Include.NON_DEFAULT)
046  private long key;
047
048  private UUID datasetKey;
049  private String datasetTitle;
050  private int attempt;
051  private long numberRecords;
052
053  @JsonSerialize(using = LocalDateTimeSerDe.LocalDateTimeSerializer.class)
054  @JsonDeserialize(using = LocalDateTimeSerDe.LocalDateTimeDeserializer.class)
055  private LocalDateTime created;
056
057  private String createdBy;
058  private Set<PipelineStep> steps = new TreeSet<>(STEPS_COMPARATOR);
059
060  public long getKey() {
061    return key;
062  }
063
064  public UUID getDatasetKey() {
065    return datasetKey;
066  }
067
068  public PipelineProcess setDatasetKey(UUID datasetKey) {
069    this.datasetKey = datasetKey;
070    return this;
071  }
072
073  public String getDatasetTitle() {
074    return datasetTitle;
075  }
076
077  public void setDatasetTitle(String datasetTitle) {
078    this.datasetTitle = datasetTitle;
079  }
080
081  public int getAttempt() {
082    return attempt;
083  }
084
085  public PipelineProcess setAttempt(int attempt) {
086    this.attempt = attempt;
087    return this;
088  }
089
090  public long getNumberRecords() {
091    return numberRecords;
092  }
093
094  public void setNumberRecords(long numberRecords) {
095    this.numberRecords = numberRecords;
096  }
097
098  public LocalDateTime getCreated() {
099    return created;
100  }
101
102  public PipelineProcess setCreated(LocalDateTime created) {
103    this.created = created;
104    return this;
105  }
106
107  public String getCreatedBy() {
108    return createdBy;
109  }
110
111  public PipelineProcess setCreatedBy(String createdBy) {
112    this.createdBy = createdBy;
113    return this;
114  }
115
116  public Set<PipelineStep> getSteps() {
117    return steps;
118  }
119
120  public void setSteps(Set<PipelineStep> steps) {
121    this.steps.clear();
122    this.steps.addAll(steps);
123  }
124
125  public void addStep(PipelineStep step) {
126    steps.add(step);
127  }
128
129  @Override
130  public String toString() {
131    return new StringJoiner(", ", PipelineProcess.class.getSimpleName() + "[", "]")
132        .add("key=" + key)
133        .add("datasetKey=" + datasetKey)
134        .add("datasetTitle=" + datasetTitle)
135        .add("attempt=" + attempt)
136        .add("numberRecords=" + numberRecords)
137        .add("created=" + created)
138        .add("createdBy='" + createdBy + "'")
139        .add("steps=" + steps)
140        .toString();
141  }
142
143  @Override
144  public boolean equals(Object o) {
145    if (this == o) return true;
146    if (o == null || getClass() != o.getClass()) return false;
147    PipelineProcess process = (PipelineProcess) o;
148    return attempt == process.attempt && Objects.equals(datasetKey, process.datasetKey);
149  }
150
151  @Override
152  public int hashCode() {
153    return Objects.hash(datasetKey, attempt);
154  }
155}
