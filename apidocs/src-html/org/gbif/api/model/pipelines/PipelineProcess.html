001package org.gbif.api.model.pipelines;
002
003import org.gbif.api.jackson.LocalDateTimeSerDe;
004import org.gbif.common.shaded.com.fasterxml.jackson.annotation.JsonInclude;
005
006import java.io.Serializable;
007import java.time.LocalDateTime;
008import java.util.*;
009
010import org.codehaus.jackson.map.annotate.JsonDeserialize;
011import org.codehaus.jackson.map.annotate.JsonSerialize;
012
013import static org.gbif.api.model.pipelines.PipelineStep.STEPS_BY_START_AND_FINISH_ASC;
014import static org.gbif.api.model.pipelines.PipelineStep.Status.RUNNING;
015
016/** Base POJO model for the Pipelines status service */
017public class PipelineProcess implements Serializable {
018
019  private static final long serialVersionUID = -3992826055732414678L;
020
021  @JsonInclude(JsonInclude.Include.NON_DEFAULT)
022  private long key;
023
024  private UUID datasetKey;
025  private String datasetTitle;
026  private int attempt;
027  private long numberRecords;
028
029  @JsonSerialize(using = LocalDateTimeSerDe.LocalDateTimeSerializer.class)
030  @JsonDeserialize(using = LocalDateTimeSerDe.LocalDateTimeDeserializer.class)
031  private LocalDateTime created;
032
033  private String createdBy;
034  private Set<PipelineStep> steps = new TreeSet<>(STEPS_BY_START_AND_FINISH_ASC);
035
036  /**
037   * Comparator that sorts pipeline processes by the start date of their latest step in an ascending
038   * order. The steps that are running have preference.
039   */
040  public static final Comparator<PipelineProcess> PIPELINE_PROCESS_BY_LATEST_STEP_ASC =
041      (p1, p2) -> {
042        Optional<PipelineStep> lastStepOpt1 = Optional.empty();
043        if (p1 != null && p1.getSteps() != null) {
044          lastStepOpt1 =
045              p1.getSteps().stream()
046                  .filter(p -> p.getState() != null)
047                  .max(Comparator.comparing(PipelineStep::getStarted));
048        }
049
050        Optional<PipelineStep> lastStepOpt2 = Optional.empty();
051        if (p2 != null && p2.getSteps() != null) {
052          lastStepOpt2 =
053              p2.getSteps().stream()
054                  .filter(p -> p.getState() != null)
055                  .max(Comparator.comparing(PipelineStep::getStarted));
056        }
057
058        if (!lastStepOpt1.isPresent()) {
059          return !lastStepOpt2.isPresent() ? 0 : 1;
060        } else if (!lastStepOpt2.isPresent()) {
061          return -1;
062        }
063
064        PipelineStep step1 = lastStepOpt1.get();
065        PipelineStep step2 = lastStepOpt2.get();
066
067        if (step1.getStarted() == null) {
068          return step2.getStarted() == null ? 0 : 1;
069        } else if (step2.getStarted() == null) {
070          return -1;
071        }
072
073        // steps that are running have preference
074        if (step1.getState() == RUNNING) {
075          return step2.getState() == RUNNING ? step1.getStarted().compareTo(step2.getStarted()) : 1;
076        } else if (step2.getState() == RUNNING) {
077          return -1;
078        } else {
079          return step1.getStarted().compareTo(step2.getStarted());
080        }
081      };
082
083  public long getKey() {
084    return key;
085  }
086
087  public UUID getDatasetKey() {
088    return datasetKey;
089  }
090
091  public PipelineProcess setDatasetKey(UUID datasetKey) {
092    this.datasetKey = datasetKey;
093    return this;
094  }
095
096  public String getDatasetTitle() {
097    return datasetTitle;
098  }
099
100  public void setDatasetTitle(String datasetTitle) {
101    this.datasetTitle = datasetTitle;
102  }
103
104  public int getAttempt() {
105    return attempt;
106  }
107
108  public PipelineProcess setAttempt(int attempt) {
109    this.attempt = attempt;
110    return this;
111  }
112
113  public long getNumberRecords() {
114    return numberRecords;
115  }
116
117  public void setNumberRecords(long numberRecords) {
118    this.numberRecords = numberRecords;
119  }
120
121  public LocalDateTime getCreated() {
122    return created;
123  }
124
125  public PipelineProcess setCreated(LocalDateTime created) {
126    this.created = created;
127    return this;
128  }
129
130  public String getCreatedBy() {
131    return createdBy;
132  }
133
134  public PipelineProcess setCreatedBy(String createdBy) {
135    this.createdBy = createdBy;
136    return this;
137  }
138
139  public Set<PipelineStep> getSteps() {
140    return steps;
141  }
142
143  public void setSteps(Set<PipelineStep> steps) {
144    this.steps.clear();
145    this.steps.addAll(steps);
146  }
147
148  public void addStep(PipelineStep step) {
149    steps.add(step);
150  }
151
152  @Override
153  public String toString() {
154    return new StringJoiner(", ", PipelineProcess.class.getSimpleName() + "[", "]")
155        .add("key=" + key)
156        .add("datasetKey=" + datasetKey)
157        .add("datasetTitle=" + datasetTitle)
158        .add("attempt=" + attempt)
159        .add("numberRecords=" + numberRecords)
160        .add("created=" + created)
161        .add("createdBy='" + createdBy + "'")
162        .add("steps=" + steps)
163        .toString();
164  }
165
166  @Override
167  public boolean equals(Object o) {
168    if (this == o) return true;
169    if (o == null || getClass() != o.getClass()) return false;
170    PipelineProcess process = (PipelineProcess) o;
171    return attempt == process.attempt && Objects.equals(datasetKey, process.datasetKey);
172  }
173
174  @Override
175  public int hashCode() {
176    return Objects.hash(datasetKey, attempt);
177  }
178}
