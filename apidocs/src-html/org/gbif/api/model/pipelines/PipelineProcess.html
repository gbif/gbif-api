001package org.gbif.api.model.pipelines;
002
003import org.gbif.api.jackson.LocalDateTimeSerDe;
004import org.gbif.common.shaded.com.fasterxml.jackson.annotation.JsonInclude;
005
006import java.io.Serializable;
007import java.time.LocalDateTime;
008import java.util.*;
009
010import org.codehaus.jackson.map.annotate.JsonDeserialize;
011import org.codehaus.jackson.map.annotate.JsonSerialize;
012
013/** Base POJO model for the Pipelines status service */
014public class PipelineProcess implements Serializable {
015
016  private static final long serialVersionUID = -3992826055732414678L;
017
018  @JsonInclude(JsonInclude.Include.NON_DEFAULT)
019  private long key;
020
021  private UUID datasetKey;
022  private int attempt;
023
024  @JsonSerialize(using = LocalDateTimeSerDe.LocalDateTimeSerializer.class)
025  @JsonDeserialize(using = LocalDateTimeSerDe.LocalDateTimeDeserializer.class)
026  private LocalDateTime created;
027
028  private String createdBy;
029  private Set<PipelineStep> steps = new TreeSet<>(Comparator.comparing(PipelineStep::getStarted));
030
031  public long getKey() {
032    return key;
033  }
034
035  public UUID getDatasetKey() {
036    return datasetKey;
037  }
038
039  public PipelineProcess setDatasetKey(UUID datasetKey) {
040    this.datasetKey = datasetKey;
041    return this;
042  }
043
044  public int getAttempt() {
045    return attempt;
046  }
047
048  public PipelineProcess setAttempt(int attempt) {
049    this.attempt = attempt;
050    return this;
051  }
052
053  public LocalDateTime getCreated() {
054    return created;
055  }
056
057  public PipelineProcess setCreated(LocalDateTime created) {
058    this.created = created;
059    return this;
060  }
061
062  public String getCreatedBy() {
063    return createdBy;
064  }
065
066  public PipelineProcess setCreatedBy(String createdBy) {
067    this.createdBy = createdBy;
068    return this;
069  }
070
071  public Set<PipelineStep> getSteps() {
072    return steps;
073  }
074
075  public void setSteps(Set<PipelineStep> steps) {
076    this.steps.clear();
077    this.steps.addAll(steps);
078  }
079
080  public void addStep(PipelineStep step) {
081    steps.add(step);
082  }
083}
