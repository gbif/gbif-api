001package org.gbif.api.model.pipelines;
002
003import org.gbif.api.jackson.LocalDateTimeSerDe;
004import org.gbif.api.model.registry.LenientEquals;
005import org.gbif.common.shaded.com.fasterxml.jackson.annotation.JsonInclude;
006
007import java.io.Serializable;
008import java.time.LocalDateTime;
009import java.util.HashSet;
010import java.util.Objects;
011import java.util.Set;
012import java.util.StringJoiner;
013
014import org.codehaus.jackson.map.annotate.JsonDeserialize;
015import org.codehaus.jackson.map.annotate.JsonSerialize;
016
017/** Models a step in pipelines. */
018public class PipelineStep implements LenientEquals<PipelineStep>, Serializable {
019
020  private static final long serialVersionUID = 460047082156621661L;
021
022  @JsonInclude(JsonInclude.Include.NON_DEFAULT)
023  private long key;
024
025  private StepType type;
026  private StepRunner runner;
027
028  @JsonSerialize(using = LocalDateTimeSerDe.LocalDateTimeSerializer.class)
029  @JsonDeserialize(using = LocalDateTimeSerDe.LocalDateTimeDeserializer.class)
030  private LocalDateTime started;
031
032  @JsonSerialize(using = LocalDateTimeSerDe.LocalDateTimeSerializer.class)
033  @JsonDeserialize(using = LocalDateTimeSerDe.LocalDateTimeDeserializer.class)
034  private LocalDateTime finished;
035
036  private Status state;
037  private String message;
038  private String rerunReason;
039  private String createdBy;
040
041  @JsonSerialize(using = LocalDateTimeSerDe.LocalDateTimeSerializer.class)
042  @JsonDeserialize(using = LocalDateTimeSerDe.LocalDateTimeDeserializer.class)
043  private LocalDateTime modified;
044
045  private String modifiedBy;
046  private Set<MetricInfo> metrics = new HashSet<>();
047
048  public long getKey() {
049    return key;
050  }
051
052  public StepType getType() {
053    return type;
054  }
055
056  public PipelineStep setType(StepType type) {
057    this.type = type;
058    return this;
059  }
060
061  public StepRunner getRunner() {
062    return runner;
063  }
064
065  public PipelineStep setRunner(StepRunner runner) {
066    this.runner = runner;
067    return this;
068  }
069
070  public LocalDateTime getStarted() {
071    return started;
072  }
073
074  public PipelineStep setStarted(LocalDateTime started) {
075    this.started = started;
076    return this;
077  }
078
079  public LocalDateTime getFinished() {
080    return finished;
081  }
082
083  public PipelineStep setFinished(LocalDateTime finished) {
084    this.finished = finished;
085    return this;
086  }
087
088  public Status getState() {
089    return state;
090  }
091
092  public PipelineStep setState(Status state) {
093    this.state = state;
094    return this;
095  }
096
097  public String getMessage() {
098    return message;
099  }
100
101  public PipelineStep setMessage(String message) {
102    this.message = message;
103    return this;
104  }
105
106  public String getRerunReason() {
107    return rerunReason;
108  }
109
110  public PipelineStep setRerunReason(String rerunReason) {
111    this.rerunReason = rerunReason;
112    return this;
113  }
114
115  public String getCreatedBy() {
116    return createdBy;
117  }
118
119  public PipelineStep setCreatedBy(String createdBy) {
120    this.createdBy = createdBy;
121    return this;
122  }
123
124  public LocalDateTime getModified() {
125    return modified;
126  }
127
128  public PipelineStep setModified(LocalDateTime modified) {
129    this.modified = modified;
130    return this;
131  }
132
133  public String getModifiedBy() {
134    return modifiedBy;
135  }
136
137  public PipelineStep setModifiedBy(String modifiedBy) {
138    this.modifiedBy = modifiedBy;
139    return this;
140  }
141
142  public Set<MetricInfo> getMetrics() {
143    return metrics;
144  }
145
146  public PipelineStep setMetrics(Set<MetricInfo> metrics) {
147    this.metrics = metrics;
148    return this;
149  }
150
151  public void addMetricInfo(MetricInfo metricInfo) {
152    metrics.add(metricInfo);
153  }
154
155  /** Enum to represent the status of a step. */
156  public enum Status {
157    RUNNING,
158    FAILED,
159    COMPLETED
160  }
161
162  /** Inner class to store metrics. */
163  public static class MetricInfo implements Serializable {
164
165    private static final long serialVersionUID = 1872427841009786709L;
166
167    private String name;
168    private String value;
169
170    public MetricInfo(String name, String value) {
171      this.name = name;
172      this.value = value;
173    }
174
175    public String getName() {
176      return name;
177    }
178
179    public String getValue() {
180      return value;
181    }
182
183    @Override
184    public boolean equals(Object o) {
185      if (this == o) {
186        return true;
187      }
188      if (o == null || getClass() != o.getClass()) {
189        return false;
190      }
191      MetricInfo that = (MetricInfo) o;
192      return Objects.equals(name, that.name) && Objects.equals(value, that.value);
193    }
194
195    @Override
196    public int hashCode() {
197      return Objects.hash(name, value);
198    }
199
200    @Override
201    public String toString() {
202      return new StringJoiner(", ", MetricInfo.class.getSimpleName() + "[", "]")
203          .add("name='" + name + "'")
204          .add("value='" + value + "'")
205          .toString();
206    }
207  }
208
209  @Override
210  public boolean equals(Object o) {
211    if (this == o) return true;
212    if (o == null || getClass() != o.getClass()) return false;
213    PipelineStep that = (PipelineStep) o;
214    return key == that.key
215        && Objects.equals(type, that.type)
216        && Objects.equals(runner, that.runner)
217        && Objects.equals(started, that.started)
218        && Objects.equals(finished, that.finished)
219        && state == that.state
220        && Objects.equals(message, that.message)
221        && Objects.equals(metrics, that.metrics)
222        && Objects.equals(rerunReason, that.rerunReason)
223        && Objects.equals(createdBy, that.createdBy)
224        && Objects.equals(modified, that.modified)
225        && Objects.equals(modifiedBy, that.modifiedBy);
226  }
227
228  @Override
229  public int hashCode() {
230    return Objects.hash(
231        key,
232        type,
233        runner,
234        started,
235        finished,
236        state,
237        message,
238        metrics,
239        rerunReason,
240        createdBy,
241        modified,
242        modifiedBy);
243  }
244
245  @Override
246  public String toString() {
247    return new StringJoiner(", ", PipelineStep.class.getSimpleName() + "[", "]")
248        .add("key=" + key)
249        .add("type=" + type)
250        .add("runner=" + runner)
251        .add("started=" + started)
252        .add("finished=" + finished)
253        .add("state=" + state)
254        .add("message='" + message + "'")
255        .add("rerunReason='" + rerunReason + "'")
256        .add("createdBy='" + createdBy + "'")
257        .add("modified=" + modified)
258        .add("modifiedBy='" + modifiedBy + "'")
259        .add("metrics=" + metrics)
260        .toString();
261  }
262
263  @Override
264  public boolean lenientEquals(PipelineStep other) {
265    if (this == other) return true;
266    return Objects.equals(type, other.type)
267        && Objects.equals(runner, other.runner)
268        && Objects.equals(finished, other.finished)
269        && state == other.state
270        && Objects.equals(message, other.message)
271        && Objects.equals(metrics, other.metrics)
272        && Objects.equals(rerunReason, other.rerunReason);
273  }
274}
