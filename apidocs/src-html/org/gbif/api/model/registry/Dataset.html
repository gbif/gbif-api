755  @Override
756  public boolean lenientEquals(Dataset other) {
757    if (this == other) {
758      return true;
759    }
760    return Objects.equal(this.parentDatasetKey, other.parentDatasetKey)
761      && Objects.equal(this.duplicateOfDatasetKey, other.duplicateOfDatasetKey)
762      && Objects.equal(this.installationKey, other.installationKey)
763      && Objects.equal(this.publishingOrganizationKey, other.publishingOrganizationKey)
764      && Objects.equal(this.doi, other.doi)
765      && Objects.equal(this.external, other.external)
766      && Objects.equal(this.type, other.type)
767      && Objects.equal(this.subtype, other.subtype)
768      && Objects.equal(this.title, other.title)
769      && Objects.equal(this.alias, other.alias)
770      && Objects.equal(this.abbreviation, other.abbreviation)
771      && Objects.equal(this.description, other.description)
772      && Objects.equal(this.language, other.language)
773      && Objects.equal(this.homepage, other.homepage)
