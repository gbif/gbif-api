052/**

768        && Objects.equals(doi, dataset.doi)

770        && type == dataset.type
771        && subtype == dataset.subtype
772        && Objects.equals(title, dataset.title)
773        && Objects.equals(alias, dataset.alias)
774        && Objects.equals(abbreviation, dataset.abbreviation)
775        && Objects.equals(description, dataset.description)
776        && language == dataset.language
777        && Objects.equals(homepage, dataset.homepage)
778        && Objects.equals(logoUrl, dataset.logoUrl)
779        && Objects.equals(citation, dataset.citation)
780        && Objects.equals(rights, dataset.rights)
781        && Objects.equals(createdBy, dataset.createdBy)
782        && Objects.equals(modifiedBy, dataset.modifiedBy)
783        && Objects.equals(created, dataset.created)
784        && Objects.equals(modified, dataset.modified)
785        && Objects.equals(deleted, dataset.deleted)
786        && Objects.equals(contacts, dataset.contacts)

789        && Objects.equals(tags, dataset.tags)
790        && Objects.equals(identifiers, dataset.identifiers)
791        && Objects.equals(comments, dataset.comments)

793        && Objects.equals(curatorialUnits, dataset.curatorialUnits)

795        && Objects.equals(geographicCoverageDescription, dataset.geographicCoverageDescription)
796        && Objects.equals(geographicCoverages, dataset.geographicCoverages)
797        && Objects.equals(temporalCoverages, dataset.temporalCoverages)
798        && Objects.equals(keywordCollections, dataset.keywordCollections)
