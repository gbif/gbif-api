016package org.gbif.api.model.registry;
017
018import org.gbif.api.vocabulary.TagName;
019
020import java.io.Serializable;
021import java.util.Date;
022import java.util.Objects;
023import java.util.StringJoiner;
024
025import javax.validation.constraints.NotNull;
026import javax.validation.constraints.Null;
027import javax.validation.constraints.Size;
028
029/**
030 * A tag that has a namespace, name and a value. {@code created} and {@code createdBy} are automatically set upon
031 * persisting.
032 */
033// TODO: Document the rules regarding duplicate names and values
034public class MachineTag implements LenientEquals<MachineTag>, Serializable {
035
036  private static final long serialVersionUID = 3475968899219274852L;
037
038  private Integer key;
039  private String namespace;
040  private String name;
041  private String value;
042  private String createdBy;
043  private Date created;
044
045  public static MachineTag newInstance(String namespace, String name, String value) {
046    return new MachineTag(namespace, name, value);
047  }
048
049  public static MachineTag newInstance(TagName tagName, String value) {
050    return new MachineTag(tagName, value);
051  }
052
053  public MachineTag() {
054    // Needed
055  }
056
057  /**
058   * This is the default constructor to create new Machine Tags which takes all user settable properties.
059   */
060  public MachineTag(String namespace, String name, String value) {
061    this.namespace = namespace;
062    this.name = name;
063    this.value = value;
064  }
065
066  /**
067   * This is the other constructor to create new Machine Tags which takes all user settable properties.
068   */
069  public MachineTag(TagName tagName, String value) {
070    this(tagName.getNamespace().getNamespace(), tagName.getName(), value);
071  }
072
073  @Null(groups = {PrePersist.class})
074  @NotNull(groups = {PostPersist.class})
075  public Integer getKey() {
076    return key;
077  }
078
079  public void setKey(Integer key) {
080    this.key = key;
081  }
082
083  @NotNull
084  @Size(min = 1, max = 255)
085  public String getNamespace() {
086    return namespace;
087  }
088
089  public void setNamespace(String namespace) {
090    this.namespace = namespace;
091  }
092
093  @NotNull
094  @Size(min = 1, max = 255)
095  public String getName() {
096    return name;
097  }
098
099  public void setName(String name) {
100    this.name = name;
101  }
102
103  @NotNull
104  @Size(min = 1, max = 700)
105  public String getValue() {
106    return value;
107  }
108
109  public void setValue(String value) {
110    this.value = value;
111  }
112
113  @Size(min = 3)
114  public String getCreatedBy() {
115    return createdBy;
116  }
117
118  public void setCreatedBy(String createdBy) {
119    this.createdBy = createdBy;
120  }
121
122  @Null(groups = {PrePersist.class})
123  @NotNull(groups = {PostPersist.class})
124  public Date getCreated() {
125    return created;
126  }
127
128  public void setCreated(Date created) {
129    this.created = created;
130  }
131
132  @Override
133  public boolean equals(Object o) {
134    if (this == o) {
135      return true;
136    }
137    if (o == null || getClass() != o.getClass()) {
138      return false;
139    }
140    MachineTag that = (MachineTag) o;
141    return Objects.equals(key, that.key) &&
142      Objects.equals(namespace, that.namespace) &&
143      Objects.equals(name, that.name) &&
144      Objects.equals(value, that.value) &&
145      Objects.equals(createdBy, that.createdBy) &&
146      Objects.equals(created, that.created);
147  }
148
149  @Override
150  public int hashCode() {
151    return Objects.hash(key, namespace, name, value, createdBy, created);
152  }
153
154  @Override
155  public String toString() {
156    return new StringJoiner(", ", MachineTag.class.getSimpleName() + "[", "]")
157      .add("key=" + key)
158      .add("namespace='" + namespace + "'")
159      .add("name='" + name + "'")
160      .add("value='" + value + "'")
161      .add("createdBy='" + createdBy + "'")
162      .add("created=" + created)
163      .toString();
164  }
165
166  /**
167   * This implementation of the {@link #equals(Object)} method does only check <em>business equality</em> and disregards
168   * automatically set and maintained fields like {@code createdBy, key} and possibly others in the future.
169   */
170  @Override
171  public boolean lenientEquals(MachineTag other) {
172    if (this == other) {
173      return true;
174    }
175
176    return Objects.equals(this.namespace, other.namespace)
177      && Objects.equals(this.name, other.name)
178      && Objects.equals(this.value, other.value);
179  }
180}
