001package org.gbif.api.model.registry.search;
002
003import org.gbif.api.model.common.paging.Pageable;
004import org.gbif.api.model.common.search.FacetedSearchRequest;
005import org.gbif.api.model.registry.Tag;
006import org.gbif.api.vocabulary.Continent;
007import org.gbif.api.vocabulary.Country;
008import org.gbif.api.vocabulary.DatasetSubtype;
009import org.gbif.api.vocabulary.DatasetType;
010
011import java.util.UUID;
012
013/**
014 * A dataset specific search request with convenience methods to add facet filters.
015 */
016public class DatasetSearchRequest extends FacetedSearchRequest<DatasetSearchParameter> {
017
018  public DatasetSearchRequest() {
019  }
020
021  public DatasetSearchRequest(Pageable page) {
022    super(page);
023  }
024
025  public DatasetSearchRequest(long offset, int limit) {
026    super(offset, limit);
027  }
028
029  /**
030   * Filters dataset by the owning organizations country.
031   */
032  public void addPublishingCountryFilter(Country country) {
033    addParameter(DatasetSearchParameter.PUBLISHING_COUNTRY, country.getIso2LetterCode());
034  }
035
036  /**
037   * Filters dataset by a country of the geospatial coverage.
038   * 
039   * @param country appearing in geospatial coverage
040   */
041  public void addCountryFilter(Country country) {
042    addParameter(DatasetSearchParameter.COUNTRY, country.getIso2LetterCode());
043  }
044
045  /**
046   * Filters dataset by a continent of the geospatial coverage.
047   * 
048   * @param continent appearing in geospatial coverage
049   */
050  public void addContinentFilter(Continent continent) {
051    addParameter(DatasetSearchParameter.CONTINENT, continent);
052  }
053
054  /**
055   * Filters datasets by their temporal coverage broken down to decades.
056   * 
057   * @param decade the decade given as a 4 digit integer
058   */
059  public void addDecadeFilter(int decade) {
060    addParameter(DatasetSearchParameter.DECADE, decade);
061  }
062
063  public void addHostingOrgFilter(UUID orgKey) {
064    addParameter(DatasetSearchParameter.HOSTING_ORG, orgKey.toString());
065  }
066
067  /**
068   * Filters datasets by a keyword as generated through {@link org.gbif.api.model.registry.Dataset#getKeywords()} by
069   * merging tags, the keywordCollections and temporalCoverages property.
070   * 
071   * @param keyword a plain keyword e.g. created by Tag.toString()
072   */
073  // TODO: Fix Javadoc, what does this do? Keywords don't exist
074  public void addKeywordFilter(String keyword) {
075    addParameter(DatasetSearchParameter.KEYWORD, keyword);
076  }
077
078  /**
079   * Filters dataset by a tag.
080   * 
081   * @param keyword given as a tag
082   */
083  public void addKeywordFilter(Tag keyword) {
084    addParameter(DatasetSearchParameter.KEYWORD, keyword.toString());
085  }
086
087  public void addPublishingOrgFilter(UUID orgKey) {
088    addParameter(DatasetSearchParameter.PUBLISHING_ORG, orgKey.toString());
089  }
090
091  public void addSubTypeFilter(DatasetSubtype subtype) {
092    addParameter(DatasetSearchParameter.SUBTYPE, subtype);
093  }
094
095  public void addTypeFilter(DatasetType type) {
096    addParameter(DatasetSearchParameter.TYPE, type);
097  }
098
099}
