172  @Override
173  public int hashCode() {
174    return Objects
175      .hashCode(key, title, description, type, subtype, fullText, hostingOrganizationKey, hostingOrganizationTitle,
176        publisherTitle, countryCoverage, continent, publishingCountry, publishingOrganizationKey, publishingOrganizationTitle,
177        decades, keywords);
178  }
179
180  @Override
181  public boolean equals(Object object) {
182    if (object instanceof DatasetSearchResult) {
183      DatasetSearchResult that = (DatasetSearchResult) object;
184      return Objects.equal(this.key, that.key)
185        && Objects.equal(this.title, that.title)
186        && Objects.equal(this.description, that.description)
187        && Objects.equal(this.type, that.type)
188        && Objects.equal(this.subtype, that.subtype)
189        && Objects.equal(this.fullText, that.fullText)
190        && Objects.equal(this.hostingOrganizationKey, that.hostingOrganizationKey)
191        && Objects.equal(this.hostingOrganizationTitle, that.hostingOrganizationTitle)
192        && Objects.equal(this.publisherTitle, that.publisherTitle)
193        && Objects.equal(this.countryCoverage, that.countryCoverage)
194        && Objects.equal(this.continent, that.continent)
195        && Objects.equal(this.publishingCountry, that.publishingCountry)
196        && Objects.equal(this.publishingOrganizationKey, that.publishingOrganizationKey)
197        && Objects.equal(this.publishingOrganizationTitle, that.publishingOrganizationTitle)
198        && Objects.equal(this.decades, that.decades)
199        && Objects.equal(this.keywords, that.keywords);
200    }
201    return false;
202  }

208      .add("title", title)
209      .add("description", description)
210      .add("type", type)
211      .add("subtype", subtype)
212      .add("fullText", fullText)
213      .add("hostingOrganizationKey", hostingOrganizationKey)
214      .add("hostingOrganizationTitle", hostingOrganizationTitle)
215      .add("publisherTitle", publisherTitle)
216      .add("countryCoverage", countryCoverage)
217      .add("continent", continent)
218      .add("publishingCountry", publishingCountry)
219      .add("publishingOrganizationKey", publishingOrganizationKey)
220      .add("publishingOrganizationTitle", publishingOrganizationTitle)
221      .add("decades", decades)
222      .add("keywords", keywords)
223      .toString();
224  }
