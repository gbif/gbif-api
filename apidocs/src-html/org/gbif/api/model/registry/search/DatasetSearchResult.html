215  @Override
216  public boolean equals(Object o) {
217    if (this == o) {
218      return true;
219    }
220    if (o == null || getClass() != o.getClass()) {
221      return false;
222    }
223    DatasetSearchResult that = (DatasetSearchResult) o;
224    return Objects.equals(key, that.key) &&
225      Objects.equals(title, that.title) &&
226      Objects.equals(description, that.description) &&
227      type == that.type &&
228      subtype == that.subtype &&
229      Objects.equals(fullText, that.fullText) &&
230      Objects.equals(hostingOrganizationKey, that.hostingOrganizationKey) &&
231      Objects.equals(hostingOrganizationTitle, that.hostingOrganizationTitle) &&
232      Objects.equals(publisherTitle, that.publisherTitle) &&
233      Objects.equals(countryCoverage, that.countryCoverage) &&
234      Objects.equals(continent, that.continent) &&
235      publishingCountry == that.publishingCountry &&
236      Objects.equals(publishingOrganizationKey, that.publishingOrganizationKey) &&
237      Objects.equals(publishingOrganizationTitle, that.publishingOrganizationTitle) &&
238      license == that.license &&
239      Objects.equals(decades, that.decades) &&
240      Objects.equals(keywords, that.keywords) &&
241      Objects.equals(projectIdentifier, that.projectIdentifier) &&
242      Objects.equals(recordCount, that.recordCount);
243  }
244

251  }

272      .add("keywords=" + keywords)
273      .add("projectIdentifier='" + projectIdentifier + "'")

275      .toString();
276  }
