001package org.gbif.api.model.registry.search.collections;
002
003import java.util.Objects;
004import java.util.StringJoiner;
005import java.util.UUID;
006
007/**
008 * A utility container for holding the key, code and name of an entity.
009 */
010public class KeyCodeNameResult {
011
012  private UUID key;
013  private String code;
014  private String name;
015
016  public KeyCodeNameResult() {
017
018  }
019
020  public KeyCodeNameResult(UUID key, String code, String name) {
021    this.key = key;
022    this.code = code;
023    this.name = name;
024  }
025
026  public UUID getKey() {
027    return key;
028  }
029
030  public void setKey(UUID key) {
031    this.key = key;
032  }
033
034  public String getCode() {
035    return code;
036  }
037
038  public void setCode(String code) {
039    this.code = code;
040  }
041
042  public String getName() {
043    return name;
044  }
045
046  public void setName(String name) {
047    this.name = name;
048  }
049
050  @Override
051  public boolean equals(Object o) {
052    if (this == o) return true;
053    if (o == null || getClass() != o.getClass()) return false;
054    KeyCodeNameResult that = (KeyCodeNameResult) o;
055    return Objects.equals(key, that.key) && Objects.equals(code, that.code) && Objects.equals(name, that.name);
056  }
057
058  @Override
059  public int hashCode() {
060    return Objects.hash(key, code, name);
061  }
062
063  @Override
064  public String toString() {
065    return new StringJoiner(", ", KeyCodeNameResult.class.getSimpleName() + "[", "]").add("key=" + key)
066      .add("code='" + code + "'")
067      .add("name='" + name + "'")
068      .toString();
069  }
070}
