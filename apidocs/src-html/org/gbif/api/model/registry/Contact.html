103  /**
104   * Adds a new user id that is assembled from a directory name and a local id within it.
105   * Format used by EML, though see https://github.com/gbif/gbif-api/issues/30.
106   * The directory should be a valid URI, if it's not, it will be ignored by this method.
107   *
108   * @param directory identifier for the directory, preferably a URL domain like http://orcid.org
109   * @param id the identifier in that directory
110   */
111  public void addUserId(String directory, String id) {
112    if (StringUtils.isNotEmpty(id)) {
113      if (StringUtils.isEmpty(directory)) {
114        userId.add(id);
115      } else {
116        try {
117          URI dir = URI.create(directory);
118          if (dir.isAbsolute()) {
119            String dir2 = dir.toString();
120            if (!dir2.endsWith("/") && !dir2.endsWith("=")) {
121              dir2 = dir2 + "/";
122            }
123
124            // Check if the id is already prefixed with the directory URI, either HTTP or HTTPS.
125            //noinspection HttpUrlsUsage
126            if (id.startsWith(dir2)
127                || id.startsWith(dir2.replace("http://", "https://"))
128                || id.startsWith(dir2.replace("https://", "http://"))) {
129              userId.add(id);
130              // Check if the id is prefixed with the hostname.
131            } else if (id.startsWith(dir.getHost())) {
132              userId.add(dir.getScheme() + "://" + id);
133            } else {
134              userId.add(dir2 + id);
135            }
136          } else {
137            if (id.startsWith(dir.toString())) {
138              userId.add(id);
139            } else {
140              userId.add(dir + ":" + id);
141            }
142          }
143        } catch (IllegalArgumentException iaEx) {
144          // in case the directory is not a valid URL keep only the user id
145          userId.add(id);
146        }
147      }
148    }
149  }

171  /**
172   * Compute and returns the complete name in the form: FirstName LastName.
173   * Since all parts are optional, this method can return an empty string (but never null)
174   *
175   * @return the non-empty parts of FirstName LastName or empty string if none
176   */
177  public String computeCompleteName() {
178    return Stream.of(firstName, lastName)
179        .map(StringUtils::trimToNull)
180        .filter(Objects::nonNull)
181        .collect(Collectors.joining(" "));
182  }

353  @Override
354  public boolean equals(Object o) {
355    if (this == o) {
356      return true;
357    }
358    if (o == null || getClass() != o.getClass()) {
359      return false;
360    }
361    Contact contact = (Contact) o;
362    return primary == contact.primary
363        && Objects.equals(key, contact.key)
364        && type == contact.type
365        && Objects.equals(userId, contact.userId)
366        && Objects.equals(firstName, contact.firstName)
367        && Objects.equals(lastName, contact.lastName)
368        && Objects.equals(position, contact.position)
369        && Objects.equals(description, contact.description)
370        && Objects.equals(email, contact.email)
371        && Objects.equals(phone, contact.phone)
372        && Objects.equals(homepage, contact.homepage)
373        && Objects.equals(organization, contact.organization)
374        && Objects.equals(address, contact.address)
375        && Objects.equals(city, contact.city)
376        && Objects.equals(province, contact.province)
377        && country == contact.country
378        && Objects.equals(postalCode, contact.postalCode)
379        && Objects.equals(createdBy, contact.createdBy)
380        && Objects.equals(modifiedBy, contact.modifiedBy)
381        && Objects.equals(created, contact.created)
382        && Objects.equals(modified, contact.modified);
383  }

438  /**
439   * This implementation of the {@link #equals(Object)} method does only check <em>business equality</em> and disregards
440   * automatically set and maintained fields like {@code createdBy, key} and others.
441   */
442  @Override
443  public boolean lenientEquals(Contact contact) {
444    if (this == contact) {
445      return true;
446    }
447
448    return Objects.equals(type, contact.type)
449        && Objects.equals(primary, contact.primary)
450        && Objects.equals(userId, contact.userId)
451        && Objects.equals(firstName, contact.firstName)
452        && Objects.equals(lastName, contact.lastName)
453        && Objects.equals(position, contact.position)
454        && Objects.equals(description, contact.description)
455        && Objects.equals(email, contact.email)
456        && Objects.equals(phone, contact.phone)
457        && Objects.equals(homepage, contact.homepage)
458        && Objects.equals(organization, contact.organization)
459        && Objects.equals(address, contact.address)
460        && Objects.equals(city, contact.city)
461        && Objects.equals(province, contact.province)
462        && Objects.equals(country, contact.country)
463        && Objects.equals(postalCode, contact.postalCode);
464  }
