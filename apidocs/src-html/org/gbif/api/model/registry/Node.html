394  @Override
395  public boolean equals(Object o) {
396    if (this == o) {
397      return true;
398    }
399    if (o == null || getClass() != o.getClass()) {
400      return false;
401    }
402    Node node = (Node) o;
403    return Objects.equals(key, node.key) &&
404      type == node.type &&
405      participationStatus == node.participationStatus &&
406      Objects.equals(participantSince, node.participantSince) &&
407      Objects.equals(dateSignedMOU, node.dateSignedMOU) &&
408      gbifRegion == node.gbifRegion &&
409      continent == node.continent &&
410      Objects.equals(title, node.title) &&
411      Objects.equals(participantTitle, node.participantTitle) &&
412      Objects.equals(abbreviation, node.abbreviation) &&
413      Objects.equals(description, node.description) &&
414      Objects.equals(email, node.email) &&
415      Objects.equals(phone, node.phone) &&
416      Objects.equals(homepage, node.homepage) &&
417      Objects.equals(logoUrl, node.logoUrl) &&
418      Objects.equals(organization, node.organization) &&
419      Objects.equals(address, node.address) &&
420      Objects.equals(city, node.city) &&
421      Objects.equals(province, node.province) &&
422      country == node.country &&
423      Objects.equals(postalCode, node.postalCode) &&
424      Objects.equals(createdBy, node.createdBy) &&
425      Objects.equals(modifiedBy, node.modifiedBy) &&
426      Objects.equals(created, node.created) &&
427      Objects.equals(modified, node.modified) &&
428      Objects.equals(deleted, node.deleted) &&
429      Objects.equals(contacts, node.contacts) &&
430      Objects.equals(endpoints, node.endpoints) &&
431      Objects.equals(machineTags, node.machineTags) &&
432      Objects.equals(tags, node.tags) &&
433      Objects.equals(identifiers, node.identifiers) &&
434      Objects.equals(comments, node.comments);
435  }

485  /**
486   * Compares the entities for business logic equality using a very lenient approach.
487   * Only fields persisted in the registry database are used in the comparison, excluding any nested properties and
488   * supplementary information which can be added from external databases such as the IMS. Server controlled values such
489   * as key and createdBy are omitted.
490   */
491  @Override
492  public boolean lenientEquals(Node other) {
493    if (this == other) {
494      return true;
495    }
496    return Objects.equals(this.type, other.type)
497      && Objects.equals(this.participationStatus, other.participationStatus)
498      && Objects.equals(this.gbifRegion, other.gbifRegion)
499      && Objects.equals(this.continent, other.continent)
500      && Objects.equals(this.title, other.title)
501      && Objects.equals(this.participantTitle, other.participantTitle)
502      && Objects.equals(this.country, other.country);
503  }
