431  @Override
432  public boolean equals(Object o) {
433    if (this == o) {
434      return true;
435    }
436    if (o == null || getClass() != o.getClass()) {
437      return false;
438    }
439    Organization that = (Organization) o;
440    return endorsementApproved == that.endorsementApproved
441        && endorsementStatus == that.endorsementStatus
442        && numPublishedDatasets == that.numPublishedDatasets
443        && Objects.equals(key, that.key)
444        && Objects.equals(endorsingNodeKey, that.endorsingNodeKey)
445        && Objects.equals(password, that.password)
446        && Objects.equals(title, that.title)
447        && Objects.equals(abbreviation, that.abbreviation)
448        && Objects.equals(description, that.description)
449        && language == that.language
450        && Objects.equals(email, that.email)
451        && Objects.equals(phone, that.phone)
452        && Objects.equals(homepage, that.homepage)
453        && Objects.equals(logoUrl, that.logoUrl)
454        && Objects.equals(address, that.address)
455        && Objects.equals(city, that.city)
456        && Objects.equals(province, that.province)
457        && country == that.country
458        && Objects.equals(postalCode, that.postalCode)
459        && Objects.equals(latitude, that.latitude)
460        && Objects.equals(longitude, that.longitude)
461        && Objects.equals(createdBy, that.createdBy)
462        && Objects.equals(modifiedBy, that.modifiedBy)
463        && Objects.equals(created, that.created)
464        && Objects.equals(modified, that.modified)
465        && Objects.equals(deleted, that.deleted)
466        && Objects.equals(endorsed, that.endorsed)
467        && Objects.equals(contacts, that.contacts)
468        && Objects.equals(endpoints, that.endpoints)
469        && Objects.equals(machineTags, that.machineTags)
470        && Objects.equals(tags, that.tags)
471        && Objects.equals(identifiers, that.identifiers)
472        && Objects.equals(comments, that.comments);
473  }

552  /**
553   * A lenient equality check for business logic uniqueness. Does not include server side controlled fields such as key
554   * and createdBy, nested objects and does not include password for security reasons.
555   */
556  @Override
557  public boolean lenientEquals(Organization other) {
558    if (this == other) {
559      return true;
560    }
561    return Objects.equals(this.endorsingNodeKey, other.endorsingNodeKey)
562        && Objects.equals(this.endorsementApproved, other.endorsementApproved)
563        && Objects.equals(this.endorsementStatus, other.endorsementStatus)
564        && Objects.equals(this.title, other.title)
565        && Objects.equals(this.abbreviation, other.abbreviation)
566        && Objects.equals(this.description, other.description)
567        && Objects.equals(this.language, other.language)
568        && Objects.equals(this.email, other.email)
569        && Objects.equals(this.phone, other.phone)
570        && Objects.equals(this.homepage, other.homepage)
571        && Objects.equals(this.logoUrl, other.logoUrl)
572        && Objects.equals(this.address, other.address)
573        && Objects.equals(this.city, other.city)
574        && Objects.equals(this.province, other.province)
