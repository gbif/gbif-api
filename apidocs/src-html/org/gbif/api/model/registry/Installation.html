033/**
034 * A technical installation which can serve dataset(s).
035 * Note: An Installation may be marked as disabled, meaning that some process has identified it is out of action.
036 * For the GBIF crawling infrastructure, this means it will not be "metasynced" nor will any dataset associated be
037 * eligble for crawling.
038 */
039// TODO: Only allow adding of Endpoints of the correct type, I would argue for removing all the set(List) methods
040public class Installation implements NetworkEntity, Contactable, Endpointable, MachineTaggable, Taggable, Commentable,
041  Identifiable, LenientEquals<Installation> {
042
043  private UUID key;
044  private UUID organizationKey;
045  private String password;
046  private InstallationType type;
047  private String title;
048  private String description;
049  private String createdBy;
050  private String modifiedBy;
051  private Date created;
052  private Date modified;
053  private Date deleted;
054  private boolean disabled;
055  private List<Contact> contacts = new ArrayList<>();
056  private List<Endpoint> endpoints = new ArrayList<>();
057  private List<MachineTag> machineTags = new ArrayList<>();
058  private List<Tag> tags = new ArrayList<>();
059  private List<Identifier> identifiers = new ArrayList<>();
060  private List<Comment> comments = new ArrayList<>();
061

063  @NotNull(groups = {PostPersist.class})
064  @Override
065  public UUID getKey() {
066    return key;
067  }
068
069  @Override
070  public void setKey(UUID key) {
071    this.key = key;
072  }
073
074  @Override
075  public String getTitle() {
076    return title;
077  }
078
079  public boolean isDisabled() {
080    return disabled;
081  }
082
083  public void setDisabled(boolean disabled) {
084    this.disabled = disabled;
085  }
086
087  @Override
088  public void setTitle(String title) {
089    this.title = title;
090  }
091
092  @Override
093  public String getDescription() {
094    return description;
095  }
096
097  @Override
098  public void setDescription(String description) {
099    this.description = description;
100  }
101
102  @Null(groups = {PrePersist.class})
103  @NotNull(groups = {PostPersist.class})
104  @Override
105  public Date getCreated() {
106    return created;
107  }
108
109  @Override
110  public void setCreated(Date created) {
111    this.created = created;
112  }
113
114  @Null(groups = {PrePersist.class})
115  @NotNull(groups = {PostPersist.class})
116  @Override
117  public Date getModified() {
118    return modified;
119  }
120
121  @Override
122  public void setModified(Date modified) {
123    this.modified = modified;
124  }
125
126  @Override
127  public Date getDeleted() {
128    return deleted;
129  }
130
131  @Override
132  public void setDeleted(Date deleted) {
133    this.deleted = deleted;
134  }
135
136  @NotNull
137  public UUID getOrganizationKey() {
138    return organizationKey;
139  }
140
141  public void setOrganizationKey(UUID organizationKey) {
142    this.organizationKey = organizationKey;
143  }
144
145  /**
146   * Get the installation password. This method is to be ignored on serialization, so that the password is not
147   * revealed in the web service response.
148   *
149   * @return organization password
150   */
151  @JsonIgnore
152  @Nullable
153  public String getPassword() {
154    return password;
155  }
156
157  public void setPassword(String password) {
158    this.password = password;
159  }
160
161  @NotNull
162  public InstallationType getType() {
163    return type;
164  }
165
166  public void setType(InstallationType type) {
167    this.type = type;
168  }
169
170  @Override
171  public String getCreatedBy() {
172    return createdBy;
173  }
174
175  @Override
176  public void setCreatedBy(String createdBy) {
177    this.createdBy = createdBy;
178  }
179
180  @Override
181  public String getModifiedBy() {
182    return modifiedBy;
183  }
184
185  @Override
186  public void setModifiedBy(String modifiedBy) {
187    this.modifiedBy = modifiedBy;
188  }
189

191  public List<Contact> getContacts() {
192    return contacts;
193  }
194

196  public void setContacts(List<Contact> contacts) {
197    this.contacts = contacts;

201  public List<Endpoint> getEndpoints() {

250  @Override

252    return comments;
253  }
254
255  @Override
256  public void setComments(List<Comment> comments) {
257    this.comments = comments;
258  }
259
260  @Override
261  public boolean equals(Object o) {
262    if (this == o) {
263      return true;
264    }
265    if (o == null || getClass() != o.getClass()) {
266      return false;
267    }
268    Installation that = (Installation) o;
269    return disabled == that.disabled &&
270      Objects.equals(key, that.key) &&
271      Objects.equals(organizationKey, that.organizationKey) &&
272      Objects.equals(password, that.password) &&
273      type == that.type &&
274      Objects.equals(title, that.title) &&
275      Objects.equals(description, that.description) &&
276      Objects.equals(createdBy, that.createdBy) &&
277      Objects.equals(modifiedBy, that.modifiedBy) &&
278      Objects.equals(created, that.created) &&
279      Objects.equals(modified, that.modified) &&
280      Objects.equals(deleted, that.deleted) &&
281      Objects.equals(contacts, that.contacts) &&
282      Objects.equals(endpoints, that.endpoints) &&
283      Objects.equals(machineTags, that.machineTags) &&
284      Objects.equals(tags, that.tags) &&
285      Objects.equals(identifiers, that.identifiers) &&
286      Objects.equals(comments, that.comments);
287  }
288
289  @Override
290  public int hashCode() {
291    return Objects
292      .hash(key, organizationKey, password, type, title, description, createdBy, modifiedBy,
293        created, modified, deleted, disabled, contacts, endpoints, machineTags, tags, identifiers,
294        comments);
295  }
296
297  @Override
298  public String toString() {
299    return new StringJoiner(", ", Installation.class.getSimpleName() + "[", "]")
300      .add("key=" + key)
301      .add("organizationKey=" + organizationKey)
302      .add("password='" + password + "'")
303      .add("type=" + type)
304      .add("title='" + title + "'")
305      .add("description='" + description + "'")
306      .add("createdBy='" + createdBy + "'")
307      .add("modifiedBy='" + modifiedBy + "'")
308      .add("created=" + created)
309      .add("modified=" + modified)
310      .add("deleted=" + deleted)
311      .add("disabled=" + disabled)
312      .add("contacts=" + contacts)
313      .add("endpoints=" + endpoints)
314      .add("machineTags=" + machineTags)
315      .add("tags=" + tags)
316      .add("identifiers=" + identifiers)
317      .add("comments=" + comments)
318      .toString();
319  }
320
321  /**
322   * Does not include the nested properties, or server controlled values (key, createdBy etc) or the password,
323   * for security reasons.
324   */
325  @Override
326  public boolean lenientEquals(Installation other) {
327    if (this == other) {
328      return true;
329    }
330    return Objects.equals(this.organizationKey, other.organizationKey)
331      && Objects.equals(this.type, other.type)
332      && Objects.equals(this.title, other.title)
333      && Objects.equals(this.description, other.description);
334  }
