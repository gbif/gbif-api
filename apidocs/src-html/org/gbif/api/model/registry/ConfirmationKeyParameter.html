001package org.gbif.api.model.registry;
002
003import java.util.UUID;
004import javax.validation.constraints.NotNull;
005
006/**
007 * Confirmation key is a UUID used to confirm an entity.
008 */
009public class ConfirmationKeyParameter {
010
011  private UUID confirmationKey;
012
013  public ConfirmationKeyParameter(){}
014
015  public ConfirmationKeyParameter(UUID confirmationKey) {
016    this.confirmationKey = confirmationKey;
017  }
018
019  @NotNull
020  public UUID getConfirmationKey() {
021    return confirmationKey;
022  }
023
024  public void setConfirmationKey(UUID confirmationKey) {
025    this.confirmationKey = confirmationKey;
026  }
027}
