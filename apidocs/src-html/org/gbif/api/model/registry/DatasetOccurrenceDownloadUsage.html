016package org.gbif.api.model.registry;
017
018import org.gbif.api.model.common.DOI;
019import org.gbif.api.model.occurrence.Download;
020
021import java.io.Serializable;
022import java.util.Objects;
023import java.util.StringJoiner;
024import java.util.UUID;
025
026import javax.validation.constraints.NotNull;
027
028/**
029 * Represents the information about the usage of one dataset in an occurrence download.
030 * Provides information about the number of records that the dataset provided to the download; additionally, provide the
031 * references to the download and dataset entities.
032 */
033public class DatasetOccurrenceDownloadUsage implements Serializable {
034
035  private String downloadKey;
036  private UUID datasetKey;
037  private String datasetTitle;
038  private DOI datasetDOI;
039  private String datasetCitation;
040  private long numberRecords;
041  private Download download;
042
043  /**
044   * Dataset key.
045   */
046  @NotNull
047  public UUID getDatasetKey() {
048    return datasetKey;
049  }
050
051  /**
052   * @return the download
053   */
054  public Download getDownload() {
055    return download;
056  }
057
058  /**
059   * Occurrence download key.
060   */
061  @NotNull
062  public String getDownloadKey() {
063    return downloadKey;
064  }
065
066  /**
067   * Number of records that the dataset has provided to the occurrence download.
068   */
069  @NotNull
070  public long getNumberRecords() {
071    return numberRecords;
072  }
073
074  /**
075   * Dataset title at the moment when the download was created.
076   */
077  public String getDatasetTitle() {
078    return datasetTitle;
079  }
080
081  /**
082   * Dataset DOI at the moment when the download was created.
083   */
084  public DOI getDatasetDOI() {
085    return datasetDOI;
086  }
087
088  /**
089   * Dataset citation at the moment when the download was created.
090   */
091  public String getDatasetCitation() {
092    return datasetCitation;
093  }
094
095  public void setDatasetKey(UUID datasetKey) {
096    this.datasetKey = datasetKey;
097  }
098
099  public void setDownload(Download download) {
100    this.download = download;
101  }
102
103  public void setDownloadKey(String downloadKey) {
104    this.downloadKey = downloadKey;
105  }
106
107  public void setNumberRecords(long numberOfRecords) {
108    this.numberRecords = numberOfRecords;
109  }
110
111  public void setDatasetTitle(String datasetTitle) {
112    this.datasetTitle = datasetTitle;
113  }
114
115  public void setDatasetDOI(DOI datasetDOI) {
116    this.datasetDOI = datasetDOI;
117  }
118
119  public void setDatasetCitation(String datasetCitation) {
120    this.datasetCitation = datasetCitation;
121  }
122
123  @Override
124  public boolean equals(Object o) {
125    if (this == o) {
126      return true;
127    }
128    if (o == null || getClass() != o.getClass()) {
129      return false;
130    }
131    DatasetOccurrenceDownloadUsage that = (DatasetOccurrenceDownloadUsage) o;
132    return numberRecords == that.numberRecords &&
133      Objects.equals(downloadKey, that.downloadKey) &&
134      Objects.equals(datasetKey, that.datasetKey) &&
135      Objects.equals(datasetTitle, that.datasetTitle) &&
136      Objects.equals(datasetDOI, that.datasetDOI) &&
137      Objects.equals(datasetCitation, that.datasetCitation) &&
138      Objects.equals(download, that.download);
139  }
140
141  @Override
142  public int hashCode() {
143    return Objects
144      .hash(downloadKey, datasetKey, datasetTitle, datasetDOI, datasetCitation, numberRecords,
145        download);
146  }
147
148  @Override
149  public String toString() {
150    return new StringJoiner(", ",
151      DatasetOccurrenceDownloadUsage.class.getSimpleName() + "[", "]")
152      .add("downloadKey='" + downloadKey + "'")
153      .add("datasetKey=" + datasetKey)
154      .add("datasetTitle='" + datasetTitle + "'")
155      .add("datasetDOI=" + datasetDOI)
156      .add("datasetCitation='" + datasetCitation + "'")
157      .add("numberRecords=" + numberRecords)
158      .add("download=" + download)
159      .toString();
160  }
161}
