016package org.gbif.api.model.registry.eml;
017
018import org.gbif.api.model.registry.Contact;
019
020import java.io.Serializable;
021import java.util.LinkedList;
022import java.util.List;
023
024import com.google.common.base.Objects;
025
026
027/**
028 * A dataset can be part of a project.
029 */
030public class Project implements Serializable {
031
032  private static final long serialVersionUID = -2625204169061362016L;
033
034  private String title;
035  private String description;
036
037  private List<Contact> contacts;
038  private String funding;
039  private String studyAreaDescription;
040  private String designDescription;
041
042  public Project() {
043  }
044
045  public Project(
046    String title, List<Contact> contacts, String funding, String studyAreaDescription, String designDescription
047  ) {
048    this.title = title;
049    this.contacts = contacts;
050    this.funding = funding;
051    this.studyAreaDescription = studyAreaDescription;
052    this.designDescription = designDescription;
053  }
054
055  public List<Contact> getContacts() {
056    return contacts;
057  }
058
059  public void setContacts(List<Contact> contacts) {
060    this.contacts = contacts;
061  }
062
063  public String getDesignDescription() {
064    return designDescription;
065  }
066
067  public void setDesignDescription(String designDescription) {
068    this.designDescription = designDescription;
069  }
070
071  public String getFunding() {
072    return funding;
073  }
074
075  public void setFunding(String funding) {
076    this.funding = funding;
077  }
078
079  public String getStudyAreaDescription() {
080    return studyAreaDescription;
081  }
082
083  public void setStudyAreaDescription(String studyAreaDescription) {
084    this.studyAreaDescription = studyAreaDescription;
085  }
086
087  public String getTitle() {
088    return title;
089  }
090
091  public void setTitle(String title) {
092    this.title = title;
093  }
094
095  public String getDescription() {
096    return description;
097  }
098
099  public void setDescription(String description) {
100    this.description = description;
101  }
102
103  /**
104   * Add contact to Contact List.
105   */
106  public void addContact(Contact contact) {
107    if (contacts == null) {
108      contacts = new LinkedList<Contact>();
109    }
110    contacts.add(contact);
111  }
112
113  @Override
114  public boolean equals(Object obj) {
115    if (this == obj) {
116      return true;
117    }
118    if (!(obj instanceof Project)) {
119      return false;
120    }
121
122    Project that = (Project) obj;
123    return Objects.equal(this.title, that.title)
124           && Objects.equal(this.description, that.description)
125           && Objects.equal(this.contacts, that.contacts)
126           && Objects.equal(this.funding, that.funding)
127           && Objects.equal(this.studyAreaDescription, that.studyAreaDescription)
128           && Objects.equal(this.designDescription, that.designDescription);
129  }
130
131  @Override
132  public int hashCode() {
133    return Objects.hashCode(title, contacts, funding, studyAreaDescription, designDescription);
134  }
135
136  @Override
137  public String toString() {
138    return Objects.toStringHelper(this)
139      .add("title", title)
140      .add("description", description)
141      .add("contacts", contacts)
142      .add("funding", funding)
143      .add("studyAreaDescription", studyAreaDescription)
144      .add("designDescription", designDescription)
145      .toString();
146  }
147
148}
