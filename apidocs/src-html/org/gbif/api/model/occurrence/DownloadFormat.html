003/**
004 * Download output requested format.
005 */
006public enum DownloadFormat {
007  // Darwin Core Archive file that is a zip of all indexed fields in both verbatim and interpreted, plus multimedia and metadata files.
008  DWCA,
009
010  // Zipped text file of the most common indexed terms, but note that it is delimited by tabs, not commas.
011  SIMPLE_CSV,
012
013  // AVRO (with Deflate compression codec) format export of the most common indexed terms.
014  SIMPLE_AVRO,
015  
016  //CSV format export of the distinct species and taxonomic field associated to it.
017  SPECIES_LIST,
018  
019  //Executes SQL command and provides download
020  SQL;
021}
