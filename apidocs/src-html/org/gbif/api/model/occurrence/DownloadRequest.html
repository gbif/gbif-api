001package org.gbif.api.model.occurrence;
002
003import org.gbif.api.jackson.DownloadRequestSerde;
004
005import java.util.Collection;
006import java.util.Collections;
007import java.util.Set;
008import javax.annotation.Nullable;
009import org.codehaus.jackson.annotate.JsonIgnore;
010import org.codehaus.jackson.annotate.JsonProperty;
011import org.codehaus.jackson.map.annotate.JsonDeserialize;
012import com.google.common.base.Joiner;
013import com.google.common.base.MoreObjects;
014import com.google.common.base.Objects;
015import com.google.common.base.Splitter;
016import com.google.common.collect.ImmutableSet;
017import com.google.common.collect.Sets;
018
019/**
020 * Represents a request to download occurrence records.
021 * This is the base class for specific type of downloads: predicate based downloads and SQL downloads..
022 */
023@JsonDeserialize(using = DownloadRequestSerde.class)
024public abstract class DownloadRequest {
025
026  private static final String DELIMITER = ",";
027  private static final Joiner COMMA_JOINER = Joiner.on(DELIMITER).skipNulls();
028  private static final Splitter COMMA_SPLITTER = Splitter.on(DELIMITER).omitEmptyStrings().trimResults();
029
030  @JsonProperty("creator")
031  private String creator;
032
033  @JsonProperty("notificationAddresses")
034  private Set<String> notificationAddresses;
035
036  @JsonProperty("sendNotification")
037  private boolean sendNotification;
038
039  @JsonProperty("format")
040  private DownloadFormat format;
041
042  /**
043   * Default constructor.
044   */
045  public DownloadRequest() {
046    // Empty constructor required to create instances from the data access layer.
047  }
048
049  public DownloadRequest(String creator, Collection<String> notificationAddresses, boolean sendNotification, DownloadFormat format) {
050    this.creator = creator;
051    this.notificationAddresses = notificationAddresses == null ? Collections.emptySet() : ImmutableSet.copyOf(notificationAddresses);
052    this.sendNotification = sendNotification;
053    this.format = format;
054  }
055
056  /**
057   * @return the user account that initiated the download
058   */
059  @Nullable
060  public String getCreator() {
061    return creator;
062  }
063
064  /**
065   * @return set of email addresses for notifications
066   */
067  @Nullable
068  public Set<String> getNotificationAddresses() {
069    return notificationAddresses;
070  }
071
072  /**
073   * Returns the notification addresses as single string. The emails are separated by ','.
074   */
075  @Nullable
076  @JsonIgnore
077  public String getNotificationAddressesAsString() {
078    if (notificationAddresses != null) {
079      return COMMA_JOINER.join(notificationAddresses);
080    }
081    return null;
082  }
083
084  public void setCreator(String creator) {
085    this.creator = creator;
086  }
087
088
089  public void setNotificationAddresses(Set<String> notificationAddresses) {
090    this.notificationAddresses = notificationAddresses;
091  }
092
093  /**
094   * Sets the notificationAddresses using a single String value that is split by ','.
095   */
096  public void setNotificationAddressesAsString(String notificationAddressesAsString) {
097    if (notificationAddressesAsString != null) {
098      notificationAddresses = Sets.newHashSet(COMMA_SPLITTER.split(notificationAddressesAsString));
099    }
100  }
101
102  public boolean getSendNotification() {
103    return sendNotification;
104  }
105
106  /**
107   * This parameter determines if the requested download must be notified to the created once it's ready.
108   */
109  public void setSendNotification(boolean sendNotification) {
110    this.sendNotification = sendNotification;
111  }
112
113  public DownloadFormat getFormat() {
114    return format;
115  }
116
117  /**
118   * This parameter determines the output format of the requested download.
119   */
120  public void setFormat(DownloadFormat format) {
121    this.format = format;
122  }
123
124  @Override
125  public boolean equals(Object obj) {
126    if (this == obj) {
127      return true;
128    }
129    if (!(obj instanceof DownloadRequest)) {
130      return false;
131    }
132
133    DownloadRequest that = (DownloadRequest) obj;
134    return Objects.equal(this.creator, that.creator)
135      && Objects.equal(this.notificationAddresses, that.notificationAddresses)
136      && Objects.equal(this.sendNotification, that.sendNotification)
137      && Objects.equal(this.format, that.format);
138  }
139
140  @Override
141  public int hashCode() {
142    return Objects.hashCode(creator, notificationAddresses, sendNotification, format);
143  }
144
145  @Override
146  public String toString() {
147    return MoreObjects.toStringHelper(this)
148      .add("creator", creator)
149      .add("notificationAddresses", notificationAddresses)
150      .add("emailNotification", sendNotification)
151      .add("format", format).toString();
152  }
153
154}
