047/**
048 * An extended map holding all core terms of an occurrence record.
049 * Major extensions that we index are also supported, i.e. media, identifiers and measurements or facts.
050 */
051@SuppressWarnings("unused")
052public class VerbatimOccurrence {
053
054  private Long key;
055  private UUID datasetKey;
056  private UUID publishingOrgKey;
057  private List<UUID> networkKeys;
058  private UUID installationKey;
059  private Country publishingCountry;
060  private EndpointType protocol;
061  private Date lastCrawled;
062  private Date lastParsed;
063  private Integer crawlId;
064  //GBIF Participation: Programme and Project
065  private String projectId;
066  private String programmeAcronym;
067
068  // the verbatim fields for the occurrence
069  private Map<Term, String> verbatimFields = new HashMap<>();
070  // verbatim extension data
071  private Map<Extension, List<Map<Term, String>>> extensions = new HashMap<>();
072
073  /**
074   * Get the value of a specific field (Term).
075   */
076  @Nullable
077  public String getVerbatimField(Term term) {
078    Objects.requireNonNull(term, "term can't be null");
079    return verbatimFields.get(term);
080  }
081
082  /**
083   * @return true if a verbatim field exists and is not null or an empty string
084   */
085  public boolean hasVerbatimField(Term term) {
086    Objects.requireNonNull(term, "term can't be null");
087    return StringUtils.isNotEmpty(verbatimFields.get(term));
088  }
089
090  /**
091   * For setting a specific field without having to replace the entire verbatimFields Map.
092   *
093   * @param term       the field to set
094   * @param fieldValue the field's value
095   */
096  public void setVerbatimField(Term term, @Nullable String fieldValue) {
097    Objects.requireNonNull(term, "term can't be null");
098    verbatimFields.put(term, fieldValue);
099  }
100
101  /**
102   * The GBIF assigned, persistent key to the occurrence record.
103   * OccurrenceID itself is kept in the verbatim verbatimFields map.
104   */
105  @NotNull
106  public Long getKey() {
107    return key;
108  }
109
110  public void setKey(Long key) {
111    this.key = key;
112  }
113
114  @NotNull
115  public UUID getDatasetKey() {
116    return datasetKey;
117  }
118
119  public void setDatasetKey(UUID datasetKey) {
120    this.datasetKey = datasetKey;
121  }
122
123  @NotNull
124  public UUID getPublishingOrgKey() {
125    return publishingOrgKey;
126  }
127
128  public void setPublishingOrgKey(UUID publishingOrgKey) {
129    this.publishingOrgKey = publishingOrgKey;
130  }
131
132  /**
133   * The GBIF Network associated to the publishing dataset.
134   */
135  @Nullable
136  public List<UUID> getNetworkKeys() {
137    return networkKeys;
138  }
139
140  public void setNetworkKeys(List<UUID> networkKeys) {
141    this.networkKeys = networkKeys;
142  }
143
144  /**
145   * Technical installation that publishes this occurrence record.
146   */
147  @Nullable
148  public UUID getInstallationKey() {
149    return installationKey;
150  }
151
152  public void setInstallationKey(UUID installationKey) {
153    this.installationKey = installationKey;
154  }
155
156  /**
157   * The country of the organization that publishes the dataset to which the occurrence belongs.
158   */
159  @Nullable
160  public Country getPublishingCountry() {
161    return publishingCountry;
162  }
163
164  public void setPublishingCountry(Country publishingCountry) {
165    this.publishingCountry = publishingCountry;
166  }
167
168  @NotNull
169  public EndpointType getProtocol() {
170    return protocol;
171  }
172
173  public void setProtocol(EndpointType protocol) {
174    this.protocol = protocol;
175  }
176
177  /**
178   * The date this record was last crawled/harvested from the endpoint.
179   */
180  @Nullable
181  public Date getLastCrawled() {
182    return lastCrawled == null ? null : new Date(lastCrawled.getTime());
183  }
184
185  public void setLastCrawled(@Nullable Date lastCrawled) {
186    this.lastCrawled = lastCrawled == null ? null : new Date(lastCrawled.getTime());
187  }
188
189  /**
190   * The date this record was last parsed from raw xml/json into verbatim verbatimFields.
191   */
192  @Nullable
193  public Date getLastParsed() {
194    return lastParsed;
195  }
196
197  public void setLastParsed(@Nullable Date lastParsed) {
198    this.lastParsed = lastParsed == null ? null : new Date(lastParsed.getTime());
199  }
200
201  /**
202   * Crawling attempt id.
203   */
204  @Nullable
205  public Integer getCrawlId() {
206    return crawlId;
207  }
208
209  public void setCrawlId(Integer crawlId) {
210    this.crawlId = crawlId;
211  }
212
213  /**
214   * GBIF project identifier.
215   */
216  @Nullable
217  public String getProjectId() {
218    return projectId;
219  }
220
221  public void setProjectId(String projectId) {
222    this.projectId = projectId;
223  }
224
225  /**
226   * GBIF programme acronym/identifier.
227   */
228  @Nullable
229  public String getProgrammeAcronym() {
230    return programmeAcronym;
231  }
232
233  public void setProgrammeAcronym(String programmeAcronym) {
234    this.programmeAcronym = programmeAcronym;
235  }
236
237  /**
238   * A map holding all verbatim core terms.
239   */
240  @NotNull
241  @JsonIgnore
242  public Map<Term, String> getVerbatimFields() {
243    return verbatimFields;
244  }
245
246  public void setVerbatimFields(Map<Term, String> verbatimFields) {
247    this.verbatimFields = verbatimFields;
248  }
249
250  /**
251   * A map holding all verbatim extension terms.
252   */
253  @NotNull
254  @JsonSerialize(keyUsing = ExtensionSerializer.class, contentUsing = TermMapListSerializer.class)
255  @JsonDeserialize(keyUsing = ExtensionKeyDeserializer.class, contentUsing = TermMapListDeserializer.class)
256  public Map<Extension, List<Map<Term, String>>> getExtensions() {
257    return extensions;
258  }
259
260  public void setExtensions(Map<Extension, List<Map<Term, String>>> extensions) {
261    this.extensions = extensions;
262  }
263
264  @Override
265  public boolean equals(Object o) {
266    if (this == o) {
267      return true;
268    }
269    if (o == null || getClass() != o.getClass()) {
270      return false;
271    }
272    VerbatimOccurrence that = (VerbatimOccurrence) o;
273    return Objects.equals(key, that.key) &&
274      Objects.equals(datasetKey, that.datasetKey) &&
275      Objects.equals(publishingOrgKey, that.publishingOrgKey) &&
276      Objects.equals(networkKeys, that.networkKeys) &&
277      Objects.equals(installationKey, that.installationKey) &&
278      publishingCountry == that.publishingCountry &&
279      protocol == that.protocol &&
280      Objects.equals(lastCrawled, that.lastCrawled) &&
281      Objects.equals(lastParsed, that.lastParsed) &&
282      Objects.equals(crawlId, that.crawlId) &&
283      Objects.equals(projectId, that.projectId) &&
284      Objects.equals(programmeAcronym, that.programmeAcronym) &&
285      Objects.equals(verbatimFields, that.verbatimFields) &&
286      Objects.equals(extensions, that.extensions);
287  }
288
289  @Override
290  public int hashCode() {
291    return Objects
292      .hash(key, datasetKey, publishingOrgKey, networkKeys, installationKey, publishingCountry,
293        protocol, lastCrawled, lastParsed, crawlId, projectId, programmeAcronym, verbatimFields,
294        extensions);
295  }
296
297  @Override
298  public String toString() {
299    return new StringJoiner(", ", VerbatimOccurrence.class.getSimpleName() + "[", "]")
300      .add("key=" + key)
301      .add("datasetKey=" + datasetKey)
302      .add("publishingOrgKey=" + publishingOrgKey)
303      .add("networkKeys=" + networkKeys)
304      .add("installationKey=" + installationKey)
305      .add("publishingCountry=" + publishingCountry)
306      .add("protocol=" + protocol)
307      .add("lastCrawled=" + lastCrawled)
308      .add("lastParsed=" + lastParsed)
309      .add("crawlId=" + crawlId)
310      .add("projectId='" + projectId + "'")
311      .add("programmeAcronym='" + programmeAcronym + "'")
312      .add("extensions=" + extensions)
313      .toString();
314  }
315
316  /**
317   * This private method is only for deserialization via jackson and not exposed anywhere else!
318   */
319  @JsonAnySetter
320  private void addJsonVerbatimField(String key, String value) {
321    if(StringUtils.isNotEmpty(value)) {
322      Term t = TermFactory.instance().findTerm(key);
323      verbatimFields.put(t, value);
324    }
325  }
326
327  /**
328   * This private method is only for serialization via jackson and not exposed anywhere else!
329   * It maps the verbatimField terms into properties with their full qualified name.
330   */
331  @JsonAnyGetter
332  private Map<String, String> jsonVerbatimFields() { // note: for 1.6.0 MUST use non-getter name; otherwise doesn't matter
333    Map<String, String> extendedProps = new HashMap<>();
334    for (Map.Entry<Term, String> prop : verbatimFields.entrySet()) {
335      extendedProps.put(prop.getKey().qualifiedName(), prop.getValue());
336    }
337    return extendedProps;
338  }
