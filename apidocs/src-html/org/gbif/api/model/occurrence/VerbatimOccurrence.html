040/**
041 * An extended map holding all core terms of an occurrence record.
042 * Major extensions that we index are also supported, i.e. media, identifiers and measurements or facts.
043 */
044@SuppressWarnings("unused")
045public class VerbatimOccurrence {
046

048    description = "Unique GBIF key for the occurrence.\n\n" +
049      "We aim to keep these keys stable, but this is not possible in every case."
050  )
051  private Long key;
052
053  @Schema(
054    description = "The UUID of the GBIF dataset containing this occurrence."

056  private UUID datasetKey;
057
058  @Schema(
059    description = "The UUID of the organization which publishes the dataset containing this occurrence."
060  )
061  private UUID publishingOrgKey;
062
063  @Schema(
064    description = "Any networks to which the dataset containing this occurrence is registered."
065  )
066  private List<UUID> networkKeys;
067
068  @Schema(
069    description = "The UUID of the technical installation hosted the dataset containing this occurrence."
070  )
071  private UUID installationKey;
072
073  @Schema(
074    description = "The UUID of the publishing organization which operates the technical installation hosting the " +
075      "dataset containing this occurrence."
076  )
077  private UUID hostingOrganizationKey;
078
079  @Schema(
080    description = "The country, territory or island based on ISO-3166 of the organization publishing the dataset " +
081      "containing this occurrence."
082  )
083  private Country publishingCountry;
084
085  @Schema(
086    description = "The technical protocol by which this occurrence was retrieved from the publisher's systems."
087  )
088  private EndpointType protocol;
089
090  @Schema(
091    description = "The time this occurrence was last retrieved from the publisher's systems."
092  )
093  private Date lastCrawled;
094
095  @Schema(
096    description = "The time this occurrence was last processed by GBIF's interpretation system “Pipelines”.\n\n" +
097      "This is the time the record was last changed in GBIF, **not** the time the record was last changed by the " +
098      "publisher.  Data is also reprocessed when we changed the taxonomic backbone, geographic data sources or " +
099      "other interpretation procedures.\n\n" +
100      "An earlier interpretation system distinguished between “parsing” and “interpretation”, but in the current " +
101      "system there is only one process — the two dates will always be the same."
102  )
103  private Date lastParsed;
104
105  @Schema(
106    description = "The sequence number of the attempt by GBIF to download (”crawl”), interpret and index the dataset " +
107      "to which this occurrence belongs."
108  )
109  private Integer crawlId;
110
111  /** GBIF Participation: Programme and Project */
112  @Schema(
113    description = "The identifier for a project, often assigned by a funded programme."
114  )
115  private String projectId;
116
117  @Schema(
118    description = "The identifier for a programme which funded the digitization of this occurrence."
119  )
120  private String programmeAcronym;
121
122  // the verbatim fields for the occurrence
123  @Schema(
124    description = "The verbatim fields for the occurrence, with Darwin Core terms as keys."
125  )
126  private Map<Term, String> verbatimFields = new HashMap<>();
127
128  // verbatim extension data
129  @Schema(
130    description = "The verbatim Darwin Core Archive extension fields for this occurrence.\n\n" +
131      "The main key is the record class term (the row type in Darwin Core Archive), within that are " +
132      " values with extension terms as keys."
133  )
134  private Map<String, List<Map<Term, String>>> extensions = new HashMap<>();
135
136  /**
137   * Get the value of a specific field (Term).
138   */
139  @Nullable
140  public String getVerbatimField(Term term) {
141    Objects.requireNonNull(term, "term can't be null");
142    return verbatimFields.get(term);
143  }
144
145  /**
146   * @return true if a verbatim field exists and is not null or an empty string
147   */
148  public boolean hasVerbatimField(Term term) {
149    Objects.requireNonNull(term, "term can't be null");
150    return StringUtils.isNotEmpty(verbatimFields.get(term));
151  }
152
153  /**
154   * For setting a specific field without having to replace the entire verbatimFields Map.
155   *
156   * @param term       the field to set
157   * @param fieldValue the field's value
158   */
159  public void setVerbatimField(Term term, @Nullable String fieldValue) {
160    Objects.requireNonNull(term, "term can't be null");
161    verbatimFields.put(term, fieldValue);
162  }

337  @Override
338  public boolean equals(Object o) {
339    if (this == o) {
340      return true;
341    }
342    if (o == null || getClass() != o.getClass()) {
343      return false;
344    }
345    VerbatimOccurrence that = (VerbatimOccurrence) o;
346    return Objects.equals(key, that.key) &&
347      Objects.equals(datasetKey, that.datasetKey) &&
348      Objects.equals(publishingOrgKey, that.publishingOrgKey) &&
349      Objects.equals(networkKeys, that.networkKeys) &&
350      Objects.equals(installationKey, that.installationKey) &&
351      publishingCountry == that.publishingCountry &&
352      protocol == that.protocol &&
353      Objects.equals(lastCrawled, that.lastCrawled) &&
354      Objects.equals(lastParsed, that.lastParsed) &&
355      Objects.equals(crawlId, that.crawlId) &&
356      Objects.equals(projectId, that.projectId) &&
357      Objects.equals(programmeAcronym, that.programmeAcronym) &&
358      Objects.equals(verbatimFields, that.verbatimFields) &&
359      Objects.equals(extensions, that.extensions);
360  }
361

385      .add("extensions=" + extensions)
386      .toString();
387  }

389  /**
390   * This private method is only for deserialization via jackson and not exposed anywhere else!
391   */
392  @JsonAnySetter
393  private void addJsonVerbatimField(String key, String value) {
394    if(StringUtils.isNotEmpty(value)) {
395      Term t = TermFactory.instance().findTerm(key);
396      verbatimFields.put(t, value);
397    }
398  }
399
400  /**
401   * This private method is only for serialization via jackson and not exposed anywhere else!
402   * It maps the verbatimField terms into properties with their full qualified name.
403   */
404  @JsonAnyGetter
405  private Map<String, String> jsonVerbatimFields() { // note: for 1.6.0 MUST use non-getter name; otherwise doesn't matter
406    Map<String, String> extendedProps = new HashMap<>();
407    for (Map.Entry<Term, String> prop : verbatimFields.entrySet()) {
408      extendedProps.put(prop.getKey().qualifiedName(), prop.getValue());
409    }
410    return extendedProps;
411  }
