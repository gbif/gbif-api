043/**
044 * An extended map holding all core terms of an occurrence record.
045 * Major extensions that we index are also supported, i.e. media, identifiers and measurements or facts.
046 */
047public class VerbatimOccurrence {
048
049  private Long key;
050  private UUID datasetKey;
051  private UUID publishingOrgKey;
052  private List<UUID> networkKeys;
053  private UUID installationKey;
054  private Country publishingCountry;
055  private EndpointType protocol;
056  private Date lastCrawled;
057  private Date lastParsed;
058  private Integer crawlId;
059
060  // the verbatim fields for the occurrence
061  private Map<Term, String> verbatimFields = Maps.newHashMap();
062  // verbatim extension data
063  private Map<Extension, List<Map<Term, String>>> extensions = Maps.newHashMap();
064
065  /**
066   * Get the value of a specific field (Term).
067   */
068  @Nullable
069  public String getVerbatimField(Term term) {
070    checkNotNull(term, "term can't be null");
071    return verbatimFields.get(term);
072  }
073
074  /**
075   * @return true if a verbatim field exists and is not null or an empty string
076   */
077  public boolean hasVerbatimField(Term term) {
078    checkNotNull(term, "term can't be null");
079    return !Strings.isNullOrEmpty(verbatimFields.get(term));
080  }
081
082  /**
083   * For setting a specific field without having to replace the entire verbatimFields Map.
084   *
085   * @param term the field to set
086   * @param fieldValue the field's value
087   */
088  public void setVerbatimField(Term term, @Nullable String fieldValue) {
089    checkNotNull(term, "term can't be null");
090    verbatimFields.put(term, fieldValue);
091  }
092
093  /**
094   * The GBIF assigned, persistent key to the occurrence record.
095   * OccurrenceID itself is kept in the verbatim verbatimFields map.
096   */
097  @NotNull
098  public Long getKey() {
099    return key;
100  }
101
102  public void setKey(Long key) {
103    this.key = key;
104  }
105
106  @NotNull
107  public UUID getDatasetKey() {
108    return datasetKey;
109  }
110
111  public void setDatasetKey(UUID datasetKey) {
112    this.datasetKey = datasetKey;
113  }
114
115  @NotNull
116  public UUID getPublishingOrgKey() {
117    return publishingOrgKey;
118  }
119
120  public void setPublishingOrgKey(UUID publishingOrgKey) {
121    this.publishingOrgKey = publishingOrgKey;
122  }
123
124  /**
125   * The GBIF Network associated to the publishing dataset.
126   */
127  @Nullable
128  public List<UUID> getNetworkKeys() {
129    return networkKeys;
130  }
131
132  public void setNetworkKeys(List<UUID> networkKeys) {
133    this.networkKeys = networkKeys;
134  }
135
136  /**
137   * Technical installation that publishes this occurrence record.
138   */
139  @Nullable
140  public UUID getInstallationKey() {
141    return installationKey;
142  }
143
144  public void setInstallationKey(UUID installationKey) {
145    this.installationKey = installationKey;
146  }
147
148  /**
149   * The country of the organization that publishes the dataset to which the occurrence belongs.
150   */
151  @Nullable
152  public Country getPublishingCountry() {
153    return publishingCountry;
154  }
155
156  public void setPublishingCountry(Country publishingCountry) {
157    this.publishingCountry = publishingCountry;
158  }
159
160  @NotNull
161  public EndpointType getProtocol() {
162    return protocol;
163  }
164
165  public void setProtocol(EndpointType protocol) {
166    this.protocol = protocol;
167  }
168
169  @Nullable
170  /**
171   * The date this record was last crawled/harvested from the endpoint.
172   */
173  public Date getLastCrawled() {
174    return lastCrawled == null ? null : new Date(lastCrawled.getTime());
175  }
176
177  public void setLastCrawled(@Nullable Date lastCrawled) {
178    this.lastCrawled = lastCrawled == null ? null : new Date(lastCrawled.getTime());
179  }
180
181  /**
182   * The date this record was last parsed from raw xml/json into verbatim verbatimFields.
183   */
184  @Nullable
185  public Date getLastParsed() {
186    return lastParsed;
187  }
188
189  public void setLastParsed(@Nullable Date lastParsed) {
190    this.lastParsed = lastParsed == null ? null : new Date(lastParsed.getTime());
191  }
192
193  /**
194   * Crawling attempt id.
195   */
196  @Nullable
197  public Integer getCrawlId() {
198    return crawlId;
199  }
200
201  public void setCrawlId(Integer crawlId) {
202    this.crawlId = crawlId;
203  }
204
205  /**
206   * A map holding all verbatim core terms.
207   */
208  @NotNull
209  @JsonIgnore
210  public Map<Term, String> getVerbatimFields() {
211    return verbatimFields;
212  }
213
214  public void setVerbatimFields(Map<Term, String> verbatimFields) {
215    this.verbatimFields = verbatimFields;
216  }
217
218  /**
219   * A map holding all verbatim extension terms.
220   */
221  @NotNull
222  @JsonSerialize(keyUsing = ExtensionSerializer.class, contentUsing = TermMapListSerializer.class)
223  @JsonDeserialize(keyUsing = ExtensionKeyDeserializer.class, contentUsing = TermMapListDeserializer.class)
224  public Map<Extension, List<Map<Term, String>>> getExtensions() {
225    return extensions;
226  }
227
228  public void setExtensions(Map<Extension, List<Map<Term, String>>> extensions) {
229    this.extensions = extensions;
230  }
231
232  @Override
233  public String toString() {
234    return Objects.toStringHelper(this)
235      .add("key", key)
236      .add("lastParsed", lastParsed)
237      .add("datasetKey", datasetKey)
238      .add("publishingOrgKey", publishingOrgKey)
239      .add("publishingCountry", publishingCountry)
240      .add("installationKey", installationKey)
241      .add("networkKeys", networkKeys)
242      .add("protocol", protocol)
243      .add("crawlId",crawlId)
244      .add("lastCrawled", lastCrawled)
245      .add("extensions", extensions)
246      .toString();
247  }
248
249  @Override
250  public int hashCode() {
251    return Objects
252      .hashCode(key, datasetKey, publishingOrgKey, publishingCountry, protocol, lastCrawled, lastParsed, crawlId,
253                verbatimFields, extensions);
254  }
255
256  @Override
257  public boolean equals(Object obj) {
258    if (this == obj) {
259      return true;
260    }
261    if (obj == null || getClass() != obj.getClass()) {
262      return false;
263    }
264    final VerbatimOccurrence other = (VerbatimOccurrence) obj;
265    return Objects.equal(this.key, other.key)
266      && Objects.equal(this.datasetKey, other.datasetKey)
267      && Objects.equal(this.publishingOrgKey, other.publishingOrgKey)
268      && Objects.equal(this.publishingCountry, other.publishingCountry)
269      && Objects.equal(this.installationKey, other.installationKey)
270      && Objects.equal(this.networkKeys, other.networkKeys)
271      && Objects.equal(this.protocol, other.protocol)
272      && Objects.equal(this.lastCrawled, other.lastCrawled)
273      && Objects.equal(this.lastParsed, other.lastParsed)
274      && Objects.equal(this.verbatimFields, other.verbatimFields)
275      && Objects.equal(this.extensions, other.extensions)
276      && Objects.equal(this.crawlId, other.crawlId);
277  }
278
279  /**
280   * This private method is only for deserialization via jackson and not exposed anywhere else!
281   */
282  @JsonAnySetter
283  private void addJsonVerbatimField(String key, String value) {
284    if(!Strings.isNullOrEmpty(value)) {
285      Term t = TermFactory.instance().findTerm(key);
286      verbatimFields.put(t, value);
287    }
288  }
289
290  /**
291   * This private method is only for serialization via jackson and not exposed anywhere else!
292   * It maps the verbatimField terms into properties with their full qualified name.
293   */
294  @JsonAnyGetter
295  private Map<String, String> jsonVerbatimFields() { // note: for 1.6.0 MUST use non-getter name; otherwise doesn't matter
296    Map<String, String> extendedProps = Maps.newHashMap();
297    for (Map.Entry<Term, String> prop : verbatimFields.entrySet()) {
298      extendedProps.put(prop.getKey().qualifiedName(), prop.getValue());
299    }
300    return extendedProps;
301  }
