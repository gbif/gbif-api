013package org.gbif.api.model.occurrence;
014
015import org.gbif.api.jackson.ExtensionKeyDeserializer;
016import org.gbif.api.jackson.ExtensionSerializer;
017import org.gbif.api.jackson.TermMapListDeserializer;
018import org.gbif.api.jackson.TermMapListSerializer;
019import org.gbif.api.vocabulary.Country;
020import org.gbif.api.vocabulary.EndpointType;
021import org.gbif.api.vocabulary.Extension;
022import org.gbif.dwc.terms.Term;
023import org.gbif.dwc.terms.TermFactory;
024
025import java.util.Date;
026import java.util.List;
027import java.util.Map;
028import java.util.UUID;
029
030import javax.annotation.Nullable;
031import javax.validation.constraints.NotNull;
032
033import com.google.common.base.Objects;
034import com.google.common.base.Strings;
035import com.google.common.collect.Maps;
036import org.codehaus.jackson.annotate.JsonAnyGetter;
037import org.codehaus.jackson.annotate.JsonAnySetter;
038import org.codehaus.jackson.annotate.JsonIgnore;
039import org.codehaus.jackson.map.annotate.JsonDeserialize;
040import org.codehaus.jackson.map.annotate.JsonSerialize;
041import static com.google.common.base.Preconditions.checkNotNull;
042
043/**
044 * An extended map holding all core terms of an occurrence record.
045 * Major extensions that we index are also supported, i.e. media, identifiers and measurements or facts.
046 */
047public class VerbatimOccurrence {
048
049  private Integer key;
050  private UUID datasetKey;
051  private UUID publishingOrgKey;
052  private Country publishingCountry;
053  private EndpointType protocol;
054  private Date lastCrawled;
055  private Date lastParsed;
056
057  // the verbatim fields for the occurrence
058  private Map<Term, String> verbatimFields = Maps.newHashMap();
059  // verbatim extension data
060  private Map<Extension, List<Map<Term, String>>> extensions = Maps.newHashMap();
061
062  /**
063   * Get the value of a specific field (Term).
064   */
065  @Nullable
066  public String getVerbatimField(Term term) {
067    checkNotNull(term, "term can't be null");
068    return verbatimFields.get(term);
069  }
070
071  /**
072   * @return true if a verbatim field exists and is not null or an empty string
073   */
074  public boolean hasVerbatimField(Term term) {
075    checkNotNull(term, "term can't be null");
076    return !Strings.isNullOrEmpty(verbatimFields.get(term));
077  }
078
079  /**
080   * For setting a specific field without having to replace the entire verbatimFields Map.
081   *
082   * @param term the field to set
083   * @param fieldValue the field's value
084   */
085  public void setVerbatimField(Term term, @Nullable String fieldValue) {
086    checkNotNull(term, "term can't be null");
087    verbatimFields.put(term, fieldValue);
088  }
089
090  /**
091   * The GBIF assigned, persistent key to the occurrence record.
092   * OccurrenceID itself is kept in the verbatim verbatimFields map.
093   */
094  @NotNull
095  public Integer getKey() {
096    return key;
097  }
098
099  public void setKey(Integer key) {
100    this.key = key;
101  }
102
103  @NotNull
104  public UUID getDatasetKey() {
105    return datasetKey;
106  }
107
108  public void setDatasetKey(UUID datasetKey) {
109    this.datasetKey = datasetKey;
110  }
111
112  @NotNull
113  public UUID getPublishingOrgKey() {
114    return publishingOrgKey;
115  }
116
117  public void setPublishingOrgKey(UUID publishingOrgKey) {
118    this.publishingOrgKey = publishingOrgKey;
119  }
120
121  /**
122   * The country of the organization that publishes the dataset to which the occurrence belongs.
123   */
124  @Nullable
125  public Country getPublishingCountry() {
126    return publishingCountry;
127  }
128
129  public void setPublishingCountry(Country publishingCountry) {
130    this.publishingCountry = publishingCountry;
131  }
132
133  @NotNull
134  public EndpointType getProtocol() {
135    return protocol;
136  }
137
138  public void setProtocol(EndpointType protocol) {
139    this.protocol = protocol;
140  }
141
142  @Nullable
143  /**
144   * The date this record was last crawled/harvested from the endpoint.
145   */
146  public Date getLastCrawled() {
147    return lastCrawled == null ? null : new Date(lastCrawled.getTime());
148  }
149
150  public void setLastCrawled(@Nullable Date lastCrawled) {
151    this.lastCrawled = lastCrawled == null ? null : new Date(lastCrawled.getTime());
152  }
153
154  /**
155   * The date this record was last parsed from raw xml/json into verbatim verbatimFields.
156   */
157  @Nullable
158  public Date getLastParsed() {
159    return lastParsed;
160  }
161
162  public void setLastParsed(@Nullable Date lastParsed) {
163    this.lastParsed = lastParsed == null ? null : new Date(lastParsed.getTime());
164  }
165
166  /**
167   * A map holding all verbatim core terms.
168   */
169  @NotNull
170  @JsonIgnore
171  public Map<Term, String> getVerbatimFields() {
172    return verbatimFields;
173  }
174
175  public void setVerbatimFields(Map<Term, String> verbatimFields) {
176    this.verbatimFields = verbatimFields;
177  }
178
179  /**
180   * A map holding all verbatim extension terms.
181   */
182  @NotNull
183  @JsonSerialize(keyUsing = ExtensionSerializer.class, contentUsing = TermMapListSerializer.class)
184  @JsonDeserialize(keyUsing = ExtensionKeyDeserializer.class, contentUsing = TermMapListDeserializer.class)
185  public Map<Extension, List<Map<Term, String>>> getExtensions() {
186    return extensions;
187  }
188
189  public void setExtensions(Map<Extension, List<Map<Term, String>>> extensions) {
190    this.extensions = extensions;
191  }
192
193  @Override
194  public String toString() {
195    return Objects.toStringHelper(this).add("lastParsed", lastParsed).add("key", key).add("datasetKey", datasetKey)
196      .add("publishingOrgKey", publishingOrgKey).add("publishingCountry", publishingCountry).add("protocol", protocol)
197      .add("lastCrawled", lastCrawled)
198      .add("extensions", extensions)
199      .toString();
200  }
201
202  @Override
203  public int hashCode() {
204    return Objects
205      .hashCode(key, datasetKey, publishingOrgKey, publishingCountry, protocol, lastCrawled, lastParsed,
206        verbatimFields,
207        extensions);
208  }
209
210  @Override
211  public boolean equals(Object obj) {
212    if (this == obj) {
213      return true;
214    }
215    if (obj == null || getClass() != obj.getClass()) {
216      return false;
217    }
218    final VerbatimOccurrence other = (VerbatimOccurrence) obj;
219    return Objects.equal(this.key, other.key)
220      && Objects.equal(this.datasetKey, other.datasetKey)
221      && Objects.equal(this.publishingOrgKey, other.publishingOrgKey)
222      && Objects.equal(this.publishingCountry, other.publishingCountry)
223      && Objects.equal(this.protocol, other.protocol)
224      && Objects.equal(this.lastCrawled, other.lastCrawled)
225      && Objects.equal(this.lastParsed, other.lastParsed)
226      && Objects.equal(this.verbatimFields, other.verbatimFields)
227      && Objects.equal(this.extensions, other.extensions);
228  }
229
230  /**
231   * This private method is only for deserialization via jackson and not exposed anywhere else!
232   */
233  @JsonAnySetter
234  private void addJsonVerbatimField(String key, String value) {
235    if(!Strings.isNullOrEmpty(value)) {
236      Term t = TermFactory.instance().findTerm(key);
237      verbatimFields.put(t, value);
238    }
239  }
240
241  /**
242   * This private method is only for serialization via jackson and not exposed anywhere else!
243   * It maps the verbatimField terms into properties with their full qualified name.
244   */
245  @JsonAnyGetter
246  private Map<String, String> jsonVerbatimFields() { // note: for 1.6.0 MUST use non-getter name; otherwise doesn't matter
247    Map<String, String> extendedProps = Maps.newHashMap();
248    for (Map.Entry<Term, String> prop : verbatimFields.entrySet()) {
249      extendedProps.put(prop.getKey().qualifiedName(), prop.getValue());
250    }
251    return extendedProps;
252  }
253}
