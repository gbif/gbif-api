001package org.gbif.api.model.occurrence;
002
003import java.util.Collection;
004import javax.annotation.Nullable;
005import javax.validation.Valid;
006import org.codehaus.jackson.annotate.JsonCreator;
007import org.codehaus.jackson.annotate.JsonProperty;
008import org.gbif.api.model.occurrence.predicate.Predicate;
009import com.google.common.base.MoreObjects;
010import com.google.common.base.Objects;
011
012/**
013 * An occurrence download request whose filters are based on predicates ( see {@link Predicate}).
014 */
015public class PredicateDownloadRequest extends DownloadRequest {
016
017
018  //Default download format.
019  private static final DownloadFormat DEFAULT_DOWNLOAD_FORMAT = DownloadFormat.SIMPLE_CSV;
020
021  private Predicate predicate;
022
023  public PredicateDownloadRequest() {
024
025  }
026
027  /**
028   * Full constructor. Used to create instances using JSON serialization.
029   */
030  @JsonCreator
031  public PredicateDownloadRequest(@JsonProperty("predicate") Predicate predicate,
032                                  @JsonProperty("creator") @Nullable String creator,
033                                  @JsonProperty("notificationAddresses") @Nullable Collection<String> notificationAddresses,
034                                  @JsonProperty("sendNotification") boolean sendNotification,
035                                  @JsonProperty("format") DownloadFormat format) {
036    super(creator, notificationAddresses, sendNotification, format == null ? DEFAULT_DOWNLOAD_FORMAT : format);
037    this.predicate = predicate;
038  }
039
040  /**
041   * @return the download filter
042   */
043  @Nullable
044  @Valid
045  public Predicate getPredicate() {
046    return predicate;
047  }
048
049  public void setPredicate(Predicate predicate) {
050    this.predicate = predicate;
051  }
052  
053  @Override
054  public boolean equals(Object obj) {
055    if (this == obj) {
056      return true;
057    }
058    if (!(obj instanceof PredicateDownloadRequest)) {
059      return false;
060    }
061
062    PredicateDownloadRequest that = (PredicateDownloadRequest) obj;
063    return super.equals(that) && Objects.equal(this.predicate, that.predicate);
064  }
065
066  @Override
067  public int hashCode() {
068    return Objects.hashCode(super.hashCode(), predicate);
069  }
070
071  @Override
072  public String toString() {
073    return MoreObjects.toStringHelper(this).addValue(super.toString()).add("predicate", predicate).toString();
074  }
075}
