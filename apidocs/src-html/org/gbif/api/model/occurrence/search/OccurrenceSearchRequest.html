034/**

039  private Boolean matchCase;

041  public OccurrenceSearchRequest() {
042    // empty block

045
046  /**
047   * This flag enables the use of case-sensitive matches and aggregations on certain search parameters.
048   * <p>
049   * Fields that support this feature are: occurrenceId, recordedBy, samplingProtocol, catalogNumber, collectionCode,
050   * institutionCode, eventId, parentEventId, waterBody, stateProvince, recordNumber, identifiedBy, organismId and locality.
051   * <p>
052   * This is an experimental feature and its implementation map change or be removed at any time.
053   * <p>
054   * Be aware that this is not a per-field flag, all possible fields will match case sensitively.
055   */
056  @Experimental
057  public Boolean isMatchCase() {
058    return Optional.ofNullable(matchCase).orElse(Boolean.FALSE);
059  }
060
061  @Experimental
062  public void setMatchCase(Boolean matchCase) {
063    this.matchCase = matchCase;
064  }
065

067    super(offset, limit);
068  }
