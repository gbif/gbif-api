013package org.gbif.api.model.occurrence.search;
014
015
016import org.gbif.api.model.common.search.SearchParameter;
017import org.gbif.api.vocabulary.BasisOfRecord;
018import org.gbif.api.vocabulary.Continent;
019import org.gbif.api.vocabulary.Country;
020import org.gbif.api.vocabulary.EstablishmentMeans;
021import org.gbif.api.vocabulary.MediaType;
022import org.gbif.api.vocabulary.OccurrenceIssue;
023import org.gbif.api.vocabulary.TypeStatus;
024
025import java.util.Date;
026import java.util.UUID;
027
028/**
029 * Supported query parameters by the occurrence search and download service.
030 * For download predicates only the numerical types support comparisons other than equals.
031 */
032public enum OccurrenceSearchParameter implements SearchParameter {
033
034  /**
035   * The dataset key as a uuid.
036   */
037  DATASET_KEY(UUID.class),
038
039  /**
040   * The 4 digit year. A year of 98 will be 98 after christ, no 1998.
041   * This parameter accepts comma separated range values, e.g.:
042   * <dl>
043   * <dt>*,1810</dt>
044   * <dd>Found before or equal to 1810</dd>
045   * <dt>1848,1933</dt>
046   * <dd>Year between 1848 and 1933</dd>
047   * </dl>
048   */
049  YEAR(Integer.class),
050
051  /**
052   * The month of the year, starting with 1 for January.
053   * A month query can be used to retrieve seasonal records when used without a year or a year range.
054   * This parameter accepts comma separated range values, e.g.:
055   * <dl>
056   * <dt>4,8</dt>
057   * <dd>Month between April and August</dd>
058   * </dl>
059   */
060  MONTH(Integer.class),
061
062
063  /**
064   * Event date (date the occurrence was recorded) in ISO 8601 formats:yyyy, yyyy-MM, yyyy-MM-dd and MM-dd.
065   * This parameter accepts comma separated range values, examples of valid ranges are:
066   * <dl>
067   * <dt>2001-02-11,2010-01-10</dt>
068   * <dd>Dates between 2001-02-11 and 2010-01-10</dd>
069   * <dt>2001-02,2010-01</dt>
070   * <dd>Dates between first day of 2001-02 and last day of 2010-01</dd>
071   * <dt>2001,2010</dt>
072   * <dd>Dates between first day of 2001 and last day of 2010</dd>
073   * <dt>2001,2010-01</dt>
074   * <dd>Dates between first day of 2001 and last day of 2010-01</dd>
075   * <dt>2001-01-10,2010</dt>
076   * <dd>Dates between 2001-01-10 and last day of 2010</dd>
077   * <dt>2001-01-10,*</dt>
078   * <dd>Dates after 2001-01-10</dd>
079   * <dt>*,2001-01-10</dt>
080   * <dd>Dates before 2001-01-10</dd>
081   * <dt>*</dt>
082   * <dd>all dates</dd>
083   * </dl>
084   */
085  EVENT_DATE(Date.class),
086
087  /**
088   * Last interpreted date in ISO 8601 formats:yyyy, yyyy-MM, yyyy-MM-dd and MM-dd.
089   * This parameter accepts comma separated range values, examples of valid ranges are:
090   * <dl>
091   * <dt>2001-02-11,2010-01-10</dt>
092   * <dd>Dates between 2001-02-11 and 2010-01-10</dd>
093   * <dt>2001-02,2010-01</dt>
094   * <dd>Dates between first day of 2001-02 and last day of 2010-01</dd>
095   * <dt>2001,2010</dt>
096   * <dd>Dates between first day of 2001 and last day of 2010</dd>
097   * <dt>2001,2010-01</dt>
098   * <dd>Dates between first day of 2001 and last day of 2010-01</dd>
099   * <dt>2001-01-10,2010</dt>
100   * <dd>Dates between 2001-01-10 and last day of 2010</dd>
101   * <dt>2001-01-10,*</dt>
102   * <dd>Dates after 2001-01-10</dd>
103   * <dt>*,2001-01-10</dt>
104   * <dd>Dates before 2001-01-10</dd>
105   * <dt>*</dt>
106   * <dd>all dates</dd>
107   * </dl>
108   */
109  LAST_INTERPRETED(Date.class),
110
111  /**
112   * Latitude in decimals between -90 and 90 based on WGS 84.
113   */
114  DECIMAL_LATITUDE(Double.class),
115
116  /**
117   * Longitude in decimals between -180 and 180 based on WGS 84.
118   */
119  DECIMAL_LONGITUDE(Double.class),
120
121  /**
122   * Country the occurrence was recorded in.
123   */
124  COUNTRY(Country.class),
125
126  /**
127   * Continent the occurrence was recorded in.
128   */
129  CONTINENT(Continent.class),
130
131  /**
132   * The country of the organization that publishes the dataset the occurrence belongs to.
133   */
134  PUBLISHING_COUNTRY(Country.class),
135
136  /**
137   * Altitude/elevation in meters above sea level.
138   * This parameter accepts comma separated range values, e.g.:
139   * <dl>
140   * <dt>*,100</dt>
141   * <dd>Altitude below or equals 100m</dd>
142   * <dt>100,*</dt>
143   * <dd>Altitude above or equals 100m</dd>
144   * <dt>-2,8.8</dt>
145   * <dd>Altitude between or equals -2m and 8.8m</dd>
146   * </dl>
147   */
148  ELEVATION(Double.class),
149
150  /**
151   * Depth in meters relative to altitude. For example 10 meters below a lake surface with given altitude.
152   * This parameter accepts comma separated range values, e.g.:
153   * <dl>
154   * <dt>*,10</dt>
155   * <dd>Depth below or equals 10m</dd>
156   * <dt>100,*</dt>
157   * <dd>Depth above or equals 100m</dd>
158   * <dt>12.1,28.8</dt>
159   * <dd>Depth between or equals 12.1m and 28.8m</dd>
160   * </dl>
161   */
162  DEPTH(Double.class),
163
164  /**
165   * An identifier of any form assigned by the source to identify the institution
166   * the record belongs to. Not guaranteed to be unique.
167   */
168  INSTITUTION_CODE(String.class),
169
170  /**
171   * An identifier of any form assigned by the source to identify the physical collection or digital dataset
172   * uniquely within the context of an institution.
173   */
174  COLLECTION_CODE(String.class),
175
176  /**
177   * An identifier of any form assigned by the source within a physical collection or digital dataset for the record
178   * which may not be unique, but should be fairly unique in combination with the institution and collection code.
179   */
180  CATALOG_NUMBER(String.class),
181
182  /**
183   * The person who recorded the occurrence.
184   */
185  RECORDED_BY(String.class),
186
187  /**
188   * An identifier given to the Occurrence at the time it was recorded.
189   */
190  RECORD_NUMBER(String.class),
191
192  /**
193   * A basis of record enumeration value.
194   */
195  BASIS_OF_RECORD(BasisOfRecord.class),
196
197  /**
198   * A taxon key from the GBIF backbone. All included and synonym taxa are included in the search, so a search for
199   * aves with taxonKey=212 will match all birds, no matter which species.
200   */
201  TAXON_KEY(Integer.class),
202
203  /**
204   * Searches the interpreted, full scientific name of the occurrence.
205   */
206  SCIENTIFIC_NAME(String.class),
207
208
209  /**
210   * Searches for occurrence records which contain a value on its coordinate fields (latitude and longitude).
211   * HAS_COORDINATE=true searches for occurrence records with a coordinate value.
212   * HAS_COORDINATE=false searches for occurrence records without a coordinate value.
213   */
214  HAS_COORDINATE(Boolean.class),
215
216
217  /**
218   * Geometry in <a href="http://en.wikipedia.org/wiki/Well-known_text">Well Known Text</a> (WKT) format.
219   * E.g.: POLYGON ((30.0 10.0, 10.12 20.23, 20 40, 40 40, 30 10)).
220   * Multi geometries like MULTIPOLYGON are not supported and multiple parameters should be used instead.
221   * Valid geometries are:
222   * <ul>
223   * <li>POINT</li>
224   * <li>LINESTRING</li>
225   * <li>POLYGON</li>
226   * <li>LINEARRING</li>
227   * </ul>
228   */
229  GEOMETRY(String.class),
230
231
232  /**
233   * Includes/excludes occurrence records which contain geospatial issues for their coordinate.
234   * See {@link org.gbif.api.vocabulary.OccurrenceIssue#GEOSPATIAL_RULES}
235   * HAS_GEOSPATIAL_ISSUE=true include records with spatial issues.
236   * HAS_GEOSPATIAL_ISSUE=false exclude records with spatial issues.
237   * The absence of this parameter returns any record with or without spatial issues.
238   */
239  HAS_GEOSPATIAL_ISSUE(Boolean.class),
240
241  /**
242   * Searches occurrence for those that have a specific issue.
243   */
244  ISSUE(OccurrenceIssue.class),
245
246  /**
247   * Nomenclatural type (type status, typified scientific name, publication) applied to the subject.
248   */
249  TYPE_STATUS(TypeStatus.class),
250
251
252  /**
253   * The kind of media object.
254   * Recommended terms from the DCMI Type Vocabulary are StillImage, Sound or MovingImage for GBIF to index and show the
255   * media files.
256   */
257  MEDIA_TYPE(MediaType.class),
258
259  /**
260   *  An identifier for the Occurrence (as opposed to a particular digital record of the occurrence).
261   *  In the absence of a persistent global unique identifier, construct one from a combination of identifiers in the
262   *  record that will most closely make the occurrenceID globally unique.
263   */
264  OCCURRENCE_ID(String.class),
265
266  /**
267   * The process by which the biological individual(s) represented in the Occurrence became established at the location.
268   */
269  ESTABLISHMENT_MEANS(EstablishmentMeans.class);
270
271  private final Class<?> type;
272
273  private OccurrenceSearchParameter(Class<?> type) {
274    this.type = type;
275  }
276
277  /**
278   * @return the data type expected for the value of the respective search parameter
279   */
280  public Class<?> type() {
281    return type;
282  }
283
284}
