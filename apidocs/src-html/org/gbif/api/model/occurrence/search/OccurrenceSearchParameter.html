031/**
032 * Supported query parameters by the occurrence search and download service.
033 * For download predicates only the numerical types support comparisons other than equals.
034 */
035public enum OccurrenceSearchParameter implements SearchParameter {
036
037  /**
038   * The dataset key as a UUID.
039   */
040  DATASET_KEY(UUID.class),
041
042  /**
043   * The 4 digit year. A year of 98 will be 98 common era, not 1998.
044   * This parameter accepts comma separated range values, e.g.:
045   * <dl>
046   * <dt>*,1810</dt>
047   * <dd>Found before or equal to 1810</dd>
048   * <dt>1848,1933</dt>
049   * <dd>Year between 1848 and 1933</dd>
050   * </dl>
051   */
052  YEAR(Integer.class),
053
054  /**
055   * The month of the year, starting with 1 for January.
056   * A month query can be used to retrieve seasonal records when used without a year or a year range.
057   * This parameter accepts comma separated range values, e.g.:
058   * <dl>
059   * <dt>4,8</dt>
060   * <dd>Month between April and August</dd>
061   * </dl>
062   */
063  MONTH(Integer.class),
064
065  /**
066   * Event date (date the occurrence was recorded) in ISO 8601 formats:yyyy, yyyy-MM, yyyy-MM-dd and MM-dd.
067   * This parameter accepts comma separated range values, examples of valid ranges are:
068   * <dl>
069   * <dt>2001-02-11,2010-01-10</dt>
070   * <dd>Dates between 2001-02-11 and 2010-01-10</dd>
071   * <dt>2001-02,2010-01</dt>
072   * <dd>Dates between first day of 2001-02 and last day of 2010-01</dd>
073   * <dt>2001,2010</dt>
074   * <dd>Dates between first day of 2001 and last day of 2010</dd>
075   * <dt>2001,2010-01</dt>
076   * <dd>Dates between first day of 2001 and last day of 2010-01</dd>
077   * <dt>2001-01-10,2010</dt>
078   * <dd>Dates between 2001-01-10 and last day of 2010</dd>
079   * <dt>2001-01-10,*</dt>
080   * <dd>Dates after 2001-01-10</dd>
081   * <dt>*,2001-01-10</dt>
082   * <dd>Dates before 2001-01-10</dd>
083   * <dt>*</dt>
084   * <dd>all dates</dd>
085   * </dl>
086   */
087  EVENT_DATE(Date.class),
088
089  /**
090   * An identifier for the set of information associated with an Event (something that occurs at a place and time).
091   * May be a global unique identifier or an identifier specific to the data set.
092   */
093  EVENT_ID(String.class),
094
095  /**
096   * An identifier for the broader Event that groups this and potentially other Events.
097   */
098  PARENT_EVENT_ID(String.class),
099
100  /**
101   * The name of, reference to, or description of the method or protocol used during an Event.
102   */
103  SAMPLING_PROTOCOL(String.class),
104
105  /**
106   * Last interpreted date in ISO 8601 formats:yyyy, yyyy-MM, yyyy-MM-dd and MM-dd.
107   * This parameter accepts comma separated range values, examples of valid ranges are:
108   * <dl>
109   * <dt>2001-02-11,2010-01-10</dt>
110   * <dd>Dates between 2001-02-11 and 2010-01-10</dd>
111   * <dt>2001-02,2010-01</dt>
112   * <dd>Dates between first day of 2001-02 and last day of 2010-01</dd>
113   * <dt>2001,2010</dt>
114   * <dd>Dates between first day of 2001 and last day of 2010</dd>
115   * <dt>2001,2010-01</dt>
116   * <dd>Dates between first day of 2001 and last day of 2010-01</dd>
117   * <dt>2001-01-10,2010</dt>
118   * <dd>Dates between 2001-01-10 and last day of 2010</dd>
119   * <dt>2001-01-10,*</dt>
120   * <dd>Dates after 2001-01-10</dd>
121   * <dt>*,2001-01-10</dt>
122   * <dd>Dates before 2001-01-10</dd>
123   * <dt>*</dt>
124   * <dd>all dates</dd>
125   * </dl>
126   */
127  LAST_INTERPRETED(Date.class),
128
129
130  /**
131   * Modified date in ISO 8601 formats:yyyy, yyyy-MM, yyyy-MM-dd and MM-dd.
132   * This parameter accepts comma separated range values, examples of valid ranges are:
133   * <dl>
134   * <dt>2001-02-11,2010-01-10</dt>
135   * <dd>Dates between 2001-02-11 and 2010-01-10</dd>
136   * <dt>2001-02,2010-01</dt>
137   * <dd>Dates between first day of 2001-02 and last day of 2010-01</dd>
138   * <dt>2001,2010</dt>
139   * <dd>Dates between first day of 2001 and last day of 2010</dd>
140   * <dt>2001,2010-01</dt>
141   * <dd>Dates between first day of 2001 and last day of 2010-01</dd>
142   * <dt>2001-01-10,2010</dt>
143   * <dd>Dates between 2001-01-10 and last day of 2010</dd>
144   * <dt>2001-01-10,*</dt>
145   * <dd>Dates after 2001-01-10</dd>
146   * <dt>*,2001-01-10</dt>
147   * <dd>Dates before 2001-01-10</dd>
148   * <dt>*</dt>
149   * <dd>all dates</dd>
150   * </dl>
151   */
152  MODIFIED(Date.class),
153
154  /**
155   * Latitude in decimals between -90 and 90 based on WGS 84.
156   */
157  DECIMAL_LATITUDE(Double.class),
158
159  /**
160   * Longitude in decimals between -180 and 180 based on WGS 84.
161   */
162  DECIMAL_LONGITUDE(Double.class),
163
164  /**
165   * The uncertainty of the coordinate in meters.
166   * This parameter accepts comma separated range values, e.g.:
167   * <dl>
168   * <dt>*,100</dt>
169   * <dd>Uncertainty below or equals 100m</dd>
170   * <dt>10000,*</dt>
171   * <dd>Uncertainty above or equals 10,000m</dd>
172   * <dt>1000,5000</dt>
173   * <dd>Uncertainty between or equals 1000m and 5000m
174   * </dd>
175   * </dl>
176   */
177  COORDINATE_UNCERTAINTY_IN_METERS(Double.class),
178
179  /**
180   * Country the occurrence was recorded in.
181   */
182  COUNTRY(Country.class),
183
184  /**
185   * Continent the occurrence was recorded in.
186   */
187  CONTINENT(Continent.class),
188
189  /**
190   * The country of the organization that publishes the dataset the occurrence belongs to.
191   */
192  PUBLISHING_COUNTRY(Country.class),
193
194  /**
195   * Altitude/elevation in meters above sea level.
196   * This parameter accepts comma separated range values, e.g.:
197   * <dl>
198   * <dt>*,100</dt>
199   * <dd>Altitude below or equals 100m</dd>
200   * <dt>100,*</dt>
201   * <dd>Altitude above or equals 100m</dd>
202   * <dt>-2,8.8</dt>
203   * <dd>Altitude between or equals -2m and 8.8m</dd>
204   * </dl>
205   */
206  ELEVATION(Double.class),
207
208  /**
209   * Depth in meters relative to altitude. For example 10 meters below a lake surface with given altitude.
210   * This parameter accepts comma separated range values, e.g.:
211   * <dl>
212   * <dt>*,10</dt>
213   * <dd>Depth below or equals 10m</dd>
214   * <dt>100,*</dt>
215   * <dd>Depth above or equals 100m</dd>
216   * <dt>12.1,28.8</dt>
217   * <dd>Depth between or equals 12.1m and 28.8m</dd>
218   * </dl>
219   */
220  DEPTH(Double.class),

223   * An identifier of any form assigned by the source to identify the institution

225   */
226  INSTITUTION_CODE(String.class),
227
228  /**
229   * An identifier of any form assigned by the source to identify the physical collection or digital dataset
230   * uniquely within the context of an institution.

232  COLLECTION_CODE(String.class),
233
234  /**
235   * An identifier of any form assigned by the source within a physical collection or digital dataset for the record
236   * which may not be unique, but should be fairly unique in combination with the institution and collection code.
237   */
238  CATALOG_NUMBER(String.class),

244

246   * The person who identified the occurrence.

264  TAXON_KEY(Integer.class),

269  ACCEPTED_TAXON_KEY(Integer.class),

272   * A kingdom key from the GBIF backbone.
273   */
274  KINGDOM_KEY(Integer.class),

277   * A phylum key from the GBIF backbone.

279  PHYLUM_KEY(Integer.class),

282   * A class key from the GBIF backbone.

284  CLASS_KEY(Integer.class),

287   * A order key from the GBIF backbone.
288   */
289  ORDER_KEY(Integer.class),
290
291  /**
292   * A family key from the GBIF backbone.
293   */
294  FAMILY_KEY(Integer.class),

296  /**
297   * A genus key from the GBIF backbone.
298   */
299  GENUS_KEY(Integer.class),

302   * A subgenus key from the GBIF backbone.

327   * The status of the use of the  GBIF Backbone taxonKey.
328   */

331  /**
332   * Searches for occurrence records which contain a value on its coordinate fields (latitude and longitude).
333   * HAS_COORDINATE=true searches for occurrence records with a coordinate value.
334   * HAS_COORDINATE=false searches for occurrence records without a coordinate value.
335   */
336  HAS_COORDINATE(Boolean.class),
337
338  /**
339   * Geometry in <a href="https://en.wikipedia.org/wiki/Well-known_text">Well Known Text</a> (WKT) format.

341   * Multi geometries like MULTIPOLYGON are not supported and multiple parameters should be used instead.
342   * Valid geometries are:
343   * <ul>
344   * <li>POINT</li>
345   * <li>LINESTRING</li>
346   * <li>POLYGON</li>
347   * <li>LINEARRING</li>
348   * </ul>
349   */
350  GEOMETRY(String.class),
351
352  /**
353   * Use in combination of LATITUDE and LONGITUDE parameters matches within a given distance.
354   * E.g.: geo_distance=100m,40,90 geo_distance=100km,40,90 geo_distance=100mi,40,90.
355   * See supported units in {@link org.gbif.api.model.occurrence.geo.DistanceUnit}.
356   */
357  GEO_DISTANCE(String.class),
358
359  /**
360   * The distance from a known centroid, e.g. a country centroid.
361   */
362  DISTANCE_FROM_CENTROID_IN_METERS(Double.class),
363
364  /**
365   * Includes/excludes occurrence records which contain geospatial issues for their coordinate.
366   * See {@link org.gbif.api.vocabulary.OccurrenceIssue#GEOSPATIAL_RULES}
367   * HAS_GEOSPATIAL_ISSUE=true include records with spatial issues.
368   * HAS_GEOSPATIAL_ISSUE=false exclude records with spatial issues.
369   * The absence of this parameter returns any record with or without spatial issues.
370   */
371  HAS_GEOSPATIAL_ISSUE(Boolean.class),
372
373  /**
374   * Searches occurrence for those that have a specific issue.
375   */
376  ISSUE(OccurrenceIssue.class),
377
378  /**
379   * Nomenclatural type (type status, typified scientific name, publication) applied to the subject.
380   */
381  TYPE_STATUS(TypeStatus.class),
382
383  /**
384   * The kind of media object.
385   * Recommended terms from the DCMI Type Vocabulary are StillImage, Sound or MovingImage for GBIF to index and show the
386   * media files.
387   */
388  MEDIA_TYPE(MediaType.class),
389
390  /**
391   *  An identifier for the Occurrence (as opposed to a particular digital record of the occurrence).
392   *  In the absence of a persistent global unique identifier, construct one from a combination of identifiers in the
393   *  record that will most closely make the occurrenceID globally unique.
394   */
395  OCCURRENCE_ID(String.class),
396
397  /**
398   * The process by which the biological individual(s) represented in the Occurrence became established at the location.
399   */
400  ESTABLISHMENT_MEANS(String.class),
401
402  /**
403   * Provides the controlled vocabulary for information about degree to which an Organism survives, reproduces, and expands its range at the given place and time.
404   */
405  DEGREE_OF_ESTABLISHMENT(String.class),
406
407  /**
408   * Provides the controlled vocabulary for information about the process by which an Organism came to be in a given place at a given time.

415   */
416  REPATRIATED(Boolean.class),
