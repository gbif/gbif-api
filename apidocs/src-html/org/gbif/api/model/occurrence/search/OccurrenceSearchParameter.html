019import org.gbif.api.vocabulary.BasisOfRecord;

021import org.gbif.api.vocabulary.Country;
022import org.gbif.api.vocabulary.EndpointType;
023import org.gbif.api.vocabulary.EstablishmentMeans;
024import org.gbif.api.vocabulary.License;
025import org.gbif.api.vocabulary.MediaType;

030import java.util.Date;

033/**
034 * Supported query parameters by the occurrence search and download service.
035 * For download predicates only the numerical types support comparisons other than equals.
036 */
037public enum OccurrenceSearchParameter implements SearchParameter {
038
039  /**
040   * The dataset key as a UUID.
041   */
042  DATASET_KEY(UUID.class),
043
044  /**
045   * The 4 digit year. A year of 98 will be 98 common era, not 1998.
046   * This parameter accepts comma separated range values, e.g.:
047   * <dl>
048   * <dt>*,1810</dt>
049   * <dd>Found before or equal to 1810</dd>
050   * <dt>1848,1933</dt>
051   * <dd>Year between 1848 and 1933</dd>
052   * </dl>
053   */
054  YEAR(Integer.class),
055
056  /**
057   * The month of the year, starting with 1 for January.
058   * A month query can be used to retrieve seasonal records when used without a year or a year range.
059   * This parameter accepts comma separated range values, e.g.:
060   * <dl>
061   * <dt>4,8</dt>
062   * <dd>Month between April and August</dd>
063   * </dl>
064   */
065  MONTH(Integer.class),
066
067  /**
068   * Event date (date the occurrence was recorded) in ISO 8601 formats:yyyy, yyyy-MM, yyyy-MM-dd and MM-dd.
069   * This parameter accepts comma separated range values, examples of valid ranges are:
070   * <dl>
071   * <dt>2001-02-11,2010-01-10</dt>
072   * <dd>Dates between 2001-02-11 and 2010-01-10</dd>
073   * <dt>2001-02,2010-01</dt>
074   * <dd>Dates between first day of 2001-02 and last day of 2010-01</dd>
075   * <dt>2001,2010</dt>
076   * <dd>Dates between first day of 2001 and last day of 2010</dd>
077   * <dt>2001,2010-01</dt>
078   * <dd>Dates between first day of 2001 and last day of 2010-01</dd>
079   * <dt>2001-01-10,2010</dt>
080   * <dd>Dates between 2001-01-10 and last day of 2010</dd>
081   * <dt>2001-01-10,*</dt>
082   * <dd>Dates after 2001-01-10</dd>
083   * <dt>*,2001-01-10</dt>
084   * <dd>Dates before 2001-01-10</dd>
085   * <dt>*</dt>
086   * <dd>all dates</dd>
087   * </dl>
088   */
089  EVENT_DATE(Date.class),
090
091  /**
092   * An identifier for the set of information associated with an Event (something that occurs at a place and time).
093   * May be a global unique identifier or an identifier specific to the data set.
094   */
095  EVENT_ID(String.class),
096
097  /**
098   * An identifier for the broader Event that groups this and potentially other Events.
099   */
100  PARENT_EVENT_ID(String.class),
101
102  /**
103   * The name of, reference to, or description of the method or protocol used during an Event.
104   */
105  SAMPLING_PROTOCOL(String.class),
106
107  /**
108   * Last interpreted date in ISO 8601 formats:yyyy, yyyy-MM, yyyy-MM-dd and MM-dd.
109   * This parameter accepts comma separated range values, examples of valid ranges are:
110   * <dl>
111   * <dt>2001-02-11,2010-01-10</dt>
112   * <dd>Dates between 2001-02-11 and 2010-01-10</dd>
113   * <dt>2001-02,2010-01</dt>
114   * <dd>Dates between first day of 2001-02 and last day of 2010-01</dd>
115   * <dt>2001,2010</dt>
116   * <dd>Dates between first day of 2001 and last day of 2010</dd>
117   * <dt>2001,2010-01</dt>
118   * <dd>Dates between first day of 2001 and last day of 2010-01</dd>
119   * <dt>2001-01-10,2010</dt>
120   * <dd>Dates between 2001-01-10 and last day of 2010</dd>
121   * <dt>2001-01-10,*</dt>
122   * <dd>Dates after 2001-01-10</dd>
123   * <dt>*,2001-01-10</dt>
124   * <dd>Dates before 2001-01-10</dd>
125   * <dt>*</dt>
126   * <dd>all dates</dd>
127   * </dl>
128   */
129  LAST_INTERPRETED(Date.class),
130
131  /**
132   * Latitude in decimals between -90 and 90 based on WGS 84.
133   */
134  DECIMAL_LATITUDE(Double.class),
135
136  /**
137   * Longitude in decimals between -180 and 180 based on WGS 84.
138   */
139  DECIMAL_LONGITUDE(Double.class),
140
141  /**
142   * The uncertainty of the coordinate in meters.
143   * This parameter accepts comma separated range values, e.g.:
144   * <dl>
145   * <dt>*,100</dt>
146   * <dd>Uncertainty below or equals 100m</dd>
147   * <dt>10000,*</dt>
148   * <dd>Uncertainty above or equals 10,000m</dd>
149   * <dt>1000,5000</dt>
150   * <dd>Uncertainty between or equals 1000m and 5000m
151   * </dd>
152   * </dl>
153   */
154  COORDINATE_UNCERTAINTY_IN_METERS(Double.class),
155
156  /**
157   * Country the occurrence was recorded in.
158   */
159  COUNTRY(Country.class),
160
161  /**
162   * Continent the occurrence was recorded in.
163   */
164  CONTINENT(Continent.class),
165
166  /**
167   * The country of the organization that publishes the dataset the occurrence belongs to.
168   */
169  PUBLISHING_COUNTRY(Country.class),
170
171  /**
172   * Altitude/elevation in meters above sea level.
173   * This parameter accepts comma separated range values, e.g.:
174   * <dl>
175   * <dt>*,100</dt>
176   * <dd>Altitude below or equals 100m</dd>
177   * <dt>100,*</dt>
178   * <dd>Altitude above or equals 100m</dd>
179   * <dt>-2,8.8</dt>
180   * <dd>Altitude between or equals -2m and 8.8m</dd>
181   * </dl>
182   */
183  ELEVATION(Double.class),
184
185  /**
186   * Depth in meters relative to altitude. For example 10 meters below a lake surface with given altitude.
187   * This parameter accepts comma separated range values, e.g.:
188   * <dl>
189   * <dt>*,10</dt>
190   * <dd>Depth below or equals 10m</dd>
191   * <dt>100,*</dt>
192   * <dd>Depth above or equals 100m</dd>
193   * <dt>12.1,28.8</dt>
194   * <dd>Depth between or equals 12.1m and 28.8m</dd>
195   * </dl>
196   */
197  DEPTH(Double.class),
198
199  /**
200   * An identifier of any form assigned by the source to identify the institution
201   * the record belongs to. Not guaranteed to be unique.
202   */
203  INSTITUTION_CODE(String.class),
204
205  /**
206   * An identifier of any form assigned by the source to identify the physical collection or digital dataset
207   * uniquely within the context of an institution.
208   */
209  COLLECTION_CODE(String.class),
210
211  /**
212   * An identifier of any form assigned by the source within a physical collection or digital dataset for the record
213   * which may not be unique, but should be fairly unique in combination with the institution and collection code.
214   */
215  CATALOG_NUMBER(String.class),
216
217  /**
218   * The person who recorded the occurrence.
219   */
220  RECORDED_BY(String.class),
221
222  /**
223   * An identifier given to the Occurrence at the time it was recorded.
224   */
225  RECORD_NUMBER(String.class),
226
227  /**
228   * A basis of record enumeration value.
229   */
230  BASIS_OF_RECORD(BasisOfRecord.class),
231
232  /**
233   * A taxon key from the GBIF backbone. All included and synonym taxa are included in the search, so a search for
234   * aves with taxonKey=212 will match all birds, no matter which species.
235   */
236  TAXON_KEY(Integer.class),
237
238  /**
239   * A taxon key from the GBIF backbone for the name usage of the currently valid or accepted taxon.
240   */
241  ACCEPTED_TAXON_KEY(Integer.class),
242
243  /**
244   * A kingdom key from the GBIF backbone.
245   */
246  KINGDOM_KEY(Integer.class),
247
248  /**
249   * A phylum key from the GBIF backbone.
250   */
251  PHYLUM_KEY(Integer.class),
252
253  /**
254   * A class key from the GBIF backbone.
255   */
256  CLASS_KEY(Integer.class),
257
258  /**
259   * A order key from the GBIF backbone.
260   */
261  ORDER_KEY(Integer.class),
262
263  /**
264   * A family key from the GBIF backbone.
265   */
266  FAMILY_KEY(Integer.class),
267
268  /**
269   * A genus key from the GBIF backbone.
270   */
271  GENUS_KEY(Integer.class),
272
273  /**
274   * A subgenus key from the GBIF backbone.
275   */
276  SUBGENUS_KEY(Integer.class),
277
278  /**
279   * A species key from the GBIF backbone.
280   */
281  SPECIES_KEY(Integer.class),
282
283  /**
284   * Searches the interpreted, full scientific name of the occurrence.
285   */
286  SCIENTIFIC_NAME(String.class),
287
288  /**
289   * Scientific name as provided byt the source.
290   */
291  VERBATIM_SCIENTIFIC_NAME(String.class),
292
293  /**
294   * Verbatim identifier for the set of taxon information. May be a global unique identifier or an identifier specific to the data set.
295   */
296  TAXON_ID(String.class),
297
298  /**
299   * The status of the use of the  GBIF Backbone taxonKey.
300   */
301  TAXONOMIC_STATUS(TaxonomicStatus.class),
302
303  /**
304   * Searches for occurrence records which contain a value on its coordinate fields (latitude and longitude).
305   * HAS_COORDINATE=true searches for occurrence records with a coordinate value.
306   * HAS_COORDINATE=false searches for occurrence records without a coordinate value.
307   */
308  HAS_COORDINATE(Boolean.class),
309
310  /**
311   * Geometry in <a href="https://en.wikipedia.org/wiki/Well-known_text">Well Known Text</a> (WKT) format.
312   * E.g.: POLYGON ((30.0 10.0, 10.12 20.23, 20 40, 40 40, 30 10)).
313   * Multi geometries like MULTIPOLYGON are not supported and multiple parameters should be used instead.
314   * Valid geometries are:
315   * <ul>
316   * <li>POINT</li>
317   * <li>LINESTRING</li>
318   * <li>POLYGON</li>
319   * <li>LINEARRING</li>
320   * </ul>
321   */
322  GEOMETRY(String.class),
323
324  /**
325   * Includes/excludes occurrence records which contain geospatial issues for their coordinate.
326   * See {@link org.gbif.api.vocabulary.OccurrenceIssue#GEOSPATIAL_RULES}
327   * HAS_GEOSPATIAL_ISSUE=true include records with spatial issues.
328   * HAS_GEOSPATIAL_ISSUE=false exclude records with spatial issues.
329   * The absence of this parameter returns any record with or without spatial issues.
330   */
331  HAS_GEOSPATIAL_ISSUE(Boolean.class),
332
333  /**
334   * Searches occurrence for those that have a specific issue.
335   */
336  ISSUE(OccurrenceIssue.class),
337
338  /**
339   * Nomenclatural type (type status, typified scientific name, publication) applied to the subject.
340   */
341  TYPE_STATUS(TypeStatus.class),
342
343  /**
344   * The kind of media object.
345   * Recommended terms from the DCMI Type Vocabulary are StillImage, Sound or MovingImage for GBIF to index and show the
346   * media files.
347   */
348  MEDIA_TYPE(MediaType.class),
349
350  /**
351   *  An identifier for the Occurrence (as opposed to a particular digital record of the occurrence).
352   *  In the absence of a persistent global unique identifier, construct one from a combination of identifiers in the
353   *  record that will most closely make the occurrenceID globally unique.
354   */
355  OCCURRENCE_ID(String.class),
356
357  /**
358   * The process by which the biological individual(s) represented in the Occurrence became established at the location.
359   */
360  ESTABLISHMENT_MEANS(EstablishmentMeans.class),
361
362  /**
363   * Searches for records whose publishing country is different to the country where the record was recorded in.
364   */
365  REPATRIATED(Boolean.class),
366
367  /**
368   * An identifier for the Organism instance (as opposed to a particular digital record of the Organism).
369   * May be a globally unique identifier or an identifier specific to the data set.
370   */
371  ORGANISM_ID(String.class),
372
373  /**
374   * The name of the next smaller administrative region than country in which the Location occurs.
375   */
376  STATE_PROVINCE(String.class),
377
378  /**
379   * The name of the water body in which the Location occurs.
380   */
381  WATER_BODY(String.class),
382
383  /**
384   * The specific description of the place.
385   * It may contain information modified from the original to correct perceived errors or standardize the description.
386   */
387  LOCALITY(String.class),
388
389  /**
390   * Protocol used to provide the occurrence record.
391   */
392  PROTOCOL(EndpointType.class),
393
394  /**
395   * The license applied to the dataset.
396   */
397  LICENSE(License.class),
398
399  /**
400   * The owning organizations uuid key.
401   */
402  PUBLISHING_ORG(UUID.class),
403
404  /**
405   * The GBIF network that the publishing organisation belongs to.
406   */
407  NETWORK_KEY(UUID.class),
408
409  /**
410   * The technical installation key that hosts/publishes this record.
411   */
412  INSTALLATION_KEY(UUID.class),
413
414  /**
415   * Crawl attempt that harvested this record.
416   */
417  CRAWL_ID(Integer.class),
418
419  /**
420   * GBIF ProjectId.
421   */
422  PROJECT_ID(String.class),
423
424  /**
425   * GBIF Programme Acronym.
426   */
427  PROGRAMME(String.class),
428
429  /**
430   * A number or enumeration value for the quantity of organisms.
431   */
432  ORGANISM_QUANTITY(Double.class),
433
434  /**
435   * The type of quantification system used for the quantity of organisms.
436   */
437  ORGANISM_QUANTITY_TYPE(String.class),
438
439  /**
440   * The unit of measurement of the size (time duration, length, area, or volume) of a sample in a sampling event.
441   */
442  SAMPLE_SIZE_UNIT(String.class),
443
444  /**
445   * A numeric value for a measurement of the size (time duration, length, area, or volume) of a sample in a sampling event.
446   */
447  SAMPLE_SIZE_VALUE(Double.class),
448
449  /**
450   * Calculated organismQuantity relative to the sampleSizeValue i.e. -> organismQuantity / sampleSizeValue.
451   */
452  RELATIVE_ORGANISM_QUANTITY(Double.class),
453
454  /**
455   * Collection key as UUID. It links to the collection to which this record belongs.
456   */
457  COLLECTION_KEY(UUID.class),
458
459  /**
460   * Institution key as UUID. It links to the institution that maintains, recorded or digitized  this record.
461   */
462  INSTITUTION_KEY(UUID.class),
463
464  /**
465   * Agent identifiers from GbifTerm.recordedByID
466   */
467  RECORDED_BY_ID(String.class),
468
469  /**
470   * Agent identifiers from GbifTerm.identifiedByID
471   */
472  IDENTIFIED_BY_ID(String.class);
473
474  private final Class<?> type;
475
476  OccurrenceSearchParameter(Class<?> type) {
477    this.type = type;
478  }
479
480  /**
481   * @return the data type expected for the value of the respective search parameter
482   */
483  public Class<?> type() {
484    return type;
485  }
