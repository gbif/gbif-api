001package org.gbif.api.model.occurrence.sql.serde;
002
003import org.gbif.api.model.occurrence.sql.Query;
004import org.gbif.api.model.occurrence.sql.Query.Issue;
005
006import java.io.IOException;
007
008import org.codehaus.jackson.JsonGenerator;
009import org.codehaus.jackson.map.JsonSerializer;
010import org.codehaus.jackson.map.SerializerProvider;
011
012/**
013 * Json serializer, to use issue name with description and comment upon serialization.
014 */
015public class QueryIssueSerde extends JsonSerializer<Query.Issue> {
016
017  @Override
018  public void serialize(Issue value, JsonGenerator generator, SerializerProvider provider) throws IOException {
019    generator.writeString(String.format("%s: %s %s", value.name(), value.description(), value.comment()));
020  }
021}
