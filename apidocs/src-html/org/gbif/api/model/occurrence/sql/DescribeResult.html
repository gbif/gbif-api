001package org.gbif.api.model.occurrence.sql;
002
003/**
004 * Describe HDFS table response.
005 */
006public class DescribeResult {
007
008  private String columnName;
009  private String dataType;
010  private String comment;
011
012  /**
013   * Empty constructor, required for serialization.
014   */
015  public DescribeResult() {}
016
017  /**
018   * Full constructor.
019   */
020  public DescribeResult(String columnName, String dataType, String comment) {
021    this.columnName = columnName;
022    this.dataType = dataType;
023    this.comment = comment;
024  }
025
026  public String getColumnName() {
027    return columnName;
028  }
029
030  public void setColumnName(String columnName) {
031    this.columnName = columnName;
032  }
033
034  public String getDataType() {
035    return dataType;
036  }
037
038  public void setDataType(String dataType) {
039    this.dataType = dataType;
040  }
041
042  public String getComment() {
043    return comment;
044  }
045
046  public void setComment(String comment) {
047    this.comment = comment;
048  }
049}
