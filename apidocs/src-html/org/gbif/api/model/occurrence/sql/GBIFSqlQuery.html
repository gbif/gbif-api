001package org.gbif.api.model.occurrence.sql;
002
003import javax.validation.ValidationException;
004
005/**
006 * Splits GBIF SQL Query to sql part and export format.
007 * Example GBIF Query:
008 * SELECT gbifid, countrycode, datasetkey, license, month, year FROM occurrence WHERE month=3 AND year = 2018 EXPORT AS AVRO;
009 */
010public class GBIFSqlQuery {
011
012  private static final String EXPORT_AS = "(?i)EXPORT\\s+AS";
013
014  private final String uncheckedHiveQuery;
015  private final SqlDownloadExportFormat sqlDownloadExportFormat;
016
017  /**
018   * Creates GBIF SQL Query with appropriate hive SQL and export format.
019   * @param sqlQuery gbif defined sql query.
020   * @return GBIFSqlQuery object.
021   */
022  public static GBIFSqlQuery create(String sqlQuery) {
023    SqlDownloadExportFormat exportFormat = SqlDownloadExportFormat.TSV;
024
025    String[] sqlSplit = sqlQuery.split(EXPORT_AS);
026
027    if (sqlSplit.length == 2) {
028      try {
029        exportFormat = SqlDownloadExportFormat.valueOf(sqlSplit[1].trim().toUpperCase());
030      } catch (IllegalArgumentException iex) {
031        throw new ValidationException("Failure reading export format please specify TSV or AVRO");
032      }
033    }
034
035    return new GBIFSqlQuery(sqlSplit[0].trim(), exportFormat);
036  }
037
038  private GBIFSqlQuery(String uncheckedHiveQuery, SqlDownloadExportFormat sqlDownloadExportFormat) {
039    this.uncheckedHiveQuery = uncheckedHiveQuery;
040    this.sqlDownloadExportFormat = sqlDownloadExportFormat;
041  }
042
043  public String getUncheckedHiveQuery() {
044    return uncheckedHiveQuery;
045  }
046
047  public SqlDownloadExportFormat getSqlDownloadExportFormat() {
048    return sqlDownloadExportFormat;
049  }
050}
