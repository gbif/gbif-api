001package org.gbif.api.model.occurrence.sql;
002
003import org.gbif.api.model.occurrence.sql.serde.QueryIssueSerde;
004
005import org.codehaus.jackson.map.annotate.JsonSerialize;
006
007/**
008 * SQL Download API Query Issues listed with there descriptions.
009 */
010public class Query {
011
012  private static final String FORMAT =
013    "Format of LEGAL query is : SELECT ... \nFROM occurrence \nWHERE ... \nGROUP BY ...";
014
015  @JsonSerialize(using = QueryIssueSerde.class)
016  public enum Issue {
017
018    NO_ISSUE("No Issue."),
019    EMPTY_SQL("SQL query cannot be empty."),
020    ONLY_ONE_SELECT_ALLOWED("SQL query should be a SELECT query with SELECT keyword used only once."),
021    DDL_JOINS_UNION_NOT_ALLOWED("SQL query cannot use ORDER BY, INSERT, UPDATE, DELETE, MERGE, PROCEDURE_CALL, EXCEPT, INTERSECT, UNION, JOIN or AS."),
022    DATASET_AND_LICENSE_REQUIRED("SQL query should select on 'datasetkey' and 'license' fields as they are required for citations."),
023    CANNOT_EXECUTE("Query cannot be executed because of"),
024    TABLE_NAME_NOT_OCCURRENCE("Query should have table name as 'occurrence'."),
025    PARSE_FAILED(String.format("Cannot parse the query, the sql query is invalid, Make sure provided query should follow the format %n %s.", FORMAT)),
026    CANNOT_USE_ALLFIELDS("Usage of '*' for selecting all fields not allowed, Please explicitly add the desired fields, refer the fields from occurrence/download/request/sql/describe endpoint."),
027    HAVING_CLAUSE_NOT_SUPPORTED("SQL Download API donot support query with HAVING clause");
028
029    Issue(String description) {
030      this.description = description;
031    }
032
033    private final String description;
034    private String comment = "";
035
036    public String description() {
037      return description;
038    }
039
040    public String comment() {
041      return comment;
042    }
043
044    public Issue withComment(String comment) {
045      this.comment = comment;
046      return this;
047    }
048  }
049}
