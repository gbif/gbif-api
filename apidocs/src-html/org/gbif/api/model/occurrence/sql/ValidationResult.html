001package org.gbif.api.model.occurrence.sql;
002
003import org.gbif.api.model.occurrence.sql.Query.Issue;
004
005import java.util.List;
006
007/**
008 * Result of a SQL Query Validation.
009 */
010public class   ValidationResult {
011
012  private String sql;
013  private List<Query.Issue> issues;
014  private boolean success;
015  private List<String> explain;
016
017  public ValidationResult() {}
018
019  /**
020   * Full constructor.
021   */
022  public ValidationResult(String sql, List<Issue> issues, List<String> queryExplanation, boolean success) {
023    this.sql = sql;
024    this.issues = issues;
025    this.success = success;
026    this.explain = queryExplanation;
027  }
028
029  public boolean isSuccess() {
030    return success;
031  }
032
033  public String getSql() {
034    return sql;
035  }
036
037  public void setSql(String sql) {
038    this.sql = sql;
039  }
040
041  public List<Query.Issue> getIssues() {
042    return issues;
043  }
044
045  public void setIssues(List<Query.Issue> issues) {
046    this.issues = issues;
047  }
048
049  public List<String> getExplain() {
050    return explain;
051  }
052
053  public void setExplain(List<String> explain) {
054    this.explain = explain;
055  }
056
057  public void setSuccess(boolean success) {
058    this.success = success;
059  }
060
061  public static class Builder {
062
063    private String sql;
064    private List<Query.Issue> issues;
065    private boolean success;
066    private List<String> explain;
067
068    public Builder sql(String sql) {
069      this.sql = sql;
070      return this;
071    }
072
073    public Builder issues(List<Query.Issue> issues) {
074      this.issues = issues;
075      return this;
076    }
077
078    public Builder success(boolean success) {
079      this.success = success;
080      return this;
081    }
082
083    public Builder explain(List<String> explain) {
084      this.explain = explain;
085      return this;
086    }
087
088    public ValidationResult build() {
089      ValidationResult validationResult = new ValidationResult();
090      validationResult.sql = sql;
091      validationResult.issues = issues;
092      validationResult.success = success;
093      validationResult.explain = explain;
094      return validationResult;
095    }
096  }
097}
