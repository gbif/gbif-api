016package org.gbif.api.model.occurrence;
017
018import java.util.Objects;
019
020/**
021 * GADM level.
022 */
023public class Gadm {
024
025  private GadmFeature level0;
026  private GadmFeature level1;
027  private GadmFeature level2;
028  private GadmFeature level3;
029
030  public GadmFeature getLevel0() {
031    return level0;
032  }
033
034  public void setLevel0(GadmFeature level0) {
035    this.level0 = level0;
036  }
037
038  public GadmFeature getLevel1() {
039    return level1;
040  }
041
042  public void setLevel1(GadmFeature level1) {
043    this.level1 = level1;
044  }
045
046  public GadmFeature getLevel2() {
047    return level2;
048  }
049
050  public void setLevel2(GadmFeature level2) {
051    this.level2 = level2;
052  }
053
054  public GadmFeature getLevel3() {
055    return level3;
056  }
057
058  public void setLevel3(GadmFeature level3) {
059    this.level3 = level3;
060  }
061
062  @Override
063  public boolean equals(Object o) {
064    if (this == o) return true;
065    if (o == null || getClass() != o.getClass()) return false;
066    Gadm gadm = (Gadm) o;
067    return Objects.equals(level0, gadm.level0) &&
068      Objects.equals(level1, gadm.level1) &&
069      Objects.equals(level2, gadm.level2) &&
070      Objects.equals(level3, gadm.level3);
071  }
072
073  @Override
074  public int hashCode() {
075    return Objects.hash(level0, level1, level2, level3);
076  }
077
078  @Override
079  public String toString() {
080    return "Gadm{" +
081      "level0=" + level0 +
082      ", level1=" + level1 +
083      ", level2=" + level2 +
084      ", level3=" + level3 +
085      '}';
086  }
087}
