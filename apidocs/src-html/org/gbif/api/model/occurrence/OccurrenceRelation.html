016package org.gbif.api.model.occurrence;
017
018import java.util.Objects;
019import java.util.StringJoiner;
020
021/**
022 * Occurrence relations based on the dwc:ResourceRelationship model.
023 * See <a href="http://darwincore.googlecode.com/svn/trunk/terms/index.htm#relindex">Darwin Core</a>.
024 */
025public class OccurrenceRelation {
026  private String id;  // resourceRelationshipID
027  private long occurrenceId;  // "from" resourceID
028  private long relatedOccurrenceId;  // "to" relatedResourceID
029  private String type;  // relationshipOfResource
030  private String accordingTo; // relationshipAccordingTo
031  private String establishedDate;  // relationshipEstablishedDate
032  private String remarks;  // relationshipRemarks
033
034  public String getId() {
035    return id;
036  }
037
038  public void setId(String id) {
039    this.id = id;
040  }
041
042  public long getOccurrenceId() {
043    return occurrenceId;
044  }
045
046  public void setOccurrenceId(long occurrenceId) {
047    this.occurrenceId = occurrenceId;
048  }
049
050  public long getRelatedOccurrenceId() {
051    return relatedOccurrenceId;
052  }
053
054  public void setRelatedOccurrenceId(long relatedOccurrenceId) {
055    this.relatedOccurrenceId = relatedOccurrenceId;
056  }
057
058  public String getType() {
059    return type;
060  }
061
062  public void setType(String type) {
063    this.type = type;
064  }
065
066  public String getAccordingTo() {
067    return accordingTo;
068  }
069
070  public void setAccordingTo(String accordingTo) {
071    this.accordingTo = accordingTo;
072  }
073
074  public String getEstablishedDate() {
075    return establishedDate;
076  }
077
078  public void setEstablishedDate(String establishedDate) {
079    this.establishedDate = establishedDate;
080  }
081
082  public String getRemarks() {
083    return remarks;
084  }
085
086  public void setRemarks(String remarks) {
087    this.remarks = remarks;
088  }
089
090  @Override
091  public boolean equals(Object o) {
092    if (this == o) {
093      return true;
094    }
095    if (o == null || getClass() != o.getClass()) {
096      return false;
097    }
098    OccurrenceRelation that = (OccurrenceRelation) o;
099    return occurrenceId == that.occurrenceId &&
100      relatedOccurrenceId == that.relatedOccurrenceId &&
101      Objects.equals(id, that.id) &&
102      Objects.equals(type, that.type) &&
103      Objects.equals(accordingTo, that.accordingTo) &&
104      Objects.equals(establishedDate, that.establishedDate) &&
105      Objects.equals(remarks, that.remarks);
106  }
107
108  @Override
109  public int hashCode() {
110    return Objects
111      .hash(id, occurrenceId, relatedOccurrenceId, type, accordingTo, establishedDate, remarks);
112  }
113
114  @Override
115  public String toString() {
116    return new StringJoiner(", ", OccurrenceRelation.class.getSimpleName() + "[", "]")
117      .add("id='" + id + "'")
118      .add("occurrenceId=" + occurrenceId)
119      .add("relatedOccurrenceId=" + relatedOccurrenceId)
120      .add("type='" + type + "'")
121      .add("accordingTo='" + accordingTo + "'")
122      .add("establishedDate='" + establishedDate + "'")
123      .add("remarks='" + remarks + "'")
124      .toString();
125  }
126}
