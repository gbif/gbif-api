016package org.gbif.api.model.checklistbank;
017
018import java.util.Objects;
019import java.util.StringJoiner;
020
021import javax.annotation.Nullable;
022import javax.validation.constraints.NotNull;
023
024/**
025 * Reference Model Object represents a literature reference stating a bibliography for a taxon.
026 *
027 * Since the initial release of the GBIF API version 1.0 ChecklistBank has been modified to only store the entire citation string of a reference and
028 * none of the atomised fields which are now deprecated.
029 *
030 * @see <a href="http://rs.gbif.org/extension/gbif/1.0/references.xml">Reference Definition</a>
031 */
032@SuppressWarnings("unused")
033public class Reference implements NameUsageExtension {
034
035  private Integer taxonKey;
036  private String citation;
037  private String link;
038  private String doi;
039  private String type;
040  private String remarks;
041  private String source;
042  private Integer sourceTaxonKey;
043
044  @Deprecated
045  private String title;
046  @Deprecated
047  private String author;
048  @Deprecated
049  private String date;
050
051  /**
052   * The name usage "taxon" key this description belongs to.
053   */
054  @Override
055  public Integer getTaxonKey() {
056    return taxonKey;
057  }
058
059  @Override
060  public void setTaxonKey(Integer taxonKey) {
061    this.taxonKey = taxonKey;
062  }
063
064  /**
065   * The author or authors of the referenced work.
066   * <blockquote>
067   * <p>
068   * <i>Example:</i> Patricia Hartge.
069   * </p>
070   * </blockquote>
071   *
072   * @return the author
073   */
074  @Nullable
075  @Deprecated
076  public String getAuthor() {
077    return author;
078  }
079
080  /**
081   * @param author the author to set
082   */
083  @Deprecated
084  public void setAuthor(String author) {
085    this.author = author;
086  }
087
088  /**
089   * A text string referring to an un-parsed bibliographic citation.
090   * The full citation given here should include all the details potentially found in the other atomised fields,
091   * i.e. includes authorship, title, etc.
092   * <p/>
093   * <blockquote>
094   * <p>
095   * <i>Example:</i> Hartge, P., Genetics of reproductive lifespan. Nature Genetics 41, 637 - 638 (2009).
096   * </p>
097   * </blockquote>
098   *
099   * @return the citation
100   */
101  @NotNull
102  public String getCitation() {
103    return citation;
104  }
105
106  /**
107   * @param citation the citation to set
108   */
109  public void setCitation(String citation) {
110    this.citation = citation;
111  }
112
113  /**
114   * Date of publication, recommended ISO format YYYY or YYYY-MM-DD.
115   *
116   * @return the publication date
117   */
118  @Nullable
119  @Deprecated
120  public String getDate() {
121    return date;
122  }
123
124  @Deprecated
125  public void setDate(String date) {
126    this.date = date;
127  }
128
129  /**
130   * The pure DOI for the publication without potential http resolver or URI prefix.
131   * For example {@code 10.1038/ng0609-637} instead of {@code doi:10.1038/ng0609-637} or {@code
132   * http://dx.doi.org/10.1038/ng0609-637}. A valid DOI always starts with {@code 10.}.
133   *
134   * @return the doi
135   *
136   * @see <a href="http://www.crossref.org/01company/15doi_info.html">Crossref DOI Info</a>
137   * @see <a href="http://www.doi.org/hb.html">DOI Handbook</a>
138   * @see <a href="http://de.wikipedia.org/wiki/Digital_Object_Identifier">Wikipedia</a>
139   */
140  @Nullable
141  public String getDoi() {
142    return doi;
143  }
144
145  public void setDoi(String doi) {
146    this.doi = doi;
147  }
148
149  /**
150   * Link.
151   *
152   * @return the link
153   */
154  @Nullable
155  public String getLink() {
156    return link;
157  }
158
159  /**
160   * @param link the link to set
161   */
162  public void setLink(String link) {
163    this.link = link;
164  }
165
166  /**
167   * Annotation of taxon-specific information related to the referenced publication.
168   * <p/>
169   * <blockquote>
170   * <p>
171   * <i>Examples:</i> "transferred H. nigritarsus to Acanolonia"; "Type specimen is a skeleton"
172   * </p>
173   * </blockquote>
174   *
175   * @return the taxonomic remarks
176   */
177  @Nullable
178  public String getRemarks() {
179    return remarks;
180  }
181
182  public void setRemarks(String remarks) {
183    this.remarks = remarks;
184  }
185
186  /**
187   * Title of book or article.
188   * <blockquote>
189   * <p>
190   * <i>Example:</i> "Genetics of reproductive lifespan", "Field Guide to Moths of Eastern North America".
191   * </p>
192   * </blockquote>
193   *
194   * @return the title
195   */
196  @Nullable
197  @Deprecated
198  public String getTitle() {
199    return title;
200  }
201
202  /**
203   * @param title the title to set
204   */
205  @Deprecated
206  public void setTitle(String title) {
207    this.title = title;
208  }
209
210  /**
211   * Used to assign a bibliographic reference to list of taxonomic or nomenclatural categories. Best practice is to use
212   * a controlled vocabulary.
213   * <blockquote>
214   * <p>
215   * <i>Examples:</i> Original publication of new combination (comb nov.)
216   * </p>
217   * </blockquote>
218   *
219   * @return the publication type
220   */
221  @Nullable
222  public String getType() {
223    return type;
224  }
225
226  public void setType(String type) {
227    this.type = type;
228  }
229
230  @Nullable
231  @Override
232  public String getSource() {
233    return source;
234  }
235
236  @Override
237  public void setSource(String source) {
238    this.source = source;
239  }
240
241  @Nullable
242  @Override
243  public Integer getSourceTaxonKey() {
244    return sourceTaxonKey;
245  }
246
247  @Override
248  public void setSourceTaxonKey(Integer sourceTaxonKey) {
249    this.sourceTaxonKey = sourceTaxonKey;
250  }
251
252  @Override
253  public boolean equals(Object o) {
254    if (this == o) {
255      return true;
256    }
257    if (o == null || getClass() != o.getClass()) {
258      return false;
259    }
260    Reference reference = (Reference) o;
261    return Objects.equals(citation, reference.citation) &&
262      Objects.equals(link, reference.link) &&
263      Objects.equals(doi, reference.doi) &&
264      Objects.equals(type, reference.type) &&
265      Objects.equals(remarks, reference.remarks) &&
266      Objects.equals(source, reference.source) &&
267      Objects.equals(sourceTaxonKey, reference.sourceTaxonKey) &&
268      Objects.equals(title, reference.title) &&
269      Objects.equals(author, reference.author) &&
270      Objects.equals(date, reference.date);
271  }
272
273  @Override
274  public int hashCode() {
275    return Objects
276      .hash(citation, link, doi, type, remarks, source, sourceTaxonKey, title, author,
277        date);
278  }
279
280  @Override
281  public String toString() {
282    return new StringJoiner(", ", Reference.class.getSimpleName() + "[", "]")
283      .add("taxonKey=" + taxonKey)
284      .add("citation='" + citation + "'")
285      .add("link='" + link + "'")
286      .add("doi='" + doi + "'")
287      .add("type='" + type + "'")
288      .add("remarks='" + remarks + "'")
289      .add("source='" + source + "'")
290      .add("sourceTaxonKey=" + sourceTaxonKey)
291      .add("title='" + title + "'")
292      .add("author='" + author + "'")
293      .add("date='" + date + "'")
294      .toString();
295  }
296}
