016package org.gbif.api.model.checklistbank;
017
018import javax.annotation.Nullable;
019
020import com.google.common.base.Objects;
021
022
023/**
024 * SpeciesProfile Model Object represents a species profile which describes basic species characteristics.
025 *
026 * @see <a href="http://rs.gbif.org/extension/gbif/1.0/speciesprofile.xml">Species Profile Definition</a>
027 */
028public class SpeciesProfile implements NameUsageExtension {
029
030  private String livingPeriod;
031  private String lifeForm;
032  private String habitat;
033  private Boolean marine;
034  private Boolean freshwater;
035  private Boolean terrestrial;
036  private Boolean extinct;
037  private Boolean hybrid;
038  private Integer ageInDays;
039  private Integer sizeInMillimeter;
040  private Integer massInGram;
041  private String source;
042  private Integer sourceTaxonKey;
043
044  /**
045   * Maximum observed age of an organism given as number of days.
046   * <blockquote>
047   * <p>
048   * <i>Example:</i> 5.
049   * </p>
050   * </blockquote>
051   *
052   * @return the ageInDays
053   */
054  @Nullable
055  public Integer getAgeInDays() {
056    return ageInDays;
057  }
058
059  /**
060   * @param ageInDays the ageInDays to set
061   */
062  public void setAgeInDays(Integer ageInDays) {
063    this.ageInDays = ageInDays;
064  }
065
066  /**
067   * A Boolean flag indicating whether the taxon is a freshwater organism.
068   *
069   * @return freshwater flag
070   */
071  @Nullable
072  public Boolean isFreshwater() {
073    return freshwater;
074  }
075
076  public void setFreshwater(Boolean freshwater) {
077    this.freshwater = freshwater;
078  }
079
080  /**
081   * Comma seperated list of mayor habitat classification as defined by IUCN in which a species is known to exist:
082   * <a href="http://www.iucnredlist.org/static/major_habitats">http://www.iucnredlist.org/static/major_habitats</a>
083   * <blockquote>
084   * <p>
085   * <i>Example:</i> 1.1.
086   * </p>
087   * </blockquote>
088   *
089   * @return the habitat
090   */
091  @Nullable
092  public String getHabitat() {
093    return habitat;
094  }
095
096  /**
097   * @param habitat the habitat to set.
098   */
099  public void setHabitat(String habitat) {
100    this.habitat = habitat;
101  }
102
103  /**
104   * A term describing the growth/lifeform of an organism. Should be based on a vocabulary like Raunkiær for plants:
105   * <a href="http://en.wikipedia.org/wiki/Raunkiær_plant_life-form">http://en.wikipedia.org/wiki/Raunkiær_plant_life-
106   * form</a>. Recommended vocabulary:
107   * <a href="http://rs.gbif.org/vocabulary/gbif/life_form.xml">http://rs.gbif.org/vocabulary/gbif/life_form.xml</a>
108   * <blockquote>
109   * <p>
110   * <i>Example:</i> Phanerophyte.
111   * </p>
112   * </blockquote>
113   *
114   * @return the lifeForm
115   */
116  @Nullable
117  public String getLifeForm() {
118    return lifeForm;
119  }
120
121  /**
122   * @param lifeForm the lifeForm to set
123   */
124  public void setLifeForm(String lifeForm) {
125    this.lifeForm = lifeForm;
126  }
127
128  /**
129   * The (geological) time a currently extinct organism is known to have lived. For geological times of fossils ideally
130   * based on a vocabulary like <a
131   * href="http://en.wikipedia.org/wiki/Geologic_column">http://en.wikipedia.org/wiki/Geologic_column</a>
132   * <blockquote>
133   * <p>
134   * <i>Example:</i> Cretaceous.
135   * </p>
136   * </blockquote>
137   *
138   * @return the livingPeriod
139   */
140  @Nullable
141  public String getLivingPeriod() {
142    return livingPeriod;
143  }
144
145  /**
146   * @param livingPeriod the livingPeriod to set
147   */
148  public void setLivingPeriod(String livingPeriod) {
149    this.livingPeriod = livingPeriod;
150  }
151
152  /**
153   * Maximum observed weight of an organism in grams.
154   * <blockquote>
155   * <p>
156   * <i>Example:</i> 12.
157   * </p>
158   * </blockquote>
159   *
160   * @return the massInGram
161   */
162  @Nullable
163  public Integer getMassInGram() {
164    return massInGram;
165  }
166
167  /**
168   * @param massInGram the massInGram to set
169   */
170  public void setMassInGram(Integer massInGram) {
171    this.massInGram = massInGram;
172  }
173
174  /**
175   * Maximum observed size of an organism in millimeter. Can be either height, length or width, whichever is greater.
176   * <blockquote>
177   * <p>
178   * <i>Example:</i> 1500.
179   * </p>
180   * </blockquote>
181   *
182   * @return the sizeInMillimeter
183   */
184  @Nullable
185  public Integer getSizeInMillimeter() {
186    return sizeInMillimeter;
187  }
188
189  /**
190   * @param sizeInMillimeter the sizeInMillimeter to se
191   */
192  public void setSizeInMillimeter(Integer sizeInMillimeter) {
193    this.sizeInMillimeter = sizeInMillimeter;
194  }
195
196  /**
197   * Flag indicating an extinct organism. Details about the timeperiod the organism has lived in can be supplied below.
198   * <blockquote>
199   * <p>
200   * <i>Example:</i> true - false.
201   * </p>
202   * </blockquote>
203   *
204   * @return the extinct
205   */
206  @Nullable
207  public Boolean isExtinct() {
208    return extinct;
209  }
210
211  /**
212   * @param extinct the extinct to set
213   */
214  public void setExtinct(Boolean extinct) {
215    this.extinct = extinct;
216  }
217
218  /**
219   * Flag indicating a hybrid organism. This does not have to be reflected in the name, but can be based on other
220   * studies like chromosome numbers etc.
221   * <blockquote>
222   * <p>
223   * <i>Example:</i> true - false.
224   * </p>
225   * </blockquote>
226   *
227   * @return the hybrid
228   */
229  @Nullable
230  public Boolean isHybrid() {
231    return hybrid;
232  }
233
234  /**
235   * @param hybrid the hybrid to set
236   */
237  public void setHybrid(Boolean hybrid) {
238    this.hybrid = hybrid;
239  }
240
241  /**
242   * A Boolean flag indicating whether the taxon is a marine organism, i.e. can be found in/above sea water
243   * <blockquote>
244   * <p>
245   * <i>Example:</i> true - false.
246   * </p>
247   * </blockquote>
248   *
249   * @return is marine flag
250   */
251  @Nullable
252  public Boolean isMarine() {
253    return marine;
254  }
255
256  /**
257   * @param marine the marine to set
258   */
259  public void setMarine(Boolean marine) {
260    this.marine = marine;
261  }
262
263  /**
264   * A Boolean flag indicating the taxon is a terrestial organism, i.e. occurrs on land as opposed to the sea.
265   * <blockquote>
266   * <p>
267   * <i>Example:</i> true - false.
268   * </p>
269   * </blockquote>
270   *
271   * @return the terrestrial
272   */
273  @Nullable
274  public Boolean isTerrestrial() {
275    return terrestrial;
276  }
277
278  /**
279   * @param terrestrial the terrestrial to set
280   */
281  public void setTerrestrial(Boolean terrestrial) {
282    this.terrestrial = terrestrial;
283  }
284
285  @Nullable
286  public String getSource() {
287    return source;
288  }
289
290  public void setSource(String source) {
291    this.source = source;
292  }
293
294  @Nullable
295  public Integer getSourceTaxonKey() {
296    return sourceTaxonKey;
297  }
298
299  public void setSourceTaxonKey(Integer sourceTaxonKey) {
300    this.sourceTaxonKey = sourceTaxonKey;
301  }
302
303  @Override
304  public boolean equals(Object object) {
305    if (this == object) {
306      return true;
307    }
308    if (!(object instanceof SpeciesProfile)) {
309      return false;
310    }
311
312    SpeciesProfile that = (SpeciesProfile) object;
313    return Objects.equal(this.livingPeriod, that.livingPeriod)
314           && Objects.equal(this.lifeForm, that.lifeForm)
315           && Objects.equal(this.habitat, that.habitat)
316           && Objects.equal(this.marine, that.marine)
317           && Objects.equal(this.terrestrial, that.terrestrial)
318           && Objects.equal(this.extinct, that.extinct)
319           && Objects.equal(this.hybrid, that.hybrid)
320           && Objects.equal(this.ageInDays, that.ageInDays)
321           && Objects.equal(this.sizeInMillimeter, that.sizeInMillimeter)
322           && Objects.equal(this.massInGram, that.massInGram)
323           && Objects.equal(this.source, that.source)
324           && Objects.equal(this.sourceTaxonKey, that.sourceTaxonKey);
325  }
326
327  @Override
328  public int hashCode() {
329    return Objects.hashCode(livingPeriod,
330                            lifeForm,
331                            habitat,
332                            marine,
333                            terrestrial,
334                            extinct,
335                            hybrid,
336                            ageInDays,
337                            sizeInMillimeter,
338                            massInGram,
339                            source,
340                            sourceTaxonKey);
341  }
342
343  @Override
344  public String toString() {
345    return Objects.toStringHelper(this)
346      .add("livingPeriod", livingPeriod)
347      .add("lifeForm", lifeForm)
348      .add("habitat", habitat)
349      .add("marine", marine)
350      .add("terrestrial", terrestrial)
351      .add("extinct", extinct)
352      .add("hybrid", hybrid)
353      .add("ageInDays", ageInDays)
354      .add("sizeInMillimeter", sizeInMillimeter)
355      .add("massInGram", massInGram)
356      .add("source", source)
357      .add("sourceTaxonKey", sourceTaxonKey)
358      .toString();
359  }
