016package org.gbif.api.model.checklistbank;
017
018import java.util.Objects;
019import java.util.StringJoiner;
020
021import javax.annotation.Nullable;
022
023/**
024 * SpeciesProfile Model Object represents a species profile which describes basic species characteristics.
025 *
026 * @see <a href="http://rs.gbif.org/extension/gbif/1.0/speciesprofile.xml">Species Profile Definition</a>
027 */
028@SuppressWarnings("unused")
029public class SpeciesProfile implements NameUsageExtension {
030
031  private Integer taxonKey;
032  private String livingPeriod;
033  private String lifeForm;
034  private String habitat;
035  private Boolean marine;
036  private Boolean freshwater;
037  private Boolean terrestrial;
038  private Boolean extinct;
039  private Boolean hybrid;
040  private Integer ageInDays;
041  private Integer sizeInMillimeter;
042  private Integer massInGram;
043  private String source;
044  private Integer sourceTaxonKey;
045
046  /**
047   * The name usage "taxon" key this description belongs to.
048   */
049  @Override
050  public Integer getTaxonKey() {
051    return taxonKey;
052  }
053
054  @Override
055  public void setTaxonKey(Integer taxonKey) {
056    this.taxonKey = taxonKey;
057  }
058
059  /**
060   * Maximum observed age of an organism given as number of days.
061   * <blockquote>
062   * <p>
063   * <i>Example:</i> 5.
064   * </p>
065   * </blockquote>
066   *
067   * @return the ageInDays
068   */
069  @Nullable
070  public Integer getAgeInDays() {
071    return ageInDays;
072  }
073
074  /**
075   * @param ageInDays the ageInDays to set
076   */
077  public void setAgeInDays(Integer ageInDays) {
078    this.ageInDays = ageInDays;
079  }
080
081  /**
082   * A Boolean flag indicating whether the taxon is a freshwater organism.
083   *
084   * @return freshwater flag
085   */
086  @Nullable
087  public Boolean isFreshwater() {
088    return freshwater;
089  }
090
091  public void setFreshwater(Boolean freshwater) {
092    this.freshwater = freshwater;
093  }
094
095  /**
096   * Comma seperated list of mayor habitat classification as defined by IUCN in which a species is known to exist:
097   * <a href="http://www.iucnredlist.org/static/major_habitats">http://www.iucnredlist.org/static/major_habitats</a>
098   * <blockquote>
099   * <p>
100   * <i>Example:</i> 1.1.
101   * </p>
102   * </blockquote>
103   *
104   * @return the habitat
105   */
106  @Nullable
107  public String getHabitat() {
108    return habitat;
109  }
110
111  /**
112   * @param habitat the habitat to set.
113   */
114  public void setHabitat(String habitat) {
115    this.habitat = habitat;
116  }
117
118  /**
119   * A term describing the growth/lifeform of an organism. Should be based on a vocabulary like Raunkiær for plants:
120   * <a href="http://en.wikipedia.org/wiki/Raunkiær_plant_life-form">http://en.wikipedia.org/wiki/Raunkiær_plant_life-
121   * form</a>. Recommended vocabulary:
122   * <a href="http://rs.gbif.org/vocabulary/gbif/life_form.xml">http://rs.gbif.org/vocabulary/gbif/life_form.xml</a>
123   * <blockquote>
124   * <p>
125   * <i>Example:</i> Phanerophyte.
126   * </p>
127   * </blockquote>
128   *
129   * @return the lifeForm
130   */
131  @Nullable
132  public String getLifeForm() {
133    return lifeForm;
134  }
135
136  /**
137   * @param lifeForm the lifeForm to set
138   */
139  public void setLifeForm(String lifeForm) {
140    this.lifeForm = lifeForm;
141  }
142
143  /**
144   * The (geological) time a currently extinct organism is known to have lived. For geological times of fossils ideally
145   * based on a vocabulary like <a
146   * href="http://en.wikipedia.org/wiki/Geologic_column">http://en.wikipedia.org/wiki/Geologic_column</a>
147   * <blockquote>
148   * <p>
149   * <i>Example:</i> Cretaceous.
150   * </p>
151   * </blockquote>
152   *
153   * @return the livingPeriod
154   */
155  @Nullable
156  public String getLivingPeriod() {
157    return livingPeriod;
158  }
159
160  /**
161   * @param livingPeriod the livingPeriod to set
162   */
163  public void setLivingPeriod(String livingPeriod) {
164    this.livingPeriod = livingPeriod;
165  }
166
167  /**
168   * Maximum observed weight of an organism in grams.
169   * <blockquote>
170   * <p>
171   * <i>Example:</i> 12.
172   * </p>
173   * </blockquote>
174   *
175   * @return the massInGram
176   */
177  @Nullable
178  public Integer getMassInGram() {
179    return massInGram;
180  }
181
182  /**
183   * @param massInGram the massInGram to set
184   */
185  public void setMassInGram(Integer massInGram) {
186    this.massInGram = massInGram;
187  }
188
189  /**
190   * Maximum observed size of an organism in millimeter. Can be either height, length or width, whichever is greater.
191   * <blockquote>
192   * <p>
193   * <i>Example:</i> 1500.
194   * </p>
195   * </blockquote>
196   *
197   * @return the sizeInMillimeter
198   */
199  @Nullable
200  public Integer getSizeInMillimeter() {
201    return sizeInMillimeter;
202  }
203
204  /**
205   * @param sizeInMillimeter the sizeInMillimeter to se
206   */
207  public void setSizeInMillimeter(Integer sizeInMillimeter) {
208    this.sizeInMillimeter = sizeInMillimeter;
209  }
210
211  /**
212   * Flag indicating an extinct organism. Details about the timeperiod the organism has lived in can be supplied below.
213   * <blockquote>
214   * <p>
215   * <i>Example:</i> true - false.
216   * </p>
217   * </blockquote>
218   *
219   * @return the extinct
220   */
221  @Nullable
222  public Boolean isExtinct() {
223    return extinct;
224  }
225
226  /**
227   * @param extinct the extinct to set
228   */
229  public void setExtinct(Boolean extinct) {
230    this.extinct = extinct;
231  }
232
233  /**
234   * Flag indicating a hybrid organism. This does not have to be reflected in the name, but can be based on other
235   * studies like chromosome numbers etc.
236   * <blockquote>
237   * <p>
238   * <i>Example:</i> true - false.
239   * </p>
240   * </blockquote>
241   *
242   * @return the hybrid
243   */
244  @Nullable
245  public Boolean isHybrid() {
246    return hybrid;
247  }
248
249  /**
250   * @param hybrid the hybrid to set
251   */
252  public void setHybrid(Boolean hybrid) {
253    this.hybrid = hybrid;
254  }
255
256  /**
257   * A Boolean flag indicating whether the taxon is a marine organism, i.e. can be found in/above sea water
258   * <blockquote>
259   * <p>
260   * <i>Example:</i> true - false.
261   * </p>
262   * </blockquote>
263   *
264   * @return is marine flag
265   */
266  @Nullable
267  public Boolean isMarine() {
268    return marine;
269  }
270
271  /**
272   * @param marine the marine to set
273   */
274  public void setMarine(Boolean marine) {
275    this.marine = marine;
276  }
277
278  /**
279   * A Boolean flag indicating the taxon is a terrestial organism, i.e. occurrs on land as opposed to the sea.
280   * <blockquote>
281   * <p>
282   * <i>Example:</i> true - false.
283   * </p>
284   * </blockquote>
285   *
286   * @return the terrestrial
287   */
288  @Nullable
289  public Boolean isTerrestrial() {
290    return terrestrial;
291  }
292
293  /**
294   * @param terrestrial the terrestrial to set
295   */
296  public void setTerrestrial(Boolean terrestrial) {
297    this.terrestrial = terrestrial;
298  }
299
300  @Nullable
301  @Override
302  public String getSource() {
303    return source;
304  }
305
306  @Override
307  public void setSource(String source) {
308    this.source = source;
309  }
310
311  @Nullable
312  @Override
313  public Integer getSourceTaxonKey() {
314    return sourceTaxonKey;
315  }
316
317  @Override
318  public void setSourceTaxonKey(Integer sourceTaxonKey) {
319    this.sourceTaxonKey = sourceTaxonKey;
320  }
321
322  @Override
323  public boolean equals(Object o) {
324    if (this == o) {
325      return true;
326    }
327    if (o == null || getClass() != o.getClass()) {
328      return false;
329    }
330    SpeciesProfile that = (SpeciesProfile) o;
331    return Objects.equals(livingPeriod, that.livingPeriod) &&
332      Objects.equals(lifeForm, that.lifeForm) &&
333      Objects.equals(habitat, that.habitat) &&
334      Objects.equals(marine, that.marine) &&
335      Objects.equals(terrestrial, that.terrestrial) &&
336      Objects.equals(extinct, that.extinct) &&
337      Objects.equals(hybrid, that.hybrid) &&
338      Objects.equals(ageInDays, that.ageInDays) &&
339      Objects.equals(sizeInMillimeter, that.sizeInMillimeter) &&
340      Objects.equals(massInGram, that.massInGram) &&
341      Objects.equals(source, that.source) &&
342      Objects.equals(sourceTaxonKey, that.sourceTaxonKey);
343  }
344
345  @Override
346  public int hashCode() {
347    return Objects
348      .hash(livingPeriod, lifeForm, habitat, marine, terrestrial, extinct, hybrid, ageInDays,
349        sizeInMillimeter, massInGram, source, sourceTaxonKey);
350  }
351
352  @Override
353  public String toString() {
354    return new StringJoiner(", ", SpeciesProfile.class.getSimpleName() + "[", "]")
355      .add("livingPeriod='" + livingPeriod + "'")
356      .add("lifeForm='" + lifeForm + "'")
357      .add("habitat='" + habitat + "'")
358      .add("marine=" + marine)
359      .add("freshwater=" + freshwater)
360      .add("terrestrial=" + terrestrial)
361      .add("extinct=" + extinct)
362      .add("hybrid=" + hybrid)
363      .add("ageInDays=" + ageInDays)
364      .add("sizeInMillimeter=" + sizeInMillimeter)
365      .add("massInGram=" + massInGram)
366      .add("source='" + source + "'")
367      .add("sourceTaxonKey=" + sourceTaxonKey)
368      .toString();
369  }
370}
