016package org.gbif.api.model.checklistbank;
017
018import org.gbif.api.vocabulary.Country;
019import org.gbif.api.vocabulary.Language;
020import org.gbif.api.vocabulary.LifeStage;
021import org.gbif.api.vocabulary.Sex;
022
023import javax.annotation.Nullable;
024import javax.validation.constraints.NotNull;
025
026import com.google.common.base.Objects;
027
028
029/**
030 * VernacularName Model Object represents a vernacular name for a scientific taxon.
031 *
032 * @see <a href="http://rs.gbif.org/extension/gbif/1.0/vernacularname.xml">Vernacular Name Definition</a>
033 */
034public class VernacularName implements NameUsageExtension {
035
036  private String vernacularName;
037  private Language language;
038  private LifeStage lifeStage;
039  private Sex sex;
040  private Country country;
041  private String area;
042  private String source;
043  private Integer sourceTaxonKey;
044  private Boolean preferred;
045  private Boolean plural;
046
047  /**
048   * The area for the vernacular name.
049   *
050   * @return the area
051   */
052  @Nullable
053  public String getArea() {
054    return area;
055  }
056
057  /**
058   * @param area the area to set
059   */
060  public void setArea(String area) {
061    this.area = area;
062  }
063
064  /**
065   * The standard code for the country in which the vernacular name is used. Recommended best practice is to use the
066   * ISO 3166-1-alpha-2 country codes available as a vocabulary at http://rs.gbif.org/vocabulary/iso/3166-1_alpha2.xml.
067   * For multiple countries separate values with a comma ",".
068   * <blockquote>
069   * <p>
070   * <i>Example:</i> "AR" for Argentina, "SV" for El Salvador. "AR,CR,SV" for Argentina, Costa Rica, and El Salvador
071   * combined.
072   * </p>
073   * </blockquote>
074   *
075   * @return the country
076   */
077  @Nullable
078  public Country getCountry() {
079    return country;
080  }
081
082  /**
083   * @param country the country to set
084   */
085  public void setCountry(Country country) {
086    this.country = country;
087  }
088
089  /**
090   * ISO 639-1 language code used for the vernacular name value.
091   * <blockquote>
092   * <p>
093   * <i>Example:</i> es
094   * </p>
095   * </blockquote>
096   *
097   * @return the language
098   */
099  @Nullable
100  public Language getLanguage() {
101    return language;
102  }
103
104  /**
105   * @param language the language to set
106   */
107  public void setLanguage(Language language) {
108    this.language = language;
109  }
110
111  /**
112   * The age class or life stage of the species for which the vernacular name applies. Best practice is to utilise a
113   * controlled list of terms for this value.
114   * <blockquote>
115   * <p>
116   * <i>Example:</i> juvenile" is the life stage of the fish Pomatomus saltatrix for which the name "snapper blue"
117   * refers.
118   * </p>
119   * </blockquote>
120   *
121   * @return the lifeStage
122   *
123   * @see <a href="http://rs.gbif.org/vocabulary/gbif/life_stage.xml">Life Stage GBIF Vocabulary</a>
124   */
125  @Nullable
126  public LifeStage getLifeStage() {
127    return lifeStage;
128  }
129
130  /**
131   * @param lifeStage the lifeStage to set
132   */
133  public void setLifeStage(LifeStage lifeStage) {
134    this.lifeStage = lifeStage;
135  }
136
137  /**
138   * The sex (gender) of the taxon for which the vernacular name applies when the vernacular name is limited to a
139   * specific gender of a species. If not limited sex should be empty. For example the vernacular name "Buck" applies
140   * to the "Male" gender of the species, Odocoileus virginianus.
141   * <blockquote>
142   * <p>
143   * <i>Example:</i> male.
144   * </p>
145   * </blockquote>
146   *
147   * @return the sex
148   *
149   * @see <a href="http://rs.gbif.org/vocabulary/gbif/sex.xml">Sex GBIF Vocabulary</a>
150   */
151  @Nullable
152  public Sex getSex() {
153    return sex;
154  }
155
156  /**
157   * @param sex the sex to set
158   */
159  public void setSex(Sex sex) {
160    this.sex = sex;
161  }
162
163  /**
164   * Bibliographic citation referencing a source where the vernacular name refers to the cited species.
165   * <blockquote>
166   * <p>
167   * <i>Example:</i> Peterson Field Guide to the Eastern Seashore, Houghton Mifflin Co, 1961, p131.
168   * </p>
169   * </blockquote>
170   *
171   * @return the source
172   */
173  @Nullable
174  public String getSource() {
175    return source;
176  }
177
178  /**
179   * @param source the source to set
180   */
181  public void setSource(String source) {
182    this.source = source;
183  }
184
185  @Nullable
186  public Integer getSourceTaxonKey() {
187    return sourceTaxonKey;
188  }
189
190  public void setSourceTaxonKey(Integer sourceTaxonKey) {
191    this.sourceTaxonKey = sourceTaxonKey;
192  }
193
194  /**
195   * A common or vernacular name.
196   * <blockquote>
197   * <p>
198   * <i>Example:</i> Andean Condor", "Condor Andino", "American Eagle", "Gänsegeier".
199   * </p>
200   * </blockquote>
201   *
202   * @return the vernacularName
203   */
204  @NotNull
205  public String getVernacularName() {
206    return vernacularName;
207  }
208
209  /**
210   * @param vernacularName the vernacularName to set
211   */
212  public void setVernacularName(String vernacularName) {
213    this.vernacularName = vernacularName;
214  }
215
216  /**
217   * This value is true if the vernacular name it qualifies refers to a plural form of the name.
218   * <blockquote>
219   * <p>
220   * <i>Example:</i> The term "Schoolies" is the plural form of a name used along the coastal Northeastern U.S. for
221   * groups of juvenile fish of the species, Morone saxatilis.
222   * </p>
223   * </blockquote>
224   *
225   * @return the plural
226   *
227   * @see <a href="http://rs.gbif.org/vocabulary/basic/boolean.xml">Boolean Vocabulary</a>
228   */
229  @Nullable
230  public Boolean isPlural() {
231    return plural;
232  }
233
234  /**
235   * @param plural the plural to set
236   */
237  public void setPlural(Boolean plural) {
238    this.plural = plural;
239  }
240
241  /**
242   * This term is true if the source citing the use of this vernacular name indicates the usage has some preference or
243   * specific standing over other possible vernacular names used for the species.
244   * <blockquote>
245   * <p>
246   * <i>Example:</i> Some organisations have attempted to assign specific and unique vernacular names for particular
247   * taxon groups in a systematic attempt to bring order and consistency to the use of these names. For example, the
248   * American Ornithological Union assigns the name "Pearl Kite" for the taxon, Gampsonyx swainsonii. The value of
249   * isPreferredName for this record would be true.
250   * </p>
251   * </blockquote>
252   *
253   * @return the preferred
254   *
255   * @see <a href="http://rs.gbif.org/vocabulary/basic/boolean.xml">Boolean Vocabulary</a>
256   */
257  @Nullable
258  public Boolean isPreferred() {
259    return preferred;
260  }
261
262  /**
263   * @param preferred the preferred to set
264   */
265  public void setPreferred(Boolean preferred) {
266    this.preferred = preferred;
267  }
268
269  @Override
270  public boolean equals(Object object) {
271    if (this == object) {
272      return true;
273    }
274    if (!(object instanceof VernacularName)) {
275      return false;
276    }
277
278    VernacularName that = (VernacularName) object;
279    return Objects.equal(this.vernacularName, that.vernacularName)
280           && Objects.equal(this.language, that.language)
281           && Objects.equal(this.lifeStage, that.lifeStage)
282           && Objects.equal(this.sex, that.sex)
283           && Objects.equal(this.country, that.country)
284           && Objects.equal(this.area, that.area)
285           && Objects.equal(this.source, that.source)
286           && Objects.equal(this.sourceTaxonKey, that.sourceTaxonKey)
287           && Objects.equal(this.preferred, that.preferred)
288           && Objects.equal(this.plural, that.plural);
289  }
290
291  @Override
292  public int hashCode() {
293    return Objects.hashCode(vernacularName,
294                            language,
295                            lifeStage,
296                            sex,
297                            country,
298                            area,
299                            source,
300                            sourceTaxonKey,
301                            preferred,
302                            plural);
303  }
304
305  @Override
306  public String toString() {
307    return Objects.toStringHelper(this)
308      .add("vernacularName", vernacularName)
309      .add("language", language)
310      .add("lifeStage", lifeStage)
311      .add("sex", sex)
312      .add("country", country)
313      .add("area", area)
314      .add("source", source)
315      .add("sourceTaxonKey", sourceTaxonKey)
316      .add("preferred", preferred)
317      .add("plural", plural)
318      .toString();
319  }
