032/**
033 * A container of a taxon name that is atomised into it's relevant separate parts.
034 * Hybrid formulas that consist of multiple genera, binomials or species
035 * epitheta are relatively poor represented. A simple boolean flag indicates a
036 * hybrid formula, e.g. Polygala vulgaris X Polygala epinema but not named
037 * hybrids that are have an x or notho as part of a well formed
038 * mono/bi/trinomial e.g. xPolygala vulgaris. In the case of hybrid formulas,
039 * i.e. isHybrid=true, the first name in the formula is tried to be kept, at
040 * least the genus.
041 * <p/>
042 * A container of a taxon name that is only atomised into three name parts maximum plus rank and a notho property
043 * indicating the name part of named hybrids that is considered to be the hybrid. No authorship is kept. For subgenera
044 * we don't use parenthesis to indicate the subgenus, but use explicit rank markers instead.
045 */

274  /**
275   * build a name controlling all available flags for name parts to be included in the resulting name.
276   *
277   * @param hybridMarker    include the hybrid marker with the name if existing
278   * @param rankMarker      include the infraspecific or infrageneric rank marker with the name if existing
279   * @param authorship      include the names authorship (authorteam and year)
280   * @param infrageneric include the infrageneric name in brackets for species or infraspecies
281   * @param genusForInfrageneric include the genus name in front of an infrageneric name (not a species)
282   * @param abbreviateGenus if true abreviate the genus with its first character
283   * @param decomposition   decompose unicode ligatures into their corresponding ascii ones, e.g. Ã¦ beomes ae
284   * @param asciiOnly       transform unicode letters into their corresponding ascii ones, e.g. Ã¸ beomes o and Ã¼ u
285   * @param showIndet       if true include the rank marker for incomplete determinations, for example Puma spec.
286   * @param nomNote         include nomenclatural notes
287   * @param remarks         include informal remarks
288   */
289  public String buildName(
290    boolean hybridMarker,
291    boolean rankMarker,
292    boolean authorship,
293    boolean infrageneric,
294    boolean genusForInfrageneric,
295    boolean abbreviateGenus,
296    boolean decomposition,
297    boolean asciiOnly,
298    boolean showIndet,
299    boolean nomNote,
300    boolean remarks,
301    boolean showSensu,
302    boolean showCultivar,
303    boolean showStrain
304  ) {
305    StringBuilder sb = new StringBuilder();
306    Rank rnk = getRank();
307
308    if (NameType.CANDIDATUS == type) {
309      sb.append("Candidatus ");
310    }
311
312    if (genusOrAbove != null && (genusForInfrageneric || infraGeneric == null || specificEpithet != null)) {
313      if (hybridMarker && NamePart.GENERIC == notho) {
314        sb.append(HYBRID_MARKER);
315      }
316      if (abbreviateGenus) {
317        sb.append(genusOrAbove.substring(0, 1)).append('.');
318      } else {
319        sb.append(genusOrAbove);
320      }
321    }
322    if (specificEpithet == null) {
323      if (Rank.SPECIES == rnk) {
324        // no species epitheton given, but rank=species. Indetermined species!
325        if (showIndet) {
326          sb.append(" spec.");
327        }
328      } else if (rnk != null && rnk.isInfraspecific()) {
329        // no species epitheton given, but rank below species. Indetermined!
330        if (showIndet) {
331          sb.append(' ');
332          sb.append(rnk.getMarker());
333        }
334      } else if (infraGeneric != null) {
335        // this is the terminal name part - always show it!
336        if (rankMarker && rank != null) {
337          // If we know the rank we use explicit rank markers
338          // this is how botanical infrageneric names are formed, see http://www.iapt-taxon.org/nomen/main.php?page=art21
339          sb.append(' ')
340            .append(rank)
341            .append(' ')
342            .append(infraGeneric);
343        } else {
344            if (genusForInfrageneric && genusOrAbove != null) {
345              // if we have shown the genus already and we do not know the rank we use parenthesis to indicate an infrageneric
346              sb.append(" (")
347                      .append(infraGeneric)
348                      .append(")");
349            } else {
350              // no genus shown yet, just show the plain infrageneric name
351              sb.append(infraGeneric);
352            }
353        }
354      }
355      // genus/infrageneric authorship
356      if (authorship) {
357        appendAuthorship(sb);
358      }
359    } else {
360      if (infrageneric && infraGeneric != null && (rank == null || getRank() == Rank.GENUS)) {
361        // only show subgenus if requested
362        sb.append(" (");
363        sb.append(infraGeneric);
364        sb.append(')');
365      }
366
367      // species part
368      sb.append(' ');
369      if (hybridMarker && NamePart.SPECIFIC == notho) {
370        sb.append(HYBRID_MARKER);
371      }
372      String epi = specificEpithet.replaceAll("[ _-]", "-");
373      sb.append(epi);
374
375      if (infraSpecificEpithet == null) {
376        // Indetermined? Only show indet cultivar marker if no cultivar epithet exists
377        if (showIndet && rnk != null && rnk.isInfraspecific() && (Rank.CULTIVAR != rnk || cultivarEpithet == null)) {
378          // no infraspecific epitheton given, but rank below species. Indetermined!
379          sb.append(' ');
380          sb.append(rnk.getMarker());
381        }
382
383        // species authorship
384        if (authorship) {
385          appendAuthorship(sb);
386        }
387      } else {
388        // infraspecific part
389        sb.append(' ');
390        if (hybridMarker && NamePart.INFRASPECIFIC == notho) {
391          if (rankMarker) {
392            sb.append("notho");
393          } else {
394            sb.append(HYBRID_MARKER);
395          }
396        }
397        if (rankMarker && rank != null) {
398          sb.append(rank);
399          sb.append(' ');
400        }
401        epi = infraSpecificEpithet.replaceAll("[ _-]", "-");
402        sb.append(epi);
403        // non autonym authorship ?
404        if (authorship && !isAutonym()) {
405          appendAuthorship(sb);
406        }
407      }
408    }
409
410    // add cultivar name
411    if (showStrain && strain != null) {
412      sb.append(" ");
413      sb.append(strain);
414    }
415
416    // add cultivar name
417    if (showCultivar && cultivarEpithet != null) {
418      sb.append(" '");
419      sb.append(cultivarEpithet);
420      sb.append("'");
421    }
422
423    // add sensu/sec reference
424    if (showSensu && sensu != null) {
425      sb.append(" ");
426      sb.append(sensu);
427    }
428
429    // add nom status
430    if (nomNote && nomStatus != null) {
431      sb.append(", ");
432      sb.append(nomStatus);
433    }
434
435    // add remarks
436    if (remarks && this.remarks != null) {
437      sb.append(" [");
438      sb.append(this.remarks);
439      sb.append("]");
440    }
441
442    String name = sb.toString().trim();
443    if (decomposition) {
444      name = UnicodeUtils.decompose(name);
445    }
446    if (asciiOnly) {
447      name = UnicodeUtils.ascii(name);
448    }
449    return Strings.emptyToNull(name);
450  }

459  private void appendAuthorship(StringBuilder sb) {
460    if (bracketAuthorship == null) {
461      if (bracketYear != null) {
462        sb.append(" (");
463        sb.append(bracketYear);
464        sb.append(")");
465      }
466    } else {
467      sb.append(" (");
468      sb.append(bracketAuthorship);
469      if (bracketYear != null) {
470        sb.append(", ");
471        sb.append(bracketYear);
472      }
473      sb.append(")");
474    }
475    if (authorship != null) {
476      sb.append(" ").append(authorship);
477    }
478    if (year != null) {
479      sb.append(", ");
480      sb.append(year);
481    }
482  }

485   * The canonical name sensu strictu with nothing else but 3 name parts at max (genus, species, infraspecific). No
486   * rank or hybrid markers and no authorship, cultivar or strain information.
487   * Infrageneric names are represented without a leading genus.
488   * Unicode characters will be replaced by their matching ASCII characters.
489   * <p/>
490   * For example:
491   * Abies alba
492   * Abies alba alpina
493   * Abies Bracteata
494   * Heucherella tiarelloides
495   *
496   * @return the 1,2 or 3 parted name as a single string
497   */

663  public void setRankMarker(String rankMarker) {
664    if (rankMarker == null) {
665      rank = null;
666    } else {
667      rankMarker = rankMarker.trim().toLowerCase();
668      if (rankMarker.startsWith("notho")) {
669        rankMarker = rankMarker.substring(5);
670        notho = NamePart.INFRASPECIFIC;
671      }
672      Rank r = Rank.inferRank(rankMarker);
673      if (r == null || Rank.UNCOMPARABLE_RANKS.contains(r)) {
674        this.rank = rankMarker;
675      } else {
676        this.rank = r.getMarker();
677      }
678    }
679  }

690  @Override
691  public boolean equals(Object obj) {
692    if (this == obj) {
693      return true;
694    }
695    if (!(obj instanceof ParsedName)) {
696      return false;
697    }
698    ParsedName o = (ParsedName) obj;
699    return equal(key, o.key)
700           && equal(scientificName, o.scientificName)
701           && equal(type, o.type)
702           && equal(genusOrAbove, o.genusOrAbove)
703           && equal(infraGeneric, o.infraGeneric)
704           && equal(specificEpithet, o.specificEpithet)
705           && equal(infraSpecificEpithet, o.infraSpecificEpithet)
706           && equal(cultivarEpithet, o.cultivarEpithet)
707           && equal(strain, o.strain)
708           && equal(authorship, o.authorship)
709           && equal(year, o.year)
710           && equal(bracketAuthorship, o.bracketAuthorship)
711           && equal(bracketYear, o.bracketYear)
712           && equal(rank, o.rank);

734  @Override
735  public String toString() {
736    StringBuilder sb = new StringBuilder();
737    sb.append(scientificName);
738    if (key != null) {
739      sb.append(" [");
740      sb.append(key);
741      sb.append("]");
742    }
743    if (genusOrAbove != null) {
744      sb.append(" G:").append(genusOrAbove);
745    }
746    if (infraGeneric != null) {
747      sb.append(" IG:").append(infraGeneric);
748    }
749    if (specificEpithet != null) {
750      sb.append(" S:").append(specificEpithet);
751    }
752    if (rank != null) {
