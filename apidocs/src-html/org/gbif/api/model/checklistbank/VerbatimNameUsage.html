016package org.gbif.api.model.checklistbank;
017
018import org.gbif.api.vocabulary.Extension;
019
020import java.util.List;
021import java.util.Map;
022
023import com.google.common.collect.Maps;
024
025public class VerbatimNameUsage {
026
027  private Integer key;
028
029  // the verbatim taxon fields for the usage
030  private Map<String, String> fields = Maps.newHashMap();
031
032  // the verbatim extension records as read by a dwc star record, keyed on the extension
033  private Map<Extension, List<Map<String, String>>> extensions = Maps.newHashMap();
034
035  public Map<Extension, List<Map<String, String>>> getExtensions() {
036    return extensions;
037  }
038
039  public void setExtensions(Map<Extension, List<Map<String, String>>> extensions) {
040    this.extensions = extensions;
041  }
042
043  public Map<String, String> getFields() {
044    return fields;
045  }
046
047  public void setFields(Map<String, String> fields) {
048    this.fields = fields;
049  }
050
051  public Integer getKey() {
052    return key;
053  }
054
055  public void setKey(Integer key) {
056    this.key = key;
057  }
058
059}
