431  @Override
432  public boolean equals(Object obj) {
433    if (this == obj) {
434      return true;
435    }
436    if (obj == null) {
437      return false;
438    }
439    if (getClass() != obj.getClass()) {
440      return false;
441    }
442    final NameUsageMatch other = (NameUsageMatch) obj;
443    return Objects.equal(this.usageKey, other.usageKey)
444           && Objects.equal(this.acceptedUsageKey, other.acceptedUsageKey)
445           && Objects.equal(this.scientificName, other.scientificName)
446           && Objects.equal(this.canonicalName, other.canonicalName)
447           && Objects.equal(this.rank, other.rank)
448           && Objects.equal(this.status, other.status)
449           && Objects.equal(this.confidence, other.confidence)
450           && Objects.equal(this.note, other.note)
451           && Objects.equal(this.matchType, other.matchType)
452           && Objects.equal(this.alternatives, other.alternatives)
453           && Objects.equal(this.kingdom, other.kingdom)
454           && Objects.equal(this.phylum, other.phylum)
455           && Objects.equal(this.clazz, other.clazz)
456           && Objects.equal(this.order, other.order)
457           && Objects.equal(this.family, other.family)
458           && Objects.equal(this.genus, other.genus)
459           && Objects.equal(this.subgenus, other.subgenus)
460           && Objects.equal(this.species, other.species)
461           && Objects.equal(this.kingdomKey, other.kingdomKey)
462           && Objects.equal(this.phylumKey, other.phylumKey)
463           && Objects.equal(this.classKey, other.classKey)

503  public enum MatchType {
504
505    /**
506     * An exact, straight match.
507     */
508    EXACT,
509    /**
510     * A fuzzy, non exact match.
511     */
512    FUZZY,
513    /**
514     * Matching on a higher rank than the lowest name given.
515     */
516    HIGHERRANK,
517    /**
518     * No match or matching several names with too little information to keep apart.
519     */
520    NONE
521
522  }
