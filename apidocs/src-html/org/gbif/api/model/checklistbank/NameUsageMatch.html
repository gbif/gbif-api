402  @Override
403  public boolean equals(Object o) {
404    if (this == o) {
405      return true;
406    }
407    if (o == null || getClass() != o.getClass()) {
408      return false;
409    }
410    NameUsageMatch that = (NameUsageMatch) o;
411    return Objects.equals(usageKey, that.usageKey) &&
412      Objects.equals(acceptedUsageKey, that.acceptedUsageKey) &&
413      Objects.equals(scientificName, that.scientificName) &&
414      Objects.equals(canonicalName, that.canonicalName) &&
415      rank == that.rank &&
416      status == that.status &&
417      Objects.equals(confidence, that.confidence) &&
418      Objects.equals(note, that.note) &&
419      matchType == that.matchType &&
420      Objects.equals(alternatives, that.alternatives) &&
421      Objects.equals(kingdom, that.kingdom) &&
422      Objects.equals(phylum, that.phylum) &&
423      Objects.equals(clazz, that.clazz) &&
424      Objects.equals(order, that.order) &&
425      Objects.equals(family, that.family) &&
426      Objects.equals(genus, that.genus) &&
427      Objects.equals(subgenus, that.subgenus) &&
428      Objects.equals(species, that.species) &&
429      Objects.equals(kingdomKey, that.kingdomKey) &&
430      Objects.equals(phylumKey, that.phylumKey) &&
431      Objects.equals(classKey, that.classKey) &&
432      Objects.equals(orderKey, that.orderKey) &&
433      Objects.equals(familyKey, that.familyKey) &&
434      Objects.equals(genusKey, that.genusKey) &&
435      Objects.equals(subgenusKey, that.subgenusKey) &&
436      Objects.equals(speciesKey, that.speciesKey);
437  }

480  public enum MatchType {
481
482    /**
483     * An exact, straight match.
484     */
485    EXACT,
486    /**
487     * A fuzzy, non exact match.
488     */
489    FUZZY,
490    /**
491     * Matching on a higher rank than the lowest name given.
492     */
493    HIGHERRANK,
494    /**
495     * No match or matching several names with too little information to keep apart.
496     */
497    NONE
498  }
