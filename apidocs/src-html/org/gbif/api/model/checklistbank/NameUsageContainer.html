037/**
038 * An extension to a NameUsage adding all further properties that are not eagerly loaded.
039 * Use this class to store the various subresources as you need them side by side with the NameUsage.
040 * Note that properties are not automatically/lazy loaded in any way.
041 * This is just a simple container class with a few convenience methods which needs to be populated manually via its
042 * setters or the constructor.
043 */
044public class NameUsageContainer extends NameUsage {
045  private static final ObjectMapper mapper = new ObjectMapper();
046  static {
047      mapper.configure(DeserializationConfig.Feature.FAIL_ON_UNKNOWN_PROPERTIES, false);
048  }
049
050  private List<Description> descriptions = newArrayList();
051  private List<Distribution> distributions = newArrayList();
052  private List<Identifier> identifiers = newArrayList();
053  private List<NameUsageMediaObject> media = newArrayList();
054  private List<Reference> referenceList = newArrayList();
055  private List<SpeciesProfile> speciesProfiles = newArrayList();
056  private List<NameUsage> synonyms = newArrayList();
057  private List<NameUsage> combinations = newArrayList();
058  private List<TypeSpecimen> typeSpecimens = newArrayList();
059  private List<VernacularName> vernacularNames = newArrayList();
060
061  public NameUsageContainer() {
062
063  }
064
065  /**
066   * Constructs a NameUsageContainer from an existing NameUsage instance.
067   */
068  public NameUsageContainer(NameUsage usage) {
069    try {
070      JsonNode propTree = mapper.convertValue(usage, JsonNode.class);
071      mapper.readerForUpdating(this).readValue(propTree);
072    } catch (IOException e) {
073      throw new IllegalStateException("Failed to copy NameUsage properties to NameUsageContainer", e);
074    }
075  }

108   * Convenience method that analyzes all species profile records and returns the distinct list of known habitats.
109   *
110   * @return list of unique habitats
111   */
112  @NotNull
113  public Set<String> getHabitats() {
114    Set<String> habitats = Sets.newLinkedHashSet();
115    for (SpeciesProfile sp : speciesProfiles) {
116      if (!Strings.isNullOrEmpty(sp.getHabitat())) {
117        habitats.add(sp.getHabitat());
118      }
119    }
120    return habitats;
121  }

280  /**
281   * Convenience method that analyzes all species profile records for extinct.
282   * If several records contradict use the majority or if numbers are equal true.
283   *
284   * @return true if extinct
285   */
286  public Boolean isExtinct() {
287    int ctrue = 0;
288    int cfalse = 0;
289    for (SpeciesProfile sp : speciesProfiles) {
290      if (sp.isExtinct() == null) {
291        continue;
292      }
293
294      if (sp.isExtinct()) {
295        ctrue++;
296      } else if (!sp.isExtinct()) {
297        cfalse++;
298      }
299    }
300    return ctrue + cfalse == 0 ? null : ctrue >= cfalse;
301  }

369  @Override
370  public boolean equals(Object object) {
371    if (object instanceof NameUsageContainer) {
372      if (!super.equals(object)) {
373        return false;
374      }
375      final NameUsageContainer other = (NameUsageContainer) object;
376      return Objects.equal(this.descriptions, other.descriptions)
377             && Objects.equal(this.distributions, other.distributions)
378             && Objects.equal(this.media, other.media)
379             && Objects.equal(this.referenceList, other.referenceList)
380             && Objects.equal(this.speciesProfiles, other.speciesProfiles)
381             && Objects.equal(this.synonyms, other.synonyms)
382             && Objects.equal(this.typeSpecimens, other.typeSpecimens)
383             && Objects.equal(this.vernacularNames, other.vernacularNames);
384    }
385    return false;
386  }
