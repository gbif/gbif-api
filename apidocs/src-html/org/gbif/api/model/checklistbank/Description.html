016package org.gbif.api.model.checklistbank;
017
018import org.gbif.api.vocabulary.Language;
019
020import javax.annotation.Nullable;
021
022import com.google.common.base.Objects;
023
024
025/**
026 * Description Model Object represents a taxon description.
027 *
028 * @see <a href="http://rs.gbif.org/extension/gbif/1.0/description.xml">Description Definition</a>
029 */
030public class Description implements NameUsageExtension {
031
032  private Integer key;
033  private String type;
034  private Language language;
035  private String description;
036  private String source;
037  private Integer sourceTaxonKey;
038  private String creator;
039  private String contributor;
040  private String license;
041
042  /**
043   * A unique GBIF identifier for any description.
044   * This key is used in the table of contents to retrieve the detailed description.
045   */
046  public Integer getKey() {
047    return key;
048  }
049
050  public void setKey(Integer key) {
051    this.key = key;
052  }
053
054  /**
055   * An entity responsible for making contributions to the textual information provided for a description.
056   *
057   * @return the contributor
058   */
059  @Nullable
060  public String getContributor() {
061    return contributor;
062  }
063
064  /**
065   * @param contributor the contributor to set
066   */
067  public void setContributor(String contributor) {
068    this.contributor = contributor;
069  }
070
071  /**
072   * The author(s) of the textual information provided for a description.
073   *
074   * @return the creator
075   */
076  @Nullable
077  public String getCreator() {
078    return creator;
079  }
080
081  /**
082   * @param creator the creator to set.
083   */
084  public void setCreator(String creator) {
085    this.creator = creator;
086  }
087
088  /**
089   * Any descriptive free text matching the category given as dc:type. The text should be either plain text or
090   * formatted with basic html tags, i.e. h1-4,p,i,b,a,img,ul and li. All other tags should be removed.
091   *
092   * @return the description
093   */
094  @Nullable
095  public String getDescription() {
096    return description;
097  }
098
099  /**
100   * @param description the description to set
101   */
102  public void setDescription(String description) {
103    this.description = description;
104  }
105
106  /**
107   * ISO 639-1 language code used for the vernacular name value.
108   *
109   * @return the language
110   */
111  @Nullable
112  public Language getLanguage() {
113    return language;
114  }
115
116  /**
117   * @param language the language to set
118   */
119  public void setLanguage(Language language) {
120    this.language = language;
121  }
122
123  /**
124   * Official permission to do something with the resource. Please use Creative Commons URIs if you can. <br/>
125   * <blockquote>
126   * <p>
127   * <i>Example:</i> CC-BY
128   * </p>
129   * </blockquote>
130   *
131   * @return the license.
132   */
133  @Nullable
134  public String getLicense() {
135    return license;
136  }
137
138  /**
139   * @param license the license to set
140   */
141  public void setLicense(String license) {
142    this.license = license;
143  }
144
145  /**
146   * Source reference of this description, a URL or full publication citation.
147   *
148   * @return the source
149   */
150  @Nullable
151  public String getSource() {
152    return source;
153  }
154
155  /**
156   * @param source the source to set
157   */
158  public void setSource(String source) {
159    this.source = source;
160  }
161
162  @Nullable
163  public Integer getSourceTaxonKey() {
164    return sourceTaxonKey;
165  }
166
167  public void setSourceTaxonKey(Integer sourceTaxonKey) {
168    this.sourceTaxonKey = sourceTaxonKey;
169  }
170
171  /**
172   * Returns the type used to categorize paragraphs of a taxon description.
173   * Given the list of types is so broad, an Enum is not used to maintain it. Rather it is kept as plain text.
174   *
175   * @return the type
176   *
177   * @see <a href="http://rs.gbif.org/vocabulary/gbif/description_type.xml">Description type definition</a>
178   */
179  @Nullable
180  public String getType() {
181    return type;
182  }
183
184  /**
185   * @param type the type to set
186   */
187  public void setType(String type) {
188    this.type = type;
189  }
190
191  @Override
192  public boolean equals(Object object) {
193    if (this == object) {
194      return true;
195    }
196    if (!(object instanceof Description)) {
197      return false;
198    }
199
200    Description that = (Description) object;
201    return Objects.equal(this.type, that.type)
202           && Objects.equal(this.language, that.language)
203           && Objects.equal(this.description, that.description)
204           && Objects.equal(this.source, that.source)
205           && Objects.equal(this.sourceTaxonKey, that.sourceTaxonKey)
206           && Objects.equal(this.creator, that.creator)
207           && Objects.equal(this.contributor, that.contributor)
208           && Objects.equal(this.license, that.license);
209  }
210
211  @Override
212  public int hashCode() {
213    return Objects.hashCode(type, language, description, source, creator, contributor, license);
214  }
215
216  @Override
217  public String toString() {
218    return Objects.toStringHelper(this)
219      .add("type", type)
220      .add("language", language)
221      .add("description", description)
222      .add("source", source)
223      .add("sourceTaxonKey", sourceTaxonKey)
224      .add("creator", creator)
225      .add("contributor", contributor)
226      .add("license", license)
227      .toString();
228  }
229
230}
