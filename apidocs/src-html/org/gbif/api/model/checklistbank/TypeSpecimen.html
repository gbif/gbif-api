027/**
028 * TypeSpecimen Model for specimens and types, including type specimens, type species and type genera and simple
029 * specimens unrelated to types.
030 *
031 * @see <a href="http://rs.gbif.org/extension/gbif/1.0/typesandspecimen.xml">Types And Specimen Definition</a>
032 */
033public class TypeSpecimen extends NameUsageComponent {

413  @Override
414  public boolean equals(Object object) {
415    if (this == object) {
416      return true;
417    }
418    if (!(object instanceof TypeSpecimen)) {
419      return false;
420    }
421    if (!super.equals(object)) {
422      return false;
423    }
424    TypeSpecimen that = (TypeSpecimen) object;
425    return Objects.equal(this.source, that.source)
426           && Objects.equal(this.citation, that.citation)
427           && Objects.equal(this.typeStatus, that.typeStatus)
428           && Objects.equal(this.typeDesignationType, that.typeDesignationType)
429           && Objects.equal(this.typeDesignatedBy, that.typeDesignatedBy)
430           && Objects.equal(this.scientificName, that.scientificName)
431           && Objects.equal(this.verbatimLabel, that.verbatimLabel)
432           && Objects.equal(this.locality, that.locality)
433           && Objects.equal(this.recordedBy, that.recordedBy)
434           && Objects.equal(this.verbatimEventDate, that.verbatimEventDate)
435           && Objects.equal(this.verbatimLongitude, that.verbatimLongitude)
436           && Objects.equal(this.verbatimLatitude, that.verbatimLatitude)
437           && Objects.equal(this.occurrenceId, that.occurrenceId)
438           && Objects.equal(this.taxonRank, that.taxonRank)
439           && Objects.equal(this.institutionCode, that.institutionCode)
440           && Objects.equal(this.collectionCode, that.collectionCode)
441           && Objects.equal(this.catalogNumber, that.catalogNumber);
442  }
