044/**
045 * A usage of a <em>scientific name</em> according to one particular Checklist including the GBIF Taxonomic Backbone,
046 * the nub. It is shown as species in the new Portal.
047 * All nub usages will have a nubKey equal to their key and belong to the nub checklist registry resource.
048 * Name usages from other checklists with names that also exist in the nub will have
049 * a nubKey that points to the related usage in the nub.
050 * To store not eagerly loaded subresources such as vernacular names or synonyms with a usage please use
051 * the {@link NameUsageContainer} class.
052 */

757  /**
758   * The status of the use of the scientificName as a label for the taxon.
759   * If it's not set explicit the isSynonym and isProParte flags are used to calculate values.
760   * <blockquote>
761   * <p>
762   * <i>Example:</i> "invalid", "misapplied", "homotypic synonym", "accepted"
763   * </p>
764   * </blockquote>
765   *
766   * @return the taxonomicStatus, can be null
767   */
768  @Nullable
769  public TaxonomicStatus getTaxonomicStatus() {
770    if (taxonomicStatus != null) {
771      return taxonomicStatus;
772    }
773
774    if (isSynonym) {
775      return isProParte() ? TaxonomicStatus.PROPARTE_SYNONYM : TaxonomicStatus.SYNONYM;
776    }
777
778    return null;
779  }
