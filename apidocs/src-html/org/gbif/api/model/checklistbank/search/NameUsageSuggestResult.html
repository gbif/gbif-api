289  /**
290   * An ordered map with entries for all higher Linnean ranks down to the actual direct parent of this usage.
291   * The map starts with the highest rank, e.g. the kingdom and maps the name usage key to its canonical name.
292   * The name usage itself is never included, even though a higher rank might point to the usage itself.
293   * 
294   * @return map of higher ranks
295   */
296  @NotNull
297  public LinkedHashMap<Integer, String> getHigherClassificationMap() {
298    return ClassificationUtils.getHigherClassificationMap(this, key, parentKey, parent);

309  @Override
310  public boolean equals(Object obj) {
311    if (obj == null) {
312      return false;
313    }
314    if (getClass() != obj.getClass()) {
315      return false;
316    }
317    final NameUsageSuggestResult other = (NameUsageSuggestResult) obj;
318    return Objects.equal(this.key, other.key)
319      && Objects.equal(this.parentKey, other.parentKey)
320      && Objects.equal(this.parent, other.parent)
321      && Objects.equal(this.kingdom, other.kingdom)
322      && Objects.equal(this.phylum, other.phylum)
323      && Objects.equal(this.clazz, other.clazz)
324      && Objects.equal(this.order, other.order)
325      && Objects.equal(this.family, other.family)
326      && Objects.equal(this.genus, other.genus)
327      && Objects.equal(this.subgenus, other.subgenus)
328      && Objects.equal(this.species, other.species)
329      && Objects.equal(this.kingdomKey, other.kingdomKey)
330      && Objects.equal(this.phylumKey, other.phylumKey)
331      && Objects.equal(this.classKey, other.classKey)
332      && Objects.equal(this.orderKey, other.orderKey)
333      && Objects.equal(this.familyKey, other.familyKey)
334      && Objects.equal(this.genusKey, other.genusKey)
335      && Objects.equal(this.subgenusKey, other.subgenusKey)
336      && Objects.equal(this.speciesKey, other.speciesKey)
337      && Objects.equal(this.canonicalName, other.canonicalName)
338      && Objects.equal(this.rank, other.rank)
339      && Objects.equal(this.nubKey, other.nubKey);
340  }
