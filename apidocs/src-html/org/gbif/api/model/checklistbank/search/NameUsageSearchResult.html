507  /**
508   * An ordered map with entries for all higher Linnean ranks down to the actual direct parent of this usage.
509   * The map starts with the highest rank, e.g. the kingdom and maps the name usage key to its canonical name.
510   * The name usage itself is never included, even though a higher rank might point to the usage itself.
511   *
512   * @return map of higher ranks
513   */
514  @NotNull
515  public LinkedHashMap<Integer, String> getHigherClassificationMap() {
516    return ClassificationUtils.getHigherClassificationMap(this, key, parentKey, parent);
517  }
