014package org.gbif.api.model.common.search;
015
016import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
017
018import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
019
020/**
021 * Interface to be implemented by all search parameter enumerations.
022 * The enumeration member name should avoid the following names which are already defined as
023 * standard (faceted) search parameters:
024 * <ul>
025 * <li>limit</li>
026 * <li>offset</li>
027 * <li>q</li>
028 * <li>hl</li>
029 * <li>language</li>
030 * <li>facet</li>
031 * </ul>
032 */
033@JsonDeserialize(as = OccurrenceSearchParameter.class)
034public interface SearchParameter {
035
036  String name();
037
038  Class<?> type();
039}
