016package org.gbif.api.model.common.search;
017
018import org.gbif.api.model.common.paging.Pageable;
019import org.gbif.api.model.common.paging.PagingRequest;
020
021import java.util.Arrays;
022import java.util.HashMap;
023import java.util.HashSet;
024import java.util.Map;
025import java.util.Set;
026
027/**
028 * Generic request class for search operations requesting facets.
029 * It extends a search request with a list of desired facets and optional settings.
030 */
031@SuppressWarnings("unused")
032public class FacetedSearchRequest<P extends SearchParameter> extends SearchRequest<P> {
033
034  private Set<P> facets = new HashSet<>();
035
036  private boolean multiSelectFacets;
037  private Integer facetMinCount;
038  private Integer facetLimit = 10;
039  private Integer facetOffset;
040
041  //Holds the paging configuration for each requested facet
042  private Map<P, Pageable> facetPages = new HashMap<>();
043
044  public FacetedSearchRequest() {
045  }
046
047  public FacetedSearchRequest(Pageable page) {
048    super(page);
049  }
050
051  public FacetedSearchRequest(SearchRequest<P> searchRequest) {
052    super(searchRequest);
053    setHighlight(searchRequest.isHighlight());
054    setParameters(searchRequest.getParameters());
055    setQ(searchRequest.getQ());
056  }
057
058  public FacetedSearchRequest(long offset, int limit) {
059    super(offset, limit);
060  }
061
062  /**
063   * Min count of facet to return, if the facet count is less than this number the facet won't be included.
064   */
065  public Integer getFacetMinCount() {
066    return facetMinCount;
067  }
068
069  public void setFacetMinCount(Integer facetMinCount) {
070    this.facetMinCount = facetMinCount;
071  }
072
073  /**
074   * Gets the list of requested facets by the search operation.
075   * The facets are a list of search parameters.
076   */
077  public Set<P> getFacets() {
078    return facets;
079  }
080
081  /**
082   * Sets the list of facets.
083   */
084  public void setFacets(Set<P> facets) {
085    this.facets = facets;
086  }
087
088  /**
089   * @return the multiSelectFacets
090   */
091  public boolean isMultiSelectFacets() {
092    return multiSelectFacets;
093  }
094
095  /**
096   * @param multiSelectFacets the multiSelectFacets to set
097   */
098  public void setMultiSelectFacets(boolean multiSelectFacets) {
099    this.multiSelectFacets = multiSelectFacets;
100  }
101
102  /**
103   * Page size of the facet request.
104   */
105  public Integer getFacetLimit() {
106    return facetLimit;
107  }
108
109  public void setFacetLimit(Integer facetLimit) {
110    this.facetLimit = facetLimit;
111  }
112
113  /**
114   * Holds the paging configuration for each requested facet.
115   */
116  public Map<P, Pageable> getFacetPages() {
117    return facetPages;
118  }
119
120  public void setFacetPages(Map<P, Pageable> facetPages) {
121    this.facetPages = facetPages;
122  }
123
124  /**
125   * Sets the paging setting of facet parameter.
126   */
127  public void addFacetPage(P parameter, int facetOffset, int facetLimit){
128    facetPages.put(parameter, new PagingRequest(facetOffset,facetLimit));
129  }
130
131  /**
132   * Gets the paging configuration of a facet parameter.
133   */
134  public Pageable getFacetPage(P parameter) {
135    return facetPages.get(parameter);
136  }
137
138  /**
139   * Offset of the facet request.
140   */
141  public Integer getFacetOffset() {
142    return facetOffset;
143  }
144
145  public void setFacetOffset(Integer facetOffset) {
146    this.facetOffset = facetOffset;
147  }
148
149  public void addFacets(P... facets) {
150    if (this.facets == null) {
151      this.facets = new HashSet<>(Arrays.asList(facets));
152    } else {
153      this.facets.addAll(new HashSet<>(Arrays.asList(facets)));
154    }
155  }
156}
