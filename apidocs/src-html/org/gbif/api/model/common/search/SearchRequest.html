016package org.gbif.api.model.common.search;
017
018import org.gbif.api.model.common.paging.Pageable;
019import org.gbif.api.model.common.paging.PageableBase;
020
021import java.text.DateFormat;
022import java.text.SimpleDateFormat;
023import java.util.Date;
024import java.util.HashMap;
025import java.util.HashSet;
026import java.util.Map;
027import java.util.Set;
028import java.util.StringJoiner;
029
030import static org.gbif.api.model.common.paging.PagingConstants.DEFAULT_PARAM_LIMIT;
031import static org.gbif.api.model.common.paging.PagingConstants.DEFAULT_PARAM_OFFSET;
032
033/**
034 * Generic request class for search operations. This class contains a list of parameters,
035 * a list of desired facets and paging options (page size and offset).
036 */
037@SuppressWarnings("unused")
038public class SearchRequest<P extends SearchParameter> extends PageableBase {
039
040  private Map<P, Set<String>> parameters = new HashMap<>();
041  private String q;
042  private boolean highlight;
043  private boolean spellCheck;
044  private int spellCheckCount;
045
046  /**
047   * Constructor with default paging offset & limit.
048   */
049  public SearchRequest() {
050    super(DEFAULT_PARAM_OFFSET, DEFAULT_PARAM_LIMIT);
051  }
052
053  /**
054   * Simple query constructor with default paging offset & limit.
055   *
056   * @param query string for request
057   */
058  public SearchRequest(String query) {
059    super(DEFAULT_PARAM_OFFSET, DEFAULT_PARAM_LIMIT);
060    q = query;
061  }
062
063  /**
064   * Minimal paging constructor.
065   */
066  public SearchRequest(Pageable page) {
067    super(page.getOffset(), page.getLimit());
068  }
069
070  /**
071   * Minimal paging constructor.
072   */
073  public SearchRequest(long offset, int limit) {
074    super(offset, limit);
075  }
076
077
078  /**
079   * @return true if highlighted search matches are requested
080   */
081  public boolean isHighlight() {
082    return highlight;
083  }
084
085  /**
086   * @param highlight the highlight to set
087   */
088  public void setHighlight(boolean highlight) {
089    this.highlight = highlight;
090  }
091
092  /**
093   *
094   * @return true if spellCheck search is requested
095   */
096  public boolean isSpellCheck() {
097    return spellCheck;
098  }
099
100  /**
101   * @param spellCheck the highlight to set
102   */
103  public void setSpellCheck(boolean spellCheck) {
104    this.spellCheck = spellCheck;
105  }
106
107  /**
108   *
109   * @return max number of spell check suggestions requested
110   */
111  public int getSpellCheckCount() {
112    return spellCheckCount;
113  }
114
115  /**
116   *
117   * @param spellCheckCount number of spell check suggestions
118   */
119  public void setSpellCheckCount(int spellCheckCount) {
120    this.spellCheckCount = spellCheckCount;
121  }
122
123  /**
124   * List of input parameters of the search operation.
125   * The parameters are handled as the parameter name and the string representation of its value.
126   *
127   * @return the list of parameters
128   */
129  public Map<P, Set<String>> getParameters() {
130    return parameters;
131  }
132
133  /**
134   * Sets the list of parameters.
135   */
136  public void setParameters(Map<P, Set<String>> parameters) {
137    this.parameters = parameters;
138  }
139
140  /**
141   * Query parameter.
142   *
143   * @return the q
144   */
145  public String getQ() {
146    return q;
147  }
148
149  /**
150   * @param q the q to set
151   */
152  public void setQ(String q) {
153    this.q = q;
154  }
155
156  /**
157   * Adds the specified parameter.
158   *
159   * @param parameter parameter to add values for
160   * @param values    list of values of the parameter to add
161   */
162  public void addParameter(P parameter, Iterable<String> values) {
163    if (parameters.containsKey(parameter)) {
164      Set<String> paramValues = parameters.get(parameter);
165      values.forEach(paramValues::add);
166    } else {
167      Set<String> paramValues = new HashSet<>();
168      values.forEach(paramValues::add);
169      parameters.put(parameter, paramValues);
170    }
171  }
172
173  /**
174   * Adds the specified parameter.
175   *
176   * @param parameter parameter to add values for
177   * @param values    list of values of the parameter to add
178   */
179  public void addParameter(P parameter, String... values) {
180    for (String value : values) {
181      addParameter(parameter, value);
182    }
183  }
184
185  /**
186   * Adds the specified parameter.
187   *
188   * @param parameter parameter to add values for
189   * @param value     value of the parameter to add
190   */
191  public void addParameter(P parameter, String value) {
192    if (value != null) {
193      if (parameters.containsKey(parameter)) {
194        Set<String> paramValues = parameters.get(parameter);
195        paramValues.add(value);
196      } else {
197        Set<String> paramValues = new HashSet<>();
198        paramValues.add(value);
199        parameters.put(parameter, paramValues);
200      }
201    }
202  }
203
204  /**
205   * Adds the specified long parameter.
206   *
207   * @param parameter parameter to add values for
208   * @param value     value of the parameter to add
209   */
210  public void addParameter(P parameter, long value) {
211    addParameter(parameter, String.valueOf(value));
212  }
213
214  /**
215   * Adds the specified int parameter.
216   *
217   * @param parameter parameter to add values for
218   * @param value     value of the parameter to add
219   */
220  public void addParameter(P parameter, int value) {
221    addParameter(parameter, String.valueOf(value));
222  }
223
224  /**
225   * Adds the specified double parameter.
226   *
227   * @param parameter parameter to add values for
228   * @param value     value of the parameter to add
229   */
230  public void addParameter(P parameter, double value) {
231    addParameter(parameter, String.valueOf(value));
232  }
233
234  /**
235   * Adds the specified boolean parameter.
236   *
237   * @param parameter parameter to add values for
238   * @param value     value of the parameter to add
239   */
240  public void addParameter(P parameter, boolean value) {
241    addParameter(parameter, String.valueOf(value));
242  }
243
244  /**
245   * Adds the specified parameter.
246   *
247   * @param parameter parameter to add values for
248   * @param value     enum value of the parameter to add
249   */
250  public void addParameter(P parameter, Enum<?> value) {
251    if (value != null) {
252      addParameter(parameter, value.name());
253    }
254  }
255
256  /**
257   * Adds the specified date parameter as an ISO date.
258   *
259   * @param parameter parameter to add date for
260   * @param value     date value of the parameter to add
261   */
262  public void addParameter(P parameter, Date value) {
263    if (value != null) {
264      // not thread safe, new instance
265      DateFormat iso = new SimpleDateFormat("yyyy-MM-dd");
266      addParameter(parameter, iso.format(value));
267    }
268  }
269
270  @Override
271  public String toString() {
272    return new StringJoiner(", ", SearchRequest.class.getSimpleName() + "[", "]")
273      .add("parameters=" + parameters)
274      .add("q='" + q + "'")
275      .add("highlight=" + highlight)
276      .add("spellCheck=" + spellCheck)
277      .add("spellCheckCount=" + spellCheckCount)
278      .add("offset=" + offset)
279      .add("limit=" + limit)
280      .toString();
281  }
282}
