016package org.gbif.api.model.common.search;
017
018import org.gbif.api.model.common.paging.Pageable;
019import org.gbif.api.model.common.paging.PageableBase;
020
021import java.text.DateFormat;
022import java.text.SimpleDateFormat;
023import java.util.Date;
024
025import com.google.common.collect.HashMultimap;
026import com.google.common.collect.Multimap;
027
028import static org.gbif.api.model.common.paging.PagingConstants.DEFAULT_PARAM_LIMIT;
029import static org.gbif.api.model.common.paging.PagingConstants.DEFAULT_PARAM_OFFSET;
030
031/**
032 * Generic request class for search operations. This class contains a list of parameters,
033 * a list of desired facets and paging options (page size and offset).
034 */
035public class SearchRequest<P extends SearchParameter> extends PageableBase {
036
037  private Multimap<P, String> parameters = HashMultimap.create();
038  private String q;
039  private boolean highlight;
040
041  /**
042   * Constructor with default paging offset & limit.
043   */
044  public SearchRequest() {
045    super(DEFAULT_PARAM_OFFSET, DEFAULT_PARAM_LIMIT);
046  }
047
048  /**
049   * Simple query constructor with default paging offset & limit.
050   *
051   * @param query string for request
052   */
053  public SearchRequest(String query) {
054    super(DEFAULT_PARAM_OFFSET, DEFAULT_PARAM_LIMIT);
055    q = query;
056  }
057
058  /**
059   * Minimal paging constructor.
060   */
061  public SearchRequest(Pageable page) {
062    super(page.getOffset(), page.getLimit());
063  }
064
065  /**
066   * Minimal paging constructor.
067   */
068  public SearchRequest(long offset, int limit) {
069    super(offset, limit);
070  }
071
072
073  /**
074   * @return true if highlighted search matches are requested
075   */
076  public boolean isHighlight() {
077    return highlight;
078  }
079
080  /**
081   * @param highlight the highlight to set
082   */
083  public void setHighlight(boolean highlight) {
084    this.highlight = highlight;
085  }
086
087  /**
088   * List of input parameters of the search operation.
089   * The parameters are handled as the parameter name and the string representation of its value.
090   *
091   * @return the list of parameters
092   */
093  public Multimap<P, String> getParameters() {
094    return parameters;
095  }
096
097  /**
098   * Sets the list of parameters.
099   */
100  public void setParameters(Multimap<P, String> parameters) {
101    this.parameters = parameters;
102  }
103
104  /**
105   * Query parameter.
106   *
107   * @return the q
108   */
109  public String getQ() {
110    return q;
111  }
112
113  /**
114   * @param q the q to set
115   */
116  public void setQ(String q) {
117    this.q = q;
118  }
119
120  /**
121   * Adds the specified parameter.
122   *
123   * @param parameter parameter to add values for
124   * @param values    list of values of the parameter to add
125   */
126  public void addParameter(P parameter, Iterable<String> values) {
127    for (String parameterValue : values) {
128      parameters.put(parameter, parameterValue);
129    }
130  }
131
132  /**
133   * Adds the specified parameter.
134   *
135   * @param parameter parameter to add values for
136   * @param values    list of values of the parameter to add
137   */
138  public void addParameter(P parameter, String... values) {
139    for (String value : values) {
140      addParameter(parameter, value);
141    }
142  }
143
144  /**
145   * Adds the specified parameter.
146   *
147   * @param parameter parameter to add values for
148   * @param value     value of the parameter to add
149   */
150  public void addParameter(P parameter, String value) {
151    if (value != null) {
152      parameters.put(parameter, value);
153    }
154  }
155
156  /**
157   * Adds the specified long parameter.
158   *
159   * @param parameter parameter to add values for
160   * @param value     value of the parameter to add
161   */
162  public void addParameter(P parameter, long value) {
163    parameters.put(parameter, String.valueOf(value));
164  }
165
166  /**
167   * Adds the specified int parameter.
168   *
169   * @param parameter parameter to add values for
170   * @param value     value of the parameter to add
171   */
172  public void addParameter(P parameter, int value) {
173    parameters.put(parameter, String.valueOf(value));
174  }
175
176  /**
177   * Adds the specified double parameter.
178   *
179   * @param parameter parameter to add values for
180   * @param value     value of the parameter to add
181   */
182  public void addParameter(P parameter, double value) {
183    parameters.put(parameter, String.valueOf(value));
184  }
185
186  /**
187   * Adds the specified boolean parameter.
188   *
189   * @param parameter parameter to add values for
190   * @param value     value of the parameter to add
191   */
192  public void addParameter(P parameter, boolean value) {
193    parameters.put(parameter, String.valueOf(value));
194  }
195
196  /**
197   * Adds the specified parameter.
198   *
199   * @param parameter parameter to add values for
200   * @param value     enum value of the parameter to add
201   */
202  public void addParameter(P parameter, Enum<?> value) {
203    if (value != null) {
204      parameters.put(parameter, value.name());
205    }
206  }
207
208
209  /**
210   * Adds the specified date parameter as an ISO date.
211   *
212   * @param parameter parameter to add date for
213   * @param value     date value of the parameter to add
214   */
215  public void addParameter(P parameter, Date value) {
216    if (value != null) {
217      // not thread safe, new instance
218      DateFormat iso = new SimpleDateFormat("yyyy-MM-dd");
219      parameters.put(parameter, iso.format(value));
220    }
221  }
222
223
224}
