023import java.util.Objects;
024import java.util.regex.Matcher;
025import java.util.regex.Pattern;
026
027import javax.validation.constraints.NotNull;
028
029import org.apache.commons.lang3.StringUtils;
030import org.slf4j.Logger;
031import org.slf4j.LoggerFactory;
032
033import com.fasterxml.jackson.core.JsonGenerator;
034import com.fasterxml.jackson.core.JsonParser;
035import com.fasterxml.jackson.databind.DeserializationContext;
036import com.fasterxml.jackson.databind.JsonDeserializer;
037import com.fasterxml.jackson.databind.JsonSerializer;
038import com.fasterxml.jackson.databind.SerializerProvider;

041
042import static org.gbif.api.util.PreconditionUtils.checkArgument;

044/**
045 * Class representing a single Digital Object Identifier (DOI) breaking it down to a prefix and suffix.
046 * For the syntax of DOI names see the <a href="http://www.doi.org/doi_handbook/2_Numbering.html#2.2">DOI Handbook</a>.
047 * All parsing is case insensitive and resulting components will all be upper cased.
048 */
049@JsonSerialize(using = DOI.DoiSerializer.class)
050@JsonDeserialize(using = DOI.DoiDeserializer.class)
051public class DOI implements Serializable {
052
053  private static final Logger LOG = LoggerFactory.getLogger(DOI.class);
054
055  /**
056   * Encoding to create URLs.
057   */
058  private static final String CHAR_ENCODING = "UTF-8";
059
060  /**
061   * The DOI prefix registered with DataCite to be used by GBIF issued production DOIs.
062   */
063  public static final String GBIF_PREFIX = "10.15468";
064
065  /**
066   * A DOI prefix provided by DataCite to be used in tests.
067   */
068  public static final String TEST_PREFIX = "10.21373";
069
070  private static final Pattern HTTP = Pattern.compile("^https?://(dx\\.)?doi\\.org/"
071    + "(urn:)?(doi:)?", Pattern.CASE_INSENSITIVE);
072  private static final Pattern PARSER = Pattern.compile("^(?:urn:)?(?:doi:)?"           // optional
073    + "(10(?:\\.[0-9]+)+)"
074    + "/(.+)$", Pattern.CASE_INSENSITIVE);
075
076  private static final String RESOLVER = "https://doi.org/";
077  private static final String SCHEME = "doi:";
078  private String prefix;
079  private String suffix;
080
081  /**
082   * Returns true only if the source can be parsed into a DOI.
083   */
084  public static boolean isParsable(String source) {
085    if (StringUtils.isNotEmpty(source)) {
086      try {
087        return PARSER.matcher(decodeUrl(source)).find();
088      } catch (IllegalArgumentException iaEx) {
089        LOG.debug("Can not decode URL from the following DOI: {}", source);
090      }
091    }
092    return false;
093  }
094
095  /**
096   * Do not use this constructor.
097   * Required by JAXB to marshall this object without introducing adapters.
098   */
099  public DOI() {
100  }
101
102  /**
103   * Parses a simple DOI string of various forms incl URN, URL or plain DOI names.
104   *
105   * @param doi the full simple DOI string
106   * @throws java.lang.IllegalArgumentException if invalid DOI string is passed
107   */
108  public DOI(String doi) {
109    Objects.requireNonNull(doi, "DOI required");
110    Matcher m = PARSER.matcher(decodeUrl(doi));
111    if (m.find()) {
112      this.prefix = m.group(1).toLowerCase();
113      this.suffix = m.group(2).toLowerCase();
114    } else {
115      throw new IllegalArgumentException(doi + " is not a valid DOI");
116    }
117  }
118
119  /**
120   * Parses a simple DOI string of various forms incl URN, URL or plain DOI names.
121   *
122   * @param prefix a simple DOI prefix starting with 10.
123   * @param suffix arbitrary suffix part of the DOI
124   * @throws java.lang.IllegalArgumentException if invalid DOI prefix is given
125   */
126  public DOI(String prefix, String suffix) {
127    this.prefix = Objects.requireNonNull(prefix, "DOI prefix required").toLowerCase();
128    checkArgument(prefix.startsWith("10."));
129    this.suffix = Objects.requireNonNull(suffix, "DOI suffix required").toLowerCase();
130  }
131
132  /**
133   * If the doi is encoded as a URL this method strips the resolver and decodes the URL encoded string entities.
134   *
135   * @param doi not null doi represented as a String
136   * @return the path part if the doi is a URL otherwise the doi is returned as is.
137   * @throws IllegalArgumentException
138   */
139  private static String decodeUrl(@NotNull String doi) {
140    Matcher m = HTTP.matcher(doi);
141    if (m.find()) {
142      // strip resolver incl potentially starting paths using a badly encoded urn:doi
143      // (the colon would need to be encoded in a proper URL)
144      doi = m.replaceFirst("");
145      // now decode the URL path, we cannot possibly have query parameters or anchors as the DOIs encoded as a URL
146      // will just be a path
147      try {
148        return URI.create(URLEncoder.encode(doi, CHAR_ENCODING)).getPath();
149      } catch (UnsupportedEncodingException e) {
150        throw new IllegalArgumentException("Unsupported DOI encoding", e);
151      }
152    }
153    return doi;
154  }
155
156  public String getPrefix() {
157    return prefix;
158  }
159
160  public void setPrefix(String prefix) {
161    this.prefix = prefix;
162  }
163
164  public String getSuffix() {
165    return suffix;
166  }
167
168  public void setSuffix(String suffix) {
169    this.suffix = suffix;
170  }
171
172  /**
173   * See <a href="http://www.doi.org/doi_handbook/2_Numbering.html#2.6">DOI Handbook, Visual presentation and other representation of DOI names</a>.
174   *
175   * @return the resolved DOI using https://doi.org/
176   * @throws IllegalStateException if the encoding of the DOI is not supported
177   */
178  public URI getUrl() {
179    try {
180      return URI.create(RESOLVER + prefix + '/' + URLEncoder.encode(suffix, CHAR_ENCODING));
181    } catch (UnsupportedEncodingException e) {
182      throw new IllegalStateException("Unsupported DOI encoding", e);
183    }
184  }
185
186  /**
187   * @return the DOI name prefixed with "doi:", as recommended by the DOI Handbook.
188   */
189  public String getDoiString() {
190    return SCHEME + getDoiName();
191  }
192
193  /**
194   * @return the pure DOI name without any initial scheme name starting with the prefix, i.e. 10.
195   */
196  public String getDoiName() {
197    return prefix + '/' + suffix;
198  }
199
200  @Override
201  public String toString() {
202    return getDoiName();
203  }
204
205  @Override
206  public int hashCode() {
207    return Objects.hash(prefix, suffix);
208  }
209
210  @Override
211  public boolean equals(Object obj) {
212    if (this == obj) {
213      return true;
214    }
215    if (obj == null || getClass() != obj.getClass()) {
216      return false;
217    }
218    final DOI other = (DOI) obj;
219    // prefix and suffix are always upper cased so we can do simple equals here
220    return Objects.equals(this.prefix, other.prefix) && Objects.equals(this.suffix, other.suffix);
221  }
222
223  public static class DoiSerializer extends JsonSerializer<DOI> {
224
225    @Override
226    public void serialize(DOI value, JsonGenerator gen, SerializerProvider serializers) throws IOException {
227      gen.writeString(value.toString());
228    }
229  }
230
231  public static class DoiDeserializer extends JsonDeserializer<DOI> {
232
233    @Override
234    public DOI deserialize(JsonParser p, DeserializationContext ctxt) throws IOException {
235      if (p != null && p.getTextLength() > 0) {
236        return new DOI(p.getText());
237      }
238      return null;
239    }
240  }
