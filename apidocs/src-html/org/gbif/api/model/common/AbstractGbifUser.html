016package org.gbif.api.model.common;
017
018import org.gbif.api.vocabulary.UserRole;
019
020import java.util.Date;
021import java.util.HashMap;
022import java.util.HashSet;
023import java.util.Map;
024import java.util.Objects;
025import java.util.Set;
026import java.util.StringJoiner;
027import java.util.stream.Collectors;
028import java.util.stream.Stream;
029
030import javax.validation.constraints.NotNull;
031import javax.validation.constraints.Pattern;
032import javax.validation.constraints.Size;
033
034import com.fasterxml.jackson.annotation.JsonIgnore;
035
036/**
037 * An abstract GBIF user account.
038 * The main purpose of this abstraction is to let subclasses handle key and password information only if required.
039 * By doing so, it is possible to have classes working for user information without having to carry those
040 * information around.
041 */
042public abstract class AbstractGbifUser {
043  protected static final String EMAIL_PATTERN =
044    "^[_A-Za-z0-9-\\+]+(\\.[_A-Za-z0-9-]+)*@[A-Za-z0-9-]+(\\.[A-Za-z0-9-]+)*(\\.[A-Za-z]{2,})$";
045
046  protected String userName;
047  protected String firstName;
048  protected String lastName;
049  protected String email;
050  protected Set<UserRole> roles = new HashSet<>();
051
052//  //country of the user (user for stats)
053//  protected Country country;
054
055  protected Map<String, String> settings = new HashMap<>();
056
057  //settings that the user will not set directly
058  protected Map<String, String> systemSettings = new HashMap<>();
059
060  protected Date deleted;
061
062  @NotNull
063  @Pattern(regexp = EMAIL_PATTERN)
064  public String getEmail() {
065    return email;
066  }
067
068  public void setEmail(String email) {
069    this.email = email;
070  }
071
072  /**
073   * The unique, immutable drupal user account name.
074   * This name should be used for referring to a user.
075   * The account name is made of ASCII lower case alphanumerics, underscore, dash or dots and is in particular void
076   * of whitespace.
077   */
078  @NotNull
079  @Pattern(regexp = "^[a-z0-9_.-]+$")
080  @Size(min = 3, max = 64)
081  public String getUserName() {
082    return userName;
083  }
084
085  public void setUserName(String userName) {
086    this.userName = userName;
087  }
088
089  /**
090   * @return the first name of a person
091   */
092  public String getFirstName() {
093    return firstName;
094  }
095
096  public void setFirstName(String firstName) {
097    this.firstName = firstName;
098  }
099
100  /**
101   * @return the last name of the user
102   */
103  public String getLastName() {
104    return lastName;
105  }
106
107  public void setLastName(String lastName) {
108    this.lastName = lastName;
109  }
110
111  /**
112   * @return the first and last name of the user concatenated with a space
113   */
114  @JsonIgnore
115  public String getName() {
116    return Stream.of(firstName, lastName)
117      .filter(Objects::nonNull)
118      .collect(Collectors.joining(" "));
119  }
120
121  @NotNull
122  public Set<UserRole> getRoles() {
123    return roles;
124  }
125
126  public void setRoles(Set<UserRole> roles) {
127    this.roles = roles;
128  }
129
130  public void addRole(UserRole role) {
131    roles.add(role);
132  }
133
134  /**
135   * Checks if the user has the given user role.
136   *
137   * @param role
138   * @return true if the user has the requested role
139   */
140  public boolean hasRole(UserRole role) {
141    return role != null && roles.contains(role);
142  }
143
144  /**
145   * Sets the settings object, setting an empty map if null is provided.
146   */
147  public void setSettings(Map<String, String> settings) {
148    // safeguard against misuse to avoid NPE
149    this.settings = settings == null ? new HashMap<>() : settings;
150  }
151
152  /**
153   * Gets the settings which may be empty but never null.
154   *
155   * @return
156   */
157  @NotNull
158  public Map<String, String> getSettings() {
159    return settings;
160  }
161
162  /**
163   * Sets the settings object, setting an empty map if null is provided.
164   */
165  public void setSystemSettings(Map<String, String> systemSettings) {
166    // safeguard against misuse to avoid NPE
167    this.systemSettings = systemSettings == null ? new HashMap<>() : systemSettings;
168  }
169
170  /**
171   * Gets the settings which may be empty but never null.
172   *
173   * @return
174   */
175  @NotNull
176  public Map<String, String> getSystemSettings() {
177    return systemSettings;
178  }
179
180  public Date getDeleted() {
181    return deleted;
182  }
183
184  public void setDeleted(Date deleted) {
185    this.deleted = deleted;
186  }
187
188  @Override
189  public boolean equals(Object o) {
190    if (this == o) {
191      return true;
192    }
193    if (o == null || getClass() != o.getClass()) {
194      return false;
195    }
196    AbstractGbifUser that = (AbstractGbifUser) o;
197    return Objects.equals(userName, that.userName) &&
198      Objects.equals(firstName, that.firstName) &&
199      Objects.equals(lastName, that.lastName) &&
200      Objects.equals(email, that.email) &&
201      Objects.equals(roles, that.roles) &&
202      Objects.equals(settings, that.settings) &&
203      Objects.equals(systemSettings, that.systemSettings) &&
204      Objects.equals(deleted, that.deleted);
205  }
206
207  @Override
208  public int hashCode() {
209    return Objects
210      .hash(userName, firstName, lastName, email, roles, settings, systemSettings, deleted);
211  }
212
213  @Override
214  public String toString() {
215    return new StringJoiner(", ", AbstractGbifUser.class.getSimpleName() + "[", "]")
216      .add("userName='" + userName + "'")
217      .add("firstName='" + firstName + "'")
218      .add("lastName='" + lastName + "'")
219      .add("email='" + email + "'")
220      .add("roles=" + roles)
221      .add("settings=" + settings)
222      .add("systemSettings=" + systemSettings)
223      .add("deleted=" + deleted)
224      .toString();
225  }
226}
