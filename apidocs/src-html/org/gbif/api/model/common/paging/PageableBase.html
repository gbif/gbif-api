016package org.gbif.api.model.common.paging;
017
018import java.util.Objects;
019import java.util.StringJoiner;
020
021import static org.gbif.api.model.common.paging.PagingConstants.DEFAULT_PARAM_LIMIT;
022import static org.gbif.api.model.common.paging.PagingConstants.DEFAULT_PARAM_OFFSET;
023
024/**
025 * Generically is a class that contains attributes used by operations that are aware of pagination.
026 * Its properties are final to allow calculations in other setter methods.
027 */
028public class PageableBase implements Pageable {
029
030  protected long offset;
031  protected int limit;
032
033  /**
034   * Default constructor with default paging values.
035   */
036  protected PageableBase() {
037    offset = DEFAULT_PARAM_OFFSET;
038    limit = DEFAULT_PARAM_LIMIT;
039  }
040
041  /**
042   * Full constructor based on a request.
043   *
044   * @throws IllegalArgumentException if negative offset or limit.
045   */
046  protected PageableBase(Pageable page) {
047    setOffset(page.getOffset());
048    setLimit(page.getLimit());
049  }
050
051  /**
052   * Full constructor.
053   *
054   * @throws IllegalArgumentException if negative offset or limit.
055   */
056  protected PageableBase(long offset, int limit) {
057    setOffset(offset);
058    setLimit(limit);
059  }
060
061  /**
062   * Total of rows that are returned.
063   */
064  @Override
065  public int getLimit() {
066    return limit;
067  }
068
069  /**
070   * @param limit the non negative limit to set
071   *
072   * @throws IllegalArgumentException if negative
073   */
074  public void setLimit(int limit) {
075    if (limit < 0) {
076      throw new IllegalArgumentException("Limit cannot be negative");
077    }
078    this.limit = limit;
079  }
080
081  /**
082   * Defines how many items to skip before beginning to return rows.
083   */
084  @Override
085  public long getOffset() {
086    return offset;
087  }
088
089  /**
090   * @param offset the non negative offset to set
091   *
092   * @throws IllegalArgumentException if negative
093   */
094  public void setOffset(long offset) {
095    if (offset < 0) {
096      throw new IllegalArgumentException("Offset cannot be negative");
097    }
098    this.offset = offset;
099  }
100
101  /**
102   * Adds to existing offset, setting offset to zero if it would be negative.
103   *
104   * @param offsetDiff to be added to existing offset.
105   */
106  public void addOffset(long offsetDiff) {
107    offset += offsetDiff;
108    if (offset < 0) {
109      offset = 0;
110    }
111  }
112
113  /**
114   * Utility method to copy paging values.
115   */
116  public void copyPagingValues(Pageable pageable) {
117    limit = pageable.getLimit();
118    offset = pageable.getOffset();
119  }
120
121  @Override
122  public boolean equals(Object o) {
123    if (this == o) {
124      return true;
125    }
126    if (o == null || getClass() != o.getClass()) {
127      return false;
128    }
129    PageableBase that = (PageableBase) o;
130    return offset == that.offset &&
131      limit == that.limit;
132  }
133
134  @Override
135  public int hashCode() {
136    return Objects.hash(offset, limit);
137  }
138
139  @Override
140  public String toString() {
141    return new StringJoiner(", ", PageableBase.class.getSimpleName() + "[", "]")
142      .add("offset=" + offset)
143      .add("limit=" + limit)
144      .toString();
145  }
146}
