016package org.gbif.api.model.common;
017
018import org.gbif.api.vocabulary.UserRole;
019
020import java.util.Date;
021import java.util.Set;
022import javax.annotation.Nullable;
023import javax.validation.constraints.NotNull;
024import javax.validation.constraints.Pattern;
025import javax.validation.constraints.Size;
026
027import com.google.common.base.Objects;
028import com.google.common.collect.Sets;
029import org.codehaus.jackson.annotate.JsonIgnore;
030
031/**
032 * A GBIF user account registered in the drupal user database.
033 */
034public class User {
035
036  private static final String EMAIL_PATTERN =
037                "^[_A-Za-z0-9-\\+]+(\\.[_A-Za-z0-9-]+)*@"
038                + "[A-Za-z0-9-]+(\\.[A-Za-z0-9]+)*(\\.[A-Za-z]{2,})$";
039
040  private Integer key;
041  private String userName;
042  private String firstName;
043  private String lastName;
044  private String email;
045  private String passwordHash;
046  private Set<UserRole> roles = Sets.newHashSet();
047  private Date lastLogin;
048
049  @NotNull
050  public Integer getKey() {
051    return key;
052  }
053
054  public void setKey(Integer key) {
055    this.key = key;
056  }
057
058  @NotNull
059  @Pattern(regexp = EMAIL_PATTERN)
060  public String getEmail() {
061    return email;
062  }
063
064  public void setEmail(String email) {
065    this.email = email;
066  }
067
068  @Nullable
069  public Date getLastLogin() {
070    return lastLogin;
071  }
072
073  public void setLastLogin(Date lastLogin) {
074    this.lastLogin = lastLogin;
075  }
076
077
078  /**
079   * The unique, immutable drupal user account name.
080   * This name should be used for referring to a user.
081   * The account name is made of ASCII lower case alphanumerics, underscore, dash or dots and is in particular void
082   * of whitespace.
083   */
084  @NotNull
085  @Pattern(regexp = "^[a-z0-9_.-]+$")
086  @Size(min = 3, max = 64)
087  public String getUserName() {
088    return userName;
089  }
090
091  public void setUserName(String userName) {
092    this.userName = userName;
093  }
094
095  /**
096   * @return the first name of a person
097   */
098  @NotNull
099  public String getFirstName() {
100    return firstName;
101  }
102
103  public void setFirstName(String firstName) {
104    this.firstName = firstName;
105  }
106
107  /**
108   * @return the last name of the user
109   */
110  @NotNull
111  public String getLastName() {
112    return lastName;
113  }
114
115  public void setLastName(String lastName) {
116    this.lastName = lastName;
117  }
118
119  /**
120   * @return the first and last name of the user concatenated with a space
121   */
122  @NotNull
123  @JsonIgnore
124  public String getName() {
125    return firstName + " " + lastName;
126  }
127
128  /**
129   * @return the drupal hashed version of the user password.
130   */
131  public String getPasswordHash() {
132    return passwordHash;
133  }
134
135  public void setPasswordHash(String passwordHash) {
136    this.passwordHash = passwordHash;
137  }
138
139  @NotNull
140  public Set<UserRole> getRoles() {
141    return roles;
142  }
143
144  public void setRoles(Set<UserRole> roles) {
145    this.roles = roles;
146  }
147
148  public void addRole(UserRole role) {
149    roles.add(role);
150  }
151
152  /**
153   * Checks if the user has the given user role.
154   * @param role
155   *
156   * @return true if the user has the requested role
157   */
158  public boolean hasRole(UserRole role) {
159    if (role != null && roles.contains(role)) {
160      return true;
161    }
162    return false;
163  }
164
165  public boolean isAdmin() {
166    return roles.contains(UserRole.ADMIN);
167  }
168
169  @Override
170  public boolean equals(Object obj) {
171    if (this == obj) {
172      return true;
173    }
174    if (!(obj instanceof User)) {
175      return false;
176    }
177
178    User that = (User) obj;
179    return Objects.equal(this.key, that.key)
180           && Objects.equal(this.userName, that.userName)
181           && Objects.equal(this.firstName, that.firstName)
182           && Objects.equal(this.lastName, that.lastName)
183           && Objects.equal(this.email, that.email)
184           && Objects.equal(this.roles, that.roles)
185           && Objects.equal(this.lastLogin, that.lastLogin)
186           && Objects.equal(this.passwordHash, that.passwordHash);
187  }
188
189  @Override
190  public int hashCode() {
191    return Objects.hashCode(key, userName, firstName, lastName, email, roles, lastLogin, passwordHash);
192  }
193
194  @Override
195  public String toString() {
196    return Objects.toStringHelper(this)
197      .add("key", key)
198      .add("accountName", userName)
199      .add("firstName", firstName)
200      .add("lastName", lastName)
201      .add("email", email)
202      .add("roles", roles)
203      .add("lastLogin", lastLogin)
204      .add("passwordHash", passwordHash)
205      .toString();
206  }
207
208}
