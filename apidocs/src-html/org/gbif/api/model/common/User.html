016package org.gbif.api.model.common;
017
018import org.gbif.api.vocabulary.UserRole;
019
020import java.util.Date;
021import java.util.HashMap;
022import java.util.HashSet;
023import java.util.Map;
024import java.util.Objects;
025import java.util.Set;
026import java.util.StringJoiner;
027
028import javax.annotation.Nullable;
029import javax.validation.constraints.NotNull;
030import javax.validation.constraints.Pattern;
031import javax.validation.constraints.Size;
032
033import com.fasterxml.jackson.annotation.JsonIgnore;
034
035/**
036 * A GBIF user account registered in the Drupal user database.
037 * @deprecated replaced by {@link GbifUser}
038 */
039@Deprecated
040public class User {
041
042  private static final String EMAIL_PATTERN =
043    "^[_A-Za-z0-9-\\+]+(\\.[_A-Za-z0-9-]+)*@"
044      + "[A-Za-z0-9-]+(\\.[A-Za-z0-9]+)*(\\.[A-Za-z]{2,})$";
045
046  private Integer key;
047  private String userName;
048  private String firstName;
049  private String lastName;
050  private String email;
051  private String passwordHash;
052  private Set<UserRole> roles = new HashSet<>();
053  private Date lastLogin;
054  // Note: Settings was introduced in the system developed to replace Drupal
055  private Map<String, String> settings = new HashMap<>();
056
057  @NotNull
058  public Integer getKey() {
059    return key;
060  }
061
062  public void setKey(Integer key) {
063    this.key = key;
064  }
065
066  @NotNull
067  @Pattern(regexp = EMAIL_PATTERN)
068  public String getEmail() {
069    return email;
070  }
071
072  public void setEmail(String email) {
073    this.email = email;
074  }
075
076  @Nullable
077  public Date getLastLogin() {
078    return lastLogin;
079  }
080
081  public void setLastLogin(Date lastLogin) {
082    this.lastLogin = lastLogin;
083  }
084
085  /**
086   * The unique, immutable drupal user account name.
087   * This name should be used for referring to a user.
088   * The account name is made of ASCII lower case alphanumerics, underscore, dash or dots and is in particular void
089   * of whitespace.
090   */
091  @NotNull
092  @Pattern(regexp = "^[a-z0-9_.-]+$")
093  @Size(min = 3, max = 64)
094  public String getUserName() {
095    return userName;
096  }
097
098  public void setUserName(String userName) {
099    this.userName = userName;
100  }
101
102  /**
103   * @return the first name of a person
104   */
105  @NotNull
106  public String getFirstName() {
107    return firstName;
108  }
109
110  public void setFirstName(String firstName) {
111    this.firstName = firstName;
112  }
113
114  /**
115   * @return the last name of the user
116   */
117  @NotNull
118  public String getLastName() {
119    return lastName;
120  }
121
122  public void setLastName(String lastName) {
123    this.lastName = lastName;
124  }
125
126  /**
127   * @return the first and last name of the user concatenated with a space
128   */
129  @NotNull
130  @JsonIgnore
131  public String getName() {
132    return firstName + " " + lastName;
133  }
134
135  /**
136   * @return the drupal hashed version of the user password.
137   */
138  public String getPasswordHash() {
139    return passwordHash;
140  }
141
142  public void setPasswordHash(String passwordHash) {
143    this.passwordHash = passwordHash;
144  }
145
146  @NotNull
147  public Set<UserRole> getRoles() {
148    return roles;
149  }
150
151  public void setRoles(Set<UserRole> roles) {
152    this.roles = roles;
153  }
154
155  public void addRole(UserRole role) {
156    roles.add(role);
157  }
158
159  /**
160   * Checks if the user has the given user role.
161   *
162   * @param role
163   * @return true if the user has the requested role
164   */
165  public boolean hasRole(UserRole role) {
166    if (role != null && roles.contains(role)) {
167      return true;
168    }
169    return false;
170  }
171
172  public boolean isAdmin() {
173    return roles.contains(UserRole.ADMIN);
174  }
175
176  /**
177   * Gets the settings which may be empty but never null.
178   *
179   * @return
180   */
181  @NotNull
182  public Map<String, String> getSettings() {
183    return settings;
184  }
185
186  /**
187   * Sets the settings object, setting an empty map if null is provided.
188   */
189  public void setSettings(Map<String, String> settings) {
190    // safeguard against misuse to avoid NPE
191    this.settings = settings == null ? new HashMap<>() : settings;
192  }
193
194  @Override
195  public boolean equals(Object o) {
196    if (this == o) {
197      return true;
198    }
199    if (o == null || getClass() != o.getClass()) {
200      return false;
201    }
202    User user = (User) o;
203    return Objects.equals(key, user.key) &&
204      Objects.equals(userName, user.userName) &&
205      Objects.equals(firstName, user.firstName) &&
206      Objects.equals(lastName, user.lastName) &&
207      Objects.equals(email, user.email) &&
208      Objects.equals(passwordHash, user.passwordHash) &&
209      Objects.equals(roles, user.roles) &&
210      Objects.equals(lastLogin, user.lastLogin) &&
211      Objects.equals(settings, user.settings);
212  }
213
214  @Override
215  public int hashCode() {
216    return Objects
217      .hash(key, userName, firstName, lastName, email, passwordHash, roles, lastLogin, settings);
218  }
219
220  @Override
221  public String toString() {
222    return new StringJoiner(", ", User.class.getSimpleName() + "[", "]")
223      .add("key=" + key)
224      .add("userName='" + userName + "'")
225      .add("firstName='" + firstName + "'")
226      .add("lastName='" + lastName + "'")
227      .add("email='" + email + "'")
228      .add("roles=" + roles)
229      .add("lastLogin=" + lastLogin)
230      .add("settings=" + settings)
231      .toString();
232  }
233}
