001package org.gbif.api.model.common;
002
003import java.io.IOException;
004import java.net.URI;
005import java.util.Objects;
006import java.util.regex.Matcher;
007import java.util.regex.Pattern;
008
009import com.google.common.base.Preconditions;
010import org.codehaus.jackson.JsonGenerator;
011import org.codehaus.jackson.JsonParser;
012import org.codehaus.jackson.map.DeserializationContext;
013import org.codehaus.jackson.map.JsonDeserializer;
014import org.codehaus.jackson.map.SerializerProvider;
015import org.codehaus.jackson.map.annotate.JsonDeserialize;
016import org.codehaus.jackson.map.annotate.JsonSerialize;
017import org.codehaus.jackson.map.ser.std.SerializerBase;
018
019/**
020 * Class representing a single Digital Object Identifier (DOI) breaking it down to a prefix and suffix.
021 * For the syntax of DOI names see the <a href="http://www.doi.org/doi_handbook/2_Numbering.html#2.2">DOI Handbook</a>.
022 * All parsing is case insensitive and resulting components will all be upper cased.
023 */
024@JsonSerialize(using = DOI.Serializer.class)
025@JsonDeserialize(using = DOI.Deserializer.class)
026public class DOI {
027  private static final Pattern HTTP = Pattern.compile("^https?://(dx\\.)?doi\\.org/"
028                                                      + "(urn:)?(doi:)?", Pattern.CASE_INSENSITIVE);
029  private static final Pattern PARSER = Pattern.compile("^(?:urn:)?(?:doi:)?"           // optional
030                                                        + "(10(?:\\.[0-9]+)+)"
031                                                        + "/(.+)$", Pattern.CASE_INSENSITIVE);
032
033  private static String RESOLVER = "http://doi.org/";
034  private static String SCHEME = "doi:";
035  private String prefix;
036  private String suffix;
037
038  /**
039   * Parses a simple DOI string of various forms incl URN, URL or plain DOI names.
040   * @param doi the full simple DOI string
041   * @throws java.lang.IllegalArgumentException if invalid DOI string is passed
042   */
043  public DOI(String doi) {
044    Preconditions.checkNotNull(doi, "DOI required");
045    Matcher m = PARSER.matcher(decodeUrl(doi));
046    if (m.find()) {
047      this.prefix = m.group(1).toUpperCase();
048      this.suffix = m.group(2).toUpperCase();
049    } else {
050      throw new IllegalArgumentException(doi + " is not a valid DOI");
051    }
052  }
053
054  /**
055   * Parses a simple DOI string of various forms incl URN, URL or plain DOI names.
056   * @param prefix a simple DOI prefix starting with 10.
057   * @param suffix arbitrary suffix part of the DOI
058   * @throws java.lang.IllegalArgumentException if invalid DOI prefix is given
059   */
060  public DOI(String prefix, String suffix) {
061    this.prefix = Preconditions.checkNotNull(prefix, "DOI prefix required").toUpperCase();
062    Preconditions.checkArgument(prefix.startsWith("10."));
063    this.suffix = Preconditions.checkNotNull(suffix, "DOI suffix required").toUpperCase();
064  }
065
066  /**
067   * If the doi is encoded as a URL this method strips the resolver and decodes the URL encoded string entities.
068   */
069  private static String decodeUrl(String doi) {
070    Matcher m = HTTP.matcher(doi);
071    if (m.find()) {
072      // strip resolver incl potentially starting paths using a badly encoded urn:doi
073      // (the colon would need to be encoded in a proper URL)
074      doi = m.replaceFirst("");
075      // now decode the URL path, we cannot possibly have query parameters or anchors as the DOIs encoded as a URL
076      // will just be a path
077      return URI.create(doi).getPath();
078    }
079    return doi;
080  }
081
082  public String getPrefix() {
083    return prefix;
084  }
085
086  public void setPrefix(String prefix) {
087    this.prefix = prefix;
088  }
089
090  public String getSuffix() {
091    return suffix;
092  }
093
094  public void setSuffix(String suffix) {
095    this.suffix = suffix;
096  }
097
098  /**
099   * See <a href="http://www.doi.org/doi_handbook/2_Numbering.html#2.6">DOI Hanbook, Visual presentation and other representation of DOI names</a>.
100   * @return the resolved DOI using http://dx.doi.org
101   */
102  public URI getUrl() {
103    return URI.create(RESOLVER + prefix + '/' + suffix);
104  }
105
106  @Override
107  public String toString() {
108    return SCHEME + prefix + '/' + suffix;
109  }
110
111  @Override
112  public int hashCode() {
113    return Objects.hash(prefix, suffix);
114  }
115
116  @Override
117  public boolean equals(Object obj) {
118    if (this == obj) {
119      return true;
120    }
121    if (obj == null || getClass() != obj.getClass()) {
122      return false;
123    }
124    final DOI other = (DOI) obj;
125    // prefix and suffix are always upper cased so we can do simple equals here
126    return Objects.equals(this.prefix, other.prefix) && Objects.equals(this.suffix, other.suffix);
127  }
128
129
130
131  /**
132   * Serializes a DOI as doi name with a doi: scheme.
133   * For example doi:10.1038/nature.2014.16460
134   */
135  public static class Serializer extends SerializerBase<DOI> {
136
137    public Serializer() {
138      super(DOI.class);
139    }
140
141    @Override
142    public void serialize(DOI value, JsonGenerator jgen, SerializerProvider provider) throws IOException {
143      jgen.writeString(value.toString());
144    }
145  }
146
147  /**
148   * Deserializes a DOI from various string based formats.
149   * See DOI constructor for details.
150   */
151  public static class Deserializer extends JsonDeserializer<DOI> {
152
153    @Override
154    public DOI deserialize(JsonParser jp, DeserializationContext ctxt) throws IOException {
155      if (jp != null && jp.getTextLength() > 0) {
156        return new DOI(jp.getText());
157      }
158      return null;
159    }
160  }
