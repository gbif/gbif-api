001package org.gbif.api.model.common;
002
003import java.io.IOException;
004import java.net.URI;
005import java.util.Objects;
006import java.util.regex.Matcher;
007import java.util.regex.Pattern;
008
009import com.google.common.base.Preconditions;
010import com.google.common.base.Strings;
011import org.codehaus.jackson.JsonGenerator;
012import org.codehaus.jackson.JsonParser;
013import org.codehaus.jackson.map.DeserializationContext;
014import org.codehaus.jackson.map.JsonDeserializer;
015import org.codehaus.jackson.map.SerializerProvider;
016import org.codehaus.jackson.map.annotate.JsonDeserialize;
017import org.codehaus.jackson.map.annotate.JsonSerialize;
018import org.codehaus.jackson.map.ser.std.SerializerBase;
019
020/**
021 * Class representing a single Digital Object Identifier (DOI) breaking it down to a prefix and suffix.
022 * For the syntax of DOI names see the <a href="http://www.doi.org/doi_handbook/2_Numbering.html#2.2">DOI Handbook</a>.
023 * All parsing is case insensitive and resulting components will all be upper cased.
024 */
025@JsonSerialize(using = DOI.Serializer.class)
026@JsonDeserialize(using = DOI.Deserializer.class)
027public class DOI {
028
029  /**
030   * The DOI prefix registered with DataCite to be used by GBIF issued production DOIs.
031   */
032  public static final String GBIF_PREFIX = "10.15468";
033
034  /**
035   * A DOI prefix provided by DataCite to be used in tests.
036   */
037  public static final String TEST_PREFIX = "10.5072";
038
039  private static final Pattern HTTP = Pattern.compile("^https?://(dx\\.)?doi\\.org/"
040                                                      + "(urn:)?(doi:)?", Pattern.CASE_INSENSITIVE);
041  private static final Pattern PARSER = Pattern.compile("^(?:urn:)?(?:doi:)?"           // optional
042                                                        + "(10(?:\\.[0-9]+)+)"
043                                                        + "/(.+)$", Pattern.CASE_INSENSITIVE);
044
045  private static String RESOLVER = "http://doi.org/";
046  private static String SCHEME = "doi:";
047  private String prefix;
048  private String suffix;
049
050  /**
051   * Returns true only if the source can be parsed into a DOI.
052   */
053  public static boolean isParsable(String source) {
054    if (!Strings.isNullOrEmpty(source)) {
055      return PARSER.matcher(decodeUrl(source)).find();
056    }
057    return false;
058  }
059
060  /**
061   * Do not use this constructor.
062   * Required by JAXB to marshall this object without introducing adapters.
063   */
064  public DOI() {
065  }
066
067  /**
068   * Parses a simple DOI string of various forms incl URN, URL or plain DOI names.
069   * @param doi the full simple DOI string
070   * @throws java.lang.IllegalArgumentException if invalid DOI string is passed
071   */
072  public DOI(String doi) {
073    Preconditions.checkNotNull(doi, "DOI required");
074    Matcher m = PARSER.matcher(decodeUrl(doi));
075    if (m.find()) {
076      this.prefix = m.group(1).toLowerCase();
077      this.suffix = m.group(2).toLowerCase();
078    } else {
079      throw new IllegalArgumentException(doi + " is not a valid DOI");
080    }
081  }
082
083  /**
084   * Parses a simple DOI string of various forms incl URN, URL or plain DOI names.
085   * @param prefix a simple DOI prefix starting with 10.
086   * @param suffix arbitrary suffix part of the DOI
087   * @throws java.lang.IllegalArgumentException if invalid DOI prefix is given
088   */
089  public DOI(String prefix, String suffix) {
090    this.prefix = Preconditions.checkNotNull(prefix, "DOI prefix required").toLowerCase();
091    Preconditions.checkArgument(prefix.startsWith("10."));
092    this.suffix = Preconditions.checkNotNull(suffix, "DOI suffix required").toLowerCase();
093  }
094
095  /**
096   * If the doi is encoded as a URL this method strips the resolver and decodes the URL encoded string entities.
097   */
098  private static String decodeUrl(String doi) {
099    Matcher m = HTTP.matcher(doi);
100    if (m.find()) {
101      // strip resolver incl potentially starting paths using a badly encoded urn:doi
102      // (the colon would need to be encoded in a proper URL)
103      doi = m.replaceFirst("");
104      // now decode the URL path, we cannot possibly have query parameters or anchors as the DOIs encoded as a URL
105      // will just be a path
106      return URI.create(doi).getPath();
107    }
108    return doi;
109  }
110
111  public String getPrefix() {
112    return prefix;
113  }
114
115  public void setPrefix(String prefix) {
116    this.prefix = prefix;
117  }
118
119  public String getSuffix() {
120    return suffix;
121  }
122
123  public void setSuffix(String suffix) {
124    this.suffix = suffix;
125  }
126
127  /**
128   * See <a href="http://www.doi.org/doi_handbook/2_Numbering.html#2.6">DOI Hanbook, Visual presentation and other representation of DOI names</a>.
129   * @return the resolved DOI using http://dx.doi.org
130   */
131  public URI getUrl() {
132    return URI.create(RESOLVER + prefix + '/' + suffix);
133  }
134
135  /**
136   * @return the pure DOI name without any initial scheme name starting with the prefix, i.e. 10.
137   */
138  public String getDoiName() {
139    return prefix + '/' + suffix;
140  }
141
142  @Override
143  public String toString() {
144    return SCHEME + prefix + '/' + suffix;
145  }
146
147  @Override
148  public int hashCode() {
149    return Objects.hash(prefix, suffix);
150  }
151
152  @Override
153  public boolean equals(Object obj) {
154    if (this == obj) {
155      return true;
156    }
157    if (obj == null || getClass() != obj.getClass()) {
158      return false;
159    }
160    final DOI other = (DOI) obj;
161    // prefix and suffix are always upper cased so we can do simple equals here
162    return Objects.equals(this.prefix, other.prefix) && Objects.equals(this.suffix, other.suffix);
163  }
164
165
166
167  /**
168   * Serializes a DOI as doi name with a doi: scheme.
169   * For example doi:10.1038/nature.2014.16460
170   */
171  public static class Serializer extends SerializerBase<DOI> {
172
173    public Serializer() {
174      super(DOI.class);
175    }
176
177    @Override
178    public void serialize(DOI value, JsonGenerator jgen, SerializerProvider provider) throws IOException {
179      jgen.writeString(value.toString());
180    }
181  }
182
183  /**
184   * Deserializes a DOI from various string based formats.
185   * See DOI constructor for details.
186   */
187  public static class Deserializer extends JsonDeserializer<DOI> {
188
189    @Override
190    public DOI deserialize(JsonParser jp, DeserializationContext ctxt) throws IOException {
191      if (jp != null && jp.getTextLength() > 0) {
192        return new DOI(jp.getText());
193      }
194      return null;
195    }
196  }
