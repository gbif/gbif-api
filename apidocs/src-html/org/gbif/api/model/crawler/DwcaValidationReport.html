001package org.gbif.api.model.crawler;
002
003import java.util.List;
004import java.util.UUID;
005import javax.annotation.concurrent.Immutable;
006import javax.annotation.concurrent.ThreadSafe;
007
008import com.google.common.base.Joiner;
009import com.google.common.base.Objects;
010import com.google.common.base.Strings;
011import com.google.common.collect.Lists;
012import org.codehaus.jackson.annotate.JsonCreator;
013import org.codehaus.jackson.annotate.JsonProperty;
014
015import static com.google.common.base.Preconditions.checkNotNull;
016
017/**
018 * A report of the state of a dwc archive, mainly testing the existance and use of unique identifiers.
019 * It combines checklist and occurrence reports.
020 */
021@Immutable
022@ThreadSafe
023public class DwcaValidationReport {
024  private final UUID datasetKey;
025
026  private final OccurrenceValidationReport occurrenceReport;
027
028  private final ChecklistValidationReport checklistReport;
029
030  private final String invalidationReason;
031
032  public boolean isValid() {
033    return invalidationReason == null
034      && (occurrenceReport == null || occurrenceReport.isValid())
035      && (checklistReport == null || checklistReport.isValid());
036  }
037
038  public DwcaValidationReport(@JsonProperty("datasetKey") UUID datasetKey, OccurrenceValidationReport occurrenceReport,
039    ChecklistValidationReport checklistReport) {
040    this.datasetKey = checkNotNull(datasetKey, "datasetKey can't be null");
041    this.occurrenceReport = checkNotNull(occurrenceReport, "occurrenceReport can't be null");
042    this.checklistReport = checkNotNull(checklistReport, "checklistReport can't be null");
043    this.invalidationReason = null;
044  }
045
046  public DwcaValidationReport(@JsonProperty("datasetKey") UUID datasetKey, OccurrenceValidationReport occurrenceReport) {
047    this.datasetKey = checkNotNull(datasetKey, "datasetKey can't be null");
048    this.occurrenceReport = checkNotNull(occurrenceReport, "occurrenceReport can't be null");
049    this.checklistReport = null;
050    this.invalidationReason = null;
051  }
052
053  public DwcaValidationReport(@JsonProperty("datasetKey") UUID datasetKey, ChecklistValidationReport checklistReport) {
054    this.datasetKey = checkNotNull(datasetKey, "datasetKey can't be null");
055    this.occurrenceReport = null;
056    this.checklistReport = checkNotNull(checklistReport, "checklistReport can't be null");
057    this.invalidationReason = null;
058  }
059
060  public DwcaValidationReport(@JsonProperty("datasetKey") UUID datasetKey, String invalidationReason) {
061    this.datasetKey = checkNotNull(datasetKey, "datasetKey can't be null");
062    this.occurrenceReport = null;
063    this.checklistReport = null;
064    this.invalidationReason = checkNotNull(invalidationReason, "invalidationReason can't be null");
065  }
066
067  public UUID getDatasetKey() {
068    return datasetKey;
069  }
070
071  public String getInvalidationReason() {
072    StringBuilder sb = new StringBuilder();
073    if (invalidationReason != null) {
074      sb.append(invalidationReason);
075    }
076    if (occurrenceReport != null && !occurrenceReport.isValid()) {
077      if (sb.length() > 1) {
078        sb.append("\n");
079      }
080      sb.append("Invalid Occurrences: ");
081      sb.append(occurrenceReport.getInvalidationReason());
082    }
083    if (checklistReport != null && !checklistReport.isValid()) {
084      if (sb.length() > 1) {
085        sb.append("\n");
086      }
087      sb.append("Invalid Checklist: ");
088      sb.append(checklistReport.getInvalidationReason());
089    }
090    return Strings.emptyToNull(sb.toString());
091  }
092
093  public OccurrenceValidationReport getOccurrenceReport() {
094    return occurrenceReport;
095  }
096
097  public ChecklistValidationReport getChecklistReport() {
098    return checklistReport;
099  }
100}
