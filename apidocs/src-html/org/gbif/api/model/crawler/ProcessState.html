001package org.gbif.api.model.crawler;
002
003/**
004 * This enum lists the state of an occurrence or checklist crawl.
005 */
006public enum ProcessState {
007
008  /**
009   * There is no data and nothing to do for further processing.
010   */
011  EMPTY,
012
013  /**
014   * Processing is currently running.
015   */
016  RUNNING,
017
018  /**
019   * Processing has finished, either successfully or with an error.
020   * See a crawls' FinishedReason for more information.
021   */
022  FINISHED
023}
