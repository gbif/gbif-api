001package org.gbif.api.model.crawler;
002
003import java.util.List;
004import java.util.UUID;
005
006import com.google.common.base.Joiner;
007import com.google.common.base.Objects;
008import com.google.common.collect.Lists;
009import org.codehaus.jackson.annotate.JsonCreator;
010import org.codehaus.jackson.annotate.JsonProperty;
011
012import static com.google.common.base.Preconditions.checkNotNull;
013
014/**
015 * The rules followed here should match the document at:
016 * http://dev.gbif.org/wiki/display/INT/Identifier+problems+and+how+to+solve+them.
017 */
018public class OccurrenceValidationReport {
019  // if the percentage of invalid triplets (eg missing catalog number) is greater than this, the archive is invalid
020  private static final double INVALID_TRIPLET_THRESHOLD = 0.25;
021
022  // the number of occurrence records checked in the validation
023  private final int checkedRecords;
024
025  // the number of triplets that were unique
026  private final int uniqueTriplets;
027
028  /**
029   * the number of triplets that were invalid (because one or more of institutionCode, collectionCode or catalogNumber
030   * were null or empty)
031   */
032  private final int recordsWithInvalidTriplets;
033
034  // the number of occurrenceIds that were unique (therefore also == the number of records with unique occurrenceId)
035  private final int uniqueOccurrenceIds;
036
037  // records that had no occurrenceId
038  private final int recordsMissingOccurrenceId;
039
040  // false if we had to stop at our memory-saving limit
041  private final boolean allRecordsChecked;
042
043  // if the archive is not valid this will hold a readable reason
044  private String invalidationReason;
045
046  // is this archive valid
047  private final boolean valid;
048
049  @JsonCreator
050  public OccurrenceValidationReport(@JsonProperty("checkedRecords") int checkedRecords,
051    @JsonProperty("uniqueTriplets") int uniqueTriplets,
052    @JsonProperty("invalidTriplets") int recordsWithInvalidTriplets,
053    @JsonProperty("uniqueOccIds") int uniqueOccurrenceIds,
054    @JsonProperty("missingOccIds") int recordsMissingOccurrenceId,
055    @JsonProperty("allRecordsChecked") boolean allRecordsChecked) {
056    this.checkedRecords = checkedRecords;
057    this.uniqueTriplets = uniqueTriplets;
058    this.recordsWithInvalidTriplets = recordsWithInvalidTriplets;
059    this.uniqueOccurrenceIds = uniqueOccurrenceIds;
060    this.recordsMissingOccurrenceId = recordsMissingOccurrenceId;
061    this.allRecordsChecked = allRecordsChecked;
062    this.valid = validate();
063  }
064
065  /**
066   * At the moment the only truly fatal conditions are:
067   * - whole archive is empty or unreadable
068   * - triplets are invalid (% invalid > than our threshold) && occIds are invalid (must be 100% coverage and unique)
069   * - any duplicate triplets && occIds are invalid
070   */
071  private boolean validate() {
072    boolean hasRecords = checkedRecords > 0;
073    double invalidRatio = hasRecords ? (double) recordsWithInvalidTriplets / checkedRecords : 0;
074    boolean invalidTripletsBelowLimit = invalidRatio <= INVALID_TRIPLET_THRESHOLD;
075    boolean hasUniqueTriplets = uniqueTriplets == checkedRecords - recordsWithInvalidTriplets;
076    boolean hasUniqueOccIds = uniqueOccurrenceIds == checkedRecords - recordsMissingOccurrenceId;
077    boolean hasGoodOccIds = uniqueOccurrenceIds == checkedRecords;
078    boolean looksValid = invalidTripletsBelowLimit && hasUniqueTriplets || hasGoodOccIds;
079
080    if (!looksValid) {
081      List<String> reasons = Lists.newArrayList();
082      if (!invalidTripletsBelowLimit) {
083        reasons.add(Math.round(100 * invalidRatio) + "% invalid triplets is > than threshold of " + Math
084          .round(100 * INVALID_TRIPLET_THRESHOLD) + '%');
085      }
086      if (!hasUniqueTriplets) {
087        reasons.add((checkedRecords - recordsWithInvalidTriplets - uniqueTriplets) + " duplicate triplets detected");
088      }
089      if (!hasGoodOccIds) {
090        if (recordsMissingOccurrenceId != 0) {
091          reasons.add(recordsMissingOccurrenceId + " records without an occurrence id (should be 0)");
092        }
093        if (!hasUniqueOccIds) {
094          reasons.add(
095            (checkedRecords - recordsMissingOccurrenceId - uniqueOccurrenceIds) + " duplicate occurrence ids detected");
096        }
097      }
098      String reason = Joiner.on("; ").join(reasons);
099      invalidationReason = "Archive invalid because [" + reason + ']';
100    }
101
102    return looksValid;
103  }
104
105  public int getCheckedRecords() {
106    return checkedRecords;
107  }
108
109  public int getUniqueTriplets() {
110    return uniqueTriplets;
111  }
112
113  public int getRecordsWithInvalidTriplets() {
114    return recordsWithInvalidTriplets;
115  }
116
117  public int getUniqueOccurrenceIds() {
118    return uniqueOccurrenceIds;
119  }
120
121  public int getRecordsMissingOccurrenceId() {
122    return recordsMissingOccurrenceId;
123  }
124
125  public boolean isAllRecordsChecked() {
126    return allRecordsChecked;
127  }
128
129  public String getInvalidationReason() {
130    return invalidationReason;
131  }
132
133  public boolean isValid() {
134    return valid;
135  }
136
137  @Override
138  public int hashCode() {
139    return Objects.hashCode(checkedRecords, uniqueTriplets, recordsWithInvalidTriplets, uniqueOccurrenceIds,
140      recordsMissingOccurrenceId, allRecordsChecked, valid);
141  }
142
143  @Override
144  public boolean equals(Object obj) {
145    if (this == obj) {
146      return true;
147    }
148    if (obj == null || getClass() != obj.getClass()) {
149      return false;
150    }
151    final OccurrenceValidationReport other = (OccurrenceValidationReport) obj;
152    return Objects.equal(this.checkedRecords, other.checkedRecords)
153           && Objects.equal(this.uniqueTriplets, other.uniqueTriplets)
154           && Objects.equal(this.recordsWithInvalidTriplets, other.recordsWithInvalidTriplets)
155           && Objects.equal(this.uniqueOccurrenceIds, other.uniqueOccurrenceIds)
156           && Objects.equal(this.recordsMissingOccurrenceId, other.recordsMissingOccurrenceId)
157           && Objects.equal(this.allRecordsChecked, other.allRecordsChecked)
158           && Objects.equal(this.valid, other.valid);
159  }
160
161  @Override
162  public String toString() {
163    return Objects.toStringHelper(this).add("checkedRecords", checkedRecords)
164      .add("uniqueTriplets", uniqueTriplets).add("recordsWithInvalidTriplets", recordsWithInvalidTriplets)
165      .add("uniqueOccurrenceIds", uniqueOccurrenceIds).add("recordsMissingOccurrenceId", recordsMissingOccurrenceId)
166      .add("allRecordsChecked", allRecordsChecked).add("invalidationReason", invalidationReason).add("valid", valid)
167      .toString();
168  }
169}
