001package org.gbif.api.model.crawler;
002
003import org.gbif.api.vocabulary.Origin;
004import org.gbif.api.vocabulary.Rank;
005
006import java.util.List;
007import java.util.Map;
008
009import com.google.common.base.Objects;
010import org.codehaus.jackson.annotate.JsonCreator;
011import org.codehaus.jackson.annotate.JsonProperty;
012
013/**
014 * Statistics of a checklist normalization result.
015 */
016public class NormalizerStats {
017
018  private final int roots;
019  private final int depth;
020  private final int synonyms;
021  private final int ignored;
022  private final Map<Origin, Integer> countByOrigin;
023  private final Map<Rank, Integer> countByRank;
024  private final List<String> cycles;
025
026  @JsonCreator
027  public NormalizerStats(@JsonProperty("roots") int roots, @JsonProperty("depth") int depth,
028    @JsonProperty("synonyms") int synonyms, @JsonProperty("ignored") int ignored,
029    @JsonProperty("countByOrigin") Map<Origin, Integer> countByOrigin,
030    @JsonProperty("countByRank") Map<Rank, Integer> countByRank, @JsonProperty("cycles") List<String> cycles) {
031    this.roots = roots;
032    this.depth = depth;
033    this.synonyms = synonyms;
034    this.ignored = ignored;
035    this.countByOrigin = countByOrigin;
036    this.countByRank = countByRank;
037    this.cycles = cycles;
038  }
039
040  /**
041   * @return list of cycles, each given as one taxonID of the loop
042   */
043  public List<String> getCycles() {
044    return cycles;
045  }
046
047  /**
048   * @return total count of name usages existing as neo nodes, both accepted and synonyms and regardless of their Origin
049   */
050  public int getCount() {
051    int total = 0;
052    for (int x : countByOrigin.values()) {
053      total += x;
054    }
055    return total;
056  }
057
058  /**
059   * @return the number of root taxa without a parent
060   */
061  public int getRoots() {
062    return roots;
063  }
064
065  /**
066   * @return maximum depth of the classification
067   */
068  public int getDepth() {
069    return depth;
070  }
071
072  /**
073   * @return the number of synonym nodes in this checklist
074   */
075  public int getSynonyms() {
076    return synonyms;
077  }
078
079  public int getIgnored() {
080    return ignored;
081  }
082
083  public int getCountByRank(Rank rank) {
084    if (countByRank.containsKey(rank)) {
085      return countByRank.get(rank);
086    }
087    return 0;
088  }
089
090  public int getCountByOrigin(Origin o) {
091    return countByOrigin.get(o);
092  }
093
094  public Map<Origin, Integer> getCountByOrigin() {
095    return countByOrigin;
096  }
097
098  public Map<Rank, Integer> getCountByRank() {
099    return countByRank;
100  }
101
102  @Override
103  public String toString() {
104    return "NormalizerStats{roots=" + roots +
105           ", depth=" + depth +
106           ", synonyms=" + synonyms +
107           ", ignored=" + ignored+
108           ", cycles=" + cycles.size() +
109           ", countByOrigin=" + countByOrigin +
110           ", countByRank=" + countByRank +
111           '}';
112  }
113
114  @Override
115  public int hashCode() {
116    return Objects.hashCode(roots, depth, synonyms, ignored, cycles, countByOrigin, countByRank);
117  }
118
119  @Override
120  public boolean equals(Object obj) {
121    if (this == obj) {
122      return true;
123    }
124    if (obj == null || getClass() != obj.getClass()) {
125      return false;
126    }
127    final NormalizerStats other = (NormalizerStats) obj;
128    return Objects.equal(this.roots, other.roots)
129           && Objects.equal(this.depth, other.depth)
130           && Objects.equal(this.synonyms, other.synonyms)
131           && Objects.equal(this.ignored, other.ignored)
132           && Objects.equal(this.cycles, other.cycles)
133           && Objects.equal(this.countByOrigin, other.countByOrigin)
134           && Objects.equal(this.countByRank, other.countByRank);
135  }
136
137}
