001package org.gbif.api.model.crawler;
002
003import org.gbif.api.vocabulary.Origin;
004import org.gbif.api.vocabulary.Rank;
005
006import java.util.List;
007import java.util.Map;
008
009import com.google.common.base.Objects;
010import org.codehaus.jackson.annotate.JsonCreator;
011import org.codehaus.jackson.annotate.JsonProperty;
012
013/**
014 * Statistics of a checklist normalization result.
015 */
016public class NormalizerStats {
017
018  private final int records;
019  private final int roots;
020  private final int depth;
021  private final int synonyms;
022  private final Map<Origin, Integer> countByOrigin;
023  private final Map<Rank, Integer> countByRank;
024  private final List<String> cycles;
025
026  @JsonCreator
027  public NormalizerStats(@JsonProperty("records") int records,
028    @JsonProperty("roots") int roots,
029    @JsonProperty("depth") int depth,
030    @JsonProperty("synonyms") int synonyms,
031    @JsonProperty("countByOrigin") Map<Origin, Integer> countByOrigin,
032    @JsonProperty("countByRank") Map<Rank, Integer> countByRank,
033    @JsonProperty("cycles") List<String> cycles) {
034    this.records = records;
035    this.roots = roots;
036    this.depth = depth;
037    this.synonyms = synonyms;
038    this.countByOrigin = countByOrigin;
039    this.countByRank = countByRank;
040    this.cycles = cycles;
041  }
042
043  /**
044   * @return list of cycles, each given as one taxonID of the loop
045   */
046  public List<String> getCycles() {
047    return cycles;
048  }
049
050  public void incOrigin(Origin origin) {
051    if (origin != null) {
052      if (!countByOrigin.containsKey(origin)) {
053        countByOrigin.put(origin, 1);
054      } else {
055        countByOrigin.put(origin, countByOrigin.get(origin) + 1);
056      }
057    }
058  }
059
060  public void incRank(Rank rank) {
061    if (rank != null) {
062      if (!countByRank.containsKey(rank)) {
063        countByRank.put(rank, 1);
064      } else {
065        countByRank.put(rank, countByRank.get(rank) + 1);
066      }
067    }
068  }
069
070  /**
071   * @return raw number of processed source records in dwc archive
072   */
073  public int getRecords() {
074    return records;
075  }
076
077  /**
078   * @return total number of name usages existing as neo nodes, both accepted and synonyms
079   */
080  public int getUsages() {
081    int total = 0;
082    for (int x : countByOrigin.values()) {
083      total += x;
084    }
085    return total;
086  }
087
088  /**
089   * @return the number of root taxa without a parent
090   */
091  public int getRoots() {
092    return roots;
093  }
094
095  /**
096   * @return maximum depth of the classification
097   */
098  public int getDepth() {
099    return depth;
100  }
101
102  /**
103   * @return the number of synonym nodes in this checklist
104   */
105  public int getSynonyms() {
106    return synonyms;
107  }
108
109  public int getCountByRank(Rank rank) {
110    if (countByRank.containsKey(rank)) {
111      return countByRank.get(rank);
112    }
113    return 0;
114  }
115
116  public int getCountByOrigin(Origin o) {
117    return countByOrigin.get(o);
118  }
119
120  public Map<Origin, Integer> getCountByOrigin() {
121    return countByOrigin;
122  }
123
124  public Map<Rank, Integer> getCountByRank() {
125    return countByRank;
126  }
127
128  @Override
129  public String toString() {
130    return "NormalizerStats{" +
131           "records=" + records +
132           ", roots=" + roots +
133           ", depth=" + depth +
134           ", synonyms=" + synonyms +
135           ", cycles=" + cycles.size() +
136           ", countByOrigin=" + countByOrigin +
137           ", countByRank=" + countByRank +
138           '}';
139  }
140
141  @Override
142  public int hashCode() {
143    return Objects.hashCode(records, roots, depth, synonyms, cycles, countByOrigin, countByRank);
144  }
145
146  @Override
147  public boolean equals(Object obj) {
148    if (this == obj) {
149      return true;
150    }
151    if (obj == null || getClass() != obj.getClass()) {
152      return false;
153    }
154    final NormalizerStats other = (NormalizerStats) obj;
155    return Objects.equal(this.records, other.records)
156           && Objects.equal(this.roots, other.roots)
157           && Objects.equal(this.depth, other.depth)
158           && Objects.equal(this.synonyms, other.synonyms)
159           && Objects.equal(this.cycles, other.cycles)
160           && Objects.equal(this.countByOrigin, other.countByOrigin)
161           && Objects.equal(this.countByRank, other.countByRank);
162  }
163
164}
