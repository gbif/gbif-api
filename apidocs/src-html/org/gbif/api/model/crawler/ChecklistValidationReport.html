001package org.gbif.api.model.crawler;
002
003import java.util.List;
004
005import com.google.common.base.Objects;
006import com.google.common.base.Preconditions;
007import org.codehaus.jackson.annotate.JsonCreator;
008import org.codehaus.jackson.annotate.JsonProperty;
009
010public class ChecklistValidationReport {
011  // the number of records checked in the validation
012  private final int checkedRecords;
013
014  // false if we had to stop at our memory-saving limit
015  private final boolean allRecordsChecked;
016
017  // a sample of non unique core/taxon identifier
018  private final List<String> duplicateIds;
019
020  // a sample of data file line numbers with records missing a core/taxon identifier
021  private final List<Integer> missingIds;
022
023  // if the archive is not valid this will hold a readable reason
024  private String invalidationReason;
025
026  // is this archive valid
027  private final boolean valid;
028
029  @JsonCreator
030  public ChecklistValidationReport(@JsonProperty("checkedRecords") int checkedRecords,
031    @JsonProperty("allRecordsChecked") boolean allRecordsChecked,
032    @JsonProperty("duplicateIds") List<String> duplicateIds,
033    @JsonProperty("missingIds") List<Integer> missingIds) {
034    this.checkedRecords = checkedRecords;
035    this.allRecordsChecked = allRecordsChecked;
036    this.duplicateIds = Preconditions.checkNotNull(duplicateIds, "duplicateIds cannot be null");
037    this.missingIds = Preconditions.checkNotNull(missingIds, "missingIds cannot be null");
038    this.valid = validate();
039  }
040
041  private boolean validate() {
042    if (!duplicateIds.isEmpty()) {
043      invalidationReason = "Non unique taxon ids";
044      return false;
045    }
046    if (!missingIds.isEmpty()) {
047      invalidationReason = "Missing taxon ids";
048      return false;
049    }
050    return true;
051  }
052
053  public int getCheckedRecords() {
054    return checkedRecords;
055  }
056
057  public boolean isAllRecordsChecked() {
058    return allRecordsChecked;
059  }
060
061  public List<String> getDuplicateIds() {
062    return duplicateIds;
063  }
064
065  public List<Integer> getMissingIds() {
066    return missingIds;
067  }
068
069  public String getInvalidationReason() {
070    return invalidationReason;
071  }
072
073  public boolean isValid() {
074    return valid;
075  }
076
077  @Override
078  public int hashCode() {
079    return Objects.hashCode(checkedRecords, allRecordsChecked, duplicateIds, missingIds, invalidationReason,valid);
080  }
081
082  @Override
083  public boolean equals(Object obj) {
084    if (this == obj) {
085      return true;
086    }
087    if (obj == null || getClass() != obj.getClass()) {
088      return false;
089    }
090    final ChecklistValidationReport other = (ChecklistValidationReport) obj;
091    return Objects.equal(this.checkedRecords, other.checkedRecords)
092           && Objects.equal(this.allRecordsChecked, other.allRecordsChecked)
093           && Objects.equal(this.duplicateIds, other.duplicateIds)
094           && Objects.equal(this.missingIds, other.missingIds)
095           && Objects.equal(this.invalidationReason, other.invalidationReason)
096           && Objects.equal(this.valid, other.valid);
097  }
098
099  @Override
100  public String toString() {
101    return Objects.toStringHelper(this)
102      .add("checkedRecords", checkedRecords)
103      .add("duplicateIds", duplicateIds)
104      .add("missingIds", missingIds)
105      .add("allRecordsChecked", allRecordsChecked)
106      .add("invalidationReason", invalidationReason)
107      .add("valid", valid)
108      .toString();
109  }
110}
