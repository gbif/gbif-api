001package org.gbif.api.model.collections;
002
003import org.gbif.api.model.registry.LenientEquals;
004
005import java.util.Date;
006import java.util.Objects;
007import java.util.StringJoiner;
008import java.util.UUID;
009import javax.validation.Valid;
010import javax.validation.constraints.NotNull;
011
012/**
013 * Person associated to a collection or institution.
014 */
015public class Person implements CollectionEntity, LenientEquals<Person> {
016
017  private UUID key;
018  private String firstName;
019  private String lastName;
020  private String position;
021  private String areaResponsibility;
022  private String researchPursuits;
023  private String phone;
024  private String fax;
025  private String email;
026  private Address mailingAddress;
027  private UUID primaryInstitutionKey;
028  private UUID primaryCollectionKey;
029  private String createdBy;
030  private String modifiedBy;
031  private Date created;
032  private Date modified;
033  private Date deleted;
034
035  /**
036   * GBIF Unique identifier.
037   */
038  @Override
039  public UUID getKey() {
040    return key;
041  }
042
043  @Override
044  public void setKey(UUID key) {
045    this.key = key;
046  }
047
048  /**
049   * Firs name, it can also include the first and last name in some cases.
050   */
051  @NotNull
052  public String getFirstName() {
053    return firstName;
054  }
055
056  public void setFirstName(String firstName) {
057    this.firstName = firstName;
058  }
059
060  /**
061   * Last and secondary names.
062   */
063  public String getLastName() {
064    return lastName;
065  }
066
067  public void setLastName(String lastName) {
068    this.lastName = lastName;
069  }
070
071  /**
072   * Textual description  of the role performed.
073   */
074  public String getPosition() {
075    return position;
076  }
077
078  public void setPosition(String position) {
079    this.position = position;
080  }
081
082  /**
083   * Description of the main responsibility assigned to a person.
084   */
085  public String getAreaResponsibility() {
086    return areaResponsibility;
087  }
088
089  public void setAreaResponsibility(String areaResponsibility) {
090    this.areaResponsibility = areaResponsibility;
091  }
092
093  /**
094   * Description of main research activities.
095   */
096  public String getResearchPursuits() {
097    return researchPursuits;
098  }
099
100  public void setResearchPursuits(String researchPursuits) {
101    this.researchPursuits = researchPursuits;
102  }
103
104  /**
105   * Primary telephone number.
106   */
107  public String getPhone() {
108    return phone;
109  }
110
111  public void setPhone(String phone) {
112    this.phone = phone;
113  }
114
115  /**
116   * Fax number.
117   */
118  public String getFax() {
119    return fax;
120  }
121
122  public void setFax(String fax) {
123    this.fax = fax;
124  }
125
126  /**
127   * Person email.
128   */
129  public String getEmail() {
130    return email;
131  }
132
133  public void setEmail(String email) {
134    this.email = email;
135  }
136
137  /**
138   * Address used to send physical mail.
139   */
140  @Valid
141  public Address getMailingAddress() {
142    return mailingAddress;
143  }
144
145  public void setMailingAddress(Address mailingAddress) {
146    this.mailingAddress = mailingAddress;
147  }
148
149  /**
150   * Primary institution where the person works.
151   */
152  public UUID getPrimaryInstitutionKey() {
153    return primaryInstitutionKey;
154  }
155
156  public void setPrimaryInstitutionKey(UUID primaryInstitutionKey) {
157    this.primaryInstitutionKey = primaryInstitutionKey;
158  }
159
160  /**
161   * Primary collection linked of the person.
162   */
163  public UUID getPrimaryCollectionKey() {
164    return primaryCollectionKey;
165  }
166
167  public void setPrimaryCollectionKey(UUID primaryCollectionKey) {
168    this.primaryCollectionKey = primaryCollectionKey;
169  }
170
171  @Override
172  public String getCreatedBy() {
173    return createdBy;
174  }
175
176  @Override
177  public void setCreatedBy(String createdBy) {
178    this.createdBy = createdBy;
179  }
180
181  @Override
182  public String getModifiedBy() {
183    return modifiedBy;
184  }
185
186  @Override
187  public void setModifiedBy(String modifiedBy) {
188    this.modifiedBy = modifiedBy;
189  }
190
191  @Override
192  public Date getCreated() {
193    return created;
194  }
195
196  @Override
197  public void setCreated(Date created) {
198    this.created = created;
199  }
200
201  @Override
202  public Date getModified() {
203    return modified;
204  }
205
206  @Override
207  public void setModified(Date modified) {
208    this.modified = modified;
209  }
210
211  @Override
212  public Date getDeleted() {
213    return deleted;
214  }
215
216  @Override
217  public void setDeleted(Date deleted) {
218    this.deleted = deleted;
219  }
220
221  @Override
222  public boolean equals(Object o) {
223    if (this == o) return true;
224    if (o == null || getClass() != o.getClass()) return false;
225    Person person = (Person) o;
226    return Objects.equals(key, person.key)
227           && Objects.equals(firstName, person.firstName)
228           && Objects.equals(lastName,
229                             person.lastName)
230           && Objects.equals(position, person.position)
231           && Objects.equals(areaResponsibility, person.areaResponsibility)
232           && Objects.equals(researchPursuits, person.researchPursuits)
233           && Objects.equals(phone, person.phone)
234           && Objects.equals(fax, person.fax)
235           && Objects.equals(email, person.email)
236           && Objects.equals(mailingAddress, person.mailingAddress)
237           && Objects.equals(primaryInstitutionKey, person.primaryInstitutionKey)
238           && Objects.equals(primaryCollectionKey, person.primaryCollectionKey)
239           && Objects.equals(createdBy, person.createdBy)
240           && Objects.equals(modifiedBy, person.modifiedBy)
241           && Objects.equals(created, person.created)
242           && Objects.equals(modified, person.modified)
243           && Objects.equals(deleted, person.deleted);
244  }
245
246  @Override
247  public int hashCode() {
248    return Objects.hash(key,
249                        firstName,
250                        lastName,
251                        position,
252                        areaResponsibility,
253                        researchPursuits,
254                        phone,
255                        fax,
256                        email,
257                        mailingAddress,
258                        primaryInstitutionKey,
259                        primaryCollectionKey,
260                        createdBy,
261                        modifiedBy,
262                        created,
263                        modified,
264                        deleted);
265  }
266
267  @Override
268  public String toString() {
269    return new StringJoiner(", ", Person.class.getSimpleName() + "[", "]").add("key=" + key)
270      .add("firstName='" + firstName + "'")
271      .add("lastName='" + lastName + "'")
272      .add("position='" + position + "'")
273      .add("areaResponsibility='" + areaResponsibility + "'")
274      .add("researchPursuits='" + researchPursuits + "'")
275      .add("phone='" + phone + "'")
276      .add("fax='" + fax + "'")
277      .add("email='" + email + "'")
278      .add("mailingAddress=" + mailingAddress)
279      .add("institutionKey=" + primaryInstitutionKey)
280      .add("collectionKey=" + primaryCollectionKey)
281      .add("createdBy='" + createdBy + "'")
282      .add("modifiedBy='" + modifiedBy + "'")
283      .add("created=" + created)
284      .add("modified=" + modified)
285      .add("deleted=" + deleted)
286      .toString();
287  }
288
289  @Override
290  public boolean lenientEquals(Person other) {
291    if (this == other) {
292      return true;
293    }
294    return Objects.equals(key, other.key)
295           && Objects.equals(firstName, other.firstName)
296           && Objects.equals(lastName, other.lastName)
297           && Objects.equals(position, other.position)
298           && Objects.equals(areaResponsibility, other.areaResponsibility)
299           && Objects.equals(researchPursuits, other.researchPursuits)
300           && Objects.equals(phone, other.phone)
301           && Objects.equals(fax, other.fax)
302           && Objects.equals(email, other.email)
303           && Objects.equals(mailingAddress, other.mailingAddress)
304           && Objects.equals(primaryInstitutionKey, other.primaryInstitutionKey)
305           && Objects.equals(primaryCollectionKey, other.primaryCollectionKey)
306           && Objects.equals(deleted, other.deleted);
307  }
308}
