499  @Override
500  public boolean equals(Object o) {
501    if (this == o) return true;
502    if (o == null || getClass() != o.getClass()) return false;
503    Collection that = (Collection) o;
504    return active == that.active
505        && personalCollection == that.personalCollection
506        && Objects.equals(key, that.key)
507        && Objects.equals(code, that.code)
508        && Objects.equals(name, that.name)
509        && Objects.equals(description, that.description)
510        && Objects.equals(contentTypes, that.contentTypes)
511        && Objects.equals(doi, that.doi)
512        && Objects.equals(email, that.email)
513        && Objects.equals(phone, that.phone)
514        && Objects.equals(homepage, that.homepage)
515        && Objects.equals(catalogUrl, that.catalogUrl)
516        && Objects.equals(apiUrl, that.apiUrl)
517        && Objects.equals(preservationTypes, that.preservationTypes)
518        && accessionStatus == that.accessionStatus
519        && Objects.equals(institutionKey, that.institutionKey)
520        && Objects.equals(mailingAddress, that.mailingAddress)
521        && Objects.equals(address, that.address)
522        && Objects.equals(createdBy, that.createdBy)
523        && Objects.equals(modifiedBy, that.modifiedBy)
524        && Objects.equals(created, that.created)
525        && Objects.equals(modified, that.modified)
526        && Objects.equals(deleted, that.deleted)
527        && Objects.equals(tags, that.tags)
528        && Objects.equals(identifiers, that.identifiers)
529        && Objects.equals(contacts, that.contacts)
530        && indexHerbariorumRecord == that.indexHerbariorumRecord
531        && Objects.equals(numberSpecimens, that.numberSpecimens)
532        && Objects.equals(machineTags, that.machineTags)
533        && Objects.equals(taxonomicCoverage, that.taxonomicCoverage)
534        && Objects.equals(geography, that.geography)
535        && Objects.equals(notes, that.notes)
536        && Objects.equals(incorporatedCollections, that.incorporatedCollections)
537        && Objects.equals(importantCollectors, that.importantCollectors)
538        && Objects.equals(collectionSummary, that.collectionSummary)
539        && Objects.equals(alternativeCodes, that.alternativeCodes)
540        && Objects.equals(comments, that.comments)
541        && Objects.equals(occurrenceMappings, that.occurrenceMappings)
542        && Objects.equals(replacedBy, that.replacedBy);
543  }
