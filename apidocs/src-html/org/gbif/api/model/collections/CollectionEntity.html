014package org.gbif.api.model.collections;
015
016import org.gbif.api.model.registry.Commentable;
017import org.gbif.api.model.registry.Identifiable;
018import org.gbif.api.model.registry.MachineTaggable;
019import org.gbif.api.model.registry.PrePersist;
020import org.gbif.api.model.registry.Taggable;
021import org.gbif.api.vocabulary.collections.MasterSourceType;
022
023import java.io.Serializable;
024import java.util.Date;
025import java.util.UUID;
026
027import javax.annotation.Nullable;
028import javax.validation.constraints.NotNull;
029import javax.validation.constraints.Size;
030
031/** Entity . */
032public interface CollectionEntity
033    extends Contactable,
034        Taggable,
035        MachineTaggable,
036        Identifiable,
037        Commentable,
038        OccurrenceMappeable,
039        Serializable {
040
041  /** Unique identifier. */
042  UUID getKey();
043
044  void setKey(UUID key);
045
046  /** Creator of the database record. */
047  String getCreatedBy();
048
049  void setCreatedBy(String createdBy);
050
051  /** Person or agent that modified the database record. */
052  String getModifiedBy();
053
054  void setModifiedBy(String modifiedBy);
055
056  /** Date when the records as created. */
057  Date getCreated();
058
059  void setCreated(Date created);
060
061  /** Date when the records was last modified. */
062  Date getModified();
063
064  void setModified(Date modified);
065
066  /** Date when the records was (logically) deleted. */
067  @Nullable
068  Date getDeleted();
069
070  void setDeleted(Date deleted);
071
072  /** Identifies an entity at the owner's location. */
073  @NotNull(groups = PrePersist.class)
074  String getCode();
075
076  void setCode(String code);
077
078  /** Descriptive name of an entity. */
079  @NotNull
080  String getName();
081
082  void setName(String name);
083
084  /** Textual description/summary of the contents of an entity. */
085  @Size(min = 1)
086  String getDescription();
087
088  void setDescription(String description);
089
090  /** Is this entity currently active/maintained. */
091  boolean isActive();
092
093  void setActive(boolean active);
094
095  /** Replacement of the entity (if applies). */
096  UUID getReplacedBy();
097
098  void setReplacedBy(UUID replacedBy);
099
100  /** Master source of the entity. */
101  MasterSourceType getMasterSource();
102
103  void setMasterSource(MasterSourceType masterSource);
104
105  /** Master source metadata */
106  MasterSourceMetadata getMasterSourceMetadata();
107
108  void setMasterSourceMetadata(MasterSourceMetadata masterSourceMetadata);
109
110  /** Flag to display the entity in the NHC portal. */
111  Boolean getDisplayOnNHCPortal();
112
113  void setDisplayOnNHCPortal(Boolean displayOnNHCPortal);
114}
