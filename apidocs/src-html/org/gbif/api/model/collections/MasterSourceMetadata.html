001package org.gbif.api.model.collections;
002
003import org.gbif.api.model.registry.PostPersist;
004import org.gbif.api.model.registry.PrePersist;
005import org.gbif.api.vocabulary.collections.Source;
006
007import java.io.Serializable;
008import java.util.Date;
009import java.util.Objects;
010import java.util.StringJoiner;
011
012import javax.validation.constraints.NotNull;
013import javax.validation.constraints.Null;
014
015/** Metadata to sync GRSciColl entities with their master sources. */
016public class MasterSourceMetadata implements Serializable {
017
018  @Null(groups = {PrePersist.class})
019  @NotNull(groups = {PostPersist.class})
020  private Integer key;
021
022  @NotNull private Source source;
023
024  @NotNull private String sourceId;
025
026  private String createdBy;
027  private Date created;
028
029  public MasterSourceMetadata() {}
030
031  public MasterSourceMetadata(Source source, String sourceId) {
032    this.source = source;
033    this.sourceId = sourceId;
034  }
035
036  public Integer getKey() {
037    return key;
038  }
039
040  public void setKey(Integer key) {
041    this.key = key;
042  }
043
044  public Source getSource() {
045    return source;
046  }
047
048  public void setSource(Source source) {
049    this.source = source;
050  }
051
052  public String getSourceId() {
053    return sourceId;
054  }
055
056  public void setSourceId(String sourceId) {
057    this.sourceId = sourceId;
058  }
059
060  public String getCreatedBy() {
061    return createdBy;
062  }
063
064  public void setCreatedBy(String createdBy) {
065    this.createdBy = createdBy;
066  }
067
068  public Date getCreated() {
069    return created;
070  }
071
072  public void setCreated(Date created) {
073    this.created = created;
074  }
075
076  @Override
077  public boolean equals(Object o) {
078    if (this == o) {
079      return true;
080    }
081    if (o == null || getClass() != o.getClass()) {
082      return false;
083    }
084    MasterSourceMetadata metadata = (MasterSourceMetadata) o;
085    return Objects.equals(key, metadata.key)
086        && source == metadata.source
087        && Objects.equals(sourceId, metadata.sourceId)
088        && Objects.equals(createdBy, metadata.createdBy)
089        && Objects.equals(created, metadata.created);
090  }
091
092  @Override
093  public int hashCode() {
094    return Objects.hash(key, source, sourceId, createdBy, created);
095  }
096
097  @Override
098  public String toString() {
099    return new StringJoiner(", ", MasterSourceMetadata.class.getSimpleName() + "[", "]")
100        .add("key=" + key)
101        .add("source=" + source)
102        .add("sourceId='" + sourceId + "'")
103        .add("createdBy='" + createdBy + "'")
104        .add("created=" + created)
105        .toString();
106  }
107}
