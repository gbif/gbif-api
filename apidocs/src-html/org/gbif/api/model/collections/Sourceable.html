001package org.gbif.api.model.collections;
002
003import java.lang.annotation.Documented;
004import java.lang.annotation.ElementType;
005import java.lang.annotation.Repeatable;
006import java.lang.annotation.Retention;
007import java.lang.annotation.RetentionPolicy;
008import java.lang.annotation.Target;
009
010import org.gbif.api.vocabulary.collections.MasterSourceType;
011
012/**
013 * Annotation to indicate that a field can be populated from external sources.
014 *
015 * <p>This is used in GRSciColl to flag fields whose value comes from other sources rather than
016 * GRSciColl such as IH, datasets or organizations.
017 */
018@Documented
019@Retention(RetentionPolicy.RUNTIME)
020@Target({ElementType.FIELD})
021@Repeatable(value = Sourceables.class)
022public @interface Sourceable {
023
024  MasterSourceType[] masterSources() default {};
025
026  /**
027   * Indicates that only some parts are sourceable. If this field is not set it's assumed that the
028   * whole field is sourceable.
029   *
030   * <p>This applies only to lists or similar. For example, a list of identifiers where only the DOI
031   * identifiers are sourceable would set this field as {DOI}.
032   */
033  String[] sourceableParts() default {};
034
035  /**
036   * Indicates that a field is initially filled from an external source but it can be overwritten
037   * afterwards.
038   *
039   * <p>E.g.: the names of IH collections.
040   */
041  boolean overridable() default false;
042}
