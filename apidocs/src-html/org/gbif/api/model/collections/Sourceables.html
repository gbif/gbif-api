014package org.gbif.api.model.collections;
015
016import java.lang.annotation.Documented;
017import java.lang.annotation.ElementType;
018import java.lang.annotation.Retention;
019import java.lang.annotation.RetentionPolicy;
020import java.lang.annotation.Target;
021
022/**
023 * Container annotation for multiple {@link Sourceable} annotations.
024 *
025 * <p>This is needed for cases where we have multiple master sources and each of them has different
026 * sourceable parts. For example, for GBIF_REGISTRY the DOI identifiers are sourceable and for IH
027 * only the IH_IRN are sourceable.
028 */
029@Documented
030@Retention(RetentionPolicy.RUNTIME)
031@Target({ElementType.FIELD})
032public @interface Sourceables {
033
034  Sourceable[] value();
035}
