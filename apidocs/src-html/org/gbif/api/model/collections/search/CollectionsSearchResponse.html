016package org.gbif.api.model.collections.search;
017
018import java.util.Objects;
019import java.util.Set;
020import java.util.UUID;
021
022/** Models the response for the Collections search. */
023public class CollectionsSearchResponse {
024
025  private String type;
026  private UUID key;
027  private String code;
028  private String name;
029  private UUID institutionKey;
030  private String institutionCode;
031  private String institutionName;
032  private Set<Match> matches;
033
034  public String getType() {
035    return type;
036  }
037
038  public void setType(String type) {
039    this.type = type;
040  }
041
042  public UUID getKey() {
043    return key;
044  }
045
046  public void setKey(UUID key) {
047    this.key = key;
048  }
049
050  public String getCode() {
051    return code;
052  }
053
054  public void setCode(String code) {
055    this.code = code;
056  }
057
058  public String getName() {
059    return name;
060  }
061
062  public void setName(String name) {
063    this.name = name;
064  }
065
066  public UUID getInstitutionKey() {
067    return institutionKey;
068  }
069
070  public void setInstitutionKey(UUID institutionKey) {
071    this.institutionKey = institutionKey;
072  }
073
074  public String getInstitutionCode() {
075    return institutionCode;
076  }
077
078  public void setInstitutionCode(String institutionCode) {
079    this.institutionCode = institutionCode;
080  }
081
082  public String getInstitutionName() {
083    return institutionName;
084  }
085
086  public void setInstitutionName(String institutionName) {
087    this.institutionName = institutionName;
088  }
089
090  public Set<Match> getMatches() {
091    return matches;
092  }
093
094  public void setMatches(Set<Match> matches) {
095    this.matches = matches;
096  }
097
098  public static class Match {
099    private String field;
100    private String snippet;
101
102    public String getField() {
103      return field;
104    }
105
106    public void setField(String field) {
107      this.field = field;
108    }
109
110    public String getSnippet() {
111      return snippet;
112    }
113
114    public void setSnippet(String snippet) {
115      this.snippet = snippet;
116    }
117
118    @Override
119    public boolean equals(Object o) {
120      if (this == o) return true;
121      if (o == null || getClass() != o.getClass()) return false;
122      Match match = (Match) o;
123      return Objects.equals(field, match.field);
124    }
125
126    @Override
127    public int hashCode() {
128      return Objects.hash(field);
129    }
130  }
131
132  @Override
133  public boolean equals(Object o) {
134    if (this == o) {
135      return true;
136    }
137    if (o == null || getClass() != o.getClass()) {
138      return false;
139    }
140    CollectionsSearchResponse response = (CollectionsSearchResponse) o;
141    return Objects.equals(type, response.type)
142        && Objects.equals(key, response.key)
143        && Objects.equals(code, response.code)
144        && Objects.equals(name, response.name)
145        && Objects.equals(institutionKey, response.institutionKey)
146        && Objects.equals(institutionCode, response.institutionCode)
147        && Objects.equals(institutionName, response.institutionName)
148        && Objects.equals(matches, response.matches);
149  }
150
151  @Override
152  public int hashCode() {
153    return Objects.hash(
154        type, key, code, name, institutionKey, institutionCode, institutionName, matches);
155  }
156}
