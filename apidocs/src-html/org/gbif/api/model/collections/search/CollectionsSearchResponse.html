014package org.gbif.api.model.collections.search;
015
016import lombok.Getter;
017
018import lombok.Setter;
019
020import org.gbif.api.vocabulary.Country;
021
022import java.util.Objects;
023import java.util.Set;
024import java.util.UUID;
025
026/** Models the response for the Collections search. */
027@Getter
028@Setter
029public class CollectionsSearchResponse {
030
031  private String type;
032  private UUID key;
033  private String code;
034  private String name;
035  private UUID institutionKey;
036  private String institutionCode;
037  private String institutionName;
038
039  private boolean displayOnNHCPortal;
040
041  private Country country;
042
043  private Country mailingCountry;
044
045  private Set<Match> matches;
046
047  @Setter
048  @Getter
049  public static class Match {
050    private String field;
051    private String snippet;
052
053    @Override
054    public boolean equals(Object o) {
055      if (this == o) return true;
056      if (o == null || getClass() != o.getClass()) return false;
057      Match match = (Match) o;
058      return Objects.equals(field, match.field);
059    }
060
061    @Override
062    public int hashCode() {
063      return Objects.hash(field);
064    }
065  }
066
067  @Override
068  public boolean equals(Object o) {
069    if (this == o) {
070      return true;
071    }
072    if (o == null || getClass() != o.getClass()) {
073      return false;
074    }
075    CollectionsSearchResponse response = (CollectionsSearchResponse) o;
076    return Objects.equals(type, response.type)
077        && Objects.equals(key, response.key)
078        && Objects.equals(code, response.code)
079        && Objects.equals(name, response.name)
080        && Objects.equals(institutionKey, response.institutionKey)
081        && Objects.equals(institutionCode, response.institutionCode)
082        && Objects.equals(institutionName, response.institutionName)
083        && Objects.equals(displayOnNHCPortal, response.displayOnNHCPortal)
084        && Objects.equals(country, response.country)
085        && Objects.equals(mailingCountry, response.mailingCountry)
086        && Objects.equals(matches, response.matches);
087  }
088
089  @Override
090  public int hashCode() {
091    return Objects.hash(
092        type,
093        key,
094        code,
095        name,
096        institutionKey,
097        institutionCode,
098        institutionName,
099        displayOnNHCPortal,
100        country,
101        mailingCountry,
102        matches);
103  }
104}
