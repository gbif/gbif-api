001package org.gbif.api.model.collections.search;
002
003import lombok.Getter;
004import lombok.Setter;
005
006import java.util.Objects;
007
008@Setter
009@Getter
010public class Highlight {
011  private String field;
012  private String snippet;
013
014  @Override
015  public boolean equals(Object o) {
016    if (this == o) return true;
017    if (o == null || getClass() != o.getClass()) return false;
018    Highlight highlight = (Highlight) o;
019    return Objects.equals(field, highlight.field);
020  }
021
022  @Override
023  public int hashCode() {
024    return Objects.hash(field);
025  }
026}
