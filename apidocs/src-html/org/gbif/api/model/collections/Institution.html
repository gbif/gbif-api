413  @Override
414  public boolean equals(Object o) {
415    if (this == o) return true;
416    if (o == null || getClass() != o.getClass()) return false;
417    Institution that = (Institution) o;
418    return active == that.active
419           && numberSpecimens == that.numberSpecimens
420           && indexHerbariorumRecord == that.indexHerbariorumRecord
421           && Objects.equals(key, that.key)
422           && Objects.equals(code, that.code)
423           && Objects.equals(name, that.name)
424           && Objects.equals(description, that.description)
425           && type == that.type
426           && Objects.equals(homepage, that.homepage)
427           && Objects.equals(catalogUrl, that.catalogUrl)
428           && Objects.equals(apiUrl, that.apiUrl)
429           && institutionalGovernance == that.institutionalGovernance
430           && Objects.equals(disciplines, that.disciplines)
431           && Objects.equals(latitude, that.latitude)
432           && Objects.equals(longitude, that.longitude)
433           && Objects.equals(mailingAddress, that.mailingAddress)
434           && Objects.equals(address, that.address)
435           && Objects.equals(additionalNames, that.additionalNames)
436           && Objects.equals(foundingDate, that.foundingDate)
437           && Objects.equals(geographicDescription, that.geographicDescription)
438           && Objects.equals(taxonomicDescription, that.taxonomicDescription)
439           && Objects.equals(logoUrl, that.logoUrl)
440           && Objects.equals(citesPermitNumber, that.citesPermitNumber)
441           && Objects.equals(createdBy, that.createdBy)
442           && Objects.equals(modifiedBy, that.modifiedBy)
443           && Objects.equals(created, that.created)
444           && Objects.equals(modified, that.modified)
445           && Objects.equals(deleted, that.deleted)
446           && Objects.equals(tags, that.tags)
447           && Objects.equals(identifiers, that.identifiers)
448           && Objects.equals(contacts, that.contacts);
449  }

529           && indexHerbariorumRecord == other.indexHerbariorumRecord
530           && Objects.equals(key, other.key)
531           && Objects.equals(code, other.code)
