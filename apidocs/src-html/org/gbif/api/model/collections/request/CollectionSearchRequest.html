016package org.gbif.api.model.collections.request;
017
018import org.gbif.api.model.common.paging.Pageable;
019import org.gbif.api.vocabulary.Country;
020import org.gbif.api.vocabulary.IdentifierType;
021import org.gbif.api.vocabulary.collections.AccessionStatus;
022import org.gbif.api.vocabulary.collections.CollectionContentType;
023import org.gbif.api.vocabulary.collections.PreservationType;
024
025import java.util.List;
026import java.util.UUID;
027
028import javax.annotation.Nullable;
029
030public class CollectionSearchRequest extends SearchRequest {
031
032  @Nullable private UUID institution;
033  @Nullable private List<CollectionContentType> contentTypes;
034  @Nullable private List<PreservationType> preservationTypes;
035  @Nullable private AccessionStatus accessionStatus;
036  @Nullable private Boolean personalCollection;
037
038  @Nullable
039  public UUID getInstitution() {
040    return institution;
041  }
042
043  public void setInstitution(@Nullable UUID institution) {
044    this.institution = institution;
045  }
046
047  @Nullable
048  public List<CollectionContentType> getContentTypes() {
049    return contentTypes;
050  }
051
052  public void setContentTypes(@Nullable List<CollectionContentType> contentTypes) {
053    this.contentTypes = contentTypes;
054  }
055
056  @Nullable
057  public List<PreservationType> getPreservationTypes() {
058    return preservationTypes;
059  }
060
061  public void setPreservationTypes(@Nullable List<PreservationType> preservationTypes) {
062    this.preservationTypes = preservationTypes;
063  }
064
065  @Nullable
066  public AccessionStatus getAccessionStatus() {
067    return accessionStatus;
068  }
069
070  public void setAccessionStatus(@Nullable AccessionStatus accessionStatus) {
071    this.accessionStatus = accessionStatus;
072  }
073
074  @Nullable
075  public Boolean getPersonalCollection() {
076    return personalCollection;
077  }
078
079  public void setPersonalCollection(@Nullable Boolean personalCollection) {
080    this.personalCollection = personalCollection;
081  }
082
083  public static Builder builder() {
084    return new Builder();
085  }
086
087  public static class Builder {
088
089    UUID institution;
090    UUID contact;
091    String q;
092    String code;
093    String name;
094    String alternativeCode;
095    String machineTagNamespace;
096    String machineTagName;
097    String machineTagValue;
098    IdentifierType identifierType;
099    String identifier;
100    Country country;
101    String city;
102    String fuzzyName;
103    Boolean active;
104    List<CollectionContentType> contentTypes;
105    List<PreservationType> preservationTypes;
106    AccessionStatus accessionStatus;
107    Boolean personalCollection;
108    Pageable page;
109
110    public Builder institution(UUID institution) {
111      this.institution = institution;
112      return this;
113    }
114
115    public Builder contact(UUID contact) {
116      this.contact = contact;
117      return this;
118    }
119
120    public Builder query(String q) {
121      this.q = q;
122      return this;
123    }
124
125    public Builder code(String code) {
126      this.code = code;
127      return this;
128    }
129
130    public Builder name(String name) {
131      this.name = name;
132      return this;
133    }
134
135    public Builder alternativeCode(String alternativeCode) {
136      this.alternativeCode = alternativeCode;
137      return this;
138    }
139
140    public Builder machineTagNamespace(String machineTagNamespace) {
141      this.machineTagNamespace = machineTagNamespace;
142      return this;
143    }
144
145    public Builder machineTagName(String machineTagName) {
146      this.machineTagName = machineTagName;
147      return this;
148    }
149
150    public Builder machineTagValue(String machineTagValue) {
151      this.machineTagValue = machineTagValue;
152      return this;
153    }
154
155    public Builder identifierType(IdentifierType identifierType) {
156      this.identifierType = identifierType;
157      return this;
158    }
159
160    public Builder identifier(String identifier) {
161      this.identifier = identifier;
162      return this;
163    }
164
165    public Builder country(Country country) {
166      this.country = country;
167      return this;
168    }
169
170    public Builder city(String city) {
171      this.city = city;
172      return this;
173    }
174
175    public Builder fuzzyName(String fuzzyName) {
176      this.fuzzyName = fuzzyName;
177      return this;
178    }
179
180    public Builder active(boolean active) {
181      this.active = active;
182      return this;
183    }
184
185    public Builder contentTypes(List<CollectionContentType> contentTypes) {
186      this.contentTypes = contentTypes;
187      return this;
188    }
189
190    public Builder preservationTypes(List<PreservationType> preservationTypes) {
191      this.preservationTypes = preservationTypes;
192      return this;
193    }
194
195    public Builder accessionStatus(AccessionStatus accessionStatus) {
196      this.accessionStatus = accessionStatus;
197      return this;
198    }
199
200    public Builder personalCollection(boolean personalCollection) {
201      this.personalCollection = personalCollection;
202      return this;
203    }
204
205    public Builder page(Pageable page) {
206      this.page = page;
207      return this;
208    }
209
210    public CollectionSearchRequest build() {
211      CollectionSearchRequest req = new CollectionSearchRequest();
212      req.setInstitution(institution);
213      req.setContact(contact);
214      req.setQ(q);
215      req.setCode(code);
216      req.setName(name);
217      req.setAlternativeCode(alternativeCode);
218      req.setMachineTagNamespace(machineTagNamespace);
219      req.setMachineTagName(machineTagName);
220      req.setMachineTagValue(machineTagValue);
221      req.setIdentifierType(identifierType);
222      req.setIdentifier(identifier);
223      req.setCountry(country);
224      req.setCity(city);
225      req.setFuzzyName(fuzzyName);
226      req.setActive(active);
227      req.setContentTypes(contentTypes);
228      req.setPreservationTypes(preservationTypes);
229      req.setAccessionStatus(accessionStatus);
230      req.setPersonalCollection(personalCollection);
231      if (page != null) {
232        req.setLimit(page.getLimit());
233        req.setOffset(page.getOffset());
234      }
235      return req;
236    }
237  }
238}
