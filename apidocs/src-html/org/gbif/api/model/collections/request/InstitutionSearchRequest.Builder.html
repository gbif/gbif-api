016package org.gbif.api.model.collections.request;
017
018import org.gbif.api.model.common.paging.Pageable;
019import org.gbif.api.vocabulary.Country;
020import org.gbif.api.vocabulary.IdentifierType;
021
022import java.util.UUID;
023
024public class InstitutionSearchRequest extends SearchRequest {
025
026  public static Builder builder() {
027    return new Builder();
028  }
029
030  public static class Builder {
031
032    UUID contact;
033    String q;
034    String code;
035    String name;
036    String alternativeCode;
037    String machineTagNamespace;
038    String machineTagName;
039    String machineTagValue;
040    IdentifierType identifierType;
041    String identifier;
042    Country country;
043    Pageable page;
044
045    public Builder contact(UUID contact) {
046      this.contact = contact;
047      return this;
048    }
049
050    public Builder query(String q) {
051      this.q = q;
052      return this;
053    }
054
055    public Builder code(String code) {
056      this.code = code;
057      return this;
058    }
059
060    public Builder name(String name) {
061      this.name = name;
062      return this;
063    }
064
065    public Builder alternativeCode(String alternativeCode) {
066      this.alternativeCode = alternativeCode;
067      return this;
068    }
069
070    public Builder machineTagNamespace(String machineTagNamespace) {
071      this.machineTagNamespace = machineTagNamespace;
072      return this;
073    }
074
075    public Builder machineTagName(String machineTagName) {
076      this.machineTagName = machineTagName;
077      return this;
078    }
079
080    public Builder machineTagValue(String machineTagValue) {
081      this.machineTagValue = machineTagValue;
082      return this;
083    }
084
085    public Builder identifierType(IdentifierType identifierType) {
086      this.identifierType = identifierType;
087      return this;
088    }
089
090    public Builder identifier(String identifier) {
091      this.identifier = identifier;
092      return this;
093    }
094
095    public Builder country(Country country) {
096      this.country = country;
097      return this;
098    }
099
100    public Builder page(Pageable page) {
101      this.page = page;
102      return this;
103    }
104
105    public InstitutionSearchRequest build() {
106      InstitutionSearchRequest req = new InstitutionSearchRequest();
107      req.setContact(contact);
108      req.setQ(q);
109      req.setCode(code);
110      req.setName(name);
111      req.setAlternativeCode(alternativeCode);
112      req.setMachineTagNamespace(machineTagNamespace);
113      req.setMachineTagName(machineTagName);
114      req.setMachineTagValue(machineTagValue);
115      req.setIdentifierType(identifierType);
116      req.setIdentifier(identifier);
117      req.setCountry(country);
118      if (page != null) {
119        req.setLimit(page.getLimit());
120        req.setOffset(page.getOffset());
121      }
122      return req;
123    }
124  }
125}
