016package org.gbif.api.model.collections.request;
017
018import org.gbif.api.model.collections.request.CollectionSearchRequest.Builder;
019import org.gbif.api.model.common.paging.Pageable;
020import org.gbif.api.vocabulary.Country;
021import org.gbif.api.vocabulary.IdentifierType;
022import org.gbif.api.vocabulary.collections.Discipline;
023import org.gbif.api.vocabulary.collections.InstitutionGovernance;
024import org.gbif.api.vocabulary.collections.InstitutionType;
025import org.gbif.api.vocabulary.collections.MasterSourceType;
026
027import java.util.List;
028import java.util.UUID;
029
030import javax.annotation.Nullable;
031
032public class InstitutionSearchRequest extends SearchRequest {
033
034  @Nullable private InstitutionType type;
035
036  @Nullable private InstitutionGovernance institutionalGovernance;
037
038  @Nullable private List<Discipline> disciplines;
039
040  @Nullable
041  public InstitutionType getType() {
042    return type;
043  }
044
045  public void setType(@Nullable InstitutionType type) {
046    this.type = type;
047  }
048
049  @Nullable
050  public InstitutionGovernance getInstitutionalGovernance() {
051    return institutionalGovernance;
052  }
053
054  public void setInstitutionalGovernance(@Nullable InstitutionGovernance institutionalGovernance) {
055    this.institutionalGovernance = institutionalGovernance;
056  }
057
058  @Nullable
059  public List<Discipline> getDisciplines() {
060    return disciplines;
061  }
062
063  public void setDisciplines(@Nullable List<Discipline> disciplines) {
064    this.disciplines = disciplines;
065  }
066
067  public static Builder builder() {
068    return new Builder();
069  }
070
071  public static class Builder {
072
073    UUID contact;
074    String q;
075    String code;
076    String name;
077    String alternativeCode;
078    String machineTagNamespace;
079    String machineTagName;
080    String machineTagValue;
081    IdentifierType identifierType;
082    String identifier;
083    Country country;
084    String city;
085    String fuzzyName;
086    Boolean active;
087    InstitutionType type;
088    InstitutionGovernance institutionalGovernance;
089    List<Discipline> disciplines;
090    MasterSourceType masterSourceType;
091    Pageable page;
092
093    public Builder contact(UUID contact) {
094      this.contact = contact;
095      return this;
096    }
097
098    public Builder query(String q) {
099      this.q = q;
100      return this;
101    }
102
103    public Builder code(String code) {
104      this.code = code;
105      return this;
106    }
107
108    public Builder name(String name) {
109      this.name = name;
110      return this;
111    }
112
113    public Builder alternativeCode(String alternativeCode) {
114      this.alternativeCode = alternativeCode;
115      return this;
116    }
117
118    public Builder machineTagNamespace(String machineTagNamespace) {
119      this.machineTagNamespace = machineTagNamespace;
120      return this;
121    }
122
123    public Builder machineTagName(String machineTagName) {
124      this.machineTagName = machineTagName;
125      return this;
126    }
127
128    public Builder machineTagValue(String machineTagValue) {
129      this.machineTagValue = machineTagValue;
130      return this;
131    }
132
133    public Builder identifierType(IdentifierType identifierType) {
134      this.identifierType = identifierType;
135      return this;
136    }
137
138    public Builder identifier(String identifier) {
139      this.identifier = identifier;
140      return this;
141    }
142
143    public Builder country(Country country) {
144      this.country = country;
145      return this;
146    }
147
148    public Builder city(String city) {
149      this.city = city;
150      return this;
151    }
152
153    public Builder fuzzyName(String fuzzyName) {
154      this.fuzzyName = fuzzyName;
155      return this;
156    }
157
158    public Builder active(boolean active) {
159      this.active = active;
160      return this;
161    }
162
163    public Builder type(InstitutionType type) {
164      this.type = type;
165      return this;
166    }
167
168    public Builder institutionalGovernance(InstitutionGovernance institutionalGovernance) {
169      this.institutionalGovernance = institutionalGovernance;
170      return this;
171    }
172
173    public Builder disciplines(List<Discipline> disciplines) {
174      this.disciplines = disciplines;
175      return this;
176    }
177
178    public Builder masterSourceType(MasterSourceType masterSourceType) {
179      this.masterSourceType = masterSourceType;
180      return this;
181    }
182
183    public Builder page(Pageable page) {
184      this.page = page;
185      return this;
186    }
187
188    public InstitutionSearchRequest build() {
189      InstitutionSearchRequest req = new InstitutionSearchRequest();
190      req.setContact(contact);
191      req.setQ(q);
192      req.setCode(code);
193      req.setName(name);
194      req.setAlternativeCode(alternativeCode);
195      req.setMachineTagNamespace(machineTagNamespace);
196      req.setMachineTagName(machineTagName);
197      req.setMachineTagValue(machineTagValue);
198      req.setIdentifierType(identifierType);
199      req.setIdentifier(identifier);
200      req.setCountry(country);
201      req.setCity(city);
202      req.setFuzzyName(fuzzyName);
203      req.setActive(active);
204      req.setType(type);
205      req.setInstitutionalGovernance(institutionalGovernance);
206      req.setDisciplines(disciplines);
207      req.setMasterSourceType(masterSourceType);
208      if (page != null) {
209        req.setLimit(page.getLimit());
210        req.setOffset(page.getOffset());
211      }
212      return req;
213    }
214  }
215}
