001package org.gbif.api.ws.mixin;
002
003import com.google.common.base.Predicate;
004import com.google.common.collect.ImmutableMap;
005import com.google.common.collect.Maps;
006import org.gbif.api.model.occurrence.Download;
007import org.gbif.api.model.occurrence.Occurrence;
008import org.gbif.api.model.registry.Dataset;
009import org.gbif.api.model.registry.search.DatasetSearchResult;
010
011import java.util.Map;
012
013/**
014 * Mixins are typically used to leave serialization-oriented annotations outside the models to avoid
015 * introducing coupling on a specific SerDe.
016 * <p>
017 * This class provides access to predefined mixins used in the GBIF web service application (client and server).
018 */
019public class Mixins {
020
021  // utility class
022  private Mixins() {
023  }
024
025  private static final ImmutableMap<Class<?>, Class<?>> PREDEFINED_MIXINS =
026    ImmutableMap.of(
027      Dataset.class, DatasetMixin.class,
028      DatasetSearchResult.class, DatasetMixin.class,
029      Download.class, LicenseMixin.class,
030      Occurrence.class, OccurrenceMixin.class
031    );
032
033  /**
034   * Return an immutable map of the predefined Jackson Mixins used by the web services.
035   *
036   * @return immutable map of the predefined Jackson Mixins
037   */
038  public static Map<Class<?>, Class<?>> getPredefinedMixins() {
039    return PREDEFINED_MIXINS;
040  }
041
042  /**
043   * Get an immutable map of the predefined Jackson Mixins after the provided keyFilter is applied.
044   *
045   * @param keyFilter predicate to filter the predefined mixins based on the model class
046   * @return immutable map of the predefined Jackson Mixins after applying the predicate
047   */
048  public static Map<Class<?>, Class<?>> getPredefinedMixins(Predicate<Class<?>> keyFilter) {
049    return ImmutableMap.copyOf(Maps.filterKeys(PREDEFINED_MIXINS, keyFilter));
050  }
051}
