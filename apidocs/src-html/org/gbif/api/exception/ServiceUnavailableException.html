016package org.gbif.api.exception;
017
018
019/**
020 * An unchecked exception that indicates that the service is not available.
021 * <p/>
022 * As this class is a runtime exception, there is no need for user code to catch it or
023 * subclasses if any error is to be considered fatal (the usual case).
024 */
025public class ServiceUnavailableException extends RuntimeException {
026
027  private static final long serialVersionUID = 9148212047393750135L;
028
029  /**
030   * Constructor for ServiceUnavailableException.
031   *
032   * @param message The message for the failure
033   * @param cause   The cause of the failure
034   */
035  public ServiceUnavailableException(String message, Throwable cause) {
036    super(message, cause);
037  }
038
039  /**
040   * @param message The message for the failure
041   */
042  public ServiceUnavailableException(String message) {
043    super(message);
044  }
045
046}
