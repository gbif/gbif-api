013package org.gbif.api.service.occurrence;
014
015import org.gbif.api.model.occurrence.DownloadRequest;
016
017import java.io.File;
018import java.io.InputStream;
019
020import javax.annotation.Nullable;
021import javax.validation.constraints.NotNull;
022
023/**
024 * Provides services to manage download requests and retrieve the download file. The services provided are:
025 * <ul>
026 * <li><strong>request</strong> inits a new download and assigns it a key</li> *
027 * <li><strong>cancel</strong> cancels an existing download</li>
028 * <li><strong>getResult</strong> retrieves the download file stream, if it is ready</li>
029 * <li><strong>getResultFile</strong> retrieves the download file, if it is ready</li>
030 * </ul>
031 */
032public interface DownloadRequestService {
033
034  /**
035   * Cancels running download. If the download is not running throws an IllegalStateException.
036   */
037  void cancel(@NotNull String downloadKey);
038
039  /**
040   * Creates a download request. If the request is successfully created its key is returned.
041   */
042  String create(@NotNull DownloadRequest downloadRequest);
043
044  /**
045   * @param downloadKey of the corresponding download request
046   * @return the input stream of the zipped download result file or null if it's not existing or ready yet
047   */
048  @Nullable
049  InputStream getResult(String downloadKey);
050
051  /**
052   * @param downloadKey of the corresponding download request
053   * @return the zipped download result file or null if it's not existing or ready yet
054   */
055  @Nullable
056  File getResultFile(String downloadKey);
057}
