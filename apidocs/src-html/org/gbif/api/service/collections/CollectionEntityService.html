014package org.gbif.api.service.collections;
015
016import org.gbif.api.model.collections.CollectionEntity;
017import org.gbif.api.model.collections.MasterSourceMetadata;
018import org.gbif.api.model.common.paging.Pageable;
019import org.gbif.api.model.common.paging.PagingResponse;
020import org.gbif.api.service.registry.CommentService;
021import org.gbif.api.service.registry.IdentifierService;
022import org.gbif.api.service.registry.MachineTagService;
023import org.gbif.api.service.registry.TagService;
024import org.gbif.api.vocabulary.collections.Source;
025
026import java.util.List;
027import java.util.Optional;
028import java.util.UUID;
029
030import javax.annotation.Nullable;
031import javax.validation.Valid;
032import javax.validation.constraints.NotNull;
033
034public interface CollectionEntityService<T extends CollectionEntity>
035    extends CrudService<T>,
036        IdentifierService,
037        TagService,
038        MachineTagService,
039        CommentService,
040        ContactService,
041        OccurrenceMappingService {
042
043  /**
044   * Lists the entities having a machine tag in the provided namespace, with the provided name and
045   * value.
046   *
047   * @return a pageable response of collection entities, with accurate counts for the machine tag
048   *     provided
049   */
050  PagingResponse<T> listByMachineTag(
051      String namespace, @Nullable String name, @Nullable String value, @Nullable Pageable page);
052
053  /** Replaces a entity with another. The entity replaced is also deleted. */
054  void replace(UUID entityToReplaceKey, UUID replacementKey);
055
056  /**
057   * Adds {@link MasterSourceMetadata} to an entity.
058   *
059   * @param targetEntityKey key of the entity to add the metadata to
060   * @param masterSourceMetadata metadata to add
061   * @return key of the created metadata
062   */
063  int addMasterSourceMetadata(UUID targetEntityKey, MasterSourceMetadata masterSourceMetadata);
064
065  /**
066   * Removes the {@link MasterSourceMetadata} from an entity.
067   *
068   * @param targetEntityKey key of the entity whose metadata will be deleted
069   */
070  void deleteMasterSourceMetadata(UUID targetEntityKey);
071
072  /**
073   * Returns the {@link MasterSourceMetadata} of the entity.
074   *
075   * @param targetEntityKey key of the entity
076   * @return {@link MasterSourceMetadata}
077   */
078  MasterSourceMetadata getMasterSourceMetadata(@NotNull UUID targetEntityKey);
079
080  /**
081   * Finds the collection entity whose master data metadata matches with the parameters received.
082   *
083   * @param source source of the metadata
084   * @param sourceId source Id of the metadata
085   * @return {@link Optional} with the collection entity found
086   */
087  List<T> findByMasterSource(Source source, String sourceId);
088
089  /**
090   * Updates an existing entity.
091   *
092   * @param entity that will replace the existing entity.
093   * @param lockFields indicates if fields that come from an external master source has to be
094   *     locked.
095   */
096  void update(@NotNull @Valid T entity, boolean lockFields);
097}
