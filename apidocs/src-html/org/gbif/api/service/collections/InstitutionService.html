001package org.gbif.api.service.collections;
002
003import org.gbif.api.model.collections.Institution;
004import org.gbif.api.model.common.paging.Pageable;
005import org.gbif.api.model.common.paging.PagingResponse;
006import org.gbif.api.service.registry.IdentifierService;
007import org.gbif.api.service.registry.TagService;
008
009import java.util.UUID;
010import javax.annotation.Nullable;
011
012/**
013 * Service for institutions in the collections context.
014 */
015public interface InstitutionService
016  extends CrudService<Institution>, ContactService, TagService, IdentifierService {
017
018  /**
019   * Pages {@link Institution} entities based on the parameters received.
020   *
021   * To iterate over <em>all</em> entities you can use code like this:
022   * {@code
023   * PagingRequest req = new PagingRequest();
024   * PagingResponse<T> response;
025   * do {
026   *   response = service.list(req);
027   *   for (T obj : response.getResults()) {
028   *     doStuff();
029   *   }
030   *   req.nextPage();
031   * } while (!response.isEndOfRecords());
032   * }
033   *
034   * @param query          to make a full text search
035   * @param contactKey     to filter by a contact
036   * @param page           paging parameters
037   *
038   * @return a list of entities ordered by their creation date, newest coming first
039   */
040  PagingResponse<Institution> list(@Nullable String query, @Nullable UUID contactKey, @Nullable Pageable page);
041
042}
