001package org.gbif.api.service.collections;
002
003import org.gbif.api.model.collections.Person;
004import org.gbif.api.model.common.paging.Pageable;
005import org.gbif.api.model.common.paging.PagingResponse;
006import org.gbif.api.model.registry.search.collections.PersonSuggestResult;
007
008import java.util.List;
009import java.util.UUID;
010import javax.annotation.Nullable;
011
012/**
013 * API Service for the collections-related staff.
014 */
015public interface PersonService extends CrudService<Person> {
016
017  /**
018   * Pages {@link Person} entities based on the parameters received.
019   *
020   * To iterate over <em>all</em> entities you can use code like this:
021   * {@code
022   * PagingRequest req = new PagingRequest();
023   * PagingResponse<T> response;
024   * do {
025   *   response = service.list(req);
026   *   for (T obj : response.getResults()) {
027   *     doStuff();
028   *   }
029   *   req.nextPage();
030   * } while (!response.isEndOfRecords());
031   * }
032   *
033   * @param query          to make a full text search
034   * @param institutionKey key of an institution to filter by
035   * @param collectionKey  key of a collection to filter by
036   * @param page           paging parameters
037   *
038   * @return a list of entities ordered by their creation date, newest coming first
039   */
040  PagingResponse<Person> list(@Nullable String query, @Nullable UUID institutionKey, @Nullable UUID collectionKey, @Nullable Pageable page);
041
042  /**
043   * Provides access to deleted persons.
044   */
045  PagingResponse<Person> listDeleted(@Nullable Pageable page);
046
047  /**
048   * Provides a simple suggest service.
049   */
050  List<PersonSuggestResult> suggest(@Nullable String q);
051}
