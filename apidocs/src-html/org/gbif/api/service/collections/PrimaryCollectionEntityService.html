014package org.gbif.api.service.collections;
015
016import org.gbif.api.model.collections.MasterSourceMetadata;
017import org.gbif.api.model.collections.PrimaryCollectionEntity;
018import org.gbif.api.vocabulary.collections.Source;
019
020import java.util.List;
021import java.util.Optional;
022import java.util.UUID;
023
024import javax.validation.Valid;
025import javax.validation.constraints.NotNull;
026
027public interface PrimaryCollectionEntityService<T extends PrimaryCollectionEntity>
028    extends CollectionEntityService<T>, ContactService, OccurrenceMappingService {
029
030  /** Replaces a entity with another. The entity replaced is also deleted. */
031  void replace(UUID entityToReplaceKey, UUID replacementKey);
032
033  /**
034   * Adds {@link MasterSourceMetadata} to an entity.
035   *
036   * @param targetEntityKey key of the entity to add the metadata to
037   * @param masterSourceMetadata metadata to add
038   * @return key of the created metadata
039   */
040  int addMasterSourceMetadata(UUID targetEntityKey, MasterSourceMetadata masterSourceMetadata);
041
042  /**
043   * Removes the {@link MasterSourceMetadata} from an entity.
044   *
045   * @param targetEntityKey key of the entity whose metadata will be deleted
046   */
047  void deleteMasterSourceMetadata(UUID targetEntityKey);
048
049  /**
050   * Returns the {@link MasterSourceMetadata} of the entity.
051   *
052   * @param targetEntityKey key of the entity
053   * @return {@link MasterSourceMetadata}
054   */
055  MasterSourceMetadata getMasterSourceMetadata(@NotNull UUID targetEntityKey);
056
057  /**
058   * Finds the collection entity whose master data metadata matches with the parameters received.
059   *
060   * @param source source of the metadata
061   * @param sourceId source Id of the metadata
062   * @return {@link Optional} with the collection entity found
063   */
064  List<T> findByMasterSource(Source source, String sourceId);
065
066  /**
067   * Updates an existing entity.
068   *
069   * @param entity that will replace the existing entity.
070   * @param lockFields indicates if fields that come from an external master source has to be
071   *     locked.
072   */
073  void update(@NotNull @Valid T entity, boolean lockFields);
074}
