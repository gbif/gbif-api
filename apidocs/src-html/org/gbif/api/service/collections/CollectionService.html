001package org.gbif.api.service.collections;
002
003import org.gbif.api.model.collections.Collection;
004import org.gbif.api.model.common.paging.Pageable;
005import org.gbif.api.model.common.paging.PagingResponse;
006import org.gbif.api.model.registry.search.collections.KeyCodeNameResult;
007import org.gbif.api.service.registry.IdentifierService;
008import org.gbif.api.service.registry.TagService;
009
010import java.util.List;
011import java.util.UUID;
012import javax.annotation.Nullable;
013
014/**
015 * API Service to work with collections.
016 */
017public interface CollectionService
018    extends CrudService<Collection>, ContactService, TagService, IdentifierService {
019
020  /**
021   * Pages {@link Collection} entities based on the parameters received.
022   *
023   * To iterate over <em>all</em> entities you can use code like this:
024   * {@code
025   * PagingRequest req = new PagingRequest();
026   * PagingResponse<T> response;
027   * do {
028   *   response = service.list(req);
029   *   for (T obj : response.getResults()) {
030   *     doStuff();
031   *   }
032   *   req.nextPage();
033   * } while (!response.isEndOfRecords());
034   * }
035   *
036   * @param query          to make a full text search
037   * @param institutionKey key of an institution to filter by
038   * @param contactKey     to filter by a contact
039   * @param page           paging parameters
040   *
041   * @return a list of entities ordered by their creation date, newest coming first
042   */
043  PagingResponse<Collection> list(@Nullable String query, @Nullable UUID institutionKey, @Nullable UUID contactKey, @Nullable Pageable page);
044
045  /**
046   * Provides access to deleted collections.
047   */
048  PagingResponse<Collection> listDeleted(@Nullable Pageable page);
049
050  /**
051   * Provides a simple suggest service.
052   */
053  List<KeyCodeNameResult> suggest(@Nullable String q);
054}
