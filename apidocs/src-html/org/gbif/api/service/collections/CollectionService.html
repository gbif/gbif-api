016package org.gbif.api.service.collections;
017
018import org.gbif.api.model.collections.Collection;
019import org.gbif.api.model.common.paging.Pageable;
020import org.gbif.api.model.common.paging.PagingResponse;
021import org.gbif.api.model.registry.search.collections.KeyCodeNameResult;
022import org.gbif.api.service.registry.IdentifierService;
023import org.gbif.api.service.registry.MachineTagService;
024import org.gbif.api.service.registry.TagService;
025
026import java.util.List;
027import java.util.UUID;
028
029import javax.annotation.Nullable;
030
031/**
032 * API Service to work with collections.
033 */
034public interface CollectionService
035    extends CrudService<Collection>, ContactService, TagService, IdentifierService, MachineTagService {
036
037  /**
038   * Pages {@link Collection} entities based on the parameters received.
039   *
040   * <p>To iterate over <em>all</em> entities you can use code like this: {@code PagingRequest req =
041   * new PagingRequest(); PagingResponse<T> response; do { response = service.list(req); for (T obj
042   * : response.getResults()) { doStuff(); } req.nextPage(); } while (!response.isEndOfRecords()); }
043   *
044   * @param query to make a full text search
045   * @param institutionKey key of an collection to filter by
046   * @param contactKey to filter by a contact
047   * @param code code of the collection
048   * @param name name of the collection
049   * @param alternativeCode alternative code of the collection
050   * @param page paging parameters
051   * @return a list of entities ordered by their creation date, newest coming first
052   */
053  PagingResponse<Collection> list(
054      @Nullable String query,
055      @Nullable UUID institutionKey,
056      @Nullable UUID contactKey,
057      @Nullable String code,
058      @Nullable String name,
059      @Nullable String alternativeCode,
060      @Nullable Pageable page);
061
062  /**
063   * Provides access to deleted collections.
064   */
065  PagingResponse<Collection> listDeleted(@Nullable Pageable page);
066
067  /**
068   * Provides a simple suggest service.
069   */
070  List<KeyCodeNameResult> suggest(@Nullable String q);
071}
