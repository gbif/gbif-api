001package org.gbif.api.service.registry;
002
003import java.util.UUID;
004
005public interface PrimaryIdentifierService extends IdentifierService {
006
007  /**
008   * Updates the specified Identifier for a target entity, setting its primary status.
009   *
010   * @param targetEntityKey the key of the target entity to which the identifier belongs
011   * @param identifierKey the key of the identifier to be updated
012   * @param isPrimary a boolean indicating whether the identifier should be set as primary
013   *                  (true) or non-primary (false)
014   *
015   * @return the key of the updated Identifier
016   */
017  int updateIdentifier(UUID targetEntityKey, int identifierKey, boolean isPrimary);
018}
