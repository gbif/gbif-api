014package org.gbif.api.service.registry;
015
016import org.gbif.api.model.common.paging.Pageable;
017import org.gbif.api.model.common.paging.PagingResponse;
018import org.gbif.api.model.registry.DatasetOccurrenceDownloadUsage;
019
020import java.util.UUID;
021
022import javax.annotation.Nullable;
023import javax.validation.constraints.NotNull;
024
025/**
026 * Interface to access and persists information about occurrence download events. This interface was
027 * initially implemented as a internal GBIF component that shouldn't be exposed for the public use.
028 */
029@SuppressWarnings("unused")
030public interface DatasetOccurrenceDownloadUsageService {
031  /** Retrieves a pageable result of the downloads which contain data taken from a dataset. */
032  PagingResponse<DatasetOccurrenceDownloadUsage> listByDataset(
033      @NotNull UUID datasetKey, Boolean showDownloadDetails, @Nullable Pageable page);
034}
