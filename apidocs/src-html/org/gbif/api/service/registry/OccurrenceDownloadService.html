014package org.gbif.api.service.registry;
015
016import org.gbif.api.model.common.paging.Pageable;
017import org.gbif.api.model.common.paging.PagingResponse;
018import org.gbif.api.model.occurrence.Download;
019import org.gbif.api.model.occurrence.DownloadStatistics;
020import org.gbif.api.model.registry.DatasetOccurrenceDownloadUsage;
021import org.gbif.api.vocabulary.Country;
022
023import java.util.Date;
024import java.util.Map;
025import java.util.Set;
026import java.util.UUID;
027
028import javax.annotation.Nullable;
029import javax.validation.Valid;
030import javax.validation.constraints.NotNull;
031
032/**
033 * Interface to access and persist information about occurrence download events.
034 */
035@SuppressWarnings("unused")
036public interface OccurrenceDownloadService {
037
038  /**
039   * Persists the occurrence download object. The object must contain a unique key, the persistence
040   * storage doesn't generate one for it.
041   */
042  void create(@NotNull @Valid Download download);
043
044  /**
045   * Retrieves a occurrence download by its unique key or DOI.
046   */
047  Download get(@NotNull String keyOrDoi);
048
049  /**
050   * Retrieves a pageable result of all the downloads, optionally the downloads can be filtered by
051   * status.
052   */
053  PagingResponse<Download> list(@Nullable Pageable page, @Nullable Set<Download.Status> status);
054
055  /**
056   * Retrieves a pageable result of the downloads created by a user in a given status.
057   */
058  PagingResponse<Download> listByUser(@NotNull String user, @Nullable Pageable page,
059    @Nullable Set<Download.Status> status);
060
061  /**
062   * Retrieves a pageable result of the downloads created by a user in a given status.
063   *
064   * Internal use only; behaviour may change without notice.
065   */
066  PagingResponse<Download> listByEraseAfter(@Nullable Pageable page,
067                                            @Nullable String eraseAfter,
068                                            @Nullable Long size,
069                                            @Nullable String erasureNotification);
070
071  /**
072   * Update an existing occurrence download.
073   */
074  void update(@NotNull @Valid Download download);
075
076  /**
077   * Retrieves a pageable result of the dataset usages in a occurrence download.
078   * <p>
079   * The Downloads in the DatasetOccurrenceDownloadUsages are null, to avoid redundant repetition of
080   * potentially large objects.
081   */
082  PagingResponse<DatasetOccurrenceDownloadUsage> listDatasetUsages(@NotNull String keyOrDoi,
083    @Nullable Pageable page);
084
085  /**
086   * Retrieve citation details of a download by its unique key or DOI.
087   */
088  String getCitation(@NotNull String keyOrDoi);
089
090  /**
091   * Retrieves downloads monthly stats by country (user and publishing country) and dataset.
092   */
093  Map<Integer, Map<Integer, Long>> getDownloadsByUserCountry(@Nullable Date fromDate,
094    @Nullable Date toDate,
095    @Nullable Country userCountry);
096
097  /**
098   * Retrieves downloaded records monthly stats by country (user and publishing country) and dataset.
099   */
100  Map<Integer, Map<Integer, Long>> getDownloadedRecordsByDataset(@Nullable Date fromDate,
101    @Nullable Date toDate,
102    @Nullable Country publishingCountry,
103    @Nullable UUID datasetKey,
104    @Nullable UUID publishingOrgKey);
105
106  /**
107   * Retrieves downloads monthly stats by country (user and publishing country) and dataset.
108   */
109  Map<Integer, Map<Integer, Long>> getDownloadsByDataset(@Nullable Date fromDate,
110                                                         @Nullable Date toDate,
111                                                         @Nullable Country publishingCountry,
112                                                         @Nullable UUID datasetKey,
113                                                         @Nullable UUID publishingOrgKey);
114
115  /**
116   * Retrieves downloads monthly stats by country (user and publishing country) and dataset.
117   */
118  PagingResponse<DownloadStatistics> getDownloadStatistics(@Nullable Date fromDate,
119                                                           @Nullable Date toDate,
120                                                           @Nullable Country publishingCountry,
121                                                           @Nullable UUID datasetKey,
122                                                           @Nullable UUID publishingOrgKey,
123                                                           @Nullable Pageable page);
124
125  /**
126   * Persists usages of datasets in an occurrence download.
127   *
128   * @param downloadKey      downloadkey of the datasets' usage information.
129   * @param datasetCitations map of datasetkey as key and number of records as value.
130   */
131  void createUsages(@NotNull String downloadKey, @NotNull Map<UUID, Long> datasetCitations);
132}
