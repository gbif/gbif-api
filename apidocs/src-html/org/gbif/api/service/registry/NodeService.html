013package org.gbif.api.service.registry;
014
015import org.gbif.api.model.common.paging.Pageable;
016import org.gbif.api.model.common.paging.PagingResponse;
017import org.gbif.api.model.registry.Dataset;
018import org.gbif.api.model.registry.Installation;
019import org.gbif.api.model.registry.Node;
020import org.gbif.api.model.registry.Organization;
021import org.gbif.api.model.registry.search.KeyTitleResult;
022import org.gbif.api.vocabulary.Country;
023
024import java.util.List;
025import java.util.UUID;
026import javax.annotation.Nullable;
027import javax.validation.constraints.NotNull;
028
029/**
030 * Actions on a GBIF node.
031 * Be aware that of 2013 Nodes DO NOT support the ContactInterface although advertised in this interface!!!
032 */
033public interface NodeService extends NetworkEntityService<Node> {
034
035  /**
036   * Provides access to the organizations endorsed by a single node.
037   */
038  PagingResponse<Organization> endorsedOrganizations(@NotNull UUID nodeKey, @Nullable Pageable page);
039
040  /**
041   * Provides access to the organizations that are awaiting an endorsement approval.
042   */
043  PagingResponse<Organization> pendingEndorsements(@Nullable Pageable page);
044
045  /**
046   * Provides access to the organizations that are awaiting an endorsement approval for the given node.
047   */
048  PagingResponse<Organization> pendingEndorsements(@NotNull UUID nodeKey, @Nullable Pageable page);
049
050  /**
051   * Provides the installations that are registered to organizations with an approved endorsement from the node.
052   */
053  PagingResponse<Installation> installations(@NotNull UUID nodeKey, @Nullable Pageable page);
054
055  /**
056   * Returns a node for a given country.
057   * 
058   * @return the countries node or null if none exists
059   */
060  Node getByCountry(Country country);
061
062  /**
063   * Returns a list of all countries which do have a GBIF node.
064   * 
065   * @return list of distinct countries having a GBIF node
066   */
067  List<Country> listNodeCountries();
068
069  /**
070   * Returns those countries considered active in GBIF.
071   * To be active a country must have a Node of type country present, that is either of voting or associate status.
072   *  @return A list of countries ordered by iso code
073    */
074   List<Country> listActiveCountries();
075
076  /**
077   * Provides paging service to list datasets published, i.e. owned by organizations endorsed by the given node.
078   * 
079   * @return list of datasets ordered by creation date with latest coming first
080   */
081  PagingResponse<Dataset> endorsedDatasets(@NotNull UUID nodeKey, @Nullable Pageable page);
082
083  /**
084   * Provides a simple suggest service.
085   */
086  List<KeyTitleResult> suggest(@Nullable String q);
087}
