001package org.gbif.api.vocabulary;
002
003/**
004 * Enumeration of issues for each name usage record encountered during checklist processing.
005 */
006public enum NameUsageIssue {
007  PARENT_NAME_USAGE_ID_INVALID,
008  ACCEPTED_NAME_USAGE_ID_INVALID,
009  ORIGINAL_NAME_USAGE_ID_INVALID,
010
011  /**
012   * dwc:taxonRank could not be interpreted
013   */
014  RANK_INVALID,
015
016  /**
017   * dwc:nomenclaturalStatus could not be interpreted
018   */
019  NOMENCLATURAL_STATUS_INVALID,
020
021  /**
022   * dwc:taxonomicStatus could not be interpreted
023   */
024  TAXONOMIC_STATUS_INVALID,
025
026  /**
027   * The scientific name was assembled from the individual name parts and not given as a whole string.
028   */
029  SCIENTIFIC_NAME_ASSEMBLED,
030
031  /**
032   * If a synonym points to another synonym as its accepted taxon the chain is resolved.
033   */
034  CHAINED_SYNOYM,
035
036  /**
037   * The authorship of the original name does not match the authorshipin brackets of the actual name.
038   */
039  BASIONYM_AUTHOR_MISMATCH,
040
041  TAXONOMIC_STATUS_MISMATCH,
042
043  /**
044   * The child parent classification resulted into a cycle that needed to be resolved/cut.
045   */
046  PARENT_CYCLE,
047
048  CLASSIFICATION_RANK_ORDER_INVALID,
049
050  /**
051   * The denormalized classification could not be applied to the name usage.
052   * For example if the id based classification has no ranks.
053   */
054  CLASSIFICATION_NOT_APPLIED,
055
056  /**
057   * At least one vernacular name extension record attached to this name usage is invalid.
058   */
059  VERNACULAR_NAME_INVALID,
060
061  /**
062   * At least one description extension record attached to this name usage is invalid.
063   */
064  DESCRIPTION_INVALID,
065
066  /**
067   * At least one distribution extension record attached to this name usage is invalid.
068   */
069  DISTRIBUTION_INVALID,
070
071  /**
072   * At least one species profile extension record attached to this name usage is invalid.
073   */
074  SPECIES_PROFILE_INVALID,
075
076  /**
077   * At least one multimedia extension record attached to this name usage is invalid.
078   * This covers multimedia coming in through various extensions including
079   * Audubon core, Simple images or multimedia or EOL media.
080   */
081  MULTIMEDIA_INVALID,
082
083  /**
084   * At least one bibliographic reference extension record attached to this name usage is invalid.
085   */
086  BIB_REFERENCE_INVALID,
087
088  /**
089   * At least one alternative identifier extension record attached to this name usage is invalid.
090   */
091  ALT_IDENTIFIER_INVALID,
092
093  /**
094   * Name usage could not be matched to the GBIF backbone.
095   */
096  BACKBONE_MATCH_NONE,
097
098  /**
099   * Name usage could only be matched to the GBIF backbone using fuzzy matching.
100   */
101  BACKBONE_MATCH_FUZZY
102}
