001package org.gbif.api.vocabulary;
002
003/**
004 * Enumeration of issues for each name usage record encountered during checklist processing.
005 */
006public enum NameUsageIssue {
007  /**
008   * The value for dwc:parentNameUsageID could not be resolved.
009   */
010  PARENT_NAME_USAGE_ID_INVALID,
011
012  /**
013   * The value for dwc:acceptedNameUsageID could not be resolved.
014   */
015  ACCEPTED_NAME_USAGE_ID_INVALID,
016
017  /**
018   * The value for dwc:originalNameUsageID could not be resolved.
019   */
020  ORIGINAL_NAME_USAGE_ID_INVALID,
021
022  /**
023   * Synonym lacking an accepted name.
024   */
025  ACCEPTED_NAME_MISSING,
026
027  /**
028   * dwc:taxonRank could not be interpreted
029   */
030  RANK_INVALID,
031
032  /**
033   * dwc:nomenclaturalStatus could not be interpreted
034   */
035  NOMENCLATURAL_STATUS_INVALID,
036
037  /**
038   * dwc:taxonomicStatus could not be interpreted
039   */
040  TAXONOMIC_STATUS_INVALID,
041
042  /**
043   * The scientific name was assembled from the individual name parts and not given as a whole string.
044   */
045  SCIENTIFIC_NAME_ASSEMBLED,
046
047  /**
048   * If a synonym points to another synonym as its accepted taxon the chain is resolved.
049   */
050  CHAINED_SYNOYM,
051
052  /**
053   * The authorship of the original name does not match the authorship in brackets of the actual name.
054   */
055  BASIONYM_AUTHOR_MISMATCH,
056
057  TAXONOMIC_STATUS_MISMATCH,
058
059  /**
060   * The child parent classification resulted into a cycle that needed to be resolved/cut.
061   */
062  PARENT_CYCLE,
063
064  CLASSIFICATION_RANK_ORDER_INVALID,
065
066  /**
067   * The denormalized classification could not be applied to the name usage.
068   * For example if the id based classification has no ranks.
069   */
070  CLASSIFICATION_NOT_APPLIED,
071
072  /**
073   * At least one vernacular name extension record attached to this name usage is invalid.
074   */
075  VERNACULAR_NAME_INVALID,
076
077  /**
078   * At least one description extension record attached to this name usage is invalid.
079   */
080  DESCRIPTION_INVALID,
081
082  /**
083   * At least one distribution extension record attached to this name usage is invalid.
084   */
085  DISTRIBUTION_INVALID,
086
087  /**
088   * At least one species profile extension record attached to this name usage is invalid.
089   */
090  SPECIES_PROFILE_INVALID,
091
092  /**
093   * At least one multimedia extension record attached to this name usage is invalid.
094   * This covers multimedia coming in through various extensions including
095   * Audubon core, Simple images or multimedia or EOL media.
096   */
097  MULTIMEDIA_INVALID,
098
099  /**
100   * At least one bibliographic reference extension record attached to this name usage is invalid.
101   */
102  BIB_REFERENCE_INVALID,
103
104  /**
105   * At least one alternative identifier extension record attached to this name usage is invalid.
106   */
107  ALT_IDENTIFIER_INVALID,
108
109  /**
110   * Name usage could not be matched to the GBIF backbone.
111   */
112  BACKBONE_MATCH_NONE,
113
114  /**
115   * Name usage could only be matched to the GBIF backbone using fuzzy matching.
116   * @deprecated because there should be no fuzzy matching being used anymore for matching checklist names
117   */
118  @Deprecated
119  BACKBONE_MATCH_FUZZY,
120
121  /**
122   * Synonym has a verbatim accepted name which is not unique and refers to several records.
123   */
124  ACCEPTED_NAME_NOT_UNIQUE,
125
126  /**
127   * Record has a verbatim parent name which is not unique and refers to several records.
128   */
129  PARENT_NAME_NOT_UNIQUE,
130
131  /**
132   * Record has a verbatim original name (basionym) which is not unique and refers to several records.
133   */
134  ORIGINAL_NAME_NOT_UNIQUE,
135
136  /**
137   * There were problems representing all name usage relationships,
138   * i.e. the link to the parent, accepted and/or original name.
139   * The interpreted record in ChecklistBank is lacking some of the original source relation.
140   */
141  RELATIONSHIP_MISSING,
142
143  /**
144   * Record has a original name (basionym) relationship which was derived from name & authorship comparison, but did not exist explicitly in the data.
145   * This should only be flagged in programmatically generated GBIF backbone usages.
146   * GBIF backbone specific issue.
147   */
148  ORIGINAL_NAME_DERIVED,
149
150  /**
151   * There have been more than one accepted name in a homotypical basionym group of names.
152   * GBIF backbone specific issue.
153   */
154  CONFLICTING_BASIONYM_COMBINATION,
155
156  /**
157   * The group (currently only genera are tested) are lacking any accepted species
158   * GBIF backbone specific issue.
159   */
160  NO_SPECIES,
161
162  /**
163   * The (accepted) bi/trinomial name does not match the parent name and should be recombined into the parent genus/species.
164   * For example the species Picea alba with a parent genus Abies is a mismatch and should be replaced by Abies alba.
165   * GBIF backbone specific issue.
166   */
167  NAME_PARENT_MISMATCH,
168
169  /**
170   * A potential orthographic variant exists in the backbone.
171   * GBIF backbone specific issue.
172   */
173  ORTHOGRAPHIC_VARIANT,
174
175  /**
176   * A not synonymized homonym exists for this name in some other backbone source which have been ignored at build time.
177   */
178  HOMONYM,
179
180  /**
181   * A bi/trinomial name published earlier than the parent genus was published.
182   * This might indicate that the name should rather be a recombination.
183   */
184  PUBLISHED_BEFORE_GENUS
185}
