003/**
004 * Mechanism of preservation applied to a collection.
005 *
006 * Note: The descriptions are the original values migrated from GRBIO.
007 */
008public enum PreservationType {
009
010  STORAGE_OUTDOORS("Storage Environment: Ambient uncontrolled environment (outdoors)"),
011  STORAGE_INDOORS("Storage Environment: Ambient controlled environment (indoors)"),
012  STORAGE_CONTROLLED_ATMOSPHERE("Storage Environment: Controlled atmosphere (N, C02, humidity)"),
013  STORAGE_FROZEN_MINUS_20("Storage Environment: Frozen (-20)"),
014  STORAGE_FROZEN_MINUS_80("Storage Environment: Frozen (-80)"),
015  STORAGE_FROZEN_BETWEEN_MINUS_132_AND_MINUS_196("Storage Environment: Frozen (-132 - -196)"),
016  STORAGE_OTHER("Storage Environment: Other (please define)"),
017  STORAGE_REFRIGERATED("Storage Environment: Refrigerated (+4)"),
018  STORAGE_RECORDED("Storage Environment: Recorded (digital, paper, film, audio, etc.)"),
019  STORAGE_VACUUM("Storage Environment: Vacuum"),
020  SAMPLE_CRYOPRESERVED("Sample Treatment: Cryopreserved"),
021  SAMPLE_DRIED("Sample Treatment: Dried"),
022  SAMPLE_EMBEDDED("Sample Treatment: Embedded"),
023  SAMPLE_FLUID_PRESERVED("Sample Treatment: Fluid preserved"),
024  SAMPLE_PINNED("Sample Treatment: Pinned"),
025  SAMPLE_PRESSED("Sample Treatment: Pressed"),
026  SAMPLE_SKELETONIZED("Sample Treatment: Skeletonized"),
027  SAMPLE_SLIDE_MOUNT("Sample Treatment: Slide mount"),
028  SAMPLE_SURFACE_COATING("Sample Treatment: Surface coating"),
029  SAMPLE_TANNED("Sample Treatment: Tanned"),
030  SAMPLE_WAX_BLOCK("Sample Treatment: Wax Block"),
031  SAMPLE_OTHER("Sample Treatment: Other (please define)");
032
033  private String description;
034
035  PreservationType(String description) {
036    this.description = description;
037  }
038
039  public String getDescription() {
040    return description;
041  }
042}
