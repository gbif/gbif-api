016package org.gbif.api.vocabulary;
017
018import org.gbif.api.util.VocabularyUtils;
019
020import java.util.HashMap;
021import java.util.Map;
022import javax.annotation.Nullable;
023
024import com.google.common.collect.ImmutableMap;
025
026/**
027 * Enumeration for all contact types.
028 * A utility to infer types is provided which has historically been used during data migration activities from legacy
029 * systems such as the previous "GBRDS" which was built on MySQL.
030 *
031 * @see <a href="http://rs.gbif.org/vocabulary/gbif/agent_role.xml">The IPT vocabulary</a>
032 */
033public enum ContactType {
034
035  TECHNICAL_POINT_OF_CONTACT,
036  ADMINISTRATIVE_POINT_OF_CONTACT,
037  POINT_OF_CONTACT,
038  ORIGINATOR,
039  METADATA_AUTHOR,
040  PRINCIPAL_INVESTIGATOR,
041  AUTHOR,
042  CONTENT_PROVIDER,
043  CUSTODIAN_STEWARD,
044  DISTRIBUTOR,
045  EDITOR,
046  OWNER,
047  PROCESSOR,
048  PUBLISHER,
049  USER,
050  PROGRAMMER,
051  DATA_ADMINISTRATOR,
052  SYSTEM_ADMINISTRATOR,
053  HEAD_OF_DELEGATION,
054  TEMPORARY_HEAD_OF_DELEGATION,
055  ADDITIONAL_DELEGATE,
056  TEMPORARY_DELEGATE,
057  REGIONAL_NODE_REPRESENTATIVE,
058  NODE_MANAGER,
059  NODE_STAFF;
060
061  /**
062   * @return the matching ContactType or null
063   */
064  public static ContactType fromString(String contactType) {
065    return (ContactType) VocabularyUtils.lookupEnum(contactType, ContactType.class);
066  }
067
068  // deliberate typos have existed and might be in genuine use
069  private static final ImmutableMap<String, ContactType> TYPE_LOOKUP = ImmutableMap.<String, ContactType> builder()
070    .put("administrative", ADMINISTRATIVE_POINT_OF_CONTACT)
071    .put("technical", TECHNICAL_POINT_OF_CONTACT)
072    .put("pointofcontact", POINT_OF_CONTACT)
073    .put("originator", ORIGINATOR)
074    .put("metadataprovider", METADATA_AUTHOR)
075    .put("principleinvestigator", PRINCIPAL_INVESTIGATOR)
076    .put("author", AUTHOR)
077    .put("contentprovider", CONTENT_PROVIDER)
078    .put("custodiansteward", CUSTODIAN_STEWARD)
079    .put("distributor", DISTRIBUTOR)
080    .put("editor", EDITOR)
081    .put("owner", OWNER)
082    .put("processor", PROCESSOR)
083    .put("publisher", PUBLISHER)
084    .put("user", USER)
085    .put("programmer", PROGRAMMER)
086    .put("data administrator", DATA_ADMINISTRATOR)
087    .put("system adminsitrator", SYSTEM_ADMINISTRATOR) // deliberate typo
088    .put("system administrator", SYSTEM_ADMINISTRATOR)
089  .build();
090
091  /**
092   * Tries its best to infer a ContactType from a given string. This can for example be used for the various contact
093   * types from DiGIR, TAPIR and BioCASe.
094   *
095   * @return the inferred ContactType
096   */
097  public static ContactType inferType(@Nullable String type) {
098    if (type != null) {
099      ContactType contactType = TYPE_LOOKUP.get(type.toLowerCase());
100      if (contactType != null) {
101        return contactType;
102      }
103      return fromString(type);
104    }
105    return null;
106  }
107}
