016package org.gbif.api.vocabulary;
017
018import org.gbif.api.util.VocabularyUtils;
019
020import java.util.Optional;
021
022import javax.annotation.Nullable;
023
024import org.apache.commons.lang3.StringUtils;
025
026/**
027 * Enumeration of the set of licenses GBIF supports for applying to a dataset. The license provides a standardised way
028 * to define appropriate uses of a dataset.
029 * </br>
030 * GBIF's recommended best practice is to use the most recent license version, which for CC-BY and CC-BY-NC is 4.0.
031 * This is in line with the recommendation from Creative Commons.
032 * The ordinal number in the Enum implicitly defines the level of restriction, see LicenseTest.
033 *
034 * @see <a href="https://creativecommons.org/faq/#why-should-i-use-the-latest-version-of-the-creative-commons-licenses">Creative
035 * Commons recommendation</a>
036 * @see <a href="http://www.gbif.org/terms/licences">GBIF Licensing</a>
037 */
038public enum License {
039
040  /**
041   * Creative Commons Zero / Public Domain version 1.0. Technically a waiver, not a license.
042   *
043   * @see <a href="http://creativecommons.org/publicdomain/zero/1.0/legalcode">legal document</a>
044   */
045  CC0_1_0("Public Domain (CC0 1.0)", "http://creativecommons.org/publicdomain/zero/1.0/legalcode"),
046  /**
047   * Creative Commons Attribution version 4.0.
048   *
049   * @see <a href="http://creativecommons.org/licenses/by/4.0/legalcode">legal document</a>
050   */
051  CC_BY_4_0("Creative Commons Attribution (CC-BY) 4.0", "http://creativecommons.org/licenses/by/4.0/legalcode"),
052  /**
053   * Creative Commons Attribution-NonCommercial version 4.0.
054   *
055   * @see <a href="http://creativecommons.org/licenses/by-nc/4.0/legalcode">legal document</a>
056   */
057  CC_BY_NC_4_0("Creative Commons Attribution Non Commercial (CC-BY-NC) 4.0", "http://creativecommons.org/licenses/by-nc/4.0/legalcode"),
058  /**
059   * No license has been specified.
060   */
061  UNSPECIFIED(null, null),
062  /**
063   * A license not supported by GBIF.
064   */
065  UNSUPPORTED(null, null);
066
067  private final String licenseTitle;
068  private final String licenseUrl;
069
070  /**
071   * Get an {@link License} from its name as String.
072   *
073   * @return instance of Optional, never null
074   */
075  public static Optional<License> fromString(String license) {
076    return VocabularyUtils.lookup(license, License.class);
077  }
078
079  /**
080   * Lookup a License by either its a) legal code URL or b) human readable summary URL, with HTTP or HTTPS.
081   * For any parsing see LicenseParser in GBIF parsers project.
082   *
083   * @param licenseUrl the case insensitive URL for the license.
084   *
085   * @return instance of Optional, never null
086   */
087  public static Optional<License> fromLicenseUrl(String licenseUrl) {
088    if (StringUtils.isNotEmpty(licenseUrl)) {
089      licenseUrl = licenseUrl.trim().toLowerCase();
090      licenseUrl = StringUtils.removeEnd(licenseUrl, "/");
091      licenseUrl = licenseUrl.replace("https://", "http://");
092
093      // do lookup by legal code URL or human readable summary URL (excluding "/legalcode")
094      for (License license : License.values()) {
095        if (license.getLicenseUrl() != null && (licenseUrl.equals(license.getLicenseUrl())
096                || license.getLicenseUrl().startsWith(licenseUrl))) {
097          return Optional.of(license);
098        }
099      }
100    }
101    return Optional.empty();
102  }
103
104  /**
105   * Get the most restrictive license between the 2 provided licenses.
106   * If one or the two licenses are null or not concrete, this method returns the fallback License.
107   *
108   * @param fallback License to return if one or the two licenses are null or not concrete
109   *
110   * @return the most restrictive License or the fallback License
111   */
112  public static License getMostRestrictive(License license1, License license2, License fallback) {
113    if (license1 == null || license2 == null || !license1.isConcrete() || !license2.isConcrete()) {
114      return fallback;
115    }
116    return (license1.compareTo(license2) > 0) ? license1 : license2;
117  }
118
119  License(@Nullable String licenseTitle, @Nullable String licenseUrl) {
120    this.licenseTitle = licenseTitle;
121    // ensure license information is kept in lowercase internally
122    if (licenseUrl != null) {
123      this.licenseUrl = licenseUrl.toLowerCase();
124    } else {
125      this.licenseUrl = null;
126    }
127  }
128
129  /**
130   * @return the License URL
131   */
132  public String getLicenseUrl() {
133    return licenseUrl;
134  }
135
136  /**
137   * @return the License title
138   */
139  public String getLicenseTitle() {
140    return licenseTitle;
141  }
142
143  /**
144   * Indicates if a license is a concrete license (true) or an abstracted license (false) like
145   * UNSPECIFIED or UNSUPPORTED.
146   * @return the license if concrete or not
147   */
148  public boolean isConcrete() {
149    return licenseUrl != null;
150  }
151}
