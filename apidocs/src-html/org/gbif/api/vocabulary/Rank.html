398  private final String marker;
399
400  Rank() {
401    this(null);
402  }
403
404  Rank(String marker) {
405    this.marker = marker;
406  }
407
408  public String getMarker() {
409    return marker;
410  }
411
412  /**
413   * @return true for infraspecific ranks.
414   */
415  public boolean isInfraspecific() {
416    return this != SPECIES && isSpeciesOrBelow();
417  }
418
419  /**
420   * @return true for infra subspecific ranks.
421   */
422  public boolean isInfrasubspecific() {
423    return ordinal() > SUBSPECIES.ordinal() && notOtherOrUnknown();
424  }
425
426  /**
427   * @return true for rank is below genus. Also incluse species and infraspecific ranks
428   */
429  public boolean isInfrageneric() {
430    return ordinal() > GENUS.ordinal() && notOtherOrUnknown();
431  }
432
433  /**
434   * @return true for real infrageneric ranks with an infragenericEpithet below genus and above species aggregate.
435   */
436  public boolean isInfragenericStrictly() {
437    return isInfrageneric() && ordinal() < SPECIES_AGGREGATE.ordinal();
438  }
439
440  /**
441   * True for all mayor Linnéan ranks, ie kingdom,phylum,class,order,family,genus and species.
442   */
443  public boolean isLinnean() {
444    for (Rank r : LINNEAN_RANKS) {
445      if (r == this) {
446        return true;
447      }
448    }
449    return false;
450  }
451
452  public boolean isSpeciesOrBelow() {
453    return ordinal() >= SPECIES.ordinal() && notOtherOrUnknown();
454  }
455
456  public boolean isSpeciesAggregateOrBelow() {
457    return ordinal() >= SPECIES_AGGREGATE.ordinal() && notOtherOrUnknown();
458  }
459
460  public boolean notOtherOrUnknown() {
461    return this != OTHER && this != UNRANKED;
462  }
463
464  /**
465   * @return true if the rank is above genus.
466   */
467  public boolean isSuprageneric() {
468    return ordinal() < GENUS.ordinal();
469  }
470
471  /**
472   * @return true if the rank is above rank species.
473   */
474  public boolean isSupraspecific() {
475    return ordinal() < SPECIES.ordinal();
476  }
477
478  /**
479   * True for names of informal ranks that represent a range of ranks really and therefore cannot safely be compared to
480   * other ranks in all cases.
481   * Example ranks are INFRASPECIFIC_NAME or INFRAGENERIC_NAME
482   *
483   * @return true if uncomparable
484   */
485  public boolean isUncomparable() {
486    return UNCOMPARABLE_RANKS.contains(this);
487  }
488
489  /**
490   * @return true if the rank is considered a legacy rank not used anymore in current nomenclature.
491   */
492  public boolean isLegacy() {
493    return LEGACY_RANKS.contains(this);
494  }
495
496  /**
497   * @return the nomenclatural code if the rank is restricted to just one code or null otherwise
498   */
499  public NomenclaturalCode isRestrictedToCode() {
500    return RANK2CODE.get(this);
501  }
502
503  /**
504   * @return true if this rank is higher than the given other
505   */
506  public boolean higherThan(Rank other) {
507    return ordinal() < other.ordinal();
508  }
