016package org.gbif.api.vocabulary;
017
018/**
019 * A simple enumeration of all kingdoms found in the GBIF Backbone Taxonomy and also Catalogue of Life.
020 */
021public enum Kingdom {
022
023  INCERTAE_SEDIS,
024  ANIMALIA,
025  ARCHAEA,
026  BACTERIA,
027  CHROMISTA,
028  FUNGI,
029  PLANTAE,
030  PROTOZOA,
031  VIRUSES;
032
033  // the constant name id used in checklistbank for the "incertae sedis" name string
034  private static final int INCERTAE_SEDIS_NAME_ID = 9;
035
036  /**
037   * Looks up a kingdom by its nub usage id.
038   *
039   * @param usageID the nub usage id to lookup, must be < 9 to get a result.
040   *
041   * @return the matching kingdom or null
042   */
043  public static Kingdom byNubUsageId(Integer usageID) {
044    for (Kingdom term : Kingdom.values()) {
045      if (term.nubUsageID().equals(usageID)) {
046        return term;
047      }
048    }
049    return null;
050  }
051
052  /**
053   * @return a unique single capital char representing the kingdom.
054   */
055  public String scientificName() {
056    String lower = name().replaceAll("_", " ").toLowerCase();
057    return this==INCERTAE_SEDIS ? lower : Character.toUpperCase(lower.charAt(0)) + lower.substring(1);
058  }
059
060  /**
061   * @return the nub usage id for the kingdom.
062   */
063  public Integer nubUsageID() {
064    return ordinal();
065  }
066
067  /**
068   * @return the checklistbank name id for the kingdom.
069   */
070  public Integer clbNameID() {
071    return this==INCERTAE_SEDIS ? INCERTAE_SEDIS_NAME_ID : ordinal();
072  }
073
074}
