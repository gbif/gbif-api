001package org.gbif.api.vocabulary;
002
003import java.util.List;
004
005import com.google.common.collect.ImmutableList;
006
007/**
008 * An enumeration of validation rules for single occurrence records.
009 */
010public enum OccurrenceIssue {
011
012  /**
013   * Coordinate is the exact 0/0 coordinate, often indicating a bad null coordinate.
014   */
015  ZERO_COORDINATE,
016
017  /**
018   * Coordinate has invalid lat/lon values out of their decimal max range.
019   */
020  COORDINATES_OUT_OF_RANGE,
021
022  /**
023   * The interpreted occurrence coordinates fall outside of the indicated country.
024   */
025  COUNTRY_COORDINATE_MISMATCH,
026
027  /**
028   * Interpreted country for dwc:country and dwc:countryCode contradict each other.
029   */
030  COUNTRY_MISMATCH,
031
032  /**
033   * Uninterpretable country values found.
034   */
035  COUNTRY_INVALID,
036
037  /**
038   * The interpreted country is based on the coordinates, not the verbatim string information.
039   */
040  COUNTRY_DERIVED_FROM_COORDINATES,
041
042  /**
043   * The interpreted continent and country do not match up.
044   */
045  CONTINENT_COUNTRY_MISMATCH,
046
047  /**
048   * Uninterpretable continent values found.
049   */
050  CONTINENT_INVALID,
051
052  /**
053   * The interpreted continent is based on the coordinates, not the verbatim string information.
054   */
055  CONTINENT_DERIVED_FROM_COORDINATES,
056
057  /**
058   * Latitude and longitude appear to be swapped.
059   */
060  PRESUMED_SWAPPED_COORDINATE,
061
062  /**
063   * Longitude appears to be negated, e.g. 32.3 instead of -32.3
064   */
065  PRESUMED_NEGATED_LONGITUDE,
066
067  /**
068   * Latitude appears to be negated, e.g. 32.3 instead of -32.3
069   */
070  PRESUMED_NEGATED_LATITUDE,
071
072  /**
073   * The recording date specified as the eventDate string and the individual year, month, day are contradicting.
074   */
075  RECORDED_DATE_MISMATCH,
076
077  /**
078   * A (partial) invalid date is given, such as a non existing date, invalid zero month, etc.
079   */
080  RECORDED_DATE_INVALID,
081
082  /**
083   * The recording date is highly unlikely, falling either into the future
084   * or represents a very old date before 1600 that predates modern taxonomy.
085   */
086  RECORDED_DATE_UNLIKELY,
087
088  /**
089   * Matching to the taxonomic backbone can only be done using a fuzzy, non exact match.
090   */
091  TAXON_MATCH_FUZZY,
092
093  /**
094   * Matching to the taxonomic backbone can only be done on a higher rank and not the scientific name.
095   */
096  TAXON_MATCH_HIGHERRANK,
097
098  /**
099   * Matching to the taxonomic backbone cannot be done cause there was no match at all
100   * or several matches with too little information to keep them apart (homonyms).
101   */
102  TAXON_MATCH_NONE,
103
104  /**
105   * Set if supplied depth is not given in the metric system, for example using feet instead of meters
106   */
107  DEPTH_NOT_METRIC,
108
109  /**
110   * Set if depth is larger than 11.000m or negative.
111   */
112  DEPTH_UNLIKELY,
113
114  /**
115   * Set if supplied min>max
116   */
117  DEPTH_MIN_MAX_SWAPPED,
118
119  /**
120   * Set if depth is a non numeric value
121   */
122  DEPTH_NON_NUMERIC,
123
124  /**
125   * Set if elevation is above the troposphere (17km) or below 11km (Mariana Trench).
126   */
127  ELEVATION_UNLIKELY,
128
129  /**
130   * Set if supplied min > max elevation
131   */
132  ELEVATION_MIN_MAX_SWAPPED,
133
134  /**
135   * Set if supplied elevation is not given in the metric system, for example using feet instead of meters
136   */
137  ELEVATION_NOT_METRIC,
138
139  /**
140   * Set if elevation is a non numeric value
141   */
142  ELEVATION_NON_NUMERIC,
143
144  /**
145   * A (partial) invalid date is given for dc:modified, such as a non existing date, invalid zero month, etc.
146   */
147  MODIFIED_DATE_INVALID,
148
149  /**
150   * The date given for dc:modified is in the future or predates unix time (1970).
151   */
152  MODIFIED_DATE_UNLIKELY,
153
154  /**
155   * The date given for dwc:dateIdentified is in the future or before Linnean times (1700).
156   */
157  IDENTIFIED_DATE_UNLIKELY,
158
159  /**
160   * The date given for dwc:dateIdentified is invalid and cant be interpreted at all.
161   */
162  IDENTIFIED_DATE_INVALID,
163
164  /**
165   * The given basis of record is impossible to interpret or seriously different from the recommended vocabulary.
166   */
167  BASIS_OF_RECORD_INVALID,
168
169  /**
170   * The given type status is impossible to interpret or seriously different from the recommended vocabulary.
171   */
172  TYPE_STATUS_INVALID,
173
174  /**
175   * An invalid date is given for dc:created of a multimedia object.
176   */
177  MULTIMEDIA_DATE_INVALID,
178
179  /**
180   * An invalid uri is given for a multimedia object.
181   */
182  MULTIMEDIA_URI_INVALID,
183
184  /**
185   * An invalid uri is given for dc:references.
186   */
187  REFERENCES_URI_INVALID,
188
189  /**
190   * The geodetic datum given is no interpretable value according to
191   */
192  GEODETIC_DATUM_INVALID;
193
194  /**
195   * All issues that indicate problems with the coordinates and thus should not be shown on maps.
196   */
197  public static final List<OccurrenceIssue> GEOSPATIAL_RULES = ImmutableList.of(ZERO_COORDINATE,
198                                                                                   COORDINATES_OUT_OF_RANGE,
199                                                                                   COUNTRY_COORDINATE_MISMATCH,
200                                                                                   PRESUMED_SWAPPED_COORDINATE,
201                                                                                   PRESUMED_NEGATED_LATITUDE,
202                                                                                   PRESUMED_NEGATED_LONGITUDE);
203  public static final List<OccurrenceIssue> TAXONOMIC_RULES = ImmutableList.of(TAXON_MATCH_FUZZY,
204                                                                               TAXON_MATCH_HIGHERRANK,
205                                                                               TAXON_MATCH_NONE);
206
207}
