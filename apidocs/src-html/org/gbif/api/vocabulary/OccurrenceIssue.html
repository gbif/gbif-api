001package org.gbif.api.vocabulary;
002
003import java.util.List;
004
005import com.google.common.collect.ImmutableList;
006
007/**
008 * An enumeration of validation rules for single occurrence records.
009 */
010public enum OccurrenceIssue {
011
012  /**
013   * Coordinate is the exact 0/0 coordinate, often indicating a bad null coordinate.
014   */
015  ZERO_COORDINATE,
016
017  /**
018   * Coordinate has invalid lat/lon values out of their decimal max range.
019   */
020  COORDINATE_OUT_OF_RANGE,
021
022  /**
023   * Coordinate value given in some form but GBIF is unable to interpret it.
024   */
025  COORDINATE_INVALID,
026
027  /**
028   * Original coordinate modified by rounding to 5 decimals.
029   */
030  COORDINATE_ROUNDED,
031
032  /**
033   * The geodetic datum given could not be interpreted.
034   */
035  GEODETIC_DATUM_INVALID,
036
037  /**
038   * Indicating that the interpreted coordinates assume they are based on WGS84 datum as the datum was either not
039   * indicated or interpretable. See GEODETIC_DATUM_INVALID.
040   */
041  GEODETIC_DATUM_ASSUMED_WGS84,
042
043  /**
044   * The original coordinate was successfully reprojected from a different geodetic datum to WGS84.
045   */
046  COORDINATE_REPROJECTED,
047
048  /**
049   * The given decimal latitude and longitude could not be reprojected to WGS84 based on the provided datum.
050   */
051  COORDINATE_REPROJECTION_FAILED,
052
053  /**
054   * Indicates successful coordinate reprojection according to provided datum, but which results in a datum shift
055   * larger than 0.1 decimal degrees.
056   */
057  COORDINATE_REPROJECTION_SUSPICIOUS,
058
059  /**
060   * Indicates an invalid or very unlikely coordinate accuracy derived from precision or uncertainty in meters.
061   */
062  @Deprecated //see POR-3061
063  COORDINATE_ACCURACY_INVALID,
064
065  /**
066   * Indicates an invalid or very unlikely coordinatePrecision
067   */
068  COORDINATE_PRECISION_INVALID,
069
070  /**
071   * Indicates an invalid or very unlikely dwc:uncertaintyInMeters.
072   */
073  COORDINATE_UNCERTAINTY_METERS_INVALID,
074
075  /**
076   * There is a mismatch between coordinate uncertainty in meters and coordinate precision.
077   */
078  @Deprecated //see POR-1804
079  COORDINATE_PRECISION_UNCERTAINTY_MISMATCH,
080
081  /**
082   * The interpreted occurrence coordinates fall outside of the indicated country.
083   */
084  COUNTRY_COORDINATE_MISMATCH,
085
086  /**
087   * Interpreted country for dwc:country and dwc:countryCode contradict each other.
088   */
089  COUNTRY_MISMATCH,
090
091  /**
092   * Uninterpretable country values found.
093   */
094  COUNTRY_INVALID,
095
096  /**
097   * The interpreted country is based on the coordinates, not the verbatim string information.
098   */
099  COUNTRY_DERIVED_FROM_COORDINATES,
100
101  /**
102   * The interpreted continent and country do not match up.
103   */
104  CONTINENT_COUNTRY_MISMATCH,
105
106  /**
107   * Uninterpretable continent values found.
108   */
109  CONTINENT_INVALID,
110
111  /**
112   * The interpreted continent is based on the coordinates, not the verbatim string information.
113   */
114  CONTINENT_DERIVED_FROM_COORDINATES,
115
116  /**
117   * Latitude and longitude appear to be swapped.
118   */
119  PRESUMED_SWAPPED_COORDINATE,
120
121  /**
122   * Longitude appears to be negated, e.g. 32.3 instead of -32.3
123   */
124  PRESUMED_NEGATED_LONGITUDE,
125
126  /**
127   * Latitude appears to be negated, e.g. 32.3 instead of -32.3
128   */
129  PRESUMED_NEGATED_LATITUDE,
130
131  /**
132   * The recording date specified as the eventDate string and the individual year, month, day are contradicting.
133   */
134  RECORDED_DATE_MISMATCH,
135
136  /**
137   * A (partial) invalid date is given, such as a non existing date, invalid zero month, etc.
138   */
139  RECORDED_DATE_INVALID,
140
141  /**
142   * The recording date is highly unlikely, falling either into the future
143   * or represents a very old date before 1600 that predates modern taxonomy.
144   */
145  RECORDED_DATE_UNLIKELY,
146
147  /**
148   * Matching to the taxonomic backbone can only be done using a fuzzy, non exact match.
149   */
150  TAXON_MATCH_FUZZY,
151
152  /**
153   * Matching to the taxonomic backbone can only be done on a higher rank and not the scientific name.
154   */
155  TAXON_MATCH_HIGHERRANK,
156
157  /**
158   * Matching to the taxonomic backbone cannot be done cause there was no match at all
159   * or several matches with too little information to keep them apart (homonyms).
160   */
161  TAXON_MATCH_NONE,
162
163  /**
164   * Set if supplied depth is not given in the metric system, for example using feet instead of meters
165   */
166  DEPTH_NOT_METRIC,
167
168  /**
169   * Set if depth is larger than 11.000m or negative.
170   */
171  DEPTH_UNLIKELY,
172
173  /**
174   * Set if supplied min>max
175   */
176  DEPTH_MIN_MAX_SWAPPED,
177
178  /**
179   * Set if depth is a non numeric value
180   */
181  DEPTH_NON_NUMERIC,
182
183  /**
184   * Set if elevation is above the troposphere (17km) or below 11km (Mariana Trench).
185   */
186  ELEVATION_UNLIKELY,
187
188  /**
189   * Set if supplied min > max elevation
190   */
191  ELEVATION_MIN_MAX_SWAPPED,
192
193  /**
194   * Set if supplied elevation is not given in the metric system, for example using feet instead of meters
195   */
196  ELEVATION_NOT_METRIC,
197
198  /**
199   * Set if elevation is a non numeric value
200   */
201  ELEVATION_NON_NUMERIC,
202
203  /**
204   * A (partial) invalid date is given for dc:modified, such as a non existing date, invalid zero month, etc.
205   */
206  MODIFIED_DATE_INVALID,
207
208  /**
209   * The date given for dc:modified is in the future or predates unix time (1970).
210   */
211  MODIFIED_DATE_UNLIKELY,
212
213  /**
214   * The date given for dwc:dateIdentified is in the future or before Linnean times (1700).
215   */
216  IDENTIFIED_DATE_UNLIKELY,
217
218  /**
219   * The date given for dwc:dateIdentified is invalid and cant be interpreted at all.
220   */
221  IDENTIFIED_DATE_INVALID,
222
223  /**
224   * The given basis of record is impossible to interpret or seriously different from the recommended vocabulary.
225   */
226  BASIS_OF_RECORD_INVALID,
227
228  /**
229   * The given type status is impossible to interpret or seriously different from the recommended vocabulary.
230   */
231  TYPE_STATUS_INVALID,
232
233  /**
234   * An invalid date is given for dc:created of a multimedia object.
235   */
236  MULTIMEDIA_DATE_INVALID,
237
238  /**
239   * An invalid uri is given for a multimedia object.
240   */
241  MULTIMEDIA_URI_INVALID,
242
243  /**
244   * An invalid uri is given for dc:references.
245   */
246  REFERENCES_URI_INVALID,
247
248  /**
249   * An error occurred during interpretation, leaving the record interpretion incomplete.
250   */
251  INTERPRETATION_ERROR,
252
253  /**
254   * Individual count value not parsable into an integer.
255   */
256  INDIVIDUAL_COUNT_INVALID;
257
258  /**
259   * All issues that indicate problems with the coordinates and thus should not be shown on maps.
260   */
261  public static final List<OccurrenceIssue> GEOSPATIAL_RULES = ImmutableList.of(ZERO_COORDINATE,
262                                                                                COORDINATE_OUT_OF_RANGE,
263                                                                                   COUNTRY_COORDINATE_MISMATCH);
264  public static final List<OccurrenceIssue> TAXONOMIC_RULES = ImmutableList.of(TAXON_MATCH_FUZZY,
265                                                                               TAXON_MATCH_HIGHERRANK,
266                                                                               TAXON_MATCH_NONE);
267
268}
