036/**
037 * An enumeration of validation rules for single occurrence records.
038 */
039public enum OccurrenceIssue implements InterpretationRemark {
040
041  /**
042   * Coordinate is the exact 0°, 0° coordinate, often indicating a bad null coordinate.
043   */
044  ZERO_COORDINATE(WARNING, TermsGroup.COORDINATES_TERMS_NO_DATUM),
045
046  /**
047   * Coordinate has a latitude and/or longitude value beyond the maximum (or minimum) decimal value.
048   */
049  COORDINATE_OUT_OF_RANGE(WARNING, TermsGroup.COORDINATES_TERMS_NO_DATUM),
050
051  /**
052   * Coordinate value is given in some form but GBIF is unable to interpret it.
053   */
054  COORDINATE_INVALID(WARNING, TermsGroup.COORDINATES_TERMS_NO_DATUM),
055
056  /**
057   * Original coordinate modified by rounding to 5 decimals.
058   */
059  COORDINATE_ROUNDED(INFO, TermsGroup.COORDINATES_TERMS_NO_DATUM),
060
061  /**
062   * The geodetic datum given could not be interpreted.
063   */
064  GEODETIC_DATUM_INVALID(WARNING, DwcTerm.geodeticDatum),
065
066  /**
067   * Indicating that the interpreted coordinates assume they are based on WGS84 datum as the datum
068   * was either not indicated or interpretable. See GEODETIC_DATUM_INVALID.
069   */
070  GEODETIC_DATUM_ASSUMED_WGS84(INFO, DwcTerm.geodeticDatum),
071
072  /**
073   * The original coordinate was successfully reprojected from a different geodetic datum to WGS84.
074   */
075  COORDINATE_REPROJECTED(INFO, TermsGroup.COORDINATES_TERMS),
076
077  /**
078   * The given decimal latitude and longitude could not be reprojected to WGS84 based on the
079   * provided datum.
080   */
081  COORDINATE_REPROJECTION_FAILED(WARNING, TermsGroup.COORDINATES_TERMS),
082
083  /**
084   * Indicates successful coordinate reprojection according to provided datum, but which results in
085   * a datum shift larger than 0.1 decimal degrees.
086   */
087  COORDINATE_REPROJECTION_SUSPICIOUS(WARNING, TermsGroup.COORDINATES_TERMS),
088
089  /**
090   * Indicates an invalid or very unlikely coordinate accuracy derived from precision or uncertainty
091   * in meters.
092   */
093  @Deprecated //see POR-3061
094  COORDINATE_ACCURACY_INVALID(WARNING),
095
096  /**
097   * Indicates an invalid or very unlikely coordinatePrecision
098   */
099  COORDINATE_PRECISION_INVALID(WARNING, DwcTerm.coordinatePrecision),
100
101  /**
102   * Indicates an invalid or very unlikely dwc:uncertaintyInMeters.
103   */
104  COORDINATE_UNCERTAINTY_METERS_INVALID(WARNING, DwcTerm.coordinateUncertaintyInMeters),
105
106  /**
107   * There is a mismatch between coordinate uncertainty in meters and coordinate precision.
108   */
109  @Deprecated //see POR-1804
110  COORDINATE_PRECISION_UNCERTAINTY_MISMATCH(WARNING),
111
112  /**
113   * The Footprint Spatial Reference System given could not be interpreted.
114   */
115  FOOTPRINT_SRS_INVALID(WARNING, DwcTerm.footprintSRS),
116
117  /**
118   * The Footprint Well-Known-Text given could not be interpreted.
119   */
120  FOOTPRINT_WKT_INVALID(WARNING, DwcTerm.footprintWKT),
121
122  /**
123   * The interpreted occurrence coordinates fall outside of the indicated country.
124   */
125  COUNTRY_COORDINATE_MISMATCH(WARNING, TermsGroup.COORDINATES_COUNTRY_TERMS),
126
127  /**
128   * Interpreted country for dwc:country and dwc:countryCode contradict each other.
129   */
130  COUNTRY_MISMATCH(WARNING, TermsGroup.COUNTRY_TERMS),
131
132  /**
133   * Uninterpretable country values found.
134   */
135  COUNTRY_INVALID(WARNING, TermsGroup.COUNTRY_TERMS),
136
137  /**
138   * The interpreted country is based on the coordinates, not the verbatim string information.
139   */
140  COUNTRY_DERIVED_FROM_COORDINATES(WARNING, TermsGroup.COORDINATES_COUNTRY_TERMS),
141
142  /**
143   * The interpreted continent and country do not match.
144   */
145  CONTINENT_COUNTRY_MISMATCH(WARNING),
146
147  /**
148   * Uninterpretable continent values found.
149   */
150  CONTINENT_INVALID(WARNING),
151
152  /**
153   * The interpreted continent is based on the coordinates, not the verbatim string information.
154   */
155  CONTINENT_DERIVED_FROM_COORDINATES(WARNING),
156
157  /**
158   * Latitude and longitude appear to be swapped.
159   */
160  PRESUMED_SWAPPED_COORDINATE(WARNING, TermsGroup.COORDINATES_TERMS_NO_DATUM),
161
162  /**
163   * Longitude appears to be negated, e.g. 32.3 instead of -32.3
164   */
165  PRESUMED_NEGATED_LONGITUDE(WARNING, TermsGroup.COORDINATES_TERMS_NO_DATUM),
166
167  /**
168   * Latitude appears to be negated, e.g. 32.3 instead of -32.3
169   */
170  PRESUMED_NEGATED_LATITUDE(WARNING, TermsGroup.COORDINATES_TERMS_NO_DATUM),
171
172  /**
173   * The recorded date specified as the eventDate string and the individual year, month, day are
174   * contradictory.
175   */
176  RECORDED_DATE_MISMATCH(WARNING, TermsGroup.RECORDED_DATE_TERMS),
177
178  /**
179   * A (partial) invalid date is given, such as a non existing date, zero month, etc.
180   */
181  RECORDED_DATE_INVALID(WARNING, TermsGroup.RECORDED_DATE_TERMS),
182
183  /**
184   * The recorded date is highly unlikely, falling either into the future or representing a very old
185   * date before 1600 thus predating modern taxonomy.
186   */
187  RECORDED_DATE_UNLIKELY(WARNING, TermsGroup.RECORDED_DATE_TERMS),
188
189  /**
190   * Matching to the taxonomic backbone can only be done using a fuzzy, non exact match.
191   */
192  TAXON_MATCH_FUZZY(WARNING, TermsGroup.TAXONOMY_TERMS),
193
194  /**
195   * Matching to the taxonomic backbone can only be done on a higher rank and not the scientific
196   * name.
197   */
198  TAXON_MATCH_HIGHERRANK(WARNING, TermsGroup.TAXONOMY_TERMS),
199
200  /**
201   * Matching to the taxonomic backbone can only be done on a species level,
202   * but the occurrence was in fact considered a broader species aggregate/complex.
203   * @see <a href="https://github.com/gbif/portal-feedback/issues/2935">gbif/portal-feedback#2935</a>
204   */
205  TAXON_MATCH_AGGREGATE(WARNING, TermsGroup.TAXONOMY_TERMS),
206
207  /**
208   * Matching to the taxonomic backbone cannot be done because there was no match at all, or several
209   * matches with too little information to keep them apart (potentially homonyms).
210   */
211  TAXON_MATCH_NONE(WARNING, TermsGroup.TAXONOMY_TERMS),
212
213  /**
214   * Set if supplied depth is not given in the metric system, for example using feet instead of
215   * meters
216   */
217  DEPTH_NOT_METRIC(WARNING, DwcTerm.minimumDepthInMeters, DwcTerm.maximumDepthInMeters),
218
219  /**
220   * Set if depth is larger than 11,000m or negative.
221   */
222  DEPTH_UNLIKELY(WARNING, DwcTerm.minimumDepthInMeters, DwcTerm.maximumDepthInMeters),
223
224  /**
225   * Set if supplied minimum depth > maximum depth
226   */
227  DEPTH_MIN_MAX_SWAPPED(WARNING, DwcTerm.minimumDepthInMeters, DwcTerm.maximumDepthInMeters),
228
229  /**
230   * Set if depth is a non-numeric value
231   */
232  DEPTH_NON_NUMERIC(WARNING, DwcTerm.minimumDepthInMeters, DwcTerm.maximumDepthInMeters),
233
234  /**
235   * Set if elevation is above the troposphere (17km) or below 11km (Mariana Trench).
236   */
237  ELEVATION_UNLIKELY(WARNING, DwcTerm.minimumElevationInMeters, DwcTerm.maximumElevationInMeters),
238
239  /**
240   * Set if supplied minimum elevation > maximum elevation
241   */
242  ELEVATION_MIN_MAX_SWAPPED(WARNING, DwcTerm.minimumElevationInMeters,
243    DwcTerm.maximumElevationInMeters),
244
245  /**
246   * Set if supplied elevation is not given in the metric system, for example using feet instead of
247   * meters
248   */
249  ELEVATION_NOT_METRIC(WARNING, DwcTerm.minimumElevationInMeters, DwcTerm.maximumElevationInMeters),
250
251  /**
252   * Set if elevation is a non-numeric value
253   */
254  ELEVATION_NON_NUMERIC(WARNING, DwcTerm.minimumElevationInMeters,
255    DwcTerm.maximumElevationInMeters),
256
257  /**
258   * A (partial) invalid date is given for dc:modified, such as a nonexistent date, zero month, etc.
259   */
260  MODIFIED_DATE_INVALID(WARNING, DcTerm.modified),
261
262  /**
263   * The date given for dc:modified is in the future or predates Unix time (1970).
264   */
265  MODIFIED_DATE_UNLIKELY(WARNING, DcTerm.modified),
266
267  /**
268   * The date given for dwc:dateIdentified is in the future or before Linnean times (1700).
269   */
270  IDENTIFIED_DATE_UNLIKELY(WARNING, DwcTerm.dateIdentified),
271
272  /**
273   * The date given for dwc:dateIdentified is invalid and can't be interpreted at all.
274   */
275  IDENTIFIED_DATE_INVALID(WARNING, DwcTerm.dateIdentified),
276
277  /**
278   * The given basis of record is impossible to interpret or significantly different from the
279   * recommended vocabulary.
280   */
281  BASIS_OF_RECORD_INVALID(WARNING, DwcTerm.basisOfRecord),
282
283  /**
284   * The given type status is impossible to interpret or significantly different from the
285   * recommended vocabulary.
286   */
287  TYPE_STATUS_INVALID(WARNING, DwcTerm.typeStatus),
288
289  /**
290   * An invalid date is given for dc:created of a multimedia object.
291   */
292  MULTIMEDIA_DATE_INVALID(WARNING),
293
294  /**
295   * An invalid URI is given for a multimedia object.
296   */
297  MULTIMEDIA_URI_INVALID(WARNING),
298
299  /**
300   * An invalid URI is given for dc:references.
301   */
302  REFERENCES_URI_INVALID(WARNING, DcTerm.references),
303
304  /**
305   * An error occurred during interpretation, leaving the record interpretation incomplete.
306   */
307  INTERPRETATION_ERROR(ERROR),
308
309  /**
310   * The individual count value is not a positive integer
311   */
312  INDIVIDUAL_COUNT_INVALID(WARNING, DwcTerm.individualCount),
313
314  /**
315   * Example: individual count value > 0, but occurrence status is absent.
316   */
317  INDIVIDUAL_COUNT_CONFLICTS_WITH_OCCURRENCE_STATUS(WARNING, DwcTerm.individualCount),
318
319  /**
320   * Occurrence status value can't be assigned to {@link OccurrenceStatus}
321   */
322  OCCURRENCE_STATUS_UNPARSABLE(WARNING, DwcTerm.occurrenceStatus),
323
324  /**
325   * Occurrence status was inferred from the individual count value
326   */
327  OCCURRENCE_STATUS_INFERRED_FROM_INDIVIDUAL_COUNT(WARNING, DwcTerm.occurrenceStatus),
328
329  /**
330   * Occurrence status was inferred from basis of records
331   */
332  OCCURRENCE_STATUS_INFERRED_FROM_BASIS_OF_RECORD(WARNING, DwcTerm.occurrenceStatus),
333
334  /**
335   * The date given for dwc:georeferencedDate is in the future or before Linnean times (1700).
336   */
337  GEOREFERENCED_DATE_UNLIKELY(WARNING, DwcTerm.georeferencedDate),
338
339  /**
340   * The date given for dwc:georeferencedDate is invalid and can't be interpreted at all.
341   */
342  GEOREFERENCED_DATE_INVALID(WARNING, DwcTerm.georeferencedDate),

344  /**
345   * The given institution matches with more than 1 GRSciColl institution.
346   */
347  AMBIGUOUS_INSTITUTION(WARNING, TermsGroup.INSTITUTION_TERMS),

349  /**
350   * The given collection matches with more than 1 GRSciColl collection.
351   */
352  AMBIGUOUS_COLLECTION(WARNING, TermsGroup.COLLECTION_TERMS),

355   * The given institution couldn't be matched with any GRSciColl institution.
356   */
357  INSTITUTION_MATCH_NONE(WARNING, TermsGroup.INSTITUTION_TERMS),
358
359  /**
360   * The given collection couldn't be matched with any GRSciColl collection.
361   */
362  COLLECTION_MATCH_NONE(WARNING, TermsGroup.COLLECTION_TERMS),

364  /**

390   * The given owner institution is different than the given institution. Therefore we assume it

459
460  private final Set<Term> relatedTerms;
461  private final InterpretationRemarkSeverity severity;
462  private final boolean isDeprecated;
463
464  /**
465   * {@link OccurrenceIssue} not linked to any specific {@link Term}.
466   */
467  OccurrenceIssue(InterpretationRemarkSeverity severity) {
468    this.severity = severity;
469    this.relatedTerms = Collections.emptySet();
470    this.isDeprecated = AnnotationUtils.isFieldDeprecated(OccurrenceIssue.class, this.name());
471  }
472
473  /**
474   * {@link OccurrenceIssue} linked to the provided {@link Term}.
475   */
476  OccurrenceIssue(InterpretationRemarkSeverity severity, Term... relatedTerms) {
477    this.severity = severity;
478    this.relatedTerms = Collections.unmodifiableSet(new HashSet<>(Arrays.asList(relatedTerms)));
479    this.isDeprecated = AnnotationUtils.isFieldDeprecated(OccurrenceIssue.class, this.name());
480  }
481
482  @Override
483  public String getId() {
484    return name();
485  }
486
487  @Override
488  public Set<Term> getRelatedTerms() {
489    return relatedTerms;
490  }
491
492  @Override
493  public InterpretationRemarkSeverity getSeverity() {
494    return severity;
495  }
496
497  @Override
498  public boolean isDeprecated() {
499    return isDeprecated;
500  }
