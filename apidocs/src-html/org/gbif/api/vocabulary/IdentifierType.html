025import org.apache.commons.lang3.StringUtils;

027/**
028 * Enumeration for all possible identifier types.
029 */
030public enum IdentifierType {
031
032  URL,
033
034  /**
035   * Reference controlled by a separate system, used for example by DOI.
036   * {http://en.wikipedia.org/wiki/Handle_(computing)}
037   */
038  LSID,
039
040  HANDLER,
041
042  DOI,
043
044  UUID,
045
046  FTP,
047
048  URI,
049
050  UNKNOWN,
051
052  /**
053   * Indicates the identifier originated from an auto_increment column in the portal.data_provider or
054   * portal.data_resource table respectively.
055   */
056  GBIF_PORTAL,
057
058  /**
059   * Identifies the node (e.g: 'DK' for Denmark, 'sp2000' for Species 2000).
060   */
061  GBIF_NODE,
062
063  /**
064   * Participant identifier from the GBIF Directory.
065   */
066  GBIF_PARTICIPANT,
067
068  /**
069   * ID migrated from GrSciColl.
070   */
071  GRSCICOLL_ID,
072
073  /**
074   * Cool URI migrated from GrSciColl.
075   */
076  GRSCICOLL_URI,
077
078  /**
079   * IRN of an IH record.
080   */
081  IH_IRN,
082
083  /**
084   * Research Organization Registry.
085   * https://ror.org
086   */
087  ROR,
088
089  /**
090   * https://www.grid.ac/institutes/
091   */
092  GRID,
093
094  /**
095   * https://cites.org/
096   */
097  CITES,
098  /**
099   * Symbiota IDs to help linking GrSciColl occurrences.
100   */
101  SYMBIOTA_UUID,
102  WIKIDATA,
103
104  /**
105   * https://www.ncbi.nlm.nih.gov/
106   */
107  NCBI_BIOCOLLECTION,
108
109  /**
110   * https://slks.dk/english/work-areas/libraries-and-literature/library-standards/isil
111   */
112  ISIL,
113
114  /**
115   * ChecklistBank dataset keys (integer)
116   * https://www.checklistbank.org/dataset
117   */
118  CLB_DATASET_KEY;
119
120  // TODO: Check if this is used, it didn't exist in the new Registry2 API, but I preserved it from the old vocabulary
121  public static final List<IdentifierType> TYPES;
122
123  static {
124    TYPES = Collections.unmodifiableList(Arrays.asList(IdentifierType.values()));
125  }
126
127  /**
128   * @return the matching IdentifierType or null
129   */
130  public static IdentifierType fromString(String identifierType) {
131    return VocabularyUtils.lookupEnum(identifierType, IdentifierType.class);
132  }
133
134  /**
135   * Tries to infer the identifier type from a given identifier.
136   * Most identifiers have a URI protocol prefix or a specific structure that
137   * allows the guess.
138   *
139   * @return the inferred identifier type or Unknown if identifier is null or cant be inferred.
140   */
141  public static IdentifierType inferFrom(@Nullable String identifier) {
142    String lcIdentifier = StringUtils.trimToEmpty(identifier).toLowerCase();
143
144    if (lcIdentifier.isEmpty()) {
145      return UNKNOWN;
146    }
147
148    if (lcIdentifier.startsWith(org.gbif.api.model.common.DOI.GBIF_PREFIX)
149        || lcIdentifier.startsWith(org.gbif.api.model.common.DOI.TEST_PREFIX)) {
150      return DOI;
151    }
152    if (lcIdentifier.startsWith("10.")
153        || lcIdentifier.startsWith("doi:10.")
154        || lcIdentifier.startsWith("urn:doi:10.")
155        || lcIdentifier.startsWith("http://dx.doi.org/10.")
156        || lcIdentifier.startsWith("https://dx.doi.org/10.")
157        || lcIdentifier.startsWith("http://doi.org/10.")
158        || lcIdentifier.startsWith("https://doi.org/10.")) {
159      return DOI;
160    }
161
162    if (lcIdentifier.startsWith("https://ror.org")) {
163      return ROR;
164    }
165
166    if(IdentifierUtils.isValidWikidataIdentifier(lcIdentifier)) {
167      return WIKIDATA;
168    }
169
170    if (lcIdentifier.startsWith("http:")
171        || lcIdentifier.startsWith("https:")
172        || lcIdentifier.startsWith("www.")) {
173      return URL;
174    }
175    if (lcIdentifier.startsWith("ftp:")) {
176      return FTP;
177    }
178    if (lcIdentifier.startsWith("urn:lsid:") || lcIdentifier.startsWith("lsid:")) {
179      return LSID;
180    }
181
182    if (lcIdentifier.startsWith("urn:uuid:") || lcIdentifier.startsWith("uuid:")) {
183      return UUID;
184    }
185
186    if (lcIdentifier.startsWith("gbif:ih:irn:")) {
187      return IH_IRN;
188    }
189
190    if (lcIdentifier.startsWith("grid")) {
191      return GRID;
192    }
193
194    if (IdentifierUtils.isValidISILIdentifier(lcIdentifier)) {
195      return ISIL;
196    }
197
198   if (IdentifierUtils.isValidCLBDatasetKey(lcIdentifier)) {
199     return CLB_DATASET_KEY;
200   }
201
202    try {
203      //noinspection ResultOfMethodCallIgnored
204      java.util.UUID.fromString(lcIdentifier);
205      return UUID;
206    } catch (IllegalArgumentException ignored) {
207      // We're just trying to convert a String to anything readable. Apparently the UUID approach failed.
208    }
209
210    return UNKNOWN;
211  }
