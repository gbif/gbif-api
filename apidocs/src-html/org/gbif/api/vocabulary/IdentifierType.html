016package org.gbif.api.vocabulary;
017
018import org.gbif.api.util.VocabularyUtils;
019
020import java.util.List;
021import javax.annotation.Nullable;
022
023import com.google.common.base.Strings;
024import com.google.common.collect.ImmutableList;
025
026/**
027 * Enumeration for all possible identifier types.
028 */
029public enum IdentifierType {
030
031  URL,
032  /**
033   * Reference controlled by a separate system, used for example by DOI.
034   * {http://en.wikipedia.org/wiki/Handle_(computing)}
035   */
036  LSID,
037  HANDLER,
038  DOI,
039  UUID,
040  FTP,
041  URI,
042  UNKNOWN,
043  /**
044   * Indicates the identifier originated from an auto_increment column in the portal.data_provider or
045   * portal.data_resource table respectively.
046   */
047  GBIF_PORTAL,
048  /**
049   * Identifies the node (e.g: 'DK' for Denmark, 'sp2000' for Species 2000).
050   */
051  GBIF_NODE,
052
053  /**
054   * Participant identifier from the GBIF Directory.
055   */
056  GBIF_PARTICIPANT,
057
058  /**
059   * ID migrated from GRBIO.
060   */
061  GRBIO_ID,
062
063  /**
064   * Cool URI migrated from GBRIO.
065   */
066  GRBIO_URI;
067
068  // TODO: Check if this is used, it didn't exist in the new Registry2 API, but I preserved it from the old vocabulary
069  public static final List<IdentifierType> TYPES;
070
071  static {
072    TYPES = ImmutableList.copyOf(IdentifierType.values());
073  }
074
075  /**
076   * @return the matching IdentifierType or null
077   */
078  public static IdentifierType fromString(String identifierType) {
079    return (IdentifierType) VocabularyUtils.lookupEnum(identifierType, IdentifierType.class);
080  }
081
082
083  /**
084   * Tries to infer the identifier type from a given identifier.
085   * Most identifiers have a URI protocol prefix or a specific structure that
086   * allows the guess.
087   *
088   * @return the inferred identifier type or Unknown if identifier is null or cant be inferred.
089   */
090  public static IdentifierType inferFrom(@Nullable String identifier) {
091    String lcIdentifier = Strings.nullToEmpty(identifier).trim().toLowerCase();
092
093    if (lcIdentifier.isEmpty()) {
094      return UNKNOWN;
095    }
096
097    if (lcIdentifier.startsWith(org.gbif.api.model.common.DOI.GBIF_PREFIX)
098        || lcIdentifier.startsWith(org.gbif.api.model.common.DOI.TEST_PREFIX)) {
099      return DOI;
100    }
101    if (lcIdentifier.startsWith("10.")
102        || lcIdentifier.startsWith("doi:10.")
103        || lcIdentifier.startsWith("urn:doi:10.")
104        || lcIdentifier.startsWith("http://dx.doi.org/10.")
105        || lcIdentifier.startsWith("https://dx.doi.org/10.")
106        || lcIdentifier.startsWith("http://doi.org/10.")
107        || lcIdentifier.startsWith("https://doi.org/10.")) {
108      return DOI;
109    }
110    if (lcIdentifier.startsWith("http:")
111        || lcIdentifier.startsWith("https:")
112        || lcIdentifier.startsWith("www.")) {
113      return URL;
114    }
115    if (lcIdentifier.startsWith("ftp:")) {
116      return FTP;
117    }
118    if (lcIdentifier.startsWith("urn:lsid:") || lcIdentifier.startsWith("lsid:")) {
119      return LSID;
120    }
121
122    if (lcIdentifier.startsWith("urn:uuid:") || lcIdentifier.startsWith("uuid:")) {
123      return UUID;
124    }
125    try {
126      java.util.UUID.fromString(lcIdentifier);
127      return UUID;
128    } catch (IllegalArgumentException ignored) {
129      // We're just trying to convert a String to anything readable. Apparently the UUID approach failed.
130    }
131
132    return UNKNOWN;
133  }
134}
