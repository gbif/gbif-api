001package org.gbif.api.vocabulary;
002
003import java.util.List;
004
005import com.google.common.collect.ImmutableList;
006
007/**
008 * An enumeration of validation rules for single occurrence records.
009 * Every rule has to be answered in a simple boolean result that is kept in the validations map of an interpreted
010 * Occurrence. All rules evaluating to true should be considered to have successfully passed the validation,
011 * e.g. if COUNTRY_COORDINATE_MISMATCH is true, there is no mismatch!
012 */
013public enum OccurrenceValidationRule {
014
015  /**
016   * Coordinate is the exact 0/0 coordinate, often indicating a bad null coordinate.
017   */
018  ZERO_COORDINATE,
019
020  /**
021   * Coordinate has invalid lat/lon values out of their decimal max range.
022   */
023  COORDINATES_OUT_OF_RANGE,
024
025  /**
026   * The interpreted occurrence coordinates fall outside of the indicated country.
027   */
028  COUNTRY_COORDINATE_MISMATCH,
029
030  /**
031   * Latitude and longitude appear to be swapped.
032   */
033  PRESUMED_SWAPPED_COORDINATE,
034
035  /**
036   * Longitude appears to be negated, e.g. 32.3 instead of -32.3
037   */
038  PRESUMED_NEGATED_LONGITUDE,
039
040  /**
041   * Latitude appears to be negated, e.g. 32.3 instead of -32.3
042   */
043  PRESUMED_NEGATED_LATITUDE;
044
045  public static List<OccurrenceValidationRule> GEOSPATIAL_RULES = ImmutableList.of(ZERO_COORDINATE,
046                                                                                   COORDINATES_OUT_OF_RANGE,
047                                                                                   COUNTRY_COORDINATE_MISMATCH,
048                                                                                   PRESUMED_SWAPPED_COORDINATE,
049                                                                                   PRESUMED_NEGATED_LATITUDE,
050                                                                                   PRESUMED_NEGATED_LONGITUDE);
051  public static List<OccurrenceValidationRule> TAXONOMIC_RULES = ImmutableList.of();
052
053}
