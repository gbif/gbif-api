256  /**
257   * Get the prefix if supported by this serializer
258   *
259   * @param t
260   * @return
261   */
262  private static Optional<String> getPrefix(Term t){
263    if( t instanceof DwcTerm){
264      return Optional.of(DwcTerm.PREFIX);
265    }
266    else if( t instanceof DcTerm){
267      return Optional.of(DcTerm.PREFIX);
268    }
269    else if( t instanceof GbifTerm){
270      return Optional.of(GbifTerm.PREFIX);
271    }
272    return Optional.empty();
273  }
