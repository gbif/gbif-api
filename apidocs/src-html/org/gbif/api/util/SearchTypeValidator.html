027/**
028 * Utility class to do basic validation of all search enum based values.
029 */
030public class SearchTypeValidator {
031
032  private static final Pattern BOOLEAN = Pattern.compile("^(true|false)$", Pattern.CASE_INSENSITIVE);
033  // this regex matches a double with an optional dot separated fracture and negative signing
034  private static final String DEC = "-?\\d+(?:\\.\\d+)?";
035  private static final String DECIMAL = "(" + DEC + ")";
036
037  private static final String DECIMAL_OR_WILDCARD = "(" + DEC + "|\\*)";
038
039  private static final Range<Double> LATITUDE_RNG = Range.closed(-90.0, 90.0);
040  private static final Range<Double> LONGITUDE_RNG = Range.closed(-180.0, 180.0);
041
042  private static final String LATITUDE_ERROR_MSG = "%s is not valid value, latitude must be between -90 and 90.";
043
044  private static final String LONGITUDE_ERROR_MSG = "%s is not valid value, longitude must be between -180 and 180.";
045
046  private static final String WILD_CARD = "*";
047
048  /**
049   * Matches ranges in formats
050   *
051   * <pre>
052   * 23.1,55.2
053   * </pre>
054   *
055   * <pre>
056   * *,88
057   * </pre>
058   *
059   * and
060   *
061   * <pre>
062   * 55,*
063   * </pre>
064   *
065   * .
066   * The matcher returns 2 groups:
067   * group 1: lower bound
068   * group 2: upper bound
069   */
070  private static final Pattern DECIMAL_RANGE_PATTERN = Pattern.compile(
071    "^" + DECIMAL_OR_WILDCARD + "\\s*,\\s*" + DECIMAL_OR_WILDCARD + "$", Pattern.CASE_INSENSITIVE);
072  /**
073   * Matches without groups a space delimited coordinate pair allowing prefixing or trailing whitespace: 10.12 -90.12
074   */
075  private static final String WKT_COORD = "\\s*" + DECIMAL + "\\s+" + DECIMAL + "\\s*";
076
077  /**
078   * Matches without groups a comma separated list of coordinates enclosed in brackets:
079   * (10.12 -90.12, 30 10, 10 20, 20 40)
080   */
081  private static final String WKT_LINE = "\\s*\\(" + WKT_COORD + "(?:," + WKT_COORD + ")*\\)\\s*";
082
083  private static final List<Pattern> WKT_PATTERNS = ImmutableList.of(
084    Pattern.compile("^POINT\\s*\\(" + WKT_COORD + "\\)$", Pattern.CASE_INSENSITIVE),
085    Pattern.compile("^(?:LINESTRING|LINEARRING)\\s*" + WKT_LINE + "$", Pattern.CASE_INSENSITIVE),
086    Pattern.compile("^POLYGON\\s*\\(" + WKT_LINE + "(?:," + WKT_LINE + ")*\\)$", Pattern.CASE_INSENSITIVE));
087
088  /**
089   * Private default constructor.
090   */
091  private SearchTypeValidator() {
092    // empty block
093  }

095  /**
096   * Builds a closed range from two given values, but accepts nulls which are converted into unbound ranges
097   * in the guava instance.
098   */
099  public static <T extends Comparable<?>> Range<T> buildRange(final T lower, final T upper) {
100    if (lower == null && upper != null) {
101      return Range.atMost(upper);
102
103    } else if (lower != null && upper == null) {
104      return Range.atLeast(lower);
105
106    } else if (lower == null && upper == null) {
107      return Range.<T>all();
108
109    } else {
110      return Range.closed(lower, upper);
111    }
112  }

114  /**
115   * Determines whether the value given is a valid decimal or date range, delimiting two values by a comma.
116   *
117   * @return true if the given value is a valid range
118   */
119  public static boolean isRange(String value) {
120    if (!Strings.isNullOrEmpty(value)) {
121      // decimal range for ints or doubles
122      if (DECIMAL_RANGE_PATTERN.matcher(value).find()) {
123        return true;
124      }
125      // check date range
126      try {
127        IsoDateParsingUtils.parseDateRange(value);
128        return true;
129      } catch (Exception e) {
130        // aha, no range
131      }
132    }
133    return false;
134  }
135
136  /**
137   * Parses a range of ISO dates.
138   * The date format used is the first date format that successfully parses the lower range limit.
139   *
140   * @return the parsed range with wildcards represented as null values
141   * @throws IllegalArgumentException if value is invalid or null
142   */
143  public static Range<Date> parseDateRange(String value) {
144    return IsoDateParsingUtils.parseDateRange(value);
145  }
146
147  /**
148   * Parses a decimal range in the format 123.1,456.
149   *
150   * @return the parsed range with wildcards represented as null values
151   * @throws IllegalArgumentException if value is invalid or null
152   */
153  public static Range<Double> parseDecimalRange(String value) {
154    if (!Strings.isNullOrEmpty(value)) {
155      Matcher m = DECIMAL_RANGE_PATTERN.matcher(value);
156      if (m.find()) {
157        return buildRange(parseDouble(m.group(1)), parseDouble(m.group(2)));
158      }
159    }
160    throw new IllegalArgumentException("Invalid decimal range: " + value);
161  }
162
163  /**
164   * Parses an integer range in the format 123,456
165   *
166   * @return the parsed range with wildcards represented as null values
167   * @throws IllegalArgumentException if value is invalid or null
168   */
169  public static Range<Integer> parseIntegerRange(String value) {
170    if (!Strings.isNullOrEmpty(value)) {
171      Matcher m = DECIMAL_RANGE_PATTERN.matcher(value);
172      if (m.find()) {
173
174        return buildRange(parseInteger(m.group(1)), parseInteger(m.group(2)));
175      }
176    }
177    throw new IllegalArgumentException("Invalid integer range: " + value);
178  }

180  /**
181   * Validates that a given parameter value matches the expected type of the parameter as defined by
182   * {@link SearchParameter#type()} and throws an IllegalArgumentException otherwise.
183   *
184   * @param param the search parameter defining the expected type
185   * @param value the parameter value to be validated
186   * @throws IllegalArgumentException if the value cannot be converted to the expected type
187   */
188  public static void validate(SearchParameter param, String value) throws IllegalArgumentException {
189    final Class<?> pType = param.type();
190
191    try {
192      if (OccurrenceSearchParameter.GEOMETRY == param) {
193        validateGeometry(value);
194
195      }
196      // All the parameters except by GEOMETRY accept the wild card value
197      if (!WILD_CARD.equalsIgnoreCase(Strings.nullToEmpty(value).trim())) {
198        if (OccurrenceSearchParameter.DECIMAL_LATITUDE == param) {
199          validateLatitude(value);
200
201        } else if (OccurrenceSearchParameter.DECIMAL_LONGITUDE == param) {
202          validateLongitude(value);
203
204        } else if (UUID.class.isAssignableFrom(pType)) {
205          UUID.fromString(value);
206
207        } else if (Double.class.isAssignableFrom(pType)) {
208          validateDouble(value);
209
210        } else if (Integer.class.isAssignableFrom(pType)) {
211          Collection<Integer> intsFound = validateInteger(value);
212          if (OccurrenceSearchParameter.MONTH == param) {
213            validateMonth(intsFound);
214          }
215
216        } else if (Boolean.class.isAssignableFrom(pType)) {
217          // we cannot use Boolean.parseBoolean as this accepted anything as false
218          if (!BOOLEAN.matcher(value).find()) {
219            throw new IllegalArgumentException("Value " + value + " is no valid boolean");
220          }
221
222        } else if (Country.class.isAssignableFrom(pType)) {
223          // iso codes expected
224          Preconditions.checkNotNull(Country.fromIsoCode(value));
225
226        } else if (Language.class.isAssignableFrom(pType)) {
227          // iso codes expected
228          Preconditions.checkNotNull(Language.fromIsoCode(value));
229
230        } else if (Enum.class.isAssignableFrom(pType)) {
231          // enum value expected, cast to enum
232          @SuppressWarnings("unchecked")
233          Class<? extends Enum<?>> eType = (Class<? extends Enum<?>>) pType;
234          Preconditions.checkNotNull(VocabularyUtils.lookupEnum(value, eType));
235
236        } else if (Date.class.isAssignableFrom(pType)) {
237          // ISO date strings
238          validateDate(value);
239
240        } else if (String.class.isAssignableFrom(pType)) {
241          // any string allowed
242
243        } else {
244          // an unexpected data type - update this method!!
245          throw new IllegalArgumentException("Unknown SearchParameter data type " + pType.getCanonicalName());
246        }
247      }
248    } catch (NullPointerException e) {
249      // Preconditions.checkNotNull throws NPE but we want IllegalArgumentException
250      throw new IllegalArgumentException("Value " + value + " invalid for filter parameter " + param, e);
251    }
252  }

255  /**
256   * @return the parsed double or null for wildcards
257   * @throws NumberFormatException if invalid double
258   */
259  private static Double parseDouble(String d) {
260    if (QUERY_WILDCARD.equals(d)) {
261      return null;
262    }
263    return Double.parseDouble(d);
264  }
265
266  /**
267   * @return the parsed integer or null for wildcards
268   * @throws NumberFormatException if invalid integer
269   */
270  private static Integer parseInteger(String d) {
271    if (QUERY_WILDCARD.equals(d)) {
272      return null;
273    }
274    return Integer.parseInt(d);
275  }
276
277  /**
278   * Validates if the string value is a valid ISO 8601 format.
279   */
280  private static void validateDate(String value) {
281    if (isRange(value)) {
282      IsoDateParsingUtils.parseDateRange(value);
283    } else {
284      IsoDateParsingUtils.parseDate(value);
285    }
286  }
287
288  /**
289   * Validates if the value is a valid single double or a range of double values.
290   * If the value is a range each limit is validated and the wildcard character '*' is skipped.
291   */
292  private static void validateDouble(String value) {
293    if (Strings.isNullOrEmpty(value)) {
294      throw new IllegalArgumentException("Double cannot be null or empty");
295    }
296    try {
297      Double.parseDouble(value);
298    } catch (NumberFormatException e) {
299      parseDecimalRange(value);
300    }
301  }
302
303
304  /**
305   * Validates if the value is a valid single double and its value is between a range.
306   */
307  private static void validateDoubleInRange(String value, Range<Double> range, String errorMsg) {
308    if (Strings.isNullOrEmpty(value)) {
309      throw new IllegalArgumentException("Double cannot be null or empty");
310    }
311    try {
312      final Double doubleValue = Double.parseDouble(value);
313      if (!range.contains(doubleValue)) {
314        throw new IllegalArgumentException(String.format(errorMsg, value));
315      }
316    } catch (NumberFormatException e) {
317      if (isRange(value)) {
318        Range<Double> rangeValue = parseDecimalRange(value);
319        if (!range.encloses(rangeValue)) {
320          throw new IllegalArgumentException(String.format(errorMsg, value));
321        }
322      } else {
323        throw new IllegalArgumentException("Argument is not a valid number");
324      }
325    }
326  }

328  /**
329   * Verify that we have indeed a wellKnownText parameter.
330   * See <a href="http://en.wikipedia.org/wiki/Well-known_text">wikipedia</a> for basic WKT specs.
331   * The validation implemented does both syntactic and topological validation (for polygons only): only convex polygons
332   * are accepted.
333   */
334  private static void validateGeometry(String wellKnownText) {
335    validateGeometrySyntax(wellKnownText);
336    try {
337      Geometry geometry = new WKTReader().read(wellKnownText);
338      // Calculating the area > 0 ensures that polygons that are representing lines or points are invalidated
339      if (geometry instanceof Polygon && (!geometry.isValid() || geometry.getArea() == 0.0)) {
340        throw new IllegalArgumentException("Invalid polygon " + wellKnownText);
341      }
342    } catch (ParseException e) {
343      throw new IllegalArgumentException("Invalid simple WKT: " + wellKnownText);
344    }
345  }
346
347  /**
348   * Verify that we have indeed a wellKnownText parameter.
349   * See <a href="http://en.wikipedia.org/wiki/Well-known_text">wikipedia</a> for basic WKT specs.
350   * The validation implemented does a basic syntax check for the following geometries, but does not
351   * verify that the resulting geometries are topologically valid (see the OGC SFS specification).
352   */
353  private static void validateGeometrySyntax(String wellKnownText) {
354    if (Strings.isNullOrEmpty(wellKnownText)) {
355      throw new IllegalArgumentException("Well Known Text cannot be empty or null");
356    }
357    // test all 4 supported geometry types with their respective regex - one must match!
358    for (Pattern regex : WKT_PATTERNS) {
359      if (regex.matcher(wellKnownText).find()) {
360        return;
361      }
362    }
363    throw new IllegalArgumentException("Invalid simple WKT: " + wellKnownText);
364  }
365
366  /**
367   * Validates if the value is a valid single integer or a range of integer values.
368   * If the value is a range each limit is validated and the wildcard character '*' is skipped.
369   *
370   * @throws IllegalArgumentException if value is invalid or null
371   */
372  private static Collection<Integer> validateInteger(String value) {
373    if (Strings.isNullOrEmpty(value)) {
374      throw new IllegalArgumentException("Integer cannot be null or empty");
375    }
376    try {
377      return Lists.newArrayList(Integer.parseInt(value));
378    } catch (NumberFormatException e) {
379      Range<Integer> range = parseIntegerRange(value);
380      List<Integer> ints = Lists.newArrayList();
381      if (range.hasLowerBound()) {
382        ints.add(range.lowerEndpoint());
383      }
384      if (range.hasUpperBound()) {
385        ints.add(range.upperEndpoint());
386      }
387      return ints;
388    }
389  }
390
391  /**
392   * Validates if the parameter is a valid latitude value.
393   */
394  private static void validateLatitude(String value) {
395    validateDoubleInRange(value, LATITUDE_RNG, LATITUDE_ERROR_MSG);
396  }
397
398  /**
399   * Validates if the parameter is a valid longitude value.
400   */
401  private static void validateLongitude(String value) {
402    validateDoubleInRange(value, LONGITUDE_RNG, LONGITUDE_ERROR_MSG);
403  }
404
405  private static void validateMonth(Collection<Integer> months) {
406    for (Integer month : months) {
407      if (month != null && (month < 1 || month > 12)) {
408        throw new IllegalArgumentException("Month needs to be between 1 - 12");
409      }
410    }
411  }
