013package org.gbif.api.util.occurrence;
014
015import org.gbif.api.model.occurrence.predicate.ConjunctionPredicate;
016import org.gbif.api.model.occurrence.predicate.DisjunctionPredicate;
017import org.gbif.api.model.occurrence.predicate.EqualsPredicate;
018import org.gbif.api.model.occurrence.predicate.GreaterThanOrEqualsPredicate;
019import org.gbif.api.model.occurrence.predicate.GreaterThanPredicate;
020import org.gbif.api.model.occurrence.predicate.InPredicate;
021import org.gbif.api.model.occurrence.predicate.IsNotNullPredicate;
022import org.gbif.api.model.occurrence.predicate.LessThanOrEqualsPredicate;
023import org.gbif.api.model.occurrence.predicate.LessThanPredicate;
024import org.gbif.api.model.occurrence.predicate.LikePredicate;
025import org.gbif.api.model.occurrence.predicate.NotPredicate;
026import org.gbif.api.model.occurrence.predicate.Predicate;
027import org.gbif.api.model.occurrence.predicate.WithinPredicate;
028import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
029
030import java.lang.reflect.InvocationTargetException;
031import java.lang.reflect.Method;
032import java.net.URLEncoder;
033import java.util.LinkedList;
034import java.util.Map;
035import java.util.regex.Matcher;
036import java.util.regex.Pattern;
037
038import com.google.common.base.Strings;
039import com.google.common.collect.Lists;
040import com.google.common.collect.Maps;
041import org.slf4j.Logger;
042import org.slf4j.LoggerFactory;
043
044/**
045 * This class builds a query parameter filter usable for search links from a
046 * {@link org.gbif.api.model.occurrence.predicate.Predicate} hierarchy.
047 * This class is not thread safe, create a new instance for every use if concurrent calls to {#queryFilter()} is
048 * expected.
049 */
050public class QueryParameterFilterBuilder {
051
052  private static final Logger LOG = LoggerFactory.getLogger(QueryParameterFilterBuilder.class);
053  private static final String WILDCARD = "*";
054  private final static Pattern POLYGON_PATTERN = Pattern.compile("POLYGON\\s*\\(\\s*\\((.+)\\)\\s*\\)",
055    Pattern.CASE_INSENSITIVE);
056
057  private Map<OccurrenceSearchParameter, LinkedList<String>> filter;
058
059  private enum State {
060    ROOT, AND, OR
061  };
062
063  private State state;
064  private OccurrenceSearchParameter lastParam;
065
066  public QueryParameterFilterBuilder() {
067
068  }
069
070  public synchronized String queryFilter(Predicate p) {
071    StringBuilder b = new StringBuilder();
072    boolean first = true;
073
074    // build filter map
075    filter = Maps.newHashMap();
076    state = State.ROOT;
077    lastParam = null;
078    visit(p);
079
080    // transform filter map into query string
081    for (OccurrenceSearchParameter param : filter.keySet()) {
082      for (String val : filter.get(param)) {
083        if (first) {
084          first = false;
085        } else {
086          b.append("&");
087        }
088        b.append(param.name());
089        b.append("=");
090        b.append(URLEncoder.encode(val));
091      }
092    }
093    return b.toString();
094  }
095
096  private void visit(ConjunctionPredicate and) throws IllegalStateException {
097    // ranges are allowed underneath root - try first
098    try {
099      visitRange(and);
100      return;
101    } catch (IllegalArgumentException e) {
102      // must be a root AND
103    }
104
105    if (state != State.ROOT) {
106      throw new IllegalStateException("AND must be a root predicate or a valid range");
107    }
108    state = State.AND;
109
110    for (Predicate p : and.getPredicates()) {
111      lastParam = null;
112      visit(p);
113    }
114    state = State.ROOT;
115  }
116
117  private void visitRange(ConjunctionPredicate and) {
118    if (and.getPredicates().size() != 2) {
119      throw new IllegalArgumentException("no valid range");
120    }
121    GreaterThanOrEqualsPredicate lower = null;
122    LessThanOrEqualsPredicate upper = null;
123    for (Predicate p : and.getPredicates()) {
124      if (p instanceof GreaterThanOrEqualsPredicate) {
125        lower = (GreaterThanOrEqualsPredicate) p;
126      } else if (p instanceof LessThanOrEqualsPredicate) {
127        upper = (LessThanOrEqualsPredicate) p;
128      }
129    }
130    if (lower == null || upper == null || lower.getKey() != upper.getKey()) {
131      throw new IllegalArgumentException("no valid range");
132    }
133    addQueryParam(lower.getKey(), range(lower.getValue(), upper.getValue()));
134  }
135
136  private String range(String from, String to) {
137    if (Strings.isNullOrEmpty(from)) {
138      from = WILDCARD;
139    }
140    if (Strings.isNullOrEmpty(to)) {
141      to = WILDCARD;
142    }
143    return from + "," + to;
144  }
145
146  private void visit(DisjunctionPredicate or) throws IllegalStateException {
147    State oldState = state;
148    if (state == State.OR) {
149      throw new IllegalStateException("OR within OR filters not supported");
150    }
151    state = State.OR;
152
153    for (Predicate p : or.getPredicates()) {
154      visit(p);
155    }
156    state = oldState;
157  }
158
159  private void visit(EqualsPredicate predicate) {
160    addQueryParam(predicate.getKey(), predicate.getValue());
161  }
162
163  private void visit(IsNotNullPredicate predicate) {
164    addQueryParam(predicate.getParameter(), "*");
165  }
166
167  private void visit(LikePredicate predicate) {
168    throw new IllegalArgumentException("LIKE operator not supported in web queries");
169  }
170
171  private void visit(GreaterThanPredicate predicate) {
172    throw new IllegalArgumentException("GREATER_THAN_OPERATOR operator not supported in web queries");
173  }
174
175  private void visit(GreaterThanOrEqualsPredicate p) {
176    addQueryParam(p.getKey(), range(p.getValue(), null));
177  }
178
179  private void visit(LessThanPredicate predicate) {
180    throw new IllegalArgumentException("LESS_THAN_OPERATOR operator not supported in web queries");
181  }
182
183  private void visit(LessThanOrEqualsPredicate p) {
184    addQueryParam(p.getKey(), range(null, p.getValue()));
185  }
186
187  private void visit(WithinPredicate within) {
188    addQueryParam(OccurrenceSearchParameter.GEOMETRY, extractPolygonValues(within.getGeometry()));
189  }
190
191  private String extractPolygonValues(String withinValue) {
192    Matcher m = POLYGON_PATTERN.matcher(withinValue);
193    if (m.find()) {
194      return m.group(1);
195    }
196    throw new IllegalArgumentException("No valid polygon WKT: " + withinValue);
197  }
198
199  private void visit(InPredicate in) {
200    for (String val : in.getValues()) {
201      addQueryParam(in.getKey(), val);
202    }
203  }
204
205  private void visit(NotPredicate not) throws IllegalStateException {
206    throw new IllegalArgumentException("NOT operator not supported in web queries");
207  }
208
209  private void addQueryParam(OccurrenceSearchParameter param, String value) {
210    // verify that last param if existed was the same
211    if (lastParam != null && param != lastParam) {
212      throw new IllegalArgumentException("Mix of search params not supported");
213    }
214
215    if (!filter.containsKey(param)) {
216      filter.put(param, Lists.<String>newLinkedList());
217    }
218    filter.get(param).add(value);
219    lastParam = param;
220  }
221
222  private void visit(Predicate p) throws IllegalStateException {
223    Method method = null;
224    try {
225      method = getClass().getDeclaredMethod("visit", new Class[] {p.getClass()});
226    } catch (NoSuchMethodException e) {
227      LOG
228        .warn(
229          "Visit method could not be found. That means a Predicate has been passed in that is unknown to this "
230            + "class",
231          e);
232      throw new IllegalArgumentException("Unknown Predicate", e);
233    }
234    try {
235      method.setAccessible(true);
236      method.invoke(this, p);
237    } catch (IllegalAccessException e) {
238      LOG.error(
239        "This should never happen as we set accessible to true explicitly before. Probably a programming error", e);
240      throw new RuntimeException("Programming error", e);
241    } catch (InvocationTargetException e) {
242      LOG.info("Exception thrown while building the Hive Download", e);
243      throw new IllegalArgumentException(e);
244    }
245  }
