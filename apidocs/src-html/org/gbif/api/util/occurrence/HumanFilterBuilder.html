046/**
047 * This class builds a human readable filter from a {@link org.gbif.api.model.occurrence.predicate.Predicate} hierarchy.
048 * This class is not thread safe, create a new instance for every use if concurrent calls to {#humanFilter} is expected.
049 * The IN predicate is not yet supported and you'll get an IllegalArgumentException.
050 * This builder only supports predicates that follow our search query parameters style with multiple values for the same
051 * parameter being logically disjunct (OR) while different search parameters are logically combined (AND). Therefore
052 * the {#humanFilter(Predicate p)} result is a map of OccurrenceSearchParameter (AND'ed) to a list of values (OR'ed).
053 */
054public class HumanFilterBuilder {
055
056  private enum State {
057    ROOT, AND, OR
058  }
059
060  private static final Logger LOG = LoggerFactory.getLogger(HumanFilterBuilder.class);
061  private static final String EQUALS_OPERATOR = "";
062  private static final String GREATER_THAN_OPERATOR_XML = "&gt;";
063  private static final String GREATER_THAN_EQUALS_OPERATOR_XML = "&gt;=";
064  private static final String LESS_THAN_OPERATOR_XML = "&lt;";
065  private static final String LESS_THAN_EQUALS_OPERATOR_XML = "&lt;=";
066
067  private static final String GREATER_THAN_OPERATOR = ">";
068  private static final String GREATER_THAN_EQUALS_OPERATOR = ">=";
069  private static final String LESS_THAN_OPERATOR = "<";
070  private static final String LESS_THAN_EQUALS_OPERATOR = "<=";
071
072  private static final String IS_NOT_NULL_OPERATOR = "IS NOT NULL";
073
074  private static final String LIKE_OPERATOR = "~";
075  private Map<OccurrenceSearchParameter, LinkedList<String>> filter;;
076  private State state;
077  private OccurrenceSearchParameter lastParam;
078  private final DatasetService datasetService;
079  private final NameUsageService nameUsageService;
080  private final ResourceBundle resourceBundle;
081  private final boolean escapeXML;
082
083  public HumanFilterBuilder(ResourceBundle resourceBundle, DatasetService datasetService,
084    NameUsageService nameUsageService, boolean escapeXML) {
085    this.datasetService = datasetService;
086    this.nameUsageService = nameUsageService;
087    this.resourceBundle = resourceBundle;
088    this.escapeXML = escapeXML;
089  }
090
091  /**
092   * @param p the predicate to convert
093   * @return a list of anded parameters with multiple values to be combined with OR
094   * @throws IllegalStateException if more complex predicates than the portal handles are supplied
095   */
096  public synchronized Map<OccurrenceSearchParameter, LinkedList<String>> humanFilter(Predicate p) {
097    filter = Maps.newHashMap();
098    state = State.ROOT;
099    lastParam = null;
100    visit(p);
101    return filter;
102  }
103
104  private void addParamValue(OccurrenceSearchParameter param, String op, String value) {
105    addParamValue(param, op + getHumanValue(param, value));
106  }
107
108  /**
109   * Gets the human readable value of the parameter value.
110   */
111  private String getHumanValue(OccurrenceSearchParameter param, String value) {
112    String humanValue;
113    // lookup values
114    switch (param) {
115      case TAXON_KEY:
116        humanValue = lookupTaxonKey(value);
117        break;
118      case DATASET_KEY:
119        humanValue = lookupDatasetKey(value);
120        break;
121      case COUNTRY:
122        humanValue = lookupCountryCode(value);
123        break;
124      case BASIS_OF_RECORD:
125        humanValue = lookupBasisOfRecord(value);
126        break;
127      case MONTH:
128        humanValue = lookupMonth(value);
129        break;
130
131      default:
132        humanValue = value;
133    }
134    return humanValue;
135  }
136
137  private void addParamValue(OccurrenceSearchParameter param, String op) {
138    // verify that last param if existed was the same
139    if (lastParam != null && param != lastParam) {
140      throw new IllegalArgumentException("Mix of search params not supported");
141    }
142
143    if (!filter.containsKey(param)) {
144      filter.put(param, Lists.<String>newLinkedList());
145    }
146
147    filter.get(param).add(op);
148    lastParam = param;
149  }
150
151  /**
152   * Gets the greater-than-equals operator. If escapeXML is true returns the XML version of the operator.
153   */
154  private String getGteOperator() {
155    return escapeXML ? GREATER_THAN_EQUALS_OPERATOR_XML : GREATER_THAN_EQUALS_OPERATOR;
156  }
157
158  /**
159   * Gets the greater-than operator. If escapeXML is true returns the XML version of the operator.
160   */
161  private String getGtOperator() {
162    return escapeXML ? GREATER_THAN_OPERATOR_XML : GREATER_THAN_OPERATOR;
163  }
164
165  /**
166   * Gets the less-than-equals operator. If escapeXML is true returns the XML version of the operator.
167   */
168  private String getLteOperator() {
169    return escapeXML ? LESS_THAN_EQUALS_OPERATOR_XML : LESS_THAN_EQUALS_OPERATOR;
170  }
171
172  /**
173   * Gets the less-than operator. If escapeXML is true returns the XML version of the operator.
174   */
175  private String getLtOperator() {
176    return escapeXML ? LESS_THAN_OPERATOR_XML : LESS_THAN_OPERATOR;
177  }
178
179  private String lookupBasisOfRecord(String value) {
180    return resourceBundle.getString("enum.basisofrecord." + value);
181  }
182
183  private String lookupCountryCode(String code) {
184    Country c = Country.fromIsoCode(code);
185    if (c != null) {
186      return c.getTitle();
187    }
188    return code;
189  }
190
191  private String lookupDatasetKey(String value) {
192    try {
193      return datasetService.get(UUID.fromString(value)).getTitle();
194    } catch (Exception e) {
195      LOG.warn("Cannot get title for dataset {}", value);
196    }
197    return value;
198  }
199
200  private String lookupMonth(String month) {
201    return resourceBundle.getString("enum.month." + month);
202  }
203
204  private String lookupTaxonKey(String value) {
205    try {
206      return nameUsageService.get(Integer.parseInt(value), null).getScientificName();
207    } catch (Exception e) {
208      LOG.warn("Cannot get name for usage {}", value);
209    }
210    return value;
211  }
212
213  private void visit(ConjunctionPredicate and) throws IllegalStateException {
214    // ranges are allowed underneath root - try first
215    try {
216      visitRange(and);
217      return;
218    } catch (IllegalArgumentException e) {
219      // must be a root AND
220    }
221
222    if (state != State.ROOT) {
223      throw new IllegalStateException("AND must be a root predicate or a valid range");
224    }
225    state = State.AND;
226
227    for (Predicate p : and.getPredicates()) {
228      lastParam = null;
229      visit(p);
230    }
231    state = State.ROOT;
232  }
233
234  private void visit(DisjunctionPredicate or) throws IllegalStateException {
235    State oldState = state;
236    if (state == State.OR) {
237      throw new IllegalStateException("OR within OR filters not supported");
238    }
239    state = State.OR;
240
241    for (Predicate p : or.getPredicates()) {
242      visit(p);
243    }
244    state = oldState;
245  }
246
247  private void visit(EqualsPredicate predicate) {
248    addParamValue(predicate.getKey(), EQUALS_OPERATOR, predicate.getValue());
249  }
250
251  private void visit(GreaterThanOrEqualsPredicate predicate) {
252    addParamValue(predicate.getKey(), getGteOperator(), predicate.getValue());
253  }
254
255  private void visit(GreaterThanPredicate predicate) {
256    addParamValue(predicate.getKey(), getGtOperator(), predicate.getValue());
257  }
258
259  private void visit(InPredicate in) {
260    for (String val : in.getValues()) {
261      addParamValue(in.getKey(), EQUALS_OPERATOR, val);
262    }
263  }
264
265  private void visit(LessThanOrEqualsPredicate predicate) {
266    addParamValue(predicate.getKey(), getLteOperator(), predicate.getValue());
267  }
268
269  private void visit(LessThanPredicate predicate) {
270    addParamValue(predicate.getKey(), getLtOperator(), predicate.getValue());
271  }
272
273  private void visit(LikePredicate predicate) {
274    addParamValue(predicate.getKey(), LIKE_OPERATOR, predicate.getValue());
275  }
276
277  private void visit(NotPredicate not) throws IllegalStateException {
278    if (not.getPredicate() instanceof SimplePredicate) {
279      visit(not.getPredicate());
280      SimplePredicate sp = (SimplePredicate) not.getPredicate();
281      // now prefix the last value with NOT
282      String notValue = "NOT (" + filter.get(sp.getKey()).removeLast() + ")";
283      filter.get(sp.getKey()).add(notValue);
284
285    } else {
286      throw new IllegalArgumentException("NOT predicate must be followed by a simple predicate");
287    }
288  }
289
290  private void visit(IsNotNullPredicate predicate) {
291    addParamValue(predicate.getParameter(), IS_NOT_NULL_OPERATOR);
292  }
293
294
295  private void visit(Predicate p) throws IllegalStateException {
296    Method method = null;
297    try {
298      method = getClass().getDeclaredMethod("visit", new Class[] {p.getClass()});
299    } catch (NoSuchMethodException e) {
300      LOG
301        .warn(
302          "Visit method could not be found. That means a Predicate has been passed in that is unknown to this "
303            + "class",
304          e);
305      throw new IllegalArgumentException("Unknown Predicate", e);
306    }
307    try {
308      method.setAccessible(true);
309      method.invoke(this, p);
310    } catch (IllegalAccessException e) {
311      LOG.error(
312        "This should never happen as we set accessible to true explicitly before. Probably a programming error", e);
313      throw new RuntimeException("Programming error", e);
314    } catch (InvocationTargetException e) {
315      LOG.info("Exception thrown while building the Hive Download", e);
316      throw new IllegalArgumentException(e);
317    }
318  }
319
320  private void visit(WithinPredicate within) {
321    addParamValue(OccurrenceSearchParameter.GEOMETRY, "", within.getGeometry());
322  }
323
324  private void visitRange(ConjunctionPredicate and) {
325    if (and.getPredicates().size() != 2) {
326      throw new IllegalArgumentException("no valid range");
327    }
328    GreaterThanOrEqualsPredicate lower = null;
329    LessThanOrEqualsPredicate upper = null;
330    for (Predicate p : and.getPredicates()) {
331      if (p instanceof GreaterThanOrEqualsPredicate) {
332        lower = (GreaterThanOrEqualsPredicate) p;
333      } else if (p instanceof LessThanOrEqualsPredicate) {
334        upper = (LessThanOrEqualsPredicate) p;
335      }
336    }
337    if (lower == null || upper == null || lower.getKey() != upper.getKey()) {
338      throw new IllegalArgumentException("no valid range");
339    }
340    addParamValue(lower.getKey(), "", lower.getValue() + "-" + upper.getValue());
341  }
