051/**
052 * This class builds a human readable filter from a {@link org.gbif.api.model.occurrence.predicate.Predicate} hierarchy.
053 * This class is not thread safe, create a new instance for every use if concurrent calls to {#humanFilter} is expected.
054 * The IN predicate is not yet supported and you'll get an IllegalArgumentException.
055 * This builder only supports predicates that follow our search query parameters style with multiple values for the same
056 * parameter being logically disjunct (OR) while different search parameters are logically combined (AND). Therefore
057 * the {#humanFilter(Predicate p)} result is a map of OccurrenceSearchParameter (AND'ed) to a list of values (OR'ed).
058 */
059public class HumanFilterBuilder {
060
061  private enum State {
062    ROOT, AND, OR
063  }
064
065  private static final Logger LOG = LoggerFactory.getLogger(HumanFilterBuilder.class);
066  private static final String DEFAULT_BUNDLE = "org/gbif/api/util/occurrence/filter.properties";
067
068  private static final String EQUALS_OPERATOR = "";
069  private static final String GREATER_THAN_OPERATOR_XML = "&gt; ";
070  private static final String GREATER_THAN_EQUALS_OPERATOR_XML = "&gt;= ";
071  private static final String LESS_THAN_OPERATOR_XML = "&lt; ";
072  private static final String LESS_THAN_EQUALS_OPERATOR_XML = "&lt;= ";
073
074  private static final String GREATER_THAN_OPERATOR = "> ";
075  private static final String GREATER_THAN_EQUALS_OPERATOR = ">= ";
076  private static final String LESS_THAN_OPERATOR = "< ";
077  private static final String LESS_THAN_EQUALS_OPERATOR = "<= ";
078
079  private static final String NOT_OPERATOR = "not";
080  private static final String IS_NOT_NULL_OPERATOR = "is not null";
081
082  private static final String LIKE_OPERATOR = "~";
083  private Map<OccurrenceSearchParameter, LinkedList<String>> filter;;
084  private State state;
085  private OccurrenceSearchParameter lastParam;
086  private final DatasetService datasetService;
087  private final NameUsageService nameUsageService;
088  private final ResourceBundle resourceBundle;
089  private final boolean escapeXML;
090
091  /**
092   * Creates a new human filter builder using the default resource bundle for enumerations from the GBIF API.
093   */
094  public HumanFilterBuilder(DatasetService datasetService, NameUsageService nameUsageService, boolean escapeXML) {
095    this.datasetService = datasetService;
096    this.nameUsageService = nameUsageService;
097    this.escapeXML = escapeXML;
098    try {
099      InputStream bundleStream = HumanFilterBuilder.class.getClassLoader().getResourceAsStream(DEFAULT_BUNDLE);
100      this.resourceBundle = new PropertyResourceBundle(bundleStream);
101    } catch (IOException e) {
102      throw new RuntimeException(e);
103    }
104  }
105
106  /**
107   * Creates a new human filter builder using the supplied resource bundle.
108   * You must make sure all possible enum values for the OccurrenceSearchParameter are listed in this bundle!
109   * For default english text it is recommended to use the default bundle with the other constructor.
110   */
111  public HumanFilterBuilder(ResourceBundle resourceBundle, DatasetService datasetService,
112    NameUsageService nameUsageService, boolean escapeXML) {
113    this.datasetService = datasetService;
114    this.nameUsageService = nameUsageService;
115    this.resourceBundle = resourceBundle;
116    this.escapeXML = escapeXML;
117  }
118
119  /**
120   * @param p the predicate to convert
121   * @return a list of anded parameters with multiple values to be combined with OR
122   * @throws IllegalStateException if more complex predicates than the portal handles are supplied
123   */
124  public synchronized Map<OccurrenceSearchParameter, LinkedList<String>> humanFilter(Predicate p) {
125    filter = Maps.newLinkedHashMap();
126    state = State.ROOT;
127    lastParam = null;
128    visit(p);
129    return filter;
130  }
131
132  private void addParamValue(OccurrenceSearchParameter param, String op, String value) {
133    addParamValue(param, op + getHumanValue(param, value));
134  }
135
136  /**
137   * Gets the human readable value of the parameter value.
138   */
139  private String getHumanValue(OccurrenceSearchParameter param, String value) {
140    String humanValue;
141    // lookup values
142    switch (param) {
143      case TAXON_KEY:
144        humanValue = lookupTaxonKey(value);
145        break;
146      case DATASET_KEY:
147        humanValue = lookupDatasetKey(value);
148        break;
149      case COUNTRY:
150      case PUBLISHING_COUNTRY:
151        humanValue = lookupCountryCode(value);
152        break;
153      case CONTINENT:
154        humanValue = lookupContinent(value);
155        break;
156      case MONTH:
157        humanValue = lookupMonth(value);
158        break;
159
160      default:
161        if (param.type().isEnum()) {
162          humanValue = lookupEnum(param.type(), value);
163        } else {
164          humanValue = value;
165        }
166    }
167    // add unit symbol
168    if (param==OccurrenceSearchParameter.DEPTH || param==OccurrenceSearchParameter.ELEVATION) {
169      humanValue = humanValue + "m";
170    }
171    return humanValue;
172  }
173
174  private String lookupContinent(String value) {
175    Continent c = VocabularyUtils.lookupEnum(value, Continent.class);
176    return c.getTitle();
177  }
178
179  private void addParamValue(OccurrenceSearchParameter param, String op) {
180    // verify that last param if existed was the same
181    if (lastParam != null && param != lastParam) {
182      throw new IllegalArgumentException("Mix of search params not supported: "+param);
183    }
184
185    if (!filter.containsKey(param)) {
186      filter.put(param, Lists.<String>newLinkedList());
187    }
188
189    filter.get(param).add(op);
190    lastParam = param;
191  }
192
193  /**
194   * Gets the greater-than-equals operator. If escapeXML is true returns the XML version of the operator.
195   */
196  private String getGteOperator() {
197    return escapeXML ? GREATER_THAN_EQUALS_OPERATOR_XML : GREATER_THAN_EQUALS_OPERATOR;
198  }
199
200  /**
201   * Gets the greater-than operator. If escapeXML is true returns the XML version of the operator.
202   */
203  private String getGtOperator() {
204    return escapeXML ? GREATER_THAN_OPERATOR_XML : GREATER_THAN_OPERATOR;
205  }
206
207  /**
208   * Gets the less-than-equals operator. If escapeXML is true returns the XML version of the operator.
209   */
210  private String getLteOperator() {
211    return escapeXML ? LESS_THAN_EQUALS_OPERATOR_XML : LESS_THAN_EQUALS_OPERATOR;
212  }
213
214  /**
215   * Gets the less-than operator. If escapeXML is true returns the XML version of the operator.
216   */
217  private String getLtOperator() {
218    return escapeXML ? LESS_THAN_OPERATOR_XML : LESS_THAN_OPERATOR;
219  }
220
221  private String lookupEnum(Class clazz, String value) {
222    return resourceBundle.getString("enum." + clazz.getSimpleName().toLowerCase() + "." + value);
223  }
224
225  private String lookupCountryCode(String code) {
226    Country c = Country.fromIsoCode(code);
227    if (c != null) {
228      return c.getTitle();
229    }
230    return code;
231  }
232
233  private String lookupDatasetKey(String value) {
234    try {
235      return datasetService.get(UUID.fromString(value)).getTitle();
236    } catch (Exception e) {
237      LOG.warn("Cannot get title for dataset {}", value);
238    }
239    return value;
240  }
241
242  private String lookupMonth(String month) {
243    return resourceBundle.getString("enum.month." + month);
244  }
245
246  private String lookupTaxonKey(String value) {
247    try {
248      return nameUsageService.get(Integer.parseInt(value), null).getScientificName();
249    } catch (Exception e) {
250      LOG.warn("Cannot get name for usage {}", value);
251    }
252    return value;
253  }
254
255  private void visit(ConjunctionPredicate and) throws IllegalStateException {
256    // ranges are allowed underneath root - try first
257    try {
258      visitRange(and);
259      return;
260    } catch (IllegalArgumentException e) {
261      // must be a root AND
262    }
263
264    if (state != State.ROOT) {
265      throw new IllegalStateException("AND must be a root predicate or a valid range");
266    }
267    state = State.AND;
268
269    for (Predicate p : and.getPredicates()) {
270      lastParam = null;
271      visit(p);
272    }
273    state = State.ROOT;
274  }
275
276  private void visit(DisjunctionPredicate or) throws IllegalStateException {
277    State oldState = state;
278    if (state == State.OR) {
279      throw new IllegalStateException("OR within OR filters not supported");
280    }
281    state = State.OR;
282
283    for (Predicate p : or.getPredicates()) {
284      visit(p);
285    }
286    state = oldState;
287  }
288
289  private void visit(EqualsPredicate predicate) {
290    addParamValue(predicate.getKey(), EQUALS_OPERATOR, predicate.getValue());
291  }
292
293  private void visit(GreaterThanOrEqualsPredicate predicate) {
294    addParamValue(predicate.getKey(), getGteOperator(), predicate.getValue());
295  }
296
297  private void visit(GreaterThanPredicate predicate) {
298    addParamValue(predicate.getKey(), getGtOperator(), predicate.getValue());
299  }
300
301  private void visit(InPredicate in) {
302    for (String val : in.getValues()) {
303      addParamValue(in.getKey(), EQUALS_OPERATOR, val);
304    }
305  }
306
307  private void visit(LessThanOrEqualsPredicate predicate) {
308    addParamValue(predicate.getKey(), getLteOperator(), predicate.getValue());
309  }
310
311  private void visit(LessThanPredicate predicate) {
312    addParamValue(predicate.getKey(), getLtOperator(), predicate.getValue());
313  }
314
315  private void visit(LikePredicate predicate) {
316    addParamValue(predicate.getKey(), LIKE_OPERATOR, predicate.getValue());
317  }
318
319  private void visit(NotPredicate not) throws IllegalStateException {
320    if (not.getPredicate() instanceof SimplePredicate) {
321      visit(not.getPredicate());
322      SimplePredicate sp = (SimplePredicate) not.getPredicate();
323      // now prefix the last value with NOT
324      String notValue = NOT_OPERATOR + " (" + filter.get(sp.getKey()).removeLast() + ")";
325      filter.get(sp.getKey()).add(notValue);
326
327    } else {
328      throw new IllegalArgumentException("NOT predicate must be followed by a simple predicate");
329    }
330  }
331
332  private void visit(IsNotNullPredicate predicate) {
333    addParamValue(predicate.getParameter(), IS_NOT_NULL_OPERATOR);
334  }
335
336
337  private void visit(Predicate p) throws IllegalStateException {
338    Method method = null;
339    try {
340      method = getClass().getDeclaredMethod("visit", new Class[] {p.getClass()});
341    } catch (NoSuchMethodException e) {
342      LOG
343        .warn(
344          "Visit method could not be found. That means a Predicate has been passed in that is unknown to this "
345            + "class",
346          e);
347      throw new IllegalArgumentException("Unknown Predicate", e);
348    }
349    try {
350      method.setAccessible(true);
351      method.invoke(this, p);
352    } catch (IllegalAccessException e) {
353      LOG.error(
354        "This should never happen as we set accessible to true explicitly before. Probably a programming error", e);
355      throw new RuntimeException("Programming error", e);
356    } catch (InvocationTargetException e) {
357      LOG.info("Exception thrown while building the human query string", e);
358      throw new IllegalArgumentException(e);
359    }
360  }
361
362  private void visit(WithinPredicate within) {
363    addParamValue(OccurrenceSearchParameter.GEOMETRY, "", within.getGeometry());
364  }
365
366  private void visitRange(ConjunctionPredicate and) {
367    if (and.getPredicates().size() != 2) {
368      throw new IllegalArgumentException("no valid range");
369    }
370    GreaterThanOrEqualsPredicate lower = null;
371    LessThanOrEqualsPredicate upper = null;
372    for (Predicate p : and.getPredicates()) {
373      if (p instanceof GreaterThanOrEqualsPredicate) {
374        lower = (GreaterThanOrEqualsPredicate) p;
375      } else if (p instanceof LessThanOrEqualsPredicate) {
376        upper = (LessThanOrEqualsPredicate) p;
377      }
378    }
379    if (lower == null || upper == null || lower.getKey() != upper.getKey()) {
380      throw new IllegalArgumentException("no valid range");
381    }
382    addParamValue(lower.getKey(), "", lower.getValue() + "-" + upper.getValue());
383  }
