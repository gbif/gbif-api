001package org.gbif.api.util.iterables;
002
003import org.gbif.api.model.common.paging.PagingConstants;
004import org.gbif.api.model.registry.Dataset;
005import org.gbif.api.model.registry.Node;
006import org.gbif.api.model.registry.Organization;
007import org.gbif.api.service.registry.DatasetService;
008import org.gbif.api.service.registry.InstallationService;
009import org.gbif.api.service.registry.NetworkService;
010import org.gbif.api.service.registry.NodeService;

014
015import java.util.UUID;
016import javax.annotation.Nullable;
017
018import com.google.common.collect.ImmutableList;
019import org.slf4j.Logger;
020import org.slf4j.LoggerFactory;
021
022/**
023 * Factory constructing registry entity iterables using specific pagers under the hood.
024 */
025public class Iterables {
026    private static final Logger LOG = LoggerFactory.getLogger(Iterables.class);
027
028    /**
029     * @param key a valid dataset, organization or installation key. If null all datasets will be iterated over
030     * @throws IllegalArgumentException if given key is not existing
031     */
032    public static Iterable<Dataset> datasets(@Nullable UUID key, @Nullable DatasetType type,
033                                             DatasetService ds, OrganizationService os, InstallationService is, NetworkService ns, NodeService nos) {
034        return datasets(key, type, ds, os, is, ns, nos, PagingConstants.DEFAULT_PARAM_LIMIT);
035    }
036
037    /**
038     * Returns a dataset iterable by testing the given registry key first to see whether it is a dataset, organization or installation.
039     * In case of an organization key the published datasets will be returned.
040     *
041     * @param key a valid dataset, organization or installation key. If null all datasets will be iterated over
042     * @param pageSize to use when talking to the registry
043     * @throws IllegalArgumentException if given key is not existing
044     */
045    public static Iterable<Dataset> datasets(@Nullable UUID key, @Nullable DatasetType type,
046                                             DatasetService ds, OrganizationService os, InstallationService is, NetworkService ns, NodeService nos,
047                                             int pageSize) {
048        if (key == null) {
049            LOG.info("Iterate over all {} datasets", type == null ? "" : type);
050            return new DatasetPager(ds, type, pageSize);
051
052        } else if (isDataset(key, ds)) {
053            LOG.info("Iterate over dataset {}", key);
054            return ImmutableList.of(ds.get(key));
055
056        } else if (isOrganization(key, os)) {
057            LOG.info("Iterate over all {} datasets published by {}", type == null ? "" : type, key);
058            return new OrgPublishingPager(os, key, type, pageSize);
059
060        } else if (isInstallation(key, is)) {
061            LOG.info("Iterate over all {} datasets hosted by installation {}", type == null ? "" : type, key);
062            return new InstallationPager(is, key, type, pageSize);
063
064        } else if (isNode(key, nos)) {
065            LOG.info("Iterate over all {} datasets endorsed by node {}", type == null ? "" : type, key);
066            return new NetworkPager(ns, key, type, pageSize);
067
068        } else if (isNetwork(key, ns)) {
069            LOG.info("Iterate over all {} datasets belonging to network {}", type == null ? "" : type, key);
070            return new NodeDatasetPager(nos, key, type, pageSize);
071        }
072        throw new IllegalArgumentException("Given key is no valid GBIF registry key: " + key);
073    }
074
075    /**
076     * @param type an optional filter to just include the given dataset type
077     */
078    public static Iterable<Dataset> datasets(@Nullable DatasetType type, DatasetService service) {
079        LOG.info("Iterate over all {} datasets", type == null ? "" : type);
080        return new DatasetPager(service, type, PagingConstants.DEFAULT_PARAM_LIMIT);
081    }
082
083    /**
084     * @param key a valid organization key
085     * @param type an optional filter to just include the given dataset type
086     */
087    public static Iterable<Dataset> publishedDatasets(UUID key, @Nullable DatasetType type, OrganizationService service) {
088        LOG.info("Iterate over all {} datasets published by {}", type == null ? "" : type, key);
089        return new OrgPublishingPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
090    }
091
092    /**
093     * @param key a valid organization key
094     * @param type an optional filter to just include the given dataset type
095     */
096    public static Iterable<Dataset> hostedDatasets(UUID key, @Nullable DatasetType type, OrganizationService service) {
097        LOG.info("Iterate over all {} datasets hosted organization by {}", type == null ? "" : type, key);
098        return new OrgHostingPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
099    }
100
101    /**
102     * @param key a valid installation key
103     * @param type an optional filter to just include the given dataset type
104     */
105    public static Iterable<Dataset> hostedDatasets(UUID key, @Nullable DatasetType type, InstallationService service) {
106        LOG.info("Iterate over all {} datasets hosted organization by {}", type == null ? "" : type, key);
107        return new InstallationPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
108    }
109
110    /**
111     * @param key a valid dataset key
112     */
113    public static Iterable<Dataset> constituentDatasets(UUID key, DatasetService service) {
114        LOG.info("Iterate over all constituent datasets of {}", key);
115        return new DatasetConstituentPager(service, key, PagingConstants.DEFAULT_PARAM_LIMIT);
116    }
117
118    /**
119     * Iterates over all constituents of a given network.
120     * @param key a valid network key
121     * @param type an optional filter to just include the given dataset type
122     */
123    public static Iterable<Dataset> networkDatasets(UUID key, @Nullable DatasetType type, NetworkService service) {
124        LOG.info("Iterate over all {} datasets belonging to network {}", type == null ? "" : type, key);
125        return new NetworkPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
126    }
127
128    /**
129     * @param nodeKey a valid endorsing node key
130     * @param type an optional filter to just include the given dataset type
131     */
132    public static Iterable<Dataset> endorsedDatasets(UUID nodeKey, @Nullable DatasetType type, NodeService service) {
133        LOG.info("Iterate over all {} datasets endorsed by node {}", type == null ? "" : type, nodeKey);
134        return new NodeDatasetPager(service, nodeKey, type, PagingConstants.DEFAULT_PARAM_LIMIT);
135    }
136
137    /**
138     * @param country an optional country filter
139     */
140    public static Iterable<Organization> organizations(@Nullable Country country, OrganizationService service) {
141        LOG.info("Iterate over all organizations {}", country == null ? "" : "from country "+country);
142        return new OrganizationPager(service, country, PagingConstants.DEFAULT_PARAM_LIMIT);
143    }
144
145    /**
146     * @param nodeKey a valid endorsing node key
147     */
148    public static Iterable<Organization> endorsedOrganizations(UUID nodeKey, NodeService service) {
149        LOG.info("Iterate over all organizations endorsed by node {}", nodeKey);
150        return new NodeOrganizationPager(service, nodeKey, PagingConstants.DEFAULT_PARAM_LIMIT);
151    }
152
153    /**
154     * Iterate over all endorsing nodes
155     */
156    public static Iterable<Node> nodes(NodeService service) {
157        LOG.info("Iterate over all nodes");
158        return new NodePager(service, PagingConstants.DEFAULT_PARAM_LIMIT);
159    }
160
161    private static boolean isDataset(UUID key, DatasetService ds) {
162        return ds.get(key) != null;
163    }
164
165    private static boolean isOrganization(UUID key, OrganizationService os) {
166        return os.get(key) != null;
167    }
168
169    private static boolean isInstallation(UUID key, InstallationService is) {
170        return is.get(key) != null;
171    }
172
173    private static boolean isNetwork(UUID key, NetworkService ns) {
174        return ns.get(key) != null;
175    }
176
177    private static boolean isNode(UUID key, NodeService ns) {
178        return ns.get(key) != null;
179    }
180}
