041/**
042 * Factory constructing registry entity iterables using specific pagers under the hood.
043 */
044@SuppressWarnings("unused")
045public class Iterables {
046    private static final Logger LOG = LoggerFactory.getLogger(Iterables.class);
047
048    /**
049     * Private default constructor.
050     */
051    private Iterables() {
052        //empty private constructor
053    }
054
055    /**
056     * @param key a valid dataset, organization or installation key. If null all datasets will be iterated over
057     * @throws IllegalArgumentException if given key is not existing
058     */
059    public static Iterable<Dataset> datasets(@Nullable UUID key, @Nullable DatasetType type,
060                                             DatasetService ds, OrganizationService os, InstallationService is,
061                                             NetworkService ns, NodeService nos) {
062        return datasets(key, type, ds, os, is, ns, nos, PagingConstants.DEFAULT_PARAM_LIMIT);
063    }
064
065    /**
066     * Returns a dataset iterable by testing the given registry key first to see whether it is a dataset, organization or installation.
067     * In case of an organization key the published datasets will be returned.
068     *
069     * @param key a valid dataset, organization or installation key. If null all datasets will be iterated over
070     * @param pageSize to use when talking to the registry
071     * @throws IllegalArgumentException if given key is not existing
072     */
073    public static Iterable<Dataset> datasets(@Nullable UUID key, @Nullable DatasetType type,
074                                             DatasetService ds, OrganizationService os, InstallationService is,
075                                             NetworkService ns, NodeService nos, int pageSize) {
076        if (key == null) {
077            LOG.info("Iterate over all {} datasets", type == null ? "" : type);
078            return new DatasetPager(ds, type, pageSize);
079
080        } else if (isDataset(key, ds)) {
081            LOG.info("Iterate over dataset {}", key);
082            return Collections.unmodifiableList(Collections.singletonList(ds.get(key)));
083
084        } else if (isOrganization(key, os)) {
085            LOG.info("Iterate over all {} datasets published by {}", type == null ? "" : type, key);
086            return new OrgPublishingPager(os, key, type, pageSize);
087
088        } else if (isInstallation(key, is)) {
089            LOG.info("Iterate over all {} datasets hosted by installation {}", type == null ? "" : type, key);
090            return new InstallationPager(is, key, type, pageSize);
091
092        } else if (isNode(key, nos)) {
093            LOG.info("Iterate over all {} datasets endorsed by node {}", type == null ? "" : type, key);
094            return new NetworkPager(ns, key, type, pageSize);
095
096        } else if (isNetwork(key, ns)) {
097            LOG.info("Iterate over all {} datasets belonging to network {}", type == null ? "" : type, key);
098            return new NodeDatasetPager(nos, key, type, pageSize);
099        }
100        throw new IllegalArgumentException("Given key is no valid GBIF registry key: " + key);
101    }
102
103    /**
104     * @param type an optional filter to just include the given dataset type
105     */
106    public static Iterable<Dataset> datasets(@Nullable DatasetType type, DatasetService service) {
107        LOG.info("Iterate over all {} datasets", type == null ? "" : type);
108        return new DatasetPager(service, type, PagingConstants.DEFAULT_PARAM_LIMIT);
109    }
110
111    /**
112     * @param key a valid organization key
113     * @param type an optional filter to just include the given dataset type
114     */
115    public static Iterable<Dataset> publishedDatasets(UUID key, @Nullable DatasetType type, OrganizationService service) {
116        LOG.info("Iterate over all {} datasets published by {}", type == null ? "" : type, key);
117        return new OrgPublishingPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
118    }
119
120    /**
121     * @param key a valid organization key
122     * @param type an optional filter to just include the given dataset type
123     */
124    public static Iterable<Dataset> hostedDatasets(UUID key, @Nullable DatasetType type, OrganizationService service) {
125        LOG.info("Iterate over all {} datasets hosted by organization {}", type == null ? "" : type, key);
126        return new OrgHostingPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
127    }
128
129    /**
130     * @param key a valid installation key
131     * @param type an optional filter to just include the given dataset type
132     */
133    public static Iterable<Dataset> hostedDatasets(UUID key, @Nullable DatasetType type, InstallationService service) {
134        LOG.info("Iterate over all {} datasets hosted by installation {}", type == null ? "" : type, key);
135        return new InstallationPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
136    }
137
138    /**
139     * @param key a valid dataset key
140     */
141    public static Iterable<Dataset> constituentDatasets(UUID key, DatasetService service) {
142        LOG.info("Iterate over all constituent datasets of {}", key);
143        return new DatasetConstituentPager(service, key, PagingConstants.DEFAULT_PARAM_LIMIT);
144    }
145
146    /**
147     * Iterates over all constituents of a given network.
148     * @param key a valid network key
149     * @param type an optional filter to just include the given dataset type
150     */
151    public static Iterable<Dataset> networkDatasets(UUID key, @Nullable DatasetType type, NetworkService service) {
152        LOG.info("Iterate over all {} datasets belonging to network {}", type == null ? "" : type, key);
153        return new NetworkPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
154    }
155
156    /**
157     * @param nodeKey a valid endorsing node key
158     * @param type an optional filter to just include the given dataset type
159     */
160    public static Iterable<Dataset> endorsedDatasets(UUID nodeKey, @Nullable DatasetType type, NodeService service) {
161        LOG.info("Iterate over all {} datasets endorsed by node {}", type == null ? "" : type, nodeKey);
162        return new NodeDatasetPager(service, nodeKey, type, PagingConstants.DEFAULT_PARAM_LIMIT);
163    }
164
165    /**
166     *
167     * @param pager producer function of next page response
168     * @return a  dataset iterable based on producer function
169     */
170    public static Iterable<Dataset> datasetsIterable(Function<PagingRequest, PagingResponse<Dataset>> pager) {
171        return new DatasetBasePager(null, PagingConstants.DEFAULT_PARAM_LIMIT) {
172          @Override
173          PagingResponse<Dataset> nextPage(PagingRequest page) {
174            return pager.apply(page);
175          }
176        };
177      }
178
179    /**
180     * @param country an optional country filter
181     */
182    public static Iterable<Organization> organizations(@Nullable Country country, OrganizationService service) {
183        LOG.info("Iterate over all organizations {}", country == null ? "" : "from country "+country);
184        return new OrganizationPager(service, country, PagingConstants.DEFAULT_PARAM_LIMIT);
185    }
186
187    /**
188     * @param nodeKey a valid endorsing node key
189     */
190    public static Iterable<Organization> endorsedOrganizations(UUID nodeKey, NodeService service) {
191        LOG.info("Iterate over all organizations endorsed by node {}", nodeKey);
192        return new NodeOrganizationPager(service, nodeKey, PagingConstants.DEFAULT_PARAM_LIMIT);
193    }
194
195    /**
196     * Iterate over all endorsing nodes
197     */
198    public static Iterable<Node> nodes(NodeService service) {
199        LOG.info("Iterate over all nodes");
200        return new NodePager(service, PagingConstants.DEFAULT_PARAM_LIMIT);
201    }
202
203    private static boolean isDataset(UUID key, DatasetService ds) {
204        return ds.get(key) != null;
205    }
206
207    private static boolean isOrganization(UUID key, OrganizationService os) {
208        return os.get(key) != null;
209    }
210
211    private static boolean isInstallation(UUID key, InstallationService is) {
212        return is.get(key) != null;
213    }
214
215    private static boolean isNetwork(UUID key, NetworkService ns) {
216        return ns.get(key) != null;
217    }
218
219    private static boolean isNode(UUID key, NodeService ns) {
220        return ns.get(key) != null;
221    }
