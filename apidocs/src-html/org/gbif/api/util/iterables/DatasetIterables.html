001package org.gbif.api.util.iterables;
002
003import org.gbif.api.model.common.paging.PagingConstants;
004import org.gbif.api.model.registry.Dataset;
005import org.gbif.api.service.registry.DatasetService;
006import org.gbif.api.service.registry.InstallationService;
007import org.gbif.api.service.registry.NetworkService;
008import org.gbif.api.service.registry.OrganizationService;
009import org.gbif.api.vocabulary.DatasetType;
010
011import java.util.UUID;
012import javax.annotation.Nullable;
013
014import com.google.common.collect.ImmutableList;
015import org.slf4j.Logger;
016import org.slf4j.LoggerFactory;
017
018/**
019 * Factory constructing dataset iterables that exclude deleted datasets.
020 */
021public class DatasetIterables {
022    private static final Logger LOG = LoggerFactory.getLogger(DatasetIterables.class);
023
024    /**
025     * @param key a valid dataset, organization or installation key. If null all datasets will be iterated over
026     * @throws IllegalArgumentException if given key is not existing
027     */
028    public static Iterable<Dataset> datasets(@Nullable UUID key, @Nullable DatasetType type,
029                                             DatasetService ds, OrganizationService os, InstallationService is, NetworkService ns) {
030        return datasets(key, type, ds, os, is, ns, PagingConstants.DEFAULT_PARAM_LIMIT);
031    }
032
033    /**
034     * Returns a dataset iterable by testing the given registry key first to see whether it is a dataset, organization or installation.
035     * In case of an organization key the published datasets will be returned.
036     *
037     * @param key a valid dataset, organization or installation key. If null all datasets will be iterated over
038     * @param pageSize to use when talking to the registry
039     * @throws IllegalArgumentException if given key is not existing
040     */
041    public static Iterable<Dataset> datasets(@Nullable UUID key, @Nullable DatasetType type,
042                                             DatasetService ds, OrganizationService os, InstallationService is, NetworkService ns, int pageSize) {
043        if (key == null) {
044            LOG.info("Iterate over all {} datasets", type == null ? "" : type);
045            return new DatasetPager(ds, type, pageSize);
046
047        } else if (isDataset(key, ds)) {
048            LOG.info("Iterate over dataset {}", key);
049            return ImmutableList.of(ds.get(key));
050
051        } else if (isOrganization(key, os)) {
052            LOG.info("Iterate over all {} datasets published by {}", type == null ? "" : type, key);
053            return new OrgPublishingPager(os, key, type, pageSize);
054
055        } else if (isInstallation(key, is)) {
056            LOG.info("Iterate over all {} datasets hosted by installation {}", type == null ? "" : type, key);
057            return new InstallationPager(is, key, type, pageSize);
058
059        } else if (isNetwork(key, ns)) {
060            LOG.info("Iterate over all {} datasets belonging to network {}", type == null ? "" : type, key);
061            return new NetworkPager(ns, key, type, pageSize);
062        }
063        throw new IllegalArgumentException("Given key is no valid GBIF registry key: " + key);
064    }
065
066    /**
067     * @param type an optional filter to just include the given dataset type
068     */
069    public static Iterable<Dataset> allDatasets(@Nullable DatasetType type, DatasetService service) {
070        LOG.info("Iterate over all {} datasets", type == null ? "" : type);
071        return new DatasetPager(service, type, PagingConstants.DEFAULT_PARAM_LIMIT);
072    }
073
074    /**
075     * @param key a valid organization key
076     * @param type an optional filter to just include the given dataset type
077     */
078    public static Iterable<Dataset> publishedDatasets(UUID key, @Nullable DatasetType type, OrganizationService service) {
079        LOG.info("Iterate over all {} datasets published by {}", type == null ? "" : type, key);
080        return new OrgPublishingPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
081    }
082
083    /**
084     * @param key a valid organization key
085     * @param type an optional filter to just include the given dataset type
086     */
087    public static Iterable<Dataset> hostedDatasets(UUID key, @Nullable DatasetType type, OrganizationService service) {
088        LOG.info("Iterate over all {} datasets hosted organization by {}", type == null ? "" : type, key);
089        return new OrgHostingPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
090    }
091
092    /**
093     * @param key a valid installation key
094     * @param type an optional filter to just include the given dataset type
095     */
096    public static Iterable<Dataset> hostedDatasets(UUID key, @Nullable DatasetType type, InstallationService service) {
097        LOG.info("Iterate over all {} datasets hosted organization by {}", type == null ? "" : type, key);
098        return new InstallationPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
099    }
100
101    /**
102     * @param key a valid installation key
103     * @param type an optional filter to just include the given dataset type
104     */
105    public static Iterable<Dataset> networkDatasets(UUID key, @Nullable DatasetType type, NetworkService service) {
106        LOG.info("Iterate over all {} datasets belonging to network {}", type == null ? "" : type, key);
107        return new NetworkPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
108    }
109
110    private static boolean isDataset(UUID key, DatasetService ds) {
111        return ds.get(key) != null;
112    }
113
114    private static boolean isOrganization(UUID key, OrganizationService os) {
115        return os.get(key) != null;
116    }
117
118    private static boolean isInstallation(UUID key, InstallationService is) {
119        return is.get(key) != null;
120    }
121
122    private static boolean isNetwork(UUID key, NetworkService ns) {
123        return ns.get(key) != null;
124    }
125}
