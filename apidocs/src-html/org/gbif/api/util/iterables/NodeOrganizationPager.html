001package org.gbif.api.util.iterables;
002
003import org.gbif.api.model.common.paging.PagingRequest;
004import org.gbif.api.model.common.paging.PagingResponse;
005import org.gbif.api.model.registry.Organization;
006import org.gbif.api.service.registry.NodeService;
007
008import java.util.UUID;
009
010import org.slf4j.Logger;
011import org.slf4j.LoggerFactory;
012
013/**
014 * Pages through all organizations endorsed by a given node.
015 */
016public class NodeOrganizationPager extends EntityPager<Organization> {
017    private static final Logger LOG = LoggerFactory.getLogger(NodeOrganizationPager.class);
018
019    private final NodeService service;
020    private final UUID endorsingNodeKey;
021
022    public NodeOrganizationPager(NodeService service, UUID endorsingNodeKey, int pageSize) {
023        super(pageSize);
024        this.service = service;
025        this.endorsingNodeKey = endorsingNodeKey;
026    }
027
028    @Override
029    PagingResponse<Organization> nextPage(PagingRequest page) {
030        return service.endorsedOrganizations(endorsingNodeKey, page);
031    }
032
033}
