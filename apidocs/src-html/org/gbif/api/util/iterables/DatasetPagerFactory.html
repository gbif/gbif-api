001package org.gbif.api.util.iterables;
002
003import org.gbif.api.model.common.paging.PagingConstants;
004import org.gbif.api.model.registry.Dataset;
005import org.gbif.api.service.registry.DatasetService;
006import org.gbif.api.service.registry.InstallationService;
007import org.gbif.api.service.registry.OrganizationService;
008import org.gbif.api.vocabulary.DatasetType;
009
010import java.util.UUID;
011import javax.annotation.Nullable;
012
013import com.google.common.collect.ImmutableList;
014import org.slf4j.Logger;
015import org.slf4j.LoggerFactory;
016
017/**
018 * Factory constructing dataset iterables that exclude deleted datasets.
019 */
020public class DatasetPagerFactory {
021    private static final Logger LOG = LoggerFactory.getLogger(DatasetPagerFactory.class);
022
023    /**
024     * @param key a valid dataset, organization or installation key. If null all datasets will be iterated over
025     * @throws IllegalArgumentException if given key is not existing
026     */
027    public static Iterable<Dataset> datasets(@Nullable UUID key, @Nullable DatasetType type, DatasetService ds, OrganizationService os, InstallationService is) {
028        return datasets(key, type, ds, os, is, PagingConstants.DEFAULT_PARAM_LIMIT);
029    }
030
031    /**
032     * Returns a dataset iterable by testing the given registry key first to see whether it is a dataset, organization or installation.
033     * In case of an organization key the published datasets will be returned.
034     *
035     * @param key a valid dataset, organization or installation key. If null all datasets will be iterated over
036     * @param pageSize to use when talking to the registry
037     * @throws IllegalArgumentException if given key is not existing
038     */
039    public static Iterable<Dataset> datasets(@Nullable UUID key, @Nullable DatasetType type, DatasetService ds, OrganizationService os, InstallationService is, int pageSize) {
040        if (key == null) {
041            LOG.info("Iterate over all {} datasets", type == null ? "" : type);
042            return new DatasetPager(ds, type, pageSize);
043
044        } else if (isDataset(key, ds)) {
045            LOG.info("Iterate over dataset {}", key);
046            return ImmutableList.of(ds.get(key));
047
048        } else if (isOrganization(key, os)) {
049            LOG.info("Iterate over all {} datasets published by {}", type == null ? "" : type, key);
050            return new OrgPublishingPager(os, key, type, pageSize);
051
052        } else if (isInstallation(key, is)) {
053            LOG.info("Iterate over all {} datasets hosted by installation {}", type == null ? "" : type, key);
054            return new InstallationPager(is, key, type, pageSize);
055        }
056        throw new IllegalArgumentException("Given key is no valid GBIF registry key: " + key);
057    }
058
059    /**
060     * @param type an optional filter to just include the given dataset type
061     */
062    public static Iterable<Dataset> allDatasets(@Nullable DatasetType type, DatasetService service) {
063        LOG.info("Iterate over all {} datasets", type == null ? "" : type);
064        return new DatasetPager(service, type, PagingConstants.DEFAULT_PARAM_LIMIT);
065    }
066
067    /**
068     * @param key a valid organization key
069     * @param type an optional filter to just include the given dataset type
070     */
071    public static Iterable<Dataset> publishedDatasets(UUID key, @Nullable DatasetType type, OrganizationService service) {
072        LOG.info("Iterate over all {} datasets published by {}", type == null ? "" : type, key);
073        return new OrgPublishingPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
074    }
075
076    /**
077     * @param key a valid organization key
078     * @param type an optional filter to just include the given dataset type
079     */
080    public static Iterable<Dataset> hostedDatasets(UUID key, @Nullable DatasetType type, OrganizationService service) {
081        LOG.info("Iterate over all {} datasets hosted organization by {}", type == null ? "" : type, key);
082        return new OrgHostingPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
083    }
084
085    /**
086     * @param key a valid installation key
087     * @param type an optional filter to just include the given dataset type
088     */
089    public static Iterable<Dataset> hostedDatasets(UUID key, @Nullable DatasetType type, InstallationService service) {
090        LOG.info("Iterate over all {} datasets hosted organization by {}", type == null ? "" : type, key);
091        return new InstallationPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
092    }
093
094    private static boolean isDataset(UUID key, DatasetService ds) {
095        return ds.get(key) != null;
096    }
097
098    private static boolean isOrganization(UUID key, OrganizationService os) {
099        return os.get(key) != null;
100    }
101
102    private static boolean isInstallation(UUID key, InstallationService is) {
103        return is.get(key) != null;
104    }
105}
