001package org.gbif.api.util.iterables;
002
003import org.gbif.api.model.common.paging.PagingRequest;
004import org.gbif.api.model.common.paging.PagingResponse;
005import org.gbif.api.model.registry.Dataset;
006import org.gbif.api.service.registry.OrganizationService;
007import org.gbif.api.vocabulary.DatasetType;
008
009import java.util.UUID;
010import javax.annotation.Nullable;
011
012import org.slf4j.Logger;
013import org.slf4j.LoggerFactory;
014
015/**
016 * Iterates over all datasets hosted by a given organisation.
017 */
018public class OrgHostingPager extends DatasetBasePager {
019    private static final Logger LOG = LoggerFactory.getLogger(OrgHostingPager.class);
020
021    private final OrganizationService os;
022    private final UUID orgKey;
023
024    public OrgHostingPager(OrganizationService os, UUID orgKey, @Nullable DatasetType type, int pageSize) {
025        super(type, pageSize);
026        this.os = os;
027        this.orgKey = orgKey;
028    }
029
030    @Override
031    PagingResponse<Dataset> nextPage(PagingRequest page) {
032        return os.hostedDatasets(orgKey, page);
033    }
034
035}
