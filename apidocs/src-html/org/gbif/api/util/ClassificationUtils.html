007import java.util.LinkedHashMap;

011import com.google.common.base.Joiner;

015public class ClassificationUtils {
016
017  private static final Joiner JOINER = Joiner.on(", ").skipNulls();
018
019  /**
020   * Concatenates all higher Linnean taxa into a single dwc:higherClassification string, skipping
021   * null values.
022   *
023   * @param lc the LinneanClassification to join into the higher classification string
024   *
025   * @return the concatenated dwc:higherClassification string
026   */
027  @Nullable
028  public static String getHigherClassification(LinneanClassification lc) {
029    return Strings.emptyToNull(JOINER.join(lc.getKingdom(),
030                                           lc.getPhylum(),
031                                           lc.getClazz(),
032                                           lc.getOrder(),
033                                           lc.getFamily(),
034                                           lc.getGenus()));
035  }
036
037  /**
038   * Gets a higher taxon property by passing the rank of it.
039   * Only Linnean ranks are supported.
040   *
041   * @param lc   the LinneanClassification holding the taxon property to be retrieved
042   * @param rank the higher linnean rank to retrieve
043   *
044   * @return the name of the higher taxon or null if rank doesnt exist
045   */
046  @Nullable
047  public static String getHigherRank(LinneanClassification lc, Rank rank) {
048    switch (rank) {
049      case KINGDOM:
050        return lc.getKingdom();
051      case PHYLUM:
052        return lc.getPhylum();
053      case CLASS:
054        return lc.getClazz();
055      case ORDER:
056        return lc.getOrder();
057      case FAMILY:
058        return lc.getFamily();
059      case GENUS:
060        return lc.getGenus();
061      case SUBGENUS:
062        return lc.getSubgenus();
063      case SPECIES:
064        return lc.getSpecies();
065      default:
066        return null;
067    }
068  }
069
070  /**
071   * Sets a higher taxon property by passing the rank of it.
072   *
073   * @param lc   the LinneanClassification on which to set the new property
074   * @param rank the higher linnean rank to set
075   * @param name the higher ranks name
076   */
077  public static void setHigherRank(LinneanClassification lc, Rank rank, String name) {
078    switch (rank) {
079      case KINGDOM:
080        lc.setKingdom(name);
081        break;
082      case PHYLUM:
083        lc.setPhylum(name);
084        break;
085      case CLASS:
086        lc.setClazz(name);
087        break;
088      case ORDER:
089        lc.setOrder(name);
090        break;
091      case FAMILY:
092        lc.setFamily(name);
093        break;
094      case GENUS:
095        lc.setGenus(name);
096        break;
097      case SUBGENUS:
098        lc.setSubgenus(name);
099        break;
100      case SPECIES:
101        lc.setSpecies(name);
102        break;
103    }
104  }
105
106  /**
107   * An ordered map with entries for all higher Linnean ranks down to species which are not null.
108   * The map starts with the highest rank, e.g. the kingdom and maps the name usage key to its canonical name.
109   *
110   * @param lc the object that implements both LinneanClassification and LinneanClassificationKeys from which to build
111   *           the map
112   *
113   * @return map of higher ranks
114   */
115  @NotNull
116  public static <T extends LinneanClassification & LinneanClassificationKeys> LinkedHashMap<Integer, String> getHigherClassificationMap(
117    T lc) {
118    return getHigherClassificationBaseMap(lc);
119  }
120
121  /**
122   * An ordered map with entries for all higher Linnean ranks down to the actual direct parent of this usage.
123   * The map starts with the highest rank, e.g. the kingdom and maps the name usage key to its canonical name.
124   * The name usage itself is never included, even though a higher rank might point to the usage itself.
125   *
126   * @param lc the object that implements both LinneanClassification and LinneanClassificationKeys from which to build
127   *           the map
128   *
129   * @return map of higher ranks
130   */
131  @NotNull
132  public static <T extends LinneanClassification & LinneanClassificationKeys> LinkedHashMap<Integer, String> getHigherClassificationMap(
133    T lc, int key, Integer parentKey, String parent) {
134    LinkedHashMap<Integer, String> map = getHigherClassificationBaseMap(lc);
135
136    if (parentKey != null) {
137      map.put(parentKey, parent);
138    }
139    // remove notion to this usage
140    map.remove(key);
141
142    return map;
143  }
144
145  private static <T extends LinneanClassification & LinneanClassificationKeys> LinkedHashMap<Integer, String> getHigherClassificationBaseMap(
146    T lc) {
147    LinkedHashMap<Integer, String> map = Maps.newLinkedHashMap();
148    if (lc.getKingdomKey() != null) {
149      map.put(lc.getKingdomKey(), lc.getKingdom());
150    }
151    if (lc.getPhylumKey() != null) {
152      map.put(lc.getPhylumKey(), lc.getPhylum());
153    }
154    if (lc.getClassKey() != null) {
155      map.put(lc.getClassKey(), lc.getClazz());
156    }
157    if (lc.getOrderKey() != null) {
158      map.put(lc.getOrderKey(), lc.getOrder());
159    }
160    if (lc.getFamilyKey() != null) {
161      map.put(lc.getFamilyKey(), lc.getFamily());
162    }
163    if (lc.getGenusKey() != null) {
164      map.put(lc.getGenusKey(), lc.getGenus());
165    }
166    if (lc.getSpeciesKey() != null) {
167      map.put(lc.getSpeciesKey(), lc.getSpecies());
168    }
169    return map;
170  }

172  /**
173   * Gets a higher taxon key by passing the rank of it.
174   * Only Linnean ranks are supported.
175   *
176   * @param lck  the classification that holds the taxon key
177   * @param rank the higher linnean rank to retrieve
178   *
179   * @return the key of the higher taxon or null if rank doesnt exist
180   */
181  @Nullable
182  public static Integer getHigherRankKey(LinneanClassificationKeys lck, Rank rank) {
183    switch (rank) {
184      case KINGDOM:
185        return lck.getKingdomKey();
186      case PHYLUM:
187        return lck.getPhylumKey();
188      case CLASS:
189        return lck.getClassKey();
190      case ORDER:
191        return lck.getOrderKey();
192      case FAMILY:
193        return lck.getFamilyKey();
194      case GENUS:
195        return lck.getGenusKey();
196      case SUBGENUS:
197        return lck.getSubgenusKey();
198      case SPECIES:
199        return lck.getSpeciesKey();
200      default:
201        return null;
202    }
203  }

205  /**
206   * Sets a higher taxon property by passing the rank of it.
207   *
208   * @param lck      the classification on which to set the taxon property
209   * @param rank     the higher rank to set
210   * @param usageKey key of the higher ranks usage
211   */
212  public static void setHigherRankKey(LinneanClassificationKeys lck, Rank rank, Integer usageKey) {
213    switch (rank) {
214      case KINGDOM:
215        lck.setKingdomKey(usageKey);
216        break;
217      case PHYLUM:
218        lck.setPhylumKey(usageKey);
219        break;
220      case CLASS:
221        lck.setClassKey(usageKey);
222        break;
223      case ORDER:
224        lck.setOrderKey(usageKey);
225        break;
226      case FAMILY:
227        lck.setFamilyKey(usageKey);
228        break;
229      case GENUS:
230        lck.setGenusKey(usageKey);
231        break;
232      case SUBGENUS:
233        lck.setSubgenusKey(usageKey);
234        break;
235      case SPECIES:
236        lck.setSpeciesKey(usageKey);
237        break;
238    }
239  }
240
241  /**
242   * Sets a higher taxon property by passing the rank of it.
243   *
244   * @param lc       the object that implements both LinneanClassification and LinneanClassificationKeys on which to set
245   *                 the taxon property
246   * @param rank     the higher linnean rank to set
247   * @param name     the higher ranks name
248   * @param usageKey key of the higher ranks usage
249   */
250  public static <T extends LinneanClassification & LinneanClassificationKeys> void setHigherRank(
251    T lc, Rank rank, String name, Integer usageKey
252  ) {
253    setHigherRank(lc, rank, name);
254    setHigherRankKey(lc, rank, usageKey);
255  }
256
257  /**
258   * @return true if at least one higher rank of the classification is given, i.e. not null or empty
259   */
260  public static boolean hasContent(LinneanClassification lc) {
261    for (Rank qr : Rank.DWC_RANKS) {
262      if (!Strings.isNullOrEmpty(getHigherRank(lc, qr))) {
263        return true;
264      }
265    }
266    return false;
267  }
268
269  /**
270   * Copies all linnean classification based higher taxon keys from one instance to another.
271   */
272  public static void copyLinneanClassificationKeys(LinneanClassificationKeys source, LinneanClassificationKeys target) {
273    for (Rank r : Rank.DWC_RANKS) {
274      setHigherRankKey(target, r, source.getHigherRankKey(r));
275    }
276  }
277
278  /**
279   * Copies all linnean classification based higher taxon names from one instance to another.
280   */
281  public static void copyLinneanClassification(LinneanClassification source, LinneanClassification target) {
282    for (Rank r : Rank.DWC_RANKS) {
283      setHigherRank(target, r, source.getHigherRank(r));
284    }
285  }
