001package org.gbif.api.util.comparators;
002
003import java.io.Serializable;
004import java.util.Comparator;
005import java.util.List;
006
007import org.gbif.api.model.registry.Endpoint;
008import org.gbif.api.vocabulary.EndpointType;
009
010import com.google.common.collect.ImmutableList;
011import com.google.common.collect.Ordering;
012
013/**
014 * Compares two Endpoints.
015 * <p></p>
016 * It does so by using a priority list of Endpoint Types.
017 * This Comparator will throw a {@link ClassCastException} exception if a non supported EndpointType is passed in. It
018 * also does not support {@code null} values.
019 * <p></p>
020 * The priority list is as follows (most to least important):
021 * <ol>
022 * <li>DwC-A</li>
023 * <li>TAPIR</li>
024 * <li>BioCASe</li>
025 * <li>DiGIR</li>
026 * <li>DiGIR (Manis)</li>
027 * <li>EML</li>
028 * </ol>
029 */
030public class EndpointPriorityComparator implements Comparator<Endpoint>, Serializable {
031
032  /* Note: Due to legacy reasons this is a Comparator itself instead of just using the PRIORITY_COMPARATOR.
033   It should be easy to remove this class or change it into a factory */
034
035  // Priorities from lowest to highest
036  public static final List<EndpointType> PRIORITIES = ImmutableList.of(
037      EndpointType.EML,
038      EndpointType.DIGIR_MANIS,
039      EndpointType.DIGIR,
040      EndpointType.BIOCASE,
041      EndpointType.TAPIR,
042      EndpointType.BIOCASE_XML_ARCHIVE,
043      EndpointType.DWC_ARCHIVE
044  );
045
046  private static final long serialVersionUID = 8085216142750609841L;
047  private static final Ordering<EndpointType> PRIORITY_COMPARATOR = Ordering.explicit(PRIORITIES);
048
049  @Override
050  public int compare(Endpoint endpoint1, Endpoint endpoint2) {
051    return PRIORITY_COMPARATOR.compare(endpoint1.getType(), endpoint2.getType());
052  }
053
054}
