001package org.gbif.api.util;
002
003import org.gbif.api.model.registry.LenientEquals;
004
005public class LenientEqualsUtils {
006
007  private LenientEqualsUtils() {}
008
009  public static <T extends LenientEquals<T>> boolean lenientEquals(T a, T b) {
010    return (a == b) || (a != null && a.lenientEquals(b));
011  }
012}
