013package org.gbif.api.util;
014
015import org.gbif.api.vocabulary.ContactType;
016import org.gbif.api.vocabulary.EndpointType;
017import org.gbif.api.vocabulary.IdentifierType;
018import org.gbif.api.vocabulary.TechnicalInstallationType;
019
020import com.google.common.base.Strings;
021
022public final class VocabularyUtils {
023
024  public static ContactType parseContactType(String type) {
025    return (ContactType) lookupEnum(type, ContactType.class);
026  }
027
028  public static EndpointType parseEndpointType(String type) {
029    return (EndpointType) lookupEnum(type, EndpointType.class);
030  }
031
032  public static IdentifierType parseIdentifierType(String type) {
033    return (IdentifierType) lookupEnum(type, IdentifierType.class);
034  }
035
036  public static TechnicalInstallationType parseTechnicalInstallationType(String type) {
037    return (TechnicalInstallationType) lookupEnum(type, TechnicalInstallationType.class);
038  }
039
040  /**
041   * Generic method to lookup an enumeration value for a given string based on the name of the enum member.
042   * The lookup is case insensitive and ignore whitespaces, underscores and dashes.
043   *
044   * @param name the enum members name to lookup
045   * @param vocab the enumeration class
046   * @return the matching enum member or null if {@code name} is null or empty (see http://dev.gbif.org/issues/browse/POR-2858)
047   * @throws IllegalArgumentException if the name cannot be parsed into a known name
048   */
049  public static <T extends Enum<?>> T lookupEnum(String name, Class<T> vocab) {
050    if (Strings.isNullOrEmpty(name)) {
051      return null;
052    }
053    final String normedType = name.toUpperCase().replaceAll("[. _-]", "");
054    T[] values = vocab.getEnumConstants();
055    if (values != null) {
056      for (T val : values) {
057        final String normedVal = val.name().toUpperCase().replaceAll("[. _-]", "");
058        if (normedType.equals(normedVal)) {
059          return val;
060        }
061      }
062    }
063    throw new IllegalArgumentException("Cannot parse " + name + " into a known " + vocab.getSimpleName());
064  }
065
066  /**
067   * Looks up an enumeration by class name. One can get the classname using the likes of:
068   * <p/>
069   *
070   * <pre>
071   * {@code
072   * Country.class.getName()
073   * }
074   * </pre>
075   *
076   * @param fullyQualifiedClassName Which should name the enumeration (e.g. org.gbif.api.vocabulary.Country)
077   * @return The enumeration or null if {@code fullyQualifiedClassName} is null or empty (see http://dev.gbif.org/issues/browse/POR-2858)
078   * @throws IllegalArgumentException if {@code fullyQualifiedClassName} class cannot be located
079   */
080  @SuppressWarnings("unchecked")
081  public static Class<? extends Enum<?>> lookupVocabulary(String fullyQualifiedClassName) {
082    if (!Strings.isNullOrEmpty(fullyQualifiedClassName)) {
083      try {
084        Class<?> cl = Class.forName(fullyQualifiedClassName);
085        if (Enum.class.isAssignableFrom(cl)) {
086          return (Class<? extends Enum<?>>) cl;
087        }
088      } catch (Exception e) {
089        throw new IllegalArgumentException("Unable to lookup the vocabulary: " + fullyQualifiedClassName, e);
090      }
091    }
092    return null;
093  }
094
095  /**
096   * A static utils class.
097   */
098  private VocabularyUtils() {
099  }
