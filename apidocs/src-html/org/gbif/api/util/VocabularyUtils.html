050  /**
051   * Generic method to lookup an enumeration value for a given string based on the name of the enum member.
052   * The lookup is case insensitive and ignore whitespaces, underscores and dashes.
053   *
054   * @param name the enum members name to lookup
055   * @param vocab the enumeration class
056   * @return the matching enum member or null if {@code name} is null or empty (see http://dev.gbif.org/issues/browse/POR-2858)
057   * @throws IllegalArgumentException if the name cannot be parsed into a known name
058   */
059  public static <T extends Enum<?>> T lookupEnum(String name, Class<T> vocab) {
060    if (Strings.isNullOrEmpty(name)) {
061      return null;
062    }
063    final String normedType = name.toUpperCase().replaceAll("[. _-]", "");
064    T[] values = vocab.getEnumConstants();
065    if (values != null) {
066      for (T val : values) {
067        final String normedVal = val.name().toUpperCase().replaceAll("[. _-]", "");
068        if (normedType.equals(normedVal)) {
069          return val;
070        }
071      }
072    }
073    throw new IllegalArgumentException("Cannot parse " + name + " into a known " + vocab.getSimpleName());
074  }
075
076  /**
077   * Same as {@link #lookupEnum(String, Class)} } without IllegalArgumentException.
078   * On failure, this method will return Optional.absent().
079   *
080   * @param name
081   * @param vocab
082   * @param <T>
083   * @return instance of com.google.common.base.Optional, never null.
084   */
085  public static <T extends Enum<?>> Optional<T> lookup(String name, Class<T> vocab) {
086    T result = null;
087    // this try/catch in needed until we replace all calls to lookupEnum() in favor of this method
088    try{
089      result = lookupEnum(name, vocab);
090    }
091    catch (IllegalArgumentException iaEx){/*ignore*/}
092    return Optional.fromNullable(result);
093  }
094
095  /**
096   * Looks up an enumeration by class name. One can get the classname using the likes of:
097   * <p/>
098   *
099   * <pre>
100   * {@code
101   * Country.class.getName()
102   * }
103   * </pre>
104   *
105   * @param fullyQualifiedClassName Which should name the enumeration (e.g. org.gbif.api.vocabulary.Country)
106   * @return The enumeration or null if {@code fullyQualifiedClassName} is null or empty (see http://dev.gbif.org/issues/browse/POR-2858)
107   * @throws IllegalArgumentException if {@code fullyQualifiedClassName} class cannot be located
108   */
109  @SuppressWarnings("unchecked")
110  public static Class<? extends Enum<?>> lookupVocabulary(String fullyQualifiedClassName) {
111    if (!Strings.isNullOrEmpty(fullyQualifiedClassName)) {
112      try {
113        Class<?> cl = Class.forName(fullyQualifiedClassName);
114        if (Enum.class.isAssignableFrom(cl)) {
115          return (Class<? extends Enum<?>>) cl;
116        }
117      } catch (Exception e) {
118        throw new IllegalArgumentException("Unable to lookup the vocabulary: " + fullyQualifiedClassName, e);
119      }
120    }
121    return null;
122  }
123
124  /**
125   * Utility method to get a map of all enumerations within a package.
126   * The map will use the enumeration class simple name as key and the enum itself as value.
127   *
128   * @return a map of all enumeration within the package or an empty map in all other cases.
129   */
130  public static Map<String, Enum<?>[]> listEnumerations(String packageName) {
131    try {
132      ClassPath cp = ClassPath.from(VocabularyUtils.class.getClassLoader());
133      ImmutableMap.Builder<String, Enum<?>[]> builder = ImmutableMap.builder();
134
135      List<ClassPath.ClassInfo> infos = cp.getTopLevelClasses(packageName).asList();
136      for (ClassPath.ClassInfo info : infos) {
137        Class<? extends Enum<?>> vocab = lookupVocabulary(info.getName());
138        // verify that it is an Enumeration
139        if (vocab != null && vocab.getEnumConstants() != null) {
140          builder.put(info.getSimpleName(), vocab.getEnumConstants());
141        }
142      }
143      return builder.build();
144    } catch (Exception e) {
145      LOG.error("Unable to read the classpath for enumerations", e);
146      return ImmutableMap.<String, Enum<?>[]>of(); // empty
147    }
148  }
