016package org.gbif.api.util;
017
018import java.lang.annotation.ElementType;
019import java.lang.annotation.Retention;
020import java.lang.annotation.RetentionPolicy;
021import java.lang.annotation.Target;
022
023import javax.validation.Constraint;
024import javax.validation.Payload;
025
026/**
027 * Validates that the URI field is absolute, beginning with either http or https.
028 * To use, apply to the URI fields intended for validation.
029 * The field must be:
030 * <ul>
031 * <li>A URI object</li>
032 * </ul>
033 */
034@Retention(RetentionPolicy.RUNTIME)
035@Target({ElementType.METHOD, ElementType.FIELD, ElementType.TYPE_USE})
036@Constraint(validatedBy = URIValidator.class)
037public @interface HttpURI {
038
039  Class<?>[] groups() default {};
040
041  Class<? extends Payload>[] payload() default {};
042
043  String message() default "Invalid URI";
044}
