016package org.gbif.api.util;
017
018import java.text.DateFormat;
019import java.text.ParseException;
020import java.text.ParsePosition;
021import java.text.SimpleDateFormat;
022import java.util.Calendar;
023import java.util.Date;
024import java.util.regex.Pattern;
025
026import org.apache.commons.lang3.StringUtils;
027
028import static org.gbif.api.model.common.search.SearchConstants.QUERY_WILDCARD;
029
030/**
031 * Utility class that parses date values, the allowed date formats are: "yyyy-MM-dd", "yyyy-MM" or "yyyy".
032 */
033public class IsoDateParsingUtils {
034
035  public static final String SIMPLE_ISO_DATE_STR_PATTERN = "\\d{4}(?:-\\d{2}(?:-\\d{2})?)?";
036
037  // match formats 'yyyy', 'yyyy-MM' and 'yyyy-MM-dd'
038  public static final Pattern SIMPLE_ISO_PATTERN = Pattern.compile(SIMPLE_ISO_DATE_STR_PATTERN);
039
040  /**
041   * Enumerations with the allowed date formats by the occurrence search service.
042   */
043  public enum IsoDateFormat {
044    FULL("yyyy-MM-dd"),
045    YEAR_MONTH("yyyy-MM"),
046    YEAR("yyyy");
047
048    private final String datePattern;
049
050    /**
051     * Private constructors.
052     * Receives a pattern and creates DateFormat instance with the pattern parameter.
053     */
054    IsoDateFormat(String datePattern) {
055      this.datePattern = datePattern;
056    }
057
058    /**
059     * @return the dateFormat that parses and formats date values.
060     */
061    public DateFormat getDateFormat() {
062      DateFormat dateFormat = new SimpleDateFormat(datePattern);
063      dateFormat.setLenient(false);
064      return dateFormat;
065    }
066
067    /**
068     * Checks if the parameter "value" can be parsed using the date format.
069     */
070    public boolean isValidDate(String value) {
071      try {
072        return (parseDate(value) != null);
073      } catch (ParseException e) {
074        return false;
075      }
076    }
077
078    /**
079     * Try to parse a string with the current date format.
080     */
081    public Date parseDate(String value) throws ParseException {
082      ParsePosition position = new ParsePosition(0);
083      Date date = getDateFormat().parse(value, position);
084      if (position.getIndex() != value.length()) {
085        throw new ParseException(value + " is not a valid date", position.getIndex());
086      }
087      return date;
088    }
089  }
090
091  /**
092   * Default private constructor.
093   */
094  private IsoDateParsingUtils() {
095    // empty block
096  }
097
098  /**
099   * Iterates over all the OccurrenceDateFormat's and returns the first one that parses successfully the value.
100   * @throws IllegalArgumentException in case of unparsable dates
101   */
102  public static IsoDateFormat getFirstDateFormatMatch(String value) throws IllegalArgumentException {
103    // at least 4 digits for a year must exist
104    if (SIMPLE_ISO_PATTERN.matcher(value).find()) {
105      for (IsoDateFormat dateFormat : IsoDateFormat.values()) {
106        if (dateFormat.isValidDate(value)) {
107          return dateFormat;
108        }
109      }
110    }
111    throw new IllegalArgumentException(value + " is not a valid date");
112  }
113
114  /**
115   * Parses a single date value. The date is interpreted using the first available date format that successfully parses
116   * the value.
117   * @return the lowest possible date according to the precision given or null in case of *
118   * @throws IllegalArgumentException in case of unparsable dates
119   */
120  public static Date parseDate(String value) {
121    if (StringUtils.isEmpty(value)) {
122      throw new IllegalArgumentException("Date parameter can't be null or empty");
123    }
124
125    // could be a wildcard
126    if (QUERY_WILDCARD.equals(value)) {
127      return null;
128    }
129
130    try {
131      return getFirstDateFormatMatch(value).parseDate(value);
132    } catch (ParseException e) {
133      throw new IllegalArgumentException(String.format("%s is not a valid date parameter", value));
134    }
135  }
136
137  /**
138   * Parses a range of dates. The date format used is the first date format that successfully parses the lower range
139   * limit.
140   */
141  public static Range<Date> parseDateRange(String value) {
142    if (StringUtils.isEmpty(value)) {
143      throw new IllegalArgumentException("Date parameter can't be null or empty");
144    }
145    final String[] dateValues = value.split(",");
146    if (dateValues.length != 2) {
147      throw new IllegalArgumentException("Date value must be a single value or a range");
148    }
149
150    final Date lowerDate = parseDate(dateValues[0]);
151
152    Date upperDate = parseDate(dateValues[1]);
153    // in case we have a real upper date check its precision and use the highest possible date, not lowest
154    if (upperDate != null) {
155      final IsoDateFormat upperDateFormat = getFirstDateFormatMatch(dateValues[1]);
156      if (upperDateFormat == IsoDateFormat.YEAR_MONTH) {
157        upperDate = toLastDayOfMonth(upperDate);
158
159      } else if (upperDateFormat == IsoDateFormat.YEAR) {
160        upperDate = toLastDayOfYear(upperDate);
161      }
162    }
163
164    return Range.closed(lowerDate, upperDate);
165  }
166
167  /**
168   * Calculates the last day of the month for the date parameter and return it a new date instance.
169   */
170  public static Date toLastDayOfMonth(Date value) {
171    return toLastDayOf(value, Calendar.DAY_OF_MONTH);
172  }
173
174  /**
175   * Calculates the last day of the year for the date parameter and return it a new date instance.
176   */
177  public static Date toLastDayOfYear(Date value) {
178    return toLastDayOf(value, Calendar.DAY_OF_YEAR);
179  }
180
181  /**
182   * Calculates the last day of a year or month for the date parameter and return it a new date instance.
183   */
184  public static Date toLastDayOf(Date value, IsoDateFormat isoDateFormat) {
185    if (IsoDateFormat.YEAR_MONTH == isoDateFormat) {
186      return toLastDayOf(value, Calendar.DAY_OF_MONTH);
187    } else if(IsoDateFormat.YEAR == isoDateFormat){
188      return toLastDayOf(value, Calendar.DAY_OF_YEAR);
189    }
190    return value;
191  }
192
193  /**
194   * Gets the actual maximum value for a field of a date value.
195   */
196  private static Date toLastDayOf(Date value, int field) {
197    Calendar calendar = Calendar.getInstance();
198    calendar.setTime(value);
199    calendar.set(field, calendar.getActualMaximum(field));
200    return calendar.getTime();
201  }
202}
