016package org.gbif.api.util;
017
018import java.text.ParseException;
019import java.time.DateTimeException;
020import java.time.LocalDate;
021import java.time.Year;
022import java.time.YearMonth;
023import java.time.ZoneOffset;
024import java.time.chrono.IsoChronology;
025import java.time.format.DateTimeFormatter;
026import java.time.format.DateTimeFormatterBuilder;
027import java.time.format.DateTimeParseException;
028import java.time.format.ResolverStyle;
029import java.time.format.SignStyle;
030import java.time.temporal.ChronoField;
031import java.time.temporal.TemporalAccessor;
032import java.util.regex.Pattern;
033
034import org.apache.commons.lang3.StringUtils;
035
036import static org.gbif.api.model.common.search.SearchConstants.QUERY_WILDCARD;
037
038/**
039 * Utility class that parses date values, the allowed date formats are: "yyyy-MM-dd", "yyyy-MM" or "yyyy".
040 *
041 * Note: Does not handle times or time zones.  This is used for search queries/predicates with date ranges.
042 * Date ranges are closed at the lower bound, and open at the upper bound, to allow less-than predicates
043 * to match times on the upper bound date (i.e. after midnight).
044 */
045public class IsoDateParsingUtils {
046
047  public static final String SIMPLE_ISO_DATE_STR_PATTERN = "\\d{4}(?:-\\d{1,2}(?:-\\d{1,2})?)?";
048
049  // match formats 'yyyy', 'yyyy-MM', 'yyyy-M', 'yyyy-MM-dd', 'yyyy-MM-d' and 'yyyy-M-d'.
050  public static final Pattern SIMPLE_ISO_PATTERN = Pattern.compile(SIMPLE_ISO_DATE_STR_PATTERN);
051
052  // Format as yyyy-MM-dd.
053  public static final DateTimeFormatter ISO_DATE_FORMATTER = DateTimeFormatter.ISO_LOCAL_DATE.withZone(ZoneOffset.UTC);
054
055  /**
056   * Enumerations with the allowed date formats by the occurrence search service.
057   */
058  public enum IsoDateFormat {
059    FULL(new DateTimeFormatterBuilder()
060      .appendValue(ChronoField.YEAR, 4, 10, SignStyle.EXCEEDS_PAD)
061      .appendLiteral('-')
062      .appendValue(ChronoField.MONTH_OF_YEAR, 1, 2, SignStyle.NEVER)
063      .appendLiteral('-')
064      .appendValue(ChronoField.DAY_OF_MONTH, 1, 2, SignStyle.NEVER)
065      .toFormatter()),
066    YEAR_MONTH(new DateTimeFormatterBuilder()
067      .appendValue(ChronoField.YEAR, 4, 10, SignStyle.EXCEEDS_PAD)
068      .appendLiteral('-')
069      .appendValue(ChronoField.MONTH_OF_YEAR, 1, 2, SignStyle.NEVER)
070      .toFormatter()),
071    YEAR(new DateTimeFormatterBuilder()
072      .appendValue(ChronoField.YEAR, 4, 10, SignStyle.EXCEEDS_PAD)
073      .toFormatter());
074
075    private final DateTimeFormatter dateFormatter;
076
077    /**
078     * Private constructors.
079     * Receives a pattern and creates a strict DateTimeFormatter.
080     */
081    IsoDateFormat(DateTimeFormatter dateFormatter) {
082      this.dateFormatter = dateFormatter
083        .withZone(ZoneOffset.UTC)
084        .withResolverStyle(ResolverStyle.STRICT)
085        .withChronology(IsoChronology.INSTANCE);
086    }
087
088    /**
089     * Checks if the parameter "value" can be parsed using the date format.
090     */
091    public boolean isValidDate(String value) {
092      try {
093        return (parseDate(value) != null);
094      } catch (ParseException e) {
095        return false;
096      }
097    }
098
099    /**
100     * Try to parse a string with the current date format.
101     */
102    public TemporalAccessor parseDate(String value) throws ParseException {
103      if (QUERY_WILDCARD.equals(value)) {
104        return null;
105      }
106
107      try {
108        switch (this) {
109          case FULL:
110            return dateFormatter.parse(value, LocalDate::from);
111
112          case YEAR_MONTH:
113            return dateFormatter.parse(value, YearMonth::from);
114
115          case YEAR:
116            return dateFormatter.parse(value, Year::from);
117        }
118      } catch (DateTimeException e) {
119        throw new ParseException(value + " is not a valid date", 0);
120      }
121      throw new ParseException(value + " is not a valid date", 0);
122    }
123
124    /**
125     * Returns the earliest date of a possible closed range, e.g. 2000-01-01 for 2000.
126     */
127    public LocalDate earliestDate(String value) throws ParseException {
128      if (QUERY_WILDCARD.equals(value)) {
129        return null;
130      }
131
132      TemporalAccessor ta = parseDate(value);
133      switch (this) {
134        case FULL:
135          return LocalDate.from(ta);
136
137        case YEAR_MONTH:
138          YearMonth yearMonth = YearMonth.from(parseDate(value));
139          return yearMonth.atDay(1);
140
141        case YEAR:
142          Year year = Year.from(parseDate(value));
143          return year.atDay(1);
144      }
145
146      throw new ParseException(value + " is not a valid date", 0);
147    }
148
149    /**
150     * Returns the latest date of a possible open range, e.g. 2001-01-01 for 2000.
151     */
152    public LocalDate latestDate(String value) throws ParseException {
153      if (QUERY_WILDCARD.equals(value)) {
154        return null;
155      }
156
157      TemporalAccessor ta = parseDate(value);
158      switch (this) {
159        case FULL:
160          return LocalDate.from(ta).plusDays(1);
161
162        case YEAR_MONTH:
163          YearMonth yearMonth = YearMonth.from(parseDate(value));
164          return yearMonth.atEndOfMonth().plusDays(1);
165
166        case YEAR:
167          Year year = Year.from(parseDate(value));
168          return year.atDay(year.isLeap() ? 366 : 365).plusDays(1);
169      }
170
171      throw new ParseException(value + " is not a valid date", 0);
172    }
173  }
174
175  /**
176   * Default private constructor.
177   */
178  private IsoDateParsingUtils() {
179    // empty block
180  }
181
182  /**
183   * Iterates over all the OccurrenceDateFormat's and returns the first one that parses successfully the value.
184   * @throws IllegalArgumentException in case of unparsable dates
185   */
186  public static IsoDateFormat getFirstDateFormatMatch(String value) throws IllegalArgumentException {
187    // 4 digits for a year must exist
188    if (SIMPLE_ISO_PATTERN.matcher(value).find()) {
189      for (IsoDateFormat dateFormat : IsoDateFormat.values()) {
190        if (dateFormat.isValidDate(value)) {
191          return dateFormat;
192        }
193      }
194    }
195    throw new IllegalArgumentException(value + " is not a valid date");
196  }
197
198  /**
199   * Parses a single date value. The date is interpreted using the first available date format that successfully parses
200   * the value.
201   * @return the lowest possible date according to the precision given or null in case of *
202   * @throws IllegalArgumentException in case of unparsable dates
203   */
204  public static LocalDate parseDate(String value) {
205    if (StringUtils.isEmpty(value)) {
206      throw new IllegalArgumentException("Date parameter can't be null or empty");
207    }
208
209    // could be a wildcard
210    if (QUERY_WILDCARD.equals(value)) {
211      return null;
212    }
213
214    try {
215      return getFirstDateFormatMatch(value).earliestDate(value);
216    } catch (DateTimeParseException | ParseException e) {
217      throw new IllegalArgumentException(String.format("%s is not a valid date parameter", value));
218    }
219  }
220
221  /**
222   * Parses a closed-open range of dates, using the most specific format for each side of the range.
223   * Returns null for an unbounded side of the range (*).
224   */
225  public static Range<LocalDate> parseDateRange(String value) {
226    if (StringUtils.isEmpty(value)) {
227      throw new IllegalArgumentException("Date parameter can't be null or empty");
228    }
229    final String[] dateValues = value.split(",");
230
231    if (dateValues.length == 1) {
232      try {
233        final LocalDate lowerDate = parseDate(dateValues[0]);
234        final LocalDate upperDate = getFirstDateFormatMatch(dateValues[0]).latestDate(dateValues[0]);
235
236        return Range.closed(lowerDate, upperDate);
237      } catch (ParseException e) {
238        throw new IllegalArgumentException(String.format("%s is not a valid date parameter", value));
239      }
240    }
241
242    if (dateValues.length == 2) {
243      try {
244        final LocalDate lowerDate = parseDate(dateValues[0]);
245        LocalDate upperDate = parseDate(dateValues[1]);
246
247        if (upperDate != null) {
248          upperDate = getFirstDateFormatMatch(dateValues[1]).latestDate(dateValues[1]);
249        }
250
251        return Range.closed(lowerDate, upperDate);
252      } catch (ParseException e) {
253        throw new IllegalArgumentException(String.format("%s is not a valid date parameter", value));
254      }
255    }
256
257    throw new IllegalArgumentException("Date value must be a single value or a range");
258  }
