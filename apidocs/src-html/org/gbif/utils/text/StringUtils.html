038  /**
039   * Removes accents & diacretics and converts ligatures into several chars
040   * @param x string to fold into ASCII
041   * @return string converted to ASCII equivalent, expanding common ligatures
042   */
043  public static String foldToAscii(String x) {
044    if (x == null) {
045      return null;
046    }
047    x = replaceSpecialCases(x);
048    // use java unicode normalizer to remove accents
049    x = Normalizer.normalize(x, Normalizer.Form.NFD);
050    return MARKER.matcher(x).replaceAll("");
051  }

053  /**
054   * The Normalizer misses a few cases and 2 char ligatures which we deal with here
055   */
056  private static String replaceSpecialCases(String x) {
057    StringBuffer sb = new StringBuffer();
058
059    for (int i = 0; i < x.length(); i++) {
060      char c = x.charAt(i);
061      switch (c) {
062        case 'ß':
063          sb.append("ss");
064          break;
065        case 'Æ':
066          sb.append("AE");
067          break;
068        case 'æ':
069          sb.append("ae");
070          break;
071        case 'Ð':
072          sb.append("D");
073          break;
074        case 'đ':
075          sb.append("d");
076          break;
077        case 'ð':
078          sb.append("d");
079          break;
080        case 'Ø':
081          sb.append("O");
082          break;
083        case 'ø':
084          sb.append("o");
085          break;
086        case 'Œ':
087          sb.append("OE");
088          break;
089        case 'œ':
090          sb.append("oe");
091          break;
092        case 'Ŧ':
093          sb.append("T");
094          break;
095        case 'ŧ':
096          sb.append("t");
097          break;
098        case 'Ł':
099          sb.append("L");
100          break;
101        case 'ł':
102          sb.append("l");
103          break;
104        default:
105          sb.append(c);
106      }
107    }
108    return sb.toString();
109  }

111  /**
112   * Increase a given string by 1, i.e. increase the last char in that string by one.
113   * If its a z or Z the char before is increased instead and a new char a is appended.
114   * Only true letters are increased, but spaces, punctuation or numbers remain unchanged.
115   * Null values stay null and empty strings empty.
116   * The case of existing characters will be kept and the appended chars will use the case of the last char of the
117   * original string.
118   *
119   * For example "Carlb" becomes "Carla", "Aua" "Atz", "zZz" "aAaa" or "Abies zzz" "Abiet aaa".
120   *
121   * @param x
122   * @return
123   */
124  public static String increase(String x) {
125    if (x == null) {
126      return null;
127    }
128    if (x.equals("")) {
129      return x;
130    }
131
132    char[] chars = x.toCharArray();
133    int idx = chars.length - 1;
134    boolean appendingNeeded = false;
135    Character lastOriginalChar = null;
136
137    while (idx >= 0){
138      char c = chars[idx];
139      if (!Character.isLetter(c)){
140        idx--;
141        continue;
142      }
143
144      if (lastOriginalChar == null){
145        lastOriginalChar = c;
146      }
147
148      if (c == 'z'){
149        chars[idx] = 'a';
150        appendingNeeded = true;
151
152      } else if (c == 'Z'){
153        chars[idx] = 'A';
154        appendingNeeded = true;
155
156      } else {
157        c++;
158        chars[idx] = c;
159        appendingNeeded = false;
160        break;
161      }
162      idx--;
163    }
164
165    // first char, also append to end
166    if (appendingNeeded){
167      char append = (lastOriginalChar==null || Character.isLowerCase(lastOriginalChar)) ? 'a' : 'A';
168      return String.valueOf(chars) + append;
169
170    } else {
171      return String.valueOf(chars);
172    }
173  }

199  /**
200   * Creates a random string in upper case of given length with purely latin characters only.
201   * Vocals are used much more frequently than consonants
202   * @param len
203   * @return a random string in upper case
204   */
205  public static String randomString(int len) {
206    StringBuilder sb = new StringBuilder(len);
207    for (int i = 0; i < len; i++) {
208      if (rnd.nextInt(3) > 1) {
209        sb.append(CONS.charAt(rnd.nextInt(CONS.length())));
210      } else {
211        sb.append(VOC.charAt(rnd.nextInt(VOC.length())));
212      }
213    }
214
215    return sb.toString();
216  }

242  /**
243   * Simple boolean parsing method that understands yes,y,true,t or 1 as true and respective values for false.
244   * It does not throw any exception but returns null instead.
245   *
246   * @param x
247   * @return the parsed integer or null
248   */
249  public static Boolean parseBoolean(String x) {
250    x = org.apache.commons.lang3.StringUtils.trimToEmpty(x).toLowerCase();
251    if (x.equals("true") || x.equals("t") || x.equals("1") || x.equals("yes") || x.equals("y")) {
252      return true;
253    }
254    if (x.equals("false") || x.equals("f") || x.equals("0") || x.equals("no") || x.equals("n")) {
255      return false;
256    }
257    return null;
258  }

260  /**
261   * Unescapes various unicode escapes if existing:
262   *
263   * java unicode escape, four hexadecimal digits
264   * \ uhhhh
265   *
266   * octal escape
267   * \nnn
268   * The octal value nnn, where nnn stands for 1 to 3 digits between ‘0’ and ‘7’. For example, the code for the ASCII
269   * ESC (escape) character is ‘\033’.
270   *
271   * hexadecimal escape
272   * \xhh...
273   * The hexadecimal value hh, where hh stands for a sequence of hexadecimal digits (‘0’–‘9’, and either ‘A’–‘F’ or
274   * ‘a’–‘f’).Like the same construct in ISO C, the escape sequence continues until the first nonhexadecimal digit is seen.
275   * However, using more than two hexadecimal digits produces undefined results. (The ‘\x’ escape sequence is not allowed
276   * in POSIX awk.)
277   *
278   * @param text string potentially containing unicode escape chars
279   * @return the unescaped string
280   */
281  public static String unescapeUnicodeChars(String text) {
282    if (text == null) {
283      return null;
284    }
285    // replace unicode, hexadecimal or octal character encodings by iterating over the chars once
286    //
287    // java unicode escape, four hexadecimal digits
288    // \ uhhhh
289    //
290    // octal escape
291    // \nnn
292    // The octal value nnn, where nnn stands for 1 to 3 digits between ‘0’ and ‘7’. For example, the code for the ASCII
293    // ESC (escape) character is ‘\033’.
294    //
295    // hexadecimal escape
296    // \xhh...
297    // The hexadecimal value hh, where hh stands for a sequence of hexadecimal digits (‘0’–‘9’, and either ‘A’–‘F’ or
298    // ‘a’–‘f’).
299    // Like the same construct in ISO C, the escape sequence continues until the first nonhexadecimal digit is seen.
300    // However, using more than two hexadecimal digits produces undefined results. (The ‘\x’ escape sequence is not allowed
301    // in POSIX awk.)
302    int i = 0, len = text.length();
303    char c;
304    StringBuffer sb = new StringBuffer(len);
305    while (i < len) {
306      c = text.charAt(i++);
307      if (c == '\\') {
308        if (i < len) {
309          c = text.charAt(i++);
310          try {
311            if (c == 'u' && text.length() >= i + 4) {
312              // make sure we have only hexadecimals
313              String hex = text.substring(i, i + 4);
314              if (HEX.matcher(hex).find()) {
315                c = (char) Integer.parseInt(hex, 16);
316                i += 4;
317              } else {
318                throw new NumberFormatException("No hex value: " + hex);
319              }
320            } else if (c == 'n' && text.length() >= i + 2) {
321              // make sure we have only 0-7 digits
322              String oct = text.substring(i, i + 2);
323              if (OCT.matcher(oct).find()) {
324                c = (char) Integer.parseInt(oct, 8);
325                i += 2;
326              } else {
327                throw new NumberFormatException("No octal value: " + oct);
328              }
329            } else if (c == 'x' && text.length() >= i + 2) {
330              // make sure we have only hexadecimals
331              String hex = text.substring(i, i + 2);
332              if (HEX.matcher(hex).find()) {
333                c = (char) Integer.parseInt(hex, 16);
334                i += 2;
335              } else {
336                throw new NumberFormatException("No hex value: " + hex);
337              }
338            } else if (c == 'r' || c == 'n' || c == 't') {
339              // escaped newline or tab. Replace with simple space
340              c = ' ';
341            } else {
342              throw new NumberFormatException("No char escape");
343            }
344          } catch (NumberFormatException e) {
345            // keep original characters including \ if escape sequence was invalid
346            // but replace \n with space instead
347            if (c == 'n') {
348              c = ' ';
349            } else {
350              c = '\\';
351              i--;
352            }
353          }
354        }
355      } // fall through: \ escapes itself, quotes any character but u
356      sb.append(c);
357    }
358    return sb.toString();
359  }

384  /**
385   * Joins a list of objects into a string, skipping null values and calling toString on each object.
386   * @param delimiter to join the values with
387   * @param values to be joined
388   * @return
389   */
390  public static String joinIfNotNull(String delimiter, Object... values) {
391    return Joiner.on(delimiter).skipNulls().join(values);
392  }

394  /**
395   * Uppercases all keys and values in a Map.
396   * If keys clash only one entry will remain, which is not guaranteed.
397   *
398   * @param map
399   * @return new map with keys and values upper cased.
400   */
401  public static Map<String, String> upper(Map<String, String> map) {
402    Map<String, String> upperMap = Maps.newHashMap();
403    for (String k : map.keySet()) {
404      String v = map.get(k);
405      if (v != null) {
406        v = v.trim().toUpperCase();
407      }
408      upperMap.put(k.toUpperCase(), v);
409    }
410    return upperMap;
411  }
