042  /**
043   * Removes accents & diacretics and converts ligatures into several chars
044   * @param x string to fold into ASCII
045   * @return string converted to ASCII equivalent, expanding common ligatures
046   */
047  public static String foldToAscii(String x) {
048    if (x == null) {
049      return null;
050    }
051    x = replaceSpecialCases(x);
052    // use java unicode normalizer to remove accents
053    x = Normalizer.normalize(x, Normalizer.Form.NFD);
054    return MARKER.matcher(x).replaceAll("");
055  }

058  /**
059   * Apply a function then join the result using a space if not null.
060   * E.g. can be used with apache.commons.lang3.StringUtils::trimToNull to compose a name when some parts are
061   * optionals.
062   *
063   * @param fct   the function to apply or Function.identity() if none
064   * @param parts
065   *
066   * @return a String that represents all parts joined by a space or empty String. Never null.
067   */
068  public static String thenJoin(Function<String, String> fct, String... parts) {
069    Objects.requireNonNull(fct, "fct shall be provided, use Function.identity() is you want to use the String as is");
070    return Arrays.stream(parts != null ? parts : new String[0])
071            .map(fct)
072            .filter(Objects::nonNull)
073            .collect(Collectors.joining(" "));
074  }

076  /**
077   * The Normalizer misses a few cases and 2 char ligatures which we deal with here
078   */
079  private static String replaceSpecialCases(String x) {
080    StringBuilder sb = new StringBuilder();
081
082    for (int i = 0; i < x.length(); i++) {
083      char c = x.charAt(i);
084      switch (c) {
085        case 'ß':
086          sb.append("ss");
087          break;
088        case 'Æ':
089          sb.append("AE");
090          break;
091        case 'æ':
092          sb.append("ae");
093          break;
094        case 'Ð':
095          sb.append("D");
096          break;
097        case 'đ':
098          sb.append("d");
099          break;
100        case 'ð':
101          sb.append("d");
102          break;
103        case 'Ø':
104          sb.append("O");
105          break;
106        case 'ø':
107          sb.append("o");
108          break;
109        case 'Œ':
110          sb.append("OE");
111          break;
112        case 'œ':
113          sb.append("oe");
114          break;
115        case 'Ŧ':
116          sb.append("T");
117          break;
118        case 'ŧ':
119          sb.append("t");
120          break;
121        case 'Ł':
122          sb.append("L");
123          break;
124        case 'ł':
125          sb.append("l");
126          break;
127        default:
128          sb.append(c);
129      }
130    }
131    return sb.toString();
132  }

134  /**
135   * Increase a given string by 1, i.e. increase the last char in that string by one.
136   * If its a z or Z the char before is increased instead and a new char a is appended.
137   * Only true letters are increased, but spaces, punctuation or numbers remain unchanged.
138   * Null values stay null and empty strings empty.
139   * The case of existing characters will be kept and the appended chars will use the case of the last char of the
140   * original string.
141   *
142   * For example "Carlb" becomes "Carla", "Aua" "Atz", "zZz" "aAaa" or "Abies zzz" "Abiet aaa".
143   *
144   * @param x
145   * @return
146   */
147  public static String increase(String x) {
148    if (x == null) {
149      return null;
150    }
151    if (x.equals("")) {
152      return x;
153    }
154
155    char[] chars = x.toCharArray();
156    int idx = chars.length - 1;
157    boolean appendingNeeded = false;
158    Character lastOriginalChar = null;
159
160    while (idx >= 0){
161      char c = chars[idx];
162      if (!Character.isLetter(c)){
163        idx--;
164        continue;
165      }
166
167      if (lastOriginalChar == null){
168        lastOriginalChar = c;
169      }
170
171      if (c == 'z'){
172        chars[idx] = 'a';
173        appendingNeeded = true;
174
175      } else if (c == 'Z'){
176        chars[idx] = 'A';
177        appendingNeeded = true;
178
179      } else {
180        c++;
181        chars[idx] = c;
182        appendingNeeded = false;
183        break;
184      }
185      idx--;
186    }
187
188    // first char, also append to end
189    if (appendingNeeded){
190      char append = (lastOriginalChar==null || Character.isLowerCase(lastOriginalChar)) ? 'a' : 'A';
191      return String.valueOf(chars) + append;
192
193    } else {
194      return String.valueOf(chars);
195    }
196  }

222  /**
223   * Creates a random string in upper case of given length with purely latin characters only.
224   * Vocals are used much more frequently than consonants
225   * @param len
226   * @return a random string in upper case
227   */
228  public static String randomString(int len) {
229    StringBuilder sb = new StringBuilder(len);
230    for (int i = 0; i < len; i++) {
231      if (rnd.nextInt(3) > 1) {
232        sb.append(CONS.charAt(rnd.nextInt(CONS.length())));
233      } else {
234        sb.append(VOC.charAt(rnd.nextInt(VOC.length())));
235      }
236    }
237
238    return sb.toString();
239  }

265  /**
266   * Simple boolean parsing method that understands yes,y,true,t or 1 as true and respective values for false.
267   * It does not throw any exception but returns null instead.
268   *
269   * @param x
270   * @return the parsed integer or null
271   */
272  public static Boolean parseBoolean(String x) {
273    x = org.apache.commons.lang3.StringUtils.trimToEmpty(x).toLowerCase();
274    if (x.equals("true") || x.equals("t") || x.equals("1") || x.equals("yes") || x.equals("y")) {
275      return true;
276    }
277    if (x.equals("false") || x.equals("f") || x.equals("0") || x.equals("no") || x.equals("n")) {
278      return false;
279    }
280    return null;
281  }

283  /**
284   * Unescapes various unicode escapes if existing:
285   *
286   * java unicode escape, four hexadecimal digits
287   * \ uhhhh
288   *
289   * octal escape
290   * \nnn
291   * The octal value nnn, where nnn stands for 1 to 3 digits between ‘0’ and ‘7’. For example, the code for the ASCII
292   * ESC (escape) character is ‘\033’.
293   *
294   * hexadecimal escape
295   * \xhh...
296   * The hexadecimal value hh, where hh stands for a sequence of hexadecimal digits (‘0’–‘9’, and either ‘A’–‘F’ or
297   * ‘a’–‘f’).Like the same construct in ISO C, the escape sequence continues until the first nonhexadecimal digit is seen.
298   * However, using more than two hexadecimal digits produces undefined results. (The ‘\x’ escape sequence is not allowed
299   * in POSIX awk.)
300   *
301   * @param text string potentially containing unicode escape chars
302   * @return the unescaped string
303   */
304  public static String unescapeUnicodeChars(String text) {
305    if (text == null) {
306      return null;
307    }
308    // replace unicode, hexadecimal or octal character encodings by iterating over the chars once
309    //
310    // java unicode escape, four hexadecimal digits
311    // \ uhhhh
312    //
313    // octal escape
314    // \nnn
315    // The octal value nnn, where nnn stands for 1 to 3 digits between ‘0’ and ‘7’. For example, the code for the ASCII
316    // ESC (escape) character is ‘\033’.
317    //
318    // hexadecimal escape
319    // \xhh...
320    // The hexadecimal value hh, where hh stands for a sequence of hexadecimal digits (‘0’–‘9’, and either ‘A’–‘F’ or
321    // ‘a’–‘f’).
322    // Like the same construct in ISO C, the escape sequence continues until the first nonhexadecimal digit is seen.
323    // However, using more than two hexadecimal digits produces undefined results. (The ‘\x’ escape sequence is not allowed
324    // in POSIX awk.)
325    int i = 0, len = text.length();
326    char c;
327    StringBuffer sb = new StringBuffer(len);
328    while (i < len) {
329      c = text.charAt(i++);
330      if (c == '\\') {
331        if (i < len) {
332          c = text.charAt(i++);
333          try {
334            if (c == 'u' && text.length() >= i + 4) {
335              // make sure we have only hexadecimals
336              String hex = text.substring(i, i + 4);
337              if (HEX.matcher(hex).find()) {
338                c = (char) Integer.parseInt(hex, 16);
339                i += 4;
340              } else {
341                throw new NumberFormatException("No hex value: " + hex);
342              }
343            } else if (c == 'n' && text.length() >= i + 2) {
344              // make sure we have only 0-7 digits
345              String oct = text.substring(i, i + 2);
346              if (OCT.matcher(oct).find()) {
347                c = (char) Integer.parseInt(oct, 8);
348                i += 2;
349              } else {
350                throw new NumberFormatException("No octal value: " + oct);
351              }
352            } else if (c == 'x' && text.length() >= i + 2) {
353              // make sure we have only hexadecimals
354              String hex = text.substring(i, i + 2);
355              if (HEX.matcher(hex).find()) {
356                c = (char) Integer.parseInt(hex, 16);
357                i += 2;
358              } else {
359                throw new NumberFormatException("No hex value: " + hex);
360              }
361            } else if (c == 'r' || c == 'n' || c == 't') {
362              // escaped newline or tab. Replace with simple space
363              c = ' ';
364            } else {
365              throw new NumberFormatException("No char escape");
366            }
367          } catch (NumberFormatException e) {
368            // keep original characters including \ if escape sequence was invalid
369            // but replace \n with space instead
370            if (c == 'n') {
371              c = ' ';
372            } else {
373              c = '\\';
374              i--;
375            }
376          }
377        }
378      } // fall through: \ escapes itself, quotes any character but u
379      sb.append(c);
380    }
381    return sb.toString();
382  }

417  /**
418   * Uppercases all keys and values in a Map.
419   * If keys clash only one entry will remain, which is not guaranteed.
420   *
421   * @param map
422   * @return new map with keys and values upper cased.
423   */
424  public static Map<String, String> upper(Map<String, String> map) {
425    Map<String, String> upperMap = new HashMap<>();
426    for (String k : map.keySet()) {
427      String v = map.get(k);
428      if (v != null) {
429        v = v.trim().toUpperCase();
430      }
431      upperMap.put(k.toUpperCase(), v);
432    }
433    return upperMap;
434  }
