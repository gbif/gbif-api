001package org.gbif.utils.file;
002
003import java.io.ByteArrayOutputStream;
004import java.io.IOException;
005import java.io.InputStream;
006import java.io.UnsupportedEncodingException;
007import java.net.URL;
008import java.nio.charset.Charset;
009import java.nio.charset.StandardCharsets;
010
011import org.slf4j.Logger;
012import org.slf4j.LoggerFactory;
013
014public class InputStreamUtils {
015
016  private static final Logger LOG = LoggerFactory.getLogger(InputStreamUtils.class);
017
018  public InputStream classpathStream(String path) {
019    InputStream in = null;
020    // relative path. Use classpath instead
021    URL url = getClass().getClassLoader().getResource(path);
022    if (url != null) {
023      try {
024        in = url.openStream();
025      } catch (IOException e) {
026        LOG.warn("Cant open classpath input stream " + path, e);
027      }
028    }
029    return in;
030  }
031
032  /**
033   * Converts an entire InputStream to a single String with UTF8 as the character encoding.
034   *
035   * @param source source input stream to convert
036   *
037   * @return the string representing the entire input stream
038   */
039  public String readEntireStream(InputStream source) {
040    return readEntireStream(source, FileUtils.UTF8);
041  }
042
043  /**
044   * Converts an entire InputStream to a single String with explicitly provided character encoding.
045   *
046   * @param source   source input stream to convert
047   * @param encoding the stream's character encoding
048   *
049   * @return the string representing the entire input stream
050   */
051  public String readEntireStream(InputStream source, String encoding) {
052    if (!Charset.isSupported(encoding)) {
053      throw new IllegalArgumentException("Unsupported encoding " + encoding);
054    };
055
056    ByteArrayOutputStream result = new ByteArrayOutputStream();
057
058    try {
059      byte[] buffer = new byte[1024];
060      int length;
061      while ((length = source.read(buffer)) != -1) {
062        result.write(buffer, 0, length);
063      }
064    } catch (IOException e) {
065      LOG.error("Caught exception", e);
066    } finally {
067      try {
068        source.close();
069      } catch (IOException e) {
070        LOG.error("Caught exception", e);
071      }
072    }
073
074    // StandardCharsets.UTF_8.name() > JDK 7
075    try {
076      return result.toString(StandardCharsets.UTF_8.name());
077    } catch (UnsupportedEncodingException e) {
078      throw new IllegalArgumentException("Could not decode stream as " + encoding);
079    }
080  }
081}
