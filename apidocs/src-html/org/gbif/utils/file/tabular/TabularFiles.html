001package org.gbif.utils.file.tabular;
002
003import java.io.InputStream;
004import java.nio.charset.Charset;
005import java.util.List;
006
007import com.google.common.base.Charsets;
008import com.google.common.base.Preconditions;
009import org.supercsv.prefs.CsvPreference;
010
011/**
012 * Static utility methods related to {@link TabularDataFileReader} instances.
013 *
014 */
015public class TabularFiles {
016
017
018  /**
019   * Get a new TabularDataFileReader.
020   *
021   * @param in
022   * @param quoteChar
023   * @param delimiterChar
024   * @param endOfLineSymbols
025   * @param charset
026   * @param headerLine
027   * @return
028   */
029  public static TabularDataFileReader<List<String>> newTabularFileReader(InputStream in, char quoteChar, char delimiterChar,
030                                                           String endOfLineSymbols, Charset charset, boolean headerLine){
031
032    Preconditions.checkNotNull(in, "An InputStream must be provided");
033    Preconditions.checkNotNull(charset, "A Charset must be provided");
034    return new SuperCsvFileReader(in, quoteChar, delimiterChar, endOfLineSymbols, charset, headerLine);
035  }
036
037
038  /**
039   * Get a new TabularDataFileReader using UTF-8 charset, default quote char and default endOfLineSymbols.
040   *
041   * @param in
042   * @param delimiterChar
043   * @param headerLine
044   */
045  public static TabularDataFileReader<List<String>> newTabularFileReader(InputStream in, char delimiterChar,
046                                                                         boolean headerLine){
047    return newTabularFileReader(in, CsvPreference.STANDARD_PREFERENCE.getQuoteChar(),
048            delimiterChar, CsvPreference.STANDARD_PREFERENCE.getEndOfLineSymbols(), Charsets.UTF_8, headerLine);
049  }
050
051}
