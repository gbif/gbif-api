001package org.gbif.utils.file.tabular;
002
003import java.nio.charset.Charset;
004
005/**
006 * Data about a tabular data file.
007 */
008public class TabularFileMetadata {
009
010  private Charset encoding;
011  private Character delimiter;
012  private Character quotedBy;
013
014  public Charset getEncoding() {
015    return encoding;
016  }
017
018  public void setEncoding(Charset encoding) {
019    this.encoding = encoding;
020  }
021
022  public Character getDelimiter() {
023    return delimiter;
024  }
025
026  public void setDelimiter(Character delimiter) {
027    this.delimiter = delimiter;
028  }
029
030  public Character getQuotedBy() {
031    return quotedBy;
032  }
033
034  public void setQuotedBy(Character quotedBy) {
035    this.quotedBy = quotedBy;
036  }
037}
