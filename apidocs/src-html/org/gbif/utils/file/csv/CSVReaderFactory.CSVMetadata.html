036  private static final Logger LOG = LoggerFactory.getLogger(CSVReaderFactory.class);
037  private static final String[] POTENTIAL_DELIMITERS = {",", "\t", ";", "|"};
038  private static final int ROWS_TO_INSPECT = 10;

040  /**

064  /**
065   * Build a CSVReader with specific delimiter and presence or not of a header line.
066   * Encoding detection will be attempted.

070   * @return

074    return new CSVReader(source, detectEncoding(source), delimiter, null, header ? 1 : 0);

123  /**
124   * Build a CSVReader and try to detect the encoding, delimiter and quotes.
125   *
126   * @param source
127   * @param headerRows
128   * @return
129   * @throws IOException
130   */
131  public static CSVReader build(File source, Integer headerRows) throws IOException {
132    String encoding = detectEncoding(source);
133    CSVMetadata csvMeta  = extractCsvMetadata(source, encoding);
134    return new CSVReader(source, encoding, csvMeta.getDelimiter(), csvMeta.getQuotedBy(), headerRows);
135  }

156  /**
157   * Extract metadata from a CSV file.
158   * Metadata includes delimiter and quotes character.
159   * 
160   * @param source
161   * @param encoding
162   * @return
163   * @throws UnkownDelimitersException
164   */
165  public static CSVMetadata extractCsvMetadata(File source, String encoding) throws UnkownDelimitersException {
166    CSVMetadata csvMetadata = new CSVMetadata();
167    // try csv, tab and then other popular delimiters
168    // keep number of resulting columns for comparisons
169    int maxColumns = 0;
170
171    for (String delim : POTENTIAL_DELIMITERS) {
172      // test with various quotes including a dynamic one if the first char in each field is consistently the same
173      List<Character> potentialQuotes = new ArrayList<Character>();
174
175      CSVReader reader;
176      try {
177        reader = build(source, encoding, delim, null, 1);
178        Character firstChar = likelyQuoteChar(reader);
179        reader.close();
180        if (firstChar != null) {
181          potentialQuotes.add(firstChar);
182        }
183      } catch (IOException ignored) {
184      }
185      // prefer quotes for CSVs
186      if (delim.equals(",")) {
187        potentialQuotes.add('"');
188        potentialQuotes.add('\'');
189        potentialQuotes.add(null);
190      } else {
191        potentialQuotes.add(null);
192        potentialQuotes.add('"');
193        potentialQuotes.add('\'');
194      }
195
196      for (Character quote : potentialQuotes) {
197        try {
198          reader = build(source, encoding, delim, quote, 0);
199          int x = consistentRowSize(reader);
200          // try to find the delimiter and quote that will give us the maximum number of rows
201          if (x > maxColumns) {
202            csvMetadata.setDelimiter(delim);
203            csvMetadata.setQuotedBy(quote);
204            maxColumns = x;
205          }
206          reader.close();
207        } catch (IOException ignored) {
208          // swallow, maybe different delimiters work
209          // if all fail we will throw an exception at the end
210        }
211      }
212    }
213
214    if (maxColumns < 1) {
215      throw new UnkownDelimitersException("Unable to detect field delimiter");
216    }
217
218    return csvMetadata;
219  }
220
221  /**
222   * @return the number of consistent columns, -1 if non consistent or column numbers-2 in case the column numbers only
223   * differ by 1 at max.
224   */
225  private static int consistentRowSize(CSVReader reader) {
226    int rowNum = 0;
227    int columns = 0;
228    boolean plusMinusOne = false;
229    while (reader.hasNext() && rowNum < ROWS_TO_INSPECT) {
230      String[] row = reader.next();
231      if (rowNum == 0) {
232        columns = row.length;
233      }
234      if (Math.abs(columns - row.length) > 1) {
235        return -1;
236      }
237      if (columns != row.length) {
238        plusMinusOne = true;
239      }
240      rowNum++;
241    }
242    if (plusMinusOne) {
243      return columns - 2;
244    }
245    return columns;
246  }
247
248  private static String detectEncoding(File source) throws UnkownCharsetException {
249    Charset encoding;
250    try {
251      encoding = CharsetDetection.detectEncoding(source, 16384);
252      if (encoding == null) {
253        throw new UnkownCharsetException("Unable to detect the files character encoding");
254      }
255    } catch (IOException e) {
256      throw new UnkownCharsetException(e);
257    }
258    return encoding.displayName();
259  }

261  /**
262   * Checks if all non empty/null fields start with the same character.
263   *
264   * @return the first character if consistent, otherwise null
265   */
266  private static Character likelyQuoteChar(CSVReader reader) {
267    Character quote = null;
268    int line = 0;
269    while (reader.hasNext() && line < 10) {
270      line++;
271      String[] row = reader.next();
272      if (row != null) {
273        for (String col : row) {
274          if (col != null && col.length() > 0) {
275            // same char at start & end?
276            if (col.length() > 1 && col.charAt(0) == col.charAt(col.length() - 1)) {
277              // only consider non alphanumerics
278              char potQuote = col.charAt(0);
279              if (Character.isLetterOrDigit(potQuote)) {
280                break;
281              }
282              if (quote == null) {
283                quote = potQuote;
284              } else {
285                if (!quote.equals(potQuote)) {
286                  quote = null;
287                  break;
288                }
289              }
290            }
291          }
292        }
293      }
294    }
295    return quote;
296  }
