016package org.gbif.utils.file.csv;
017
018import org.gbif.utils.file.ClosableReportingIterator;
019
020import java.io.BufferedReader;
021import java.io.File;
022import java.io.FileInputStream;
023import java.io.IOException;
024import java.io.InputStream;
025import java.io.InputStreamReader;
026import java.util.Map;
027import java.util.Set;
028
029import com.google.common.base.Strings;
030import com.google.common.cache.Cache;
031import com.google.common.cache.CacheBuilder;
032import org.apache.commons.lang3.text.StrTokenizer;
033import org.slf4j.Logger;
034import org.slf4j.LoggerFactory;
035
036/**
037 *
038 */
039public class CSVReader implements ClosableReportingIterator<String[]>, Iterable<String[]> {
040
041  private static final Logger LOG = LoggerFactory.getLogger(CSVReader.class);
042  public final int headerRows;
043  public final String encoding;
044  public final String delimiter;
045  public final Character quoteChar;
046  public final String[] header;
047  private final StrTokenizer tokenizer;
048  private String row;
049  private int rows;
050  private int readRows;
051  private final Map<Integer, String> emptyLines;
052  private final BufferedReader br;
053  private boolean rowError;
054  private String errorMessage;
055  private Exception exception;
056
057  public CSVReader(File source, String encoding, String delimiter, Character quotes, Integer headerRows)
058          throws IOException {
059    this(new FileInputStream(source), encoding, delimiter, quotes, headerRows);
060  }
061
062  public CSVReader(InputStream stream, String encoding, String delimiter, Character quotes, Integer headerRows)
063          throws IOException {
064    Cache<Integer, String> cache = CacheBuilder.newBuilder().maximumSize(1000).build();
065    this.emptyLines = cache.asMap();
066    this.rows = 0;
067    this.readRows = 0;
068    this.delimiter = delimiter;
069    this.encoding = encoding;
070    this.quoteChar = quotes;
071    this.headerRows = headerRows == null || headerRows < 0 ? 0 : headerRows;
072    tokenizer = new StrTokenizer();
073    tokenizer.setDelimiterString(delimiter);
074    if (quotes != null) {
075      tokenizer.setQuoteChar(quotes);
076    }
077    tokenizer.setIgnoreEmptyTokens(false);
078    tokenizer.reset();
079    InputStreamReader reader = new InputStreamReader(stream, encoding);
080    br = new BufferedReader(reader);
081    row = br.readLine();
082    // parse header row
083    if (row == null) {
084      header = null;
085    } else {
086      tokenizer.reset(row);
087      header = tokenizer.getTokenArray();
088    }
089    // skip initial header rows?
090    while (headerRows > 0) {
091      headerRows--;
092      row = br.readLine();
093    }
094  }
095
096  public void close() {
097    try {
098      br.close();
099    } catch (IOException e) {
100      LOG.debug("Exception caught", e);
101    }
102  }
103
104  /**
105   * @return the current line number of the String[] iterator
106   */
107  public int currLineNumber() {
108    return rows;
109  }
110
111  /**
112   * @return a set of the line numbers of the firsts empty rows found in the file
113   */
114  public Set<Integer> getEmptyLines() {
115    return emptyLines.keySet();
116  }
117
118  /**
119   * @return the number of rows of data that were correctly read from the file
120   */
121  public int getReadRows() {
122    return readRows;
123  }
124
125  /*
126   * (non-Javadoc)
127   * @see java.util.Iterator#hasNext()
128   */
129  public boolean hasNext() {
130    return row != null;
131  }
132
133  public ClosableReportingIterator<String[]> iterator() {
134    return this;
135  }
136
137  /*
138   * (non-Javadoc)
139   * @see java.util.Iterator#next()
140   */
141  public String[] next() {
142    if (row == null) {
143      return null;
144    }
145    tokenizer.reset(row);
146    resetReportingIterator();
147    try {
148      row = br.readLine();
149      rows++;
150      // skip empty lines
151      while (row != null && row.length() == 0) {
152        // save line number of empty line
153        emptyLines.put(rows + headerRows + 1, "");
154        row = br.readLine();
155        rows++;
156      }
157      readRows++;
158    } catch (IOException e) {
159      LOG.debug("Exception caught", e);
160      rowError = true;
161      exception = e;
162
163      // construct error message showing exception and problem row
164      StringBuilder msg = new StringBuilder();
165      msg.append("Exception caught: ");
166      msg.append(e.getMessage());
167      if (!Strings.isNullOrEmpty(row)) {
168        msg.append("\n");
169        msg.append("Row: ");
170        msg.append(row);
171      }
172      errorMessage = msg.toString();
173
174      // ensure iteration terminates
175      row = null;
176    }
177    return tokenizer.getTokenArray();
178  }
179
180  /**
181   * Reset all reporting parameters.
182   */
183  private void resetReportingIterator() {
184    rowError = false;
185    exception = null;
186    errorMessage = null;
187  }
188
189  public void remove() {
190    throw new UnsupportedOperationException("Remove not supported");
191  }
192
193  public boolean hasRowError() {
194    return rowError;
195  }
196
197  public String getErrorMessage() {
198    return errorMessage;
199  }
200
201  public Exception getException() {
202    return exception;
203  }
204}
