016package org.gbif.utils.file.csv;
017
018import org.gbif.utils.file.ClosableReportingIterator;
019
020import java.io.BufferedReader;
021import java.io.File;
022import java.io.FileInputStream;
023import java.io.IOException;
024import java.io.InputStream;
025import java.io.InputStreamReader;
026import java.util.HashSet;
027import java.util.Map;
028import java.util.Set;
029
030import com.google.common.base.Strings;
031import com.google.common.cache.Cache;
032import com.google.common.cache.CacheBuilder;
033import org.apache.commons.lang3.text.StrTokenizer;
034import org.slf4j.Logger;
035import org.slf4j.LoggerFactory;
036
037/**
038 *
039 */
040public class CSVReader implements ClosableReportingIterator<String[]> {
041
042  private static final Logger LOG = LoggerFactory.getLogger(CSVReader.class);
043  public final int headerRows;
044  public final String encoding;
045  public final String delimiter;
046  public final Character quoteChar;
047  public final String[] header;
048  private final StrTokenizer tokenizer;
049  private String row;
050  private int rows;
051  private int readRows;
052  private final Map<Integer, String> emptyLines;
053  private final BufferedReader br;
054  private boolean rowError;
055  private String errorMessage;
056  private Exception exception;
057
058  public CSVReader(File source, String encoding, String delimiter, Character quotes, Integer headerRows)
059          throws IOException {
060    this(new FileInputStream(source), encoding, delimiter, quotes, headerRows);
061  }
062
063  public CSVReader(InputStream stream, String encoding, String delimiter, Character quotes, Integer headerRows)
064          throws IOException {
065    this(stream, encoding, delimiter, quotes, headerRows, 1000);
066  }
067
068  public CSVReader(InputStream stream, String encoding, String delimiter, Character quotes, Integer headerRows, int emptyLineCache)
069          throws IOException {
070    if (emptyLineCache > 0) {
071      Cache<Integer, String> cache = CacheBuilder.newBuilder().maximumSize(emptyLineCache).build();
072      this.emptyLines = cache.asMap();
073    } else {
074      emptyLines = null;
075    }
076    this.rows = 0;
077    this.readRows = 0;
078    this.delimiter = delimiter;
079    this.encoding = encoding;
080    this.quoteChar = quotes;
081    this.headerRows = headerRows == null || headerRows < 0 ? 0 : headerRows;
082    tokenizer = new StrTokenizer();
083    tokenizer.setDelimiterString(delimiter);
084    if (quotes != null) {
085      tokenizer.setQuoteChar(quotes);
086    }
087    tokenizer.setIgnoreEmptyTokens(false);
088    tokenizer.reset();
089    InputStreamReader reader = new InputStreamReader(stream, encoding);
090    br = new BufferedReader(reader);
091    row = br.readLine();
092    // parse header row
093    if (row == null) {
094      header = null;
095    } else {
096      tokenizer.reset(row);
097      header = tokenizer.getTokenArray();
098    }
099    // skip initial header rows?
100    while (headerRows > 0) {
101      headerRows--;
102      row = br.readLine();
103    }
104  }
105
106  /**
107   * Get the header, or null if none
108   * @return
109   */
110  public String[] getHeader() {
111    return header;
112  }
113
114  @Override
115  public void close() {
116    try {
117      br.close();
118    } catch (IOException e) {
119      LOG.debug("Exception caught", e);
120    }
121  }
122
123  /**
124   * @return the current line number of the String[] iterator
125   */
126  public int currLineNumber() {
127    return rows;
128  }
129
130  /**
131   * @return a set of the line numbers of the firsts empty rows found in the file
132   */
133  public Set<Integer> getEmptyLines() {
134    return emptyLines == null ? new HashSet<>() : emptyLines.keySet();
135  }
136
137  /**
138   * @return the number of rows of data that were correctly read from the file
139   */
140  public int getReadRows() {
141    return readRows;
142  }
143
144  /*
145   * (non-Javadoc)
146   * @see java.util.Iterator#hasNext()
147   */
148  @Override
149  public boolean hasNext() {
150    return row != null;
151  }
152
153  /*
154   * (non-Javadoc)
155   * @see java.util.Iterator#next()
156   */
157  @Override
158  public String[] next() {
159    if (row == null) {
160      return null;
161    }
162    tokenizer.reset(row);
163    resetReportingIterator();
164    try {
165      row = br.readLine();
166      rows++;
167      // skip empty lines
168      while (row != null && row.length() == 0) {
169        // save line number of empty line
170        if (emptyLines != null) {
171          emptyLines.put(rows + headerRows + 1, "");
172        }
173        row = br.readLine();
174        rows++;
175      }
176      readRows++;
177    } catch (IOException e) {
178      LOG.debug("Exception caught", e);
179      rowError = true;
180      exception = e;
181
182      // construct error message showing exception and problem row
183      StringBuilder msg = new StringBuilder();
184      msg.append("Exception caught: ");
185      msg.append(e.getMessage());
186      if (!Strings.isNullOrEmpty(row)) {
187        msg.append("\n");
188        msg.append("Row: ");
189        msg.append(row);
190      }
191      errorMessage = msg.toString();
192
193      // ensure iteration terminates
194      row = null;
195    }
196    return tokenizer.getTokenArray();
197  }
198
199  /**
200   * Reset all reporting parameters.
201   */
202  private void resetReportingIterator() {
203    rowError = false;
204    exception = null;
205    errorMessage = null;
206  }
207
208  @Override
209  public void remove() {
210    throw new UnsupportedOperationException("Remove not supported");
211  }
212
213  @Override
214  public boolean hasRowError() {
215    return rowError;
216  }
217
218  @Override
219  public String getErrorMessage() {
220    return errorMessage;
221  }
222
223  @Override
224  public Exception getException() {
225    return exception;
226  }
227}
