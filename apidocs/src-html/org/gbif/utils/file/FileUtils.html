075  /**
076   * Reads a file and returns a unique set of multiple columns from lines which are no comments (starting with #) and
077   * trims whitespace.
078   *
079   * @param source the UTF-8 encoded text file with tab delimited columns
080   * @param resultSet the set implementation to be used. Will not be cleared before reading!
081   * @param column variable length argument of column indices to process
082   * @return set of column rows
083   */
084  public static Set<String> columnsToSet(InputStream source, Set<String> resultSet, int... column) throws IOException {
085    LineIterator lines = getLineIterator(source);
086    int maxCols = 0;
087    for (int c : column) {
088      if (c > maxCols) {
089        maxCols = c;
090      }
091    }
092    while (lines.hasNext()) {
093      String line = lines.nextLine().trim();
094      // ignore comments
095      if (!ignore(line)) {
096        String[] parts = TAB_DELIMITED.split(line);
097        if (maxCols <= parts.length) {
098          for (int c : column) {
099            String cell = parts[c].trim();
100            resultSet.add(cell);
101          }
102        }
103      }
104    }
105    return resultSet;
106  }

108  public static void copyStreams(InputStream in, OutputStream out) throws IOException {
109    // write the file to the file specified
110    int bytesRead;
111    byte[] buffer = new byte[8192];
112
113    while ((bytesRead = in.read(buffer, 0, 8192)) != -1) {
114      out.write(buffer, 0, bytesRead);
115    }
116
117    out.close();
118    in.close();
119  }

223  /**
224   * Converts the byte size into human-readable format.
225   * Support both SI and byte format.
226   */
227  public static String humanReadableByteCount(long bytes, boolean si) {
228    int unit = si ? 1000 : 1024;
229    if (bytes < unit) {
230      return bytes + " B";
231    }
232    int exp = (int) (Math.log(bytes) / Math.log(unit));
233    String pre = (si ? "kMGTPE" : "KMGTPE").charAt(exp - 1) + (si ? "" : "i");
234    return String.format("%.1f %sB", bytes / Math.pow(unit, exp), pre);
235  }

256  public static ByteBuffer readByteBuffer(File file, int bufferSize) throws IOException {
257    ByteBuffer bbuf = ByteBuffer.allocate(bufferSize);
258    BufferedInputStream f = new BufferedInputStream(new FileInputStream(file), bufferSize);
259
260    int b;
261    while ((b = f.read()) != -1) {
262      if (!bbuf.hasRemaining()) {
263        break;
264      }
265      bbuf.put((byte) b);
266    }
267    f.close();
268
269    return bbuf;
270  }

346  /**
347   * Read a hashmap from a tab delimited utf8 input stream using the row number as an integer value and the entire row
348   * as the value. Ignores commented rows starting with #.
349   *
350   * @param source tab delimited text file to read
351   */
352  public static Map<String, String> streamToMap(InputStream source, Map<String, String> result) throws IOException {
353    LineIterator lines = getLineIterator(source);
354    Integer row = 0;
355    while (lines.hasNext()) {
356      row++;
357      String line = lines.nextLine().trim();
358      // ignore comments
359      if (!ignore(line)) {
360        result.put(line, row.toString());
361      }
362    }
363    return result;
364  }

366  /**
367   * Read a hashmap from a tab delimited utf8 file, ignoring commented rows starting with #.
368   *
369   * @param source tab delimited input stream to read
370   * @param key column number to use as key
371   * @param value column number to use as value
372   * @param trimToNull if true trims map entries to null
373   */
374  public static Map<String, String> streamToMap(InputStream source, Map<String, String> result, int key, int value,
375    boolean trimToNull) throws IOException {
376    LineIterator lines = getLineIterator(source);
377    int maxCols = key > value ? key : value + 1;
378    while (lines.hasNext()) {
379      String line = lines.nextLine();
380      // ignore comments
381      if (!ignore(line)) {
382        String[] parts = TAB_DELIMITED.split(line);
383        if (maxCols <= parts.length) {
384          if (trimToNull) {
385            result.put(StringUtils.trimToNull(parts[key]), StringUtils.trimToNull(parts[value]));
386          } else {
387            result.put(parts[key], parts[value]);
388          }
389        }
390      }
391    }
392    return result;
393  }

435  /**
436   * For the given list, finds the index of the lowest value using the given comparator.
437   *
438   * @param values To compare
439   * @param comparator To use
440   * @return The index of the lowest value, or -1 if they are all null
441   */
442  static int lowestValueIndex(List<String> values, Comparator<String> comparator) {
443    int index = 0;
444    String lowestValue = null;
445    for (int i = 0; i < values.size(); i++) {
446      String value = values.get(i);
447      if (lowestValue == null) {
448        lowestValue = value;
449        index = i;
450      } else if (comparator.compare(lowestValue, value) > 0) {
451        lowestValue = value;
452        index = i;
453      }
454    }
455
456    return lowestValue == null ? -1 : index;
457  }

480  /**
481   * Merges a list of intermediary sort chunk files into a single sorted file. On completion, the intermediary sort
482   * chunk files are deleted.
483   *
484   * @param sortFiles sort chunk files to merge
485   * @param sortedFileWriter writer to merge to. Can already be open and contain data
486   * @param lineComparator To use when determining the order (reuse the one that was used to sort the individual
487   *        files)
488   */
489  public void mergedSortedFiles(List<File> sortFiles, FileWriter sortedFileWriter, Comparator<String> lineComparator)
490    throws IOException {
491    List<BufferedReader> partReaders = new LinkedList<BufferedReader>();
492    try {
493      List<String> partReaderLine = new LinkedList<String>();
494      for (File f : sortFiles) {
495        partReaders.add(new BufferedReader(new FileReader(f)));
496      }
497      boolean moreData = false;
498      // load first line in
499      for (BufferedReader partReader : partReaders) {
500        String partLine = partReader.readLine();
501        if (partLine != null) {
502          moreData = true;
503        }
504        // we still add the "null" to keep the partReaders and partLineReader indexes in sync - ALWAYS
505        partReaderLine.add(partLine);
506      }
507      // keep going until all readers are exhausted
508      while (moreData) {
509        int index = lowestValueIndex(partReaderLine, lineComparator);
510        if (index >= 0) {
511          sortedFileWriter.write(partReaderLine.get(index));
512          sortedFileWriter.write("\n");
513          BufferedReader r = partReaders.get(index);
514          String partLine = r.readLine();
515          // TODO: Synchronization on local variable?
516          synchronized (partReaderLine) {
517            partReaderLine.add(index, partLine);
518            partReaderLine.remove(index + 1);
519          }
520        } else {
521          moreData = false;
522        }
523      }
524    } finally {
525      for (BufferedReader b : partReaders) {
526        try {
527          b.close();
528        } catch (RuntimeException e) {
529        }
530      }
531      // I assume it periodically flushes anyway, so only need to do once at end...
532      sortedFileWriter.flush();
533      sortedFileWriter.close();
534      // delete (intermediary) sort chunk files, only the sorted file remains
535      for (File f : sortFiles) {
536        f.delete();
537      }
538    }
539  }

541  /**
542   * Sorts the input file into the output file using the supplied delimited line parameters.
543   *
544   * This method is not reliable when the sort field may contain Unicode codepoints outside the Basic Multilingual Plane,
545   * i.e. above \uFFFF. In that case, the sort order differs from Java's String sort order.  This should not be a problem
546   * for most usage; the Supplementary Multilingual Planes contain ancient scripts, emojis, arrows and so on.
547   *
548   * @param input To sort
549   * @param sorted The sorted version of the input excluding ignored header lines (see ignoreHeaderLines)
550   * @param column the column that keeps the values to sort on
551   * @param columnDelimiter the delimiter that separates columns in a row
552   * @param enclosedBy optional column enclosing character, e.g. a double quote for CSVs
553   * @param newlineDelimiter the chars used for new lines, usually \n, \n\r or \r
554   * @param ignoreHeaderLines number of beginning lines to ignore, e.g. headers
555   */
556  public void sort(File input, File sorted, String encoding, int column, String columnDelimiter, Character enclosedBy,
557    String newlineDelimiter, int ignoreHeaderLines) throws IOException {
558    Comparator<String> lineComparator;
559    if (enclosedBy == null) {
560      lineComparator = new LineComparator(column, columnDelimiter);
561    } else {
562      lineComparator = new LineComparator(column, columnDelimiter, enclosedBy);
563    }
564    sort(input, sorted, encoding, column, columnDelimiter, enclosedBy, newlineDelimiter, ignoreHeaderLines,
565      lineComparator, false);
566  }

569  /**
570   * Sorts the input file into the output file using the supplied delimited line parameters.
571   *
572   * This method is not reliable when the sort field may contain Unicode codepoints outside the Basic Multilingual Plane,
573   * i.e. above \uFFFF. In that case, the sort order differs from Java's String sort order.  This should not be a problem
574   * for most usage; the Supplementary Multilingual Planes contain ancient scripts, emojis, arrows and so on.
575   *
576   * @param input To sort
577   * @param sorted The sorted version of the input excluding ignored header lines (see ignoreHeaderLines)
578   * @param column the column that keeps the values to sort on
579   * @param columnDelimiter the delimiter that separates columns in a row
580   * @param enclosedBy optional column enclosing character, e.g. a double quote for CSVs
581   * @param newlineDelimiter the chars used for new lines, usually \n, \r\n or \r
582   * @param ignoreHeaderLines number of beginning lines to ignore, e.g. headers
583   * @param lineComparator used to sort the output
584   * @param ignoreCase ignore case order, this parameter couldn't have any effect if the LineComparator is used
585   */
586  public void sort(File input, File sorted, String encoding, int column, String columnDelimiter, Character enclosedBy,
587    String newlineDelimiter, int ignoreHeaderLines, Comparator<String> lineComparator, boolean ignoreCase)
588    throws IOException {
589    LOG.debug("sorting " + input.getAbsolutePath() + " as new file " + sorted.getAbsolutePath());
590    if (encoding == null) {
591      LOG.warn("No encoding specified, assume UTF-8");
592      encoding = FileUtils.UTF8;
593    }
594    // if the id is in the first column, first try sorting via shell as its the fastest we can get
595    if (!sortInGnu(input, sorted, encoding, ignoreHeaderLines, column, columnDelimiter, newlineDelimiter, ignoreCase)) {
596      LOG.debug("No GNU sort available, using native Java sorting");
597      sortInJava(input, sorted, encoding, lineComparator, ignoreHeaderLines);
598    }
599  }

601  /**
602   * Sorts the input file into the output file using the supplied lineComparator.
603   *
604   * @param input To sort
605   * @param sorted The sorted version of the input excluding ignored header lines (see ignoreHeaderLines)
606   * @param lineComparator To use during comparison
607   * @param ignoreHeaderLines number of beginning lines to ignore, e.g. headers
608   */
609  public void sortInJava(File input, File sorted, String encoding, Comparator<String> lineComparator,
610    int ignoreHeaderLines) throws IOException {
611    LOG.debug("Sorting File[" + input.getAbsolutePath() + ']');
612    long start = System.currentTimeMillis();
613    List<File> sortFiles = new LinkedList<File>();
614    BufferedReader br = new BufferedReader(new InputStreamReader(new FileInputStream(input), encoding));
615    List<String> headerLines = new LinkedList<String>();
616    try {
617      String line = br.readLine();
618      int fileCount = 0;
619
620      List<String> linesToSort = new LinkedList<String>();
621      while (line != null) {
622        if (ignoreHeaderLines > 0) {
623          headerLines.add(line);
624          ignoreHeaderLines--;
625        } else {
626          linesToSort.add(line);
627
628          // if buffer is full, then sort and write to file
629          if (linesToSort.size() == linesPerMemorySort) {
630            sortFiles.add(sortAndWrite(input, encoding, lineComparator, fileCount, linesToSort));
631            linesToSort = new LinkedList<String>();
632            fileCount++;
633          }
634        }
635        line = br.readLine();
636      }
637      // catch the last lot
638      if (!linesToSort.isEmpty()) {
639        sortFiles.add(sortAndWrite(input, encoding, lineComparator, fileCount, linesToSort));
640      }
641    } finally {
642      br.close();
643    }
644    LOG.debug(
645      sortFiles.size() + " sorted file chunks created in " + (System.currentTimeMillis() - start) / 1000 + " secs");
646
647    // now merge the sorted files into one single sorted file
648    FileWriter sortedFileWriter = new FileWriter(sorted);
649    // first write the old header lines if existing
650    for (String h : headerLines) {
651      sortedFileWriter.write(h);
652      sortedFileWriter.write("\n");
653    }
654    mergedSortedFiles(sortFiles, sortedFileWriter, lineComparator);
655
656    LOG.debug(
657      "File " + input.getAbsolutePath() + " sorted successfully using " + sortFiles.size() + " parts to do sorting in "
658        + (System.currentTimeMillis() - start) / 1000 + " secs");
659  }

662  /**
663   * Splits the supplied file into files of set line size and with a suffix.
664   *
665   * @param input To split up
666   * @param linesPerOutput Lines per split file
667   * @param extension The file extension to use - e.g. ".txt"
668   * @return The split files
669   */
670  public List<File> split(File input, int linesPerOutput, String extension) throws IOException {
671    LOG.debug("Splitting File[" + input.getAbsolutePath() + ']');
672    long timer = System.currentTimeMillis();
673    List<File> splitFiles = new LinkedList<File>();
674    BufferedReader br = new BufferedReader(new FileReader(input));
675    String line = br.readLine();
676    int fileCount = 0;
677    File splitFile = getChunkFile(input, fileCount);
678    fileCount++;
679    splitFiles.add(splitFile);
680    FileWriter fw = new FileWriter(splitFile);
681    try {
682      int lineCount = 0;
683      while (line != null) {
684        if (lineCount == linesPerOutput) {
685          fw.flush();
686          fw.close();
687          splitFile = getChunkFile(input, fileCount);
688          splitFiles.add(splitFile);
689          // is ok to reuse, as last one is closed, and this will always get closed - see finally below
690          fw = new FileWriter(splitFile);
691          fileCount++;
692          lineCount = 0;
693        }
694        fw.write(line);
695        fw.write("\n");
696        line = br.readLine();
697        lineCount++;
698      }
699      fw.flush();
700    } finally {
701      fw.close();
702    }
703    LOG.debug("File[" + input.getAbsolutePath() + "] split successfully into[" + splitFiles.size() + "] parts in secs["
704      + (1 + System.currentTimeMillis() - timer) / 1000 + "]");
705    return splitFiles;
706  }

708  /**
709   * Test whether we have a new enough version of GNU Sort that supports (primarily) the -k option with a start and end
710   * column.
711   *
712   * Mac OS only includes an old version of GNU sort, and will fail this test.
713   */
714  private boolean gnuSortAvailable() {
715    if (gnuSortAvailable != null) {
716      return gnuSortAvailable;
717    }
718
719    try {
720      String command = "sort -k1,1 -t',' --ignore-case /dev/null";
721      LOG.debug("Testing capability of GNU sort with command: {}", command);
722
723      Process process = new ProcessBuilder("/bin/sh", "-c", command).start();
724      int exitValue = process.waitFor();
725
726      if (exitValue == 0) {
727        LOG.debug("GNU sort is capable");
728        gnuSortAvailable = true;
729      } else {
730        LOG.warn("GNU sort does not exist or is too old, and will not be used.  Sorting large files will be slow.",
731            new InputStreamUtils().readEntireStream(process.getErrorStream()).replace('\n', ' '));
732        gnuSortAvailable = false;
733      }
734    } catch (Exception e) {
735      LOG.warn("GNU sort does not exist or is too old, and will not be used.  Sorting large files will be slow.", e);
736      gnuSortAvailable = false;
737    }
738
739    return gnuSortAvailable;
740  }

742  /**
743   * sort a text file via an external GNU sort command:
744   * sorting tabs at 3rd column, numerical reverse order
745   * sort -t$'\t' -k3 -o sorted.txt col2007.txt
746   * <p/>
747   * The GNU sort based sorting is extremely efficient and much, much faster than the current sortInJava method. It is
748   * locale aware though and we only want the native C sorting locale. See
749   * http://www.gnu.org/software/coreutils/faq/coreutils-faq.html#Sort-does-not-sort-in-normal-order_0021
750   * <p/>
751   * Example C sort order:
752   * <p/>
753   * <pre>
754   * 1 oOdontoceti
755   * 10 gGlobicephala melaena melaena Traill
756   * 100 gGlobicephala melaena melaena Traill
757   * 101 gGlobicephala melaena melaena Traill
758   * 11 pPontoporia Gray
759   * 12 pPontoporia blainvillei Gervais and d'Orbigny
760   * 120 iInia d'Orbigny
761   * 121 iInia geoffrensis Blainville
762   * 2 sSusuidae
763   * 20 cCetacea
764   * Amphiptera
765   * Amphiptera pacifica Giglioli
766   * Anarnak Lacpde
767   * Balaena mangidach Chamisso
768   * amphiptera
769   * amphiptera pacifica Giglioli
770   * anarnak Lacpde
771   * balaena mangidach Chamisso
772   * </pre>
773   */
774  protected boolean sortInGnu(File input, File sorted, String encoding, int ignoreHeaderLines, int column,
775    String columnDelimiter, String lineDelimiter, boolean ignoreCase) throws IOException {
776    String command;
777    // GNU sort is checked for use when:
778    //  line delimiter is \n
779    //  column delimiter is not \n or '
780    //. sort version is sufficient to include start and end column (-k 1,1).
781    // Use the --debug option to sort if working on this code.
782    if (lineDelimiter == null || !lineDelimiter.contains("\n") ||
783        (columnDelimiter != null && (columnDelimiter.contains("\n") || columnDelimiter.contains("'"))) ||
784        !gnuSortAvailable()) {
785      LOG.debug("Cannot use GNU sort on this file");
786      return false;
787    }
788
789    // keep header rows
790    boolean success = false;
791    try {
792      LinkedList<String> cmds = new LinkedList<String>();
793      cmds.add("/bin/sh");
794      cmds.add("-c");
795      cmds.add("");
796      ProcessBuilder pb = new ProcessBuilder(cmds);
797      Map<String, String> env = pb.environment();
798      env.clear();
799      // make sure we use the C locale for sorting
800      env.put("LC_ALL", "C");
801
802      String sortArgs = String.format(" %s -k%d,%d -t'%s'",
803        ignoreCase ? "--ignore-case" : "", column+1, column+1, columnDelimiter);
804
805      if (ignoreHeaderLines > 0) {
806        // copy header lines
807        command = "head -n " + ignoreHeaderLines + ' ' + input.getAbsolutePath() + " > " + sorted.getAbsolutePath();
808        LOG.debug("Issue external command: {}", command);
809        cmds.removeLast();
810        cmds.add(command);
811        Process process = pb.start();
812        int exitValue = process.waitFor();
813        if (exitValue != 0) {
814          LOG.warn("Error sorting file (copying header lines) with GNU head");
815          return false;
816        }
817
818        // do the sorting ignoring the header rows
819        command = "sed " + ignoreHeaderLines + "d " + input.getAbsolutePath() + " | "
820            + "sort " + sortArgs
821            + " >> " + sorted.getAbsolutePath();
822      } else {
823        // do sorting directly, we don't have header rows
824        command = "sort " + sortArgs + " -o " + sorted.getAbsolutePath() + ' ' + input.getAbsolutePath();
825      }
826
827      LOG.debug("Issue external command: {}", command);
828      cmds.removeLast();
829      cmds.add(command);
830      Process process = pb.start();
831      // get the stdout and stderr from the command that was run
832      InputStream err = process.getErrorStream();
833      int exitValue = process.waitFor();
834      if (exitValue == 0) {
835        LOG.debug("Successfully sorted file with GNU sort");
836        success = true;
837      } else {
838        LOG.warn("Error sorting file with GNU sort");
839        InputStreamUtils isu = new InputStreamUtils();
840        System.err.append(isu.readEntireStream(err));
841      }
842    } catch (Exception e) {
843      LOG.warn("Caught Exception using GNU sort", e);
844    }
845    return success;
846  }

848  /**
849   * Sorts the lines and writes to file using the
850   *
851   * @param input File to base the name on
852   * @param lineComparator To compare the lines for sorting
853   * @param fileCount Used for the file name
854   * @param linesToSort To actually sort
855   * @return The written file
856   */
857  private File sortAndWrite(File input, String encoding, Comparator<String> lineComparator, int fileCount,
858    List<String> linesToSort) throws IOException {
859    long start = System.currentTimeMillis();
860    Collections.sort(linesToSort, lineComparator);
861    // When implementing a comparator, make it SUPER quick!!!
862    LOG.debug(
863      "Collections.sort took msec[" + (System.currentTimeMillis() - start) + "] to sort records[" + linesToSort.size()
864        + ']');
865    File sortFile = getChunkFile(input, fileCount);
866    Writer fw = new OutputStreamWriter(new FileOutputStream(sortFile), encoding);
867    try {
868      for (String s : linesToSort) {
869        fw.write(s);
870        fw.write("\n");
871      }
872    } finally {
873      fw.close();
874    }
875    return sortFile;
876  }
