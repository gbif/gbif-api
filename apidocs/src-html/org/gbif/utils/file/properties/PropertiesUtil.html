053  /**
054   * Loads a properties file.
055   * The file should be available in the classpath, the default {@link ClassLoader} is used to load the file.
056   *
057   * @throws IOException Should there be an issue in loading the file
058   * @throws IllegalArgumentException If the file does not exist
059   */
060  public static Properties loadProperties(String propertiesFile) throws IOException, IllegalArgumentException {
061    Properties tempProperties = new Properties();
062    Closer closer = Closer.create();
063    try {
064      File file = new File(propertiesFile);
065      if (file.exists()) {// first tries to load the file as a external file
066        tempProperties.load(closer.register(new FileInputStream(file)));
067      } else { // tries to load the file as a resource
068        URL configFileURL = Resources.getResource(propertiesFile);
069        tempProperties.load(closer.register(Resources.asByteSource(configFileURL).openStream()));
070      }
071    } finally {
072      closer.close();
073    }
074    return tempProperties;
075  }
076
077  /**
078   * Reads a property file from an absolute filepath.
079   */
080  public static Properties readFromFile(String filepath) throws IOException, IllegalArgumentException {
081    if (Strings.isNullOrEmpty(filepath)) {
082      throw new IllegalArgumentException("No properties file given");
083    }
084    File pf = new File(filepath);
085    if (!pf.exists()) {
086      throw new IllegalArgumentException("Cannot find properties file " + filepath);
087    }
088    Properties properties = new Properties();
089
090    Closer closer = Closer.create();
091    try {
092      FileReader reader = closer.register(new FileReader(pf));
093      properties.load(reader);
094    } finally {
095      closer.close();
096    }
097    return properties;
098  }

100  /**
101   * Reads and casts the named property as an Double.
102   *
103   * @param p The properties file to read from.
104   * @param key To read the value of.
105   * @param exceptionForNull If true, and the property is not found an IAE is thrown, otherwise defaultValue is
106   *        returned
107   * @param defaultValue If the property is not found, and exceptionForNull is false, this is returned for missing
108   *        properties.
109   * @return The property at the key as an Double
110   * @throws IllegalArgumentException if the property is invalid (can't be cast to a double) or not found and we are
111   *         instructed to throw it.
112   */
113  public static Double propertyAsDouble(Properties p, String key, boolean exceptionForNull, Double defaultValue)
114    throws IllegalArgumentException {
115    String v = p.getProperty(key);
116    if (v != null) {
117      try {
118        return Double.parseDouble(v);
119      } catch (NumberFormatException e) {
120        throw new IllegalArgumentException("Invalid value[" + v + "] supplied for " + key);
121      }
122    } else {
123      if (exceptionForNull) {
124        throw new IllegalArgumentException("Missing property for " + key);
125      } else {
126        return defaultValue;
127      }
128    }
129  }

194  /**
195   * Reads and casts the named property as a boolean.
196   * Case insensitive values for 'true', 'on', 'yes', 't' and 'y' return true values,
197   * 'false', 'off', 'no', 'f' and 'n' return false.
198   * Otherwise or in case of a missing property the default will be used.
199   *
200   * @param p The properties file to read from.
201   * @param key To read the value of.
202   * @param defaultValue If the property is not found this is returned for missing properties.
203   * @return The property at the key as a boolean
204   */
205  public static boolean propertyAsBool(Properties p, String key, boolean defaultValue) {
206    Boolean val = BooleanUtils.toBooleanObject(p.getProperty(key, null));
207    return val == null ? defaultValue : val;
208  }

210  /**
211   * Reads and converts the named property as UTF8 bytes.
212   *
213   * @param p The properties file to read from.
214   * @param key To read the value of.
215   * @param exceptionForNull If true, and the property is not found an IAE is thrown, otherwise defaultValue is
216   *        returned
217   * @param defaultValue If the property is not found, and exceptionForNull is false, this is returned for missing
218   *        properties.
219   * @return The property at the key as byte[]t
220   * @throws IllegalArgumentException if the property is not found and we are instructed to throw it.
221   */
222  public static byte[] propertyAsUTF8Bytes(Properties p, String key, boolean exceptionForNull, byte[] defaultValue)
223    throws IllegalArgumentException {
224    String v = p.getProperty(key);
225    if (v != null) {
226      try {
227        return v.getBytes(UTF8_ENCODING);
228      } catch (UnsupportedEncodingException e) {
229        // never one would hope
230        throw new RuntimeException("System does not support " + UTF8_ENCODING + " encoding");
231      }
232    } else {
233      if (exceptionForNull) {
234        throw new IllegalArgumentException("Missing property for " + key);
235      } else {
236        return defaultValue;
237      }
238    }
239  }

241  /**
242   * Filters and translates Properties with a prefix.
243   * The resulting Properties will only include the properties that start with the provided prefix with that prefix
244   * removed (e.g. myprefix.key1 will be returned as key1 if prefix = "myprefix.")
245   *
246   * @param properties to filter and translate
247   * @param prefix prefix used to filter the properties. (e.g. "myprefix.")
248   * @return new Properties object with filtered and translated properties. Never null.
249   */
250  public static Properties filterProperties(final Properties properties, String prefix) {
251    Preconditions.checkNotNull(properties, "Can't filter a null Properties");
252    Preconditions.checkState(StringUtils.isNotBlank(prefix), "Can't filter using a blank prefix", properties);
253
254    Properties filtered = new Properties();
255    for (String key : properties.stringPropertyNames()) {
256      if (key.startsWith(prefix)) {
257        filtered.setProperty(key.substring(prefix.length()), properties.getProperty(key));
258      }
259    }
260    return filtered;
261  }

263  /**
264   * Returns a new Properties object that contains only the elements where the key starts by the provided
265   * prefix. The same keys will be used in the returned Properties.
266   * @param original
267   * @param prefix
268   * @return
269   */
270  public static Properties subsetProperties(final Properties original, String prefix) {
271    return propertiesByPrefix(original, prefix, false);
272  }
273
274  /**
275   * Remove properties from the original object and return the removed element(s) as new Properties object.
276   * The same keys will be used in the returned Properties.
277   * @param original original object in which the element will be removed if key starts with provided prefix.
278   * @param prefix
279   * @return
280   */
281  public static Properties removeProperties(final Properties original, String prefix) {
282    return propertiesByPrefix(original, prefix, true);
283  }
284
285  /**
286   * Get a a new Properties object that only contains the elements that start with the prefix.
287   * The same keys will be used in the returned Properties.
288   * @param original
289   * @param prefix
290   * @param remove should the element(s) be removed from the original Properties object
291   * @return
292   */
293  private static Properties propertiesByPrefix(final Properties original, String prefix, boolean remove) {
294    Preconditions.checkNotNull(original, "Can't filter a null Properties");
295    Preconditions.checkState(StringUtils.isNotBlank(prefix), "Can't filter using a blank prefix", original);
296
297    Properties filtered = new Properties();
298
299    if(original.isEmpty()){
300      return filtered;
301    }
302
303    Iterator<Object> keysIt = original.keySet().iterator();
304    String key;
305    while (keysIt.hasNext()) {
306      key = String.valueOf(keysIt.next());
307      if (key.startsWith(prefix)) {
308        filtered.setProperty(key, original.getProperty(key));
309        if(remove){
310          keysIt.remove();
311        }
312      }
313    }
314    return filtered;
315  }
