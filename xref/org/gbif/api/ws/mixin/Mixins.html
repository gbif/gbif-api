11  import java.util.Map;

13  /**
14   * Mixins are typically used to leave serialization-oriented annotations outside the models to avoid
15   * introducing coupling on a specific SerDe.
16   * <p>
17   * This class provides access to predefined mixins used in the GBIF web service application (client and server).
18   */
19  public class Mixins {
20  
21    // utility class
22    private Mixins() {
23    }
24  
25    private static final ImmutableMap<Class<?>, Class<?>> PREDEFINED_MIXINS =
26      ImmutableMap.of(
27        Dataset.class, DatasetMixin.class,
28        DatasetSearchResult.class, DatasetMixin.class,
29        Download.class, LicenseMixin.class,
30        Occurrence.class, OccurrenceMixin.class
31      );
32  
33    /**
34     * Return an immutable map of the predefined Jackson Mixins used by the web services.
35     *
36     * @return immutable map of the predefined Jackson Mixins
37     */
38    public static Map<Class<?>, Class<?>> getPredefinedMixins() {
39      return PREDEFINED_MIXINS;
40    }
41  
42    /**
43     * Get an immutable map of the predefined Jackson Mixins after the provided keyFilter is applied.
44     *
45     * @param keyFilter predicate to filter the predefined mixins based on the model class
46     * @return immutable map of the predefined Jackson Mixins after applying the predicate
47     */
48    public static Map<Class<?>, Class<?>> getPredefinedMixins(Predicate<Class<?>> keyFilter) {
49      return ImmutableMap.copyOf(Maps.filterKeys(PREDEFINED_MIXINS, keyFilter));
50    }
51  }
