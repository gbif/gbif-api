14  package org.gbif.api.model.predicate;
15  
16  import org.gbif.api.annotation.Experimental;
17  import org.gbif.api.model.common.search.SearchParameter;
18  
19  import jakarta.annotation.Nullable;
20  
21  import com.fasterxml.jackson.annotation.JsonCreator;
22  import com.fasterxml.jackson.annotation.JsonProperty;
23  
24  import io.swagger.v3.oas.annotations.media.Schema;
25  
26  /**
27   * This predicate checks if its {@code key} is LIKE its {@code value}.
28   * The syntax for one <code>?</code> or any <code>*</code> arbitrary matching characters
29   * is the one used by ElasticSearch, Unix/DOS shells, etc.
30   */
31  @Schema(
32    description = "This predicate checks if its `key` matches a simple pattern in the `value`.\n\n" +
33      "The character `?` matches a single character, and `*` matches zero or more characters.\n" +
34      "This is similar to the matching used by ElasticSearch, Unix/DOS shells, etc."
35  )
36  public class LikePredicate<S extends SearchParameter> extends SimplePredicate<S> {
37  
38    @Schema(
39      description = "Specify which taxonomy to use."
40    )
41    @Experimental
42    @Nullable
43    private final String checklistKey;
44  
45    public LikePredicate(S key, String value, Boolean matchCase) {
46      this(key, value, null, matchCase);
47    }
48  
49    @Experimental
50    public String getChecklistKey() {
51      return checklistKey;
52    }
53  
54    @JsonCreator
55    public LikePredicate(
56        @JsonProperty("key") S key,
57        @JsonProperty("value") String value,
58        @Nullable @JsonProperty(value = "checklistKey") String checklistKey,
59        @Nullable @JsonProperty(value = "matchCase") Boolean matchCase) {
60      super(false, key, value, matchCase);
61      this.checklistKey = checklistKey;
62      // make sure we deal with a String type
63      if (!String.class.equals(key.type())) {
64        throw new IllegalArgumentException("Like comparisons are only allowed for strings but not parameter " + key);
65      }
66    }
67  }
