14  package org.gbif.api.model.predicate;
15  
16  import org.gbif.api.model.common.search.SearchParameter;
17  import org.gbif.api.util.RangeValue;
18  
19  import java.util.Objects;
20  import java.util.StringJoiner;
21  
22  import jakarta.validation.constraints.NotNull;
23  
24  import com.fasterxml.jackson.annotation.JsonCreator;
25  import com.fasterxml.jackson.annotation.JsonProperty;
26  
27  import io.swagger.v3.oas.annotations.media.Schema;
28  
29  /**
30   * This predicate checks if its {@code key} is within the range {@code value}.
31   *
32   * Note: ALA only, not advertised as part of the public GBIF API.
33   */
34  @Schema(
35    description = "This predicate checks if its `key` is within the range `value`."
36  )
37  public class RangePredicate<S extends SearchParameter> implements Predicate {
38  
39      @Schema(description = "The search parameter to test.")
40      @NotNull
41      private final S key;
42  
43      @Schema(description = "The range value to test for.")
44      @NotNull private final RangeValue value;
45  
46      @JsonCreator
47      public RangePredicate(@JsonProperty("key") S key, @JsonProperty("value") RangeValue value) {
48          Objects.requireNonNull(key, "<key> may not be null");
49          Objects.requireNonNull(value, "<value> may not be null");
50          this.key = key;
51          this.value = value;
52      }
53  
54      public S getKey() {
55          return key;
56      }
57  
58      public RangeValue getValue() {
59          return value;
60      }
61  
62    @Override
63    public String toString() {
64      return new StringJoiner(", ", this.getClass().getSimpleName() + "[", "]")
65        .add("key=" + key)
66        .add("value='" + value + "'")
67        .toString();
68    }
69  }
