14  package org.gbif.api.model.pipelines;
15  
16  /** Enum to represent the pipelines step names. */
17  public enum StepType {
18  
19    // Generic step name used to be a replace of the following three when client doesn't know the endpoint type of a dataset
20    // Interpretaion
21    TO_VERBATIM("toVerbatim", 1),
22    DWCA_TO_VERBATIM("dwcaToVerbatim", 1),
23    XML_TO_VERBATIM("xmlToVerbatim", 1),
24    ABCD_TO_VERBATIM("abcdToVerbatim", 1),
25    VERBATIM_TO_IDENTIFIER("verbatimToIdentifier", 2),
26    VERBATIM_TO_INTERPRETED("verbatimToInterpreted", 3),
27    INTERPRETED_TO_INDEX("interpretedToIndex", 4),
28    HDFS_VIEW("hdfsView", 4),
29    FRAGMENTER("fragmenter", 4),
30    EVENTS_VERBATIM_TO_INTERPRETED("eventsVerbatimToInterpreted", 4),
31    EVENTS_INTERPRETED_TO_INDEX("eventsInterpretedToIndex", 5),
32    EVENTS_HDFS_VIEW("eventsHdfsView", 5),
33    // Validator
34    VALIDATOR_UPLOAD_ARCHIVE("validatorUploadArchive", 1),
35    VALIDATOR_VALIDATE_ARCHIVE("validatorValidateArchive", 2),
36    VALIDATOR_DWCA_TO_VERBATIM("validatorDwcaToVerbatim", 3),
37    VALIDATOR_XML_TO_VERBATIM("validatorXmlToVerbatim", 3),
38    VALIDATOR_ABCD_TO_VERBATIM("validatorAbcdToVerbatim", 3),
39    VALIDATOR_TABULAR_TO_VERBATIM("validatorTabularToVerbatim", 3),
40    VALIDATOR_VERBATIM_TO_INTERPRETED("validatorVerbatimToInterpreted", 4),
41    VALIDATOR_INTERPRETED_TO_INDEX("validatorInterpretedToIndex", 5),
42    VALIDATOR_COLLECT_METRICS("validatorCollectMetrics", 6);
43  
44    private String label;
45    private int executionOrder;
46  
47    StepType(String label, int executionOrder) {
48      this.label = label;
49      this.executionOrder = executionOrder;
50    }
51  
52    public int getExecutionOrder() {
53      return executionOrder;
54    }
55  
56    public String getLabel() {
57      return label;
58    }
59  }
