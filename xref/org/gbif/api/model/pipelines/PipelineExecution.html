14  /** Models a execution of a pipeline that can include one or more steps. */
15  public class PipelineExecution implements Serializable {
16  
17    private long key;
18    private List<StepType> stepsToRun = new ArrayList<>();
19    private String rerunReason;
20    private String remarks;
21  
22    @JsonSerialize(using = LocalDateTimeSerDe.LocalDateTimeSerializer.class)

24    private LocalDateTime created;
25  
26    private String createdBy;
27    private Set<PipelineStep> steps = new TreeSet<>(STEPS_BY_START_AND_FINISH_ASC);
28  
29    public static final Comparator<PipelineExecution> PIPELINE_EXECUTION_BY_CREATED_ASC =
30        (e1, e2) -> {
31          LocalDateTime created1 = e1 != null ? e1.getCreated() : null;
32          LocalDateTime created2 = e2 != null ? e2.getCreated() : null;
33  
34          if (created1 == null) {
35            return created2 == null ? 0 : 1;
36          } else if (created2 == null) {
37            return -1;
38          }
39  
40          return created1.compareTo(created2);
41        };

111   @Override

113     if (this == o) return true;

115     PipelineExecution that = (PipelineExecution) o;
116     return key == that.key
117         && Objects.equals(stepsToRun, that.stepsToRun)
