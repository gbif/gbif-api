106     public static Set<String> getAllInterpretationAsString() {
107       return getAllInterpretation().stream().map(RecordType::name).collect(Collectors.toSet());
108     }
109 
110     public static Set<String> getAllValidatorInterpretationAsString() {
111       Set<String> set = getAllInterpretationAsString();
112       set.add(IDENTIFIER.name());
113       set.remove(IDENTIFIER_ABSENT.name());
114       return set;
115     }
