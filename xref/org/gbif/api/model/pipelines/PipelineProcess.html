13  /** Base POJO model for the Pipelines status service */
14  public class PipelineProcess implements Serializable {
15  
16    private static final long serialVersionUID = -3992826055732414678L;
17  
18    @JsonInclude(JsonInclude.Include.NON_DEFAULT)
19    private long key;
20  
21    private UUID datasetKey;
22    private String datasetTitle;
23    private int attempt;
24    private long numberRecords;
25  
26    @JsonSerialize(using = LocalDateTimeSerDe.LocalDateTimeSerializer.class)
27    @JsonDeserialize(using = LocalDateTimeSerDe.LocalDateTimeDeserializer.class)
28    private LocalDateTime created;
29  
30    private String createdBy;
31    private Set<PipelineStep> steps = new TreeSet<>(Comparator.comparing(PipelineStep::getStarted));
32

34      return key;
35    }
36  
37    public UUID getDatasetKey() {
38      return datasetKey;
39    }
40  
41    public PipelineProcess setDatasetKey(UUID datasetKey) {
42      this.datasetKey = datasetKey;
43      return this;
44    }
45  
46    public String getDatasetTitle() {
47      return datasetTitle;
48    }
49  
50    public void setDatasetTitle(String datasetTitle) {
51      this.datasetTitle = datasetTitle;
52    }
53  
54    public int getAttempt() {
55      return attempt;
56    }
57  
58    public PipelineProcess setAttempt(int attempt) {
59      this.attempt = attempt;
60      return this;
61    }
62  
63    public long getNumberRecords() {
64      return numberRecords;
65    }
66  
67    public void setNumberRecords(long numberRecords) {
68      this.numberRecords = numberRecords;
69    }
70  
71    public LocalDateTime getCreated() {
72      return created;
73    }
74  
75    public PipelineProcess setCreated(LocalDateTime created) {
76      this.created = created;
77      return this;
78    }
79  
80    public String getCreatedBy() {
81      return createdBy;
82    }
83  
84    public PipelineProcess setCreatedBy(String createdBy) {
85      this.createdBy = createdBy;
86      return this;
87    }
88  
89    public Set<PipelineStep> getSteps() {
90      return steps;
91    }
92  
93    public void setSteps(Set<PipelineStep> steps) {
94      this.steps.clear();
95      this.steps.addAll(steps);
96    }
97  
98    public void addStep(PipelineStep step) {
99      steps.add(step);
100   }
101 
102   @Override
103   public String toString() {
104     return new StringJoiner(", ", PipelineProcess.class.getSimpleName() + "[", "]").add("key=" + key)
105       .add("datasetKey=" + datasetKey)
106       .add("datasetTitle=" + datasetTitle)
107       .add("attempt=" + attempt)
108       .add("numberRecords=" + numberRecords)
109       .add("created=" + created)
110       .add("createdBy='" + createdBy + "'")
111       .add("steps=" + steps)
112       .toString();
113   }
114 
115   @Override
116   public boolean equals(Object o) {
117     if (this == o) return true;
118     if (o == null || getClass() != o.getClass()) return false;
119     PipelineProcess process = (PipelineProcess) o;
120     return attempt == process.attempt && Objects.equals(datasetKey, process.datasetKey);
121   }
122 
123   @Override
124   public int hashCode() {
125     return Objects.hash(datasetKey, attempt);
126   }
127 }
