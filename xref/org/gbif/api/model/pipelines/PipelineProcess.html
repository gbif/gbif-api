18    public static final Comparator<PipelineStep> STEPS_COMPARATOR =
19        (s1, s2) -> {
20          LocalDateTime started1 = s1 != null ? s1.getStarted() : null;
21          LocalDateTime started2 = s2 != null ? s2.getStarted() : null;
22  
23          if (started1 == null) {
24            return (started2 == null) ? 0 : 1;
25          } else if (started2 == null) {
26            return -1;
27          } else {
28            int comparison = started1.compareTo(started2);
29            if (comparison != 0) {
30              return comparison;
31            } else {
32              LocalDateTime finished1 = s1.getFinished();
33              LocalDateTime finished2 = s2.getFinished();
34              if (finished1 == null) {
35                return (finished2 == null) ? 0 : 1;
36              } else if (finished2 == null) {
37                return -1;
38              } else {
39                return finished1.compareTo(finished2);
40              }
41            }
42          }
43        };
