14  package org.gbif.api.model.literature.search;
15  
16  import org.gbif.api.model.literature.LiteratureRelevance;
17  import org.gbif.api.model.literature.LiteratureTopic;
18  import org.gbif.api.model.literature.LiteratureType;
19  import org.gbif.api.vocabulary.Country;
20  import org.gbif.api.vocabulary.GbifRegion;
21  import org.gbif.api.vocabulary.Language;
22  
23  import java.util.ArrayList;
24  import java.util.Date;
25  import java.util.HashMap;
26  import java.util.HashSet;
27  import java.util.List;
28  import java.util.Map;
29  import java.util.Set;
30  import java.util.UUID;
31  
32  import com.fasterxml.jackson.annotation.JsonProperty;
33  
34  import io.swagger.v3.oas.annotations.media.Schema;
35  
36  import lombok.AccessLevel;
37  import lombok.Getter;
38  import lombok.Setter;
39  
40  @Getter
41  @Setter
42  @SuppressWarnings("unused")
43  public class LiteratureSearchResult {
44  
45    @Getter(AccessLevel.NONE)
46    @Setter(AccessLevel.NONE)
47    private String abstract_;
48  
49    @Schema(description = "The date the literature item was found by GBIF's staff or automated processes.")
50    private String discovered;
51  
52    @Schema(description = "The authors of the publication.")
53    private List<Map<String, Object>> authors = new ArrayList<>();
54  
55    @Schema(description = "Countries or areas of focus of the study.")
56    private Set<Country> countriesOfCoverage = new HashSet<>();
57  
58    @Schema(description = "Countries or areas of institution with which authors are affiliated.")
59    private Set<Country> countriesOfResearcher = new HashSet<>();
60  
61    @Schema(description = "The date the publication was added to the GBIF literature database.")
62    private Date added;
63  
64    @Schema(description = "The publication date of the publication.  See also `year`, `month` and `day`.")
65    private Date published;
66  
67    @Schema(description = "The day of publication. See also `published`.", minimum = "1", maximum = "31")
68    private Integer day;
69  
70    @Schema(description = "Keys of GBIF downloads referenced by the publication.")
71    private List<String> gbifDownloadKey = new ArrayList<>();
72  
73    @Schema(description = "Keys of GBIF occurrences directly mentioned by the paper.")
74    private List<Long> gbifOccurrenceKey = new ArrayList<>();
75  
76    @Schema(description = "Keys of taxa from the GBIF Backbone Taxonomy that are the focus of the paper.")
77    private List<Integer> gbifTaxonKey = new ArrayList<>();
78  
79    @Schema(description = "Keys of higher taxa from the GBIF Backbone Taxonomy that are the focus of the paper.")
80    private List<Integer> gbifHigherTaxonKey = new ArrayList<>();
81  
82    @Schema(description = "A list of GBIF network keys relevant to the publication.")
83    private List<UUID> gbifNetworkKey = new ArrayList<>();
84  
85    @Schema(description = "A list of GBIF project identifiers relevant to the publication.")
86    private List<String> gbifProjectIdentifier = new ArrayList<>();
87  
88    @Schema(description = "A list of GBIF programmes relevant to the publication.")
89    private List<String> gbifProgramme = new ArrayList<>();
90  
91    @Schema(description = "The manner in which GBIF is cited in a paper.")
92    private String citationType;
93  
94    @Schema(description = "GBIF regions (political divisions) related to the publication.")
95    private Set<GbifRegion> gbifRegion = new HashSet<>();
96  
97    @Schema(description = "GBIF identifier for this literature item.")
98    private UUID id;
99  
100   @Schema(description = "Identifiers (such as DOIs) for the literature item.")
101   private Map<String, Object> identifiers = new HashMap<>();
102 
103   @Schema(description = "Keywords assigned to the literature item.")
104   private List<String> keywords = new ArrayList<>();
105 
106   @Schema(description = "The language of the publication.")
107   private Language language;
108 
109   @Schema(description = "Type of literature, e.g. journal article.")
110   private LiteratureType literatureType;
111 
112   @Schema(description = "The month of publication. See also `published`.", minimum = "1", maximum = "12")
113   private Integer month;
114 
115   @Schema(description = "Unstructured notes.")
116   private String notes;
117 
118   @Schema(description = "Whether the publication is open access.")
119   private boolean openAccess;
120 
121   @Schema(description = "Whether the publication has been peer reviewed.")
122   private boolean peerReview;
123 
124   @Schema(description = "The publisher of the paper.")
125   private String publisher;
126 
127   @Schema(description = "Relevance to GBIF community, see [literature relevance](https://www.gbif.org/faq?question=literature-relevance).")
128   private Set<LiteratureRelevance> relevance = new HashSet<>();
129 
130   @Schema(description = "Journal of publication.")
131   private String source;
132 
133   @Schema(description = "Various tags applied to the literature.")
134   private List<String> tags = new ArrayList<>();
135 
136   @Schema(description = "The title of the publication.")
137   private String title;
138 
139   @Schema(description = "Topics of the publication.")
140   private Set<LiteratureTopic> topics = new HashSet<>();
141 
142   @Schema(description = "The date this literature entry was last modified.")
143   private Date modified;
144 
145   @Schema(description = "Websites associated with the publication.")
146   private List<String> websites = new ArrayList<>();
147 
148   @Schema(description = "The year of publication.  See also `published`.")
149   private Integer year;
150 
151   @Schema(name = "abstract", description = "The abstract from the publication.")
152   @JsonProperty("abstract")
153   public String getAbstract() {
154     return abstract_;
155   }
156 
157   public void setAbstract(String abstract_) {
158     this.abstract_ = abstract_;
159   }
160 
161 }
