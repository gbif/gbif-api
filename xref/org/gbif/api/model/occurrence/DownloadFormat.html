16  package org.gbif.api.model.occurrence;
17  
18  /**
19   * Download output requested format.
20   */
21  public enum DownloadFormat {
22    /**
23     * Darwin Core Archive file that is a zip of all indexed fields in both verbatim and interpreted, plus multimedia and metadata files.
24     */
25    DWCA(".zip"),
26  
27    /**
28     * Zipped text file of the most common indexed terms, but note that it is delimited by tabs, not commas.
29     */
30    SIMPLE_CSV(".zip"),
31  
32    /**
33     * AVRO (with Deflate compression codec) format export of the most common indexed terms.
34     */
35    SIMPLE_AVRO(".avro"),
36  
37    /**
38     * AVRO (with Deflate compression codec) format export of the most common indexed terms, plus many verbatim terms.
39     *
40     * More verbatim terms may be added upon request.
41     */
42    SIMPLE_WITH_VERBATIM_AVRO(".avro"),
43  
44    /**
45     * TSV format export of the distinct species and taxonomic field associated to each.
46     */
47    SPECIES_LIST(".zip"),
48  
49    /**
50     * Special AVRO format for the <a href="https://mol.org/">Map of Life</a> project.
51     */
52    MAP_OF_LIFE(".avro"),
53  
54    /**
55     * Special export format for the <a href="https://bionomia.net/">Bionomia</a> project.
56     */
57    BIONOMIA(".zip"),
58  
59    /**
60     * Special CSV format for the <a href="https://www.iucn.org/">International Union for Conservation of Nature</a>
61     * <a href="https://www.iucnredlist.org/">Red List</a> project.
62     */
63    IUCN(".csv");
64  
65    private final String extension;
66  
67    DownloadFormat(String extension) {
68      this.extension = extension;
69    }
70  
71    public String getExtension() {
72      return extension;
73    }
74  }
