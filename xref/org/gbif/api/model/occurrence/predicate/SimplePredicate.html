35    protected SimplePredicate(boolean checkForNonEquals, OccurrenceSearchParameter key, String value) {
36      Preconditions.checkNotNull(key, "<key> may not be null");
37      Preconditions.checkNotNull(value, "<value> may not be null");
38      Preconditions.checkArgument(!value.isEmpty(), "<value> may not be empty");
39      // make sure the value is of the right type according to the key given
40      SearchTypeValidator.validate(key, value);
41  
42      this.key = key;
43      this.value = value;
44  
45      if (checkForNonEquals) {
46        checkNonEqualsComparatorAllowed();
47      }
48    }
49  
50    public OccurrenceSearchParameter getKey() {
51      return key;
52    }
53  
54    public String getValue() {
55      return value;
56    }
57  
58    /**
59     * @throws IllegalArgumentException if the key SearchParameter allows other comparators than equals
60     */
61    private void checkNonEqualsComparatorAllowed() {
62      if (Number.class.isAssignableFrom(key.type()) || Date.class.isAssignableFrom(key.type())){
63      } else {
64        throw new IllegalArgumentException("Only equals comparisons are allowed for search parameter " + key);
65      }
66    }
67  
68    @Override
69    public int hashCode() {
70      return Objects.hashCode(key, value);
71    }
72  
73    @Override
74    public String toString() {
75      return Objects.toStringHelper(this).add("key", key).add("value", value).toString();
76    }
