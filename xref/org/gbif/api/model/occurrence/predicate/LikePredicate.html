16  package org.gbif.api.model.occurrence.predicate;
17  
18  import org.gbif.api.model.occurrence.search.OccurrenceSearchParameter;
19  
20  import javax.annotation.Nullable;
21  
22  import com.fasterxml.jackson.annotation.JsonCreator;
23  import com.fasterxml.jackson.annotation.JsonProperty;
24  
25  /**
26   * This predicate checks if its {@code key} is LIKE its {@code value}.
27   * The syntax for one (_) or any (%) arbitrary matching characters is the one used by Hive.
28   * See the <a href="https://cwiki.apache.org/confluence/display/Hive/LanguageManual+UDF#Relational_Operators">
29   * Hive Language Reference for LIKE</a>.
30   */
31  public class LikePredicate extends SimplePredicate {
32  
33    @JsonCreator
34    public LikePredicate(
35      @JsonProperty("key") OccurrenceSearchParameter key,
36      @JsonProperty("value") String value,
37      @Nullable @JsonProperty(value = "matchCase") Boolean matchCase) {
38      super(false, key, value, matchCase);
39      // make sure we deal with a String type
40      if (!String.class.equals(key.type())) {
41        throw new IllegalArgumentException("Like comparisons are only allowed for strings but not parameter " + key);
42      }
43    }
44  }
