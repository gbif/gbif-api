16  package org.gbif.api.model.occurrence.predicate;
17  
18  import java.util.Collection;
19  import javax.validation.constraints.NotNull;
20  
21  import com.google.common.base.Objects;
22  import com.google.common.base.Preconditions;
23  import com.google.common.collect.ImmutableList;
24  import org.codehaus.jackson.annotate.JsonCreator;
25  import org.codehaus.jackson.annotate.JsonProperty;

27  /**
28   * A compound predicate is a Predicate that itself contains other Predicates.
29   * This is to be used as a base class because the way the containing Predicates
30   * should be combined needs to be specified (e.g. "AND", "OR").
31   */
32  public class CompoundPredicate implements Predicate {
33  
34    @NotNull
35    private final Collection<Predicate> predicates;
36  
37    @JsonCreator
38    protected CompoundPredicate(@JsonProperty("predicates") Collection<Predicate> predicates) {
39      Preconditions.checkNotNull(predicates, "Predicates may not be null");
40      Preconditions.checkArgument(!predicates.isEmpty(), "Predicates may not be empty");
41      this.predicates = ImmutableList.copyOf(predicates);
42    }
43  
44    /**
45     * Returns all the predicates this compound predicate is made out of in an <em>immutable</em> collection.
46     *
47     * @return the immutable collection of child predicates.
48     */
49    public Collection<Predicate> getPredicates() {
50      return predicates;
51    }
52  
53    @Override
54    public boolean equals(Object obj) {
55      if (this == obj) {
56        return true;
57      }
58      if (!(obj instanceof CompoundPredicate)) {
59        return true;
60      }
61  
62      final CompoundPredicate other = (CompoundPredicate) obj;
63      return Objects.equal(this.predicates, other.predicates);
64    }
65  
66    @Override
67    public int hashCode() {
68      return Objects.hashCode(predicates);
69    }
70  
71    @Override
72    public String toString() {
73      return Objects.toStringHelper(this).add("predicates", predicates).toString();
74    }
75  }
