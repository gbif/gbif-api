32  /**
33   * Request class for issuing search request to the occurrence search service.
34   */
35  public class OccurrenceSearchRequest extends FacetedSearchRequest<OccurrenceSearchParameter> {
36  
37    public OccurrenceSearchRequest() {
38      // empty block
39      super();
40    }
41  
42    public OccurrenceSearchRequest(long offset, int limit) {
43      super(offset, limit);
44    }
45  
46    public OccurrenceSearchRequest(Pageable page) {
47      super(page);
48    }
49  
50    public void addOccurrenceIDFilter(String occurrenceID) {
51      addParameter(OccurrenceSearchParameter.OCCURRENCE_ID, occurrenceID);
52    }
53  
54    public void addEstablishmentMeansFilter(EstablishmentMeans establishmentMeans) {
55      addParameter(OccurrenceSearchParameter.ESTABLISHMENT_MEANS, establishmentMeans);
56    }
57  
58    public void addBasisOfRecordFilter(BasisOfRecord basisOfRecord) {
59      addParameter(OccurrenceSearchParameter.BASIS_OF_RECORD, basisOfRecord);
60    }
61  
62    public void addTypeStatusFilter(TypeStatus typeStatus) {
63      addParameter(OccurrenceSearchParameter.TYPE_STATUS, typeStatus);
64    }
65  
66    public void addCatalogNumberFilter(String catalogNumber) {
67      addParameter(OccurrenceSearchParameter.CATALOG_NUMBER, catalogNumber);
68    }
69  
70    public void addRecordedByFilter(String collectorName) {
71      addParameter(OccurrenceSearchParameter.RECORDED_BY, collectorName);
72    }
73  
74    public void addIdentifiedByFilter(String collectorName) {
75      addParameter(OccurrenceSearchParameter.IDENTIFIED_BY, collectorName);
76    }
77  
78    public void addRecordNumberFilter(String recordNumber) {
79      addParameter(OccurrenceSearchParameter.RECORD_NUMBER, recordNumber);
80    }
81  
82    public void addCountryFilter(Country country) {
83      addParameter(OccurrenceSearchParameter.COUNTRY, country.getIso2LetterCode());
84    }
85  
86    public void addContinentFilter(Continent continent) {
87      addParameter(OccurrenceSearchParameter.CONTINENT, continent);
88    }
89  
90    public void addDatasetKeyFilter(UUID datasetKey) {
91      addParameter(OccurrenceSearchParameter.DATASET_KEY, datasetKey.toString());

95      addParameter(OccurrenceSearchParameter.GEOMETRY, geometryAsWkt);
