27  /**
28   * Supported query parameters by the occurrence search and download service.
29   * For download predicates only the numerical types support comparisons other than equals.
30   */
31  public enum OccurrenceSearchParameter implements SearchParameter {
32  
33    /**
34     * The dataset key as a uuid.
35     */
36    DATASET_KEY(UUID.class),
37  
38    /**
39     * The 4 digit year. A year of 98 will be 98 after christ, no 1998.
40     * This parameter accepts comma separated range values, e.g.:
41     * <dl>
42     * <dt>*,1810</dt>
43     * <dd>Found before or equal to 1810</dd>
44     * <dt>1848,1933</dt>
45     * <dd>Year between 1848 and 1933</dd>
46     * </dl>
47     */
48    YEAR(Integer.class),
49  
50    /**
51     * The month of the year, starting with 1 for January.
52     * A month query can be used to retrieve seasonal records when used without a year or a year range.
53     * This parameter accepts comma separated range values, e.g.:
54     * <dl>
55     * <dt>4,8</dt>
56     * <dd>Month between April and August</dd>
57     * </dl>
58     */
59    MONTH(Integer.class),
60  
61  
62    /**
63     * Event date (date the occurrence was recorded) in ISO 8601 formats:yyyy, yyyy-MM, yyyy-MM-dd and MM-dd.
64     * This parameter accepts comma separated range values, examples of valid ranges are:
65     * <dl>
66     * <dt>2001-02-11,2010-01-10</dt>
67     * <dd>Dates between 2001-02-11 and 2010-01-10</dd>
68     * <dt>2001-02,2010-01</dt>
69     * <dd>Dates between first day of 2001-02 and last day of 2010-01</dd>
70     * <dt>2001,2010</dt>
71     * <dd>Dates between first day of 2001 and last day of 2010</dd>
72     * <dt>2001,2010-01</dt>
73     * <dd>Dates between first day of 2001 and last day of 2010-01</dd>
74     * <dt>2001-01-10,2010</dt>
75     * <dd>Dates between 2001-01-10 and last day of 2010</dd>
76     * <dt>2001-01-10,*</dt>
77     * <dd>Dates after 2001-01-10</dd>
78     * <dt>*,2001-01-10</dt>
79     * <dd>Dates before 2001-01-10</dd>
80     * <dt>*</dt>
81     * <dd>all dates</dd>
82     * </dl>
83     */
84    EVENT_DATE(Date.class),
85  
86    /**
87     * Last interpreted date in ISO 8601 formats:yyyy, yyyy-MM, yyyy-MM-dd and MM-dd.
88     * This parameter accepts comma separated range values, examples of valid ranges are:
89     * <dl>
90     * <dt>2001-02-11,2010-01-10</dt>
91     * <dd>Dates between 2001-02-11 and 2010-01-10</dd>
92     * <dt>2001-02,2010-01</dt>
93     * <dd>Dates between first day of 2001-02 and last day of 2010-01</dd>
94     * <dt>2001,2010</dt>
95     * <dd>Dates between first day of 2001 and last day of 2010</dd>
96     * <dt>2001,2010-01</dt>
97     * <dd>Dates between first day of 2001 and last day of 2010-01</dd>
98     * <dt>2001-01-10,2010</dt>
99     * <dd>Dates between 2001-01-10 and last day of 2010</dd>
100    * <dt>2001-01-10,*</dt>
101    * <dd>Dates after 2001-01-10</dd>
102    * <dt>*,2001-01-10</dt>
103    * <dd>Dates before 2001-01-10</dd>
104    * <dt>*</dt>
105    * <dd>all dates</dd>
106    * </dl>
107    */
108   LAST_INTERPRETED(Date.class),
109 
110   /**
111    * Latitude in decimals between -90 and 90 based on WGS 84.
112    */
113   DECIMAL_LATITUDE(Double.class),
114 
115   /**
116    * Longitude in decimals between -180 and 180 based on WGS 84.
117    */
118   DECIMAL_LONGITUDE(Double.class),
119 
120   /**
121    * Country the occurrence was recorded in.
122    */
123   COUNTRY(Country.class),
124 
125   /**
126    * Continent the occurrence was recorded in.
127    */
128   CONTINENT(Continent.class),
129 
130   /**
131    * The country of the organization that publishes the dataset the occurrence belongs to.
132    */
133   PUBLISHING_COUNTRY(Country.class),
134 
135   /**
136    * Altitude/elevation in meters above sea level.
137    * This parameter accepts comma separated range values, e.g.:
138    * <dl>
139    * <dt>*,100</dt>
140    * <dd>Altitude below or equals 100m</dd>
141    * <dt>100,*</dt>
142    * <dd>Altitude above or equals 100m</dd>
143    * <dt>-2,8.8</dt>
144    * <dd>Altitude between or equals -2m and 8.8m</dd>
145    * </dl>
146    */
147   ELEVATION(Double.class),
148 
149   /**
150    * Depth in meters relative to altitude. For example 10 meters below a lake surface with given altitude.
151    * This parameter accepts comma separated range values, e.g.:
152    * <dl>
153    * <dt>*,10</dt>
154    * <dd>Depth below or equals 10m</dd>
155    * <dt>100,*</dt>
156    * <dd>Depth above or equals 100m</dd>
157    * <dt>12.1,28.8</dt>
158    * <dd>Depth between or equals 12.1m and 28.8m</dd>
159    * </dl>
160    */
161   DEPTH(Double.class),
162 
163   /**
164    * An identifier of any form assigned by the source to identify the institution
165    * the record belongs to. Not guaranteed to be unique.
166    */
167   INSTITUTION_CODE(String.class),
168 
169   /**
170    * An identifier of any form assigned by the source to identify the physical collection or digital dataset
171    * uniquely within the context of an institution.
172    */
173   COLLECTION_CODE(String.class),
174 
175   /**
176    * An identifier of any form assigned by the source within a physical collection or digital dataset for the record
177    * which may not be unique, but should be fairly unique in combination with the institution and collection code.
178    */
179   CATALOG_NUMBER(String.class),
180 
181   /**
182    * The person who recorded the occurrence.
183    */
184   RECORDED_BY(String.class),
185 
186   /**
187    * An identifier given to the Occurrence at the time it was recorded.
188    */
189   RECORD_NUMBER(String.class),
190 
191   /**
192    * A basis of record enumeration value.
193    */
194   BASIS_OF_RECORD(BasisOfRecord.class),
195 
196   /**
197    * A taxon key from the GBIF backbone. All included and synonym taxa are included in the search, so a search for
198    * aves with taxonKey=212 will match all birds, no matter which species.
199    */
200   TAXON_KEY(Integer.class),
201 
202   /**
203    * Searches the interpreted, full scientific name of the occurrence.
204    */
205   SCIENTIFIC_NAME(String.class),
206 
207 
208   /**
209    * Searches for occurrence records which contain a value on its coordinate fields (latitude and longitude).
210    * HAS_COORDINATE=true searches for occurrence records with a coordinate value.
211    * HAS_COORDINATE=false searches for occurrence records without a coordinate value.
212    */
213   HAS_COORDINATE(Boolean.class),
214 
215 
216   /**
217    * Geometry in <a href="http://en.wikipedia.org/wiki/Well-known_text">Well Known Text</a> (WKT) format.
218    * E.g.: POLYGON ((30.0 10.0, 10.12 20.23, 20 40, 40 40, 30 10)).
219    * Multi geometries like MULTIPOLYGON are not supported and multiple parameters should be used instead.
220    * Valid geometries are:
221    * <ul>
222    * <li>POINT</li>
223    * <li>LINESTRING</li>
224    * <li>POLYGON</li>
225    * <li>LINEARRING</li>
226    * </ul>
227    */
228   GEOMETRY(String.class),
229 
230 
231   /**
232    * Includes/excludes occurrence records which contain spatial issues.
233    * SPATIAL_ISSUES=true include records with spatial issues.
234    * SPATIAL_ISSUES=false exclude records with spatial issues.
235    * The absence of this parameter returns any record with or without spatial issues.
236    */
237   SPATIAL_ISSUES(Boolean.class),
238 
239   /**
240    * Searches occurrence for those that have a specific issue.
241    */
242   ISSUE(OccurrenceIssue.class),
243 
244   /**
245    * Nomenclatural type (type status, typified scientific name, publication) applied to the subject.
246    */
247   TYPE_STATUS(TypeStatus.class),
248 
249 
250   /**
251    * The kind of media object.
252    * Recommended terms from the DCMI Type Vocabulary are StillImage, Sound or MovingImage for GBIF to index and show the
253    * media files.
254    */
255   MEDIA_TYPE(MediaType.class);
256

262   }

265    * @return the data type expected for the value of the respective search parameter

268     return type;
269   }
