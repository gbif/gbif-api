32  /**
33   * Supported query parameters by the occurrence search and download service.
34   * For download predicates only the numerical types support comparisons other than equals.
35   */
36  public enum OccurrenceSearchParameter implements SearchParameter {
37  
38    /**
39     * The dataset key as a UUID.
40     */
41    DATASET_KEY(UUID.class),
42  
43    /**
44     * The 4 digit year. A year of 98 will be 98 common era, not 1998.
45     * This parameter accepts comma separated range values, e.g.:
46     * <dl>
47     * <dt>*,1810</dt>
48     * <dd>Found before or equal to 1810</dd>
49     * <dt>1848,1933</dt>
50     * <dd>Year between 1848 and 1933</dd>
51     * </dl>
52     */
53    YEAR(Integer.class),
54  
55    /**
56     * The month of the year, starting with 1 for January.
57     * A month query can be used to retrieve seasonal records when used without a year or a year range.
58     * This parameter accepts comma separated range values, e.g.:
59     * <dl>
60     * <dt>4,8</dt>
61     * <dd>Month between April and August</dd>
62     * </dl>
63     */
64    MONTH(Integer.class),
65  
66    /**
67     * Event date (date the occurrence was recorded) in ISO 8601 formats:yyyy, yyyy-MM, yyyy-MM-dd and MM-dd.
68     * This parameter accepts comma separated range values, examples of valid ranges are:
69     * <dl>
70     * <dt>2001-02-11,2010-01-10</dt>
71     * <dd>Dates between 2001-02-11 and 2010-01-10</dd>
72     * <dt>2001-02,2010-01</dt>
73     * <dd>Dates between first day of 2001-02 and last day of 2010-01</dd>
74     * <dt>2001,2010</dt>
75     * <dd>Dates between first day of 2001 and last day of 2010</dd>
76     * <dt>2001,2010-01</dt>
77     * <dd>Dates between first day of 2001 and last day of 2010-01</dd>
78     * <dt>2001-01-10,2010</dt>
79     * <dd>Dates between 2001-01-10 and last day of 2010</dd>
80     * <dt>2001-01-10,*</dt>
81     * <dd>Dates after 2001-01-10</dd>
82     * <dt>*,2001-01-10</dt>
83     * <dd>Dates before 2001-01-10</dd>
84     * <dt>*</dt>
85     * <dd>all dates</dd>
86     * </dl>
87     */
88    EVENT_DATE(Date.class),
89  
90    /**
91     * An identifier for the set of information associated with an Event (something that occurs at a place and time).
92     * May be a global unique identifier or an identifier specific to the data set.
93     */
94    EVENT_ID(String.class),
95  
96    /**
97     * An identifier for the broader Event that groups this and potentially other Events.
98     */
99    PARENT_EVENT_ID(String.class),
100 
101   /**
102    * The name of, reference to, or description of the method or protocol used during an Event.
103    */
104   SAMPLING_PROTOCOL(String.class),
105 
106   /**
107    * Last interpreted date in ISO 8601 formats:yyyy, yyyy-MM, yyyy-MM-dd and MM-dd.
108    * This parameter accepts comma separated range values, examples of valid ranges are:
109    * <dl>
110    * <dt>2001-02-11,2010-01-10</dt>
111    * <dd>Dates between 2001-02-11 and 2010-01-10</dd>
112    * <dt>2001-02,2010-01</dt>
113    * <dd>Dates between first day of 2001-02 and last day of 2010-01</dd>
114    * <dt>2001,2010</dt>
115    * <dd>Dates between first day of 2001 and last day of 2010</dd>
116    * <dt>2001,2010-01</dt>
117    * <dd>Dates between first day of 2001 and last day of 2010-01</dd>
118    * <dt>2001-01-10,2010</dt>
119    * <dd>Dates between 2001-01-10 and last day of 2010</dd>
120    * <dt>2001-01-10,*</dt>
121    * <dd>Dates after 2001-01-10</dd>
122    * <dt>*,2001-01-10</dt>
123    * <dd>Dates before 2001-01-10</dd>
124    * <dt>*</dt>
125    * <dd>all dates</dd>
126    * </dl>
127    */
128   LAST_INTERPRETED(Date.class),
129 
130 
131   /**
132    * Modified date in ISO 8601 formats:yyyy, yyyy-MM, yyyy-MM-dd and MM-dd.
133    * This parameter accepts comma separated range values, examples of valid ranges are:
134    * <dl>
135    * <dt>2001-02-11,2010-01-10</dt>
136    * <dd>Dates between 2001-02-11 and 2010-01-10</dd>
137    * <dt>2001-02,2010-01</dt>
138    * <dd>Dates between first day of 2001-02 and last day of 2010-01</dd>
139    * <dt>2001,2010</dt>
140    * <dd>Dates between first day of 2001 and last day of 2010</dd>
141    * <dt>2001,2010-01</dt>
142    * <dd>Dates between first day of 2001 and last day of 2010-01</dd>
143    * <dt>2001-01-10,2010</dt>
144    * <dd>Dates between 2001-01-10 and last day of 2010</dd>
145    * <dt>2001-01-10,*</dt>
146    * <dd>Dates after 2001-01-10</dd>
147    * <dt>*,2001-01-10</dt>
148    * <dd>Dates before 2001-01-10</dd>
149    * <dt>*</dt>
150    * <dd>all dates</dd>
151    * </dl>
152    */
153   MODIFIED(Date.class),
154 
155   /**
156    * Latitude in decimals between -90 and 90 based on WGS 84.
157    */
158   DECIMAL_LATITUDE(Double.class),
159 
160   /**
161    * Longitude in decimals between -180 and 180 based on WGS 84.
162    */
163   DECIMAL_LONGITUDE(Double.class),
164 
165   /**
166    * The uncertainty of the coordinate in meters.
167    * This parameter accepts comma separated range values, e.g.:
168    * <dl>
169    * <dt>*,100</dt>
170    * <dd>Uncertainty below or equals 100m</dd>
171    * <dt>10000,*</dt>
172    * <dd>Uncertainty above or equals 10,000m</dd>
173    * <dt>1000,5000</dt>
174    * <dd>Uncertainty between or equals 1000m and 5000m
175    * </dd>
176    * </dl>
177    */
178   COORDINATE_UNCERTAINTY_IN_METERS(Double.class),
179 
180   /**
181    * Country the occurrence was recorded in.
182    */
183   COUNTRY(Country.class),
184 
185   /**
186    * Continent the occurrence was recorded in.
187    */
188   CONTINENT(Continent.class),
189 
190   /**
191    * The country of the organization that publishes the dataset the occurrence belongs to.
192    */
193   PUBLISHING_COUNTRY(Country.class),
194 
195   /**
196    * Altitude/elevation in meters above sea level.
197    * This parameter accepts comma separated range values, e.g.:
198    * <dl>
199    * <dt>*,100</dt>
200    * <dd>Altitude below or equals 100m</dd>
201    * <dt>100,*</dt>
202    * <dd>Altitude above or equals 100m</dd>
203    * <dt>-2,8.8</dt>
204    * <dd>Altitude between or equals -2m and 8.8m</dd>
205    * </dl>
206    */
207   ELEVATION(Double.class),
208 
209   /**
210    * Depth in meters relative to altitude. For example 10 meters below a lake surface with given altitude.
211    * This parameter accepts comma separated range values, e.g.:
212    * <dl>
213    * <dt>*,10</dt>
214    * <dd>Depth below or equals 10m</dd>
215    * <dt>100,*</dt>
216    * <dd>Depth above or equals 100m</dd>
217    * <dt>12.1,28.8</dt>
218    * <dd>Depth between or equals 12.1m and 28.8m</dd>
219    * </dl>
220    */
221   DEPTH(Double.class),
222 
223   /**
224    * An identifier of any form assigned by the source to identify the institution
225    * the record belongs to. Not guaranteed to be unique.
226    */
227   INSTITUTION_CODE(String.class),
228 
229   /**
230    * An identifier of any form assigned by the source to identify the physical collection or digital dataset
231    * uniquely within the context of an institution.
232    */
233   COLLECTION_CODE(String.class),
234 
235   /**
236    * An identifier of any form assigned by the source within a physical collection or digital dataset for the record
237    * which may not be unique, but should be fairly unique in combination with the institution and collection code.
238    */
239   CATALOG_NUMBER(String.class),
240 
241   /**
242    * The person who recorded the occurrence.
243    */
244   RECORDED_BY(String.class),
245 
246   /**
247    * The person who identified the occurrence.
248    */
249   IDENTIFIED_BY(String.class),
250 
251   /**
252    * An identifier given to the Occurrence at the time it was recorded.
253    */
254   RECORD_NUMBER(String.class),
255 
256   /**
257    * A basis of record enumeration value.
258    */
259   BASIS_OF_RECORD(BasisOfRecord.class),
260 
261   /**
262    * A taxon key from the GBIF backbone. All included and synonym taxa are included in the search, so a search for
263    * aves with taxonKey=212 will match all birds, no matter which species.
264    */
265   TAXON_KEY(Integer.class),
266 
267   /**
268    * A taxon key from the GBIF backbone for the name usage of the currently valid or accepted taxon.
269    */
270   ACCEPTED_TAXON_KEY(Integer.class),
271 
272   /**
273    * A kingdom key from the GBIF backbone.
274    */
275   KINGDOM_KEY(Integer.class),
276 
277   /**
278    * A phylum key from the GBIF backbone.
279    */
280   PHYLUM_KEY(Integer.class),
281 
282   /**
283    * A class key from the GBIF backbone.
284    */
285   CLASS_KEY(Integer.class),
286 
287   /**
288    * A order key from the GBIF backbone.
289    */
290   ORDER_KEY(Integer.class),
291 
292   /**
293    * A family key from the GBIF backbone.
294    */
295   FAMILY_KEY(Integer.class),
296 
297   /**
298    * A genus key from the GBIF backbone.
299    */
300   GENUS_KEY(Integer.class),
301 
302   /**
303    * A subgenus key from the GBIF backbone.
304    */
305   SUBGENUS_KEY(Integer.class),
306 
307   /**
308    * A species key from the GBIF backbone.
309    */
310   SPECIES_KEY(Integer.class),
311 
312   /**
313    * Searches the interpreted, full scientific name of the occurrence.
314    */
315   SCIENTIFIC_NAME(String.class),
316 
317   /**
318    * Scientific name as provided byt the source.
319    */
320   VERBATIM_SCIENTIFIC_NAME(String.class),
321 
322   /**
323    * Verbatim identifier for the set of taxon information. May be a global unique identifier or an identifier specific to the data set.
324    */
325   TAXON_ID(String.class),
326 
327   /**
328    * The status of the use of the  GBIF Backbone taxonKey.
329    */
330   TAXONOMIC_STATUS(TaxonomicStatus.class),
331 
332   /**
333    * Searches for occurrence records which contain a value on its coordinate fields (latitude and longitude).
334    * HAS_COORDINATE=true searches for occurrence records with a coordinate value.
335    * HAS_COORDINATE=false searches for occurrence records without a coordinate value.
336    */
337   HAS_COORDINATE(Boolean.class),
338 
339   /**
340    * Geometry in <a href="https://en.wikipedia.org/wiki/Well-known_text">Well Known Text</a> (WKT) format.
341    * E.g.: POLYGON ((30.0 10.0, 10.12 20.23, 20 40, 40 40, 30 10)).
342    * Multi geometries like MULTIPOLYGON are not supported and multiple parameters should be used instead.
343    * Valid geometries are:
344    * <ul>
345    * <li>POINT</li>
346    * <li>LINESTRING</li>
347    * <li>POLYGON</li>
348    * <li>LINEARRING</li>
349    * </ul>
350    */
351   GEOMETRY(String.class),
352 
353   /**
354    * Use in combination of LATITUDE and LONGITUDE parameters matches within a given distance.
355    * E.g.: geo_distance=100m,40,90 geo_distance=100km,40,90 geo_distance=100mi,40,90.
356    * See supported units in {@link org.gbif.api.model.occurrence.geo.DistanceUnit}.
357    */
358   GEO_DISTANCE(String.class),
359 
360   /**
361    * Includes/excludes occurrence records which contain geospatial issues for their coordinate.
362    * See {@link org.gbif.api.vocabulary.OccurrenceIssue#GEOSPATIAL_RULES}
363    * HAS_GEOSPATIAL_ISSUE=true include records with spatial issues.
364    * HAS_GEOSPATIAL_ISSUE=false exclude records with spatial issues.
365    * The absence of this parameter returns any record with or without spatial issues.
366    */
367   HAS_GEOSPATIAL_ISSUE(Boolean.class),
368 
369   /**
370    * Searches occurrence for those that have a specific issue.
371    */
372   ISSUE(OccurrenceIssue.class),
373 
374   /**
375    * Nomenclatural type (type status, typified scientific name, publication) applied to the subject.
376    */
377   TYPE_STATUS(TypeStatus.class),
378 
379   /**
380    * The kind of media object.
381    * Recommended terms from the DCMI Type Vocabulary are StillImage, Sound or MovingImage for GBIF to index and show the
382    * media files.
383    */
384   MEDIA_TYPE(MediaType.class),
385 
386   /**
387    *  An identifier for the Occurrence (as opposed to a particular digital record of the occurrence).
388    *  In the absence of a persistent global unique identifier, construct one from a combination of identifiers in the
389    *  record that will most closely make the occurrenceID globally unique.
390    */
391   OCCURRENCE_ID(String.class),
392 
393   /**
394    * The process by which the biological individual(s) represented in the Occurrence became established at the location.
395    */
396   ESTABLISHMENT_MEANS(EstablishmentMeans.class),
397 
398   /**
399    * Searches for records whose publishing country is different to the country where the record was recorded in.
400    */
401   REPATRIATED(Boolean.class),
402 
403   /**
404    * An identifier for the Organism instance (as opposed to a particular digital record of the Organism).
405    * May be a globally unique identifier or an identifier specific to the data set.
406    */
407   ORGANISM_ID(String.class),
408 
409   /**
410    * The name of the next smaller administrative region than country in which the Location occurs.
411    */
412   STATE_PROVINCE(String.class),
413 
414   /**
415    * The name of the water body in which the Location occurs.
416    */
417   WATER_BODY(String.class),
418 
419   /**
420    * The specific description of the place.
421    * It may contain information modified from the original to correct perceived errors or standardize the description.
422    */
423   LOCALITY(String.class),
424
