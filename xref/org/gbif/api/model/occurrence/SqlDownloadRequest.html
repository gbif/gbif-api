11  /**
12   * SQL Based downloads.
13   * This type of downloads accept and simple SELECT query that follows the  pattern:
14   *   SELECT field1, field2, ...fieldN FROM occurrence WHERE [multiple simple predicates] GROUP BY field1,field2..fieldN.
15   */
16  public class SqlDownloadRequest extends DownloadRequest {
17  
18    private String sql;
19  
20    public SqlDownloadRequest() {
21  
22    }
23  
24    /**
25     * Full constructor. Used to create instances using JSON serialization.
26     */
27    @JsonCreator
28    public SqlDownloadRequest(@JsonProperty("sql") String sql,
29                              @JsonProperty("creator") @Nullable String creator,
30                              @JsonProperty("notification_address") @Nullable Collection<String> notificationAddresses,
31                              @JsonProperty("send_notification") @Nullable boolean sendNotification) {
32      super(creator, notificationAddresses, sendNotification, DownloadFormat.SQL);
33      this.sql = sql;
34    }
35  
36    /**
37     *
38     * @return the sql query.
39     */
40    @Valid
41    public String getSql() {
42      return sql;
43    }
44  
45    /**
46     * This parameter when present provides the SQL query for custom download
47     * @param sql
48     */
49    public void setSql(String sql) {
50      this.sql = sql;
51    }
52  
53    @Override
54    public int hashCode() {
55      return Objects.hashCode(super.hashCode(), sql);
56    }
57  
58    @Override
59    public boolean equals(Object obj) {
60      if (this == obj) {
61        return true;
62      }
63      if (!(obj instanceof SqlDownloadRequest)) {
64        return false;
65      }
66  
67      SqlDownloadRequest that = (SqlDownloadRequest) obj;
68      return super.equals(that) && Objects.equal(this.sql, that.sql);
69    }
70  
71    @Override
72    public String toString() {
73      return MoreObjects.toStringHelper(this).addValue(super.toString())
74        .add("sql", sql).toString();
75    }
76  }
