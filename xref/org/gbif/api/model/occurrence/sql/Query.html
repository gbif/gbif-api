5   import org.codehaus.jackson.map.annotate.JsonSerialize;

7   /**
8    * SQL Download API Query Issues listed with there descriptions.
9    */
10  public class Query {
11  
12    private static final String FORMAT =
13      "Format of LEGAL query is : SELECT ... \nFROM occurrence \nWHERE ... \nGROUP BY ...";
14  
15    @JsonSerialize(using = QueryIssueSerde.class)
16    public enum Issue {
17  
18      NO_ISSUE("No Issue."),
19      EMPTY_SQL("SQL query cannot be empty."),
20      ONLY_ONE_SELECT_ALLOWED("SQL query should be a SELECT query with SELECT keyword used only once."),
21      DDL_JOINS_UNION_NOT_ALLOWED("SQL query cannot use ORDER BY, INSERT, UPDATE, DELETE, MERGE, PROCEDURE_CALL, EXCEPT, INTERSECT, UNION, JOIN or AS."),
22      DATASET_AND_LICENSE_REQUIRED("SQL query should select on 'datasetkey' and 'license' fields as they are required for citations."),
23      CANNOT_EXECUTE("Query cannot be executed because of"),
24      TABLE_NAME_NOT_OCCURRENCE("Query should have table name as 'occurrence'."),
25      PARSE_FAILED(String.format("Cannot parse the query, the sql query is invalid, Make sure provided query should follow the format %n %s.", FORMAT)),
26      CANNOT_USE_ALLFIELDS("Usage of '*' for selecting all fields not allowed, Please explicitly add the desired fields, refer the fields from occurrence/download/request/sql/describe endpoint."),
27      HAVING_CLAUSE_NOT_SUPPORTED("SQL Download API donot support query with HAVING clause");
28  
29      Issue(String description) {
30        this.description = description;
31      }
32  
33      private final String description;
34      private String comment = "";
35  
36      public String description() {
37        return description;
38      }
39  
40      public String comment() {
41        return comment;
42      }
43  
44      public Issue withComment(String comment) {
45        this.comment = comment;
46        return this;
47      }
48    }
49  }
