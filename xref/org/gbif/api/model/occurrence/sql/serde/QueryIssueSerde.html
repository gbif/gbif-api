12  /**
13   * Json serializer, to use issue name with description and comment upon serialization.
14   */
15  public class QueryIssueSerde extends JsonSerializer<Query.Issue> {
16  
17    @Override
18    public void serialize(Issue value, JsonGenerator generator, SerializerProvider provider) throws IOException {
19      generator.writeString(String.format("%s: %s %s", value.name(), value.description(), value.comment()));
20    }
21  }
