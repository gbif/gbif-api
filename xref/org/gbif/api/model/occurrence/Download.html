43    /**
44     * Reflects the possibles statuses of a download during its execution.
45     * The download statuses are:
46     * - PREPARING: the download is in a queue, waiting for other downloads to complete
47     * - RUNNING: the download is being processed
48     * - SUCCEEDED: the file is ready to be downloaded
49     * - CANCELLED: the download was cancelled by the user
50     * - KILLED: the download was killed by the workflow engine
51     * - FAILED: the download failed
52     * - SUSPENDED: the download was paused and its execution will be resumed later
53     * - FILE_ERASED: the download was successful, but the download file has been deleted
54     */
55    @Schema(
56      description = "The status of the download.\n\n" +
57        "* `PREPARING`: the download is in a queue, waiting for other downloads to complete\n" +
58        "* `RUNNING`: the download is being processed\n" +
59        "* `SUCCEEDED`: the download has completed and the file is ready to be downloaded\n" +
60        "* `CANCELLED`: the download was cancelled by the user\n" +
61        "* `KILLED`: the download was killed by the download engine\n" +
62        "* `FAILED`: the download failed\n" +
63        "* `SUSPENDED`: the download was paused and its execution will be resumed later\n" +
64        "* `FILE_ERASED`: the download was successful, but the download file " +
65        "[has since been deleted](https://www.gbif.org/faq?question=for-how-long-will-does-gbif-store-downloads)."
66    )
67    public enum Status {
68      PREPARING,
69      RUNNING,
70      SUCCEEDED,
71      CANCELLED,
72      KILLED,
73      FAILED,
74      SUSPENDED,
75      FILE_ERASED;
76  
77      /**
78       * Statuses that represent a download that that hasn't finished.
79       */
80      public static final EnumSet<Status> EXECUTING_STATUSES = EnumSet.of(PREPARING, RUNNING, SUSPENDED);
81  
82      /**
83       * Statuses that represent a download that that has finished.
84       */
85      public static final EnumSet<Status> FINISH_STATUSES = EnumSet.of(SUCCEEDED, CANCELLED, KILLED, FAILED, FILE_ERASED);
86    }
