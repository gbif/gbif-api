21    /**
22     * Reflects the possibles statuses of a download during its execution.
23     * The download statuses are:
24     * - PREPARING: the download is initiating
25     * - RUNNING: the download is being processed
26     * - SUCCEEDED: the file is ready to be downloaded
27     * - CANCELLED: the download was cancelled by the user
28     * - KILLED: the download was killed by the workflow engine
29     * - FAILED: the download failed
30     * - SUSPENDED: the download was paused and its execution will be resumed later
31     * - FILE_DELETED: the download was successful, but the download file has been deleted
32     */
33    public enum Status {
34      PREPARING,
35      RUNNING,
36      SUCCEEDED,
37      CANCELLED,
38      KILLED,
39      FAILED,
40      SUSPENDED,
41      FILE_ERASED;
42  
43      /**
44       * Statuses that represent a download that that hasn't finished.
45       */
46      public static final EnumSet<Status> EXECUTING_STATUSES = EnumSet.of(PREPARING, RUNNING, SUSPENDED);
47  
48      /**
49       * Statuses that represent a download that that has finished.
50       */
51      public static final EnumSet<Status> FINISH_STATUSES = EnumSet.of(SUCCEEDED, CANCELLED, KILLED, FAILED, FILE_ERASED);
52    }
