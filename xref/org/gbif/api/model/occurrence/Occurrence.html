159   /**
160    * Create occurrence instance from existing verbatim one, copying over all data.
161    */
162   public Occurrence(@Nullable VerbatimOccurrence verbatim) {
163     if (verbatim != null) {
164       setKey(verbatim.getKey());
165       setDatasetKey(verbatim.getDatasetKey());
166       setPublishingOrgKey(verbatim.getPublishingOrgKey());
167       setPublishingCountry(verbatim.getPublishingCountry());
168       setProtocol(verbatim.getProtocol());
169       if (verbatim.getLastCrawled() != null) {
170         setLastCrawled(new Date(verbatim.getLastCrawled().getTime()));
171       }
172       if (verbatim.getVerbatimFields() != null) {
173         getVerbatimFields().putAll(verbatim.getVerbatimFields());
174       }
175       if (verbatim.getLastParsed() != null) {
176         setLastParsed(verbatim.getLastParsed());
177       }
178       setExtensions(verbatim.getExtensions());
179     }
180   }
