178   /**
179    * Create occurrence instance from existing verbatim one, copying over all data.
180    */
181   public Occurrence(@Nullable VerbatimOccurrence verbatim) {
182     if (verbatim != null) {
183       setKey(verbatim.getKey());
184       setDatasetKey(verbatim.getDatasetKey());
185       setPublishingOrgKey(verbatim.getPublishingOrgKey());
186       setPublishingCountry(verbatim.getPublishingCountry());
187       setProtocol(verbatim.getProtocol());
188       setCrawlId(verbatim.getCrawlId());
189       if (verbatim.getLastCrawled() != null) {
190         setLastCrawled(new Date(verbatim.getLastCrawled().getTime()));
191       }
192       if (verbatim.getVerbatimFields() != null) {
193         getVerbatimFields().putAll(verbatim.getVerbatimFields());
194       }
195       if (verbatim.getLastParsed() != null) {
196         setLastParsed(verbatim.getLastParsed());
197       }
198       setExtensions(verbatim.getExtensions());
199     }
200   }

1230   /**
1231    * This private method is only for serialization via jackson and not exposed anywhere else!
1232    * It maps the verbatimField terms into properties with their simple name or qualified names for UnknownTerms.
1233    */
1234   @JsonAnyGetter
1235   private Map<String, String> jsonVerbatimFields() {
1236     Map<String, String> extendedProps = new HashMap<>();
1237     for (Map.Entry<Term, String> prop : getVerbatimFields().entrySet()) {
1238       Term t = prop.getKey();
1239       if (t instanceof UnknownTerm || PROPERTIES.contains(t.simpleName())) {
1240         extendedProps.put(t.qualifiedName(), prop.getValue());
1241       } else {
1242         // render all terms in controlled enumerations as simple names only - unless we have a property of that name already!
1243         extendedProps.put(t.simpleName(), prop.getValue());
1244       }
1245     }
1246     return extendedProps;
1247   }
