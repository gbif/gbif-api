230   /**

233   @JsonAnySetter

235     Term t = TermFactory.instance().findTerm(key);

239   /**
240    * This private method is only for serialization via jackson and not exposed anywhere else!
241    * It maps the verbatimField terms into properties with their full qualified name.
242    */
243   @JsonAnyGetter
244   private Map<String, String> jsonVerbatimFields() { // note: for 1.6.0 MUST use non-getter name; otherwise doesn't
245 // matter
246     Map<String, String> extendedProps = Maps.newHashMap();
247     for (Map.Entry<Term, String> prop : verbatimFields.entrySet()) {
248       extendedProps.put(prop.getKey().qualifiedName(), prop.getValue());
249     }
250     return extendedProps;
251   }
