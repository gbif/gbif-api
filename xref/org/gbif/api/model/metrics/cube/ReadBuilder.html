13  package org.gbif.api.model.metrics.cube;
14  
15  import org.gbif.api.vocabulary.Country;
16  import org.gbif.api.vocabulary.Language;
17  
18  import java.util.Map;
19  import java.util.UUID;
20  import javax.annotation.concurrent.NotThreadSafe;
21  
22  import com.google.common.base.Preconditions;
23  import com.google.common.collect.Maps;
24  
25  /**
26   * Provides building of addresses for reading the cube.
27   * This class ensures the type safety of dimensions as they are added to the builder.
28   */
29  @NotThreadSafe
30  public class ReadBuilder {
31  
32    private final Map<Dimension<?>, String> address = Maps.newHashMap();
33  
34    /**
35     * Adds an country type dimension to the address.
36     */
37    public ReadBuilder at(Dimension<Country> dim, Country value) {
38      Preconditions.checkNotNull(value, "Dimension cannot be null");
39      address.put(dim, value.getIso2LetterCode());
40      return this;
41    }
42  
43    /**
44     * Adds an language type dimension to the address.
45     */
46    public ReadBuilder at(Dimension<Language> dim, Language value) {
47      Preconditions.checkNotNull(value, "Dimension cannot be null");
48      address.put(dim, value.getIso2LetterCode());
49      return this;
50    }
51  
52    /**
53     * Adds an enumerated type dimension to the address.
54     */
55    public ReadBuilder at(Dimension<? extends Enum<?>> dim, Enum<?> value) {
56      Preconditions.checkNotNull(value, "Dimension cannot be null");
57      address.put(dim, value.name());
58      return this;
59    }
60  
61    /**
62     * Adds a boolen typed dimension to the address.
63     */
64    public ReadBuilder at(Dimension<Boolean> dim, boolean value) {
65      address.put(dim, String.valueOf(value));
66      return this;
67    }
68  
69    /**
70     * Adds a double typed dimension to the address.
71     */
72    public ReadBuilder at(Dimension<Double> dim, double value) {
73      address.put(dim, String.valueOf(value));
74      return this;
75    }
76  
77    /**
78     * Adds a float typed dimension to the address.
79     */
80    public ReadBuilder at(Dimension<Float> dim, float value) {
81      address.put(dim, String.valueOf(value));
82      return this;
83    }
84  
85    /**
86     * Adds an integer typed dimension to the address.
87     */
88    public ReadBuilder at(Dimension<Integer> dim, int value) {
89      address.put(dim, String.valueOf(value));
90      return this;
91    }
92  
93    /**
94     * Adds an String typed dimension to the address.
95     */
96    public ReadBuilder at(Dimension<String> dim, String value) {
97      Preconditions.checkNotNull(value, "Dimension cannot be null");
98      address.put(dim, value);
99      return this;
100   }
101 
102   /**
103    * Adds a UUID typed dimension to the address.
104    */
105   public ReadBuilder at(Dimension<UUID> dim, UUID value) {
106     Preconditions.checkNotNull(value, "Dimension cannot be null");
107     address.put(dim, value.toString());
108     return this;
109   }
110 
111   /**
112    * @return The built address.
113    */
114   public Map<Dimension<?>, String> build() {
115     return address;
116   }
117 }
