254   /**
255    * Convenience method checking if any spatial validation rule has not passed.
256    * Primarily used to indicate that the record should not be displayed on a map.
257    */
258   @JsonIgnore
259   public boolean hasSpatialIssue() {
260     for (OccurrenceIssue rule : OccurrenceIssue.GEOSPATIAL_RULES) {
261       if (issues.contains(rule)) {
262         return true;
263       }
264     }
265     return false;
266   }

280   /**
281    * This private method is only for serialization via jackson and not exposed anywhere else!
282    * It maps the verbatimField terms into properties with their simple name or qualified names for UnknownTerms.
283    */
284   @JsonAnyGetter
285   private Map<String, String> jsonVerbatimFields() {
286     Map<String, String> extendedProps = new HashMap<>();
287     for (Map.Entry<Term, String> prop : getVerbatimFields().entrySet()) {
288       Term t = prop.getKey();
289       if (t instanceof UnknownTerm || PROPERTIES.contains(t.simpleName())) {
290         extendedProps.put(t.qualifiedName(), prop.getValue());
291       } else {
292         // render all terms in controlled enumerations as simple names only - unless we have a property of that name already!
293         extendedProps.put(t.simpleName(), prop.getValue());
294       }
295     }
296     return extendedProps;
297   }
