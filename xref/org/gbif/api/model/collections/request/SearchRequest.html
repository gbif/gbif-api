16  package org.gbif.api.model.collections.request;
17  
18  import java.io.Serializable;
19  import java.util.HashMap;
20  import java.util.HashSet;
21  import java.util.List;
22  import java.util.Map;
23  import java.util.Set;
24  import java.util.UUID;
25  import javax.annotation.Nullable;
26  import lombok.Builder;
27  import lombok.Data;
28  import lombok.experimental.SuperBuilder;
29  import org.gbif.api.model.common.paging.Pageable;
30  import org.gbif.api.model.common.paging.PagingConstants;
31  import org.gbif.api.model.common.paging.PagingRequest;
32  import org.gbif.api.vocabulary.Country;
33  import org.gbif.api.vocabulary.GbifRegion;
34  import org.gbif.api.vocabulary.IdentifierType;
35  import org.gbif.api.vocabulary.SortOrder;
36  import org.gbif.api.vocabulary.collections.CollectionsFacetParameter;
37  import org.gbif.api.vocabulary.collections.CollectionsSortField;
38  import org.gbif.api.vocabulary.collections.MasterSourceType;
39  import org.gbif.api.vocabulary.collections.Source;
40  
41  @SuperBuilder
42  @Data
43  public abstract class SearchRequest<F extends CollectionsFacetParameter> implements Serializable {
44  
45    @Nullable private Boolean hl;
46    @Nullable private String q;
47    @Nullable private List<String> code;
48    @Nullable private List<String> name;
49    @Nullable private List<String> alternativeCode;
50    @Nullable private List<UUID> contact;
51    @Nullable private List<String> machineTagNamespace;
52    @Nullable private List<String> machineTagName;
53    @Nullable private List<String> machineTagValue;
54    @Nullable private List<IdentifierType> identifierType;
55    @Nullable private List<String> identifier;
56    @Nullable private List<Country> country;
57    @Nullable private List<GbifRegion> gbifRegion;
58    @Nullable private List<String> city;
59    @Nullable private List<String> fuzzyName;
60    @Nullable private List<Boolean> active;
61    @Nullable private List<MasterSourceType> masterSourceType;
62    @Nullable private String numberSpecimens;
63    @Nullable private List<Boolean> displayOnNHCPortal;
64    @Nullable private List<UUID> replacedBy;
65    @Nullable private String occurrenceCount;
66    @Nullable private String typeSpecimenCount;
67    @Nullable private CollectionsSortField sortBy;
68    @Nullable private SortOrder sortOrder;
69    @Nullable private List<UUID> institutionKeys;
70    @Nullable private List<Source> source;
71    @Nullable private List<String> sourceId;
72    @Nullable private Long offset;
73    @Nullable private Integer limit;
74  
75    @Builder.Default @Nullable private Set<F> facets = new HashSet<>();
76    @Nullable private boolean multiSelectFacets;
77    @Nullable private Integer facetMinCount;
78    @Nullable private Integer facetLimit = 10;
79    @Nullable private Integer facetOffset;
80  
81    // Holds the paging configuration for each requested facet
82    @Builder.Default @Nullable private Map<F, Pageable> facetPages = new HashMap<>();
83  
84    public Pageable getPage() {
85      return new PagingRequest(
86          offset != null ? offset : PagingConstants.DEFAULT_PARAM_OFFSET,
87          limit != null ? limit : PagingConstants.DEFAULT_PARAM_LIMIT);
88    }
89  }
