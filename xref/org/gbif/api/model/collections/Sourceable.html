1   package org.gbif.api.model.collections;

3   import java.lang.annotation.Documented;
4   import java.lang.annotation.ElementType;
5   import java.lang.annotation.Repeatable;
6   import java.lang.annotation.Retention;
7   import java.lang.annotation.RetentionPolicy;
8   import java.lang.annotation.Target;
9   
10  import org.gbif.api.vocabulary.collections.MasterSourceType;
11  
12  /**
13   * Annotation to indicate that a field can be populated from external sources.
14   *
15   * <p>This is used in GRSciColl to flag fields whose value comes from other sources rather than
16   * GRSciColl such as IH, datasets or organizations.
17   */
18  @Documented
19  @Retention(RetentionPolicy.RUNTIME)
20  @Target({ElementType.FIELD})
21  @Repeatable(value = Sourceables.class)
22  public @interface Sourceable {
23  
24    MasterSourceType[] masterSources() default {};
25  
26    /**
27     * Indicates that only some parts are sourceable. If this field is not set it's assumed that the
28     * whole field is sourceable.
29     *
30     * <p>This applies only to lists or similar. For example, a list of identifiers where only the DOI
31     * identifiers are sourceable would set this field as {DOI}.
32     */
33    String[] sourceableParts() default {};
34  
35    /**
36     * Indicates that a field is initially filled from an external source but it can be overwritten
37     * afterwards.
38     *
39     * <p>E.g.: the names of IH collections.
40     */
41    boolean overridable() default false;
42  }
