14  package org.gbif.api.model.collections;
15  
16  import org.gbif.api.vocabulary.collections.MasterSourceType;
17  
18  import java.lang.annotation.Documented;
19  import java.lang.annotation.ElementType;
20  import java.lang.annotation.Repeatable;
21  import java.lang.annotation.Retention;
22  import java.lang.annotation.RetentionPolicy;
23  import java.lang.annotation.Target;
24  
25  /**
26   * Annotation to indicate that a field can be populated from external sources.
27   *
28   * <p>This is used in GRSciColl to flag fields whose value comes from other sources rather than
29   * GRSciColl such as IH, datasets or organizations.
30   */
31  @Documented
32  @Retention(RetentionPolicy.RUNTIME)
33  @Target({ElementType.FIELD})
34  @Repeatable(value = Sourceables.class)
35  public @interface Sourceable {
36  
37    MasterSourceType[] masterSources() default {};
38  
39    /**
40     * Indicates that only some parts are sourceable. If this field is not set it's assumed that the
41     * whole field is sourceable.
42     *
43     * <p>This applies only to lists or similar. For example, a list of identifiers where only the DOI
44     * identifiers are sourceable would set this field as {DOI}.
45     */
46    String[] sourceableParts() default {};
47  
48    /**
49     * Indicates that a field is initially filled from an external source, but it can be overwritten
50     * afterwards.
51     *
52     * <p>E.g.: the names of IH collections.
53     */
54    boolean overridable() default false;
55  }
