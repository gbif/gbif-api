10  import javax.validation.constraints.NotBlank;
11  import javax.validation.constraints.NotNull;
12  
13  import lombok.Data;
14  
15  @Data
16  public class DescriptorGroup implements LenientEquals<DescriptorGroup>, Serializable {
17  
18    private long key;
19    @NotBlank private String title;
20    private String description;
21    @NotNull private UUID collectionKey;
22    private Date created;
23    private String createdBy;
24    private Date modified;
25    private String modifiedBy;
26    private Date deleted;
27  
28    @Override
29    public boolean lenientEquals(DescriptorGroup other) {
30      if (this == other) {
31        return true;
32      }
33  
34      return key == other.key
35          && Objects.equals(title, other.title)
36          && Objects.equals(description, other.description)
37          && Objects.equals(collectionKey, other.collectionKey);
38    }
39  }
