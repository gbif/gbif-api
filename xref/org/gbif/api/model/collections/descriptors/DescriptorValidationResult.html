3   import java.util.ArrayList;
4   import java.util.List;
5   
6   import lombok.Builder;
7   import lombok.Data;
8   
9   /**
10   * Result class for descriptor vocabulary validation that allows graceful handling
11   * of invalid values by leaving fields blank rather than failing the entire operation.
12   */
13  @Data
14  @Builder
15  public class DescriptorValidationResult {
16    private String validBiomeType;
17    private String validObjectClassification;
18    @Builder.Default
19    private List<String> issues = new ArrayList<>();
20  
21    public void addIssue(String issue) {
22      if (issues == null) {
23        issues = new ArrayList<>();
24      }
25      issues.add(issue);
26    }
27  
28    public boolean hasIssues() {
29      return issues != null && !issues.isEmpty();
30    }
31  }
