339   @Override
340   public boolean equals(Object o) {
341     if (this == o) return true;
342     if (o == null || getClass() != o.getClass()) return false;
343     Collection that = (Collection) o;
344     return active == that.active
345            && personalCollection == that.personalCollection
346            && Objects.equals(key, that.key)
347            && Objects.equals(code, that.code)
348            && Objects.equals(name, that.name)
349            && Objects.equals(description, that.description)
350            && Objects.equals(contentTypes, that.contentTypes)
351            && Objects.equals(doi, that.doi)
352            && Objects.equals(homepage, that.homepage)
353            && Objects.equals(catalogUrl, that.catalogUrl)
354            && Objects.equals(apiUrl, that.apiUrl)
355            && Objects.equals(preservationTypes, that.preservationTypes)
356            && accessionStatus == that.accessionStatus
357            && Objects.equals(institutionKey, that.institutionKey)
358            && Objects.equals(mailingAddress, that.mailingAddress)
359            && Objects.equals(address, that.address)
360            && Objects.equals(createdBy, that.createdBy)
361            && Objects.equals(modifiedBy, that.modifiedBy)
362            && Objects.equals(created, that.created)
363            && Objects.equals(modified, that.modified)
364            && Objects.equals(deleted, that.deleted)
365            && Objects.equals(tags, that.tags)
366            && Objects.equals(identifiers, that.identifiers)
367            && Objects.equals(contacts, that.contacts);
368   }

427   @Override
428   public boolean lenientEquals(Collection other) {
429     if (this == other) {
430       return true;
431     }
432     return active == other.active
433            && personalCollection == other.personalCollection
434            && Objects.equals(key, other.key)
435            && Objects.equals(code, other.code)
436            && Objects.equals(name, other.name)
437            && Objects.equals(description, other.description)
438            && Objects.equals(contentTypes, other.contentTypes)
439            && Objects.equals(doi, other.doi)
440            && Objects.equals(homepage, other.homepage)
441            && Objects.equals(catalogUrl, other.catalogUrl)
442            && Objects.equals(apiUrl, other.apiUrl)
443            && Objects.equals(preservationTypes, other.preservationTypes)
444            && accessionStatus == other.accessionStatus
445            && Objects.equals(institutionKey, other.institutionKey)
446            && Objects.equals(mailingAddress, other.mailingAddress)
447            && Objects.equals(address, other.address)
448            && Objects.equals(deleted, other.deleted);
