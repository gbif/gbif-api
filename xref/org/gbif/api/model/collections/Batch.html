7   import com.fasterxml.jackson.annotation.JsonIgnore;

9   import io.swagger.v3.oas.annotations.Hidden;
10  import io.swagger.v3.oas.annotations.media.Schema;
11  import lombok.Data;
12  
13  /** A batch that can contain either institutions or collections to create or update them in bulk. */
14  @Data
15  public class Batch {
16  
17    @Schema(description = "Unique key for the batch.", accessMode = Schema.AccessMode.READ_ONLY)
18    private Integer key;
19  
20    @Schema(description = "State that indicates if the batch is being processed, finished or failed.")
21    private State state;
22  
23    @Schema(description = "Indicates if the batch is for institutions or for collections.")
24    private CollectionEntityType entityType;
25  
26    @Hidden @JsonIgnore private String resultFilePath;
27  
28    @Schema(description = "Errors found that made the batch fail.")
29    private List<String> errors = new ArrayList<>();
30  
31    @Schema(
32        description = "Timestamp of when the batch was created in the GBIF registry.",
33        accessMode = Schema.AccessMode.READ_ONLY)
34    private Date created;
35  
36    @Schema(
37        description = "The GBIF username of the creator of the batch in the GBIF registry.",
38        accessMode = Schema.AccessMode.READ_ONLY)
39    private String createdBy;
40  
41    public enum State {
42      IN_PROGRESS,
43      FAILED,
44      FINISHED;
45    }
46  }
