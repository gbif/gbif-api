16  package org.gbif.api.model.registry;
17  
18  import java.io.Serializable;
19  import java.util.Date;
20  import java.util.Objects;
21  import java.util.StringJoiner;
22  
23  import javax.validation.constraints.NotNull;
24  import javax.validation.constraints.Null;
25  import javax.validation.constraints.Size;
26  
27  public class Comment implements Serializable, LenientEquals<Comment> {
28  
29    private Integer key;
30    private String content;
31    private String createdBy;
32    private String modifiedBy;
33    private Date created;
34    private Date modified;
35  
36    @Null(groups = {PrePersist.class})
37    @NotNull(groups = {PostPersist.class})
38    public Integer getKey() {
39      return key;
40    }
41  
42    public void setKey(Integer key) {
43      this.key = key;
44    }
45  
46    @NotNull
47    @Size(min = 1)
48    public String getContent() {
49      return content;
50    }
51  
52    public void setContent(String content) {
53      this.content = content;
54    }
55  
56    @Size(min = 3)
57    public String getCreatedBy() {
58      return createdBy;
59    }
60  
61    public void setCreatedBy(String createdBy) {
62      this.createdBy = createdBy;
63    }
64  
65    @Size(min = 3)
66    public String getModifiedBy() {
67      return modifiedBy;
68    }
69  
70    public void setModifiedBy(String modifiedBy) {
71      this.modifiedBy = modifiedBy;
72    }
73  
74    @Null(groups = {PrePersist.class})
75    @NotNull(groups = {PostPersist.class})
76    public Date getCreated() {
77      return created;
78    }
79  
80    public void setCreated(Date created) {
81      this.created = created;
82    }
83  
84    @Null(groups = {PrePersist.class})
85    @NotNull(groups = {PostPersist.class})
86    public Date getModified() {
87      return modified;
88    }
89  
90    public void setModified(Date modified) {
91      this.modified = modified;
92    }
93  
94    @Override
95    public boolean equals(Object o) {
96      if (this == o) {
97        return true;
98      }
99      if (o == null || getClass() != o.getClass()) {
100       return false;
101     }
102     Comment comment = (Comment) o;
103     return Objects.equals(key, comment.key)
104         && Objects.equals(content, comment.content)
105         && Objects.equals(createdBy, comment.createdBy)
106         && Objects.equals(modifiedBy, comment.modifiedBy)
107         && Objects.equals(created, comment.created)
108         && Objects.equals(modified, comment.modified);
109   }
110 
111   @Override
112   public int hashCode() {
113     return Objects.hash(key, content, createdBy, modifiedBy, created, modified);
114   }
115 
116   @Override
117   public String toString() {
118     return new StringJoiner(", ", Comment.class.getSimpleName() + "[", "]")
119         .add("key=" + key)
120         .add("content='" + content + "'")
121         .add("createdBy='" + createdBy + "'")
122         .add("modifiedBy='" + modifiedBy + "'")
123         .add("created=" + created)
124         .add("modified=" + modified)
125         .toString();
126   }
127 
128   /**
129    * A lenient equality check ignoring server side fields. If the objects are equal or have the same content, they are
130    * considered the same.
131    */
132   @Override
133   public boolean lenientEquals(Comment other) {
134     if (this == other) {
135       return true;
136     }
137     return Objects.equals(this.content, other.content);
138   }
139 }
