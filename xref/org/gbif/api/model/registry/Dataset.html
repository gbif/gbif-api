46  /**
47   * A GBIF dataset which provides occurrence data, checklist data or metadata.
48   * This Dataset class is covering all of the GBIF metadata profile, but only few properties are kept in the
49   * database table:
50   * <ul>
51   * <li>key</li>
52   * <li>parentDatasetKey</li>
53   * <li>duplicateOfDatasetKey</li>
54   * <li>installationKey</li>
55   * <li>publishingOrganizationKey</li>
56   * <li>external</li>
57   * <li>numConstituents</li>
58   * <li>type</li>
59   * <li>subtype</li>
60   * <li>title</li>
61   * <li>alias</li>
62   * <li>abbreviation</li>
63   * <li>description</li>
64   * <li>language</li>

754   @Override
755   public boolean lenientEquals(Dataset other) {
756     if (this == other) {
757       return true;
758     }
759     return Objects.equal(this.parentDatasetKey, other.parentDatasetKey)
760       && Objects.equal(this.duplicateOfDatasetKey, other.duplicateOfDatasetKey)
761       && Objects.equal(this.installationKey, other.installationKey)
762       && Objects.equal(this.publishingOrganizationKey, other.publishingOrganizationKey)
763       && Objects.equal(this.doi, other.doi)
764       && Objects.equal(this.external, other.external)
765       && Objects.equal(this.type, other.type)
766       && Objects.equal(this.subtype, other.subtype)
767       && Objects.equal(this.title, other.title)
768       && Objects.equal(this.alias, other.alias)
