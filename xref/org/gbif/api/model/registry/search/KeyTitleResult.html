3   import java.util.UUID;
4   
5   import com.google.common.base.Objects;
6   
7   /**
8    * A utility container for holding only the title and key of an entity.
9    */
10  public class KeyTitleResult {
11  
12    private UUID key;
13    private String title;
14  
15    public KeyTitleResult() {
16    }
17  
18    public KeyTitleResult(UUID key, String title) {
19      this.key = key;
20      this.title = title;
21    }
22  
23    public UUID getKey() {
24      return key;
25    }
26  
27    public void setKey(UUID key) {
28      this.key = key;
29    }
30  
31    public String getTitle() {
32      return title;
33    }
34  
35    public void setTitle(String title) {
36      this.title = title;
37    }
38  
39    @Override
40    public int hashCode() {
41      return Objects
42        .hashCode(key, title);
43    }
44  
45    @Override
46    public boolean equals(Object object) {
47      if (object instanceof KeyTitleResult) {
48        KeyTitleResult that = (KeyTitleResult) object;
49        return Objects.equal(this.key, that.key)
50          && Objects.equal(this.title, that.title);
51      }
52      return false;
53    }
54  
55    @Override
56    public String toString() {
57      return Objects.toStringHelper(this)
58        .add("key", key)
59        .add("title", title)
60        .toString();
61    }
62  }
