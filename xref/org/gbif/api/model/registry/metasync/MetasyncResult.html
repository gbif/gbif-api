3   /**
4    * Enumeration of all the different categories of results we want to report on during metadata synchronisation.
5    */
6   public enum MetasyncResult {
7   
8     /**
9      * The synchronization was successful.
10     */
11    OK,
12  
13    /**
14     * Is any kind of error establishing a connection or during a connection that's usually on the network level
15     * (connection refused, timeouts etc.).
16     */
17    IO_EXCEPTION,
18  
19    /**
20     * Any kind of HTTP error (e.g. a non 200 response code).
21     */
22    HTTP_ERROR,
23  
24    /**
25     * This means that we got a reply from the endpoint but it does not conform to what we expected (e.g. HTML instead of
26     * XML).
27     */
28    PROTOCOL_ERROR,
29  
30    /**
31     * Anything that doesn't fit in the former categories (e.g. invalid URI stored in our Registry).
32     */
33    OTHER_ERROR
34  }
