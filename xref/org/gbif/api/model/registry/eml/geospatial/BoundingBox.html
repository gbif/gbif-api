16  package org.gbif.api.model.registry.eml.geospatial;

18  import java.io.Serializable;
19  import java.util.Locale;

21  
22  /**
23   * Geographic coverage geometry in the form of 4 sided bounding box.
24   */
25  public class BoundingBox implements Serializable, Geometry {
26  
27    private static final long serialVersionUID = 9101833933868378174L;
28  
29    private static final int MIN_MIN_LONGITUDE = -180;
30    private static final int MAX_MAX_LONGITUDE = 180;
31  
32    private static final int MIN_MIN_LATITUDE = -90;
33    private static final int MAX_MAX_LATITUDE = 90;
34  
35    private double minLatitude;
36    private double maxLatitude;
37    private double minLongitude;
38    private double maxLongitude;
39    private boolean isGlobalCoverage;
40  
41  
42    public BoundingBox() {
43    }
44  
45    public BoundingBox(double minLatitude, double maxLatitude, double minLongitude, double maxLongitude) {
46      this.minLatitude = minLatitude;
47      this.maxLatitude = maxLatitude;
48      this.minLongitude = minLongitude;
49      this.maxLongitude = maxLongitude;
50      this.isGlobalCoverage = checkIsGlobalCoverage(minLatitude, maxLatitude, minLongitude, maxLongitude);
51    }
52  
53    public double getMaxLatitude() {
54      return maxLatitude;
55    }
56  
57    public void setMaxLatitude(double maxLatitude) {
58      this.maxLatitude = maxLatitude;
59    }
60  
61    public double getMaxLongitude() {
62      return maxLongitude;
63    }
64  
65    public void setMaxLongitude(double maxLongitude) {
66      this.maxLongitude = maxLongitude;
67    }
68  
69    public double getMinLatitude() {
70      return minLatitude;
71    }
72  
73    public void setMinLatitude(double minLatitude) {
74      this.minLatitude = minLatitude;
75    }
76  
77    public double getMinLongitude() {
78      return minLongitude;
79    }
80  
81    public void setMinLongitude(double minLongitude) {
82      this.minLongitude = minLongitude;
83    }
84  
85    @Override
86    public String toWellKnownText() {
87      return String.format(Locale.ENGLISH, "POLYGON ((%f %f, %f %f, %f %f, %f %f))",

96    }

98    /**
99     * @return whether the BoundingBox represents global coverage
100    */
101   public boolean isGlobalCoverage() {
102     return isGlobalCoverage;
103   }
104 
105   /**
106    * Set whether this binding box represents global coverage or not. Setter ensures all 4 required params are provided.
107    *
108    * @param minLatitude  min latitude
109    * @param maxLatitude  max latitude
110    * @param minLongitude min longitude
111    * @param maxLongitude max longitude
112    */
113   public void setGlobalCoverage(double minLatitude, double maxLatitude, double minLongitude, double maxLongitude) {
114     isGlobalCoverage = checkIsGlobalCoverage(minLatitude, maxLatitude, minLongitude, maxLongitude);
115   }
116 
117   /**
118    * Determine whether the bounding box represents global coverage. This is the case, when all North, East, South,
119    * and West lines represent the maximum possible degree values (90, 180, -90, -180).
120    *
121    * @param minLatitude  min latitude
122    * @param maxLatitude  max latitude
123    * @param minLongitude min longitude
124    * @param maxLongitude max longitude
125    *
126    * @return whether the bounding box represents global coverage
127    */
128   private boolean checkIsGlobalCoverage(
129     double minLatitude, double maxLatitude, double minLongitude, double maxLongitude
130   ) {
131     return minLatitude == MIN_MIN_LATITUDE
132            && maxLatitude == MAX_MAX_LATITUDE
133            && minLongitude == MIN_MIN_LONGITUDE
134            && maxLongitude == MAX_MAX_LONGITUDE;
135   }
