10  /**
11   * Class representing a single Digital Object Identifier (DOI) breaking it down to a prefix and suffix.
12   * For the syntax of DOI names see the <a href="http://www.doi.org/doi_handbook/2_Numbering.html#2.2">DOI Handbook</a>.
13   * All parsing is case insensitive and resulting components will all be upper cased.
14   */
15  public class DOI {
16    private static final Pattern HTTP = Pattern.compile("^https?://(dx\\.)?doi\\.org/"
17                                                        + "(urn:)?(doi:)?", Pattern.CASE_INSENSITIVE);
18    private static final Pattern PARSER = Pattern.compile("^(?:urn:)?(?:doi:)?"           // optional
19                                                          + "(10(?:\\.[0-9]+)+)"
20                                                          + "/(.+)$", Pattern.CASE_INSENSITIVE);
21  
22    private static String RESOLVER = "http://doi.org/";
23    private static String SCHEME = "doi:";
24    private String prefix;
25    private String suffix;
26  
27    /**
28     * Parses a simple DOI string of various forms incl URN, URL or plain DOI names.
29     * @param doi the full simple DOI string
30     * @throws java.lang.IllegalArgumentException if invalid DOI string is passed
31     */
32    public DOI(String doi) {
33      Preconditions.checkNotNull("DOI required", doi);
34      Matcher m = PARSER.matcher(decodeUrl(doi));
35      if (m.find()) {
36        this.prefix = m.group(1).toUpperCase();
37        this.suffix = m.group(2).toUpperCase();
38      } else {
39        throw new IllegalArgumentException(doi + " is not a valid DOI");
40      }
41    }
42  
43    /**
44     * Parses a simple DOI string of various forms incl URN, URL or plain DOI names.
45     * @param prefix a simple DOI prefix starting with 10.
46     * @param suffix arbitrary suffix part of the DOI
47     * @throws java.lang.IllegalArgumentException if invalid DOI prefix is given
48     */
49    public DOI(String prefix, String suffix) {
50      this.prefix = Preconditions.checkNotNull("DOI prefix required", prefix).toUpperCase();
51      Preconditions.checkArgument(prefix.startsWith("10."));
52      this.suffix = Preconditions.checkNotNull("DOI suffix required", suffix).toUpperCase();
53    }
54  
55    /**
56     * If the doi is encoded as a URL this method strips the resolver and decodes the URL encoded string entities.
57     */
58    private static String decodeUrl(String doi) {
59      Matcher m = HTTP.matcher(doi);
60      if (m.find()) {
61        // strip resolver incl potentially starting paths using a badly encoded urn:doi
62        // (the colon would need to be encoded in a proper URL)
63        doi = m.replaceFirst("");
64        // now decode the URL path, we cannot possibly have query parameters or anchors as the DOIs encoded as a URL
65        // will just be a path
66        return URI.create(doi).getPath();
67      }
68      return doi;
69    }

72      return prefix;
73    }

76      this.prefix = prefix;
77    }

80      return suffix;
81    }
82  
83    public void setSuffix(String suffix) {
84      this.suffix = suffix;
85    }
86  
87    /**
88     * See <a href="http://www.doi.org/doi_handbook/2_Numbering.html#2.6">DOI Hanbook, Visual presentation and other representation of DOI names</a>.
89     * @return the resolved DOI using http://dx.doi.org
90     */
91    public URI getUrl() {
92      return URI.create(RESOLVER + prefix + '/' + suffix);
93    }
94  
95    @Override
96    public String toString() {
97      return SCHEME + prefix + '/' + suffix;
98    }
