20  /**
21   * Class representing a single Digital Object Identifier (DOI) breaking it down to a prefix and suffix.
22   * For the syntax of DOI names see the <a href="http://www.doi.org/doi_handbook/2_Numbering.html#2.2">DOI Handbook</a>.
23   * All parsing is case insensitive and resulting components will all be upper cased.
24   */
25  @JsonSerialize(using = DOI.Serializer.class)
26  @JsonDeserialize(using = DOI.Deserializer.class)
27  public class DOI {
28    public static final String GBIF_PREFIX = "10.15468";
29    private static final Pattern HTTP = Pattern.compile("^https?://(dx\\.)?doi\\.org/"
30                                                        + "(urn:)?(doi:)?", Pattern.CASE_INSENSITIVE);
31    private static final Pattern PARSER = Pattern.compile("^(?:urn:)?(?:doi:)?"           // optional
32                                                          + "(10(?:\\.[0-9]+)+)"
33                                                          + "/(.+)$", Pattern.CASE_INSENSITIVE);
34  
35    private static String RESOLVER = "http://doi.org/";
36    private static String SCHEME = "doi:";
37    private String prefix;
38    private String suffix;

40    /**
41     * Returns true only if the source can be parsed into a DOI.
42     */
43    public static boolean isParsable(String source) {
44      if (!Strings.isNullOrEmpty(source)) {
45        return PARSER.matcher(decodeUrl(source)).find();
46      }
47      return false;
48    }

57    /**
58     * Parses a simple DOI string of various forms incl URN, URL or plain DOI names.
59     * @param doi the full simple DOI string
60     * @throws java.lang.IllegalArgumentException if invalid DOI string is passed
61     */
62    public DOI(String doi) {
63      Preconditions.checkNotNull(doi, "DOI required");
64      Matcher m = PARSER.matcher(decodeUrl(doi));
65      if (m.find()) {
66        this.prefix = m.group(1).toUpperCase();
67        this.suffix = m.group(2).toUpperCase();
68      } else {
69        throw new IllegalArgumentException(doi + " is not a valid DOI");
70      }
71    }
72  
73    /**
74     * Parses a simple DOI string of various forms incl URN, URL or plain DOI names.
75     * @param prefix a simple DOI prefix starting with 10.
76     * @param suffix arbitrary suffix part of the DOI
77     * @throws java.lang.IllegalArgumentException if invalid DOI prefix is given
78     */
79    public DOI(String prefix, String suffix) {
80      this.prefix = Preconditions.checkNotNull(prefix, "DOI prefix required").toUpperCase();
81      Preconditions.checkArgument(prefix.startsWith("10."));
82      this.suffix = Preconditions.checkNotNull(suffix, "DOI suffix required").toUpperCase();
83    }
84  
85    /**
86     * If the doi is encoded as a URL this method strips the resolver and decodes the URL encoded string entities.
87     */
88    private static String decodeUrl(String doi) {
89      Matcher m = HTTP.matcher(doi);
90      if (m.find()) {
91        // strip resolver incl potentially starting paths using a badly encoded urn:doi
92        // (the colon would need to be encoded in a proper URL)
93        doi = m.replaceFirst("");
94        // now decode the URL path, we cannot possibly have query parameters or anchors as the DOIs encoded as a URL
95        // will just be a path
96        return URI.create(doi).getPath();
97      }
98      return doi;
99    }
