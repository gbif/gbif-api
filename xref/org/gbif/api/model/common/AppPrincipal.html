1   package org.gbif.api.model.common;
2   
3   import java.util.Objects;
4   import javax.security.auth.Subject;
5   
6   /**
7    * Similar to {@link GbifUserPrincipal} but represents an application instead of a user.
8    * The appKey is used as the unique account name and is exposed as the principal name.
9    */
10  public class AppPrincipal implements ExtendedPrincipal {
11  
12    private final String appKey;
13    private final String appRole;
14  
15    /**
16     * {@link AppPrincipal} constructor.
17     *
18     * @param appKey  mandatory, appKey of the application that is now authenticated.
19     * @param appRole optionally, the "role" of the app as {@link String}. Mostly to use jsr250 @RolesAllowed.
20     */
21    public AppPrincipal(String appKey, String appRole) {
22      Objects.requireNonNull(appKey, "appKey shall be provided");
23      this.appKey = appKey;
24      this.appRole = appRole;
25    }
26  
27    @Override
28    public String getName() {
29      return appKey;
30    }
31  
32    @Override
33    public boolean implies(Subject subject) {
34      return false;
35    }
36  
37    @Override
38    public boolean hasRole(String role) {
39      return appRole != null && appRole.equalsIgnoreCase(role);
40    }
41  }
