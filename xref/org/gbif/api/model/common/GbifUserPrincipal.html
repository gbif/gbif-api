5   import java.util.Objects;

7   import com.google.common.base.Strings;

9   /**
10   * A wrapper class for a GBIF User that exposes the unique account name as the principal name.
11   * Replacement for {@link UserPrincipal}
12   */
13  public class GbifUserPrincipal implements ExtendedPrincipal {
14    private final GbifUser user;
15  
16    public GbifUserPrincipal(GbifUser user) {
17      Objects.requireNonNull(user, "user shall be provided");
18      this.user = user;
19    }
20  
21    @Override
22    public String getName() {
23      return user.getUserName();
24    }
25  
26    public GbifUser getUser() {
27      return user;
28    }
29  
30    /**
31     * Checks if the user has the given string based role.
32     * We use strings here and not the enum to facilitate the use of the method with the standard SecurityContext
33     * which uses Strings for roles.
34     *
35     * @param role case insensitive role
36     *
37     * @return true if the user has the requested role
38     */
39    public boolean hasRole(String role) {
40      if (!Strings.isNullOrEmpty(role)) {
41        try {
42          UserRole r = UserRole.valueOf(role.toUpperCase());
43          return user.hasRole(r);
44        } catch (IllegalArgumentException e) {
45          // ignore
46        }
47      }
48      return false;
49    }
50  }
