3   /**
4    * This enumeration represents the status of a DOI identifier.
5    * It represents both EZID and DataCite DOIs.
6    */
7   public enum DoiStatus {
8     /**
9      * The identifier is known only to the DOI registration agency.
10     * This status may be used to reserve an identifier name without advertising the identifier's existence
11     * to resolvers and other external services.
12     * A reserved identifier may be fully deleted.
13     */
14    RESERVED("reserved"),
15  
16    /**
17     * A public, registered DOI.
18     * It's target URL is known to public resolvers and other external services.
19     * It may be marked as DELETED in the future, but never again as RESERVED.
20     */
21    REGISTERED("public"),
22  
23    /**
24     * The identifier once was registered, but the object referenced by the identifier is not available.
25     * This is known as an "inactive" DOI in DataCite and "unavailable" in EZID.
26     * <br/>
27     * In EZID the identifier redirects to an EZID-provided "tombstone" page regardless of its target URL.
28     * In DataCite the original target URL is still available.
29     */
30    DELETED("unavailable"),
31  
32    /**
33     * A failed DOI status indicates we could not communicate with DataCite cause we had invalid metadata.
34     * This DOI then requires a manual cleanup.
35     */
36    FAILED("failed");
37
