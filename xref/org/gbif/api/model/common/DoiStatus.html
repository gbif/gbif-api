3   /**
4    * This enumeration represents the status of a DOI identifier.
5    * It represents both EZID and DataCite DOIs.
6    */
7   public enum DoiStatus {
8      /**
9      * A NEW DOI status indicates the DOI has been minted in GBIF only and has not yet been passed on to DataCite.
10     */
11    NEW(null),
12  
13    /**
14     * The identifier is known only to the DOI registration agency.
15     * This status may be used to reserve an identifier name without advertising the identifier's existence
16     * to resolvers and other external services.
17     * A reserved identifier may be fully deleted.
18     */
19    RESERVED("reserved"),
20  
21    /**
22     * A public, registered DOI.
23     * It's target URL is known to public resolvers and other external services.
24     * It may be marked as DELETED in the future, but never again as RESERVED.
25     */
26    REGISTERED("public"),
27  
28    /**
29     * The identifier once was registered, but the object referenced by the identifier is not available.
30     * This is known as an "inactive" DOI in DataCite and "unavailable" in EZID.
31     * <br/>
32     * In EZID the identifier redirects to an EZID-provided "tombstone" page regardless of its target URL.
33     * In DataCite the original target URL is still available.
34     */
35    DELETED("unavailable"),
36  
37    /**
38     * A failed DOI status indicates we could not communicate with DataCite cause we had invalid metadata.
39     * This DOI then requires a manual cleanup. The status is GBIF internal only!
40     */
41    FAILED(null);
42  
43    private final String ezid;
44  
45    DoiStatus(String ezid) {
46      this.ezid = ezid;
47    }
48  
49    /**
50     * @return the identifier status value used in EZID.
51     */
52    public String getEzid() {
53      return ezid;
54    }
55  
56    /**
57     * @return true if the identifier is registered
58     */
59    public boolean isRegistered() {
60      return this == REGISTERED || this == DELETED;
61    }
62  
63    public static DoiStatus fromString(String status) {
64      for (DoiStatus s : DoiStatus.values()) {
65        if (status.equalsIgnoreCase(s.name()) || (s.getEzid() != null && status.equalsIgnoreCase(s.getEzid()))) {
66          return s;
67        }
68      }
69      return null;
70    }
71  }
