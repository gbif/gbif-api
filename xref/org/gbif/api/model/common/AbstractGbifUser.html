16  package org.gbif.api.model.common;
17  
18  import org.gbif.api.vocabulary.UserRole;
19  
20  import java.util.Date;
21  import java.util.HashMap;
22  import java.util.HashSet;
23  import java.util.Locale;
24  import java.util.Map;
25  import java.util.Objects;
26  import java.util.Set;
27  import java.util.StringJoiner;
28  import java.util.stream.Collectors;
29  import java.util.stream.Stream;
30  
31  import javax.validation.constraints.NotNull;
32  import javax.validation.constraints.Pattern;
33  import javax.validation.constraints.Size;
34  
35  import com.fasterxml.jackson.annotation.JsonIgnore;
36  
37  /**
38   * An abstract GBIF user account.
39   * The main purpose of this abstraction is to let subclasses handle key and password information only if required.
40   * By doing so, it is possible to have classes working for user information without having to carry those
41   * information around.
42   */
43  @SuppressWarnings("unused")
44  public abstract class AbstractGbifUser {
45    protected static final String EMAIL_PATTERN =
46      "^[_A-Za-z0-9-+]+(\\.[_A-Za-z0-9-]+)*@[A-Za-z0-9-]+(\\.[A-Za-z0-9-]+)*(\\.[A-Za-z]{2,})$";
47  
48    protected String userName;
49    protected String firstName;
50    protected String lastName;
51    protected String email;
52    protected Set<UserRole> roles = new HashSet<>();
53    protected Map<String, String> settings = new HashMap<>();
54    //settings that the user will not set directly
55    protected Map<String, String> systemSettings = new HashMap<>();
56    protected Date deleted;
57    protected Locale locale;
58  
59    @NotNull
60    @Pattern(regexp = EMAIL_PATTERN)
61    public String getEmail() {
62      return email;
63    }
64  
65    public void setEmail(String email) {
66      this.email = email;
67    }
68  
69    /**
70     * The unique, immutable drupal user account name.
71     * This name should be used for referring to a user.
72     * The account name is made of ASCII lower case alphanumerics, underscore, dash or dots and is in particular void
73     * of whitespace.
74     */
75    @NotNull
76    @Pattern(regexp = "^[a-z0-9_.-]+$")
77    @Size(min = 3, max = 64)
78    public String getUserName() {
79      return userName;
80    }
81  
82    public void setUserName(String userName) {
83      this.userName = userName;
84    }
85  
86    /**
87     * @return the first name of a person
88     */
89    public String getFirstName() {
90      return firstName;
91    }
92  
93    public void setFirstName(String firstName) {
94      this.firstName = firstName;
95    }
96  
97    /**
98     * @return the last name of the user
99     */
100   public String getLastName() {
101     return lastName;
102   }
103 
104   public void setLastName(String lastName) {
105     this.lastName = lastName;
106   }
107 
108   /**
109    * @return the first and last name of the user concatenated with a space
110    */
111   @JsonIgnore
112   public String getName() {
113     return Stream.of(firstName, lastName)
114       .filter(Objects::nonNull)
115       .collect(Collectors.joining(" "));
116   }
117 
118   @NotNull
119   public Set<UserRole> getRoles() {
120     return roles;
121   }
122 
123   public void setRoles(Set<UserRole> roles) {
124     this.roles = roles;
125   }
126 
127   public void addRole(UserRole role) {
128     roles.add(role);
129   }
130 
131   /**
132    * Checks if the user has the given user role.
133    *
134    * @param role user role
135    * @return true if the user has the requested role
136    */
137   public boolean hasRole(UserRole role) {
138     return role != null && roles.contains(role);
139   }
140 
141   /**
142    * Sets the settings object, setting an empty map if null is provided.
143    */
144   public void setSettings(Map<String, String> settings) {
145     // safeguard against misuse to avoid NPE
146     this.settings = settings == null ? new HashMap<>() : settings;
147   }
148 
149   /**
150    * Gets the settings which may be empty but never null.
151    */
152   @NotNull
153   public Map<String, String> getSettings() {
154     return settings;
155   }
156 
157   /**
158    * Sets the settings object, setting an empty map if null is provided.
159    */
160   public void setSystemSettings(Map<String, String> systemSettings) {
161     // safeguard against misuse to avoid NPE
162     this.systemSettings = systemSettings == null ? new HashMap<>() : systemSettings;
163   }
164 
165   /**
166    * Gets the settings which may be empty but never null.
167    */
168   @NotNull
169   public Map<String, String> getSystemSettings() {
170     return systemSettings;
171   }
172 
173   public Date getDeleted() {
174     return deleted;
175   }
176 
177   public void setDeleted(Date deleted) {
178     this.deleted = deleted;
179   }
180 
181   public Locale getLocale() {
182     return locale;
183   }
184 
185   public void setLocale(Locale locale) {
186     this.locale = locale;
187   }
188 
189   @Override
190   public boolean equals(Object o) {
191     if (this == o) {
192       return true;
193     }
194     if (o == null || getClass() != o.getClass()) {
195       return false;
196     }
197     AbstractGbifUser that = (AbstractGbifUser) o;
198     return Objects.equals(userName, that.userName) &&
199       Objects.equals(firstName, that.firstName) &&
200       Objects.equals(lastName, that.lastName) &&
201       Objects.equals(email, that.email) &&
202       Objects.equals(roles, that.roles) &&
203       Objects.equals(settings, that.settings) &&
204       Objects.equals(systemSettings, that.systemSettings) &&
205       Objects.equals(deleted, that.deleted) &&
206       Objects.equals(locale, that.locale);
207   }
208 
209   @Override
210   public int hashCode() {
211     return Objects
212       .hash(userName, firstName, lastName, email, roles, settings, systemSettings, deleted, locale);
213   }
214 
215   @Override
216   public String toString() {
217     return new StringJoiner(", ", AbstractGbifUser.class.getSimpleName() + "[", "]")
218       .add("userName='" + userName + "'")
219       .add("firstName='" + firstName + "'")
220       .add("lastName='" + lastName + "'")
221       .add("email='" + email + "'")
222       .add("roles=" + roles)
223       .add("settings=" + settings)
224       .add("systemSettings=" + systemSettings)
225       .add("deleted=" + deleted)
226       .add("language=" + locale)
227       .toString();
228   }
229 }
