3   /**
4    * This enum lists the reasons why a crawl is finished.
5    */
6   public enum FinishReason {
7     /**
8      * This status means that we were able to finish the crawl without any fatal errors and without exhausting retries.
9      * There could still have been intermittent errors but we managed to reach the end of an endpoint or a DwC-A file.
10     */
11    NORMAL,
12  
13    /**
14     * This status means that the user requested an abort of the crawl and that is the reason this crawl is done.
15     */
16    USER_ABORT,
17  
18    /**
19     * This status means that we had to abort the crawl for any abnormal reason (endpoint down, not a valid archive,
20     * ...). To find the exact reason(s) you need to look at the logs to see past errors.
21     */
22    ABORT,
23  
24    /**
25     * Crawling has stopped because the source data was not modified since the last crawl.
26     */
27    NOT_MODIFIED,
28  
29    /**
30     * This status means that we don't know why the crawl aborted. This is a sign of a programming error and should be
31     * investigated.
32     */
33    UNKNOWN
34  }
