26  /**
27   * The rules followed here should match the document at:
28   * http://dev.gbif.org/wiki/display/INT/Identifier+problems+and+how+to+solve+them.
29   */
30  public class OccurrenceValidationReport {
31    // if the percentage of invalid triplets (eg missing catalog number) is greater than this, the archive is invalid
32    private static final double INVALID_TRIPLET_THRESHOLD = 0.25;
33  
34    // the number of occurrence records checked in the validation
35    private final int checkedRecords;
36  
37    // the number of triplets that were unique
38    private final int uniqueTriplets;
39  
40    /**
41     * the number of triplets that were invalid (because one or more of institutionCode, collectionCode or catalogNumber
42     * were null or empty)
43     */
44    private final int recordsWithInvalidTriplets;
45  
46    // the number of occurrenceIds that were unique (therefore also == the number of records with unique occurrenceId)
47    private final int uniqueOccurrenceIds;
48  
49    // records that had no occurrenceId
50    private final int recordsMissingOccurrenceId;
51  
52    // false if we had to stop at our memory-saving limit
53    private final boolean allRecordsChecked;
54  
55    // if the archive is not valid this will hold a readable reason
56    private String invalidationReason;
57  
58    // is this archive valid
59    private final boolean valid;
60  
61    @JsonCreator
62    public OccurrenceValidationReport(@JsonProperty("checkedRecords") int checkedRecords,
63                                      @JsonProperty("uniqueTriplets") int uniqueTriplets,
64                                      @JsonProperty("invalidTriplets") int recordsWithInvalidTriplets,
65                                      @JsonProperty("uniqueOccIds") int uniqueOccurrenceIds,
66                                      @JsonProperty("missingOccIds") int recordsMissingOccurrenceId,
67                                      @JsonProperty("allRecordsChecked") boolean allRecordsChecked) {
68      this.checkedRecords = checkedRecords;
69      this.uniqueTriplets = uniqueTriplets;
70      this.recordsWithInvalidTriplets = recordsWithInvalidTriplets;
71      this.uniqueOccurrenceIds = uniqueOccurrenceIds;
72      this.recordsMissingOccurrenceId = recordsMissingOccurrenceId;
73      this.allRecordsChecked = allRecordsChecked;
74      this.valid = validate();
75    }
76  
77    /**
78     * At the moment the only truly fatal conditions are:
79     * - whole archive is empty or unreadable
80     * - triplets are invalid (% invalid > than our threshold) && occIds are invalid (must be 100% coverage and unique)
81     * - any duplicate triplets && occIds are invalid
82     */
83    private boolean validate() {
84      boolean hasRecords = checkedRecords > 0;
85      double invalidRatio = hasRecords ? (double) recordsWithInvalidTriplets / checkedRecords : 0;
86      boolean invalidTripletsBelowLimit = invalidRatio <= INVALID_TRIPLET_THRESHOLD;
87      boolean hasUniqueTriplets = uniqueTriplets == checkedRecords - recordsWithInvalidTriplets;
88      boolean hasUniqueOccIds = uniqueOccurrenceIds == checkedRecords - recordsMissingOccurrenceId;
89      boolean hasGoodOccIds = uniqueOccurrenceIds == checkedRecords;
90      boolean looksValid = invalidTripletsBelowLimit && hasUniqueTriplets || hasGoodOccIds;
91  
92      if (!looksValid) {
93        List<String> reasons = new ArrayList<>();
94        if (!invalidTripletsBelowLimit) {
95          reasons.add(Math.round(100 * invalidRatio) + "% invalid triplets is > than threshold of " + Math
96            .round(100 * INVALID_TRIPLET_THRESHOLD) + '%');
97        }
98        if (!hasUniqueTriplets) {
99          reasons.add((checkedRecords - recordsWithInvalidTriplets - uniqueTriplets) + " duplicate triplets detected");
100       }
101       if (!hasGoodOccIds) {
102         if (recordsMissingOccurrenceId != 0) {
103           reasons.add(recordsMissingOccurrenceId + " records without an occurrence id (should be 0)");
104         }
105         if (!hasUniqueOccIds) {
106           reasons.add(
107             (checkedRecords - recordsMissingOccurrenceId - uniqueOccurrenceIds) + " duplicate occurrence ids detected");
108         }
109       }
110 
111       String reason = String.join("; ", reasons);
112       invalidationReason = "Archive invalid because [" + reason + ']';
113     }
114 
115     return looksValid;
116   }
