81      checkNotNull(builder, "builder can't be null");
82  
83      datasetKey = checkNotNull(builder.datasetKey, "datasetKey can't be null");
84      crawlJob = checkNotNull(builder.crawlJob, "crawlJob can't be null");
85      startedCrawling = builder.startedCrawling;
86      finishedCrawling = builder.finishedCrawling;
87      finishReason = builder.finishReason;
88      processStateOccurrence = builder.processStateOccurrence;
89      processStateChecklist = builder.processStateChecklist;
90      crawlContext = builder.crawlContext;
91  
92      declaredCount = builder.declaredCount;
93  
94      pagesCrawled = builder.pagesCrawled;
95  
96      pagesFragmentedSuccessful = builder.pagesFragmentedSuccessful;
97      pagesFragmentedError = builder.pagesFragmentedError;
98      fragmentsEmitted = builder.fragmentsEmitted;
99  
100     fragmentsReceived = builder.fragmentsReceived;
101     rawOccurrencesPersistedNew = builder.rawOccurrencesPersistedNew;
102     rawOccurrencesPersistedUpdated = builder.rawOccurrencesPersistedUpdated;
103     rawOccurrencesPersistedUnchanged = builder.rawOccurrencesPersistedUnchanged;
104     rawOccurrencesPersistedError = builder.rawOccurrencesPersistedError;
105     fragmentsProcessed = builder.fragmentsProcessed;
106 
107     verbatimOccurrencesPersistedSuccessful = builder.verbatimOccurrencesPersistedSuccessful;
108     verbatimOccurrencesPersistedError = builder.verbatimOccurrencesPersistedError;
109 
110     interpretedOccurrencesPersistedSuccessful = builder.interpretedOccurrencesPersistedSuccessful;
111     interpretedOccurrencesPersistedError = builder.interpretedOccurrencesPersistedError;
112 
113     checkArgument(declaredCount == null || declaredCount >= 0,
114                   "declaredCount must be either null or greater than or equal to zero");
115 
116     checkArgument(pagesCrawled >= 0, "pagesCrawled has to be greater than or equal to zero");
117 
118     checkArgument(pagesFragmentedSuccessful >= 0, "pagesFragmentedSuccessful");
119     checkArgument(pagesFragmentedError >= 0, "pagesFragmentedError has to be greater than or equal to zero");
120     checkArgument(fragmentsEmitted >= 0, "fragmentsEmitted has to be greater than or equal to zero");
121 
122     checkArgument(fragmentsReceived >= 0, "fragmentsReceived has to be greater than or equal to zero");
123     checkArgument(rawOccurrencesPersistedNew >= 0,
124                   "rawOccurrencesPersistedNew has to be greater than or equal to zero");
125     checkArgument(rawOccurrencesPersistedUpdated >= 0,
126                   "rawOccurrencesPersistedUpdated has to be greater than or equal to zero");
127     checkArgument(rawOccurrencesPersistedUnchanged >= 0,
128                   "rawOccurrencesPersistedUnchanged has to be greater than or equal to zero");
129     checkArgument(rawOccurrencesPersistedError >= 0,
130                   "rawOccurrencesPersistedError has to be greater than or equal to zero");
131     checkArgument(fragmentsProcessed >= 0, "fragmentsProcessed has to be greater than or equal to zero");
132 
133     checkArgument(verbatimOccurrencesPersistedSuccessful >= 0,
134                   "verbatimOccurrencesPersistedSuccessful has to be greater than or equal to zero");
135     checkArgument(verbatimOccurrencesPersistedError >= 0,
136                   "verbatimOccurrencesPersistedError has to be greater than or equal to zero");
137 
138     checkArgument(interpretedOccurrencesPersistedSuccessful >= 0,
139                   "interpretedOccurrencesPersistedSuccessful has to be greater than or equal to zero");

141                   "interpretedOccurrencesPersistedError has to be greater than or equal to zero");

409   @Override
410   public boolean equals(Object obj) {
411     if (obj == null) {
412       return false;
413     }
414     if (getClass() != obj.getClass()) {
415       return false;
416     }
417     final DatasetProcessStatus other = (DatasetProcessStatus) obj;
418     return Objects.equal(this.datasetKey, other.datasetKey)
419            && Objects.equal(this.crawlJob, other.crawlJob)
420            && Objects.equal(this.startedCrawling, other.startedCrawling)
421            && Objects.equal(this.finishedCrawling, other.finishedCrawling)
422            && Objects.equal(this.finishReason, other.finishReason)
423            && Objects.equal(this.processStateOccurrence, other.processStateOccurrence)
424            && Objects.equal(this.processStateChecklist, other.processStateChecklist)
425            && Objects.equal(this.crawlContext, other.crawlContext)
426            && Objects.equal(this.declaredCount, other.declaredCount)
427            && Objects.equal(this.pagesCrawled, other.pagesCrawled)
428            && Objects.equal(this.pagesFragmentedSuccessful, other.pagesFragmentedSuccessful)
429            && Objects.equal(this.pagesFragmentedError, other.pagesFragmentedError)
430            && Objects.equal(this.fragmentsEmitted, other.fragmentsEmitted)
431            && Objects.equal(this.fragmentsReceived, other.fragmentsReceived)
432            && Objects.equal(this.rawOccurrencesPersistedNew, other.rawOccurrencesPersistedNew)
433            && Objects.equal(this.rawOccurrencesPersistedUpdated, other.rawOccurrencesPersistedUpdated)
434            && Objects.equal(this.rawOccurrencesPersistedUnchanged, other.rawOccurrencesPersistedUnchanged)
435            && Objects.equal(this.rawOccurrencesPersistedError, other.rawOccurrencesPersistedError)
436            && Objects.equal(this.fragmentsProcessed, other.fragmentsProcessed)
437            && Objects.equal(this.verbatimOccurrencesPersistedSuccessful, other.verbatimOccurrencesPersistedSuccessful)
438            && Objects.equal(this.verbatimOccurrencesPersistedError, other.verbatimOccurrencesPersistedError)
439            && Objects.equal(this.interpretedOccurrencesPersistedSuccessful,
440                             other.interpretedOccurrencesPersistedSuccessful)
441            && Objects.equal(this.interpretedOccurrencesPersistedError, other.interpretedOccurrencesPersistedError);
442   }
