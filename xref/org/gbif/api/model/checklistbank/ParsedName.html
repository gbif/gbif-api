31  /**
32   * A container of a taxon name that is atomised into it's relevant separate parts.
33   * Hybrid formulas that consist of multiple genera, binomials or species
34   * epitheta are relatively poor represented. A simple boolean flag indicates a
35   * hybrid formula, e.g. Polygala vulgaris X Polygala epinema but not named
36   * hybrids that are have an x or notho as part of a well formed
37   * mono/bi/trinomial e.g. xPolygala vulgaris. In the case of hybrid formulas,
38   * i.e. isHybrid=true, the first name in the formula is tried to be kept, at
39   * least the genus.
40   * <p/>
41   * A container of a taxon name that is only atomised into three name parts maximum plus rank and a notho property
42   * indicating the name part of named hybrids that is considered to be the hybrid. No authorship is kept. For subgenera
43   * we don't use parenthesis to indicate the subgenus, but use explicit rank markers instead.
44   */
45  public class ParsedName {

75    /**
76     * Replaces all digraphs and ligatures with their underlying 2 latin letters.
77     *
78     * @param x the string to decompose
79     */
80    public static String decompose(String x) {
81      if (x == null) {
82        return x;
83      }
84      // god knows why, but on OSX with java 1.5&6 at least this doesnt work as it should! Use custom replace method
85      // return Normalizer.normalize(x, Form.NFD);
86      return x.replaceAll("æ", "ae")
87        .replaceAll("Æ", "Ae")
88        .replaceAll("œ", "oe")
89        .replaceAll("Œ", "Oe")
90        .replaceAll("Ĳ", "Ij")
91        .replaceAll("ĳ", "ij")
92        .replaceAll("ǈ", "Lj")
93        .replaceAll("ǉ", "lj")
94        .replaceAll("ȸ", "db")
95        .replaceAll("ȹ", "qp")
96        .replaceAll("ß", "ss")
97        .replaceAll("ﬆ", "st")
98        .replaceAll("ﬀ", "ff")
99        .replaceAll("ﬁ", "fi")
100       .replaceAll("ﬂ", "fl")
101       .replaceAll("ﬃ", "ffi")
102       .replaceAll("ﬄ", "ffl");
103   }

302   /**
303    * build a name controlling all available flags for name parts to be included in the resulting name.
304    *
305    * @param hybridMarker    include the hybrid marker with the name if existing
306    * @param rankMarker      include the infraspecific or infrageneric rank marker with the name if existing
307    * @param authorship      include the names authorship (authorteam and year)
308    * @param subgenus        include the subgenus in brackets in species or subspecies
309    * @param abbreviateGenus if true abreviate the genus with its first character
310    * @param decomposition   decompose unicode letters into their corresponding ascii ones, e.g. æ beomes ae
311    * @param showIndet       if true include the rank marker for incomplete determinations, for example Puma spec.
312    * @param nomNote         include nomenclatural notes
313    * @param remarks         include informal remarks
314    */
315   public String buildName(
316     boolean hybridMarker,
317     boolean rankMarker,
318     boolean authorship,
319     boolean subgenus,
320     boolean abbreviateGenus,
321     boolean decomposition,
322     boolean showIndet,
323     boolean nomNote,
324     boolean remarks,
325     boolean showSensu,
326     boolean showCultivar,
327     @Nullable NomenclaturalCode code
328   ) {
329     StringBuilder sb = new StringBuilder();
330     Rank rnk = getRank();
331 
332     if (NameType.CANDIDATUS == type) {
333       sb.append("Candidatus ");
334     }
335 
336     if (genusOrAbove != null) {
337       if (hybridMarker && NamePart.GENERIC == notho) {
338         sb.append(HYBRID_MARKER);
339       }
340       if (abbreviateGenus) {
341         sb.append(genusOrAbove.substring(0, 1)).append('.');
342       } else {
343         sb.append(genusOrAbove);
344       }
345     }
346     if (specificEpithet == null) {
347       if (Rank.SPECIES == rnk) {
348         // no species epitheton given, but rank=species. Indetermined species!
349         if (showIndet) {
350           sb.append(" spec.");
351         }
352       } else if (rnk != null && rnk.isInfraspecific()) {
353         // no species epitheton given, but rank below species. Indetermined!
354         if (showIndet) {
355           sb.append(' ');
356           sb.append(rnk.getMarker());
357         }
358       } else if (infraGeneric != null) {
359         // this is the terminal name part - always show it!
360         // We dont use parenthesis to indicate a subgenus, but use explicit rank markers instead
361         if (rankMarker) {
362           if (rank == null) {
363             // assume its subgenus. Zoological subgenera in brackets dont have a rank marker in the parsed name
364             sb.append(' ').append(Rank.SUBGENUS.getMarker());
365           } else {
366             sb.append(' ').append(rank);
367           }
368         }
369         sb.append(' ').append(infraGeneric);
370       }
371       // genus/infrageneric authorship
372       if (authorship) {
373         appendAuthorship(sb);
374       }
375     } else {
376       if (subgenus && infraGeneric != null && (rank == null || getRank() == Rank.GENUS)) {
377         // only show subgenus if requested
378         sb.append(" (");
379         sb.append(infraGeneric);
380         sb.append(')');
381       }
382 
383       // species part
384       sb.append(' ');
385       if (hybridMarker && NamePart.SPECIFIC == notho) {
386         sb.append(HYBRID_MARKER);
387       }
388       String epi = specificEpithet.replaceAll("[ _-]", "-");
389       sb.append(epi);
390 
391       if (infraSpecificEpithet == null) {
392         // Indetermined? Only show indet cultivar marker if no cultivar epithet exists
393         if (showIndet && rnk != null && rnk.isInfraspecific() && (Rank.CULTIVAR != rnk || cultivarEpithet == null)) {
394           // no infraspecific epitheton given, but rank below species. Indetermined!
395           sb.append(' ');
396           sb.append(rnk.getMarker());
397         }
398 
399         // species authorship
400         if (authorship) {
401           appendAuthorship(sb);
402         }
403       } else {
404         // infraspecific part
405         sb.append(' ');
406         if (hybridMarker && NamePart.INFRASPECIFIC == notho) {
407           if (rankMarker) {
408             sb.append("notho");
409           } else {
410             sb.append(HYBRID_MARKER);
411           }
412         }
413         if (rankMarker) {
414           String rm = getInfraspecificRankMarker(code);
415           if (rm != null) {
416             sb.append(rm);
417             sb.append(' ');
418           }
419         }
420         epi = infraSpecificEpithet.replaceAll("[ _-]", "-");
421         sb.append(epi);
422         // non autonym authorship ?
423         if (authorship && !isAutonym()) {
424           appendAuthorship(sb);
425         }
426       }
427     }
428 
429     // add cultivar name
430     if (showCultivar && cultivarEpithet != null) {
431       sb.append(" '");
432       sb.append(cultivarEpithet);
433       sb.append("'");
434     }
435 
436     // add sensu/sec reference
437     if (showSensu && sensu != null) {
438       sb.append(" ");
439       sb.append(sensu);
440     }
441 
442     // add nom status
443     if (nomNote && nomStatus != null) {
444       sb.append(", ");
445       sb.append(nomStatus);
446     }
447 
448     // add remarks
449     if (remarks && this.remarks != null) {
450       sb.append(" [");
451       sb.append(this.remarks);
452       sb.append("]");
453     }
454 
455     if (decomposition) {
456       return decompose(sb.toString().trim());
457     }
458     return sb.toString().trim();
459   }

461   private String getInfraspecificRankMarker(NomenclaturalCode code) {
462     if (getRank() == Rank.SUBSPECIES && NomenclaturalCode.ZOOLOGICAL == code) {
463       return null;
464     }
465     return rank;
466   }

468   private void appendAuthorship(StringBuilder sb) {
469     if (bracketAuthorship == null) {
470       if (bracketYear != null) {
471         sb.append(" (");
472         sb.append(bracketYear);
473         sb.append(")");
474       }
475     } else {
476       sb.append(" (");
477       sb.append(bracketAuthorship);
478       if (bracketYear != null) {
479         sb.append(", ");
480         sb.append(bracketYear);
481       }
482       sb.append(")");
483     }
484     if (authorship != null) {
485       sb.append(" ").append(authorship);
486     }
487     if (year != null) {
488       sb.append(", ");
489       sb.append(year);
490     }
491   }

510   /**
511    * The code compliant, canonical name with 3 name parts at max (genus, species, infraspecific), a rank marker for
512    * infraspecific names and cultivar or strain epithets. The canonical name can be a 1, 2 or 3 parted name, but does
513    * not include any informal notes or
514    * authorships. Notho taxa will have the hybrid marker.
515    * <p/>
516    * For example:
517    * Abies alba
518    * Abies alba subsp. alpina
519    * Abies sect. Bracteata
520    * ×Heucherella tiarelloides
521    *
522    * @return the 1,2 or 3 parted name as a single string
523    */
524   @JsonProperty
525   public String canonicalNameWithMarker() {
526     return buildName(true, true, false, false, false, true, true, false, false, false, true);
527   }

660   public void setRankMarker(String rankMarker) {
661     if (rankMarker == null) {
662       rank = null;
663     } else {
664       rankMarker = rankMarker.trim().toLowerCase();
665       if (rankMarker.startsWith("notho")) {
666         rankMarker = rankMarker.substring(5);
667         notho = NamePart.INFRASPECIFIC;
668       }
669       Rank r = Rank.inferRank(rankMarker);
670       if (r == null || Rank.UNCOMPARABLE_RANKS.contains(r)) {
671         this.rank = rankMarker;
672       } else {
673         this.rank = r.getMarker();
674       }
675     }
676   }

687   @Override
688   public boolean equals(Object obj) {
689     if (this == obj) {
690       return true;
691     }
692     if (!(obj instanceof ParsedName)) {
693       return false;
694     }
695     ParsedName o = (ParsedName) obj;
696     return equal(key, o.key)
697            && equal(scientificName, o.scientificName)
698            && equal(type, o.type)
699            && equal(genusOrAbove, o.genusOrAbove)
700            && equal(infraGeneric, o.infraGeneric)
701            && equal(specificEpithet, o.specificEpithet)
702            && equal(infraSpecificEpithet, o.infraSpecificEpithet)
703            && equal(cultivarEpithet, o.cultivarEpithet)
704            && equal(authorship, o.authorship)
705            && equal(year, o.year)
706            && equal(bracketAuthorship, o.bracketAuthorship)
707            && equal(bracketYear, o.bracketYear)
708            && equal(rank, o.rank);
709   }

729   @Override
730   public String toString() {
731     StringBuilder sb = new StringBuilder();
732     sb.append(scientificName);
733     if (key != null) {
734       sb.append(" [");
735       sb.append(key);
736       sb.append("]");
737     }
738     if (genusOrAbove != null) {
739       sb.append(" G:").append(genusOrAbove);
740     }
741     if (infraGeneric != null) {
742       sb.append(" IG:").append(infraGeneric);
743     }
744     if (specificEpithet != null) {
745       sb.append(" S:").append(specificEpithet);
746     }
747     if (rank != null) {
748       sb.append(" R:").append(rank);
749     }
750     if (infraSpecificEpithet != null) {
751       sb.append(" IS:").append(infraSpecificEpithet);
752     }
753     if (cultivarEpithet != null) {
754       sb.append(" CV:").append(cultivarEpithet);
755     }
756     if (authorship != null) {
757       sb.append(" A:").append(authorship);
758     }
