29  /**
30   * A container of a taxon name that is atomised into it's relevant separate parts.
31   * Hybrid formulas that consist of multiple genera, binomials or species
32   * epitheta are relatively poor represented. A simple boolean flag indicates a
33   * hybrid formula, e.g. Polygala vulgaris X Polygala epinema but not named
34   * hybrids that are have an x or notho as part of a well formed
35   * mono/bi/trinomial e.g. xPolygala vulgaris. In the case of hybrid formulas,
36   * i.e. isHybrid=true, the first name in the formula is tried to be kept, at
37   * least the genus.
38   * <p/>
39   * A container of a taxon name that is only atomised into three name parts maximum plus rank and a notho property
40   * indicating the name part of named hybrids that is considered to be the hybrid. No authorship is kept. For subgenera
41   * we don't use parenthesis to indicate the subgenus, but use explicit rank markers instead.
42   */
43  public class ParsedName {

74    /**
75     * Replaces all digraphs and ligatures with their underlying 2 latin letters.
76     *
77     * @param x the string to decompose
78     */
79    public static String decompose(String x) {
80      if (x == null) {
81        return x;
82      }
83      // god knows why, but on OSX with java 1.5&6 at least this doesnt work as it should! Use custom replace method
84      // return Normalizer.normalize(x, Form.NFD);
85      return x.replaceAll("æ", "ae")
86        .replaceAll("Æ", "Ae")
87        .replaceAll("œ", "oe")
88        .replaceAll("Œ", "Oe")
89        .replaceAll("Ĳ", "Ij")
90        .replaceAll("ĳ", "ij")
91        .replaceAll("ǈ", "Lj")
92        .replaceAll("ǉ", "lj")
93        .replaceAll("ȸ", "db")
94        .replaceAll("ȹ", "qp")
95        .replaceAll("ß", "ss")
96        .replaceAll("ﬆ", "st")
97        .replaceAll("ﬀ", "ff")
98        .replaceAll("ﬁ", "fi")
99        .replaceAll("ﬂ", "fl")
100       .replaceAll("ﬃ", "ffi")
101       .replaceAll("ﬄ", "ffl");
102   }

301   /**
302    * build a name controlling all available flags for name parts to be included in the resulting name.
303    *
304    * @param hybridMarker    include the hybrid marker with the name if existing
305    * @param rankMarker      include the infraspecific or infrageneric rank marker with the name if existing
306    * @param authorship      include the names authorship (authorteam and year)
307    * @param infrageneric include the infrageneric name in brackets for species or infraspecies
308    * @param genusForInfrageneric include the genus name in front of an infrageneric name (not a species)
309    * @param abbreviateGenus if true abreviate the genus with its first character
310    * @param decomposition   decompose unicode letters into their corresponding ascii ones, e.g. æ beomes ae
311    * @param showIndet       if true include the rank marker for incomplete determinations, for example Puma spec.
312    * @param nomNote         include nomenclatural notes
313    * @param remarks         include informal remarks
314    */
315   public String buildName(
316     boolean hybridMarker,
317     boolean rankMarker,
318     boolean authorship,
319     boolean infrageneric,
320     boolean genusForInfrageneric,
321     boolean abbreviateGenus,
322     boolean decomposition,
323     boolean showIndet,
324     boolean nomNote,
325     boolean remarks,
326     boolean showSensu,
327     boolean showCultivar,
328     boolean showStrain
329   ) {
330     StringBuilder sb = new StringBuilder();
331     Rank rnk = getRank();
332 
333     if (NameType.CANDIDATUS == type) {
334       sb.append("Candidatus ");
335     }
336 
337     if (genusOrAbove != null && (genusForInfrageneric || infraGeneric == null || specificEpithet != null)) {
338       if (hybridMarker && NamePart.GENERIC == notho) {
339         sb.append(HYBRID_MARKER);
340       }
341       if (abbreviateGenus) {
342         sb.append(genusOrAbove.substring(0, 1)).append('.');
343       } else {
344         sb.append(genusOrAbove);
345       }
346     }
347     if (specificEpithet == null) {
348       if (Rank.SPECIES == rnk) {
349         // no species epitheton given, but rank=species. Indetermined species!
350         if (showIndet) {
351           sb.append(" spec.");
352         }
353       } else if (rnk != null && rnk.isInfraspecific()) {
354         // no species epitheton given, but rank below species. Indetermined!
355         if (showIndet) {
356           sb.append(' ');
357           sb.append(rnk.getMarker());
358         }
359       } else if (infraGeneric != null) {
360         // this is the terminal name part - always show it!
361         if (rankMarker && rank != null) {
362           // If we know the rank we use explicit rank markers
363           // this is how botanical infrageneric names are formed, see http://www.iapt-taxon.org/nomen/main.php?page=art21
364           sb.append(' ')
365             .append(rank)
366             .append(' ')
367             .append(infraGeneric);
368         } else {
369             if (genusForInfrageneric && genusOrAbove != null) {
370               // if we have shown the genus already and we do not know the rank we use parenthesis to indicate an infrageneric
371               sb.append(" (")
372                       .append(infraGeneric)
373                       .append(")");
374             } else {
375               // no genus shown yet, just show the plain infrageneric name
376               sb.append(infraGeneric);
377             }
378         }
379       }
380       // genus/infrageneric authorship
381       if (authorship) {
382         appendAuthorship(sb);
383       }
384     } else {
385       if (infrageneric && infraGeneric != null && (rank == null || getRank() == Rank.GENUS)) {
386         // only show subgenus if requested
387         sb.append(" (");
388         sb.append(infraGeneric);
389         sb.append(')');
390       }
391 
392       // species part
393       sb.append(' ');
394       if (hybridMarker && NamePart.SPECIFIC == notho) {
395         sb.append(HYBRID_MARKER);
396       }
397       String epi = specificEpithet.replaceAll("[ _-]", "-");
398       sb.append(epi);
399 
400       if (infraSpecificEpithet == null) {
401         // Indetermined? Only show indet cultivar marker if no cultivar epithet exists
402         if (showIndet && rnk != null && rnk.isInfraspecific() && (Rank.CULTIVAR != rnk || cultivarEpithet == null)) {
403           // no infraspecific epitheton given, but rank below species. Indetermined!
404           sb.append(' ');
405           sb.append(rnk.getMarker());
406         }
407 
408         // species authorship
409         if (authorship) {
410           appendAuthorship(sb);
411         }
412       } else {
413         // infraspecific part
414         sb.append(' ');
415         if (hybridMarker && NamePart.INFRASPECIFIC == notho) {
416           if (rankMarker) {
417             sb.append("notho");
418           } else {
419             sb.append(HYBRID_MARKER);
420           }
421         }
422         if (rankMarker) {
423           sb.append(rank);
424           sb.append(' ');
425         }
426         epi = infraSpecificEpithet.replaceAll("[ _-]", "-");
427         sb.append(epi);
428         // non autonym authorship ?
429         if (authorship && !isAutonym()) {
430           appendAuthorship(sb);
431         }
432       }
433     }
434 
435     // add cultivar name
436     if (showStrain && strain != null) {
437       sb.append(" ");
438       sb.append(strain);
439     }
440 
441     // add cultivar name
442     if (showCultivar && cultivarEpithet != null) {
443       sb.append(" '");
444       sb.append(cultivarEpithet);
445       sb.append("'");
446     }
447 
448     // add sensu/sec reference
449     if (showSensu && sensu != null) {
450       sb.append(" ");
451       sb.append(sensu);
452     }
453 
454     // add nom status
455     if (nomNote && nomStatus != null) {
456       sb.append(", ");
457       sb.append(nomStatus);
458     }
459 
460     // add remarks
461     if (remarks && this.remarks != null) {
462       sb.append(" [");
463       sb.append(this.remarks);
464       sb.append("]");
465     }
466 
467     if (decomposition) {
468       return decompose(sb.toString().trim());
469     }
470     return sb.toString().trim();
471   }

480   private void appendAuthorship(StringBuilder sb) {
481     if (bracketAuthorship == null) {
482       if (bracketYear != null) {
483         sb.append(" (");
484         sb.append(bracketYear);
485         sb.append(")");
486       }
487     } else {
488       sb.append(" (");
489       sb.append(bracketAuthorship);
490       if (bracketYear != null) {
491         sb.append(", ");
492         sb.append(bracketYear);
493       }
494       sb.append(")");
495     }
496     if (authorship != null) {
497       sb.append(" ").append(authorship);
498     }
499     if (year != null) {
500       sb.append(", ");
501       sb.append(year);
502     }
503   }

505   /**
506    * The canonical name sensu strictu with nothing else but 3 name parts at max (genus, species, infraspecific). No
507    * rank or hybrid markers and no authorship, cultivar or strain information.
508    * Infrageneric names are represented without a leading genus.
509    * <p/>
510    * For example:
511    * Abies alba
512    * Abies alba alpina
513    * Abies Bracteata
514    * Heucherella tiarelloides
515    *
516    * @return the 1,2 or 3 parted name as a single string
517    */

523   /**
524    * The code compliant, canonical name with 3 name parts at max (genus, species, infraspecific), a rank marker for
525    * infraspecific names and cultivar or strain epithets. The canonical name can be a 1, 2 or 3 parted name, but does
526    * not include any informal notes or
527    * authorships. Notho taxa will have the hybrid marker.

529    * For example:
530    * Abies alba

535    * @return the 1,2 or 3 parted name as a single string

673   public void setRankMarker(String rankMarker) {
674     if (rankMarker == null) {
675       rank = null;
676     } else {
677       rankMarker = rankMarker.trim().toLowerCase();
678       if (rankMarker.startsWith("notho")) {
679         rankMarker = rankMarker.substring(5);
680         notho = NamePart.INFRASPECIFIC;
681       }
682       Rank r = Rank.inferRank(rankMarker);
683       if (r == null || Rank.UNCOMPARABLE_RANKS.contains(r)) {
684         this.rank = rankMarker;
685       } else {
686         this.rank = r.getMarker();
687       }
688     }
689   }

700   @Override
701   public boolean equals(Object obj) {
702     if (this == obj) {
703       return true;
704     }
705     if (!(obj instanceof ParsedName)) {
706       return false;
707     }
708     ParsedName o = (ParsedName) obj;
709     return equal(key, o.key)
710            && equal(scientificName, o.scientificName)
711            && equal(type, o.type)
712            && equal(genusOrAbove, o.genusOrAbove)
713            && equal(infraGeneric, o.infraGeneric)
714            && equal(specificEpithet, o.specificEpithet)
715            && equal(infraSpecificEpithet, o.infraSpecificEpithet)
716            && equal(cultivarEpithet, o.cultivarEpithet)
717            && equal(strain, o.strain)
718            && equal(authorship, o.authorship)
719            && equal(year, o.year)
720            && equal(bracketAuthorship, o.bracketAuthorship)
721            && equal(bracketYear, o.bracketYear)
722            && equal(rank, o.rank);
723   }

744   @Override
745   public String toString() {
746     StringBuilder sb = new StringBuilder();
747     sb.append(scientificName);
