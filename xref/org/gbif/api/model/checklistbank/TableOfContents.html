14  /**
15   * A simple class to represent a table of contents for multiple languages.
16   * It is used by species pages to generate a table of contents menu with asynchronous loading of full descriptions.
17   */
18  public class TableOfContents {
19    private static final String DEFAULT_TOPIC = "general";
20  
21    private final Map<Language, Map<String, List<Integer>>> toc = Maps.newTreeMap();
22  
23    public void addDescription(int key, Language lang, String topic) {
24      topic = Strings.isNullOrEmpty(topic) ? DEFAULT_TOPIC : topic.toLowerCase().trim();
25  
26      if (lang == null) {
27        // default to english
28        lang = Language.ENGLISH;
29      }
30  
31      if (!toc.containsKey(lang)) {
32        toc.put(lang, Maps.<String, List<Integer>>newTreeMap());
33      }
34      if (!toc.get(lang).containsKey(topic)) {
35        toc.get(lang).put(topic, Lists.<Integer>newArrayList());
36      }
37      toc.get(lang).get(topic).add(key);
38    }
39  
40    @JsonIgnore
41    public boolean isEmpty() {
42      return toc.isEmpty();
43    }
44  
45    /**
46     * @return list of all languages available for this ToC
47     */
48    @JsonIgnore
49    public List<Language> listLanguages() {
50      return Lists.newArrayList(toc.keySet());
51    }
52  
53    /**
54     * @return map of all topics for a given language with a list of entry keys for each language
55     */
56    @JsonIgnore
57    public Map<String, List<Integer>> listTopicEntries(Language lang) {
58      if (toc.containsKey(lang)) {
59        return toc.get(lang);
60      }
61      return Maps.newHashMap();
62    }
