44   * the nub. It is shown as species in the new Portal.
45   * All nub usages will have a nubKey equal to their key and belong to the nub checklist registry resource.

49   * the {@link NameUsageContainer} class.

761    * An optional string that can be used to disambiguate homonyms.
762    * For example Oenanthe alone can mean either an orchid or a bird.
763    * When shown in selection lists a user needs to know this difference to make a meaningful choice.
764    *
765    * @return the homonym disambiguation
766    */
767   @Nullable
768   @JsonIgnore
769   public String getHomonymDisambiguation() {
770     // TODO: use a more sophisticated logic and only show the lowest rank sufficient for disambiguation
771     // TODO: make this optional and only show a string when disambiguation is needed!
772     Joiner joiner = Joiner.on(", ").skipNulls();
773     return joiner.join(getKingdom(), getPhylum(), getClazz(), getOrder(), getFamily());

809   /**
810    * Returns the primary source record for this usage as an Integer.
811    * All nub usages are build from several underlying checklist usages,
812    * but these are sorted by priority and the usage key for the highest priority one becomes the sourceId for a nub
813    * usage.
814    * Some nub usages do not have any source record altogether.
815    * For example if there is a subspecies found, but no matching parent species,
816    * the missing species will be created nevertheless and has no primary source.
817    *
818    * @return the sourceId for a name usage as an integer or null if its not convertible
819    */
820   @JsonIgnore
821   public Integer getSourceKey() {
822     final String sourceId = getSourceId();
823     if (sourceId != null) {
824       try {
825         return Integer.valueOf(sourceId);
826       } catch (Exception ignored) {
827         // swallow
828       }
829     }
830     return null;
831   }

833   /**
834    * The status of the use of the scientificName as a label for the taxon.
835    * If it's not set explicit the isSynonym and isProParte flags are used to calculate values.
836    * <blockquote>
837    * <p>
838    * <i>Example:</i> "invalid", "misapplied", "homotypic synonym", "accepted"
839    * </p>
840    * </blockquote>
841    *
842    * @return the taxonomicStatus, can be null
843    */
844   @Nullable
845   public TaxonomicStatus getTaxonomicStatus() {
846     if (taxonomicStatus != null) {
847       return taxonomicStatus;
848     }
849 
850     if (isSynonym) {
851       return isProParte() ? TaxonomicStatus.PROPARTE_SYNONYM : TaxonomicStatus.SYNONYM;
852     }
853 
854     return null;
855   }
