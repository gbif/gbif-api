51  /**
52   * This class builds a human readable filter from a {@link org.gbif.api.model.occurrence.predicate.Predicate} hierarchy.
53   * This class is not thread safe, create a new instance for every use if concurrent calls to {#humanFilter} is expected.
54   * The IN predicate is not yet supported and you'll get an IllegalArgumentException.
55   * This builder only supports predicates that follow our search query parameters style with multiple values for the same
56   * parameter being logically disjunct (OR) while different search parameters are logically combined (AND). Therefore
57   * the {#humanFilter(Predicate p)} result is a map of OccurrenceSearchParameter (AND'ed) to a list of values (OR'ed).
58   */
59  public class HumanFilterBuilder {
60  
61    private enum State {
62      ROOT, AND, OR
63    }

119   /**
120    * @param p the predicate to convert
121    * @return a list of anded parameters with multiple values to be combined with OR
122    * @throws IllegalStateException if more complex predicates than the portal handles are supplied
123    */
124   public synchronized Map<OccurrenceSearchParameter, LinkedList<String>> humanFilter(Predicate p) {
125     filter = Maps.newLinkedHashMap();
126     state = State.ROOT;
127     lastParam = null;
128     visit(p);
129     return filter;
130   }

132   private void addParamValue(OccurrenceSearchParameter param, String op, String value) {
133     addParamValue(param, op + getHumanValue(param, value));
134   }

136   /**
137    * Gets the human readable value of the parameter value.
138    */
139   private String getHumanValue(OccurrenceSearchParameter param, String value) {
140     String humanValue;
141     // lookup values
142     switch (param) {
143       case TAXON_KEY:
144         humanValue = lookupTaxonKey(value);
145         break;
146       case DATASET_KEY:
147         humanValue = lookupDatasetKey(value);
148         break;
149       case COUNTRY:
150       case PUBLISHING_COUNTRY:
151         humanValue = lookupCountryCode(value);
152         break;
153       case CONTINENT:
154         humanValue = lookupContinent(value);
155         break;
156       case MONTH:
157         humanValue = lookupMonth(value);
158         break;
159 
160       default:
161         if (param.type().isEnum()) {
162           humanValue = lookupEnum(param.type(), value);
163         } else {
164           humanValue = value;
165         }
166     }
167     // add unit symbol
168     if (param==OccurrenceSearchParameter.DEPTH || param==OccurrenceSearchParameter.ELEVATION) {
169       humanValue = humanValue + "m";
170     }
171     return humanValue;
172   }

179   private void addParamValue(OccurrenceSearchParameter param, String op) {
180     // verify that last param if existed was the same
181     if (lastParam != null && param != lastParam) {
182       throw new IllegalArgumentException("Mix of search params not supported: "+param);
183     }
184 
185     if (!filter.containsKey(param)) {
186       filter.put(param, Lists.<String>newLinkedList());
187     }
188 
189     filter.get(param).add(op);
190     lastParam = param;
191   }

255   private void visit(ConjunctionPredicate and) throws IllegalStateException {
256     // ranges are allowed underneath root - try first
257     try {
258       visitRange(and);
259       return;
260     } catch (IllegalArgumentException e) {
261       // must be a root AND
262     }
263 
264     if (state != State.ROOT) {
265       throw new IllegalStateException("AND must be a root predicate or a valid range");
266     }
267     state = State.AND;
268 
269     for (Predicate p : and.getPredicates()) {
270       lastParam = null;
271       visit(p);
272     }
273     state = State.ROOT;
274   }
275 
276   private void visit(DisjunctionPredicate or) throws IllegalStateException {
277     State oldState = state;
278     if (state == State.OR) {
279       throw new IllegalStateException("OR within OR filters not supported");
280     }
281     state = State.OR;
282 
283     for (Predicate p : or.getPredicates()) {
284       visit(p);
285     }
286     state = oldState;
287   }

319   private void visit(NotPredicate not) throws IllegalStateException {
320     if (not.getPredicate() instanceof SimplePredicate) {
321       visit(not.getPredicate());
322       SimplePredicate sp = (SimplePredicate) not.getPredicate();
323       // now prefix the last value with NOT
324       String notValue = NOT_OPERATOR + " (" + filter.get(sp.getKey()).removeLast() + ")";
325       filter.get(sp.getKey()).add(notValue);
326 
327     } else {
328       throw new IllegalArgumentException("NOT predicate must be followed by a simple predicate");
329     }
330   }

332   private void visit(IsNotNullPredicate predicate) {

334   }

337   private void visit(Predicate p) throws IllegalStateException {
338     Method method = null;
339     try {
340       method = getClass().getDeclaredMethod("visit", new Class[] {p.getClass()});
341     } catch (NoSuchMethodException e) {
342       LOG
343         .warn(
344           "Visit method could not be found. That means a Predicate has been passed in that is unknown to this "
345             + "class",
346           e);
347       throw new IllegalArgumentException("Unknown Predicate", e);
348     }
349     try {
350       method.setAccessible(true);
351       method.invoke(this, p);
352     } catch (IllegalAccessException e) {
353       LOG.error(
354         "This should never happen as we set accessible to true explicitly before. Probably a programming error", e);
355       throw new RuntimeException("Programming error", e);
356     } catch (InvocationTargetException e) {
357       LOG.info("Exception thrown while building the human query string", e);
358       throw new IllegalArgumentException(e);
359     }
360   }

362   private void visit(WithinPredicate within) {
363     addParamValue(OccurrenceSearchParameter.GEOMETRY, "", within.getGeometry());
364   }

366   private void visitRange(ConjunctionPredicate and) {
367     if (and.getPredicates().size() != 2) {
368       throw new IllegalArgumentException("no valid range");
369     }
370     GreaterThanOrEqualsPredicate lower = null;
371     LessThanOrEqualsPredicate upper = null;
372     for (Predicate p : and.getPredicates()) {
373       if (p instanceof GreaterThanOrEqualsPredicate) {
374         lower = (GreaterThanOrEqualsPredicate) p;
375       } else if (p instanceof LessThanOrEqualsPredicate) {
376         upper = (LessThanOrEqualsPredicate) p;
377       }
378     }
379     if (lower == null || upper == null || lower.getKey() != upper.getKey()) {
380       throw new IllegalArgumentException("no valid range");
381     }
382     addParamValue(lower.getKey(), "", lower.getValue() + "-" + upper.getValue());
383   }
