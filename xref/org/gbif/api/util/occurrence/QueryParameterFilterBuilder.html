46  /**
47   * This class builds a query parameter filter usable for search links from a
48   * {@link org.gbif.api.model.occurrence.predicate.Predicate} hierarchy.
49   * This class is not thread safe, create a new instance for every use if concurrent calls to {#queryFilter()} is expected.
50   */
51  public class QueryParameterFilterBuilder {
52  
53    private static final Logger LOG = LoggerFactory.getLogger(QueryParameterFilterBuilder.class);
54    private static final String WILDCARD = "*";
55    private final static Pattern POLYGON_PATTERN = Pattern.compile("POLYGON\\s*\\(\\s*\\((.+)\\)\\s*\\)", Pattern.CASE_INSENSITIVE);
56  
57    private Map<OccurrenceSearchParameter, LinkedList<String>> filter;
58    private enum State { ROOT, AND, OR };
59    private State state;
60    private OccurrenceSearchParameter lastParam;
61  
62    public QueryParameterFilterBuilder() {
63    }
64  
65    public synchronized String queryFilter(Predicate p) {
66      StringBuilder b = new StringBuilder();
67      boolean first = true;
68  
69      // build filter map
70      filter = Maps.newHashMap();
71      state = State.ROOT;
72      lastParam = null;
73      visit(p);
74  
75      // transform filter map into query string
76      for (OccurrenceSearchParameter param : filter.keySet()) {
77        for (String val : filter.get(param)) {
78          if (first) {
79            first = false;
80          } else {
81            b.append("&");
82          }
83          b.append(param.name());
84          b.append("=");
85          b.append(URLEncoder.encode(val));
86        }
87      }
88      return b.toString();
89    }
90  
91    private void visit(ConjunctionPredicate and) throws IllegalStateException {
92      // ranges are allowed underneath root - try first
93      try {
94        visitRange(and);
95        return;
96      } catch (IllegalArgumentException e) {
97        // must be a root AND
98      }
99  
100     if (state != State.ROOT) {
101       throw new IllegalStateException("AND must be a root predicate or a valid range");
102     }
103     state = State.AND;
104 
105     for (Predicate p : and.getPredicates()) {
106       lastParam = null;
107       visit(p);
108     }
109     state = State.ROOT;
110   }
111 
112   private void visitRange(ConjunctionPredicate and){
113     if (and.getPredicates().size() != 2){
114       throw new IllegalArgumentException("no valid range");
115     }
116     GreaterThanOrEqualsPredicate lower = null;
117     LessThanOrEqualsPredicate upper = null;
118     for (Predicate p : and.getPredicates()) {
119       if (p instanceof GreaterThanOrEqualsPredicate) {
120         lower = (GreaterThanOrEqualsPredicate) p;
121       } else if (p instanceof LessThanOrEqualsPredicate) {
122         upper = (LessThanOrEqualsPredicate) p;
123       }
124     }
125     if (lower == null || upper == null || lower.getKey() != upper.getKey()) {
126       throw new IllegalArgumentException("no valid range");
127     }
128     addQueryParam(lower.getKey(), range(lower.getValue(), upper.getValue()));
129   }
130 
131   private String range(String from, String to){
132     if (Strings.isNullOrEmpty(from)) {
133       from = WILDCARD;
134     }
135     if (Strings.isNullOrEmpty(to)) {
136       to = WILDCARD;
137     }
138     return from + "," + to;
139   }
140 
141   private void visit(DisjunctionPredicate or) throws IllegalStateException {
142     State oldState = state;
143     if (state == State.OR) {
144       throw new IllegalStateException("OR within OR filters not supported");
145     }
146     state = State.OR;
147 
148     for (Predicate p : or.getPredicates()) {
149       visit(p);
150     }
151     state = oldState;
152   }
153 
154   private void visit(EqualsPredicate predicate) {
155     addQueryParam(predicate.getKey(), predicate.getValue());
156   }
157 
158   private void visit(LikePredicate predicate) {
159     throw new IllegalArgumentException("LIKE operator not supported in web queries");
160   }
161 
162   private void visit(GreaterThanPredicate predicate) {
163     throw new IllegalArgumentException("GREATER_THAN_OPERATOR operator not supported in web queries");
164   }
165 
166   private void visit(GreaterThanOrEqualsPredicate p) {
167     addQueryParam(p.getKey(), range(p.getValue(), null));
168   }
169 
170   private void visit(LessThanPredicate predicate) {
171     throw new IllegalArgumentException("LESS_THAN_OPERATOR operator not supported in web queries");
172   }
173 
174   private void visit(LessThanOrEqualsPredicate p) {
175     addQueryParam(p.getKey(), range(null, p.getValue()));
176   }
177 
178   private void visit(WithinPredicate within) {
179     addQueryParam(OccurrenceSearchParameter.GEOMETRY, extractPolygonValues(within.getGeometry()));
180   }
181 
182   private String extractPolygonValues(String withinValue) {
183     Matcher m = POLYGON_PATTERN.matcher(withinValue);
184     if (m.find()) {
185       return m.group(1);
186     }
187     throw new IllegalArgumentException("No valid polygon WKT: " + withinValue);
188   }
189 
190   private void visit(InPredicate in) {
191     for (String val : in.getValues()) {
192       addQueryParam(in.getKey(), val);
193     }
194   }
195 
196   private void visit(NotPredicate not) throws IllegalStateException {
197     throw new IllegalArgumentException("NOT operator not supported in web queries");
198   }
199 
200   private void addQueryParam(OccurrenceSearchParameter param, String value) {
201     // verify that last param if existed was the same
202     if (lastParam != null && param != lastParam) {
203       throw new IllegalArgumentException("Mix of search params not supported");
204     }
205 
206     if (!filter.containsKey(param)) {
207       filter.put(param, Lists.<String>newLinkedList());
208     }
209     filter.get(param).add(value);
210     lastParam = param;
211   }
212 
213   private void visit(Predicate p) throws IllegalStateException {
214     Method method = null;
215     try {
216       method = getClass().getDeclaredMethod("visit", new Class[] {p.getClass()});
217     } catch (NoSuchMethodException e) {
218       LOG.warn(
219         "Visit method could not be found. That means a Predicate has been passed in that is unknown to this " + "class",
220         e);
221       throw new IllegalArgumentException("Unknown Predicate", e);
222     }
223     try {
224       method.setAccessible(true);
225       method.invoke(this, p);
226     } catch (IllegalAccessException e) {
227       LOG.error("This should never happen as we set accessible to true explicitly before. Probably a programming error", e);
228       throw new RuntimeException("Programming error", e);
229     } catch (InvocationTargetException e) {
230       LOG.info("Exception thrown while building the Hive Download", e);
231       throw new IllegalArgumentException(e);
232     }
233   }
