14  package org.gbif.api.util.validators.identifierschemes;
15  
16  import java.util.Objects;
17  import java.util.regex.Pattern;
18  
19  public class ResearcherIdValidator implements IdentifierSchemeValidator {
20  
21    public static final Pattern FORMAT_PATTERN =
22        Pattern.compile(
23            "^(?<prefix>http(?:s)?:\\/\\/(?:www.)?.+)?([A-Z]{1,3}-\\d{4}-(19|20)\\d\\d)$");
24  
25    private static final Pattern PUBLONS_PATTERN =
26        Pattern.compile(
27            "^(?<prefix>http(?:s)?:\\/\\/(?:www.)?publons.com\\/researcher\\/)?(([A-Z]{1,3}-\\d{4}-(19|20)\\d\\d)|([0-9]+\\/[a-zA-Z\\-\\_]+\\/?))$");
28  
29    @Override
30    public boolean isValid(String value) {
31      if (value == null || value.isEmpty()) {
32        return false;
33      }
34      return FORMAT_PATTERN.matcher(value).matches() || PUBLONS_PATTERN.matcher(value).matches();
35    }
36  
37    @Override
38    public String normalize(String value) {
39      Objects.requireNonNull(value, "Identifier value can't be null");
40      String trimmedValue = value.trim();
41      if (FORMAT_PATTERN.matcher(trimmedValue).matches()
42          || PUBLONS_PATTERN.matcher(trimmedValue).matches()) {
43        return trimmedValue;
44      }
45      throw new IllegalArgumentException(value + " it not a valid ResearcherID");
46    }
47  }
