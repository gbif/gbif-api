14  package org.gbif.api.util.validators.identifierschemes;
15  
16  public enum IdentifierScheme {
17    ORCID("https://orcid.org"),
18    WIKIDATA("https://www.wikidata.org"),
19    ISNI("http://www.isni.org"),
20    VIAF("https://viaf.org"),
21    HUH("https://kiki.huh.harvard.edu"),
22    RESEARCHER_ID("http://www.researcherid.com"),
23    IH_IRN("http://sweetgum.nybg.org"),
24    OTHER("");
25  
26    private String schemeURI;
27  
28    IdentifierScheme(String schemeURI) {
29      this.schemeURI = schemeURI;
30    }
31  
32    public String getSchemeURI() {
33      return schemeURI;
34    }
35  }
