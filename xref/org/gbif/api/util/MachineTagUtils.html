22    /**
23     * Returns a copy of the original list of machine taggable entities, that have the machine tag.
24     */
25    public static <T extends MachineTaggable> List<T> filter(List<T> source, @Nullable final String namespace, @Nullable final String name, @Nullable final String value) {
26      return Lists.newArrayList(Iterables.filter(source, new Predicate<T>() {
27  
28        @Override
29        public boolean apply(@Nullable T e) {
30          for (MachineTag t : e.getMachineTags()) {
31            if ((namespace == null || t.getNamespace().equals(namespace)) &&
32                (name == null || t.getName().equals(name)) &&
33                (value == null || t.getValue().equals(value)))
34            {
35              return true;
36            }
37          }
38          return false;
39        }
40      }));
41    }

101   public static List<MachineTag> list(MachineTaggable taggable, String namespace, String tagName) {
102     List<MachineTag> tags = Lists.newArrayList();
103     for (MachineTag mt : taggable.getMachineTags()) {
104       if (mt.getNamespace().equals(namespace) && mt.getName().equals(tagName)) {
