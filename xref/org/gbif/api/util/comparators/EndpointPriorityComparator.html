14  package org.gbif.api.util.comparators;
15  
16  import org.gbif.api.model.registry.Endpoint;
17  import org.gbif.api.vocabulary.EndpointType;
18  
19  import java.io.Serializable;
20  import java.util.Arrays;
21  import java.util.Collections;
22  import java.util.Comparator;
23  import java.util.List;
24  
25  /**
26   * Compares two Endpoints.
27   * <p></p>
28   * It does so by using a priority list of Endpoint Types. This Comparator will throw a {@link
29   * ClassCastException} exception if a non supported EndpointType is passed in. It also does not
30   * support {@code null} values.
31   * <p></p>
32   * The priority list is as follows (most to least important):
33   * <ol>
34   * <li>CamtrapDP</li>
35   * <li>DwC-A</li>
36   * <li>TAPIR</li>
37   * <li>BioCASe</li>
38   * <li>DiGIR</li>
39   * <li>DiGIR (Manis)</li>
40   * <li>EML</li>
41   * </ol>
42   */
43  public class EndpointPriorityComparator implements Comparator<Endpoint>, Serializable {
44  
45    /* Note: Due to legacy reasons this is a Comparator itself instead of just using the PRIORITY_COMPARATOR.
46     It should be easy to remove this class or change it into a factory */
47  
48    // Priorities from lowest to highest
49    public static final List<EndpointType> PRIORITIES = Collections.unmodifiableList(
50      Arrays.asList(
51        EndpointType.EML,
52        EndpointType.DIGIR_MANIS,
53        EndpointType.DIGIR,
54        EndpointType.BIOCASE,
55        EndpointType.TAPIR,
56        EndpointType.BIOCASE_XML_ARCHIVE,
57        EndpointType.DWC_ARCHIVE,
58        EndpointType.CAMTRAP_DP
59      ));
60  
61    private static final long serialVersionUID = 8085216142750609841L;
62  
63    @Override
64    public int compare(Endpoint../../../../org/gbif/api/model/registry/Endpoint.html#Endpoint">Endpoint endpoint1, Endpoint endpoint2) {
65      int firstEndpointTypeIndex = PRIORITIES.indexOf(endpoint1.getType());
66      int secondEndpointTypeIndex = PRIORITIES.indexOf(endpoint2.getType());
67  
68      if (firstEndpointTypeIndex == -1) {
69        throw new ClassCastException("Cannot compare value: " + endpoint1.getType());
70      } else if (secondEndpointTypeIndex == -1) {
71        throw new ClassCastException("Cannot compare value: " + endpoint2.getType());
72      } else {
73        return Integer.compare(firstEndpointTypeIndex, secondEndpointTypeIndex);
74      }
75    }
76  }
