1   package org.gbif.api.util.comparators;
2   
3   import java.io.Serializable;
4   import java.util.Comparator;
5   import java.util.List;
6   
7   import org.gbif.api.model.registry.Endpoint;
8   import org.gbif.api.vocabulary.EndpointType;
9   
10  import com.google.common.collect.ImmutableList;
11  import com.google.common.collect.Ordering;
12  
13  /**
14   * Compares two Endpoints.
15   * <p></p>
16   * It does so by using a priority list of Endpoint Types.
17   * This Comparator will throw a {@link ClassCastException} exception if a non supported EndpointType is passed in. It
18   * also does not support {@code null} values.
19   * <p></p>
20   * The priority list is as follows (most to least important):
21   * <ol>
22   * <li>DwC-A</li>
23   * <li>TAPIR</li>
24   * <li>BioCASe</li>
25   * <li>DiGIR</li>
26   * <li>DiGIR (Manis)</li>
27   * <li>EML</li>
28   * </ol>
29   */
30  public class EndpointPriorityComparator implements Comparator<Endpoint>, Serializable {
31  
32    /* Note: Due to legacy reasons this is a Comparator itself instead of just using the PRIORITY_COMPARATOR.
33     It should be easy to remove this class or change it into a factory */
34  
35    // Priorities from lowest to highest
36    public static final List<EndpointType> PRIORITIES = ImmutableList.of(
37        EndpointType.EML,
38        EndpointType.DIGIR_MANIS,
39        EndpointType.DIGIR,
40        EndpointType.BIOCASE,
41        EndpointType.TAPIR,
42        EndpointType.BIOCASE_XML_ARCHIVE,
43        EndpointType.DWC_ARCHIVE
44    );
45  
46    private static final long serialVersionUID = 8085216142750609841L;
47    private static final Ordering<EndpointType> PRIORITY_COMPARATOR = Ordering.explicit(PRIORITIES);
48  
49    @Override
50    public int compare(Endpoint endpoint1, Endpoint endpoint2) {
51      return PRIORITY_COMPARATOR.compare(endpoint1.getType(), endpoint2.getType());
52    }
53  
54  }
