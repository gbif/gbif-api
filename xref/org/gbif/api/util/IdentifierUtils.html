20     * Creates a http link for an identifier if possible by passing it to some known resolvers for the specific id type.
21     * If no link can be constructed, null is returned.
22     *
23     * @param identifier Identifier's identifier
24     * @param type Identifier's type
25     *
26     * @return the url or null if it cannot be created
27     */
28    @Nullable
29    public static String getIdentifierLink(String identifier, IdentifierType type) {
30      if (identifier == null || type == null) {
31        return null;
32      }
33      switch (type) {
34        case HANDLER:
35        case URI:
36        case URL:
37        case FTP:
38          return identifier;
39        case DOI:
40          return "http://dx.doi.org/" + identifier;
41        case LSID:
42          return "http://lsid.tdwg.org/summary/" + identifier;
43        case GBIF_PORTAL:
44          return "http://data.gbif.org/datasets/resource/" + identifier;
45      }
46      return null;
47    }
