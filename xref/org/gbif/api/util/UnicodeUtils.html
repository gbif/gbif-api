1   package org.gbif.api.util;
2   
3   import java.text.Normalizer;
4   
5   import org.apache.commons.lang3.StringUtils;
6   
7   /**
8    * Utilities dealing with unicode strings
9    */
10  public class UnicodeUtils {
11  
12      /**
13       * Replaces all diacretics with their ascii counterpart.
14       */
15      public static String ascii(String x) {
16          if (x == null) {
17              return null;
18          }
19          // manually normalize characters not dealt with by the java Normalizer
20          x = StringUtils.replaceChars(x, "øØðÐ", "oOdD");
21  
22          // use java unicode normalizer to remove accents and punctuation
23          x = Normalizer.normalize(x, Normalizer.Form.NFD);
24          x = x.replaceAll("\\p{M}", "");
25          return x;
26      }
27  
28      /**
29       * Replaces all digraphs and ligatures with their underlying 2 latin letters.
30       *
31       * @param x the string to decompose
32       */
33      public static String decompose(String x) {
34          if (x == null) {
35              return null;
36          }
37          return x.replaceAll("æ", "ae")
38                  .replaceAll("Æ", "Ae")
39                  .replaceAll("œ", "oe")
40                  .replaceAll("Œ", "Oe")
41                  .replaceAll("Ĳ", "Ij")
42                  .replaceAll("ĳ", "ij")
43                  .replaceAll("ǈ", "Lj")
44                  .replaceAll("ǉ", "lj")
45                  .replaceAll("ȸ", "db")
46                  .replaceAll("ȹ", "qp")
47                  .replaceAll("ß", "ss")
48                  .replaceAll("ﬆ", "st")
49                  .replaceAll("ﬅ", "ft")
50                  .replaceAll("ﬀ", "ff")
51                  .replaceAll("ﬁ", "fi")
52                  .replaceAll("ﬂ", "fl")
53                  .replaceAll("ﬃ", "ffi")
54                  .replaceAll("ﬄ", "ffl");
55      }
56  }
