17  /**
18   * Factory constructing dataset iterables that exclude deleted datasets.
19   */
20  public class DatasetPagerFactory {
21      private static final Logger LOG = LoggerFactory.getLogger(DatasetPagerFactory.class);
22  
23      /**
24       * @param key a valid dataset, organization or installation key. If null all datasets will be iterated over
25       * @throws IllegalArgumentException if given key is not existing
26       */
27      public static Iterable<Dataset> datasets(@Nullable UUID key, @Nullable DatasetType type, DatasetService ds, OrganizationService os, InstallationService is) {
28          return datasets(key, type, ds, os, is, PagingConstants.DEFAULT_PARAM_LIMIT);
29      }
30  
31      /**
32       * Returns a dataset iterable by testing the given registry key first to see whether it is a dataset, organization or installation.
33       * In case of an organization key the published datasets will be returned.
34       *
35       * @param key a valid dataset, organization or installation key. If null all datasets will be iterated over
36       * @param pageSize to use when talking to the registry
37       * @throws IllegalArgumentException if given key is not existing
38       */
39      public static Iterable<Dataset> datasets(@Nullable UUID key, @Nullable DatasetType type, DatasetService ds, OrganizationService os, InstallationService is, int pageSize) {
40          if (key == null) {
41              LOG.info("Iterate over all {} datasets", type == null ? "" : type);
42              return new DatasetPager(ds, type, pageSize);
43  
44          } else if (isDataset(key, ds)) {
45              LOG.info("Iterate over dataset {}", key);
46              return ImmutableList.of(ds.get(key));
47  
48          } else if (isOrganization(key, os)) {
49              LOG.info("Iterate over all {} datasets published by {}", type == null ? "" : type, key);
50              return new OrgPublishingPager(os, key, type, pageSize);
51  
52          } else if (isInstallation(key, is)) {
53              LOG.info("Iterate over all {} datasets hosted by installation {}", type == null ? "" : type, key);
54              return new InstallationPager(is, key, type, pageSize);
55          }
56          throw new IllegalArgumentException("Given key is no valid GBIF registry key: " + key);
57      }
58  
59      /**
60       * @param type an optional filter to just include the given dataset type
61       */
62      public static Iterable<Dataset> allDatasets(@Nullable DatasetType type, DatasetService service) {
63          LOG.info("Iterate over all {} datasets", type == null ? "" : type);
64          return new DatasetPager(service, type, PagingConstants.DEFAULT_PARAM_LIMIT);
65      }
66  
67      /**
68       * @param key a valid organization key
69       * @param type an optional filter to just include the given dataset type
70       */
71      public static Iterable<Dataset> publishedDatasets(UUID key, @Nullable DatasetType type, OrganizationService service) {
72          LOG.info("Iterate over all {} datasets published by {}", type == null ? "" : type, key);
73          return new OrgPublishingPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
74      }
75  
76      /**
77       * @param key a valid organization key
78       * @param type an optional filter to just include the given dataset type
79       */
80      public static Iterable<Dataset> hostedDatasets(UUID key, @Nullable DatasetType type, OrganizationService service) {
81          LOG.info("Iterate over all {} datasets hosted organization by {}", type == null ? "" : type, key);
82          return new OrgHostingPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
83      }
84  
85      /**
86       * @param key a valid installation key
87       * @param type an optional filter to just include the given dataset type
88       */
89      public static Iterable<Dataset> hostedDatasets(UUID key, @Nullable DatasetType type, InstallationService service) {
90          LOG.info("Iterate over all {} datasets hosted organization by {}", type == null ? "" : type, key);
91          return new InstallationPager(service, key, type, PagingConstants.DEFAULT_PARAM_LIMIT);
92      }
93  
94      private static boolean isDataset(UUID key, DatasetService ds) {
95          return ds.get(key) != null;
96      }
97  
98      private static boolean isOrganization(UUID key, OrganizationService os) {
99          return os.get(key) != null;
100     }
101 
102     private static boolean isInstallation(UUID key, InstallationService is) {
103         return is.get(key) != null;
104     }
105 }
