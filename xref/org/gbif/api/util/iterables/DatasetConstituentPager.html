8   import java.util.UUID;

10  /**
11   * Iterates over all dataset constituents of a given super dataset.
12   */
13  public class DatasetConstituentPager extends DatasetBasePager {
14  
15      private final DatasetService service;
16      private final UUID datasetKey;
17  
18      public DatasetConstituentPager(DatasetService service, UUID datasetKey, int pageSize) {
19          super(null, pageSize);
20          this.service = service;
21          this.datasetKey = datasetKey;
22      }
23  
24      @Override
25      PagingResponse<Dataset> nextPage(PagingRequest page) {
26          return service.listConstituents(datasetKey, page);
27      }
28  
29  }
