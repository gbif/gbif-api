6   import javax.annotation.Nullable;
7   
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;
10  
11  /**
12   * Iterator over datasets from paging responses that filters out deleted and constituent datasets
13   * It also allows for an optional type filter.
14   */
15  abstract class DatasetBasePager extends EntityPager<Dataset> {
16      private static final Logger LOG = LoggerFactory.getLogger(DatasetBasePager.class);
17      private final DatasetType type;
18  
19      /**
20       * @param pageSize to use when talking to the registry
21       * @param type the accepted dataset type, null for all
22       */
23      public DatasetBasePager(@Nullable DatasetType type, int pageSize) {
24          super(pageSize);
25          this.type = type;
26      }
27  
28      @Override
29      protected boolean exclude(Dataset d) {
30          if (d.getParentDatasetKey() != null) {
31              LOG.debug("Ignore constituent dataset {} {} of parent {}", d.getKey(), d.getTitle().replaceAll("\n", " "), d.getParentDatasetKey());
32          } else if (type != null && d.getType() != type) {
33              LOG.debug("Ignore {} dataset {}: {}", d.getType(), d.getKey(), d.getTitle().replaceAll("\n", " "));
34          }
35          return false;
36      }
37  }
