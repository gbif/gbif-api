8   /**
9    * A typed dataset key for both registered and external datasets.
10   * External dataset keys are compound keys based on the originating network UUID key and a free String based
11   * local identifier. The compound external datasetKey is structured as follows in its serialized form:
12   * <pre>{NETWORK_KEY}:{DATASET_ID}</pre>
13   *
14   * This class is intended as a utility class, lacks mutability and therefore is not suitable serialization via Jackson!
15   */
16  public class DatasetKey {
17  
18    private static final int UUID_LENGTH = 36;
19    private static final int EXTERNAL_KEY_DATASET_ID_START_INDEX = UUID_LENGTH + 1;
20    private static final char DELIMITER = ':';
21  
22    private final UUID registryKey;
23    private final String datasetId;
24  
25  
26    /**
27     * Parses a datasetKey which can be either a compound external key or a simple UUID for registered datasets.
28     *
29     * @param key a simple registered dataset uuid key or an external compound dataset key
30     *
31     * @return a valid DatasetKey instance
32     *
33     * @throws IllegalArgumentException for invalid external or registered dataset keys
34     */
35    public static DatasetKey fromString(String key) {
36      Preconditions.checkNotNull(key);
37      if (key.length() == UUID_LENGTH) {
38        // a registered dataset key
39        return new DatasetKey(UUID.fromString(key));
40  
41      } else if (key.length() > EXTERNAL_KEY_DATASET_ID_START_INDEX && DELIMITER == key.charAt(UUID_LENGTH)) {
42        // an external dataset key
43        return new DatasetKey(UUID.fromString(key.substring(0, UUID_LENGTH)),
44          key.substring(EXTERNAL_KEY_DATASET_ID_START_INDEX));
45      }
46  
47      throw new IllegalArgumentException("Invalid dataset key " + key);
48    }
49  
50    /**
51     * Constructor for a registered dataset key.
52     */
53    public DatasetKey(UUID datasetKey) {
54      this.registryKey = Preconditions.checkNotNull(datasetKey);
55      this.datasetId = null;
56    }
57  
58    /**
59     * Constructor for an external dataset key.
60     *
61     * @param networkKey of the originating network
62     * @param datasetId  the local dataset id within the network
63     */
64    public DatasetKey(UUID networkKey, String datasetId) {
65      this.registryKey = Preconditions.checkNotNull(networkKey);
66      this.datasetId = datasetId.trim();
67      if (Strings.isNullOrEmpty(this.datasetId)) {
68        throw new IllegalArgumentException("datasetId has to have non whitespace characters");
69      }
70    }
71  
72    /**
73     * A registered UUID of either a dataset or the network of origin in the case of external datasets.
74     *
75     * @return the registered network or dataset key
76     */
77    public UUID getRegistryKey() {
78      return registryKey;
79    }
80  
81    /**
82     * @return the local identifier of an external dataset key or null if its a registered dataset
83     */
84    public String getDatasetId() {
85      return datasetId;
86    }
87  
88    public boolean isExternalKey() {
89      return datasetId != null;
90    }
91  
92    /**
93     * Generates the serialized dataset key as used in the {@link org.gbif.api.model.registry.Dataset} class.
94     *
95     * @return the serialized dataset key
96     */
97    public String toDatasetKey() {
98      if (isExternalKey()) {
99        return registryKey.toString() + DELIMITER + datasetId;
100     }
101     return registryKey.toString();
102   }
