47    /**
48     * Matches ranges in formats
49     *
50     * <pre>
51     * 23.1,55.2
52     * </pre>
53     *
54     * <pre>
55     * *,88
56     * </pre>
57     *
58     * and
59     *
60     * <pre>
61     * 55,*
62     * </pre>
63     *
64     * .
65     * The matcher returns 2 groups:
66     * group 1: lower bound
67     * group 2: upper bound
68     */
69    private static final Pattern DECIMAL_RANGE_PATTERN = Pattern.compile(
70      "^" + DECIMAL_OR_WILDCARD + "\\s*,\\s*" + DECIMAL_OR_WILDCARD + "$", Pattern.CASE_INSENSITIVE);
71    /**
72     * Matches without groups a space delimited coordinate pair allowing prefixing or trailing whitespace: 10.12 -90.12
73     */
74    private static final String WKT_COORD = "\\s*" + DECIMAL + "\\s+" + DECIMAL + "\\s*";
75  
76    /**
77     * Matches without groups a comma separated list of coordinates enclosed in brackets:
78     * (10.12 -90.12, 30 10, 10 20, 20 40)
79     */
80    private static final String WKT_LINE = "\\s*\\(" + WKT_COORD + "(?:," + WKT_COORD + ")*\\)\\s*";
81  
82    private static final List<Pattern> WKT_PATTERNS = ImmutableList.of(
83      Pattern.compile("^POINT\\s*\\(" + WKT_COORD + "\\)$", Pattern.CASE_INSENSITIVE),
84      Pattern.compile("^(?:LINESTRING|LINEARRING)\\s*" + WKT_LINE + "$", Pattern.CASE_INSENSITIVE),
85      Pattern.compile("^POLYGON\\s*\\(" + WKT_LINE + "(?:," + WKT_LINE + ")*\\)$", Pattern.CASE_INSENSITIVE));

94    /**
95     * Builds a closed range from two given values, but accepts nulls which are converted into unbound ranges
96     * in the guava instance.
97     */
98    public static <T extends Comparable<?>> Range<T> buildRange(final T lower, final T upper) {
99      if (lower == null && upper != null) {
100       return Ranges.atMost(upper);
101 
102     } else if (lower != null && upper == null) {
103       return Ranges.atLeast(lower);
104 
105     } else if (lower == null && upper == null) {
106       return Ranges.<T>all();
107 
108     } else {
109       return Ranges.closed(lower, upper);
110     }
111   }

113   /**
114    * Determines whether the value given is a valid decimal or date range, delimiting two values by a comma.
115    *
116    * @return true if the given value is a valid range
117    */
118   public static boolean isRange(String value) {
119     if (!Strings.isNullOrEmpty(value)) {
120       // decimal range for ints or doubles
121       if (DECIMAL_RANGE_PATTERN.matcher(value).find()) {
122         return true;
123       }
124       // check date range
125       try {
126         IsoDateParsingUtils.parseDateRange(value);
127         return true;
128       } catch (Exception e) {
129         // aha, no range
130       }
131     }
132     return false;
133   }

146   /**
147    * Parses a decimal range in the format 123.1,456.
148    *
149    * @return the parsed range with wildcards represented as null values
150    * @throws IllegalArgumentException if value is invalid or null
151    */
152   public static Range<Double> parseDecimalRange(String value) {
153     if (!Strings.isNullOrEmpty(value)) {
154       Matcher m = DECIMAL_RANGE_PATTERN.matcher(value);
155       if (m.find()) {
156         return buildRange(parseDouble(m.group(1)), parseDouble(m.group(2)));
157       }
158     }
159     throw new IllegalArgumentException("Invalid decimal range: " + value);
160   }

179   /**
180    * Validates that a given parameter value matches the expected type of the parameter as defined by
181    * {@link SearchParameter#type()} and throws an IllegalArgumentException otherwise.
182    *
183    * @param param the search parameter defining the expected type
184    * @param value the parameter value to be validated
185    * @throws IllegalArgumentException if the value cannot be converted to the expected type
186    */
187   public static void validate(SearchParameter param, String value) throws IllegalArgumentException {
188     final Class<?> pType = param.type();
189 
190     try {
191       if (OccurrenceSearchParameter.GEOMETRY == param) {
192         validateGeometry(value);
193 
194       } else if (OccurrenceSearchParameter.DECIMAL_LATITUDE == param) {
195         validateLatitude(value);
196 
197       } else if (OccurrenceSearchParameter.DECIMAL_LONGITUDE == param) {
198         validateLongitude(value);
199 
200       } else if (UUID.class.isAssignableFrom(pType)) {
201         UUID.fromString(value);
202 
203       } else if (Double.class.isAssignableFrom(pType)) {
204         validateDouble(value);
205 
206       } else if (Integer.class.isAssignableFrom(pType)) {
207         Collection<Integer> intsFound = validateInteger(value);
208         if (OccurrenceSearchParameter.MONTH == param) {
209           validateMonth(intsFound);
210         }
211 
212       } else if (Boolean.class.isAssignableFrom(pType)) {
213         // we cannot use Boolean.parseBoolean as this accepted anything as false
214         if (!BOOLEAN.matcher(value).find()) {
215           throw new IllegalArgumentException("Value " + value + " is no valid boolean");
216         }
217 
218       } else if (Country.class.isAssignableFrom(pType)) {
219         // iso codes expected
220         Preconditions.checkNotNull(Country.fromIsoCode(value));
221 
222       } else if (Language.class.isAssignableFrom(pType)) {
223         // iso codes expected
224         Preconditions.checkNotNull(Language.fromIsoCode(value));
225 
226       } else if (Enum.class.isAssignableFrom(pType)) {
227         // enum value expected, cast to enum
228         @SuppressWarnings("unchecked")
229         Class<? extends Enum<?>> eType = (Class<? extends Enum<?>>) pType;
230         Preconditions.checkNotNull(VocabularyUtils.lookupEnum(value, eType));
231 
232       } else if (Date.class.isAssignableFrom(pType)) {
233         // ISO date strings
234         validateDate(value);
235 
236       } else if (String.class.isAssignableFrom(pType)) {
237         // any string allowed
238 
239       } else {
240         // an unexpected data type - update this method!!
241         throw new IllegalArgumentException("Unknown SearchParameter data type " + pType.getCanonicalName());
242       }
243 
244     } catch (NullPointerException e) {
245       // Preconditions.checkNotNull throws NPE but we want IllegalArgumentException
246       throw new IllegalArgumentException("Value " + value + " invalid for filter parameter " + param, e);
247     }
248   }

285    * Validates if the value is a valid single double or a range of double values.
286    * If the value is a range each limit is validated and the wildcard character '*' is skipped.

288   private static void validateDouble(String value) {
289     if (Strings.isNullOrEmpty(value)) {
290       throw new IllegalArgumentException("Double cannot be null or empty");

300   /**
301    * Validates if the value is a valid single double and its value is between a range.
302    */
303   private static void validateDoubleInRange(String value, Range<Double> range, String errorMsg) {
304     if (Strings.isNullOrEmpty(value)) {
305       throw new IllegalArgumentException("Double cannot be null or empty");
306     }
307     try {
308       final Double doubleValue = Double.parseDouble(value);
309       if (!range.contains(doubleValue)) {
310         throw new IllegalArgumentException(String.format(errorMsg, value));
311       }
312     } catch (NumberFormatException e) {
313       if (isRange(value)) {
314         Range<Double> rangeValue = parseDecimalRange(value);
315         if (!range.encloses(rangeValue)) {
316           throw new IllegalArgumentException(String.format(errorMsg, value));
317         }
318       } else {
319         throw new IllegalArgumentException("Argument is not a valid number");
320       }
321     }
322   }

324   /**
325    * Verify that we have indeed a wellKnownText parameter.
326    * See <a href="http://en.wikipedia.org/wiki/Well-known_text">wikipedia</a> for basic WKT specs.
327    * The validation implemented does both syntactic and topological validation (for polygons only): only convex polygons
328    * are accepted.
329    */
330   private static void validateGeometry(String wellKnownText) {
331     validateGeometrySyntax(wellKnownText);
332     try {
333       Geometry geometry = new WKTReader().read(wellKnownText);
334       // Calculating the area > 0 ensures that polygons that are representing lines or points are invalidated
335       if (geometry instanceof Polygon && (!geometry.isValid() || geometry.getArea() == 0.0)) {
336         throw new IllegalArgumentException("Invalid polygon " + wellKnownText);
337       }
338     } catch (ParseException e) {
339       throw new IllegalArgumentException("Invalid simple WKT: " + wellKnownText);
340     }
341   }

343   /**
344    * Verify that we have indeed a wellKnownText parameter.
345    * See <a href="http://en.wikipedia.org/wiki/Well-known_text">wikipedia</a> for basic WKT specs.
346    * The validation implemented does a basic syntax check for the following geometries, but does not
347    * verify that the resulting geometries are topologically valid (see the OGC SFS specification).
348    */
349   private static void validateGeometrySyntax(String wellKnownText) {
350     if (Strings.isNullOrEmpty(wellKnownText)) {
351       throw new IllegalArgumentException("Well Known Text cannot be empty or null");
352     }
353     // test all 4 supported geometry types with their respective regex - one must match!
354     for (Pattern regex : WKT_PATTERNS) {
355       if (regex.matcher(wellKnownText).find()) {
356         return;
357       }
358     }
359     throw new IllegalArgumentException("Invalid simple WKT: " + wellKnownText);
360   }

362   /**
363    * Validates if the value is a valid single integer or a range of integer values.
364    * If the value is a range each limit is validated and the wildcard character '*' is skipped.
365    *
366    * @throws IllegalArgumentException if value is invalid or null
367    */
368   private static Collection<Integer> validateInteger(String value) {
369     if (Strings.isNullOrEmpty(value)) {
370       throw new IllegalArgumentException("Integer cannot be null or empty");
371     }
372     try {
373       return Lists.newArrayList(Integer.parseInt(value));
374     } catch (NumberFormatException e) {
375       Range<Integer> range = parseIntegerRange(value);
376       List<Integer> ints = Lists.newArrayList();
377       if (range.hasLowerBound()) {
378         ints.add(range.lowerEndpoint());
379       }
380       if (range.hasUpperBound()) {
381         ints.add(range.upperEndpoint());
382       }
383       return ints;
384     }
385   }
