3   import java.lang.annotation.Inherited;
4   import java.lang.annotation.Retention;
5   import java.lang.annotation.RetentionPolicy;
6   import java.lang.annotation.Target;
7   
8   import io.swagger.v3.oas.annotations.Parameter;
9   import io.swagger.v3.oas.annotations.enums.ParameterIn;
10  import io.swagger.v3.oas.annotations.media.Schema;
11  
12  import static java.lang.annotation.ElementType.ANNOTATION_TYPE;
13  import static java.lang.annotation.ElementType.METHOD;
14  
15  public class CommonParameters {
16  
17    /**
18     * The usual (search) limit and offset parameters
19     */
20    @Target({METHOD, ANNOTATION_TYPE})
21    @Retention(RetentionPolicy.RUNTIME)
22    @Inherited
23    @Parameter(
24      name = "hl",
25      description = "Set `hl=true` to highlight terms matching the query when in fulltext search fields. The highlight " +
26        "will be an emphasis tag of class `gbifHl`.",
27      schema = @Schema(implementation = Boolean.class),
28      in = ParameterIn.QUERY)
29    public @interface HighlightParameter {}
30  
31    /**
32     * A normal full-text search parameter.
33     */
34    @Target({METHOD, ANNOTATION_TYPE})
35    @Retention(RetentionPolicy.RUNTIME)
36    @Inherited
37    @Parameter(
38      name = "q",
39      description = "Simple full text search parameter. The value for this parameter can be a simple word or a phrase. " +
40        "Wildcards are not supported",
41      schema = @Schema(implementation = String.class),
42      in = ParameterIn.QUERY)
43    public @interface QParameter {}
44  }
