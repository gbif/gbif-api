13  package org.gbif.api.service.metrics;

15  import org.gbif.api.model.metrics.cube.ReadBuilder;
16  import org.gbif.api.model.metrics.cube.Rollup;

18  import java.util.List;

20  /**
21   * The cube API service, for reading addressable counts from a basic cube.
22   */
23  public interface CubeService {
24  
25    /**
26     * Using the supplied {@link ReadBuilder} to obtain the address, looks up the cube value.
27     * Should the cube support no dimensions (e.g. count all), then calling with an empty ReadBuilder
28     * will return this.  E.g. cubeService.get(new ReadBuilder());
29     *
30     * @param addressBuilder To obtain the address at which to look up from the cube
31     * @return The value which might be 0. A value of 0 means that the count is truly at 0
32     * @throws IllegalArgumentException Should the addressBuilder provide an address that does not exist in the cube
33     */
34    long get(ReadBuilder addressBuilder) throws IllegalArgumentException;
35  
36    /**
37     * Provides the list of rollups thus specifying the available combinations of addressable dimensions for a cube.
38     *
39     * @return The schema for the cube (defined by the queryable addresses)
40     */
41    List<Rollup> getSchema();
42  }
