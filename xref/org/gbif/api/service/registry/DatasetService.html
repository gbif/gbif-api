29  public interface DatasetService
30    extends NetworkEntityService<Dataset> {
31  
32    /**
33     * Pages through constituents of a dataset, i.e. returns datasets which have a parentDatasetKey
34     * equals to the one requested.
35     * 
36     * @param datasetKey the parent datasets key
37     */
38    PagingResponse<Dataset> listConstituents(UUID datasetKey, @Nullable Pageable page);
39  
40    /**
41     * Pages through all constituent datasets, i.e. returns datasets which have a non null parentDatasetKey.
42     */
43    PagingResponse<Dataset> listConstituents(@Nullable Pageable page);
44  
45    /**
46     * Provides paging service to list datasets published, i.e. owned by organizations from a given country.
47     * 
48     * @param country the hosting country
49     * @param type the optional dataset type filter
50     * @return list of datasets ordered by creation date with latest coming first
51     */
52    PagingResponse<Dataset> listByCountry(Country country, @Nullable DatasetType type, @Nullable Pageable page);
53  
54    /**
55     * Lists all metadata descriptions available for a dataset and optionally filters them by document type.
56     * The list is sorted by priority with the first result ranking highest.
57     * Highest priority in this sense means most relevant for augmenting/updating a dataset with EML being the most
58     * relevant cause informative type.
59     * 
60     * @return the list of metadata entries sorted by priority
61     */
62    List<Metadata> listMetadata(UUID datasetKey, @Nullable MetadataType type);
63  
64    /**
65     * Lists all networks that this dataset is a constituent of.
66     * @param datasetKey the dataset in question
67     * @return list of networks that have this dataset as a constituent
68     */
69    List<Network> listNetworks(UUID datasetKey);
70  
71    /**
72     * Get a metadata description by its key.
73     */
74    Metadata getMetadata(int metadataKey);
75  
76    /**
77     * Removes a metadata entry and its document by its key.
78     */
79    void deleteMetadata(int metadataKey);
80  
81    /**
82     * Inserts a metadata document, replacing any previously existing document of the same type.
83     * The document type is discovered by the service and returned in the Metadata instance.
84     * 
85     * @throws IllegalArgumentException if document is not parsable
86     */
87    Metadata insertMetadata(UUID datasetKey, InputStream document);
88  
89    /**
90     * Retrieves a GBIF generated EML document overlaying GBIF information with any existing metadata document data.
91     */
92    InputStream getMetadataDocument(UUID datasetKey);
93  
94    /**
95     * Gets the actual metadata document content by its key.
96     */
97    InputStream getMetadataDocument(int metadataKey);
98  
99    /**
100    * Provides access to deleted datasets.
101    */
102   PagingResponse<Dataset> listDeleted(@Nullable Pageable page);
103 
104   /**
105    * Provides access to datasets that are marked as a duplicate of another. When 2 datasets are considered duplicates,
106    * one is marked as a duplicate of the other. Only the marked dataset is returned in this call.
107    */
108   PagingResponse<Dataset> listDuplicates(@Nullable Pageable page);
109 
110   /**
111    * Provides access to internal (e.g. not marked as external) datasets, that are not sub datasets that have no
112    * endpoint.
113    */
114   PagingResponse<Dataset> listDatasetsWithNoEndpoint(@Nullable Pageable page);
