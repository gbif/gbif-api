13  package org.gbif.api.service.registry;
14  
15  import org.gbif.api.model.common.paging.Pageable;
16  import org.gbif.api.model.common.paging.PagingResponse;
17  import org.gbif.api.model.registry.Dataset;
18  import org.gbif.api.model.registry.Metadata;
19  import org.gbif.api.model.registry.Network;
20  import org.gbif.api.vocabulary.Country;
21  import org.gbif.api.vocabulary.DatasetType;
22  import org.gbif.api.vocabulary.MetadataType;
23  
24  import java.io.InputStream;
25  import java.util.List;
26  import java.util.UUID;
27  import javax.annotation.Nullable;
28  
29  public interface DatasetService
30    extends NetworkEntityService<Dataset> {
31  
32    /**
33     * Pages through constituents of a dataset, i.e. returns datasets which have a parentDatasetKey
34     * equals to the one requested.
35     *
36     * @param datasetKey the parent datasets key
37     */
38    PagingResponse<Dataset> listConstituents(UUID datasetKey, @Nullable Pageable page);
39  
40    /**
41     * Pages through all constituent datasets, i.e. returns datasets which have a non null parentDatasetKey.
42     */
43    PagingResponse<Dataset> listConstituents(@Nullable Pageable page);
44  
45    /**
46     * Provides paging service to list datasets published, i.e. owned by organizations from a given country.
47     *
48     * @param country the hosting country
49     * @param type the optional dataset type filter
50     * @return list of datasets ordered by creation date with latest coming first
51     */
52    PagingResponse<Dataset> listByCountry(Country country, @Nullable DatasetType type, @Nullable Pageable page);
53  
54    /**
55     * Provides paging service to list datasets published filtered by a particular dataset type.
56     *
57     * @param type the dataset type filter
58     * @return list of datasets ordered by creation date with latest coming first
59     */
60    PagingResponse<Dataset> listByType(DatasetType type, @Nullable Pageable page);
61  
62    /**
63     * Lists all metadata descriptions available for a dataset and optionally filters them by document type.
64     * The list is sorted by priority with the first result ranking highest.
65     * Highest priority in this sense means most relevant for augmenting/updating a dataset with EML being the most
66     * relevant cause informative type.
67     *
68     * @return the list of metadata entries sorted by priority
69     */
70    List<Metadata> listMetadata(UUID datasetKey, @Nullable MetadataType type);
71  
72    /**
73     * Lists all networks that this dataset is a constituent of.
74     * @param datasetKey the dataset in question
75     * @return list of networks that have this dataset as a constituent
76     */
77    List<Network> listNetworks(UUID datasetKey);
78  
79    /**
80     * Get a metadata description by its key.
81     */
82    Metadata getMetadata(int metadataKey);
83  
84    /**
85     * Removes a metadata entry and its document by its key.
86     */
87    void deleteMetadata(int metadataKey);
88  
89    /**
90     * Inserts a metadata document, replacing any previously existing document of the same type.
91     * Updates dataset from metadata document, but only if metadata document does not exist already.
92     * The document type is discovered by the service and returned in the Metadata instance.
93     *
94     * @param datasetKey the dataset in question
95     * @param document   metadata document to insert
96     *
97     * @throws IllegalArgumentException if document is not parsable
98     */
99    Metadata insertMetadata(UUID datasetKey, InputStream document);
100 
101   /**
102    * Retrieves a GBIF generated EML document overlaying GBIF information with any existing metadata document data.
103    */
104   InputStream getMetadataDocument(UUID datasetKey);
105 
106   /**
107    * Gets the actual metadata document content by its key.
108    */
109   InputStream getMetadataDocument(int metadataKey);
110 
111   /**
112    * Provides access to deleted datasets.
113    */
114   PagingResponse<Dataset> listDeleted(@Nullable Pageable page);
115 
116   /**
117    * Provides access to datasets that are marked as a duplicate of another. When 2 datasets are considered duplicates,
118    * one is marked as a duplicate of the other. Only the marked dataset is returned in this call.
119    */
120   PagingResponse<Dataset> listDuplicates(@Nullable Pageable page);
121 
122   /**
123    * Provides access to internal (e.g. not marked as external) datasets, that are not sub datasets that have no
124    * endpoint.
125    */
126   PagingResponse<Dataset> listDatasetsWithNoEndpoint(@Nullable Pageable page);
127 
128   /**
129    * Get a Dataset list from a DOI.
130    * GBIF assigned DOIs and alternate identifiers are returned.
131    * Multiple dataset could share the same DOI since this is not enforced.
132    */
133   PagingResponse<Dataset> listByDOI(String doi, @Nullable Pageable page);
134 }
