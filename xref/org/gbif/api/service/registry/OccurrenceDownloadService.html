13  package org.gbif.api.service.registry;
14  
15  import org.gbif.api.model.common.paging.Pageable;
16  import org.gbif.api.model.common.paging.PagingResponse;
17  import org.gbif.api.model.occurrence.Download;
18  import org.gbif.api.model.registry.DatasetOccurrenceDownloadUsage;
19  import org.gbif.api.vocabulary.Country;
20  
21  import java.util.*;
22  import javax.annotation.Nullable;
23  import javax.validation.constraints.NotNull;
24  
25  /**
26   * Interface to access and persist information about occurrence download events.
27   */
28  public interface OccurrenceDownloadService {
29  
30    /**
31     * Persists the occurrence download object. The object must contain a unique key, the persistence storage doesn't
32     * generate one for it.
33     */
34    void create(@NotNull Download download);
35  
36    /**
37     * Retrieves a occurrence download by its unique key.
38     */
39    Download get(@NotNull String key);
40  
41  
42    /**
43     * Retrieves a pageable result of all the downloads, optionally the downloads can be filtered by status.
44     */
45    PagingResponse<Download> list(@Nullable Pageable page, @Nullable Set<Download.Status> status);
46  
47    /**
48     * Retrieves a pageable result of the downloads created by a user in a given status.
49     */
50    PagingResponse<Download> listByUser(@NotNull String user, @Nullable Pageable page, @Nullable Set<Download.Status> status);
51  
52    /**
53     * Update an existing occurrence download.
54     */
55    void update(@NotNull Download download);
56  
57  
58    /**
59     * Retrieves a pageable result of the dataset usages in a occurrence download.
60     *
61     * The Downloads in the DatasetOccurrenceDownloadUsages are null, to avoid redundant repetition of potentially large
62     * objects.
63     */
64    PagingResponse<DatasetOccurrenceDownloadUsage> listDatasetUsages(@NotNull String downloadKey,
65                                                                     @Nullable Pageable page);
66  
67    /**
68     * Retrieves downloads monthly stats by country (user and publishing country) and dataset.
69     */
70    Map<Integer,Map<Integer,Long>> getDownloadsByUserCountry(@Nullable Date fromDate, @Nullable  Date toDate,
71                                                             @Nullable Country userCountry);
72  
73    /**
74     * Retrieves downloaded records monthly stats by country (user and publishing country) and dataset.
75     */
76    Map<Integer,Map<Integer,Long>> getDownloadedRecordsByDataset(@Nullable Date fromDate, @Nullable  Date toDate,
77                                                                 @Nullable Country publishingCountry,
78                                                                 @Nullable UUID datasetKey);
79    /**
80     * Persists usages of datasets in an occurrence download.
81     * @param downloadKey downloadkey of the datasets usage information.
82     * @param datasetCitations map of datasetkey as key and number of records as value.
83     */
84    void createUsages(@NotNull String downloadKey, @NotNull Map<UUID, Long> datasetCitations);
85  }
