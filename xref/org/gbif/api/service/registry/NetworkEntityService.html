13  package org.gbif.api.service.registry;
14  
15  import org.gbif.api.model.common.paging.Pageable;
16  import org.gbif.api.model.common.paging.PagingResponse;
17  import org.gbif.api.vocabulary.IdentifierType;
18  
19  import java.util.Collection;
20  import java.util.Map;
21  import java.util.UUID;
22  import javax.annotation.Nullable;
23  import javax.validation.constraints.NotNull;
24  
25  public interface NetworkEntityService<T> extends MachineTagService, TagService, CommentService, IdentifierService,
26    EndpointService, ContactService {
27  
28    UUID create(@NotNull T entity);
29  
30    // TODO: Define behavior when it is already deleted or does not exist
31    void delete(@NotNull UUID key);
32  
33    T get(@NotNull UUID key);
34  
35    /**
36     * Retrieves all titles for the requested entity keys in one go
37     */
38    Map<UUID, String> getTitles(Collection<UUID> keys);
39  
40      /**
41       * Used to retrieve a list of network entities.
42       *
43       * To iterate over <em>all</em> entities you can use code like this:
44       * {@code
45       * PagingRequest req = new PagingRequest();
46       * PagingResponse<T> response;
47       * do {
48       *   response = service.list(req);
49       *   for (T obj : response.getResults()) {
50       *     doStuff();
51       *   }
52       *   req.nextPage();
53       * } while (!response.isEndOfRecords());
54       * }
55       *
56       * @return a list of network entities ordered by their creation date, newest coming first
57       */
58    PagingResponse<T> list(@Nullable Pageable page);
59  
60    /**
61     * A simple search that supports paging.
62     *
63     * @return a pageable response of network entities, with accurate counts.
64     */
65    PagingResponse<T> search(String query, @Nullable Pageable page);
66  
67    /**
68     * Lists the entities by the provided identifier, scoped by type.
69     *
70     * @return a pageable response of network entities, with accurate counts for the identifier provided
71     */
72    PagingResponse<T> listByIdentifier(IdentifierType type, String identifier, @Nullable Pageable page);
73  
74    /**
75     * Lists the entities by the provided identifier, which may be of any type.
76     *
77     * @return a pageable response of network entities, with accurate counts for the identifier provided
78     */
79    PagingResponse<T> listByIdentifier(String identifier, @Nullable Pageable page);
80  
81    /**
82     * Lists the entities having a machine tag in the provided namespace, with the provided name and value.
83     *
84     * @return a pageable response of network entities, with accurate counts for the machine tag provided
85     */
86    PagingResponse<T> listByMachineTag(String namespace, @Nullable String name, @Nullable String value, @Nullable Pageable page);
87  
88    void update(@NotNull T entity);
89  
90  }
