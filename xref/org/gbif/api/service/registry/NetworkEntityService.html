13  package org.gbif.api.service.registry;
14  
15  import org.gbif.api.model.common.paging.Pageable;
16  import org.gbif.api.model.common.paging.PagingResponse;
17  import org.gbif.api.vocabulary.IdentifierType;
18  
19  import java.util.UUID;
20  
21  import javax.annotation.Nullable;
22  import javax.validation.constraints.NotNull;
23  
24  public interface NetworkEntityService<T> extends MachineTagService, TagService, CommentService, IdentifierService,
25    EndpointService, ContactService {
26  
27    UUID create(@NotNull T entity);
28  
29    // TODO: Define behavior when it is already deleted or does not exist
30    void delete(@NotNull UUID key);
31  
32    T get(@NotNull UUID key);
33  
34    /**
35     * Used to retrieve a list of network entities.
36     *
37     * To iterate over <em>all</em> entities you can use code like this:
38     * {@code
39     * PagingRequest req = new PagingRequest();
40     * PagingResponse<T> response;
41     * do {
42     *   response = service.list(req);
43     *   for (T obj : response.getResults()) {
44     *     doStuff();
45     *   }
46     *   req.nextPage();
47     * } while (!response.isEndOfRecords());
48     * }
49     *
50     * @return a list of network entities ordered by their creation date, newest coming first
51     */
52    PagingResponse<T> list(@Nullable Pageable page);
53  
54    /**
55     * A simple search that supports paging.
56     * 
57     * @return a pageable response of network entities, with accurate counts.
58     */
59    PagingResponse<T> search(String query, @Nullable Pageable page);
60  
61    /**
62     * Lists the entities by the provided identifier, scoped by type.
63     * 
64     * @return a pageable response of network entities, with accurate counts for the identifier provided
65     */
66    PagingResponse<T> listByIdentifier(IdentifierType type, String identifier, @Nullable Pageable page);
67  
68    /**
69     * Lists the entities by the provided identifier, which may be of any type.
70     * 
71     * @return a pageable response of network entities, with accurate counts for the identifier provided
72     */
73    PagingResponse<T> listByIdentifier(String identifier, @Nullable Pageable page);
74  
75    void update(@NotNull T entity);
76  
77  }
