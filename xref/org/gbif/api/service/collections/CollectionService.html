16  package org.gbif.api.service.collections;
17  
18  import java.util.List;
19  import java.util.UUID;
20  
21  import org.gbif.api.model.collections.Collection;
22  import org.gbif.api.model.collections.request.CollectionSearchRequest;
23  import org.gbif.api.model.collections.view.CollectionView;
24  import org.gbif.api.model.common.paging.Pageable;
25  import org.gbif.api.model.common.paging.PagingResponse;
26  import org.gbif.api.model.registry.search.collections.KeyCodeNameResult;
27  
28  import javax.annotation.Nullable;
29  import javax.validation.constraints.NotNull;
30  
31  /** API Service to work with collections. */
32  public interface CollectionService extends PrimaryCollectionEntityService<Collection> {
33  
34    /**
35     * Pages {@link Collection} entities based on the parameters received.
36     *
37     * <p>To iterate over <em>all</em> entities you can use code like this: {@code PagingRequest req =
38     * new PagingRequest(); PagingResponse<T> response; do { response = service.list(req); for (T obj
39     * : response.getResults()) { doStuff(); } req.nextPage(); } while (!response.isEndOfRecords()); }
40     *
41     * @param searchRequest {@link CollectionSearchRequest} with all the parameters
42     * @return a list of entities ordered by their creation date, newest coming first
43     */
44    PagingResponse<CollectionView> list(CollectionSearchRequest searchRequest);
45  
46    /** Provides access to deleted collections. */
47    PagingResponse<CollectionView> listDeleted(@Nullable UUID replacedBy, @Nullable Pageable page);
48  
49    /** Retrieves a {@link CollectionView} by the collection key. */
50    CollectionView getCollectionView(@NotNull UUID key);
51  
52    /** Provides a simple suggest service. */
53    List<KeyCodeNameResult> suggest(@Nullable String q);
54  
55    /**
56     * Creates a {@link Collection} from a {@link org.gbif.api.model.registry.Dataset}.
57     *
58     * @param datasetKey key of the dataset to create the collection from
59     * @param collectionCode the code to assign to the collection since it can't be inferred from the
60     *     dataset
61     * @return UUID of the created collection
62     */
63    UUID createFromDataset(UUID datasetKey, String collectionCode);
64  }
