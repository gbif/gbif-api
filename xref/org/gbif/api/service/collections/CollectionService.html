15  /**
16   * API Service to work with collections.
17   */
18  public interface CollectionService
19      extends CrudService<Collection>, ContactService, TagService, IdentifierService, MachineTagService {
20  
21    /**
22     * Pages {@link Collection} entities based on the parameters received.
23     *
24     * <p>To iterate over <em>all</em> entities you can use code like this: {@code PagingRequest req =
25     * new PagingRequest(); PagingResponse<T> response; do { response = service.list(req); for (T obj
26     * : response.getResults()) { doStuff(); } req.nextPage(); } while (!response.isEndOfRecords()); }
27     *
28     * @param query to make a full text search
29     * @param institutionKey key of an collection to filter by
30     * @param contactKey to filter by a contact
31     * @param code code of the collection
32     * @param name name of the collection
33     * @param page paging parameters
34     * @return a list of entities ordered by their creation date, newest coming first
35     */
36    PagingResponse<Collection> list(
37        @Nullable String query,
38        @Nullable UUID institutionKey,
39        @Nullable UUID contactKey,
40        @Nullable String code,
41        @Nullable String name,
42        @Nullable Pageable page);
43  
44    /**
45     * Provides access to deleted collections.
46     */
47    PagingResponse<Collection> listDeleted(@Nullable Pageable page);
48  
49    /**
50     * Provides a simple suggest service.
51     */
52    List<KeyCodeNameResult> suggest(@Nullable String q);
53  }
