1   package org.gbif.api.service.collections;
2   
3   import java.io.IOException;
4   import java.io.InputStream;
5   import java.util.UUID;
6   
7   import org.gbif.api.model.collections.descriptors.DescriptorChangeSuggestion;
8   import org.gbif.api.model.collections.descriptors.DescriptorChangeSuggestionRequest;
9   import org.gbif.api.model.collections.suggestions.Status;
10  import org.gbif.api.model.collections.suggestions.Type;
11  import org.gbif.api.model.common.paging.Pageable;
12  import org.gbif.api.model.common.paging.PagingResponse;
13  import org.gbif.api.vocabulary.Country;
14  
15  /**
16   * Service interface for managing descriptor change suggestions in collections.
17   * This service handles the creation, updating, and management of descriptor change suggestions,
18   * including file handling and status management.
19   *
20   * Note: File storage is handled internally using unique identifiers. The provided filename
21   * is used only for display and metadata purposes.
22   */
23  public interface DescriptorChangeSuggestionService {
24  
25    /**
26     * Creates a new descriptor change suggestion.
27     *
28     * @param fileStream The input stream containing the descriptor data
29     * @param filename The original filename (used for display and metadata only)
30     * @param request The suggestion request containing metadata
31     * @return The created DescriptorChangeSuggestion
32     * @throws IOException If there's an error processing the file
33     * @throws IllegalArgumentException If the request is invalid
34     */
35    DescriptorChangeSuggestion createSuggestion(InputStream fileStream, String filename, DescriptorChangeSuggestionRequest request) throws IOException;
36  
37    /**
38     * Updates an existing descriptor change suggestion.
39     * Only pending suggestions can be updated.
40     *
41     * @param key The key of the suggestion to update
42     * @param request The updated suggestion request
43     * @param fileStream Optional new file stream to replace the existing one
44     * @param filename The original filename if fileStream is provided (used for display and metadata only)
45     * @return The updated DescriptorChangeSuggestion
46     * @throws IOException If there's an error processing the file
47     * @throws IllegalArgumentException If the suggestion is not in PENDING state
48     * @throws IllegalStateException If the suggestion cannot be updated
49     */
50    DescriptorChangeSuggestion updateSuggestion(long key, DescriptorChangeSuggestionRequest request, InputStream fileStream, String filename) throws IOException;
51  
52    /**
53     * Retrieves a descriptor change suggestion by its key.
54     *
55     * @param key The key of the suggestion to retrieve
56     * @return The DescriptorChangeSuggestion or null if not found
57     */
58    DescriptorChangeSuggestion getSuggestion(long key);
59  
60    /**
61     * Retrieves the file associated with a descriptor change suggestion.
62     *
63     * @param key The key of the suggestion
64     * @return InputStream containing the suggestion file data
65     * @throws IllegalArgumentException If the suggestion is not found
66     * @throws IOException If there's an error reading the file
67     */
68    InputStream getSuggestionFile(long key) throws IOException;
69  
70    /**
71     * Applies a pending descriptor change suggestion.
72     * This will create or update the descriptor group based on the suggestion.
73     *
74     * @param key The key of the suggestion to apply
75     * @throws IOException If there's an error processing the file
76     * @throws IllegalArgumentException If the suggestion is not in PENDING state
77     * @throws IllegalStateException If the suggestion cannot be applied
78     */
79    void applySuggestion(long key) throws IOException;
80  
81    /**
82     * Discards a pending descriptor change suggestion.
83     *
84     * @param key The key of the suggestion to discard
85     * @throws IllegalArgumentException If the suggestion is not in PENDING state
86     * @throws IllegalStateException If the suggestion cannot be discarded
87     */
88    void discardSuggestion(long key);
89  
90  
91    /**
92     * Lists descriptor change suggestions with optional filters.
93     *
94     * @param pageable Pagination details
95     * @param status Filter by status (PENDING, APPROVED, DISCARDED)
96     * @param type Filter by type (CREATE, UPDATE, DELETE)
97     * @param proposerEmail Filter by proposer's email
98     * @param collectionKey Filter by collection key
99     * @param country Filter by collection's country
100    * @return PagingResponse containing the filtered suggestions
101    */
102   PagingResponse<DescriptorChangeSuggestion> list(Pageable pageable, Status status, Type type, String proposerEmail, UUID collectionKey, Country country);
103 
104   /**
105    * Counts the total number of descriptor change suggestions based on filters.
106    *
107    * @param status Filter by status
108    * @param type Filter by type
109    * @param proposerEmail Filter by proposer's email
110    * @param collectionKey Filter by collection key
111    * @param country Filter by collection's country
112    * @return Total number of matching suggestions
113    */
114   long count(Status status, Type type, String proposerEmail, UUID collectionKey, Country country);
115 }
