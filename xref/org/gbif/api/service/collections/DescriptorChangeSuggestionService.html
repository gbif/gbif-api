1   package org.gbif.api.service.collections;
2   
3   import java.io.IOException;
4   import java.io.InputStream;
5   import java.util.UUID;
6   
7   import org.gbif.api.model.collections.descriptors.DescriptorChangeSuggestion;
8   import org.gbif.api.model.collections.descriptors.DescriptorChangeSuggestionRequest;
9   import org.gbif.api.model.collections.suggestions.Status;
10  import org.gbif.api.model.collections.suggestions.Type;
11  import org.gbif.api.model.common.paging.Pageable;
12  import org.gbif.api.model.common.paging.PagingResponse;
13  
14  /**
15   * Service interface for managing descriptor change suggestions in collections.
16   * This service handles the creation, updating, and management of descriptor change suggestions,
17   * including file handling and status management.
18   *
19   * Note: File storage is handled internally using unique identifiers. The provided filename
20   * is used only for display and metadata purposes.
21   */
22  public interface DescriptorChangeSuggestionService {
23  
24    /**
25     * Creates a new descriptor change suggestion.
26     *
27     * @param fileStream The input stream containing the descriptor data
28     * @param filename The original filename (used for display and metadata only)
29     * @param request The suggestion request containing metadata
30     * @return The created DescriptorChangeSuggestion
31     * @throws IOException If there's an error processing the file
32     * @throws IllegalArgumentException If the request is invalid
33     */
34    DescriptorChangeSuggestion createSuggestion(InputStream fileStream, String filename, DescriptorChangeSuggestionRequest request) throws IOException;
35  
36    /**
37     * Updates an existing descriptor change suggestion.
38     * Only pending suggestions can be updated.
39     *
40     * @param key The key of the suggestion to update
41     * @param request The updated suggestion request
42     * @param fileStream Optional new file stream to replace the existing one
43     * @param filename The original filename if fileStream is provided (used for display and metadata only)
44     * @return The updated DescriptorChangeSuggestion
45     * @throws IOException If there's an error processing the file
46     * @throws IllegalArgumentException If the suggestion is not in PENDING state
47     * @throws IllegalStateException If the suggestion cannot be updated
48     */
49    DescriptorChangeSuggestion updateSuggestion(long key, DescriptorChangeSuggestionRequest request, InputStream fileStream, String filename) throws IOException;
50  
51    /**
52     * Retrieves a descriptor change suggestion by its key.
53     *
54     * @param key The key of the suggestion to retrieve
55     * @return The DescriptorChangeSuggestion or null if not found
56     */
57    DescriptorChangeSuggestion getSuggestion(long key);
58  
59    /**
60     * Retrieves the file associated with a descriptor change suggestion.
61     *
62     * @param key The key of the suggestion
63     * @return InputStream containing the suggestion file data
64     * @throws IllegalArgumentException If the suggestion is not found
65     * @throws IOException If there's an error reading the file
66     */
67    InputStream getSuggestionFile(long key) throws IOException;
68  
69    /**
70     * Applies a pending descriptor change suggestion.
71     * This will create or update the descriptor group based on the suggestion.
72     *
73     * @param key The key of the suggestion to apply
74     * @throws IOException If there's an error processing the file
75     * @throws IllegalArgumentException If the suggestion is not in PENDING state
76     * @throws IllegalStateException If the suggestion cannot be applied
77     */
78    void applySuggestion(long key) throws IOException;
79  
80    /**
81     * Discards a pending descriptor change suggestion.
82     *
83     * @param key The key of the suggestion to discard
84     * @throws IllegalArgumentException If the suggestion is not in PENDING state
85     * @throws IllegalStateException If the suggestion cannot be discarded
86     */
87    void discardSuggestion(long key);
88  
89  
90    /**
91     * Lists descriptor change suggestions with optional filters.
92     *
93     * @param pageable Pagination details
94     * @param status Filter by status (PENDING, APPROVED, DISCARDED)
95     * @param type Filter by type (CREATE, UPDATE, DELETE)
96     * @param proposerEmail Filter by proposer's email
97     * @param collectionKey Filter by collection key
98     * @return PagingResponse containing the filtered suggestions
99     */
100   PagingResponse<DescriptorChangeSuggestion> list(Pageable pageable, Status status, Type type, String proposerEmail, UUID collectionKey);
101 
102   /**
103    * Counts the total number of descriptor change suggestions based on filters.
104    *
105    * @param status Filter by status
106    * @param type Filter by type
107    * @param proposerEmail Filter by proposer's email
108    * @param collectionKey Filter by collection key
109    * @return Total number of matching suggestions
110    */
111   long count(Status status, Type type, String proposerEmail, UUID collectionKey);
112 }
