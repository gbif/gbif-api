15  import org.gbif.api.model.occurrence.DownloadRequest;

17  import java.io.InputStream;

19  import javax.annotation.Nullable;
20  import javax.validation.constraints.NotNull;

22  /**
23   * Provides services to manage download requests and retrieve the download file. The services provided are:
24   * <ul>
25   * <li><strong>request</strong> inits a new download and assigns it a key</li> *
26   * <li><strong>cancel</strong> cancels an existing download</li>
27   * <li><strong>getResult</strong> retrieves the download file, if it is ready</li>
28   * </ul>
29   */
30  public interface DownloadRequestService {
31  
32    /**
33     * Cancels running download. If the download is not running throws an IllegalStateException.
34     */
35    void cancel(@NotNull String downloadKey);
36  
37    /**
38     * Creates a download request. If the request is successfully created its key is returned.
39     */
40    String create(@NotNull DownloadRequest downloadRequest);
41  
42    /**
43     * @param downloadKey of the corresponding download request
44     * @return the input stream of the zipped download result file or null if it's not existing or ready yet
45     */
46    @Nullable
47    InputStream getResult(String downloadKey);
48  
49  }
