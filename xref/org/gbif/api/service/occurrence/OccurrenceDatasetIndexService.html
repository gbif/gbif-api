13  package org.gbif.api.service.occurrence;
14  
15  import org.gbif.api.vocabulary.Country;
16  
17  import java.util.SortedMap;
18  import java.util.UUID;
19  
20  /**
21   * Supports the listing of datasets having occurrence records.
22   */
23  public interface OccurrenceDatasetIndexService {
24  
25    /**
26     * This provides a map of occurrence dataset keys to the number of records matching the given country.
27     * Datasets which do not have content are not included (e.g. no zero-values will exist).
28     * That is to say the count represents the number of records in the dataset that are interpreted as
29     * the provided country, and not the total count of records in the dataset.
30     * 
31     * @param country to filter occurrence datasets by
32     * @return a map of dataset keys to counts sorted by count descending
33     */
34    SortedMap<UUID, Integer> occurrenceDatasetsForCountry(Country country);
35  
36  
37    /**
38     * This provides a map of occurrence dataset keys to the number of records matching the given nub key.
39     * Datasets which do not have content are not included (e.g. no zero-values will exist).
40     * That is to say the count represents the number of records in the dataset that are interpreted as
41     * the provided taxon (or any subordinate taxon), and not the total count of records in the dataset. E.g. a dataset of
42     * 1 million records covering everything, might report it has 300k records of birds (Aves); those records might
43     * themselves have more specific identifications (such as the species).
44     * 
45     * @param nubKey to filter occurrence datasets by
46     * @return a map of dataset keys to counts sorted by count descending
47     */
48    SortedMap<UUID, Integer> occurrenceDatasetsForNubKey(int nubKey);
49  }
