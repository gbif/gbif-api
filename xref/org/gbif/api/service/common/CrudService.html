23  /**

29   */
30  public interface CrudService<T, W, K> {
31  
32    /**
33     * Persists a new entity.
34     *
35     * @param entity to create.
36     *
37     * @return The persisted entity.
38     */
39    K create(W entity);
40  
41    /**
42     * Deletes the entity.
43     *
44     * @param key The entity key to delete.
45     */
46    void delete(K key);
47  
48    /**
49     * Get an entity by its unique key.
50     *
51     * @return the entity or null if not existing.
52     */
53    @Nullable
54    T get(K key);
55  
56  
57    /**
58     * Lists all entity.
59     *
60     * @param page to enable paging.
61     *
62     * @return The pagable list of entities.
63     */
64    PagingResponse<T> list(@Nullable Pageable page);

67     * Updates the entity.
68     *
69     * @param entity to update
70     */
71    void update(W entity);
