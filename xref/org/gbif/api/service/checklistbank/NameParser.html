1   package org.gbif.api.service.checklistbank;
2   
3   import org.gbif.api.exception.UnparsableException;
4   import org.gbif.api.model.checklistbank.ParsedName;
5   import org.gbif.api.vocabulary.Rank;
6   
7   import javax.annotation.Nullable;
8   
9   /**
10   * Interface for parsing scientific names.
11   */
12  public interface NameParser {
13  
14    /**
15     * Fully parse the supplied name also trying to extract authorships, a conceptual sec reference, remarks or notes
16     * on the nomenclatural status. In some cases the authorship parsing proves impossible and this nameparser will
17     * return null.
18     *
19     * For strings which are no scientific names and scientific names that cannot be expressed by the ParsedName class
20     * the parser will throw an UnparsableException with a given NameType and the original, unparsed name. This is the
21     * case for all virus names and proper hybrid formulas, so make sure you catch and process this exception.
22     *
23     * @param scientificName the full scientific name to parse
24     * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
25     *
26     * @return the parsed name
27     *
28     * @throws UnparsableException
29     */
30    ParsedName parse(String scientificName, @Nullable Rank rank) throws UnparsableException;
31  
32    /**
33     * Delegate method to parse a scientific name of unknown rank.
34     * @return the parsed name
35     */
36    ParsedName parse(String scientificName) throws UnparsableException;
37  
38    /**
39     * Fully parses a name using #parse(String, Rank) but converts names that throw a UnparsableException
40     * into ParsedName objects with the scientific name, rank and name type given.
41     * @return the parsed name
42     */
43    ParsedName parseQuietly(String scientificName, @Nullable Rank rank);
44  
45    /**
46     * Delegate method to parse a scientific name of unknown rank quietly.
47     * @return the parsed name
48     */
49    ParsedName parseQuietly(String scientificName);
50  
51    /**
52     * Parses the scientific name without authorship and returns the ParsedName.canonicalName() string
53     * @param scientificName the full scientific name to parse
54     * @param rank the rank of the name if it is known externally. Helps identifying infrageneric names vs bracket authors
55     *
56     * @return the canonical name or null for unparsable names
57     */
58    String parseToCanonical(String scientificName, @Nullable Rank rank);
59  
60    /**
61     * Delegate method to parse a scientific name of unknown rank and return its canonical form.
62     */
63    String parseToCanonical(String scientificName);
64  
65  }
