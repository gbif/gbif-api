16  package org.gbif.api.service.checklistbank;
17  
18  import org.gbif.api.model.checklistbank.NameUsageMatch;
19  import org.gbif.api.model.common.LinneanClassification;
20  import org.gbif.api.vocabulary.Rank;
21  
22  import javax.annotation.Nullable;
23  
24  /**
25   * A lookup service that fuzzy matches classified scientific names against a body of names.
26   * In GBIF this is in particular used for a Nub Lookup service that binds occurrence names and other external records
27   * with a scientific name to the nub.
28   * <p/>
29   * Several match signatures for essentially the same matching are provided.
30   */
31  public interface NameUsageMatchingService {
32  
33    /**
34     * Tries to match a scientific name with an optional classification given as individual parameters.
35     *
36     * @param scientificName  the name to match against
37     * @param rank            of the name to match
38     * @param classification  optional classification of the name to match
39     * @param strict          if true only tries to match the scientific name, if false (the default) the matching
40     *                        service tries also to match the lowest possible taxon from the given classification
41     * @param verbose         if true adds verbose matching information, i.e. list of alternative matches
42     *
43     * @return a match which is never null, but might have a usageKey=null if nothing could be matched
44     */
45    NameUsageMatch match(String scientificName, @Nullable Rank rank, @Nullable LinneanClassification classification,
46      boolean strict, boolean verbose);
47  
48  }
