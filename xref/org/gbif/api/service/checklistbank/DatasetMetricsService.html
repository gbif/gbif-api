16  package org.gbif.api.service.checklistbank;
17  
18  import org.gbif.api.model.checklistbank.DatasetMetrics;
19  
20  import java.util.List;
21  import java.util.UUID;
22  import javax.annotation.Nullable;
23  
24  public interface DatasetMetricsService {
25  
26    /**
27     * @return metrics about this dataset or null if none could be found
28     */
29    @Nullable
30    DatasetMetrics get(UUID datasetKey);
31  
32    /**
33     * @return the list of all metrics about this dataset ordered by ascending created timestamp.
34     *         An empty list if no matching dataset is found.
35     */
36    @Nullable
37    List<DatasetMetrics> list(UUID datasetKey);
38  
39  }
