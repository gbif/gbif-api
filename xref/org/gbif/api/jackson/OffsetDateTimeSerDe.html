1   package org.gbif.api.jackson;

3   import com.fasterxml.jackson.core.JsonGenerator;
4   import com.fasterxml.jackson.core.JsonParser;
5   import com.fasterxml.jackson.databind.DeserializationContext;
6   import com.fasterxml.jackson.databind.JsonDeserializer;
7   import com.fasterxml.jackson.databind.JsonSerializer;
8   import com.fasterxml.jackson.databind.SerializerProvider;
9   import java.io.IOException;
10  import java.time.OffsetDateTime;
11  import java.time.format.DateTimeFormatter;
12  
13  /**
14   * Jackson {@link JsonSerializer} and {@link JsonDeserializer} classes for {@link OffsetDateTime}.
15   */
16  public class OffsetDateTimeSerDe {
17  
18    private static final DateTimeFormatter FORMATTER =
19      DateTimeFormatter.ofPattern("yyyy-MM-dd HH:mm:ssXXX");
20  
21    /**
22     * Jackson {@link JsonSerializer} for {@link OffsetDateTime}.
23     */
24    public static class OffsetDateTimeSerializer extends JsonSerializer<OffsetDateTime> {
25      @Override
26      public void serialize(OffsetDateTime value, JsonGenerator jgen, SerializerProvider provider)
27        throws IOException {
28        jgen.writeString(value.format(FORMATTER));
29      }
30    }
31  
32    /**
33     * Jackson {@link JsonDeserializer} for {@link OffsetDateTime}.
34     */
35    public static class OffsetDateTimeDeserializer extends JsonDeserializer<OffsetDateTime> {
36      @Override
37      public OffsetDateTime deserialize(JsonParser jp, DeserializationContext ctxt)
38        throws IOException {
39        return OffsetDateTime.parse(jp.getText(), FORMATTER);
40      }
41    }
42  }
