16  package org.gbif.api.jackson;
17  
18  import java.io.IOException;
19  import java.util.AbstractMap;
20  import java.util.Map;
21  
22  import org.codehaus.jackson.JsonGenerator;
23  import org.codehaus.jackson.JsonParser;
24  import org.codehaus.jackson.JsonProcessingException;
25  import org.codehaus.jackson.map.DeserializationContext;
26  import org.codehaus.jackson.map.JsonDeserializer;
27  import org.codehaus.jackson.map.JsonSerializer;
28  import org.codehaus.jackson.map.SerializerProvider;
29  
30  /**
31   * Jackson Serializer and Deserializer for {@link java.util.Map.Entry}.
32   * This is mostly for pre 2.7 version of Jackson see
33   * https://github.com/fasterxml/jackson-databind/issues/565
34   *
35   * The goal is to omit the key/value field name since they are implicit
36   * for a Map.Entry.
37   *
38   * {"key":"mykey","value":18} becomes {"mykey":18}
39   *
40   * The key will use toString() and the value can only be a String or a Number (int or float) for now.
41   *
42   * <pre>
43   * {@code
44   * // Usage for lists:
45   * @JsonSerialize(contentUsing = MapEntrySerde.MapEntryJsonSerializer.class)
46   * public List<Map.Entry<String, Object>> getKeyValueList() { ... }
47   * }
48   * </pre>
49   */
50  public class MapEntrySerde {
51  
52    public static class MapEntryJsonSerializer extends JsonSerializer<Map.Entry<Object, Object>> {
53  
54      @Override
55      public void serialize(Map.Entry<Object, Object> value, JsonGenerator jgen,
56                            SerializerProvider serializerProvider) throws IOException, JsonProcessingException {
57        if(value == null){
58          jgen.writeNull();
59          return;
60        }
61        jgen.writeStartObject();
62        jgen.writeFieldName(value.getKey().toString());
63        jgen.writeObject(value.getValue());
64        jgen.writeEndObject();
65      }
66    }
67  
68    public static class MapEntryJsonDeserializer extends JsonDeserializer<Map.Entry<Object, Object>> {
69      @Override
70      public Map.Entry<Object, Object> deserialize(JsonParser jp, DeserializationContext ctxt) throws IOException {
71        String tmp = jp.getText(); // {
72        jp.nextToken();
73        String key = jp.getText();
74        jp.nextToken();
75        Object value;
76  
77        switch (jp.getCurrentToken()) {
78          case VALUE_STRING: value = jp.getText();
79            break;
80          case VALUE_NUMBER_INT: value = jp.getIntValue();
81            break;
82          case VALUE_NUMBER_FLOAT: value = jp.getFloatValue();
83            break;
84          default :
85            throw ctxt.mappingException("Expected String or Number");
86        }
87        jp.nextToken();
88        tmp = jp.getText(); // }
89  
90        return new AbstractMap.SimpleImmutableEntry<>(key, value);
91      }
92    }
93  }
