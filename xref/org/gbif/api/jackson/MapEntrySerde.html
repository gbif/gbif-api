15  /**
16   * Jackson Serializer and Deserializer for {@link java.util.Map.Entry}.
17   * This is mostly for pre 2.7 version of Jackson see
18   * https://github.com/fasterxml/jackson-databind/issues/565
19   *
20   * The goal is to omit the key/value field name since they are implicit
21   * for a Map.Entry.
22   *
23   * {"key":"mykey","value":18} becomes {"mykey":18}
24   *
25   * The key will use toString() and the value can only be a String or a Number (int or float) for now.
26   *
27   * <pre>
28   * {@code
29   * // Usage for lists:
30   * @JsonSerialize(contentUsing = MapEntrySerde.MapEntryJsonSerializer.class)
31   * public List<Map.Entry<String, Object>> getKeyValueList() { ... }
32   * }
33   * </pre>
34   */
35  public class MapEntrySerde {
36  
37    public static class MapEntryJsonSerializer extends JsonSerializer<Map.Entry<Object, Object>> {
38  
39      @Override
40      public void serialize(Map.Entry<Object, Object> value, JsonGenerator jgen,
41                            SerializerProvider serializerProvider) throws IOException, JsonProcessingException {
42        if(value == null){
43          jgen.writeNull();
44          return;
45        }
46        jgen.writeStartObject();
47        jgen.writeFieldName(value.getKey().toString());
48        jgen.writeObject(value.getValue());
49        jgen.writeEndObject();
50      }
51    }
52  
53    public static class MapEntryJsonDeserializer extends JsonDeserializer<Map.Entry<Object, Object>> {
54      @Override
55      public Map.Entry<Object, Object> deserialize(JsonParser jp, DeserializationContext ctxt) throws IOException {
56        String tmp = jp.getText(); // {
57        jp.nextToken();
58        String key = jp.getText();
59        jp.nextToken();
60        Object value;
61  
62        switch (jp.getCurrentToken()) {
63          case VALUE_STRING: value = jp.getText();
64            break;
65          case VALUE_NUMBER_INT: value = jp.getIntValue();
66            break;
67          case VALUE_NUMBER_FLOAT: value = jp.getFloatValue();
68            break;
69          default :
70            throw ctxt.mappingException("Expected String or Number");
71        }
72        jp.nextToken();
73        tmp = jp.getText(); // }
74  
75        return new AbstractMap.SimpleImmutableEntry<>(key, value);
76      }
77    }
78  }
