28      if (jp.getCurrentToken() == JsonToken.START_ARRAY) {
29        JsonDeserializer<Object> deserializer =
30          ctxt.getDeserializerProvider().findTypedValueDeserializer(ctxt.getConfig(), ctxt.constructType(List.class),
31            null);
32        List<Map<String, String>> verbatimTerms = (List<Map<String, String>>) deserializer.deserialize(jp, ctxt);
33        List<Map<Term, String>> interpretedTerms = Lists.newArrayList();
34        for (Map<String, String> verbExtension : verbatimTerms) {
35          Map<Term, String> extension = new HashMap<Term, String>();
36          for (Entry<String, String> entry : verbExtension.entrySet()) {
37            Term term = termFactory.findTerm(entry.getKey());
38            if (term == null && ctxt.getConfig().isEnabled(Feature.FAIL_ON_UNKNOWN_PROPERTIES)) {
39              throw ctxt.mappingException("Term not found " + entry.getKey());
40            }
41            extension.put(term, entry.getValue());
42          }
43          interpretedTerms.add(extension);
44        }
45        return interpretedTerms;
46      }
47      throw ctxt.mappingException("Expected JSON String");
48    }
