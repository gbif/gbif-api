24  /**
25   * Download request deserializer.
26   *
27   * For most of the time, the serialization has been to "notificationAddresses" and "sendNotification".  For a few months
28   * in 2018-2019, it was "notification_address" and "send_notification".
29   *
30   * The API documentation has previously specified "notification_address" and "sendNotification".
31   *
32   * We therefore accept all combinations.
33   *
34   * https://github.com/gbif/portal-feedback/issues/2046
35   */
36  public class DownloadRequestSerde extends JsonDeserializer<DownloadRequest> {
37  
38    private static final String PREDICATE = "predicate";
39    private static final List<String> SEND_NOTIFICATION = ImmutableList.of("sendNotification", "send_notification");
40    private static final List<String> NOTIFICATION_ADDRESSES =
41      ImmutableList.of("notificationAddresses", "notificationAddress", "notification_addresses", "notification_address");
42    private static final String CREATOR = "creator";
43    private static final String FORMAT = "format";
44    private static final Logger LOG = LoggerFactory.getLogger(DownloadRequestSerde.class);
45    private static final ObjectMapper MAPPER = new ObjectMapper();
46  
47    @Override
48    public DownloadRequest deserialize(JsonParser jp, DeserializationContext ctxt) throws IOException {
49      JsonNode node = jp.getCodec().readTree(jp);
50      LOG.debug("DownloadRequest for deserialization: {}", node);
51      //at least one element must be defined
52      if (node.size() == 0) {
53        return null;
54      }
55      DownloadFormat format = Optional.ofNullable(node.get(FORMAT))
56        .map(n -> VocabularyUtils.lookupEnum(n.asText(), DownloadFormat.class)).orElse(DownloadFormat.DWCA);
57      String creator = Optional.ofNullable(node.get(CREATOR)).map(JsonNode::asText).orElse(null);
58  
59      List<String> notificationAddresses = new ArrayList<>();
60      for (final String jsonKey : NOTIFICATION_ADDRESSES) {
61        notificationAddresses.addAll(Optional.ofNullable(node.get(jsonKey)).map(jsonNode -> {
62          try {
63            return Arrays.asList(MAPPER.readValue(jsonNode, String[].class));
64          } catch (Exception e) {
65            throw Throwables.propagate(e);
66          }
67        }).orElse(new ArrayList<>()));
68      }
69  
70      boolean sendNotification = false;
71      for (final String jsonKey : SEND_NOTIFICATION) {
72        sendNotification |= Optional.ofNullable(node.get(jsonKey)).map(JsonNode::asBoolean).orElse(Boolean.FALSE);
73      }
74      JsonNode predicate = Optional.ofNullable(node.get(PREDICATE)).orElse(null);
75      Predicate predicateObj = predicate == null ? null : MAPPER.readValue(predicate, Predicate.class);
76      return new PredicateDownloadRequest(predicateObj, creator, notificationAddresses, sendNotification, format);
77    }
78  }
