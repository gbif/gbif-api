26  import java.util.Arrays;
27  import java.util.Collections;

34  import com.fasterxml.jackson.core.JsonParser;

36  import com.fasterxml.jackson.databind.JsonDeserializer;
37  import com.fasterxml.jackson.databind.JsonNode;
38  import com.fasterxml.jackson.databind.ObjectMapper;

40  /**
41   * Download request deserializer.
42   * <p>
43   * For most of the time, the serialization has been to "notificationAddresses" and
44   * "sendNotification".  For a few months in 2018-2019, it was "notification_address" and
45   * "send_notification".
46   * <p>
47   * The API documentation has previously specified "notification_address" and "sendNotification".
48   * <p>
49   * We therefore accept all combinations.
50   * <p>
51   * https://github.com/gbif/portal-feedback/issues/2046
52   */
53  public class DownloadRequestSerde extends JsonDeserializer<DownloadRequest> {
54  
55    private static final String PREDICATE = "predicate";
56    private static final List<String> SEND_NOTIFICATION = Collections
57      .unmodifiableList(Arrays.asList("sendNotification", "send_notification"));
58    private static final List<String> NOTIFICATION_ADDRESSES =
59      Collections.unmodifiableList(
60        Arrays.asList("notificationAddresses", "notificationAddress", "notification_addresses",
61          "notification_address"));
62    private static final String CREATOR = "creator";
63    private static final String FORMAT = "format";
64    private static final Logger LOG = LoggerFactory.getLogger(DownloadRequestSerde.class);
65    private static final ObjectMapper MAPPER = new ObjectMapper();
66  
67    @Override
68    public DownloadRequest deserialize(JsonParser jp, DeserializationContext ctxt) throws IOException {
69      JsonNode node = jp.getCodec().readTree(jp);
70      LOG.debug("DownloadRequest for deserialization: {}", node);
71      //at least one element must be defined
72      if (node.size() == 0) {
73        return null;
74      }
75      DownloadFormat format = Optional.ofNullable(node.get(FORMAT))
76        .map(n -> VocabularyUtils.lookupEnum(n.asText(), DownloadFormat.class)).orElse(DownloadFormat.DWCA);
77      String creator = Optional.ofNullable(node.get(CREATOR)).map(JsonNode::asText).orElse(null);
78  
79      List<String> notificationAddresses = new ArrayList<>();
80      for (final String jsonKey : NOTIFICATION_ADDRESSES) {
81        notificationAddresses.addAll(Optional.ofNullable(node.get(jsonKey)).map(jsonNode -> {
82          try {
83            return Arrays.asList(MAPPER.treeToValue(jsonNode, String[].class));
84          } catch (Exception e) {
85            throw new RuntimeException(e);
86          }
87        }).orElse(new ArrayList<>()));
88      }
89  
90      boolean sendNotification = false;
91      for (final String jsonKey : SEND_NOTIFICATION) {
92        sendNotification |= Optional.ofNullable(node.get(jsonKey)).map(JsonNode::asBoolean).orElse(Boolean.FALSE);
93      }
94  
95      JsonNode predicate = Optional.ofNullable(node.get(PREDICATE)).orElse(null);
96      Predicate predicateObj = predicate == null ? null : MAPPER.treeToValue(predicate, Predicate.class);
97      return new PredicateDownloadRequest(predicateObj, creator, notificationAddresses, sendNotification, format);
98    }
99  }
