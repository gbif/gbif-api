295     /**
296      * Tries its best to infer a rank from a given rank marker such as subsp.
297      *
298      * @return the inferred rank or null
299      */
300     public static Rank inferRank(@Nullable String rankMarker) {
301         // first try rank marker
302         if (rankMarker != null) {
303             Rank markerRank = RANK_MARKER_MAP.get(NORMALIZE_TERM.matcher(rankMarker.toLowerCase()).replaceAll(""));
304             if (markerRank != null) {
305                 return markerRank;
306             }
307         }
308         return null;
309     }

311     /**
312      * Tries its best to infer a rank from an atomised name.
313      * As a final resort for higher monomials the suffices are inspected, but no attempt is made to disambiguate
314      * the 2 known homonym suffices -idae and -inae, but instead the far more widespread zoological versions are
315      * interpreted.
316      * TODO: pass optional nomenclatural code paraemeter to disambiguate homonym suffices.
317      *
318      * @param genusOrAbove         an optional uninomial at genus level or above
319      * @param infraGeneric         an optional subgeneric name part, e.g. the subgenus
320      * @param specificEpithet      an optional specific epithet
321      * @param rankMarker           an optional rank marker such as subsp.
322      * @param infraSpecificEpithet an optional infraspecific epithet
323      * @return the inferred rank or Unranked if it cant be found, never null.
324      */
325 
326     public static Rank inferRank(
327             @Nullable String genusOrAbove,
328             @Nullable String infraGeneric,
329             @Nullable String specificEpithet,
330             @Nullable String rankMarker,
331             @Nullable String infraSpecificEpithet
332     ) {
333         // first try rank marker
334         Rank markerRank = inferRank(rankMarker);
335         if (markerRank != null) {
336             return markerRank;
337         }
338 
339         // default if we cant find anything else
340         Rank rank = UNRANKED;
341         // detect rank based on parsed name
342         if (infraSpecificEpithet != null) {
343             // some infraspecific name
344             rank = INFRASPECIFIC_NAME;
345         } else if (specificEpithet != null) {
346             // a species
347             rank = SPECIES;
348         } else if (infraGeneric != null) {
349             // some infrageneric name
350             rank = INFRAGENERIC_NAME;
351         } else if (genusOrAbove != null) {
352             // a suprageneric name, check suffices
353             for (String suffix : SUFFICES_RANK_MAP.keySet()) {
354                 if (genusOrAbove.endsWith(suffix)) {
355                     rank = SUFFICES_RANK_MAP.get(suffix);
356                     break;
357                 }
358             }
359         }
360         return rank;
361     }

375     /**
376      * @return true for infraspecific ranks excluding cultivars and strains.
377      */
378     public boolean isInfraspecific() {
379         return isSpeciesOrBelow() && this != SPECIES && this != CULTIVAR && this != CULTIVAR_GROUP && this != STRAIN;

385     public boolean isInfrageneric() {
386         return ordinal() > GENUS.ordinal() && this != UNRANKED && this != INFORMAL;

389     /**
390      * True for all mayor Linnéan ranks, ie kingdom,phylum,class,order,family,genus and species.
391      */
392     public boolean isLinnean() {
393         for (Rank r : LINNEAN_RANKS) {
394             if (r == this) {
395                 return true;
396             }
397         }
398         return false;
399     }
