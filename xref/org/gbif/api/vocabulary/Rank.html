404   /**
405    * Tries its best to infer a rank from a given rank marker such as subsp.
406    *
407    * @return the inferred rank or null
408    */
409   public static Rank inferRank(@Nullable String rankMarker) {
410     // first try rank marker
411     if (rankMarker != null) {
412       Rank markerRank = RANK_MARKER_MAP.get(NORMALIZE_TERM.matcher(rankMarker.toLowerCase()).replaceAll(""));
413       if (markerRank != null) {
414         return markerRank;
415       }
416     }
417     return null;
418   }

420   /**
421    * Tries its best to infer a rank from an atomised name.
422    * As a final resort for higher monomials the suffices are inspected, but no attempt is made to disambiguate
423    * the 2 known homonym suffices -idae and -inae, but instead the far more widespread zoological versions are
424    * interpreted.
425    * TODO: pass optional nomenclatural code paraemeter to disambiguate homonym suffices.
426    *
427    * @param genusOrAbove         an optional uninomial at genus level or above
428    * @param infraGeneric         an optional subgeneric name part, e.g. the subgenus
429    * @param specificEpithet      an optional specific epithet
430    * @param rankMarker           an optional rank marker such as subsp.
431    * @param infraSpecificEpithet an optional infraspecific epithet
432    *
433    * @return the inferred rank or Unranked if it cant be found, never null.
434    */
435 
436   public static Rank inferRank(
437       @Nullable String genusOrAbove,
438       @Nullable String infraGeneric,
439       @Nullable String specificEpithet,
440       @Nullable String rankMarker,
441       @Nullable String infraSpecificEpithet
442   ) {
443     // first try rank marker
444     Rank markerRank = inferRank(rankMarker);
445     if (markerRank != null) {
446       return markerRank;
447     }
448 
449     // default if we cant find anything else
450     Rank rank = UNRANKED;
451     // detect rank based on parsed name
452     if (infraSpecificEpithet != null) {
453       // some infraspecific name
454       rank = INFRASPECIFIC_NAME;
455     } else if (specificEpithet != null) {
456       // a species
457       rank = SPECIES;
458     } else if (infraGeneric != null) {
459       // some infrageneric name
460       rank = INFRAGENERIC_NAME;
461     } else if (genusOrAbove != null) {
462       // a suprageneric name, check suffices
463       for (String suffix : SUFFICES_RANK_MAP.keySet()) {
464         if (genusOrAbove.endsWith(suffix)) {
465           rank = SUFFICES_RANK_MAP.get(suffix);
466           break;
467         }
468       }
469     }
470     return rank;
471   }

485   /**
486    * @return true for infraspecific ranks excluding cultivars and strains.
487    */
488   public boolean isInfraspecific() {
489     return isSpeciesOrBelow() && this != SPECIES && this != CULTIVAR && this != CULTIVAR_GROUP && this != STRAIN;
490   }
491 
492   /**
493    * @return true for rank is below genus.
494    */
495   public boolean isInfrageneric() {
496     return ordinal() > GENUS.ordinal() && this != UNRANKED && this != INFORMAL;
497   }
498 
499   /**
500    * True for all mayor LinnÃ©an ranks, ie kingdom,phylum,class,order,family,genus and species.
501    */
502   public boolean isLinnean() {
503     for (Rank r : LINNEAN_RANKS) {
504       if (r == this) {
505         return true;
506       }
507     }
508     return false;
509   }
510 
511   public boolean isSpeciesOrBelow() {
512     return SPECIES_OR_BELOW.contains(this);
513   }
514 
515   /**
516    * True for all infrasubspecific ranks used in microbiology.
517    */
518   public boolean isMicrobial() {
519     return INFRASUBSPECIFIC_MICROBIAL_RANKS.contains(this);
520   }
521 
522   /**
523    * @return true if the rank is above genus.
524    */
525   public boolean isSuprageneric() {
526     return ordinal() < GENUS.ordinal();
527   }
528 
529   /**
530    * @return true if the rank is above rank species.
531    */
532   public boolean isSupraspecific() {
533     return ordinal() < SPECIES.ordinal();
534   }
535 
536   /**
537    * True for names of informal ranks that represent a range of ranks really and therefore cannot safely be compared to
538    * other ranks in all cases.
539    * Example ranks are infraspecies or subgeneric
540    *
541    * @return true if uncomparable
542    */
543   public boolean isUncomparable() {
544     return UNCOMPARABLE_RANKS.contains(this);
545   }
546 
547   /**
548    * @return true if this rank is higher than the given other
549    */
550   public boolean higherThan(Rank other) {
551     return ordinal() < other.ordinal();
552   }
