3   /**
4    * These are the possible states of an {@link org.gbif.api.model.occurrence.Occurrence} as it moves through the processing and persistence chain. It
5    * is used both in the processing logic as well as the messages that are exchanged as part of that processing.
6    */
7   public enum OccurrencePersistenceStatus {
8     /**
9      * This occurrence has never been seen before.
10     */
11    NEW,
12  
13    /**
14     * This occurrence previously existed and has now been updated with new information.
15     */
16    UPDATED,
17  
18    /**
19     * This occurrence previously existed but there is no new information to propagate. This is typical of a crawl
20     * in which the exact record that was harvested in the previous crawl is seen again.
21     */
22    UNCHANGED,
23  
24    /**
25     * This occurrence has been deleted.
26     */
27    DELETED
28  }
