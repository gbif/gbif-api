16  package org.gbif.api.vocabulary;
17  
18  /**
19   * A simple enumeration of all kingdoms found in the GBIF Backbone Taxonomy.
20   *
21   * The Catalogue of Life has many more kingdoms these days and data keeps changing,
22   * thus this enumeration is deprecated.
23   */
24  @Deprecated
25  public enum Kingdom {
26  
27    INCERTAE_SEDIS,
28    ANIMALIA,
29    ARCHAEA,
30    BACTERIA,
31    CHROMISTA,
32    FUNGI,
33    PLANTAE,
34    PROTOZOA,
35    VIRUSES;
36  
37    /**
38     * Looks up a kingdom by its nub usage key.
39     *
40     * @param usageKey the nub usage key to lookup, must be < 9 to get a result.
41     *
42     * @return the matching kingdom or null
43     */
44    public static Kingdom byNubUsageKey(int usageKey) {
45      try {
46        return Kingdom.values()[usageKey];
47      } catch (IndexOutOfBoundsException e) {
48        throw new IllegalArgumentException("There is no kingdom with usage key " + usageKey);
49      }
50    }
51  
52    /**
53     * @return a unique single capital char representing the kingdom.
54     */
55    public String scientificName() {
56      String lower = name().replaceAll("_", " ").toLowerCase();
57      return this==INCERTAE_SEDIS ? lower : Character.toUpperCase(lower.charAt(0)) + lower.substring(1);
58    }
59  
60    public String nubUsageKey() {
61      return String.valueOf(ordinal());
62    }
63  
64    /**
65     * @deprecated please use nubUsageKey() instead
66     */
67    @Deprecated
68    public String nubUsageID() {
69      return nubUsageKey();
70    }
71  
72    /**
73     * @deprecated please use byNubUsageId(int) instead
74     */
75    @Deprecated
76    public static Kingdom byNubUsageId(int usageKey) {
77      return byNubUsageKey(usageKey);
78    }
79  }
