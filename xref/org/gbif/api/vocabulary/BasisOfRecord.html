18  /**
19   * A simple enumeration of all DarwinCore values of BasisOfRecord legal for occurrences.

21   * @see <a href="http://rs.tdwg.org/dwc/terms/type-vocabulary/index.htm">Darwin Core Type Vocabulary</a>

24  public enum BasisOfRecord {
25  
26    /**
27     * An occurrence record describing a preserved specimen.
28     */
29    PRESERVED_SPECIMEN,
30  
31    /**
32     * An occurrence record describing a fossilized specimen.
33     */
34    FOSSIL_SPECIMEN,
35  
36    /**

38     * or cultivated plants in a garden.
39     */
40    LIVING_SPECIMEN,
41  
42    /**
43     * An occurrence record describing an observation.
44     */
45    OBSERVATION,
46  
47    /**
48     * An occurrence record describing an observation made by one or more people.
49     */
50    HUMAN_OBSERVATION,
51  
52    /**
53     * An occurrence record describing an observation made by a machine.
54     */
55    MACHINE_OBSERVATION,
56  
57    /**
58     * An occurrence record based on samples taken from other specimens or the environment.
59     */
60    MATERIAL_SAMPLE,
61  
62    /**
63     * An occurrence record based on literature alone.
64     */
65    LITERATURE,
66  
67    /**

70    UNKNOWN

72  }
