16  package org.gbif.api.vocabulary;
17  
18  /**
19   * Enumeration representing the different nomenclatoral codes found in biology for scientific names.
20   * <p/>
21   * Nomenclature codes or codes of nomenclature are the various rulebooks that govern biological taxonomic
22   * nomenclature, each in their own broad field of organisms.
23   * To an end-user who only deals with names of species, with some awareness that species are assignable to
24   * families, it may not be noticeable that there is more than one code, but beyond this basic level these are rather
25   * different in the way they work.
26   *
27   * @see <a href="http://en.wikipedia.org/wiki/Nomenclature_codes">Nomenclature codes (Wikipedia)</a>
28   */
29  public enum NomenclaturalCode {
30  
31    BACTERIAL("ICNB", "International Code of Nomenclature of Bacteria", "http://www.ncbi.nlm.nih.gov/books/NBK8808/"),
32    BOTANICAL("ICBN", "International Code of Botanical Nomenclature", "http://ibot.sav.sk/icbn/main.htm"),
33    BIOCODE("BC", "BioCode", "http://www.bgbm.org/iapt/biocode/"),
34    CULTIVARS("ICNCP", "International Code of Nomenclature for Cultivated Plants", ""),
35    PHYLOCODE("PC", "Phylocode", "http://www.ohio.edu/phylocode/index.html"),
36    PHYTOSOCIOLOGY("ICPN", "International Code of Phytosociological Nomenclature", "http://www.iavs.org/pdf/Code.pdf"),
37    VIRUS("ICVCN", "International Code of Virus Classifications and Nomenclature", "http://talk.ICTVonline.org/"),
38    ZOOLOGICAL("ICZN",
39               "International Code of Zoological Nomenclature",
40               "http://www.nhm.ac.uk/hosted-sites/iczn/code/index.jsp");
41  
42    private final String title;
43    private final String acronym;
44    private final String link;
45  
46    NomenclaturalCode(String acronym, String title, String link) {
47      this.acronym = acronym;
48      this.link = link;
49      this.title = title;
50    }
51  
52    public String getAcronym() {
53      return acronym;
54    }
55  
56    public String getLink() {
57      return link;
58    }
59  
60    public String getTitle() {
61      return title;
62    }
63  
64  }
