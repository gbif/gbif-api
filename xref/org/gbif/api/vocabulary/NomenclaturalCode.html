16  package org.gbif.api.vocabulary;
17  
18  /**
19   * Enumeration representing the different nomenclatoral codes found in biology for scientific names.
20   * <p/>
21   * Nomenclature codes or codes of nomenclature are the various rulebooks that govern biological taxonomic
22   * nomenclature, each in their own broad field of organisms.
23   * To an end-user who only deals with names of species, with some awareness that species are assignable to
24   * families, it may not be noticeable that there is more than one code, but beyond this basic level these are rather
25   * different in the way they work.
26   *
27   * @see <a href="http://en.wikipedia.org/wiki/Nomenclature_codes">Nomenclature codes (Wikipedia)</a>
28   *
29   * Deprecated.
30   * Please use the name parser enum instead.
31   * https://github.com/gbif/name-parser/blob/master/name-parser-api/src/main/java/org/gbif/nameparser/api/NomCode.java
32  */
33  @Deprecated
34  public enum NomenclaturalCode {
35  
36    BACTERIAL("ICNB", "International Code of Nomenclature of Bacteria", "http://www.ncbi.nlm.nih.gov/books/NBK8808/"),
37    BOTANICAL("ICBN", "International Code of Botanical Nomenclature", "http://ibot.sav.sk/icbn/main.htm"),
38    BIOCODE("BC", "BioCode", "http://www.bgbm.org/iapt/biocode/"),
39    CULTIVARS("ICNCP", "International Code of Nomenclature for Cultivated Plants", ""),
40    PHYLOCODE("PC", "Phylocode", "http://www.ohio.edu/phylocode/index.html"),
41    PHYTOSOCIOLOGY("ICPN", "International Code of Phytosociological Nomenclature", "http://www.iavs.org/pdf/Code.pdf"),
42    VIRUS("ICVCN", "International Code of Virus Classifications and Nomenclature", "http://talk.ICTVonline.org/"),
43    ZOOLOGICAL("ICZN",
44               "International Code of Zoological Nomenclature",
45               "http://www.nhm.ac.uk/hosted-sites/iczn/code/index.jsp");
46  
47    private final String title;
48    private final String acronym;
49    private final String link;
50  
51    NomenclaturalCode(String acronym, String title, String link) {
52      this.acronym = acronym;
53      this.link = link;
54      this.title = title;
55    }
56  
57    public String getAcronym() {
58      return acronym;
59    }
60  
61    public String getLink() {
62      return link;
63    }
64  
65    public String getTitle() {
66      return title;
67    }
68  
69  }
