16  package org.gbif.api.vocabulary;
17  
18  import static org.gbif.api.vocabulary.TagNamespace.*;
19  
20  /**
21   * Known {@link org.gbif.api.model.registry.MachineTag#getName()} name value.
22   * The tag names listed here are used/recognized by different services.
23   */
24  public enum TagName {
25  
26    /**
27     * The Darwin Core term datasetID which is found in dwc archives to reference subdatasets
28     * such as the Catalogue of Life GSDs. Stored in the registry to keep a mapping between the subdataset registry key
29     * and the dwca datasetID value.
30     *
31     * @see <a href="http://rs.tdwg.org/dwc/terms/datasetID">DWC Term</a>
32     * @see <a href="http://kos.gbif.org/wiki/Dwc:datasetID">KOS wiki</a>
33     */
34    DATASET_ID("dataset_id", GBIF_HARVESTING),
35  
36    /**
37     * This is a counter starting at 1 incremented every time we try to crawl a dataset.
38     * This is even incremented when a crawl fails for any reason.
39     */
40    CRAWL_ATTEMPT("crawl_attempt", GBIF_CRAWLER),
41  
42    /**
43     * <em>Datasets</em> marked with this tag will not be in the periodic crawl.
44     */
45    OMIT_FROM_SCHEDULED_CRAWL("omitFromScheduledCrawl", GBIF_CRAWLER),
46  
47    /**
48     * Records the source of an archive endpoint, e.g. that it is from a BioCASe installation.
49     */
50    ARCHIVE_ORIGIN("archiveOrigin", GBIF_METASYNC),
51  
52    /**
53     * TaPIR or BioCASe conceptual schema.
54     */
55    CONCEPTUAL_SCHEMA("conceptualSchema", GBIF_METASYNC),
56  
57    /**
58     * Dataset title for BioCASe, no longer used.
59     */
60    @Deprecated
61    DATASET_TITLE("datasetTitle", GBIF_METASYNC),
62  
63    /**
64     * Date the records in the dataset were last updated.
65     */
66    DATE_LAST_UPDATED("dateLastUpdated", GBIF_METASYNC),
67  
68    /**
69     * Number of records declared in the dataset by the source.
70     */
71    DECLARED_COUNT("declaredCount", GBIF_METASYNC),
72  
73    /**
74     * DiGIR code.
75     */
76    DIGIR_CODE("code", GBIF_METASYNC),
77  
78    /**
79     * Not sure, no longer in use?
80     */
81    @Deprecated
82    LOCAL_ID("localId", GBIF_METASYNC),
83  
84    /**
85     * Number of records to request in each search.
86     */
87    MAX_SEARCH_RESPONSE_RECORDS("maxSearchResponseRecords", GBIF_METASYNC),
88  
89    /**
90     * Records whether an orphan dataset has been RESCUED.
91     */
92    ORPHAN_STATUS("status", GBIF_ORPHANS),
93  
94    /**
95     * The endpoint which was not working when the dataset was rescued.
96     */
97    ORPHANED_ENDPOINT("orphanEndpoint", GBIF_ORPHANS),
98  
99    /**
100    * The GBIF download used to rescue the orphaned dataset.
101    */
102   ORPHAN_DOWNLOAD("download", GBIF_ORPHANS),
103 
104   /**
105    * The modification time of the last-available Darwin Core Archive from temporary storage,
106    * used to rescue the dataset.
107    */
108   ORPHAN_DWCA_CACHE_TIME("crawlerDwcaCacheTime", GBIF_ORPHANS);
109 
110   private final String name;
111 
112   private final TagNamespace namespace;
113 
114   /**
115    * Default constructor.
116    *
117    * @param name      namespace value
118    * @param namespace, Enum namespace to which belongs the predicate
119    */
120   TagName(String name, TagNamespace namespace) {
121     this.name = name;
122     this.namespace = namespace;
123   }
124 
125   /**
126    * Namespace to which belong this tag.
127    *
128    * @return the namespace
129    */
130   public TagNamespace getNamespace() {
131     return namespace;
132   }
133 
134   /**
135    * Actual machine tag name.
136    *
137    * @return the machine tag name
138    */
139   public String getName() {
140     return name;
141   }
142 
143 }
