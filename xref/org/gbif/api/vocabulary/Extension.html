83    private final String rowType;
84  
85    /**
86     * @param rowType the case insensitive row type uri for the extension
87     *
88     * @return the matching extension or null
89     */
90    public static Extension fromRowType(String rowType) {
91      if (!Strings.isNullOrEmpty(rowType)) {
92        for (Extension extension : Extension.values()) {
93          if (rowType.equalsIgnoreCase(extension.getRowType())
94              || rowType.equalsIgnoreCase(extension.name().replaceAll("_", ""))) {
95            return extension;
96          }
97        }
98      }
99      return null;
100   }
