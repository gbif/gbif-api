16  package org.gbif.api.vocabulary;
17  
18  /**
19   * Very coarse habitat vocabulary matching the boolean biome flags found in the species profile extension.
20   * These also match the classifications provided by the WWF, BBC.
21   *
22   * http://www.ucmp.berkeley.edu/exhibits/biomes/index.php
23   * http://wwf.panda.org/about_our_earth/ecoregions/about/habitat_types/
24   *
25   * @see <a href="http://rs.gbif.org/extension/gbif/1.0/speciesprofile.xml">Species Profile Extension</a>
26   */
27  public enum Habitat {
28  
29    /**
30     * Marine habitats including the deep & open ocean, reefs and estuaries.
31     *
32     * @see <a href="http://wwf.panda.org/about_our_earth/ecoregions/about/habitat_types/selecting_marine_ecoregions/">WWF</a>
33     * @see <a href="http://www.bbc.co.uk/nature/habitats#marine">BBC</a>
34     * @see <a href="http://www.ucmp.berkeley.edu/exhibits/biomes/marine.php">UCMP Berkeley</a>
35     *
36     */
37    MARINE,
38  
39    /**
40     * Freshwater habitats including rivers, lakes, ponds, wetlands, bogs, marsh, swamp and brackish waters!
41     *
42     * @see <a href="http://wwf.panda.org/about_our_earth/ecoregions/about/habitat_types/selecting_freshwater_ecoregions/">WWF</a>
43     * @see <a href="http://www.bbc.co.uk/nature/habitats#freshwater">BBC</a>
44     * @see <a href="http://www.ucmp.berkeley.edu/exhibits/biomes/freshwater.php">UCMP Berkeley</a>
45     *
46     */
47    FRESHWATER,
48  
49    /**
50     * Terrestrial habitats cover all habitats on land
51     * including forests, deserts, grasslands, meadows, tundra, mangroves, farmland.
52     *
53     * @see <a href="http://wwf.panda.org/about_our_earth/ecoregions/about/habitat_types/selecting_terrestrial_ecoregions/">WWF</a>
54     * @see <a href="http://www.bbc.co.uk/nature/habitats#land">BBC</a>
55     * @see <a href="http://www.ucmp.berkeley.edu/exhibits/biomes/index.php">UCMP Berkeley</a>
56     *
57     */
58    TERRESTRIAL;
59  }
