386   /**
387    * Returns the corresponding {@link TranslationLanguage} for the locale specified.
388    *
389    * @param locale language locLW
390    * @return the matching TranslationLanguage or UNKNOWN
391    */
392   public static TranslationLanguage fromLocale(String locale) {
393     if (locale != null && !locale.isEmpty()) {
394       String localeCode = locale.trim();
395       for (TranslationLanguage language : TranslationLanguage.values()) {
396         if (localeCode.equalsIgnoreCase(language.getLocale())) {
397           return language;
398         }
399       }
400     }
401     return UNKNOWN;
402   }
