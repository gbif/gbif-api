1   package org.gbif.api.vocabulary;
2   
3   /**
4    * Enumeration of issues for each name usage record encountered during checklist processing.
5    */
6   public enum NameUsageIssue {
7     /**
8      * The value for dwc:parentNameUsageID could not be resolved.
9      */
10    PARENT_NAME_USAGE_ID_INVALID,
11  
12    /**
13     * The value for dwc:acceptedNameUsageID could not be resolved.
14     */
15    ACCEPTED_NAME_USAGE_ID_INVALID,
16  
17    /**
18     * The value for dwc:originalNameUsageID could not be resolved.
19     */
20    ORIGINAL_NAME_USAGE_ID_INVALID,
21  
22    /**
23     * Synonym lacking an accepted name.
24     */
25    ACCEPTED_NAME_MISSING,
26  
27    /**
28     * dwc:taxonRank could not be interpreted
29     */
30    RANK_INVALID,
31  
32    /**
33     * dwc:nomenclaturalStatus could not be interpreted
34     */
35    NOMENCLATURAL_STATUS_INVALID,
36  
37    /**
38     * dwc:taxonomicStatus could not be interpreted
39     */
40    TAXONOMIC_STATUS_INVALID,
41  
42    /**
43     * The scientific name was assembled from the individual name parts and not given as a whole string.
44     */
45    SCIENTIFIC_NAME_ASSEMBLED,
46  
47    /**
48     * If a synonym points to another synonym as its accepted taxon the chain is resolved.
49     */
50    CHAINED_SYNOYM,
51  
52    /**
53     * The authorship of the original name does not match the authorship in brackets of the actual name.
54     */
55    BASIONYM_AUTHOR_MISMATCH,
56  
57    TAXONOMIC_STATUS_MISMATCH,
58  
59    /**
60     * The child parent classification resulted into a cycle that needed to be resolved/cut.
61     */
62    PARENT_CYCLE,
63  
64    CLASSIFICATION_RANK_ORDER_INVALID,
65  
66    /**
67     * The denormalized classification could not be applied to the name usage.
68     * For example if the id based classification has no ranks.
69     */
70    CLASSIFICATION_NOT_APPLIED,
71  
72    /**
73     * At least one vernacular name extension record attached to this name usage is invalid.
74     */
75    VERNACULAR_NAME_INVALID,
76  
77    /**
78     * At least one description extension record attached to this name usage is invalid.
79     */
80    DESCRIPTION_INVALID,
81  
82    /**
83     * At least one distribution extension record attached to this name usage is invalid.
84     */
85    DISTRIBUTION_INVALID,
86  
87    /**
88     * At least one species profile extension record attached to this name usage is invalid.
89     */
90    SPECIES_PROFILE_INVALID,
91  
92    /**
93     * At least one multimedia extension record attached to this name usage is invalid.
94     * This covers multimedia coming in through various extensions including
95     * Audubon core, Simple images or multimedia or EOL media.
96     */
97    MULTIMEDIA_INVALID,
98  
99    /**
100    * At least one bibliographic reference extension record attached to this name usage is invalid.
101    */
102   BIB_REFERENCE_INVALID,
103 
104   /**
105    * At least one alternative identifier extension record attached to this name usage is invalid.
106    */
107   ALT_IDENTIFIER_INVALID,
108 
109   /**
110    * Name usage could not be matched to the GBIF backbone.
111    */
112   BACKBONE_MATCH_NONE,
113 
114   /**
115    * Name usage could only be matched to the GBIF backbone using fuzzy matching.
116    * @deprecated because there should be no fuzzy matching being used anymore for matching checklist names
117    */
118   @Deprecated
119   BACKBONE_MATCH_FUZZY,
120 
121   /**
122    * Synonym has a verbatim accepted name which is not unique and refers to several records.
123    */
124   ACCEPTED_NAME_NOT_UNIQUE,
125 
126   /**
127    * Record has a verbatim parent name which is not unique and refers to several records.
128    */
129   PARENT_NAME_NOT_UNIQUE,
130 
131   /**
132    * Record has a verbatim original name (basionym) which is not unique and refers to several records.
133    */
134   ORIGINAL_NAME_NOT_UNIQUE,
135 
136   /**
137    * There were problems representing all name usage relationships,
138    * i.e. the link to the parent, accepted and/or original name.
139    * The interpreted record in ChecklistBank is lacking some of the original source relation.
140    */
141   RELATIONSHIP_MISSING,
142 
143   /**
144    * Record has a original name (basionym) relationship which was derived from name & authorship comparison, but did not exist explicitly in the data.
145    * This should only be flagged in programmatically generated GBIF backbone usages.
146    */
147   ORIGINAL_NAME_DERIVED,
148 
149   /**
150    * The taxonomic status was inferred from the name or relations.
151    * Typically this is used to indicate that a name of a original name group has been turned into a synonym
152    * because there cannot be more than one accepted name in a homotypical group of names.
153    */
154   STATUS_DERIVED,
155 
156   /**
157    * The (accepted) bi/trinomial name does not match the parent name and should be recombined into the parent genus/species.
158    * For example the species Picea alba with a parent genus Abies is a mismatch and should be replaced by Abies alba.
159    */
160   NAME_PARENT_MISMATCH
161 }
