1   package org.gbif.api.vocabulary;

3   /**
4    * Enumeration of issues for each name usage record encountered during checklist processing.
5    */
6   public enum NameUsageIssue {
7     PARENT_NAME_USAGE_ID_INVALID,
8     ACCEPTED_NAME_USAGE_ID_INVALID,
9     ORIGINAL_NAME_USAGE_ID_INVALID,
10  
11    /**
12     * Synonym lacking an accepted name.
13     */
14    ACCEPTED_NAME_MISSING,
15  
16    /**
17     * dwc:taxonRank could not be interpreted
18     */
19    RANK_INVALID,
20  
21    /**
22     * dwc:nomenclaturalStatus could not be interpreted
23     */
24    NOMENCLATURAL_STATUS_INVALID,
25  
26    /**
27     * dwc:taxonomicStatus could not be interpreted
28     */
29    TAXONOMIC_STATUS_INVALID,
30  
31    /**
32     * The scientific name was assembled from the individual name parts and not given as a whole string.
33     */
34    SCIENTIFIC_NAME_ASSEMBLED,
35  
36    /**
37     * If a synonym points to another synonym as its accepted taxon the chain is resolved.
38     */
39    CHAINED_SYNOYM,
40  
41    /**
42     * The authorship of the original name does not match the authorshipin brackets of the actual name.
43     */
44    BASIONYM_AUTHOR_MISMATCH,
45  
46    TAXONOMIC_STATUS_MISMATCH,
47  
48    /**
49     * The child parent classification resulted into a cycle that needed to be resolved/cut.
50     */
51    PARENT_CYCLE,
52  
53    CLASSIFICATION_RANK_ORDER_INVALID,
54  
55    /**
56     * The denormalized classification could not be applied to the name usage.
57     * For example if the id based classification has no ranks.
58     */
59    CLASSIFICATION_NOT_APPLIED,
60  
61    /**
62     * At least one vernacular name extension record attached to this name usage is invalid.
63     */
64    VERNACULAR_NAME_INVALID,
65  
66    /**
67     * At least one description extension record attached to this name usage is invalid.
68     */
69    DESCRIPTION_INVALID,
70  
71    /**
72     * At least one distribution extension record attached to this name usage is invalid.
73     */
74    DISTRIBUTION_INVALID,
75  
76    /**
77     * At least one species profile extension record attached to this name usage is invalid.
78     */
79    SPECIES_PROFILE_INVALID,
80  
81    /**
82     * At least one multimedia extension record attached to this name usage is invalid.
83     * This covers multimedia coming in through various extensions including
84     * Audubon core, Simple images or multimedia or EOL media.
85     */
86    MULTIMEDIA_INVALID,
87  
88    /**
89     * At least one bibliographic reference extension record attached to this name usage is invalid.
90     */
91    BIB_REFERENCE_INVALID,
92  
93    /**
94     * At least one alternative identifier extension record attached to this name usage is invalid.
95     */
96    ALT_IDENTIFIER_INVALID,
97  
98    /**
99     * Name usage could not be matched to the GBIF backbone.
100    */
101   BACKBONE_MATCH_NONE,
102 
103   /**
104    * Name usage could only be matched to the GBIF backbone using fuzzy matching.
105    */
106   BACKBONE_MATCH_FUZZY
107 }
