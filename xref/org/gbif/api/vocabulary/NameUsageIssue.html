13  /**
14   * Enumeration of issues for each name usage record encountered during checklist processing.
15   */
16  public enum NameUsageIssue implements InterpretationRemark {
17    /**
18     * The value for dwc:parentNameUsageID could not be resolved.
19     */
20    PARENT_NAME_USAGE_ID_INVALID(ERROR, DwcTerm.parentNameUsageID),
21  
22    /**
23     * The value for dwc:acceptedNameUsageID could not be resolved.
24     */
25    ACCEPTED_NAME_USAGE_ID_INVALID(ERROR, DwcTerm.acceptedNameUsageID),
26  
27    /**
28     * The value for dwc:originalNameUsageID could not be resolved.
29     */
30    ORIGINAL_NAME_USAGE_ID_INVALID(ERROR, DwcTerm.originalNameUsageID),
31  
32    /**
33     * Synonym lacking an accepted name.
34     */
35    ACCEPTED_NAME_MISSING(DwcTerm.taxonomicStatus, DwcTerm.acceptedNameUsageID, DwcTerm.acceptedNameUsage),
36  
37    /**
38     * dwc:taxonRank could not be interpreted
39     */
40    RANK_INVALID(DwcTerm.taxonRank),
41  
42    /**
43     * dwc:nomenclaturalStatus could not be interpreted
44     */
45    NOMENCLATURAL_STATUS_INVALID(DwcTerm.nomenclaturalStatus),
46  
47    /**
48     * dwc:taxonomicStatus could not be interpreted
49     */
50    TAXONOMIC_STATUS_INVALID(DwcTerm.taxonomicStatus),
51  
52    /**
53     * The scientific name was assembled from the individual name parts and not given as a whole string.
54     */
55    SCIENTIFIC_NAME_ASSEMBLED(INFO, DwcTerm.scientificName, DwcTerm.genus, DwcTerm.specificEpithet, DwcTerm.infraspecificEpithet, DwcTerm.taxonRank, DwcTerm.scientificNameAuthorship, DwcTerm.namePublishedInYear),
56  
57    /**
58     * If a synonym points to another synonym as its accepted taxon the chain is resolved.
59     */
60    CHAINED_SYNOYM(DwcTerm.acceptedNameUsageID, DwcTerm.acceptedNameUsage),
61  
62    /**
63     * The authorship of the original name does not match the authorship in brackets of the actual name.
64     */
65    BASIONYM_AUTHOR_MISMATCH(DwcTerm.scientificName, DwcTerm.scientificNameAuthorship, DwcTerm.originalNameUsageID, DwcTerm.originalNameUsage),
66  
67    TAXONOMIC_STATUS_MISMATCH(DwcTerm.taxonomicStatus, DwcTerm.acceptedNameUsageID, DwcTerm.acceptedNameUsage),
68  
69    /**
70     * The child parent classification resulted into a cycle that needed to be resolved/cut.
71     */
72    PARENT_CYCLE(ERROR, DwcTerm.parentNameUsageID, DwcTerm.parentNameUsage),
73  
74    /**
75     * The given ranks of the names in the classification hierarchy do not follow the hierarchy of ranks.
76     */
77    CLASSIFICATION_RANK_ORDER_INVALID(ERROR, DwcTerm.parentNameUsageID, DwcTerm.parentNameUsage, DwcTerm.taxonRank),
78  
79    /**
80     * The denormalized classification could not be applied to the name usage.
81     * For example if the id based classification has no ranks.
82     */
83    CLASSIFICATION_NOT_APPLIED(DwcTerm.kingdom, DwcTerm.phylum, DwcTerm.class_, DwcTerm.order, DwcTerm.family, DwcTerm.genus),
84  
85    /**
86     * At least one vernacular name extension record attached to this name usage is invalid.
87     */
88    VERNACULAR_NAME_INVALID(DwcTerm.vernacularName),
89  
90    /**
91     * At least one description extension record attached to this name usage is invalid.
92     */
93    DESCRIPTION_INVALID(GbifTerm.Description),
94  
95    /**
96     * At least one distribution extension record attached to this name usage is invalid.
97     */
98    DISTRIBUTION_INVALID(GbifTerm.Distribution),
99  
100   /**
101    * At least one species profile extension record attached to this name usage is invalid.
102    */
103   SPECIES_PROFILE_INVALID(GbifTerm.SpeciesProfile),
104 
105   /**
106    * At least one multimedia extension record attached to this name usage is invalid.
107    * This covers multimedia coming in through various extensions including
108    * Audubon core, Simple images or multimedia or EOL media.
109    */
110   MULTIMEDIA_INVALID(GbifTerm.Multimedia),
111 
112   /**
113    * At least one bibliographic reference extension record attached to this name usage is invalid.
114    */
115   BIB_REFERENCE_INVALID(GbifTerm.Reference),
116 
117   /**
118    * At least one alternative identifier extension record attached to this name usage is invalid.
119    */
120   ALT_IDENTIFIER_INVALID(GbifTerm.Identifier),
121 
122   /**
123    * Name usage could not be matched to the GBIF backbone.
124    */
125   BACKBONE_MATCH_NONE(INFO, DwcTerm.scientificName, DwcTerm.scientificNameAuthorship, DwcTerm.kingdom, DwcTerm.taxonRank),
126 
127   /**
128    * Name usage could only be matched to the GBIF backbone using fuzzy matching.
129    * @deprecated because there should be no fuzzy matching being used anymore for matching checklist names
130    */
131   @Deprecated
132   BACKBONE_MATCH_FUZZY(DwcTerm.scientificName, DwcTerm.scientificNameAuthorship, DwcTerm.kingdom, DwcTerm.taxonRank),
133 
134   /**
135    * Synonym has a verbatim accepted name which is not unique and refers to several records.
136    */
137   ACCEPTED_NAME_NOT_UNIQUE(DwcTerm.acceptedNameUsage),
138 
139   /**
140    * Record has a verbatim parent name which is not unique and refers to several records.
141    */
142   PARENT_NAME_NOT_UNIQUE(DwcTerm.parentNameUsage),
143 
144   /**
145    * Record has a verbatim original name (basionym) which is not unique and refers to several records.
146    */
147   ORIGINAL_NAME_NOT_UNIQUE(DwcTerm.originalNameUsage),
148 
149   /**
150    * There were problems representing all name usage relationships,
151    * i.e. the link to the parent, accepted and/or original name.
152    * The interpreted record in ChecklistBank is lacking some of the original source relation.
153    */
154   RELATIONSHIP_MISSING(DwcTerm.parentNameUsageID, DwcTerm.acceptedNameUsageID, DwcTerm.originalNameUsageID, DwcTerm.parentNameUsage, DwcTerm.acceptedNameUsage, DwcTerm.originalNameUsage),
155 
156   /**
157    * Record has a original name (basionym) relationship which was derived from name & authorship comparison, but did not exist explicitly in the data.
158    * This should only be flagged in programmatically generated GBIF backbone usages.
159    * GBIF backbone specific issue.
160    */
161   ORIGINAL_NAME_DERIVED(INFO),
162 
163   /**
164    * There have been more than one accepted name in a homotypical basionym group of names.
165    * GBIF backbone specific issue.
166    */
167   CONFLICTING_BASIONYM_COMBINATION(),
168 
169   /**
170    * The group (currently only genera are tested) are lacking any accepted species
171    * GBIF backbone specific issue.
172    */
173   NO_SPECIES(INFO),
174 
175   /**
176    * The (accepted) bi/trinomial name does not match the parent name and should be recombined into the parent genus/species.
177    * For example the species Picea alba with a parent genus Abies is a mismatch and should be replaced by Abies alba.
178    * GBIF backbone specific issue.
179    */
180   NAME_PARENT_MISMATCH(),
181 
182   /**
183    * A potential orthographic variant exists in the backbone.
184    * GBIF backbone specific issue.
185    */
186   ORTHOGRAPHIC_VARIANT(INFO),
187 
188   /**
189    * A not synonymized homonym exists for this name in some other backbone source which have been ignored at build time.
190    */
191   HOMONYM(INFO, DwcTerm.scientificName),
192 
193   /**
194    * A bi/trinomial name published earlier than the parent genus was published.
195    * This might indicate that the name should rather be a recombination.
196    */
197   PUBLISHED_BEFORE_GENUS(DwcTerm.scientificName, DwcTerm.scientificNameAuthorship, DwcTerm.namePublishedInYear, DwcTerm.genus, DwcTerm.parentNameUsageID, DwcTerm.parentNameUsage),
198 
199   /**
200    * The scientific name string could not be parsed at all, but appears to be a parsable name type,
201    * i.e. it is not classified as a virus or hybrid formula.
202    */
203   UNPARSABLE(DwcTerm.scientificName),
204 
205   /**
206    * The beginning of the scientific name string was parsed,
207    * but there is additional information in the string that was not understood.
208    */
209   PARTIALLY_PARSABLE(DwcTerm.scientificName);

212   private final Set<Term> related;
213   private final InterpretationRemarkSeverity severity;
214   private final boolean isDeprecated;
215 
216   NameUsageIssue(Term ... related) {
217     this(WARNING, related);
218   }
219 
220   NameUsageIssue(InterpretationRemarkSeverity severity, Term ... related) {
221     if (related == null) {
222       this.related = ImmutableSet.of();
223     } else {
224       this.related = ImmutableSet.copyOf(related);
225     }
226     this.severity = severity;
227     this.isDeprecated = AnnotationUtils.isFieldDeprecated(NameUsageIssue.class, this.name());
228   }
229 
230   @Override
231   public String getId() {
232     return name();
233   }
234 
235   @Override
236   public Set<Term> getRelatedTerms() {
237     return related;
238   }
239 
240   @Override
241   public InterpretationRemarkSeverity getSeverity(){
242     return severity;
243   }
244 
245   @Override
246   public boolean isDeprecated(){
247     return isDeprecated;
248   }
