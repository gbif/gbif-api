19  import org.gbif.api.util.VocabularyUtils;
20  
21  /**
22   * Enumeration for all endpoint types.
23   */
24  public enum EndpointType {
25    /**
26     * A single <a href="http://knb.ecoinformatics.org/software/eml/">EML metadata document</a> in any EML version.
27     */
28    EML,
29    /**
30     * Syndication feeds like RSS or ATOM of various flavors.
31     * The feeds can serve very different purpose, for example if attached to an organisation it can be news items.
32     * Attached to a technical installation like the IPT it should return the latest dataset resources.
33     */
34    FEED,
35    /**
36     * An OGC Web Feature Service.
37     */
38    WFS,
39    /**
40     * An OGC Web Map Service.
41     */
42    WMS,
43    /**
44     * Taxon Concept data given as RDF
45     * based on the <a href="http://rs.tdwg.org/ontology/voc/TaxonConcept.rdf">TDWG ontology</a>.
46     */
47    TCS_RDF,
48    /**
49     * A Taxon Concept Schema document.
50     * @see <a href="http://www.tdwg.org/standards/117/">TDWG TCS Standard</a>.
51     */
52    TCS_XML,
53    /**
54     * A Darwin Core Archive as defined by the <a href="http://rs.tdwg.org/dwc/terms/guides/text/index.htm">Darwin Core Text Guidelines</a>.
55     */
56    DWC_ARCHIVE,
57    /**
58     * A <a href="http://digir.sourceforge.net/">DiGIR</a> service endpoint.
59     */
60    DIGIR,
61    /**
62     * A DiGIR service slightly modified for the MANIS network.
63     */
64    DIGIR_MANIS,
65    /**
66     * A <a href="http://www.tdwg.org/dav/subgroups/tapir/1.0/docs/">TAPIR</a> service.
67     */
68    TAPIR,
69    /**
70     * A <a href="http://www.biocase.org/products/protocols/index.shtml">BioCASe</a> protocl compliant service.
71     */
72    BIOCASE,
73    /**
74     * The Open Archives Initiative Protocol for Metadata Harvesting.
75     * A <a href="http://www.openarchives.org/OAI/openarchivesprotocol.html">OAI-PMH</a> compliant data provider service.
76     */
77    OAI_PMH,
78    /**
79     * Any other service not covered by this enum so far.
80     */
81    OTHER;
82  
83    /**
84     * @return the matching EndpointType or null
85     */
86    public static EndpointType fromString(String endpointType) {
87      return (EndpointType) VocabularyUtils.lookupEnum(endpointType, EndpointType.class);
88    }
89  
90  }
