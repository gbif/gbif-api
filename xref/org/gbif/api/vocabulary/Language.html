971   /**
972    * @param code the case insensitive 2 or 3 letter codes
973    *
974    * @return the matching language or UNKNOWN
975    */
976   public static Language fromIsoCode(String code) {
977     if (!Strings.isNullOrEmpty(code)) {
978       String codeLower = code.toLowerCase().trim();
979       if (codeLower.length() == 2){
980         for (Language language : Language.values()) {
981           if (codeLower.equals(language.getIso2LetterCode())) {
982             return language;
983           }
984         }
985       } else if (codeLower.length() == 3) {
986         for (Language language : Language.values()) {
987           if (codeLower.equals(language.getIso3LetterCode())) {
988             return language;
989           }
990         }
991       }
992     }
993     return UNKNOWN;
994   }
