64    /**

71    private static final ImmutableMap<String, ContactType> TYPE_LOOKUP;
72  
73    static {
74  
75      Map<String, ContactType> lookup = new HashMap<String, ContactType>();
76      lookup.put("administrative", ADMINISTRATIVE_POINT_OF_CONTACT);
77      lookup.put("technical", TECHNICAL_POINT_OF_CONTACT);
78      lookup.put("pointofcontact", POINT_OF_CONTACT);
79      lookup.put("originator", ORIGINATOR);
80      lookup.put("metadataprovider", METADATA_AUTHOR);
81      lookup.put("principleinvestigator", PRINCIPAL_INVESTIGATOR);
82      lookup.put("author", AUTHOR);
83      lookup.put("contentprovider", CONTENT_PROVIDER);
84      lookup.put("custodiansteward", CUSTODIAN_STEWARD);
85      lookup.put("distributor", DISTRIBUTOR);
86      lookup.put("editor", EDITOR);
87      lookup.put("owner", OWNER);
88      lookup.put("processor", PROCESSOR);
89      lookup.put("publisher", PUBLISHER);
90      lookup.put("user", USER);
91      lookup.put("programmer", PROGRAMMER);
92      lookup.put("data administrator", DATA_ADMINISTRATOR);
93      lookup.put("system adminsitrator", SYSTEM_ADMINISTRATOR);
94      TYPE_LOOKUP = ImmutableMap.copyOf(lookup);
95    }
96  
97    /**
98     * Tries its best to infer a ContactType from a given string. This can for example be used for the various contact
99     * types from DiGIR, TAPIR and BioCASe.
100    *
101    * @return the inferred ContactType
102    */
103   public static ContactType inferType(@Nullable String type) {
104     if (type != null) {
105       ContactType contactType = TYPE_LOOKUP.get(type.toLowerCase());
106       if (contactType != null) {
107         return contactType;
108       }
109       return fromString(type);
110     }
111     return null;
112   }
