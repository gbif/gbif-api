16  package org.gbif.api.vocabulary;
17  
18  import org.gbif.api.util.VocabularyUtils;
19  
20  import java.util.HashMap;
21  import java.util.Map;
22  import javax.annotation.Nullable;
23  
24  import com.google.common.collect.ImmutableMap;
25  
26  /**
27   * Enumeration for all contact types.
28   * A utility to infer types is provided which has historically been used during data migration activities from legacy
29   * systems such as the previous "GBRDS" which was built on MySQL.
30   *
31   * @see <a href="http://rs.gbif.org/vocabulary/gbif/agent_role.xml">The IPT vocabulary</a>
32   */
33  public enum ContactType {
34  
35    TECHNICAL_POINT_OF_CONTACT,
36    ADMINISTRATIVE_POINT_OF_CONTACT,
37    POINT_OF_CONTACT,
38    ORIGINATOR,
39    METADATA_AUTHOR,
40    PRINCIPAL_INVESTIGATOR,
41    AUTHOR,
42    CONTENT_PROVIDER,
43    CUSTODIAN_STEWARD,
44    DISTRIBUTOR,
45    EDITOR,
46    OWNER,
47    PROCESSOR,
48    PUBLISHER,
49    USER,
50    PROGRAMMER,
51    DATA_ADMINISTRATOR,
52    SYSTEM_ADMINISTRATOR,
53    HEAD_OF_DELEGATION,
54    TEMPORARY_HEAD_OF_DELEGATION,
55    ADDITIONAL_DELEGATE,
56    TEMPORARY_DELEGATE,
57    REGIONAL_NODE_REPRESENTATIVE,
58    NODE_MANAGER,
59    NODE_STAFF;
60  
61    /**
62     * @return the matching ContactType or null
63     */
64    public static ContactType fromString(String contactType) {
65      return (ContactType) VocabularyUtils.lookupEnum(contactType, ContactType.class);
66    }
67  
68    // deliberate typos have existed and might be in genuine use
69    private static final ImmutableMap<String, ContactType> TYPE_LOOKUP = ImmutableMap.<String, ContactType> builder()
70      .put("administrative", ADMINISTRATIVE_POINT_OF_CONTACT)
71      .put("technical", TECHNICAL_POINT_OF_CONTACT)
72      .put("pointofcontact", POINT_OF_CONTACT)
73      .put("originator", ORIGINATOR)
74      .put("metadataprovider", METADATA_AUTHOR)
75      .put("principleinvestigator", PRINCIPAL_INVESTIGATOR)
76      .put("author", AUTHOR)
77      .put("contentprovider", CONTENT_PROVIDER)
78      .put("custodiansteward", CUSTODIAN_STEWARD)
79      .put("distributor", DISTRIBUTOR)
80      .put("editor", EDITOR)
81      .put("owner", OWNER)
82      .put("processor", PROCESSOR)
83      .put("publisher", PUBLISHER)
84      .put("user", USER)
85      .put("programmer", PROGRAMMER)
86      .put("data administrator", DATA_ADMINISTRATOR)
87      .put("system adminsitrator", SYSTEM_ADMINISTRATOR) // deliberate typo
88      .put("system administrator", SYSTEM_ADMINISTRATOR)
89    .build();
90  
91    /**
92     * Tries its best to infer a ContactType from a given string. This can for example be used for the various contact
93     * types from DiGIR, TAPIR and BioCASe.
94     *
95     * @return the inferred ContactType
96     */
97    public static ContactType inferType(@Nullable String type) {
98      if (type != null) {
99        ContactType contactType = TYPE_LOOKUP.get(type.toLowerCase());
100       if (contactType != null) {
101         return contactType;
102       }
103       return fromString(type);
104     }
105     return null;
106   }
107 }
