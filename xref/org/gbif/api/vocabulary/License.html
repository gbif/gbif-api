11  /**
12   * Enumeration of the set of licenses GBIF supports for applying to a dataset. The license provides a standardised way
13   * to define appropriate uses of a dataset.
14   * </br>
15   * GBIF's recommended best practice is to use the most recent license version, which for CC-BY and CC-BY-NC is 4.0.
16   * This is in line with the recommendation from Creative Commons.
17   * The ordinal number in the Enum implicitly defines the level of restriction, see LicenseTest.
18   *
19   * @see <a href="https://creativecommons.org/faq/#why-should-i-use-the-latest-version-of-the-creative-commons-licenses">Creative
20   * Commons recommendation</a>
21   * @see <a href="http://www.gbif.org/terms/licences">GBIF Licensing</a>
22   */
23  public enum License {
24  
25    /**
26     * Creative Commons Zero / Public Domain version 1.0. Technically a waiver, not a license.
27     *
28     * @see <a href="http://creativecommons.org/publicdomain/zero/1.0/legalcode">legal document</a>
29     */
30    CC0_1_0("Public Domain (CC0 1.0)", "http://creativecommons.org/publicdomain/zero/1.0/legalcode"),
31    /**
32     * Creative Commons Attribution version 4.0.
33     *
34     * @see <a href="http://creativecommons.org/licenses/by/4.0/legalcode">legal document</a>
35     */
36    CC_BY_4_0("Creative Commons Attribution (CC-BY) 4.0", "http://creativecommons.org/licenses/by/4.0/legalcode"),
37    /**
38     * Creative Commons Attribution-NonCommercial version 4.0.
39     *
40     * @see <a href="http://creativecommons.org/licenses/by-nc/4.0/legalcode">legal document</a>
41     */
42    CC_BY_NC_4_0("Creative Commons Attribution Non Commercial (CC-BY-NC) 4.0", "http://creativecommons.org/licenses/by-nc/4.0/legalcode"),
43    /**
44     * No license has been specified.
45     */
46    UNSPECIFIED(null, null),
47    /**
48     * A license not supported by GBIF.
49     */
50    UNSUPPORTED(null, null);
51  
52  
53    private final String licenseTitle;
54    private final String licenseUrl;
55  
56    /**
57     * FIXME returning Guava Optional will cause issues, Java 8 Optional should be returned.
58     * Get an {@link License} from its name as String.
59     *
60     * @return instance of com.google.common.base.Optional, never null
61     */
62    public static Optional<License> fromString(String license) {
63      return VocabularyUtils.lookup(license, License.class);
64    }
65  
66    /**
67     * FIXME returning Guava Optional will cause issues, Java 8 Optional should be returned.
68     * Lookup a License by either its a) legal code URL or b) human readable summary URL.
69     * For any parsing see LicenseParser in GBIF parsers project.
70     *
71     * @param licenseUrl the case insensitive URL for the license.
72     *
73     * @return instance of com.google.common.base.Optional, never null
74     */
75    public static Optional<License> fromLicenseUrl(String licenseUrl) {
76      if (!Strings.isNullOrEmpty(licenseUrl)) {
77        licenseUrl = licenseUrl.trim().toLowerCase();
78        licenseUrl = StringUtils.removeEnd(licenseUrl, "/");
79  
80        // do lookup by legal code URL or human readable summary URL (excluding "/legalcode")
81        for (License license : License.values()) {
82          if (license.getLicenseUrl() != null && (licenseUrl.equals(license.getLicenseUrl())
83                  || license.getLicenseUrl().startsWith(licenseUrl))) {
84            return Optional.fromNullable(license);
85          }
86        }
87      }
88      return Optional.absent();
89    }
90  
91    /**
92     * Get the most restrictive license between the 2 provided licenses.
93     * If one or the two licenses are null or not concrete, this method returns the fallback License.
94     *
95     * @param fallback License to return if one or the two licenses are null or not concrete
96     *
97     * @return the most restrictive License or the fallback License
98     */
99    public static License getMostRestrictive(License license1, License license2, License fallback) {
100     if (license1 == null || license2 == null || !license1.isConcrete() || !license2.isConcrete()) {
101       return fallback;
102     }
103     return (license1.compareTo(license2) > 0) ? license1 : license2;
104   }
105 
106   License(@Nullable String licenseTitle, @Nullable String licenseUrl) {
107     this.licenseTitle = licenseTitle;
108     // ensure license information is kept in lowercase internally
109     if (licenseUrl != null) {
110       this.licenseUrl = licenseUrl.toLowerCase();
111     } else {
112       this.licenseUrl = null;
113     }
114   }
115 
116   /**
117    * @return the License URL
118    */
119   public String getLicenseUrl() {
120     return licenseUrl;
121   }
122 
123   /**
124    * @return the License title
125    */
126   public String getLicenseTitle() {
127     return licenseTitle;
128   }
129 
130   /**
131    * Indicates if a license is a concrete license (true) or an abstracted license (false) like
132    * UNSPECIFIED or UNSUPPORTED.
133    * @return the license if concrete or not
134    */
135   public boolean isConcrete() {
136     return licenseUrl != null;
137   }
138 }
