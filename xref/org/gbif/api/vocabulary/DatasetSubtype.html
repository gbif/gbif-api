17  import java.util.Set;

21  /**
22   * Enumeration for all possible dataset subtypes.
23   */
24  public enum DatasetSubtype {
25    /**
26     * A taxonomic checklist that is authoritative in its classification and synonymy.
27     */
28    TAXONOMIC_AUTHORITY,
29    /**
30     * A nomenclatoral checklist that is authoritative in the pure names and publication information.
31     */
32    NOMENCLATOR_AUTHORITY,
33    /**
34     * A thematic checklist that has a theme for grouping names, e.g. parasites of elephants.
35     */
36    INVENTORY_THEMATIC,
37    /**
38     * A regional checklist that has a regional commonality in grouping names, e.g. species in a protected area.
39     */
40    INVENTORY_REGIONAL,
41    /**
42     * A taxonomic checklist with a global, spatial coverage.
43     * This subtype is used in particular by the catalogue of life to assemble its aggregated checklist.
44     */
45    GLOBAL_SPECIES_DATASET,
46    /**
47     * A taxonomic checklist that has been derived from data that was originally occurrence data.
48     */
49    DERIVED_FROM_OCCURRENCE,
50    /**
51     * Specimen data. Possible museum collection.
52     */
53    SPECIMEN,
54    /**
55     * Observation data. Possible monitoring and field observations.
56     */
57    OBSERVATION;
58  
59    /**
60     * A set of all DatasetSubtype that belong to DatasetType OCCURRENCE.
61     *
62     * @see DatasetType#OCCURRENCE
63     */
64    public static final Set<DatasetSubtype> OCCURRENCE_DATASET_SUBTYPES = ImmutableSet.of(SPECIMEN, OBSERVATION);
65  
66    /**
67     * A set of all DatasetSubtype that belong to DatasetType CHECKLIST.

71    public static final Set<DatasetSubtype> CHECKLIST_DATASET_SUBTYPES = ImmutableSet
72      .of(TAXONOMIC_AUTHORITY, NOMENCLATOR_AUTHORITY, INVENTORY_THEMATIC, INVENTORY_REGIONAL, GLOBAL_SPECIES_DATASET,
73        DERIVED_FROM_OCCURRENCE);
74  
75    /**

77     */

79      return (DatasetSubtype) VocabularyUtils.lookupEnum(datasetSubType, DatasetSubtype.class);
80    }
