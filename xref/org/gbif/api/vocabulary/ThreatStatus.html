16  package org.gbif.api.vocabulary;
17  
18  /**
19   * The IUCN threat status.
20   * The IUCN Red List Categories and Criteria are intended to be an easily and widely understood system
21   * for classifying species at high risk of global extinction. The general aim of the system is to provide an explicit,
22   * objective framework for the classification of the broadest range of species according to their extinction risk.
23   *
24   * @see <a href="http://rs.gbif.org/vocabulary/iucn/threat_status.xml">rs.gbif.org vocabulary</a>
25   */
26  public enum ThreatStatus {
27  
28    /**
29     * A taxon is Extinct when there is no reasonable doubt that the last individual has died. A taxon is presumed Extinct
30     * when exhaustive surveys in known and/or expected habitat, at appropriate times (diurnal, seasonal, annual),
31     * throughout its historic range have failed to record an individual. Surveys should be over a time frame appropriate
32     * to the taxon's life cycle and life form.
33     */
34    EXTINCT("EX"),
35  
36    /**
37     * A taxon is Extinct in the Wild when it is known only to survive in cultivation, in captivity or as a naturalized
38     * population (or populations) well outside the past range. A taxon is presumed Extinct in the Wild when exhaustive
39     * surveys in known and/or expected habitat, at appropriate times (diurnal, seasonal, annual), throughout its historic
40     * range have failed to record an individual. Surveys should be over a time frame appropriate to the taxon's life
41     * cycle and life form.
42     */
43    EXTINCT_IN_THE_WILD("EW"),
44  
45    /**
46     * Category for a taxon when there is no reasonable doubt that the last individual potentially capable of reproduction
47     * within the region has died or has disappeared from the wild in the region, or when, if it is a former visiting
48     * taxon, the last individual has died or disappeared in the wild from the region. The setting of any time limit for
49     * listing under RE is left to the discretion of the regional Red List authority, but should not normally pre-date
50     * 1500 AD.
51     */
52    REGIONALLY_EXTINCT("RE"),
53  
54    /**
55     * A taxon is Critically Endangered when the best available evidence indicates that it meets any of the criteria A to
56     * E for Critically Endangered (see Section V), and it is therefore considered to be facing an extremely high risk of
57     * extinction in the wild.
58     */
59    CRITICALLY_ENDANGERED("CR"),
60  
61    /**
62     * A taxon is Endangered when the best available evidence indicates that it meets any of the criteria A to E for
63     * Endangered (see Section V), and it is therefore considered to be facing a very high risk of extinction in the wild.
64     */
65    ENDANGERED("EN"),
66  
67    /**
68     * A taxon is Vulnerable when the best available evidence indicates that it meets any of the criteria A to E for
69     * Vulnerable (see Section V), and it is therefore considered to be facing a high risk of extinction in the wild.
70     */
71    VULNERABLE("VU"),
72  
73    /**
74     * A taxon is Near Threatened when it has been evaluated against the criteria but does not qualify for Critically
75     * Endangered, Endangered or Vulnerable now, but is close to qualifying for or is likely to qualify for a threatened
76     * category in the near future.
77     */
78    NEAR_THREATENED("NT"),
79  
80    /**
81     * A taxon is Least Concern when it has been evaluated against the criteria and does not qualify for Critically
82     * Endangered, Endangered, Vulnerable or Near Threatened. Widespread and abundant taxa are included in this category.
83     */
84    LEAST_CONCERN("LC"),
85  
86    /**
87     * A taxon is Data Deficient when there is inadequate information to make a direct, or indirect, assessment of its
88     * risk of extinction based on its distribution and/or population status. A taxon in this category may be well
89     * studied, and its biology well known, but appropriate data on abundance and/or distribution are lacking. Data
90     * Deficient is therefore not a category of threat. Listing of taxa in this category indicates that more information
91     * is required and acknowledges the possibility that future research will show that threatened classification is
92     * appropriate. It is important to make positive use of whatever data are available. In many cases great care should
93     * be exercised in choosing between DD and a threatened status. If the range of a taxon is suspected to be relatively
94     * circumscribed, and a considerable period of time has elapsed since the last record of the taxon, threatened status
95     * may well be justified.
96     */
97    DATA_DEFICIENT("DD"),
98  
99    /**
100    * Category for a taxon deemed to be ineligible for assessment at a regional level. A taxon may be NA because it is
101    * not a wild population or not within its natural range in the region, or because it is a vagrant to the region. It
102    * may also be NA because it occurs at very low numbers in the region (i.e. when the regional Red List authority has
103    * decided to use a filter to exclude taxa before the assessment procedure) or the taxon may be classified at a lower
104    * taxonomic level (e.g. below the level of species or subspecies) than considered eligible by the regional Red List
105    * authority. In contrast to other Red List Categories, it is not mandatory to use NA for all taxa to which it
106    * applies; but is recommended for taxa where its use is informative.
107    */
108   NOT_APPLICABLE("NA"),
109 
110   /**
111    * A taxon is Not Evaluated when it is has not yet been evaluated against the criteria.
112    */
113   NOT_EVALUATED("NE");
114 
115   private final String code;
116 
117   ThreatStatus(String code) {
118     this.code = code;
119   }
120 
121   public String getCode() {
122     return code;
123   }
124 }
