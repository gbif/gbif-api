22  /**
23   * Known namespaces for tags.

25  public enum TagNamespace {

27    /**

30    PUBLIC(null),

32    /**

35    GBIF("gbif.org"),
36  
37    /**
38     * The GBIF ChecklistBank namespace.
39     */
40    CHECKLISTBANK("clb.gbif.org"),
41  
42    /**

44     */
45    CRAWLER("crawler.gbif.org"),
46  
47    /**
48     * The GBIF DwC-Archive Validator namespace.
49     */
50    VALIDATOR("validator.gbif.org"),
51  
52    /**
53     * The GBIF Harvesting and Indexing Toolkit (HIT) namespace.
54     */
55    GBIF_HARVESTING("hit.gbif.org"),
56  
57    /**
58     * The Encyclopedia of Life (EOL) namespace.
59     */
60    EOL("eol.org"),
61  
62    /**
63     * The Catalogue of Life (COL) namespace.
64     */
65    COL("catalogueoflife.org"),
66  
67    /**
68     * The Atlas of Living Australia namespace.
69     */
70    ALA("ala.org.au");
71  
72    private final String namespace;
73  
74    TagNamespace(String namespace) {
75      this.namespace = namespace;
76    }
77  
78    /**
79     * @return the unique full namespace uri in lower case.
80     */
81    public String getNamespace() {
82      return namespace;
83    }
84  
85    /**
86     * Gets the list of {@link TagName} which belongs to the namespace.
87     */
88    public List<TagName> getPredicates() {
89      List<TagName> tagNames = Lists.newArrayList();
90      for (TagName tagName : TagName.values()) {
91        if (tagName.getNamespace() == this) {
92          tagNames.add(tagName);
93        }
94      }
95      return tagNames;
96    }

98  }
