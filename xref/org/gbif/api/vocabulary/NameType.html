16  package org.gbif.api.vocabulary;
17  
18  import com.google.common.base.Strings;
19  
20  /**
21   * A short classification of scientific name strings used in Checklist Bank.
22   */
23  public enum NameType {
24  
25    /**
26     * a scientific name which is not well formed.
27     */
28    SCINAME,
29  
30    /**
31     * a well formed scientific name according to present nomenclatural rules.
32     * This is either the canonical orcanonical with authorship.
33     */
34    WELLFORMED,
35  
36    /**
37     * doubtful whether this is a scientific name at all.
38     */
39    DOUBTFUL,
40  
41    /**
42     * surely not a scientific name.
43     */
44    BLACKLISTED,
45  
46    /**
47     * a virus name.
48     */
49    VIRUS,
50  
51    /**
52     * a hybrid <b>formula</b> (not a hybrid name).
53     */
54    HYBRID,
55  
56    /**
57     * a scientific name with some informal addition like "cf." or indetermined like Abies spec.
58     */
59    INFORMAL,
60  
61    /**
62     * a cultivated plant name.
63     */
64    CULTIVAR,
65  
66    /**
67     * Candidatus is a component of the taxonomic name for a bacterium that cannot be maintained in a
68     * Bacteriology Culture Collection. It is an interim taxonomic status for noncultivable organisms.
69     * An example would be "Candidatus Phytoplasma allocasuarinae".
70     * This can be abbreviated to "Ca. Phytoplasma allocasuarinae".
71     *
72     * @see <a href="http://en.wikipedia.org/wiki/Candidatus">wikipedia</a>
73     * @see <a href="http://www.bacterio.cict.fr/candidatus.html">J.P. Euz√©by</a>
74     *
75     */
76    CANDIDATUS;
77  
78    /**
79     * Case insensitive lookup of a name type by its name that does not throw an exception but returns null
80     * for not found name types.
81     *
82     * @param nameType case insensitive name of name type
83     *
84     * @return the matching NameType or null
85     */
86    public static NameType fromString(String nameType) {
87      if (!Strings.isNullOrEmpty(nameType)) {
88        try {
89          return valueOf(nameType.toUpperCase().trim());
90        } catch (IllegalArgumentException e) {
91          // swallow
92        }
93      }
94      return null;
95    }
96  
97    /**
98     * @return true if this name type is "better", ie more correct than the given name
99     */
100   public boolean isBetterThan(NameType other) {
101     if (other == null) {
102       return true;
103     }
104     if (BLACKLISTED != this) {
105       if (INFORMAL == this && BLACKLISTED == other) {
106         return true;
107       } else if (DOUBTFUL == this && (BLACKLISTED == other || INFORMAL == other)) {
108         return true;
109       } else if ((SCINAME == this || VIRUS == this || HYBRID == this)
110           && (BLACKLISTED == other || INFORMAL == other || DOUBTFUL == other)) {
111         return true;
112       } else if (WELLFORMED == this && WELLFORMED != other) {
113         return true;
114       }
115     }
116     return false;
117   }
118 
119   public boolean isParsable() {
120     return !(this == HYBRID || VIRUS == this || BLACKLISTED == this);
121   }

123 }
