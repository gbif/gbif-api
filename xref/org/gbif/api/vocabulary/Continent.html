16  package org.gbif.api.vocabulary;
17  
18  import org.gbif.api.util.VocabularyUtils;
19  
20  /**
21   * Enumeration for all continents based on the 7 number model found on
22   * <a href="http://en.wikipedia.org/wiki/Continent#Number_of_continents">Wikipedia</a>.
23   *
24   * In particular this splits the Americas into North and South America with North America including the Caribbean and
25   * reaching down and including Panama. See the <a href="http://upload.wikimedia.org/wikipedia/commons/1/19/Continents_vide_couleurs.png">Wikipedia continent map</a> for the exact divisions.
26   */
27  public enum Continent {
28  
29    AFRICA("Africa"),
30    ANTARCTICA("Antarctica"),
31    ASIA("Asia"),
32    OCEANIA("Oceania"),
33    EUROPE("Europe"),
34    /**
35     * North America includes the Caribbean and
36     * reachies down and includes Panama.
37     */
38    NORTH_AMERICA("North America"),
39    SOUTH_AMERICA("South America");
40  
41    private final String title;
42  
43    Continent(String title) {
44      this.title = title;
45    }
46  
47    /**
48     * @return the continent name in the English language.
49     */
50    public String getTitle() {
51      return title;
52    }
53  
54    /**
55     * @param continent name
56     *
57     * @return the matching continent or null
58     */
59    public static Continent fromString(String continent) {
60      try {
61        return (Continent) VocabularyUtils.lookupEnum(continent, Continent.class);
62      } catch (IllegalArgumentException e) {
63        return null;
64      }
65    }
66  }
