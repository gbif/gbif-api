1   package org.gbif.api.vocabulary;
2   
3   import org.gbif.dwc.terms.Term;
4   
5   import java.util.Set;
6   
7   /**
8    * Represents a remark flagged during the interpretation phase.
9    */
10  public interface InterpretationRemark {
11  
12    /**
13     * Returns the identifier of the {@link InterpretationRemark}.
14     * This is normally implemented to return the result of {@link Enum#name}.
15     * Potential name clashes are detected by unit tests.
16     *
17     * @return identifier of the {@link InterpretationRemark}. Never null.
18     */
19    String getId();
20  
21    /**
22     * {@link Set} containing the {@link Term} considered to flag this remark.
23     *
24     * @return {@link Set} of {@link Term} or empty {@link Set}. Never null.
25     */
26    Set<Term> getRelatedTerms();
27  
28    /**
29     * Returns the default severity of this {@link InterpretationRemark}.
30     * Severity can be relative to the context and profile. This {@link InterpretationRemarkSeverity}
31     * should be seen as the default severity in the general context of interpretation.
32     *
33     * @return default severity of this {@link InterpretationRemark}. Never null.
34     */
35    InterpretationRemarkSeverity getSeverity();
36  
37    /**
38     * Checks if the {@link InterpretationRemark} is deprecated or not.
39     * @return true if the {@link InterpretationRemark} is marked with @Deprecated.
40     */
41    boolean isDeprecated();
42  
43  }
